24  package org.hibernate.engine.internal;

26  import java.io.Serializable;
27  import java.lang.reflect.Constructor;
28  
29  import org.hibernate.InstantiationException;
30  import org.hibernate.MappingException;
31  import org.hibernate.engine.spi.IdentifierValue;
32  import org.hibernate.engine.spi.VersionValue;
33  import org.hibernate.property.Getter;
34  import org.hibernate.type.IdentifierType;
35  import org.hibernate.type.PrimitiveType;
36  import org.hibernate.type.Type;
37  import org.hibernate.type.VersionType;
38  
39  /**
40   * @author Gavin King
41   */
42  public class UnsavedValueFactory {
43  	
44  	private static Object instantiate(Constructor constructor) {
45  		try {
46  			return constructor.newInstance( (Object[]) null );
47  		}
48  		catch (Exception e) {
49  			throw new InstantiationException( "could not instantiate test object", constructor.getDeclaringClass(), e );
50  		}
51  	}
52  	
53  	/**
54  	 * Return an IdentifierValue for the specified unsaved-value. If none is specified, 
55  	 * guess the unsaved value by instantiating a test instance of the class and
56  	 * reading it's id property, or if that is not possible, using the java default
57  	 * value for the type 
58  	 */
59  	public static IdentifierValue getUnsavedIdentifierValue(
60  			String unsavedValue, 
61  			Getter identifierGetter,
62  			Type identifierType,
63  			Constructor constructor) {
64  		
65  		if ( unsavedValue == null ) {
66  			if ( identifierGetter!=null && constructor!=null ) {
67  				// use the id value of a newly instantiated instance as the unsaved-value
68  				Serializable defaultValue = (Serializable) identifierGetter.get( instantiate(constructor) );
69  				return new IdentifierValue( defaultValue );
70  			}
71  			else if ( identifierGetter != null && (identifierType instanceof PrimitiveType) ) {
72  				Serializable defaultValue = ( ( PrimitiveType ) identifierType ).getDefaultValue();
73  				return new IdentifierValue( defaultValue );
74  			}
75  			else {
76  				return IdentifierValue.NULL;
77  			}
78  		}
79  		else if ( "null".equals( unsavedValue ) ) {
80  			return IdentifierValue.NULL;
81  		}
82  		else if ( "undefined".equals( unsavedValue ) ) {
83  			return IdentifierValue.UNDEFINED;
84  		}
85  		else if ( "none".equals( unsavedValue ) ) {
86  			return IdentifierValue.NONE;
87  		}
88  		else if ( "any".equals( unsavedValue ) ) {
89  			return IdentifierValue.ANY;
90  		}
91  		else {
92  			try {
93  				return new IdentifierValue( ( Serializable ) ( ( IdentifierType ) identifierType ).stringToObject( unsavedValue ) );
94  			}
95  			catch ( ClassCastException cce ) {
96  				throw new MappingException( "Bad identifier type: " + identifierType.getName() );
97  			}
98  			catch ( Exception e ) {
99  				throw new MappingException( "Could not parse identifier unsaved-value: " + unsavedValue );
100 			}
101 		}
102 	}
103 
104 	public static VersionValue getUnsavedVersionValue(
105 			String versionUnsavedValue, 
106 			Getter versionGetter,
107 			VersionType versionType,
108 			Constructor constructor) {
109 		
110 		if ( versionUnsavedValue == null ) {
111 			if ( constructor!=null ) {
112 				Object defaultValue = versionGetter.get( instantiate(constructor) );
113 				// if the version of a newly instantiated object is not the same
114 				// as the version seed value, use that as the unsaved-value
115 				return versionType.isEqual( versionType.seed( null ), defaultValue ) ?
116 						VersionValue.UNDEFINED :
117 						new VersionValue( defaultValue );
118 			}
119 			else {
120 				return VersionValue.UNDEFINED;
121 			}
122 		}
123 		else if ( "undefined".equals( versionUnsavedValue ) ) {
124 			return VersionValue.UNDEFINED;
125 		}
126 		else if ( "null".equals( versionUnsavedValue ) ) {
127 			return VersionValue.NULL;
128 		}
129 		else if ( "negative".equals( versionUnsavedValue ) ) {
130 			return VersionValue.NEGATIVE;
131 		}
132 		else {
133 			// this should not happen since the DTD prevents it
134 			throw new MappingException( "Could not parse version unsaved-value: " + versionUnsavedValue );
135 		}
136 		
137 	}
138 
139 }
