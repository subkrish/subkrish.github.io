48  /**
49   * Acts as a cache for compiled query plans, as well as query-parameter metadata.
50   *
51   * @see Environment#QUERY_PLAN_CACHE_PARAMETER_METADATA_MAX_SIZE
52   * @see Environment#QUERY_PLAN_CACHE_MAX_SIZE
53   *
54   * @author Steve Ebersole
55   */
56  public class QueryPlanCache implements Serializable {
57  
58      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, QueryPlanCache.class.getName());
59  	
60  	/**
61  	 * The default strong reference count.
62  	 */
63  	public static final int DEFAULT_PARAMETER_METADATA_MAX_COUNT = 128;
64  	/**
65  	 * The default soft reference count.
66  	 */
67  	public static final int DEFAULT_QUERY_PLAN_MAX_COUNT = 2048;
68  	private final SessionFactoryImplementor factory;
69  	/**
70  	 * the cache of the actual plans...
71  	 */
72  	private final BoundedConcurrentHashMap queryPlanCache;
73  	/**
74  	 * simple cache of param metadata based on query string.  Ideally, the original "user-supplied query"
75  	 * string should be used to obtain this metadata (i.e., not the para-list-expanded query string) to avoid
76  	 * unnecessary cache entries.
77  	 * <p></p>
78  	 * Used solely for caching param metadata for native-sql queries, see {@link #getSQLParameterMetadata} for a
79  	 * discussion as to why...
80  	 */
81  	private final BoundedConcurrentHashMap<String,ParameterMetadata> parameterMetadataCache;
82  
83  	public QueryPlanCache(final SessionFactoryImplementor factory) {
84  		this.factory = factory;
85  
86  		Integer maxParameterMetadataCount = ConfigurationHelper.getInteger(
87  				Environment.QUERY_PLAN_CACHE_PARAMETER_METADATA_MAX_SIZE,
88  				factory.getProperties()
89  		);
90  		if ( maxParameterMetadataCount == null ) {
91  			maxParameterMetadataCount = ConfigurationHelper.getInt(
92  					Environment.QUERY_PLAN_CACHE_MAX_STRONG_REFERENCES,
93  					factory.getProperties(),
94  					DEFAULT_PARAMETER_METADATA_MAX_COUNT
95  			);
96  		}
97  		Integer maxQueryPlanCount = ConfigurationHelper.getInteger(
98  				Environment.QUERY_PLAN_CACHE_MAX_SIZE,
99  				factory.getProperties()
100 		);
101 		if ( maxQueryPlanCount == null ) {
102 			maxQueryPlanCount = ConfigurationHelper.getInt(
103 					Environment.QUERY_PLAN_CACHE_MAX_SOFT_REFERENCES,
104 					factory.getProperties(),
105 					DEFAULT_QUERY_PLAN_MAX_COUNT
106 			);
107 		}
108 
109 		queryPlanCache = new BoundedConcurrentHashMap( maxQueryPlanCount, 20, BoundedConcurrentHashMap.Eviction.LIRS );
110 		parameterMetadataCache = new BoundedConcurrentHashMap<String, ParameterMetadata>(
111 				maxParameterMetadataCount,
112 				20,
113 				BoundedConcurrentHashMap.Eviction.LIRS
114 		);
115 
116 	}
117 
118 	/**
119 	 * Obtain the parameter metadata for given native-sql query.
120 	 * <p/>
121 	 * for native-sql queries, the param metadata is determined outside any relation to a query plan, because
122 	 * query plan creation and/or retrieval for a native-sql query depends on all of the return types having been
123 	 * set, which might not be the case up-front when param metadata would be most useful
124 	 *
125 	 * @param query The query
126 	 * @return The parameter metadata
127 	 */
128 	public ParameterMetadata getSQLParameterMetadata(final String query)  {
129 		ParameterMetadata value = parameterMetadataCache.get( query );
130 		if ( value == null ) {
131 			value = buildParameterMetadata( query );
132 			parameterMetadataCache.putIfAbsent( query, value );
133 		}
134 		return value;
135 	}
136 	
137 	private ParameterMetadata buildParameterMetadata(String query){
138 		ParamLocationRecognizer recognizer = ParamLocationRecognizer.parseLocations( query );
139 		final int size = recognizer.getOrdinalParameterLocationList().size();
140 		OrdinalParameterDescriptor[] ordinalDescriptors = new OrdinalParameterDescriptor[ size ];
141 		for ( int i = 0; i < size; i++ ) {
142 			final Integer position = recognizer.getOrdinalParameterLocationList().get( i );
143 			ordinalDescriptors[i] = new OrdinalParameterDescriptor( i, null, position );
144 		}
145 		Map<String, NamedParameterDescriptor> namedParamDescriptorMap = new HashMap<String, NamedParameterDescriptor>();
146 		Map<String, ParamLocationRecognizer.NamedParameterDescription> map = recognizer.getNamedParameterDescriptionMap();
147 		for ( final String name : map.keySet() ) {
148 			final ParamLocationRecognizer.NamedParameterDescription description = map.get( name );
149 			namedParamDescriptorMap.put(
150 					name,
151 					new NamedParameterDescriptor(
152 							name,
153 							null,
154 							description.buildPositionsArray(),
155 							description.isJpaStyle()
156 					)
157 			);
158 		}
159 		return new ParameterMetadata( ordinalDescriptors, namedParamDescriptorMap );
160 	}

162 	public HQLQueryPlan getHQLQueryPlan( String queryString, boolean shallow, Map enabledFilters)
163 			throws QueryException, MappingException {
164 		HQLQueryPlanKey key = new HQLQueryPlanKey( queryString, shallow, enabledFilters );
165 		HQLQueryPlan value = (HQLQueryPlan) queryPlanCache.get( key );
166 		if ( value == null ) {
167 			LOG.tracev( "Unable to locate HQL query plan in cache; generating ({0})", queryString );
168 			value = new HQLQueryPlan( queryString, shallow, enabledFilters, factory );
169 			queryPlanCache.putIfAbsent( key, value );
170 		} else {
171 			LOG.tracev( "Located HQL query plan in cache ({0})", queryString );
172 		}
173 		return value;
174 	}

178 	public FilterQueryPlan getFilterQueryPlan(String filterString, String collectionRole, boolean shallow, Map enabledFilters)
179 			throws QueryException, MappingException {
180 		FilterQueryPlanKey key =  new FilterQueryPlanKey( filterString, collectionRole, shallow, enabledFilters );
181 		FilterQueryPlan value = (FilterQueryPlan) queryPlanCache.get( key );
182 		if(value == null){
183 			LOG.tracev( "Unable to locate collection-filter query plan in cache; generating ({0} : {1} )",
184 										collectionRole, filterString );
185 			value = new FilterQueryPlan( filterString, collectionRole, shallow, enabledFilters,factory );
186 			queryPlanCache.putIfAbsent( key, value );
187 		} else {
188 			LOG.tracev( "Located collection-filter query plan in cache ({0} : {1})", collectionRole, filterString );
189 		}
190 		return value;
191 	}
192 
193 	public NativeSQLQueryPlan getNativeSQLQueryPlan(final NativeSQLQuerySpecification spec) {
194 		NativeSQLQueryPlan value = (NativeSQLQueryPlan) queryPlanCache.get( spec );
195 		if(value == null){
196 			LOG.tracev( "Unable to locate native-sql query plan in cache; generating ({0})", spec.getQueryString() );
197 			value = new NativeSQLQueryPlan( spec, factory);
198 			queryPlanCache.putIfAbsent( spec, value );
199 		} else {
200 			LOG.tracev( "Located native-sql query plan in cache ({0})", spec.getQueryString() );
201 		}
202 		return value;
203 	}

207 	public void cleanup() {
208 		LOG.trace( "Cleaning QueryPlan Cache" );
209 		queryPlanCache.clear();
210 		parameterMetadataCache.clear();
211 	}

213 	private static class HQLQueryPlanKey implements Serializable {
214 		private final String query;
215 		private final boolean shallow;
216 		private final Set<DynamicFilterKey> filterKeys;
217 		private final int hashCode;
218 
219 		public HQLQueryPlanKey(String query, boolean shallow, Map enabledFilters) {
220 			this.query = query;
221 			this.shallow = shallow;
222 			if ( CollectionHelper.isEmpty( enabledFilters ) ) {
223 				filterKeys = Collections.emptySet();
224 			}
225 			else {
226 				Set<DynamicFilterKey> tmp = new HashSet<DynamicFilterKey>(
227 						CollectionHelper.determineProperSizing( enabledFilters ),
228 						CollectionHelper.LOAD_FACTOR
229 				);
230 				for ( Object o : enabledFilters.values() ) {
231 					tmp.add( new DynamicFilterKey( (FilterImpl) o ) );
232 				}
233 				this.filterKeys = Collections.unmodifiableSet( tmp );
234 			}
235 
236 			int hash = query.hashCode();
237 			hash = 29 * hash + ( shallow ? 1 : 0 );
238 			hash = 29 * hash + filterKeys.hashCode();
239 			this.hashCode = hash;
240 		}
241 
242 		@Override
243 		public boolean equals(Object o) {
244 			if ( this == o ) {
245 				return true;
246 			}
247 			if ( o == null || getClass() != o.getClass() ) {
248 				return false;
249 			}
250 
251 			final HQLQueryPlanKey that = ( HQLQueryPlanKey ) o;
252 
253 			return shallow == that.shallow
254 					&& filterKeys.equals( that.filterKeys )
255 					&& query.equals( that.query );
256 
257 		}
258 
259 		@Override
260 		public int hashCode() {
261 			return hashCode;
262 		}
263 	}

265 	private static class DynamicFilterKey implements Serializable {
266 		private final String filterName;
267 		private final Map<String,Integer> parameterMetadata;
268 		private final int hashCode;
269 
270 		@SuppressWarnings({ "UnnecessaryBoxing" })
271 		private DynamicFilterKey(FilterImpl filter) {
272 			this.filterName = filter.getName();
273 			if ( filter.getParameters().isEmpty() ) {
274 				parameterMetadata = Collections.emptyMap();
275 			}
276 			else {
277 				parameterMetadata = new HashMap<String,Integer>(
278 						CollectionHelper.determineProperSizing( filter.getParameters() ),
279 						CollectionHelper.LOAD_FACTOR
280 				);
281 				for ( Object o : filter.getParameters().entrySet() ) {
282 					final Map.Entry entry = (Map.Entry) o;
283 					final String key = (String) entry.getKey();
284 					final Integer valueCount;
285 					if ( Collection.class.isInstance( entry.getValue() ) ) {
286 						valueCount = ( (Collection) entry.getValue() ).size();
287 					}
288 					else {
289 						valueCount = 1;
290 					}
291 					parameterMetadata.put( key, valueCount );
292 				}
293 			}
294 
295 			int hash = filterName.hashCode();
296 			hash = 31 * hash + parameterMetadata.hashCode();
297 			this.hashCode = hash;
298 		}
299 
300 		@Override
301 		public boolean equals(Object o) {
302 			if ( this == o ) {
303 				return true;
304 			}
305 			if ( o == null || getClass() != o.getClass() ) {
306 				return false;
307 			}
308 
309 			DynamicFilterKey that = ( DynamicFilterKey ) o;
310 
311 			return filterName.equals( that.filterName )
312 					&& parameterMetadata.equals( that.parameterMetadata );
313 
314 		}
315 
316 		@Override
317 		public int hashCode() {
318 			return hashCode;
319 		}
320 	}

323 		private final String query;
324 		private final String collectionRole;
325 		private final boolean shallow;
326 		private final Set<String> filterNames;
327 		private final int hashCode;
328 
329 		@SuppressWarnings({ "unchecked" })
330 		public FilterQueryPlanKey(String query, String collectionRole, boolean shallow, Map enabledFilters) {
331 			this.query = query;
332 			this.collectionRole = collectionRole;
333 			this.shallow = shallow;
334 
335 			if ( CollectionHelper.isEmpty( enabledFilters ) ) {
336 				this.filterNames = Collections.emptySet();
337 			}
338 			else {
339 				Set<String> tmp = new HashSet<String>();
340 				tmp.addAll( enabledFilters.keySet() );
341 				this.filterNames = Collections.unmodifiableSet( tmp );
342 
343 			}
344 
345 			int hash = query.hashCode();
346 			hash = 29 * hash + collectionRole.hashCode();
347 			hash = 29 * hash + ( shallow ? 1 : 0 );
348 			hash = 29 * hash + filterNames.hashCode();
349 			this.hashCode = hash;
350 		}
351 
352 		@Override
353 		public boolean equals(Object o) {
354 			if ( this == o ) {
355 				return true;
356 			}
357 			if ( o == null || getClass() != o.getClass() ) {
358 				return false;
359 			}
360 
361 			final FilterQueryPlanKey that = ( FilterQueryPlanKey ) o;
362 
363 			return shallow == that.shallow
364 					&& filterNames.equals( that.filterNames )
365 					&& query.equals( that.query )
366 					&& collectionRole.equals( that.collectionRole );
367 
368 		}
369 
370 		@Override
371 		public int hashCode() {
372 			return hashCode;
373 		}
