43  /**
44   * We need an entry to tell us all about the current state
45   * of a collection with respect to its persistent state
46   *
47   * @author Gavin King
48   */
49  public final class CollectionEntry implements Serializable {

51      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, CollectionEntry.class.getName());

53  	//ATTRIBUTES MAINTAINED BETWEEN FLUSH CYCLES
54  
55  	// session-start/post-flush persistent state
56  	private Serializable snapshot;
57  	// allow the CollectionSnapshot to be serialized
58  	private String role;
59  
60  	// "loaded" means the reference that is consistent
61  	// with the current database state
62  	private transient CollectionPersister loadedPersister;
63  	private Serializable loadedKey;
64  
65  	// ATTRIBUTES USED ONLY DURING FLUSH CYCLE
66  
67  	// during flush, we navigate the object graph to
68  	// collections and decide what to do with them
69  	private transient boolean reached;
70  	private transient boolean processed;
71  	private transient boolean doupdate;
72  	private transient boolean doremove;
73  	private transient boolean dorecreate;
74  	// if we instantiate a collection during the flush() process,
75  	// we must ignore it for the rest of the flush()
76  	private transient boolean ignore;
77  
78  	// "current" means the reference that was found during flush()
79  	private transient CollectionPersister currentPersister;
80  	private transient Serializable currentKey;

82  	/**
83  	 * For newly wrapped collections, or dereferenced collection wrappers
84  	 */
85  	public CollectionEntry(CollectionPersister persister, PersistentCollection collection) {
86  		// new collections that get found + wrapped
87  		// during flush shouldn't be ignored
88  		ignore = false;
89  
90  		collection.clearDirty(); //a newly wrapped collection is NOT dirty (or we get unnecessary version updates)
91  
92  		snapshot = persister.isMutable() ?
93  				collection.getSnapshot(persister) :
94  				null;
95  		collection.setSnapshot(loadedKey, role, snapshot);
96  	}

98  	/**
99  	 * For collections just loaded from the database
100 	 */
101 	public CollectionEntry(
102 			final PersistentCollection collection,
103 			final CollectionPersister loadedPersister,
104 			final Serializable loadedKey,
105 			final boolean ignore
106 	) {
107 		this.ignore=ignore;
108 
109 		//collection.clearDirty()
110 
111 		this.loadedKey = loadedKey;
112 		setLoadedPersister(loadedPersister);
113 
114 		collection.setSnapshot(loadedKey, role, null);
115 
116 		//postInitialize() will be called after initialization
117 	}

120 	 * For uninitialized detached collections
121 	 */
122 	public CollectionEntry(CollectionPersister loadedPersister, Serializable loadedKey) {
123 		// detached collection wrappers that get found + reattached
124 		// during flush shouldn't be ignored
125 		ignore = false;
126 
127 		//collection.clearDirty()
128 
129 		this.loadedKey = loadedKey;
130 		setLoadedPersister(loadedPersister);
131 	}

133 	/**
134 	 * For initialized detached collections
135 	 */
136 	public CollectionEntry(PersistentCollection collection, SessionFactoryImplementor factory) throws MappingException {
137 		// detached collections that get found + reattached
138 		// during flush shouldn't be ignored
139 		ignore = false;
140 
141 		loadedKey = collection.getKey();
142 		setLoadedPersister( factory.getCollectionPersister( collection.getRole() ) );
143 
144 		snapshot = collection.getStoredSnapshot();
145 	}

147 	/**
148 	 * Used from custom serialization.
149 	 *
150 	 * @see #serialize
151 	 * @see #deserialize
152 	 */
153 	private CollectionEntry(
154 			String role,
155 	        Serializable snapshot,
156 	        Serializable loadedKey,
157 	        SessionFactoryImplementor factory) {
158 		this.role = role;
159 		this.snapshot = snapshot;
160 		this.loadedKey = loadedKey;
161 		if ( role != null ) {
162 			afterDeserialize( factory );
163 		}
164 	}

166 	/**
167 	 * Determine if the collection is "really" dirty, by checking dirtiness
168 	 * of the collection elements, if necessary
169 	 */
170 	private void dirty(PersistentCollection collection) throws HibernateException {
171 
172 		boolean forceDirty = collection.wasInitialized() &&
173 				!collection.isDirty() && //optimization
174 				getLoadedPersister() != null &&
175 				getLoadedPersister().isMutable() && //optimization
176 				( collection.isDirectlyAccessible() || getLoadedPersister().getElementType().isMutable() ) && //optimization
177 				!collection.equalsSnapshot( getLoadedPersister() );
178 
179 		if ( forceDirty ) {
180 			collection.dirty();
181 		}
182 
183 	}

185 	public void preFlush(PersistentCollection collection) throws HibernateException {
186 		if ( loadedKey == null && collection.getKey() != null ) {
187 			loadedKey = collection.getKey();
188 		}
189 
190 		boolean nonMutableChange = collection.isDirty() &&
191 				getLoadedPersister()!=null &&
192 				!getLoadedPersister().isMutable();
193 		if (nonMutableChange) {
194 			throw new HibernateException(
195 					"changed an immutable collection instance: " +
196 					MessageHelper.collectionInfoString( getLoadedPersister().getRole(), getLoadedKey() )
197 				);
198 		}
199 
200 		dirty(collection);
201 
202 		if ( LOG.isDebugEnabled() && collection.isDirty() && getLoadedPersister() != null ) {
203 			LOG.debugf( "Collection dirty: %s",
204 					MessageHelper.collectionInfoString( getLoadedPersister().getRole(), getLoadedKey() ) );
205 		}
206 
207 		setDoupdate(false);
208 		setDoremove(false);
209 		setDorecreate(false);
210 		setReached(false);
211 		setProcessed(false);
212 	}
213 
214 	public void postInitialize(PersistentCollection collection) throws HibernateException {
215 		snapshot = getLoadedPersister().isMutable() ?
216 				collection.getSnapshot( getLoadedPersister() ) :
217 				null;
218 		collection.setSnapshot(loadedKey, role, snapshot);
219 		if ( getLoadedPersister().getBatchSize() > 1 ) {
220 			( (AbstractPersistentCollection) collection ).getSession()
221 					.getPersistenceContext()
222 					.getBatchFetchQueue()
223 					.removeBatchLoadableCollection( this );
224 		}
225 	}
226 
227 	/**
228 	 * Called after a successful flush
229 	 */
230 	public void postFlush(PersistentCollection collection) throws HibernateException {
231 		if ( isIgnore() ) {
232 			ignore = false;
233 		}
234 		else if ( !isProcessed() ) {
235 			throw new AssertionFailure( "collection [" + collection.getRole() + "] was not processed by flush()" );
236 		}
237 		collection.setSnapshot(loadedKey, role, snapshot);
238 	}
239 
240 	/**
241 	 * Called after execution of an action
242 	 */
243 	public void afterAction(PersistentCollection collection) {
244 		loadedKey = getCurrentKey();
245 		setLoadedPersister( getCurrentPersister() );
246 
247 		boolean resnapshot = collection.wasInitialized() &&
248 				( isDoremove() || isDorecreate() || isDoupdate() );
249 		if ( resnapshot ) {
250 			snapshot = loadedPersister==null || !loadedPersister.isMutable() ?
251 					null :
252 					collection.getSnapshot(loadedPersister); //re-snapshot
253 		}
254 
255 		collection.postAction();
256 	}

270 	private boolean fromMerge = false;
271 
272 	/**
273 	 * Reset the stored snapshot for both the persistent collection and this collection entry. 
274 	 * Used during the merge of detached collections.
275 	 * 
276 	 * @param collection the persistentcollection to be updated
277 	 * @param storedSnapshot the new stored snapshot
278 	 */
279 	public void resetStoredSnapshot(PersistentCollection collection, Serializable storedSnapshot) {
280 		LOG.debugf("Reset storedSnapshot to %s for %s", storedSnapshot, this);
281 
282 		if ( fromMerge ) {
283 			return; // EARLY EXIT!
284 		}
285 
286 		snapshot = storedSnapshot;
287 		collection.setSnapshot( loadedKey, role, snapshot );
288 		fromMerge = true;
289 	}

291 	private void setLoadedPersister(CollectionPersister persister) {
292 		loadedPersister = persister;
293 		setRole( persister == null ? null : persister.getRole() );
294 	}

296 	void afterDeserialize(SessionFactoryImplementor factory) {
297 		loadedPersister = ( factory == null ? null : factory.getCollectionPersister(role) );
298 	}

383 	@Override
384     public String toString() {
385 		String result = "CollectionEntry" +
386 				MessageHelper.collectionInfoString( loadedPersister.getRole(), loadedKey );
387 		if (currentPersister!=null) {
388 			result += "->" +
389 					MessageHelper.collectionInfoString( currentPersister.getRole(), currentKey );
390 		}
391 		return result;
392 	}

394 	/**
395 	 * Get the collection orphans (entities which were removed from the collection)
396 	 */
397 	public Collection getOrphans(String entityName, PersistentCollection collection)
398 	throws HibernateException {
399 		if (snapshot==null) {
400 			throw new AssertionFailure("no collection snapshot for orphan delete");
401 		}
402 		return collection.getOrphans( snapshot, entityName );
403 	}

405 	public boolean isSnapshotEmpty(PersistentCollection collection) {
406 		//TODO: does this really need to be here?
407 		//      does the collection already have
408 		//      it's own up-to-date snapshot?
409 		return collection.wasInitialized() &&
410 			( getLoadedPersister()==null || getLoadedPersister().isMutable() ) &&
411 			collection.isSnapshotEmpty( getSnapshot() );
412 	}

416 	/**
417 	 * Custom serialization routine used during serialization of a
418 	 * Session/PersistenceContext for increased performance.
419 	 *
420 	 * @param oos The stream to which we should write the serial data.
421 	 * @throws java.io.IOException
422 	 */
423 	public void serialize(ObjectOutputStream oos) throws IOException {
424 		oos.writeObject( role );
425 		oos.writeObject( snapshot );
426 		oos.writeObject( loadedKey );
427 	}

429 	/**
430 	 * Custom deserialization routine used during deserialization of a
431 	 * Session/PersistenceContext for increased performance.
432 	 *
433 	 * @param ois The stream from which to read the entry.
434 	 * @param session The session being deserialized.
435 	 * @return The deserialized CollectionEntry
436 	 * @throws IOException
437 	 * @throws ClassNotFoundException
438 	 */
439 	public static CollectionEntry deserialize(
440 			ObjectInputStream ois,
441 	        SessionImplementor session) throws IOException, ClassNotFoundException {
442 		return new CollectionEntry(
443 				( String ) ois.readObject(),
444 		        ( Serializable ) ois.readObject(),
445 		        ( Serializable ) ois.readObject(),
446 		        ( session == null ? null : session.getFactory() )
447 		);
448 	}
