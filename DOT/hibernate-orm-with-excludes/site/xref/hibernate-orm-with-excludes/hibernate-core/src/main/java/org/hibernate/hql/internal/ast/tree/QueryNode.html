25  package org.hibernate.hql.internal.ast.tree;
26  import antlr.SemanticException;
27  import antlr.collections.AST;
28  import org.jboss.logging.Logger;
29  
30  import org.hibernate.hql.internal.antlr.HqlSqlTokenTypes;
31  import org.hibernate.hql.internal.antlr.SqlTokenTypes;
32  import org.hibernate.hql.internal.ast.util.ASTUtil;
33  import org.hibernate.hql.internal.ast.util.ColumnHelper;
34  import org.hibernate.internal.CoreMessageLogger;
35  import org.hibernate.type.Type;

37  /**
38   * Defines a top-level AST node representing an HQL select statement.
39   *
40   * @author Joshua Davis
41   */
42  public class QueryNode extends AbstractRestrictableStatement implements SelectExpression {
43  
44      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, QueryNode.class.getName());
45  
46  	private OrderByClause orderByClause;
47  	private int scalarColumnIndex = -1;
48  
49  	/**
50  	 * @see Statement#getStatementType()
51  	 */
52  	public int getStatementType() {
53  		return HqlSqlTokenTypes.QUERY;
54  	}
55  
56  	/**
57  	 * @see Statement#needsExecutor()
58  	 */
59  	public boolean needsExecutor() {
60  		return false;
61  	}
62  
63  	@Override
64      protected int getWhereClauseParentTokenType() {
65  		return SqlTokenTypes.FROM;
66  	}
67  
68  	@Override
69      protected CoreMessageLogger getLog() {
70          return LOG;
71  	}
72  
73  	/**
74  	 * Locate the select clause that is part of this select statement.
75  	 * </p>
76  	 * Note, that this might return null as derived select clauses (i.e., no
77  	 * select clause at the HQL-level) get generated much later than when we
78  	 * get created; thus it depends upon lifecycle.
79  	 *
80  	 * @return Our select clause, or null.
81  	 */
82  	public final SelectClause getSelectClause() {
83  		// Due to the complexity in initializing the SelectClause, do not generate one here.
84  		// If it is not found; simply return null...
85  		//
86  		// Also, do not cache since it gets generated well after we are created.
87  		return ( SelectClause ) ASTUtil.findTypeInChildren( this, SqlTokenTypes.SELECT_CLAUSE );
88  	}
89  
90  	public final boolean hasOrderByClause() {
91  		OrderByClause orderByClause = locateOrderByClause();
92  		return orderByClause != null && orderByClause.getNumberOfChildren() > 0;
93  	}
94  
95  	public final OrderByClause getOrderByClause() {
96  		if ( orderByClause == null ) {
97  			orderByClause = locateOrderByClause();
98  
99  			// if there is no order by, make one
100 			if ( orderByClause == null ) {
101 				LOG.debug( "getOrderByClause() : Creating a new ORDER BY clause" );
102 				orderByClause = ( OrderByClause ) ASTUtil.create( getWalker().getASTFactory(), SqlTokenTypes.ORDER, "ORDER" );
103 
104 				// Find the WHERE; if there is no WHERE, find the FROM...
105 				AST prevSibling = ASTUtil.findTypeInChildren( this, SqlTokenTypes.WHERE );
106 				if ( prevSibling == null ) {
107 					prevSibling = ASTUtil.findTypeInChildren( this, SqlTokenTypes.FROM );
108 				}
109 
110 				// Now, inject the newly built ORDER BY into the tree
111 				orderByClause.setNextSibling( prevSibling.getNextSibling() );
112 				prevSibling.setNextSibling( orderByClause );
113 			}
114 		}
115 		return orderByClause;
116 	}
117 
118 	private OrderByClause locateOrderByClause() {
119 		return ( OrderByClause ) ASTUtil.findTypeInChildren( this, SqlTokenTypes.ORDER );
120 	}
121 
122 
123 	private String alias;
124 
125 	public String getAlias() {
126 		return alias;
127 	}
128 
129 	public FromElement getFromElement() {
130 		return null;
131 	}
132 
133 	public boolean isConstructor() {
134 		return false;
135 	}
136 
137 	public boolean isReturnableEntity() throws SemanticException {
138 		return false;
139 	}
140 
141 	public boolean isScalar() throws SemanticException {
142 		return true;
143 	}
144 
145 	public void setAlias(String alias) {
146 		this.alias = alias;
147 	}
148 
149 	public void setScalarColumn(int i) throws SemanticException {
150 		scalarColumnIndex = i;
151 		setScalarColumnText( i );
152 	}
153 
154 	public int getScalarColumnIndex() {
155 		return scalarColumnIndex;
156 	}
157 
158 	public void setScalarColumnText(int i) throws SemanticException {
159 		ColumnHelper.generateSingleScalarColumn( this, i );
160 	}
161 
162 	@Override
163     public Type getDataType() {
164 		return ( (SelectExpression) getSelectClause().getFirstSelectExpression() ).getDataType();
165 	}
166 
167 }
