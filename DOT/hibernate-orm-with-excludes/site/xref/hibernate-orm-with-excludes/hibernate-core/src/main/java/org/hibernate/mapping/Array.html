24  package org.hibernate.mapping;
25  
26  import org.hibernate.MappingException;
27  import org.hibernate.cfg.Mappings;
28  import org.hibernate.internal.util.ReflectHelper;
29  import org.hibernate.type.CollectionType;
30  import org.hibernate.type.PrimitiveType;
31  
32  /**
33   * An array mapping has a primary key consisting of the key columns + index column.
34   *
35   * @author Gavin King
36   */
37  public class Array extends List {
38  
39  	private String elementClassName;
40  
41  	public Array(Mappings mappings, PersistentClass owner) {
42  		super( mappings, owner );
43  	}
44  
45  	public Class getElementClass() throws MappingException {
46  		if (elementClassName==null) {
47  			org.hibernate.type.Type elementType = getElement().getType();
48  			return isPrimitiveArray() ?
49  				( (PrimitiveType) elementType ).getPrimitiveClass() :
50  				elementType.getReturnedClass();
51  		}
52  		else {
53  			try {
54  				return ReflectHelper.classForName(elementClassName);
55  			}
56  			catch (ClassNotFoundException cnfe) {
57  				throw new MappingException(cnfe);
58  			}
59  		}
60  	}
61  
62  	@Override
63      public CollectionType getDefaultCollectionType() throws MappingException {
64  		return getMappings().getTypeResolver()
65  				.getTypeFactory()
66  				.array( getRole(), getReferencedPropertyName(), getElementClass() );
67  	}
68  
69  	@Override
70      public boolean isArray() {
71  		return true;
72  	}
73  
74  	/**
75  	 * @return Returns the elementClassName.
76  	 */
77  	public String getElementClassName() {
78  		return elementClassName;
79  	}
80  	/**
81  	 * @param elementClassName The elementClassName to set.
82  	 */
83  	public void setElementClassName(String elementClassName) {
84  		this.elementClassName = elementClassName;
85  	}
86  
87  	@Override
88      public Object accept(ValueVisitor visitor) {
89  		return visitor.accept(this);
90  	}
91  }
