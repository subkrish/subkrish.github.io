24  package org.hibernate.service.internal;
25  
26  import java.util.List;
27  import java.util.Map;
28  
29  import org.hibernate.service.BootstrapServiceRegistry;
30  import org.hibernate.service.Service;
31  import org.hibernate.service.ServiceRegistry;
32  import org.hibernate.service.spi.BasicServiceInitiator;
33  import org.hibernate.service.spi.Configurable;
34  import org.hibernate.service.spi.ServiceBinding;
35  import org.hibernate.service.spi.ServiceInitiator;
36  
37  /**
38   * Hibernate implementation of the standard service registry.
39   *
40   * @author Steve Ebersole
41   */
42  public class StandardServiceRegistryImpl extends AbstractServiceRegistryImpl implements ServiceRegistry {
43  	private final Map configurationValues;
44  
45  	@SuppressWarnings( {"unchecked"})
46  	public StandardServiceRegistryImpl(
47  			BootstrapServiceRegistry bootstrapServiceRegistry,
48  			List<BasicServiceInitiator> serviceInitiators,
49  			List<ProvidedService> providedServices,
50  			Map<?, ?> configurationValues) {
51  		this( true, bootstrapServiceRegistry, serviceInitiators, providedServices, configurationValues );
52  	}
53  	
54  	@SuppressWarnings( {"unchecked"})
55  	public StandardServiceRegistryImpl(
56  			boolean autoCloseRegistry,
57  			BootstrapServiceRegistry bootstrapServiceRegistry,
58  			List<BasicServiceInitiator> serviceInitiators,
59  			List<ProvidedService> providedServices,
60  			Map<?, ?> configurationValues) {
61  		super( bootstrapServiceRegistry, autoCloseRegistry );
62  
63  		this.configurationValues = configurationValues;
64  
65  		// process initiators
66  		for ( ServiceInitiator initiator : serviceInitiators ) {
67  			createServiceBinding( initiator );
68  		}
69  
70  		// then, explicitly provided service instances
71  		for ( ProvidedService providedService : providedServices ) {
72  			createServiceBinding( providedService );
73  		}
74  	}
75  
76  	@Override
77  	public <R extends Service> R initiateService(ServiceInitiator<R> serviceInitiator) {
78  		// todo : add check/error for unexpected initiator types?
79  		return ( (BasicServiceInitiator<R>) serviceInitiator ).initiateService( configurationValues, this );
80  	}
81  
82  	@Override
83  	public <R extends Service> void configureService(ServiceBinding<R> serviceBinding) {
84  		if ( Configurable.class.isInstance( serviceBinding.getService() ) ) {
85  			( (Configurable) serviceBinding.getService() ).configure( configurationValues );
86  		}
87  	}
88  }
