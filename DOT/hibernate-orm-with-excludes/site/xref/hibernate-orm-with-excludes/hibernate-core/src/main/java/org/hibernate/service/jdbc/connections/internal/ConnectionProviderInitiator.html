101 	@Override
102 	public ConnectionProvider initiateService(Map configurationValues, ServiceRegistryImplementor registry) {
103 		final MultiTenancyStrategy strategy = MultiTenancyStrategy.determineMultiTenancyStrategy(  configurationValues );
104 		if ( strategy == MultiTenancyStrategy.DATABASE || strategy == MultiTenancyStrategy.SCHEMA ) {
105 			// nothing to do, but given the separate hierarchies have to handle this here.
106 			return null;
107 		}
108 
109 		final ClassLoaderService classLoaderService = registry.getService( ClassLoaderService.class );
110 
111 		ConnectionProvider connectionProvider = null;
112 		String providerClassName = getConfiguredConnectionProviderName( configurationValues );
113 		if ( providerClassName != null ) {
114 			connectionProvider = instantiateExplicitConnectionProvider( providerClassName, classLoaderService );
115 		}
116 		else if ( configurationValues.get( Environment.DATASOURCE ) != null ) {
117 			connectionProvider = new DatasourceConnectionProviderImpl();
118 		}
119 
120 		if ( connectionProvider == null ) {
121 			if ( c3p0ConfigDefined( configurationValues ) && c3p0ProviderPresent( classLoaderService ) ) {
122 				connectionProvider = instantiateExplicitConnectionProvider( C3P0_PROVIDER_CLASS_NAME,
123 						classLoaderService
124 				);
125 			}
126 		}
127 
128 		if ( connectionProvider == null ) {
129 			if ( proxoolConfigDefined( configurationValues ) && proxoolProviderPresent( classLoaderService ) ) {
130 				connectionProvider = instantiateExplicitConnectionProvider( PROXOOL_PROVIDER_CLASS_NAME,
131 						classLoaderService
132 				);
133 			}
134 		}
135 
136 		if ( connectionProvider == null ) {
137 			if ( configurationValues.get( Environment.URL ) != null ) {
138 				connectionProvider = new DriverManagerConnectionProviderImpl();
139 			}
140 		}
141 
142 		if ( connectionProvider == null ) {
143             LOG.noAppropriateConnectionProvider();
144 			connectionProvider = new UserSuppliedConnectionProviderImpl();
145 		}
146 
147 
148 		final Map injectionData = (Map) configurationValues.get( INJECTION_DATA );
149 		if ( injectionData != null && injectionData.size() > 0 ) {
150 			final ConnectionProvider theConnectionProvider = connectionProvider;
151 			new BeanInfoHelper( connectionProvider.getClass() ).applyToBeanInfo(
152 					connectionProvider,
153 					new BeanInfoHelper.BeanInfoDelegate() {
154 						public void processBeanInfo(BeanInfo beanInfo) throws Exception {
155 							PropertyDescriptor[] descritors = beanInfo.getPropertyDescriptors();
156 							for ( int i = 0, size = descritors.length; i < size; i++ ) {
157 								String propertyName = descritors[i].getName();
158 								if ( injectionData.containsKey( propertyName ) ) {
159 									Method method = descritors[i].getWriteMethod();
160 									method.invoke(
161 											theConnectionProvider,
162 											injectionData.get( propertyName )
163 									);
164 								}
165 							}
166 						}
167 					}
168 			);
169 		}
170 
171 		return connectionProvider;
172 	}
173 
174 	private String getConfiguredConnectionProviderName( Map configurationValues ) {
175 		String providerClassName = ( String ) configurationValues.get( Environment.CONNECTION_PROVIDER );
176 		if ( LEGACY_CONNECTION_PROVIDER_MAPPING.containsKey( providerClassName ) ) {
177 			String actualProviderClassName = LEGACY_CONNECTION_PROVIDER_MAPPING.get( providerClassName );
178             LOG.providerClassDeprecated(providerClassName, actualProviderClassName);
179 			providerClassName = actualProviderClassName;
180 		}
181 		return providerClassName;
182 	}
183 
184 	private ConnectionProvider instantiateExplicitConnectionProvider(
185 			String providerClassName,
186 			ClassLoaderService classLoaderService) {
187 		try {
188             LOG.instantiatingExplicitConnectionProvider( providerClassName );
189 			return (ConnectionProvider) classLoaderService.classForName( providerClassName ).newInstance();
190 		}
191 		catch ( Exception e ) {
192 			throw new HibernateException( "Could not instantiate connection provider [" + providerClassName + "]", e );
193 		}
194 	}
195 
196 	private boolean c3p0ProviderPresent(ClassLoaderService classLoaderService) {
197 		try {
198 			classLoaderService.classForName( C3P0_PROVIDER_CLASS_NAME );
199 		}
200 		catch ( Exception e ) {
201             LOG.c3p0ProviderClassNotFound(C3P0_PROVIDER_CLASS_NAME);
202 			return false;
203 		}
204 		return true;
205 	}
206 
207 	private static boolean c3p0ConfigDefined(Map configValues) {
208 		for ( Object key : configValues.keySet() ) {
209 			if ( String.class.isInstance( key )
210 					&& ( (String) key ).startsWith( AvailableSettings.C3P0_CONFIG_PREFIX ) ) {
211 				return true;
212 			}
213 		}
214 		return false;
215 	}
216 
217 	private boolean proxoolProviderPresent(ClassLoaderService classLoaderService) {
218 		try {
219 			classLoaderService.classForName( PROXOOL_PROVIDER_CLASS_NAME );
220 		}
221 		catch ( Exception e ) {
222             LOG.proxoolProviderClassNotFound(PROXOOL_PROVIDER_CLASS_NAME);
223 			return false;
224 		}
225 		return true;
226 	}
227 
228 	private static boolean proxoolConfigDefined(Map configValues) {
229 		for ( Object key : configValues.keySet() ) {
230 			if ( String.class.isInstance( key )
231 					&& ( (String) key ).startsWith( AvailableSettings.PROXOOL_CONFIG_PREFIX ) ) {
232 				return true;
233 			}
234 		}
235 		return false;
236 	}

238 	/**
239 	 * Build the connection properties capable of being passed to the {@link java.sql.DriverManager#getConnection}
240 	 * forms taking {@link Properties} argument.  We seek out all keys in the passed map which start with
241 	 * {@code hibernate.connection.}, using them to create a new {@link Properties} instance.  The keys in this
242 	 * new {@link Properties} have the {@code hibernate.connection.} prefix trimmed.
243 	 *
244 	 * @param properties The map from which to build the connection specific properties.
245 	 *
246 	 * @return The connection properties.
247 	 */
248 	public static Properties getConnectionProperties(Map<?,?> properties) {
249 		Properties result = new Properties();
250 		for ( Map.Entry entry : properties.entrySet() ) {
251 			if ( ! ( String.class.isInstance( entry.getKey() ) ) || ! String.class.isInstance( entry.getValue() ) ) {
252 				continue;
253 			}
254 			final String key = (String) entry.getKey();
255 			final String value = (String) entry.getValue();
256 			if ( key.startsWith( Environment.CONNECTION_PREFIX ) ) {
257 				if ( SPECIAL_PROPERTIES.contains( key ) ) {
258 					if ( Environment.USER.equals( key ) ) {
259 						result.setProperty( "user", value );
260 					}
261 				}
262 				else {
263 					result.setProperty(
264 							key.substring( Environment.CONNECTION_PREFIX.length() + 1 ),
265 							value
266 					);
267 				}
268 			}
269 			else if ( CONDITIONAL_PROPERTIES.containsKey( key ) ) {
270 				result.setProperty( CONDITIONAL_PROPERTIES.get( key ), value );
271 			}
272 		}
273 		return result;
274 	}

276 	private static final Set<String> SPECIAL_PROPERTIES;
277 
278 	static {
279 		SPECIAL_PROPERTIES = new HashSet<String>();
280 		SPECIAL_PROPERTIES.add( Environment.DATASOURCE );
281 		SPECIAL_PROPERTIES.add( Environment.URL );
282 		SPECIAL_PROPERTIES.add( Environment.CONNECTION_PROVIDER );
283 		SPECIAL_PROPERTIES.add( Environment.POOL_SIZE );
284 		SPECIAL_PROPERTIES.add( Environment.ISOLATION );
285 		SPECIAL_PROPERTIES.add( Environment.DRIVER );
286 		SPECIAL_PROPERTIES.add( Environment.USER );
287 	}
