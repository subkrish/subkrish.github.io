24  package org.hibernate.ejb.criteria.expression;
25  
26  import java.io.Serializable;
27  import java.util.ArrayList;
28  import java.util.List;
29  import javax.persistence.Tuple;
30  import javax.persistence.criteria.CompoundSelection;
31  import javax.persistence.criteria.Selection;
32  
33  import org.hibernate.ejb.criteria.CriteriaBuilderImpl;
34  import org.hibernate.ejb.criteria.CriteriaQueryCompiler;
35  import org.hibernate.ejb.criteria.ParameterRegistry;
36  import org.hibernate.ejb.criteria.Renderable;
37  import org.hibernate.ejb.criteria.TupleElementImplementor;
38  import org.hibernate.ejb.criteria.ValueHandlerFactory;
39  
40  /**
41   * The Hibernate implementation of the JPA {@link CompoundSelection}
42   * contract.
43   *
44   * @author Steve Ebersole
45   */
46  public class CompoundSelectionImpl<X>
47  		extends SelectionImpl<X>
48  		implements CompoundSelection<X>, Renderable, Serializable {
49  	private final boolean isConstructor;
50  	private List<Selection<?>> selectionItems;
51  
52  	public CompoundSelectionImpl(
53  			CriteriaBuilderImpl criteriaBuilder,
54  			Class<X> javaType,
55  			List<Selection<?>> selectionItems) {
56  		super( criteriaBuilder, javaType );
57  		this.isConstructor = !javaType.isArray() && !Tuple.class.isAssignableFrom( javaType );
58  		this.selectionItems = selectionItems;
59  	}
60  
61  	@Override
62  	public boolean isCompoundSelection() {
63  		return true;
64  	}
65  
66  	@Override
67  	public List<Selection<?>> getCompoundSelectionItems() {
68  		return selectionItems;
69  	}
70  
71  	@Override
72  	public List<ValueHandlerFactory.ValueHandler> getValueHandlers() {
73  		if ( isConstructor ) {
74  			return null;
75  		}
76  		boolean foundHandlers = false;
77  		ArrayList<ValueHandlerFactory.ValueHandler> valueHandlers = new ArrayList<ValueHandlerFactory.ValueHandler>();
78  		for ( Selection selection : getCompoundSelectionItems() ) {
79  			ValueHandlerFactory.ValueHandler valueHandler = ( (TupleElementImplementor) selection ).getValueHandler();
80  			valueHandlers.add( valueHandler );
81  			foundHandlers = foundHandlers || valueHandler != null;
82  		}
83  		return foundHandlers ? null : valueHandlers;
84  	}
85  
86  	public void registerParameters(ParameterRegistry registry) {
87  		for ( Selection selectionItem : getCompoundSelectionItems() ) {
88  			Helper.possibleParameter(selectionItem, registry);
89  		}
90  	}
91  
92  	public String render(CriteriaQueryCompiler.RenderingContext renderingContext) {
93  		StringBuilder buff = new StringBuilder();
94  		if ( isConstructor ) {
95  			buff.append( "new " ).append( getJavaType().getName() ).append( '(' );
96  		}
97  		String sep = "";
98  		for ( Selection selection : selectionItems ) {
99  			buff.append( sep )
100 					.append( ( (Renderable) selection ).renderProjection( renderingContext ) );
101 			sep = ", ";
102 		}
103 		if ( isConstructor ) {
104 			buff.append( ')' );
105 		}
106 		return buff.toString();
107 	}
108 
109 	public String renderProjection(CriteriaQueryCompiler.RenderingContext renderingContext) {
110 		return render( renderingContext );
111 	}
112 }
