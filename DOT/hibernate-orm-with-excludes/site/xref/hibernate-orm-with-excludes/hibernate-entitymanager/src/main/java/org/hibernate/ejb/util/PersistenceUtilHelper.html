29  	public static LoadState isLoadedWithoutReference(Object proxy, String property, MetadataCache cache) {
30  		Object entity;
31  		boolean sureFromUs = false;
32  		if ( proxy instanceof HibernateProxy ) {
33  			LazyInitializer li = ( ( HibernateProxy ) proxy ).getHibernateLazyInitializer();
34  			if ( li.isUninitialized() ) {
35  				return LoadState.NOT_LOADED;
36  			}
37  			else {
38  				entity = li.getImplementation();
39  			}
40  			sureFromUs = true;
41  		}
42  		else {
43  			entity = proxy;
44  		}
45  
46  		//we are instrumenting but we can't assume we are the only ones
47  		if ( FieldInterceptionHelper.isInstrumented( entity ) ) {
48  			FieldInterceptor interceptor = FieldInterceptionHelper.extractFieldInterceptor( entity );
49  			final boolean isInitialized = interceptor == null || interceptor.isInitialized( property );
50  			LoadState state;
51  			if (isInitialized && interceptor != null) {
52  				//property is loaded according to bytecode enhancement, but is it loaded as far as association?
53  				//it's ours, we can read
54  				state = isLoaded( get( entity, property, cache ) );
55  				//it's ours so we know it's loaded
56  				if (state == LoadState.UNKNOWN) state = LoadState.LOADED;
57  			}
58  			else if ( interceptor != null && (! isInitialized)) {
59  				state = LoadState.NOT_LOADED;
60  			}
61  			else if ( sureFromUs ) { //interceptor == null
62  				//property is loaded according to bytecode enhancement, but is it loaded as far as association?
63  				//it's ours, we can read
64  				state = isLoaded( get( entity, property, cache ) );
65  				//it's ours so we know it's loaded
66  				if (state == LoadState.UNKNOWN) state = LoadState.LOADED;
67  			}
68  			else {
69  				state = LoadState.UNKNOWN;
70  			}
71  
72  			return state;
73  		}
74  		else {
75  			//can't do sureFromUs ? LoadState.LOADED : LoadState.UNKNOWN;
76  			//is that an association?
77  			return LoadState.UNKNOWN;
78  		}
79  	}

81  	public static LoadState isLoadedWithReference(Object proxy, String property, MetadataCache cache) {
82  		//for sure we don't instrument and for sure it's not a lazy proxy
83  		Object object = get(proxy, property, cache);
84  		return isLoaded( object );
85  	}

87  	private static Object get(Object proxy, String property, MetadataCache cache) {
88  		final Class<?> clazz = proxy.getClass();
89  
90  		try {
91  			Member member = cache.getMember( clazz, property );
92  			if (member instanceof Field) {
93  				return ( (Field) member ).get( proxy );
94  			}
95  			else if (member instanceof Method) {
96  				return ( (Method) member ).invoke( proxy );
97  			}
98  			else {
99  				throw new AssertionFailure( "Member object neither Field nor Method: " + member);
100 			}
101 		}
102 		catch ( IllegalAccessException e ) {
103 			throw new PersistenceException( "Unable to access field or method: "
104 							+ clazz + "#"
105 							+ property, e);
106 		}
107 		catch ( InvocationTargetException e ) {
108 			throw new PersistenceException( "Unable to access field or method: "
109 							+ clazz + "#"
110 							+ property, e);
111 		}
112 	}
113 
114 	private static void setAccessibility(Member member) {
115 		//Sun's ease of use, sigh...
116 		( ( AccessibleObject ) member ).setAccessible( true );
117 	}

119 	public static LoadState isLoaded(Object o) {
120 		if ( o instanceof HibernateProxy ) {
121 			final boolean isInitialized = !( ( HibernateProxy ) o ).getHibernateLazyInitializer().isUninitialized();
122 			return isInitialized ? LoadState.LOADED : LoadState.NOT_LOADED;
123 		}
124 		else if ( o instanceof PersistentCollection ) {
125 			final boolean isInitialized = ( (PersistentCollection) o ).wasInitialized();
126 			return isInitialized ? LoadState.LOADED : LoadState.NOT_LOADED;
127 		}
128 		else {
129 			return LoadState.UNKNOWN;
130 		}
131 	}

133 	/**
134 	 * Returns the method with the specified name or <code>null</code> if it does not exist.
135 	 *
136 	 * @param clazz The class to check.
137 	 * @param methodName The method name.
138 	 *
139 	 * @return Returns the method with the specified name or <code>null</code> if it does not exist.
140 	 */
141 	private static Method getMethod(Class<?> clazz, String methodName) {
142 		try {
143 			char string[] = methodName.toCharArray();
144 			string[0] = Character.toUpperCase( string[0] );
145 			methodName = new String( string );
146 			try {
147 				return clazz.getDeclaredMethod( "get" + methodName );
148 			}
149 			catch ( NoSuchMethodException e ) {
150 				return clazz.getDeclaredMethod( "is" + methodName );
151 			}
152 		}
153 		catch ( NoSuchMethodException e ) {
154 			return null;
155 		}
156 	}

158 	/**
159 	 * Cache hierarchy and member resolution in a weak hash map
160 	 */
161 	//TODO not really thread-safe
162 	public static class MetadataCache implements Serializable {
163 		private transient Map<Class<?>, ClassCache> classCache = new WeakHashMap<Class<?>, ClassCache>();
164 
165 
166 		private void readObject(java.io.ObjectInputStream stream) {
167 			classCache = new WeakHashMap<Class<?>, ClassCache>();
168 		}
169 
170 		Member getMember(Class<?> clazz, String property) {
171 			ClassCache cache = classCache.get( clazz );
172 			if (cache == null) {
173 				cache = new ClassCache(clazz);
174 				classCache.put( clazz, cache );
175 			}
176 			Member member = cache.members.get( property );
177 			if ( member == null ) {
178 				member = findMember( clazz, property );
179 				cache.members.put( property, member );
180 			}
181 			return member;
182 		}
183 
184 		private Member findMember(Class<?> clazz, String property) {
185 			final List<Class<?>> classes = getClassHierarchy( clazz );
186 
187 			for (Class current : classes) {
188 				final Field field;
189 				try {
190 					field = current.getDeclaredField( property );
191 					setAccessibility( field );
192 					return field;
193 				}
194 				catch ( NoSuchFieldException e ) {
195 					final Method method = getMethod( current, property );
196 					if (method != null) {
197 						setAccessibility( method );
198 						return method;
199 					}
200 				}
201 			}
202 			//we could not find any match
203 			throw new PersistenceException( "Unable to find field or method: "
204 							+ clazz + "#"
205 							+ property);
206 		}
207 
208 		private List<Class<?>> getClassHierarchy(Class<?> clazz) {
209 			ClassCache cache = classCache.get( clazz );
210 			if (cache == null) {
211 				cache = new ClassCache(clazz);
212 				classCache.put( clazz, cache );
213 			}
214 			return cache.classHierarchy;
215 		}
216 
217 		private static List<Class<?>> findClassHierarchy(Class<?> clazz) {
218 			List<Class<?>> classes = new ArrayList<Class<?>>();
219 			Class<?> current = clazz;
220 			do {
221 				classes.add( current );
222 				current = current.getSuperclass();
223 			}
224 			while ( current != null );
225 			return classes;
226 		}
227 
228 		private static class ClassCache {
229 			List<Class<?>> classHierarchy;
230 			Map<String, Member> members = new HashMap<String, Member>();
231 
232 			public ClassCache(Class<?> clazz) {
233 				classHierarchy = findClassHierarchy( clazz );
234 			}
235 		}
236 	}
