25  package org.hibernate.criterion;
26  import org.hibernate.Criteria;
27  import org.hibernate.EntityMode;
28  import org.hibernate.HibernateException;
29  import org.hibernate.engine.spi.QueryParameters;
30  import org.hibernate.engine.spi.SessionFactoryImplementor;
31  import org.hibernate.engine.spi.SessionImplementor;
32  import org.hibernate.engine.spi.TypedValue;
33  import org.hibernate.internal.CriteriaImpl;
34  import org.hibernate.loader.criteria.CriteriaJoinWalker;
35  import org.hibernate.loader.criteria.CriteriaQueryTranslator;
36  import org.hibernate.persister.entity.OuterJoinLoadable;
37  import org.hibernate.type.Type;
38  
39  /**
40   * @author Gavin King
41   */
42  public abstract class SubqueryExpression implements Criterion {
43  	
44  	private CriteriaImpl criteriaImpl;
45  	private String quantifier;
46  	private String op;
47  	private QueryParameters params;
48  	private Type[] types;
49  	private CriteriaQueryTranslator innerQuery;
50  
51  	protected Type[] getTypes() {
52  		return types;
53  	}
54  	
55  	protected SubqueryExpression(String op, String quantifier, DetachedCriteria dc) {
56  		this.criteriaImpl = dc.getCriteriaImpl();
57  		this.quantifier = quantifier;
58  		this.op = op;
59  	}
60  	
61  	protected abstract String toLeftSqlString(Criteria criteria, CriteriaQuery outerQuery);
62  
63  	public String toSqlString(Criteria criteria, CriteriaQuery criteriaQuery) throws HibernateException {
64  		final SessionFactoryImplementor factory = criteriaQuery.getFactory();
65  		final OuterJoinLoadable persister =
66  				( OuterJoinLoadable ) factory.getEntityPersister( criteriaImpl.getEntityOrClassName() );
67  
68  		createAndSetInnerQuery( criteriaQuery, factory );
69  		criteriaImpl.setSession( deriveRootSession( criteria ) );
70  
71  		CriteriaJoinWalker walker = new CriteriaJoinWalker(
72  				persister,
73  				innerQuery,
74  				factory,
75  				criteriaImpl,
76  				criteriaImpl.getEntityOrClassName(),
77  				criteriaImpl.getSession().getLoadQueryInfluencers(),
78  				innerQuery.getRootSQLALias()
79  		);
80  
81  		String sql = walker.getSQLString();
82  
83  		final StringBuilder buf = new StringBuilder( toLeftSqlString(criteria, criteriaQuery) );
84  		if ( op != null ) {
85  			buf.append( ' ' ).append( op ).append( ' ' );
86  		}
87  		if ( quantifier != null ) {
88  			buf.append( quantifier ).append( ' ' );
89  		}
90  		return buf.append( '(' ).append( sql ).append( ')' )
91  				.toString();
92  	}
93  
94  	private SessionImplementor deriveRootSession(Criteria criteria) {
95  		if ( criteria instanceof CriteriaImpl ) {
96  			return ( ( CriteriaImpl ) criteria ).getSession();
97  		}
98  		else if ( criteria instanceof CriteriaImpl.Subcriteria ) {
99  			return deriveRootSession( ( ( CriteriaImpl.Subcriteria ) criteria ).getParent() );
100 		}
101 		else {
102 			// could happen for custom Criteria impls.  Not likely, but...
103 			// 		for long term solution, see HHH-3514
104 			return null;
105 		}
106 	}
107 
108 	public TypedValue[] getTypedValues(Criteria criteria, CriteriaQuery criteriaQuery) 
109 	throws HibernateException {
110 		//the following two lines were added to ensure that this.params is not null, which
111 		//can happen with two-deep nested subqueries
112 		SessionFactoryImplementor factory = criteriaQuery.getFactory();
113 		createAndSetInnerQuery(criteriaQuery, factory);
114 		
115 		Type[] ppTypes = params.getPositionalParameterTypes();
116 		Object[] ppValues = params.getPositionalParameterValues();
117 		TypedValue[] tv = new TypedValue[ppTypes.length];
118 		for ( int i=0; i<ppTypes.length; i++ ) {
119 			tv[i] = new TypedValue( ppTypes[i], ppValues[i], EntityMode.POJO );
120 		}
121 		return tv;
122 	}
123 
124 	/**
125 	 * Creates the inner query used to extract some useful information about types, since it is needed in both methods.
126 	 *
127 	 * @param criteriaQuery The criteria query
128 	 * @param factory The session factory.
129 	 */
130 	private void createAndSetInnerQuery(CriteriaQuery criteriaQuery, SessionFactoryImplementor factory) {
131 		if ( innerQuery == null ) {
132 			//with two-deep subqueries, the same alias would get generated for
133 			//both using the old method (criteriaQuery.generateSQLAlias()), so
134 			//that is now used as a fallback if the main criteria alias isn't set
135 			String alias;
136 			if ( this.criteriaImpl.getAlias() == null ) {
137 				alias = criteriaQuery.generateSQLAlias();
138 			}
139 			else {
140 				alias = this.criteriaImpl.getAlias() + "_";
141 			}
142 
143 			innerQuery = new CriteriaQueryTranslator(
144 					factory,
145 					criteriaImpl,
146 					criteriaImpl.getEntityOrClassName(), //implicit polymorphism not supported (would need a union)
147 					alias,
148 					criteriaQuery
149 				);
150 
151 			params = innerQuery.getQueryParameters();
152 			types = innerQuery.getProjectedTypes();
153 		}
154 	}
155 }
