51  /**
52   * Defines the default listener used by Hibernate for handling save-update
53   * events.
54   *
55   * @author Steve Ebersole
56   * @author Gavin King
57   */
58  public class DefaultSaveOrUpdateEventListener extends AbstractSaveEventListener implements SaveOrUpdateEventListener {
59  
60  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, DefaultSaveOrUpdateEventListener.class.getName() );
61  
62  	/**
63  	 * Handle the given update event.
64  	 *
65  	 * @param event The update event to be handled.
66  	 */
67  	public void onSaveOrUpdate(SaveOrUpdateEvent event) {
68  		final SessionImplementor source = event.getSession();
69  		final Object object = event.getObject();
70  		final Serializable requestedId = event.getRequestedId();
71  
72  		if ( requestedId != null ) {
73  			//assign the requested id to the proxy, *before*
74  			//reassociating the proxy
75  			if ( object instanceof HibernateProxy ) {
76  				( ( HibernateProxy ) object ).getHibernateLazyInitializer().setIdentifier( requestedId );
77  			}
78  		}
79  
80  		// For an uninitialized proxy, noop, don't even need to return an id, since it is never a save()
81  		if ( reassociateIfUninitializedProxy( object, source ) ) {
82  			LOG.trace( "Reassociated uninitialized proxy" );
83  		}
84  		else {
85  			//initialize properties of the event:
86  			final Object entity = source.getPersistenceContext().unproxyAndReassociate( object );
87  			event.setEntity( entity );
88  			event.setEntry( source.getPersistenceContext().getEntry( entity ) );
89  			//return the id in the event object
90  			event.setResultId( performSaveOrUpdate( event ) );
91  		}
92  
93  	}
94  
95  	protected boolean reassociateIfUninitializedProxy(Object object, SessionImplementor source) {
96  		return source.getPersistenceContext().reassociateIfUninitializedProxy( object );
97  	}
98  
99  	protected Serializable performSaveOrUpdate(SaveOrUpdateEvent event) {
100 		EntityState entityState = getEntityState(
101 				event.getEntity(),
102 				event.getEntityName(),
103 				event.getEntry(),
104 				event.getSession()
105 		);
106 
107 		switch ( entityState ) {
108 			case DETACHED:
109 				entityIsDetached( event );
110 				return null;
111 			case PERSISTENT:
112 				return entityIsPersistent( event );
113 			default: //TRANSIENT or DELETED
114 				return entityIsTransient( event );
115 		}
116 	}
117 
118 	protected Serializable entityIsPersistent(SaveOrUpdateEvent event) throws HibernateException {
119 		final boolean traceEnabled = LOG.isTraceEnabled();
120 		if ( traceEnabled ) {
121 			LOG.trace( "Ignoring persistent instance" );
122 		}
123 		EntityEntry entityEntry = event.getEntry();
124 		if ( entityEntry == null ) {
125 			throw new AssertionFailure( "entity was transient or detached" );
126 		}
127 		else {
128 
129 			if ( entityEntry.getStatus() == Status.DELETED ) {
130 				throw new AssertionFailure( "entity was deleted" );
131 			}
132 
133 			final SessionFactoryImplementor factory = event.getSession().getFactory();
134 
135 			Serializable requestedId = event.getRequestedId();
136 
137 			Serializable savedId;
138 			if ( requestedId == null ) {
139 				savedId = entityEntry.getId();
140 			}
141 			else {
142 
143 				final boolean isEqual = !entityEntry.getPersister().getIdentifierType()
144 						.isEqual( requestedId, entityEntry.getId(), factory );
145 
146 				if ( isEqual ) {
147 					throw new PersistentObjectException(
148 							"object passed to save() was already persistent: " +
149 									MessageHelper.infoString( entityEntry.getPersister(), requestedId, factory )
150 					);
151 				}
152 
153 				savedId = requestedId;
154 
155 			}
156 
157 			if ( traceEnabled ) {
158 				LOG.tracev( "Object already associated with session: {0}", MessageHelper.infoString( entityEntry.getPersister(), savedId, factory ) );
159 			}
160 
161 			return savedId;
162 
163 		}
164 	}

166 	/**
167 	 * The given save-update event named a transient entity.
168 	 * <p/>
169 	 * Here, we will perform the save processing.
170 	 *
171 	 * @param event The save event to be handled.
172 	 *
173 	 * @return The entity's identifier after saving.
174 	 */
175 	protected Serializable entityIsTransient(SaveOrUpdateEvent event) {
176 
177 		LOG.trace( "Saving transient instance" );
178 
179 		final EventSource source = event.getSession();
180 
181 		EntityEntry entityEntry = event.getEntry();
182 		if ( entityEntry != null ) {
183 			if ( entityEntry.getStatus() == Status.DELETED ) {
184 				source.forceFlush( entityEntry );
185 			}
186 			else {
187 				throw new AssertionFailure( "entity was persistent" );
188 			}
189 		}
190 
191 		Serializable id = saveWithGeneratedOrRequestedId( event );
192 
193 		source.getPersistenceContext().reassociateProxy( event.getObject(), id );
194 
195 		return id;
196 	}
197 
198 	/**
199 	 * Save the transient instance, assigning the right identifier
200 	 *
201 	 * @param event The initiating event.
202 	 *
203 	 * @return The entity's identifier value after saving.
204 	 */
205 	protected Serializable saveWithGeneratedOrRequestedId(SaveOrUpdateEvent event) {
206 		return saveWithGeneratedId(
207 				event.getEntity(),
208 				event.getEntityName(),
209 				null,
210 				event.getSession(),
211 				true
212 		);
213 	}

215 	/**
216 	 * The given save-update event named a detached entity.
217 	 * <p/>
218 	 * Here, we will perform the update processing.
219 	 *
220 	 * @param event The update event to be handled.
221 	 */
222 	protected void entityIsDetached(SaveOrUpdateEvent event) {
223 
224 		LOG.trace( "Updating detached instance" );
225 
226 		if ( event.getSession().getPersistenceContext().isEntryFor( event.getEntity() ) ) {
227 			//TODO: assertion only, could be optimized away
228 			throw new AssertionFailure( "entity was persistent" );
229 		}
230 
231 		Object entity = event.getEntity();
232 
233 		EntityPersister persister = event.getSession().getEntityPersister( event.getEntityName(), entity );
234 
235 		event.setRequestedId(
236 				getUpdateId(
237 						entity, persister, event.getRequestedId(), event.getSession()
238 				)
239 		);
240 
241 		performUpdate( event, entity, persister );
242 
243 	}
244 
245 	/**
246 	 * Determine the id to use for updating.
247 	 *
248 	 * @param entity The entity.
249 	 * @param persister The entity persister
250 	 * @param requestedId The requested identifier
251 	 * @param session The session
252 	 *
253 	 * @return The id.
254 	 *
255 	 * @throws TransientObjectException If the entity is considered transient.
256 	 */
257 	protected Serializable getUpdateId(
258 			Object entity,
259 			EntityPersister persister,
260 			Serializable requestedId,
261 			SessionImplementor session) {
262 		// use the id assigned to the instance
263 		Serializable id = persister.getIdentifier( entity, session );
264 		if ( id == null ) {
265 			// assume this is a newly instantiated transient object
266 			// which should be saved rather than updated
267 			throw new TransientObjectException(
268 					"The given object has a null identifier: " +
269 							persister.getEntityName()
270 			);
271 		}
272 		else {
273 			return id;
274 		}
275 
276 	}

278 	protected void performUpdate(
279 			SaveOrUpdateEvent event,
280 			Object entity,
281 			EntityPersister persister) throws HibernateException {
282 
283         final boolean traceEnabled = LOG.isTraceEnabled();
284 		if ( traceEnabled && !persister.isMutable() ) {
285 			LOG.trace( "Immutable instance passed to performUpdate()" );
286 		}
287 
288 		if ( traceEnabled ) {
289 			LOG.tracev( "Updating {0}",
290 					MessageHelper.infoString( persister, event.getRequestedId(), event.getSession().getFactory() ) );
291 		}
292 
293 		final EventSource source = event.getSession();
294 		final EntityKey key = source.generateEntityKey( event.getRequestedId(), persister );
295 
296 		source.getPersistenceContext().checkUniqueness(key, entity);
297 
298 		if (invokeUpdateLifecycle(entity, persister, source)) {
299             reassociate(event, event.getObject(), event.getRequestedId(), persister);
300             return;
301         }
302 
303 		// this is a transient object with existing persistent state not loaded by the session
304 
305 		new OnUpdateVisitor(source, event.getRequestedId(), entity).process(entity, persister);
306 
307 		// TODO: put this stuff back in to read snapshot from
308         // the second-level cache (needs some extra work)
309         /*Object[] cachedState = null;
310 
311         if ( persister.hasCache() ) {
312         	CacheEntry entry = (CacheEntry) persister.getCache()
313         			.get( event.getRequestedId(), source.getTimestamp() );
314             cachedState = entry==null ?
315             		null :
316             		entry.getState(); //TODO: half-assemble this stuff
317         }*/
318 
319 		source.getPersistenceContext().addEntity(
320 				entity,
321 				(persister.isMutable() ? Status.MANAGED : Status.READ_ONLY),
322 				null, // cachedState,
323 				key,
324 				persister.getVersion( entity ),
325 				LockMode.NONE,
326 				true,
327 				persister,
328 				false,
329 				true // assume true, since we don't really know, and it doesn't matter
330 				);
331 
332 		persister.afterReassociate(entity, source);
333 
334 		if ( traceEnabled ) {
335 			LOG.tracev( "Updating {0}", MessageHelper.infoString( persister, event.getRequestedId(), source.getFactory() ) );
336 		}
337 
338 		cascadeOnUpdate( event, persister, entity );
339 	}
340 
341 	protected boolean invokeUpdateLifecycle(Object entity, EntityPersister persister, EventSource source) {
342 		if ( persister.implementsLifecycle() ) {
343 			LOG.debug( "Calling onUpdate()" );
344 			if ( ( (Lifecycle) entity ).onUpdate( source ) ) {
345 				LOG.debug( "Update vetoed by onUpdate()" );
346 				return true;
347 			}
348 		}
349 		return false;
350 	}

352 	/**
353 	 * Handles the calls needed to perform cascades as part of an update request
354 	 * for the given entity.
355 	 *
356 	 * @param event The event currently being processed.
357 	 * @param persister The defined persister for the entity being updated.
358 	 * @param entity The entity being updated.
359 	 */
360 	private void cascadeOnUpdate(SaveOrUpdateEvent event, EntityPersister persister, Object entity) {
361 		EventSource source = event.getSession();
362 		source.getPersistenceContext().incrementCascadeLevel();
363 		try {
364 			new Cascade( CascadingAction.SAVE_UPDATE, Cascade.AFTER_UPDATE, source )
365 					.cascade( persister, entity );
366 		}
367 		finally {
368 			source.getPersistenceContext().decrementCascadeLevel();
369 		}
370 	}
371 
372 	@Override
373     protected CascadingAction getCascadeAction() {
374 		return CascadingAction.SAVE_UPDATE;
375 	}
