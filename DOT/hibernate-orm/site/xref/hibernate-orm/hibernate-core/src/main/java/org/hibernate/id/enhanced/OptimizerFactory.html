48  	public static enum StandardOptimizerDescriptor {
49  		NONE( "none", NoopOptimizer.class ),
50  		HILO( "hilo", HiLoOptimizer.class ),
51  		LEGACY_HILO( "legacy-hilo", LegacyHiLoAlgorithmOptimizer.class ),
52  		POOLED( "pooled", PooledOptimizer.class, true ),
53  		POOLED_LO( "pooled-lo", PooledLoOptimizer.class, true );
54  
55  		private final String externalName;
56  		private final Class<? extends Optimizer> optimizerClass;
57  		private final boolean isPooled;
58  
59  		StandardOptimizerDescriptor(String externalName, Class<? extends Optimizer> optimizerClass) {
60  			this( externalName, optimizerClass, false );
61  		}
62  
63  		StandardOptimizerDescriptor(String externalName, Class<? extends Optimizer> optimizerClass, boolean pooled) {
64  			this.externalName = externalName;
65  			this.optimizerClass = optimizerClass;
66  			this.isPooled = pooled;
67  		}
68  
69  		public String getExternalName() {
70  			return externalName;
71  		}
72  
73  		public Class<? extends Optimizer> getOptimizerClass() {
74  			return optimizerClass;
75  		}
76  
77  		public boolean isPooled() {
78  			return isPooled;
79  		}
80  
81  		public static StandardOptimizerDescriptor fromExternalName(String externalName) {
82  			if ( StringHelper.isEmpty( externalName ) ) {
83  				LOG.debug( "No optimizer specified, using NONE as default" );
84  				return NONE;
85  			}
86  			else if ( NONE.externalName.equals( externalName ) ) {
87  				return NONE;
88  			}
89  			else if ( HILO.externalName.equals( externalName ) ) {
90  				return HILO;
91  			}
92  			else if ( LEGACY_HILO.externalName.equals( externalName ) ) {
93  				return LEGACY_HILO;
94  			}
95  			else if ( POOLED.externalName.equals( externalName ) ) {
96  				return POOLED;
97  			}
98  			else if ( POOLED_LO.externalName.equals( externalName ) ) {
99  				return POOLED_LO;
100 			}
101 			else {
102 				LOG.debugf( "Unknown optimizer key [%s]; returning null assuming Optimizer impl class name", externalName );
103 				return null;
104 			}
105 		}

109 	 * Marker interface for optimizer which wish to know the user-specified initial value.
110 	 * <p/>
111 	 * Used instead of constructor injection since that is already a public understanding and
112 	 * because not all optimizers care.
113 	 */
114 	public static interface InitialValueAwareOptimizer {
115 		/**
116 		 * Reports the user specified initial value to the optimizer.
117 		 * <p/>
118 		 * <tt>-1</tt> is used to indicate that the user did not specify.
119 		 *
120 		 * @param initialValue The initial value specified by the user, or <tt>-1</tt> to indicate that the
121 		 * user did not specify.
122 		 */
123 		public void injectInitialValue(long initialValue);
124 	}

126 	public static boolean isPooledOptimizer(String type) {
127 		final StandardOptimizerDescriptor standardDescriptor = StandardOptimizerDescriptor.fromExternalName( type );
128 		return standardDescriptor != null && standardDescriptor.isPooled();
129 	}

131 	private static Class[] CTOR_SIG = new Class[] { Class.class, int.class };
132 
133 	/**
134 	 * Builds an optimizer
135 	 *
136 	 * @param type The optimizer type, either a short-hand name or the {@link Optimizer} class name.
137 	 * @param returnClass The generated value java type
138 	 * @param incrementSize The increment size.
139 	 *
140 	 * @return The built optimizer
141 	 *
142 	 * @deprecated Use {@link #buildOptimizer(String, Class, int, long)} instead
143 	 */
144 	@Deprecated
145 	@SuppressWarnings( {"UnnecessaryBoxing", "unchecked"})
146 	public static Optimizer buildOptimizer(String type, Class returnClass, int incrementSize) {
147 		final Class<? extends Optimizer> optimizerClass;
148 
149 		final StandardOptimizerDescriptor standardDescriptor = StandardOptimizerDescriptor.fromExternalName( type );
150 		if ( standardDescriptor != null ) {
151 			optimizerClass = standardDescriptor.getOptimizerClass();
152 		}
153 		else {
154 			try {
155 				optimizerClass = ReflectHelper.classForName( type );
156 			}
157 			catch( Throwable ignore ) {
158 				LOG.unableToLocateCustomOptimizerClass( type );
159 				return buildFallbackOptimizer( returnClass, incrementSize );
160 			}
161 		}
162 
163 		try {
164 			Constructor ctor = optimizerClass.getConstructor( CTOR_SIG );
165 			return ( Optimizer ) ctor.newInstance( returnClass, Integer.valueOf( incrementSize ) );
166 		}
167 		catch( Throwable ignore ) {
168             LOG.unableToInstantiateOptimizer( type );
169 		}
170 
171 		return buildFallbackOptimizer( returnClass, incrementSize );
172 	}
173 
174 	private static Optimizer buildFallbackOptimizer(Class returnClass, int incrementSize) {
175 		return new NoopOptimizer( returnClass, incrementSize );
176 	}

178 	/**
179 	 * Builds an optimizer
180 	 *
181 	 * @param type The optimizer type, either a short-hand name or the {@link Optimizer} class name.
182 	 * @param returnClass The generated value java type
183 	 * @param incrementSize The increment size.
184 	 * @param explicitInitialValue The user supplied initial-value (-1 indicates the user did not specify).
185 	 *
186 	 * @return The built optimizer
187 	 */
188 	@SuppressWarnings({ "UnnecessaryBoxing", "deprecation" })
189 	public static Optimizer buildOptimizer(String type, Class returnClass, int incrementSize, long explicitInitialValue) {
190 		final Optimizer optimizer = buildOptimizer( type, returnClass, incrementSize );
191 		if ( InitialValueAwareOptimizer.class.isInstance( optimizer ) ) {
192 			( (InitialValueAwareOptimizer) optimizer ).injectInitialValue( explicitInitialValue );
193 		}
194 		return optimizer;
195 	}

220 		 * class which is used to represent the id (e.g. {@link java.lang.Long}).
221 		 *

235 	/**
236 	 * An optimizer that performs no optimization.  The database is hit for
237 	 * every request.
238 	 *
239 	 * Using this implementation is probably not the most efficient choice.
240 	 */
241 	public static final class NoopOptimizer extends OptimizerSupport {
242 		private IntegralDataTypeHolder lastSourceValue;
243 
244 		public NoopOptimizer(Class returnClass, int incrementSize) {
245 			super( returnClass, incrementSize );
246 		}
247 
248 		@Override
249 		public Serializable generate(AccessCallback callback) {
250 			// IMPL NOTE : this method is called concurrently and is
251 			// not synchronized. It is very important to work on the
252 			// local variable: the field lastSourceValue is not
253 			// reliable as it might be mutated by multipled threads.
254 			// The lastSourceValue field is only accessed by tests,
255 			// so this is not a concern.
256 			IntegralDataTypeHolder value = null;
257 			while ( value == null || value.lt( 1 ) ) {
258 				value = callback.getNextValue();
259 			}
260 			lastSourceValue = value;
261 			return value.makeValue();
262 		}
263 
264 		@Override
265 		public IntegralDataTypeHolder getLastSourceValue() {
266 			return lastSourceValue;
267 		}
268 
269 		@Override
270 		public boolean applyIncrementSizeToSourceValues() {
271 			return false;
272 		}
273 	}

275 	/**
276 	 * Optimizer which applies a 'hilo' algorithm in memory to achieve
277 	 * optimization.
278 	 * <p/>
279 	 * A 'hilo' algorithm is simply a means for a single value stored in the
280 	 * database to represent a "bucket" of possible, contiguous values.  The
281 	 * database value identifies which particular bucket we are on.
282 	 * <p/>
283 	 * This database value must be paired with another value that defines the
284 	 * size of the bucket; the number of possible values available.
285 	 * The {@link #getIncrementSize() incrementSize} serves this purpose.  The
286 	 * naming here is meant more for consistency in that this value serves the
287 	 * same purpose as the increment supplied to the {@link PooledOptimizer}.
288 	 * <p/>
289 	 * The general algorithms used to determine the bucket are:<ol>
290 	 * <li>{@code upperLimit = (databaseValue * incrementSize) + 1}</li>
291 	 * <li>{@code lowerLimit = upperLimit - incrementSize}</li>
292 	 * </ol>
293 	 * As an example, consider a case with incrementSize of 20.  Initially the
294 	 * database holds 1:<ol>
295 	 * <li>{@code upperLimit = (1 * 20) + 1 = 21}</li>
296 	 * <li>{@code lowerLimit = 21 - 20 = 1}</li>
297 	 * </ol>
298 	 * From there we increment the value from lowerLimit until we reach the
299 	 * upperLimit, at which point we would define a new bucket.  The database
300 	 * now contains 2, though incrementSize remains unchanged:<ol>
301 	 * <li>{@code upperLimit = (2 * 20) + 1 = 41}</li>
302 	 * <li>{@code lowerLimit = 41 - 20 = 21}</li>
303 	 * </ol>
304 	 * And so on...
305 	 * <p/>
306 	 * Note, 'value' always (after init) holds the next value to return
307 	 */
308 	public static class HiLoOptimizer extends OptimizerSupport {
309 		private IntegralDataTypeHolder lastSourceValue;
310 		private IntegralDataTypeHolder upperLimit;
311 		private IntegralDataTypeHolder value;
312 
313 		public HiLoOptimizer(Class returnClass, int incrementSize) {
314 			super( returnClass, incrementSize );
315 			if ( incrementSize < 1 )
316 				throw new HibernateException( "increment size cannot be less than 1" );
317 			if ( LOG.isTraceEnabled() ) {
318 				LOG.tracev( "Creating hilo optimizer with [incrementSize={0}; returnClass={1}]", incrementSize, returnClass.getName() );
319 			}
320 		}
321 
322 		@Override
323 		public synchronized Serializable generate(AccessCallback callback) {
324 			if ( lastSourceValue == null ) {
325 				// first call, so initialize ourselves.  we need to read the database
326 				// value and set up the 'bucket' boundaries
327 				lastSourceValue = callback.getNextValue();
328 				while ( lastSourceValue.lt( 1 ) ) {
329 					lastSourceValue = callback.getNextValue();
330 				}
331 				// upperLimit defines the upper end of the bucket values
332 				upperLimit = lastSourceValue.copy().multiplyBy( incrementSize ).increment();
333 				// initialize value to the low end of the bucket
334 				value = upperLimit.copy().subtract( incrementSize );
335 			}
336 			else if ( ! upperLimit.gt( value ) ) {
337 				lastSourceValue = callback.getNextValue();
338 				upperLimit = lastSourceValue.copy().multiplyBy( incrementSize ).increment();
339 			}
340 			return value.makeValueThenIncrement();
341 		}
342 
343 		@Override
344 		public IntegralDataTypeHolder getLastSourceValue() {
345 			return lastSourceValue;
346 		}
347 
348 		@Override
349 		public boolean applyIncrementSizeToSourceValues() {
350 			return false;
351 		}
352 
353 		/**
354 		 * Getter for property 'lastValue'.
355 		 * <p/>
356 		 * Exposure intended for testing purposes.
357 		 *
358 		 * @return Value for property 'lastValue'.
359 		 */
360 		public IntegralDataTypeHolder getLastValue() {
361 			return value.copy().decrement();
362 		}
363 
364 		/**
365 		 * Getter for property 'upperLimit'.
366 		 * <p/>
367 		 * Exposure intended for testing purposes.
368 		 *
369 		 * @return Value for property 'upperLimit'.
370 		 */
371 		public IntegralDataTypeHolder getHiValue() {
372 			return upperLimit;
373 		}
374 	}

376 	public static class LegacyHiLoAlgorithmOptimizer extends OptimizerSupport {
377 		private long maxLo;
378 		private long lo;
379 		private IntegralDataTypeHolder hi;
380 
381 		private IntegralDataTypeHolder lastSourceValue;
382 		private IntegralDataTypeHolder value;
383 
384 
385 		public LegacyHiLoAlgorithmOptimizer(Class returnClass, int incrementSize) {
386 			super( returnClass, incrementSize );
387 			if ( incrementSize < 1 )
388 				throw new HibernateException( "increment size cannot be less than 1" );
389 			if ( LOG.isTraceEnabled() ) {
390 				LOG.tracev( "Creating hilo optimizer (legacy) with [incrementSize={0}; returnClass={1}]", incrementSize, returnClass.getName() );
391 			}
392 			maxLo = incrementSize;
393 			lo = maxLo+1;
394 		}
395 
396 		@Override
397 		public synchronized Serializable generate(AccessCallback callback) {
398 			if ( lo > maxLo ) {
399 				lastSourceValue = callback.getNextValue();
400 				lo = lastSourceValue.eq( 0 ) ? 1 : 0;
401 				hi = lastSourceValue.copy().multiplyBy( maxLo+1 );
402 			}
403 			value = hi.copy().add( lo++ );
404 			return value.makeValue();
405 		}
406 
407 		@Override
408 		public IntegralDataTypeHolder getLastSourceValue() {
409 			return lastSourceValue.copy();
410 		}
411 
412 		@Override
413 		public boolean applyIncrementSizeToSourceValues() {
414 			return false;
415 		}
416 
417 		/**
418 		 * Getter for property 'lastValue'.
419 		 * <p/>
420 		 * Exposure intended for testing purposes.
421 		 *
422 		 * @return Value for property 'lastValue'.
423 		 */
424 		@SuppressWarnings( {"UnusedDeclaration"})
425 		public IntegralDataTypeHolder getLastValue() {
426 			return value;
427 		}
428 	}

430 	/**
431 	 * Optimizer which uses a pool of values, storing the next low value of the
432 	 * range in the database.
433 	 * <p/>
434 	 * Note that this optimizer works essentially the same as the
435 	 * {@link HiLoOptimizer} except that here the bucket ranges are actually
436 	 * encoded into the database structures.
437 	 * <p/>
438 	 * Note if you prefer that the database value be interpreted as the bottom end of our current range,
439 	 * then use the {@link PooledLoOptimizer} strategy
440 	 */
441 	public static class PooledOptimizer extends OptimizerSupport implements InitialValueAwareOptimizer {
442 		private IntegralDataTypeHolder hiValue;
443 		private IntegralDataTypeHolder value;
444 		private long initialValue = -1;
445 
446 		public PooledOptimizer(Class returnClass, int incrementSize) {
447 			super( returnClass, incrementSize );
448 			if ( incrementSize < 1 ) {
449 				throw new HibernateException( "increment size cannot be less than 1" );
450 			}
451 			if ( LOG.isTraceEnabled() ) {
452 				LOG.tracev( "Creating pooled optimizer with [incrementSize={0}; returnClass={1}]", incrementSize, returnClass.getName() );
453 			}
454 		}
455 
456 		@Override
457 		public synchronized Serializable generate(AccessCallback callback) {
458 			if ( hiValue == null ) {
459 				value = callback.getNextValue();
460                 // unfortunately not really safe to normalize this
461                 // to 1 as an initial value like we do the others
462                 // because we would not be able to control this if
463                 // we are using a sequence...
464                 if (value.lt(1)) LOG.pooledOptimizerReportedInitialValue(value);
465                 // the call to obtain next-value just gave us the initialValue
466 				if ( ( initialValue == -1 && value.lt( incrementSize ) ) || value.eq( initialValue ) ) {
467 					hiValue = callback.getNextValue();
468 				}
469 				else {
470 					hiValue = value;
471 					value = hiValue.copy().subtract( incrementSize );
472 				}
473 			}
474 			else if ( ! hiValue.gt( value ) ) {
475 				hiValue = callback.getNextValue();
476 				value = hiValue.copy().subtract( incrementSize );
477 			}
478 			return value.makeValueThenIncrement();
479 		}
480 
481 		@Override
482 		public IntegralDataTypeHolder getLastSourceValue() {
483 			return hiValue;
484 		}
485 
486 		@Override
487 		public boolean applyIncrementSizeToSourceValues() {
488 			return true;
489 		}
490 
491 		/**
492 		 * Getter for property 'lastValue'.
493 		 * <p/>
494 		 * Exposure intended for testing purposes.
495 		 *
496 		 * @return Value for property 'lastValue'.
497 		 */
498 		public IntegralDataTypeHolder getLastValue() {
499 			return value.copy().decrement();
500 		}
501 
502 		@Override
503 		public void injectInitialValue(long initialValue) {
504 			this.initialValue = initialValue;
505 		}
506 	}

508 	public static class PooledLoOptimizer extends OptimizerSupport {
509 		private IntegralDataTypeHolder lastSourceValue; // last value read from db source
510 		private IntegralDataTypeHolder value; // the current generator value
511 		private IntegralDataTypeHolder upperLimitValue; // the value at which we'll hit the db again
512 
513 		public PooledLoOptimizer(Class returnClass, int incrementSize) {
514 			super( returnClass, incrementSize );
515 			if ( incrementSize < 1 ) {
516 				throw new HibernateException( "increment size cannot be less than 1" );
517 			}
518 			LOG.creatingPooledLoOptimizer( incrementSize, returnClass.getName() );
519 		}
520 
521 		@Override
522 		public synchronized Serializable generate(AccessCallback callback) {
523 			if ( lastSourceValue == null || ! value.lt( upperLimitValue ) ) {
524 				lastSourceValue = callback.getNextValue();
525 				upperLimitValue = lastSourceValue.copy().add( incrementSize );
526 				value = lastSourceValue.copy();
527 				// handle cases where initial-value is less that one (hsqldb for instance).
528 				while ( value.lt( 1 ) ) {
529 					value.increment();
530 				}
531 			}
532 			return value.makeValueThenIncrement();
533 		}
534 
535 		@Override
536 		public IntegralDataTypeHolder getLastSourceValue() {
537 			return lastSourceValue;
538 		}
539 
540 		@Override
541 		public boolean applyIncrementSizeToSourceValues() {
542 			return true;
543 		}
