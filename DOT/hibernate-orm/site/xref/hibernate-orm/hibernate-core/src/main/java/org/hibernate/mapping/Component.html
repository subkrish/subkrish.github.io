100 	public int getColumnSpan() {
101 		int n=0;
102 		Iterator iter = getPropertyIterator();
103 		while ( iter.hasNext() ) {
104 			Property p = (Property) iter.next();
105 			n+= p.getColumnSpan();
106 		}
107 		return n;
108 	}
109 	public Iterator getColumnIterator() {
110 		Iterator[] iters = new Iterator[ getPropertySpan() ];
111 		Iterator iter = getPropertyIterator();
112 		int i=0;
113 		while ( iter.hasNext() ) {
114 			iters[i++] = ( (Property) iter.next() ).getColumnIterator();
115 		}
116 		return new JoinedIterator(iters);
117 	}

196 	public boolean[] getColumnInsertability() {
197 		boolean[] result = new boolean[ getColumnSpan() ];
198 		Iterator iter = getPropertyIterator();
199 		int i=0;
200 		while ( iter.hasNext() ) {
201 			Property prop = (Property) iter.next();
202 			boolean[] chunk = prop.getValue().getColumnInsertability();
203 			if ( prop.isInsertable() ) {
204 				System.arraycopy(chunk, 0, result, i, chunk.length);
205 			}
206 			i+=chunk.length;
207 		}
208 		return result;
209 	}

293 	public IdentifierGenerator createIdentifierGenerator(
294 			IdentifierGeneratorFactory identifierGeneratorFactory,
295 			Dialect dialect,
296 			String defaultCatalog,
297 			String defaultSchema,
298 			RootClass rootClass) throws MappingException {
299 		if ( builtIdentifierGenerator == null ) {
300 			builtIdentifierGenerator = buildIdentifierGenerator(
301 					identifierGeneratorFactory,
302 					dialect,
303 					defaultCatalog,
304 					defaultSchema,
305 					rootClass
306 			);
307 		}
308 		return builtIdentifierGenerator;
309 	}
310 
311 	private IdentifierGenerator buildIdentifierGenerator(
312 			IdentifierGeneratorFactory identifierGeneratorFactory,
313 			Dialect dialect,
314 			String defaultCatalog,
315 			String defaultSchema,
316 			RootClass rootClass) throws MappingException {
317 		final boolean hasCustomGenerator = ! DEFAULT_ID_GEN_STRATEGY.equals( getIdentifierGeneratorStrategy() );
318 		if ( hasCustomGenerator ) {
319 			return super.createIdentifierGenerator(
320 					identifierGeneratorFactory, dialect, defaultCatalog, defaultSchema, rootClass
321 			);
322 		}
323 
324 		final Class entityClass = rootClass.getMappedClass();
325 		final Class attributeDeclarer; // what class is the declarer of the composite pk attributes
326 		CompositeNestedGeneratedValueGenerator.GenerationContextLocator locator;
327 
328 		// IMPL NOTE : See the javadoc discussion on CompositeNestedGeneratedValueGenerator wrt the
329 		//		various scenarios for which we need to account here
330 		if ( rootClass.getIdentifierMapper() != null ) {
331 			// we have the @IdClass / <composite-id mapped="true"/> case
332 			attributeDeclarer = resolveComponentClass();
333 		}
334 		else if ( rootClass.getIdentifierProperty() != null ) {
335 			// we have the "@EmbeddedId" / <composite-id name="idName"/> case
336 			attributeDeclarer = resolveComponentClass();
337 		}
338 		else {
339 			// we have the "straight up" embedded (again the hibernate term) component identifier
340 			attributeDeclarer = entityClass;
341 		}
342 
343 		locator = new StandardGenerationContextLocator( rootClass.getEntityName() );
344 		final CompositeNestedGeneratedValueGenerator generator = new CompositeNestedGeneratedValueGenerator( locator );
345 
346 		Iterator itr = getPropertyIterator();
347 		while ( itr.hasNext() ) {
348 			final Property property = (Property) itr.next();
349 			if ( property.getValue().isSimpleValue() ) {
350 				final SimpleValue value = (SimpleValue) property.getValue();
351 
352 				if ( DEFAULT_ID_GEN_STRATEGY.equals( value.getIdentifierGeneratorStrategy() ) ) {
353 					// skip any 'assigned' generators, they would have been handled by
354 					// the StandardGenerationContextLocator
355 					continue;
356 				}
357 
358 				final IdentifierGenerator valueGenerator = value.createIdentifierGenerator(
359 						identifierGeneratorFactory,
360 						dialect,
361 						defaultCatalog,
362 						defaultSchema,
363 						rootClass
364 				);
365 				generator.addGeneratedValuePlan(
366 						new ValueGenerationPlan(
367 								property.getName(),
368 								valueGenerator,
369 								injector( property, attributeDeclarer )
370 						)
371 				);
372 			}
373 		}
374 		return generator;
375 	}
376 
377 	private Setter injector(Property property, Class attributeDeclarer) {
378 		return property.getPropertyAccessor( attributeDeclarer )
379 				.getSetter( attributeDeclarer, property.getName() );
380 	}

391 	public static class StandardGenerationContextLocator
392 			implements CompositeNestedGeneratedValueGenerator.GenerationContextLocator {
393 		private final String entityName;
394 
395 		public StandardGenerationContextLocator(String entityName) {
396 			this.entityName = entityName;
397 		}
398 
399 		public Serializable locateGenerationContext(SessionImplementor session, Object incomingObject) {
400 			return session.getEntityPersister( entityName, incomingObject ).getIdentifier( incomingObject, session );
401 		}

404 	public static class ValueGenerationPlan implements CompositeNestedGeneratedValueGenerator.GenerationPlan {
405 		private final String propertyName;
406 		private final IdentifierGenerator subGenerator;
407 		private final Setter injector;
408 
409 		public ValueGenerationPlan(
410 				String propertyName,
411 				IdentifierGenerator subGenerator,
412 				Setter injector) {
413 			this.propertyName = propertyName;
414 			this.subGenerator = subGenerator;
415 			this.injector = injector;
416 		}
417 
418 		/**
419 		 * {@inheritDoc}
420 		 */
421 		public void execute(SessionImplementor session, Object incomingObject, Object injectionContext) {
422 			final Object generatedValue = subGenerator.generate( session, incomingObject );
423 			injector.set( injectionContext, generatedValue, session.getFactory() );
424 		}
425 
426 		public void registerPersistentGenerators(Map generatorMap) {
427 			if ( PersistentIdentifierGenerator.class.isInstance( subGenerator ) ) {
428 				generatorMap.put( ( (PersistentIdentifierGenerator) subGenerator ).generatorKey(), subGenerator );
429 			}
430 		}
