24  package org.hibernate.metamodel.source.annotations.global;
25  
26  import java.util.List;
27  
28  import org.jboss.jandex.AnnotationInstance;
29  import org.jboss.logging.Logger;
30  
31  import org.hibernate.AnnotationException;
32  import org.hibernate.internal.CoreMessageLogger;
33  import org.hibernate.internal.util.StringHelper;
34  import org.hibernate.metamodel.relational.Column;
35  import org.hibernate.metamodel.relational.ObjectName;
36  import org.hibernate.metamodel.relational.Schema;
37  import org.hibernate.metamodel.relational.SimpleValue;
38  import org.hibernate.metamodel.relational.Table;
39  import org.hibernate.metamodel.source.MetadataImplementor;
40  import org.hibernate.metamodel.source.annotations.AnnotationBindingContext;
41  import org.hibernate.metamodel.source.annotations.HibernateDotNames;
42  import org.hibernate.metamodel.source.annotations.JandexHelper;
43  
44  /**
45   * Binds table related information. This binder is called after the entities are bound.
46   *
47   * @author Hardy Ferentschik
48   */
49  public class TableBinder {
50  
51  	private static final CoreMessageLogger LOG = Logger.getMessageLogger(
52  			CoreMessageLogger.class,
53  			TableBinder.class.getName()
54  	);
55  
56  	private TableBinder() {
57  	}
58  
59  	/**
60  	 * Binds {@link org.hibernate.annotations.Tables} and {@link org.hibernate.annotations.Table} annotations to the supplied
61  	 * metadata.
62  	 *
63  	 * @param bindingContext the context for annotation binding
64  	 */
65  	public static void bind(AnnotationBindingContext bindingContext) {
66  		List<AnnotationInstance> annotations = bindingContext.getIndex().getAnnotations( HibernateDotNames.TABLE );
67  		for ( AnnotationInstance tableAnnotation : annotations ) {
68  			bind( bindingContext.getMetadataImplementor(), tableAnnotation );
69  		}
70  
71  		annotations = bindingContext.getIndex().getAnnotations( HibernateDotNames.TABLES );
72  		for ( AnnotationInstance tables : annotations ) {
73  			for ( AnnotationInstance table : JandexHelper.getValue( tables, "value", AnnotationInstance[].class ) ) {
74  				bind( bindingContext.getMetadataImplementor(), table );
75  			}
76  		}
77  	}
78  
79  	private static void bind(MetadataImplementor metadata, AnnotationInstance tableAnnotation) {
80  		String tableName = JandexHelper.getValue( tableAnnotation, "appliesTo", String.class );
81  		ObjectName objectName = new ObjectName( tableName );
82  		Schema schema = metadata.getDatabase().getSchema( objectName.getSchema(), objectName.getCatalog() );
83  		Table table = schema.locateTable( objectName.getName() );
84  		if ( table != null ) {
85  			bindHibernateTableAnnotation( table, tableAnnotation );
86  		}
87  	}
88  
89  	private static void bindHibernateTableAnnotation(Table table, AnnotationInstance tableAnnotation) {
90  		for ( AnnotationInstance indexAnnotation : JandexHelper.getValue(
91  				tableAnnotation,
92  				"indexes",
93  				AnnotationInstance[].class
94  		) ) {
95  			bindIndexAnnotation( table, indexAnnotation );
96  		}
97  		String comment = JandexHelper.getValue( tableAnnotation, "comment", String.class );
98  		if ( StringHelper.isNotEmpty( comment ) ) {
99  			table.addComment( comment.trim() );
100 		}
101 	}
102 
103 	private static void bindIndexAnnotation(Table table, AnnotationInstance indexAnnotation) {
104 		String indexName = JandexHelper.getValue( indexAnnotation, "appliesTo", String.class );
105 		String[] columnNames = JandexHelper.getValue( indexAnnotation, "columnNames", String[].class );
106 		if ( columnNames == null ) {
107 			LOG.noColumnsSpecifiedForIndex( indexName, table.toLoggableString() );
108 			return;
109 		}
110 		org.hibernate.metamodel.relational.Index index = table.getOrCreateIndex( indexName );
111 		for ( String columnName : columnNames ) {
112 			Column column = findColumn( table, columnName );
113 			if ( column == null ) {
114 				throw new AnnotationException( "@Index references a unknown column: " + columnName );
115 			}
116 			index.addColumn( column );
117 		}
118 	}
119 
120 	private static Column findColumn(Table table, String columnName) {
121 		Column column = null;
122 		for ( SimpleValue value : table.values() ) {
123 			if ( value instanceof Column && ( (Column) value ).getColumnName().getName().equals( columnName ) ) {
124 				column = (Column) value;
125 				break;
126 			}
127 		}
128 		return column;
129 	}
130 }
