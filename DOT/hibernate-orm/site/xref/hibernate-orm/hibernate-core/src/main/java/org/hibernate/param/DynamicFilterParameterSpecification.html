25  package org.hibernate.param;
26  import java.sql.PreparedStatement;
27  import java.sql.SQLException;
28  import java.util.Collection;
29  import java.util.Iterator;
30  
31  import org.hibernate.engine.spi.QueryParameters;
32  import org.hibernate.engine.spi.SessionImplementor;
33  import org.hibernate.type.Type;
34  
35  /**
36   * A specialized ParameterSpecification impl for dealing with a dynamic filter parameters.
37   * 
38   * @see org.hibernate.Session#enableFilter(String)
39   *
40   * @author Steve Ebersole
41   */
42  public class DynamicFilterParameterSpecification implements ParameterSpecification {
43  	private final String filterName;
44  	private final String parameterName;
45  	private final Type definedParameterType;
46  
47  	/**
48  	 * Constructs a parameter specification for a particular filter parameter.
49  	 *
50  	 * @param filterName The name of the filter
51  	 * @param parameterName The name of the parameter
52  	 * @param definedParameterType The paremeter type specified on the filter metadata
53  	 */
54  	public DynamicFilterParameterSpecification(
55  			String filterName,
56  			String parameterName,
57  			Type definedParameterType) {
58  		this.filterName = filterName;
59  		this.parameterName = parameterName;
60  		this.definedParameterType = definedParameterType;
61  	}
62  
63  	@Override
64  	public int bind(
65  			PreparedStatement statement,
66  			QueryParameters qp,
67  			SessionImplementor session,
68  			int start) throws SQLException {
69  		final int columnSpan = definedParameterType.getColumnSpan( session.getFactory() );
70  		final Object value = session.getLoadQueryInfluencers().getFilterParameterValue( filterName + '.' + parameterName );
71  		if ( Collection.class.isInstance( value ) ) {
72  			int positions = 0;
73  			Iterator itr = ( ( Collection ) value ).iterator();
74  			while ( itr.hasNext() ) {
75  				definedParameterType.nullSafeSet( statement, itr.next(), start + positions, session );
76  				positions += columnSpan;
77  			}
78  			return positions;
79  		}
80  		else {
81  			definedParameterType.nullSafeSet( statement, value, start, session );
82  			return columnSpan;
83  		}
84  	}
85  
86  	@Override
87  	public Type getExpectedType() {
88  		return definedParameterType;
89  	}
90  
91  	@Override
92  	public void setExpectedType(Type expectedType) {
93  		// todo : throw exception?  maybe warn if not the same?
94  	}
95  
96  	@Override
97  	public String renderDisplayInfo() {
98  		return "dynamic-filter={filterName=" + filterName + ",paramName=" + parameterName + "}";
99  	}
100 }
