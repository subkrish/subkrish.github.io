24  package org.hibernate.type.descriptor.java;
25  import org.hibernate.type.descriptor.WrapperOptions;
26  
27  /**
28   * Descriptor for {@link Short} handling.
29   *
30   * @author Steve Ebersole
31   */
32  public class ShortTypeDescriptor extends AbstractTypeDescriptor<Short> {
33  	public static final ShortTypeDescriptor INSTANCE = new ShortTypeDescriptor();
34  
35  	public ShortTypeDescriptor() {
36  		super( Short.class );
37  	}
38  
39  	public String toString(Short value) {
40  		return value == null ? null : value.toString();
41  	}
42  
43  	public Short fromString(String string) {
44  		return Short.valueOf( string );
45  	}
46  
47  	@SuppressWarnings({ "unchecked" })
48  	public <X> X unwrap(Short value, Class<X> type, WrapperOptions options) {
49  		if ( value == null ) {
50  			return null;
51  		}
52  		if ( Short.class.isAssignableFrom( type ) ) {
53  			return (X) value;
54  		}
55  		if ( Byte.class.isAssignableFrom( type ) ) {
56  			return (X) Byte.valueOf( value.byteValue() );
57  		}
58  		if ( Integer.class.isAssignableFrom( type ) ) {
59  			return (X) Integer.valueOf( value.intValue() );
60  		}
61  		if ( Long.class.isAssignableFrom( type ) ) {
62  			return (X) Long.valueOf( value.longValue() );
63  		}
64  		if ( Double.class.isAssignableFrom( type ) ) {
65  			return (X) Double.valueOf( value.doubleValue() );
66  		}
67  		if ( Float.class.isAssignableFrom( type ) ) {
68  			return (X) Float.valueOf( value.floatValue() );
69  		}
70  		if ( String.class.isAssignableFrom( type ) ) {
71  			return (X) value.toString();
72  		}
73  		throw unknownUnwrap( type );
74  	}
75  
76  	@SuppressWarnings({ "UnnecessaryBoxing" })
77  	public <X> Short wrap(X value, WrapperOptions options) {
78  		if ( value == null ) {
79  			return null;
80  		}
81  		if ( Short.class.isInstance( value ) ) {
82  			return (Short) value;
83  		}
84  		if ( Number.class.isInstance( value ) ) {
85  			return Short.valueOf( ( (Number) value ).shortValue() );
86  		}
87  		if ( String.class.isInstance( value ) ) {
88  			return Short.valueOf( ( (String) value ) );
89  		}
90  		throw unknownWrap( value.getClass() );
91  	}
92  }
