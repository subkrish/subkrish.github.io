60  	@Test
61  	public void testSimpleElement() throws Exception {
62  		assertEquals(
63  				"BoyFavoriteNumbers",
64  				configuration().getCollectionMapping( Boy.class.getName() + '.' + "favoriteNumbers" )
65  						.getCollectionTable().getName()
66  		);
67  		Session s = openSession();
68  		s.getTransaction().begin();
69  		Boy boy = new Boy();
70  		boy.setFirstName( "John" );
71  		boy.setLastName( "Doe" );
72  		boy.getNickNames().add( "Johnny" );
73  		boy.getNickNames().add( "Thing" );
74  		boy.getScorePerNickName().put( "Johnny", 3 );
75  		boy.getScorePerNickName().put( "Thing", 5 );
76  		int[] favNbrs = new int[4];
77  		for (int index = 0; index < favNbrs.length - 1; index++) {
78  			favNbrs[index] = index * 3;
79  		}
80  		boy.setFavoriteNumbers( favNbrs );
81  		boy.getCharacters().add( Character.GENTLE );
82  		boy.getCharacters().add( Character.CRAFTY );
83  
84  		HashMap<String,FavoriteFood> foods = new HashMap<String,FavoriteFood>();
85  		foods.put( "breakfast", FavoriteFood.PIZZA);
86  		foods.put( "lunch", FavoriteFood.KUNGPAOCHICKEN);
87  		foods.put( "dinner", FavoriteFood.SUSHI);
88  		boy.setFavoriteFood(foods);
89  		s.persist( boy );
90  		s.getTransaction().commit();
91  		s.clear();
92  		Transaction tx = s.beginTransaction();
93  		boy = (Boy) s.get( Boy.class, boy.getId() );
94  		assertNotNull( boy.getNickNames() );
95  		assertTrue( boy.getNickNames().contains( "Thing" ) );
96  		assertNotNull( boy.getScorePerNickName() );
97  		assertTrue( boy.getScorePerNickName().containsKey( "Thing" ) );
98  		assertEquals( new Integer( 5 ), boy.getScorePerNickName().get( "Thing" ) );
99  		assertNotNull( boy.getFavoriteNumbers() );
100 		assertEquals( 3, boy.getFavoriteNumbers()[1] );
101 		assertTrue( boy.getCharacters().contains( Character.CRAFTY ) );
102 		assertTrue( boy.getFavoriteFood().get("dinner").equals(FavoriteFood.SUSHI));
103 		assertTrue( boy.getFavoriteFood().get("lunch").equals(FavoriteFood.KUNGPAOCHICKEN));
104 		assertTrue( boy.getFavoriteFood().get("breakfast").equals(FavoriteFood.PIZZA));
105 		List result = s.createQuery( "select boy from Boy boy join boy.nickNames names where names = :name" )
106 				.setParameter( "name", "Thing" ).list();
107 		assertEquals( 1, result.size() );
108 		s.delete( boy );
109 		tx.commit();
110 		s.close();
111 	}

260 		isDefaultValueCollectionColumnPresent( Boy.class.getName(), "hatedNames" );
261 		isDefaultValueCollectionColumnPresent( Boy.class.getName(), "preferredNames" );

266 	private void isDefaultValueCollectionColumnPresent(String collectionOwner, String propertyName) {
267 		isCollectionColumnPresent( collectionOwner, propertyName, propertyName );
268 	}
269 
270 	private void isCollectionColumnPresent(String collectionOwner, String propertyName, String columnName) {
271 		final Collection collection = configuration().getCollectionMapping( collectionOwner + "." + propertyName );
272 		final Iterator columnIterator = collection.getCollectionTable().getColumnIterator();
273 		boolean hasDefault = false;
274 		while ( columnIterator.hasNext() ) {
275 			Column column = (Column) columnIterator.next();
276 			if ( columnName.equals( column.getName() ) ) hasDefault = true;
277 		}
278 		assertTrue( "Could not find " + columnName, hasDefault );
279 	}

376 	protected void checkDefaultJoinColumnName(
377 			Class<?> ownerEntityClass,
378 			String ownerCollectionPropertyName,
379 			String ownerForeignKeyNameExpected) {
380 		final org.hibernate.mapping.Collection ownerCollection = configuration().getCollectionMapping(
381 				ownerEntityClass.getName() + '.' + ownerCollectionPropertyName
382 		);
383 		// The default owner join column can only be computed if it has a PK with 1 column.
384 		assertEquals ( 1, ownerCollection.getOwner().getKey().getColumnSpan() );
385 		assertEquals(
386 				ownerForeignKeyNameExpected,
387 				( (Column) ownerCollection.getKey().getColumnIterator().next() ).getText()
388 		);
389 
390 		boolean hasOwnerFK = false;
391 		for ( Iterator it=ownerCollection.getCollectionTable().getForeignKeyIterator(); it.hasNext(); ) {
392 			final ForeignKey fk = (ForeignKey) it.next();
393 			assertSame( ownerCollection.getCollectionTable(), fk.getTable() );
394 			if ( fk.getColumnSpan() > 1 ) {
395 				continue;
396 			}
397 			if ( fk.getColumn( 0 ).getText().equals( ownerForeignKeyNameExpected ) ) {
398 				assertSame( ownerCollection.getOwner().getTable(), fk.getReferencedTable() );
399 				hasOwnerFK = true;
400 			}
401 		}
402 		assertTrue( hasOwnerFK );
403 	}
