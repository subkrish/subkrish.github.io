24  package org.hibernate.test.annotations.mappedsuperclass.intermediate;
25  
26  import java.math.BigDecimal;
27  
28  import org.junit.Test;
29  
30  import org.hibernate.Session;
31  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
32  
33  import static org.junit.Assert.assertEquals;
34  
35  /**
36   * @author Steve Ebersole
37   */
38  public class IntermediateMappedSuperclassTest extends BaseCoreFunctionalTestCase {
39  	@Override
40  	protected Class<?>[] getAnnotatedClasses() {
41  		return new Class[] { AccountBase.class, Account.class, SavingsAccountBase.class, SavingsAccount.class };
42  	}
43  
44  	@Test
45  	public void testGetOnIntermediateMappedSuperclass() {
46  		final BigDecimal withdrawalLimit = new BigDecimal( 1000.00 ).setScale( 2 );
47  		Session session = openSession();
48  		session.beginTransaction();
49  		SavingsAccount savingsAccount = new SavingsAccount( "123", withdrawalLimit );
50  		session.save( savingsAccount );
51  		session.getTransaction().commit();
52  		session.close();
53  
54  		session = openSession();
55  		session.beginTransaction();
56  		Account account = (Account) session.get( Account.class, savingsAccount.getId() );
57  		// Oracle returns the BigDecimal with scale=0, which is equal to 1000 (not 1000.00);
58  		// compare using BigDecimal.doubleValue;
59  		assertEquals(
60  				withdrawalLimit.doubleValue(),
61  				( (SavingsAccount) account ).getWithdrawalLimit().doubleValue(),
62  				0.001);
63  		session.delete( account );
64  		session.getTransaction().commit();
65  		session.close();
66  	}
67  }
