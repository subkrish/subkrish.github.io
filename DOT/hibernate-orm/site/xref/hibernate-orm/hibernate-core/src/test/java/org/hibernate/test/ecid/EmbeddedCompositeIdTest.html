50  	@Test
51  	public void testMerge() {
52  		Session s = openSession();
53  		Transaction t = s.beginTransaction();
54  		Course uc =  new UniversityCourse("mat2000", "Monash", "second year maths", 0);
55  		Course c =  new Course("eng5000", "BHS", "grade 5 english");
56  		s.persist(uc);
57  		s.persist(c);
58  		t.commit();
59  		s.close();
60  		
61  		c.setDescription("Grade 5 English");
62  		uc.setDescription("Second year mathematics");
63  		
64  		s = openSession();
65  		t = s.beginTransaction();
66  		s.merge(c);
67  		s.merge(uc);
68  		t.commit();
69  		s.close();
70  		
71  		s = openSession();
72  		t = s.beginTransaction();
73  		s.delete(c);
74  		s.delete(uc);
75  		t.commit();
76  		s.close();
77  	}

79  	@Test
80  	@TestForIssue( jiraKey = "HHH-799" )
81  	public void testMerging() {
82  		Session s = openSession();
83  		Transaction t = s.beginTransaction();
84  		Course course = new Course( "EN-101", "BA", "preparatory english" );
85  		s.persist( course );
86  		t.commit();
87  		s.close();
88  
89  		String newDesc = "basic preparatory english";
90  		course.setDescription( newDesc );
91  
92  		s = openSession();
93  		t = s.beginTransaction();
94  		Course c = (Course) s.merge( course );
95  		assertEquals( "description not merged", newDesc, c.getDescription() );
96  		t.commit();
97  		s.close();
98  
99  		s = openSession();
100 		t = s.beginTransaction();
101 		Course cid = new Course( "EN-101", "BA", null );
102 		course = ( Course ) s.get( Course.class, cid );
103 		assertEquals( "description not merged", newDesc, course.getDescription() );
104 		s.delete( course );
105 		t.commit();
106 		s.close();
107 	}

109 	@Test
110 	public void testPolymorphism() {
111 		Session s = openSession();
112 		Transaction t = s.beginTransaction();
113 		Course uc =  new UniversityCourse("mat2000", "Monash", "second year maths", 0);
114 		Course c =  new Course("eng5000", "BHS", "grade 5 english");
115 		s.persist(uc);
116 		s.persist(c);
117 		t.commit();
118 		s.close();
119 		
120 		s = openSession();
121 		t = s.beginTransaction();
122 		Course ucid = new Course("mat2000", "Monash", null);
123 		Course cid =  new Course("eng5000", "BHS", null);
124 		Course luc = (Course) s.load(Course.class, ucid);
125 		Course lc = (Course) s.load(Course.class, cid);
126 		assertFalse( Hibernate.isInitialized(luc) );
127 		assertFalse( Hibernate.isInitialized(lc) );
128 		assertEquals( UniversityCourse.class, Hibernate.getClass(luc) );
129 		assertEquals( Course.class, Hibernate.getClass(lc) );
130 		assertSame( ( (HibernateProxy) lc ).getHibernateLazyInitializer().getImplementation(), cid );
131 		assertEquals( c.getCourseCode(), "eng5000" );
132 		assertEquals( uc.getCourseCode(), "mat2000" );
133 		t.commit();
134 		s.close();
135 
136 		s = openSession();
137 		t = s.beginTransaction();
138 		ucid = new Course("mat2000", "Monash", null);
139 		cid =  new Course("eng5000", "BHS", null);
140 		luc = (Course) s.get(Course.class, ucid);
141 		lc = (Course) s.get(Course.class, cid);
142 		assertTrue( Hibernate.isInitialized(luc) );
143 		assertTrue( Hibernate.isInitialized(lc) );
144 		assertEquals( UniversityCourse.class, Hibernate.getClass(luc) );
145 		assertEquals( Course.class, Hibernate.getClass(lc) );
146 		assertSame( lc, cid );
147 		assertEquals( c.getCourseCode(), "eng5000" );
148 		assertEquals( uc.getCourseCode(), "mat2000" );
149 		t.commit();
150 		s.close();
151 
152 		s = openSession();
153 		t = s.beginTransaction();
154 		List list = s.createQuery("from Course order by courseCode").list();
155 		assertTrue( list.get(0) instanceof Course );
156 		assertTrue( list.get(1) instanceof UniversityCourse );
157 		c = (Course) list.get(0);
158 		uc = (UniversityCourse) list.get(1);
159 		assertEquals( c.getCourseCode(), "eng5000" );
160 		assertEquals( uc.getCourseCode(), "mat2000" );
161 		t.commit();
162 		s.close();
163 		
164 		c.setDescription("Grade 5 English");
165 		uc.setDescription("Second year mathematics");
166 		
167 		s = openSession();
168 		t = s.beginTransaction();
169 		s.saveOrUpdate(c);
170 		s.saveOrUpdate(uc);
171 		t.commit();
172 		s.close();
173 		
174 		s = openSession();
175 		t = s.beginTransaction();
176 		s.delete(c);
177 		s.delete(uc);
178 		t.commit();
179 		s.close();
180 	}
