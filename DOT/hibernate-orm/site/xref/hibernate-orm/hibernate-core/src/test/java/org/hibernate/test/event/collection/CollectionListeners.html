54  public class CollectionListeners {
55  
56  	public interface Listener extends Serializable {
57  		void addEvent(AbstractCollectionEvent event, Listener listener);
58  	}
59  
60  	public static abstract class AbstractListener implements Listener {
61  
62  		private final CollectionListeners listeners;
63  
64  		protected AbstractListener( CollectionListeners listeners ) {
65  			this.listeners = listeners;
66  		}
67  
68  		public void addEvent(AbstractCollectionEvent event, Listener listener) {
69  			listeners.addEvent( event, listener );
70  		}
71  	}
72  
73  	public static class InitializeCollectionListener
74  			extends DefaultInitializeCollectionEventListener
75  			implements Listener {
76  		private final CollectionListeners listeners;
77  		private InitializeCollectionListener(CollectionListeners listeners) {
78  			this.listeners = listeners;
79  		}
80  		public void onInitializeCollection(InitializeCollectionEvent event) {
81  			super.onInitializeCollection( event );
82  			addEvent( event, this );
83  		}
84  		public void addEvent(AbstractCollectionEvent event, Listener listener) {
85  			listeners.addEvent( event, listener );
86  		}
87  	}
88  
89  	public static class PreCollectionRecreateListener extends AbstractListener
90  			implements PreCollectionRecreateEventListener {
91  		private PreCollectionRecreateListener(CollectionListeners listeners) {
92  			super( listeners );
93  		}
94  		public void onPreRecreateCollection(PreCollectionRecreateEvent event) {
95  			addEvent( event, this );
96  		}
97  	}
98  
99  	public static class PostCollectionRecreateListener extends AbstractListener
100 			implements PostCollectionRecreateEventListener {
101 		private PostCollectionRecreateListener(CollectionListeners listeners) {
102 			super( listeners );
103 		}
104 		public void onPostRecreateCollection(PostCollectionRecreateEvent event) {
105 			addEvent( event, this );
106 		}
107 	}
108 
109 	public static class PreCollectionRemoveListener extends AbstractListener
110 			implements PreCollectionRemoveEventListener {
111 		private PreCollectionRemoveListener(CollectionListeners listeners) {
112 			super( listeners );
113 		}
114 		public void onPreRemoveCollection(PreCollectionRemoveEvent event) {
115 			addEvent( event, this );
116 		}
117 	}
118 
119 	public static class PostCollectionRemoveListener extends AbstractListener
120 			implements PostCollectionRemoveEventListener {
121 		private PostCollectionRemoveListener(CollectionListeners listeners) {
122 			super( listeners );
123 		}
124 		public void onPostRemoveCollection(PostCollectionRemoveEvent event) {
125 			addEvent( event, this );
126 		}
127 	}
128 
129 	public static class PreCollectionUpdateListener extends AbstractListener
130 			implements PreCollectionUpdateEventListener {
131 		private PreCollectionUpdateListener(CollectionListeners listeners) {
132 			super( listeners );
133 		}
134 		public void onPreUpdateCollection(PreCollectionUpdateEvent event) {
135 			addEvent( event, this );
136 		}
137 	}
138 
139 	public static class PostCollectionUpdateListener extends AbstractListener
140 			implements PostCollectionUpdateEventListener {
141 		private PostCollectionUpdateListener(CollectionListeners listeners) {
142 			super( listeners );
143 		}
144 		public void onPostUpdateCollection(PostCollectionUpdateEvent event) {
145 			addEvent( event, this );
146 		}
147 	}
148 
149 	private final PreCollectionRecreateListener preCollectionRecreateListener;
150 	private final InitializeCollectionListener initializeCollectionListener;
151 	private final PreCollectionRemoveListener preCollectionRemoveListener;
152 	private final PreCollectionUpdateListener preCollectionUpdateListener;
153 	private final PostCollectionRecreateListener postCollectionRecreateListener;
154 	private final PostCollectionRemoveListener postCollectionRemoveListener;
155 	private final PostCollectionUpdateListener postCollectionUpdateListener;
156 
157 	private List listenersCalled = new ArrayList();
158 	private List events = new ArrayList();
159 
160 	public CollectionListeners( SessionFactory sf) {
161 		preCollectionRecreateListener = new PreCollectionRecreateListener( this );
162 		initializeCollectionListener = new InitializeCollectionListener( this );
163 		preCollectionRemoveListener = new PreCollectionRemoveListener( this );
164 		preCollectionUpdateListener = new PreCollectionUpdateListener( this );
165 		postCollectionRecreateListener = new PostCollectionRecreateListener( this );
166 		postCollectionRemoveListener = new PostCollectionRemoveListener( this );
167 		postCollectionUpdateListener = new PostCollectionUpdateListener( this );
168 
169 		EventListenerRegistry registry = ( (SessionFactoryImplementor) sf ).getServiceRegistry().getService( EventListenerRegistry.class );
170 		registry.setListeners( EventType.INIT_COLLECTION, initializeCollectionListener );
171 
172 		registry.setListeners( EventType.PRE_COLLECTION_RECREATE, preCollectionRecreateListener );
173 		registry.setListeners( EventType.POST_COLLECTION_RECREATE, postCollectionRecreateListener );
174 
175 		registry.setListeners( EventType.PRE_COLLECTION_REMOVE, preCollectionRemoveListener );
176 		registry.setListeners( EventType.POST_COLLECTION_REMOVE, postCollectionRemoveListener );
177 
178 		registry.setListeners( EventType.PRE_COLLECTION_UPDATE, preCollectionUpdateListener );
179 		registry.setListeners( EventType.POST_COLLECTION_UPDATE, postCollectionUpdateListener );
180 	}
181 
182 	public void addEvent(AbstractCollectionEvent event, Listener listener) {
183 		listenersCalled.add( listener );
184 		events.add( event );
185 	}
186 
187 	public List getListenersCalled() {
188 		return listenersCalled;
189 	}
190 
191 	public List getEvents() {
192 		return events;
193 	}
194 
195 	public void clear() {
196 		listenersCalled.clear();
197 		events.clear();
198 	}
199 
200 	public PreCollectionRecreateListener getPreCollectionRecreateListener() { return preCollectionRecreateListener; }
201 	public InitializeCollectionListener getInitializeCollectionListener() { return initializeCollectionListener; }
202 	public PreCollectionRemoveListener getPreCollectionRemoveListener() { return preCollectionRemoveListener; }
203 	public PreCollectionUpdateListener getPreCollectionUpdateListener() { return preCollectionUpdateListener; }
204 	public PostCollectionRecreateListener getPostCollectionRecreateListener() { return postCollectionRecreateListener; }
205 	public PostCollectionRemoveListener getPostCollectionRemoveListener() { return postCollectionRemoveListener; }
206 	public PostCollectionUpdateListener getPostCollectionUpdateListener() { return postCollectionUpdateListener; }
