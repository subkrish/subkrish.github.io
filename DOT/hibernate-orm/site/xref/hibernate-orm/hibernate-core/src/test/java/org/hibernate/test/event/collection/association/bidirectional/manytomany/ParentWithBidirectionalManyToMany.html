22  package org.hibernate.test.event.collection.association.bidirectional.manytomany;
23  import java.util.Collection;
24  import java.util.HashSet;
25  import java.util.Iterator;
26  
27  import org.hibernate.test.event.collection.AbstractParentWithCollection;
28  import org.hibernate.test.event.collection.Child;
29  
30  /**
31   *
32   * @author Gail Badner
33   */
34  public class ParentWithBidirectionalManyToMany extends AbstractParentWithCollection {
35  	public ParentWithBidirectionalManyToMany() {
36  	}
37  
38  	public ParentWithBidirectionalManyToMany(String name) {
39  		super( name );
40  	}
41  
42  	public void newChildren(Collection children) {
43  		if ( children == getChildren() ) {
44  			return;
45  		}
46  		if ( getChildren() != null ) {
47  			for ( Iterator it = getChildren().iterator(); it.hasNext(); ) {
48  				ChildWithBidirectionalManyToMany child = ( ChildWithBidirectionalManyToMany ) it.next();
49  				child.removeParent( this );
50  			}
51  		}
52  		if ( children != null ) {
53  			for ( Iterator it = children.iterator(); it.hasNext(); ) {
54  				ChildWithBidirectionalManyToMany child = ( ChildWithBidirectionalManyToMany ) it.next();
55  				child.addParent( this );
56  			}
57  		}
58  		super.newChildren( children );
59  	}
60  
61  	public Child createChild(String name) {
62  		return new ChildWithBidirectionalManyToMany( name, new HashSet() );
63  	}
64  
65  	public void addChild(Child child) {
66  		super.addChild( child );
67  		( ( ChildWithBidirectionalManyToMany ) child ).addParent( this );
68  	}
69  
70  	public void addAllChildren(Collection children) {
71  		super.addAllChildren( children );
72  		for ( Iterator it = children.iterator(); it.hasNext(); ) {
73  			ChildWithBidirectionalManyToMany child = ( ChildWithBidirectionalManyToMany ) it.next();
74  			child.addParent( this );
75  		}
76  	}
77  
78  	public void removeChild(Child child) {
79  		// Note: if the collection is a bag, the same child can be in the collection more than once
80  		super.removeChild( child );
81  		// only remove the parent from the child's set if child is no longer in the collection
82  		if ( ! getChildren().contains( child ) ) {
83  			( ( ChildWithBidirectionalManyToMany ) child ).removeParent( this );
84  		}
85  	}
86  
87  	public void removeAllChildren(Collection children) {
88  		super.removeAllChildren( children );
89  		for ( Iterator it = children.iterator(); it.hasNext(); ) {
90  			ChildWithBidirectionalManyToMany child = ( ChildWithBidirectionalManyToMany ) it.next();
91  			child.removeParent( this );
92  		}
93  	}
94  
95  	public void clearChildren() {
96  		if ( getChildren() != null ) {
97  			for ( Iterator it = getChildren().iterator(); it.hasNext(); ) {
98  				ChildWithBidirectionalManyToMany child = ( ChildWithBidirectionalManyToMany ) it.next();
99  				child.removeParent( this );
100 			}
101 		}
102 		super.clearChildren();
103 	}
104 }
