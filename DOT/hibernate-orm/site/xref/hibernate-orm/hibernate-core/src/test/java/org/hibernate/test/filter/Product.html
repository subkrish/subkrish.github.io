92  	public void addCategory(Category category) {
93  		if ( category == null ) {
94  			return;
95  		}
96  
97  		if ( categories == null ) {
98  			categories = new HashSet();
99  		}
100 
101 		categories.add( category );
102 		if ( category.getProducts() == null ) {
103 			category.setProducts( new HashSet() );
104 		}
105 		category.getProducts().add( this );
106 	}
