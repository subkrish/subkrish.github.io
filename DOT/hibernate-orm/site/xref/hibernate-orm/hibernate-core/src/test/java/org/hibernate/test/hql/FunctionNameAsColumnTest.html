115 	@Test
116 	public void testGetMultiColumnSameNameAsArgFunctionHQL() {
117 		Session s = openSession();
118 		Transaction t = s.beginTransaction();
119 		EntityWithArgFunctionAsColumn e1 = new EntityWithArgFunctionAsColumn();
120 		e1.setLower( 3 );
121 		e1.setUpper( " abc " );
122 		EntityWithArgFunctionAsColumn e2 = new EntityWithArgFunctionAsColumn();
123 		e2.setLower( 999 );
124 		e2.setUpper( " xyz " );
125 		EntityWithFunctionAsColumnHolder holder1 = new EntityWithFunctionAsColumnHolder();
126 		holder1.getEntityWithArgFunctionAsColumns().add( e1 );
127 		EntityWithFunctionAsColumnHolder holder2 = new EntityWithFunctionAsColumnHolder();
128 		holder2.getEntityWithArgFunctionAsColumns().add( e2 );
129 		holder1.setNextHolder( holder2 );
130 		s.save( holder1 );
131 		t.commit();
132 		s.close();
133 
134 		s = openSession();
135 		t = s.beginTransaction();
136 		holder1 = ( EntityWithFunctionAsColumnHolder ) s.createQuery(
137 				"from EntityWithFunctionAsColumnHolder h left join fetch h.entityWithArgFunctionAsColumns " +
138 						"left join fetch h.nextHolder left join fetch h.nextHolder.entityWithArgFunctionAsColumns " +
139 						"where h.nextHolder is not null" )
140 				.uniqueResult();
141 		assertTrue( Hibernate.isInitialized( holder1.getEntityWithArgFunctionAsColumns() ) );
142 		assertTrue( Hibernate.isInitialized( holder1.getNextHolder() ) );
143 		assertTrue( Hibernate.isInitialized( holder1.getNextHolder().getEntityWithArgFunctionAsColumns() ) );
144 		assertEquals( 1, holder1.getEntityWithArgFunctionAsColumns().size() );
145 		e1 = ( EntityWithArgFunctionAsColumn ) holder1.getEntityWithArgFunctionAsColumns().iterator().next();
146 		assertEquals( 3, e1.getLower() );
147 		assertEquals( " abc ", e1.getUpper() );
148 		assertEquals( 1, holder1.getNextHolder().getEntityWithArgFunctionAsColumns().size() );
149 		e2 = ( EntityWithArgFunctionAsColumn ) ( holder1.getNextHolder() ).getEntityWithArgFunctionAsColumns().iterator().next();
150 		assertEquals( 999, e2.getLower() );
151 		assertEquals( " xyz ", e2.getUpper() );
152 		t.commit();
153 		s.close();
154 
155 		cleanup();
156 	}

294 	@Test
295 	public void testNoArgFcnAndColumnSameNameAsNoArgFunctionHQL() {
296 		SQLFunction function = sessionFactory().getSqlFunctionRegistry().findSQLFunction( "current_date" );
297 		if ( function == null || function.hasParenthesesIfNoArguments() ) {
298 			SkipLog.reportSkip( "current_date reuires ()", "tests noarg function that does not require ()" );
299 			return;
300 		}
301 
302 		Session s = openSession();
303 		Transaction t = s.beginTransaction();
304 		EntityWithNoArgFunctionAsColumn e1 = new EntityWithNoArgFunctionAsColumn();
305 		e1.setCurrentDate( "blah blah blah" );
306 		EntityWithNoArgFunctionAsColumn e2 = new EntityWithNoArgFunctionAsColumn();
307 		e2.setCurrentDate( "yadda yadda yadda" );
308 		EntityWithFunctionAsColumnHolder holder1 = new EntityWithFunctionAsColumnHolder();
309 		holder1.getEntityWithNoArgFunctionAsColumns().add( e1 );
310 		EntityWithFunctionAsColumnHolder holder2 = new EntityWithFunctionAsColumnHolder();
311 		holder2.getEntityWithNoArgFunctionAsColumns().add( e2 );
312 		holder1.setNextHolder( holder2 );
313 		s.save( holder1 );
314 		t.commit();
315 		s.close();
316 
317 		s = openSession();
318 		t = s.beginTransaction();
319 		List results = s.createQuery(
320 						"select str(current_date), currentDate from EntityWithNoArgFunctionAsColumn"
321 				).list();
322 		assertEquals( 2, results.size() );
323 		assertEquals( ( ( Object[] ) results.get( 0 ) )[ 0 ], ( ( Object[] ) results.get( 1 ) )[ 0 ] );
324 		assertTrue( ! ( ( Object[] ) results.get( 0 ) )[ 0 ].equals( ( ( Object[] ) results.get( 0 ) )[ 1 ] ) );
325 		assertTrue( ! ( ( Object[] ) results.get( 1 ) )[ 0 ].equals( ( ( Object[] ) results.get( 1 ) )[ 1 ] ) );
326 		assertTrue( ( ( Object[] ) results.get( 0 ) )[ 1 ].equals( e1.getCurrentDate() ) ||
327 				     ( ( Object[] ) results.get( 0 ) )[ 1 ].equals( e2.getCurrentDate() ) );
328 		assertTrue( ( ( Object[] ) results.get( 1 ) )[ 1 ].equals( e1.getCurrentDate() ) ||
329 				     ( ( Object[] ) results.get( 1 ) )[ 1 ].equals( e2.getCurrentDate() ) );
330 		assertFalse( ( ( Object[] ) results.get( 0 ) )[ 1 ].equals( ( ( Object[] ) results.get( 1 ) )[ 1 ] ) );
331 		t.commit();
332 		s.close();
333 
334 		cleanup();
335 	}
