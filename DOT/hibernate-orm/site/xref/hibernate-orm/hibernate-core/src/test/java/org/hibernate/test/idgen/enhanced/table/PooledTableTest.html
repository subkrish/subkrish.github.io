24  package org.hibernate.test.idgen.enhanced.table;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.id.enhanced.OptimizerFactory;
30  import org.hibernate.id.enhanced.TableGenerator;
31  import org.hibernate.persister.entity.EntityPersister;
32  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
33  
34  import static org.hibernate.id.IdentifierGeneratorHelper.BasicHolder;
35  import static org.hibernate.testing.junit4.ExtraAssertions.assertClassAssignability;
36  import static org.junit.Assert.assertEquals;
37  
38  /**
39   * @author Steve Ebersole
40   */
41  public class PooledTableTest extends BaseCoreFunctionalTestCase {
42  	@Override
43  	public String[] getMappings() {
44  		return new String[] { "idgen/enhanced/table/Pooled.hbm.xml" };
45  	}
46  
47  	@Test
48  	public void testNormalBoundary() {
49  		EntityPersister persister = sessionFactory().getEntityPersister( Entity.class.getName() );
50  		assertClassAssignability( TableGenerator.class, persister.getIdentifierGenerator().getClass() );
51  		TableGenerator generator = ( TableGenerator ) persister.getIdentifierGenerator();
52  		assertClassAssignability( OptimizerFactory.PooledOptimizer.class, generator.getOptimizer().getClass() );
53  		OptimizerFactory.PooledOptimizer optimizer = ( OptimizerFactory.PooledOptimizer ) generator.getOptimizer();
54  
55  		int increment = optimizer.getIncrementSize();
56  		Entity[] entities = new Entity[ increment + 1 ];
57  		Session s = openSession();
58  		s.beginTransaction();
59  		for ( int i = 0; i < increment; i++ ) {
60  			entities[i] = new Entity( "" + ( i + 1 ) );
61  			s.save( entities[i] );
62  			assertEquals( 2, generator.getTableAccessCount() ); // initialization calls seq twice
63  			assertEquals( increment + 1, ( (BasicHolder) optimizer.getLastSourceValue() ).getActualLongValue() ); // initialization calls seq twice
64  			assertEquals( i + 1, ( (BasicHolder) optimizer.getLastValue() ).getActualLongValue() );
65  			assertEquals( increment + 1, ( (BasicHolder) optimizer.getLastSourceValue() ).getActualLongValue() );
66  		}
67  		// now force a "clock over"
68  		entities[ increment ] = new Entity( "" + increment );
69  		s.save( entities[ increment ] );
70  		assertEquals( 3, generator.getTableAccessCount() ); // initialization (2) + clock over
71  		assertEquals( ( increment * 2 ) + 1, ( (BasicHolder) optimizer.getLastSourceValue() ).getActualLongValue() ); // initialization (2) + clock over
72  		assertEquals( increment + 1, ( (BasicHolder) optimizer.getLastValue() ).getActualLongValue() );
73  		s.getTransaction().commit();
74  
75  		s.beginTransaction();
76  		for ( int i = 0; i < entities.length; i++ ) {
77  			assertEquals( i + 1, entities[i].getId().intValue() );
78  			s.delete( entities[i] );
79  		}
80  		s.getTransaction().commit();
81  		s.close();
82  	}
