46  	@Test
47  	public void testHqlIdPropertyReferences() {
48  		Session s = openSession();
49  		s.beginTransaction();
50  		Person p = new Person( new Long(1), "steve", 123 );
51  		s.save( p );
52  		Order o = new Order( new Long(1), p );
53  		LineItem l = new LineItem( o, "my-product", 2 );
54  		l.setId( "456" );
55  		s.save( o );
56  		s.getTransaction().commit();
57  		s.close();
58  
59  		s = openSession();
60  		s.beginTransaction();
61  
62  		long count = extractCount( s, "select count(*) from Person p where p.id = 123" );
63  		assertEquals( "Person by id prop (non-identifier)", 1, count );
64  		count = extractCount( s, "select count(*) from Person p where p.pk = 1" );
65  		assertEquals( "Person by pk prop (identifier)", 1, count );
66  
67  		count = extractCount( s, "select count(*) from Order o where o.id = 1" );
68  		assertEquals( "Order by number prop (named identifier)", 1, count );
69  		count = extractCount( s, "select count(*) from Order o where o.number = 1" );
70  		assertEquals( "Order by id prop (virtual identifier)", 1, count );
71  
72  		count = extractCount( s, "select count(*) from LineItem l where l.id = '456'" );
73  		assertEquals( "LineItem by id prop (non-identifier", 1, count );
74  
75  		if ( getDialect().supportsRowValueConstructorSyntax() ) {
76  			Query q = s.createQuery( "select count(*) from LineItem l where l.pk = (:order, :product)" )
77  					.setEntity( "order", o )
78  					.setString( "product", "my-product" );
79  			count = extractCount( q );
80  			assertEquals( "LineItem by pk prop (named composite identifier", 1, count );
81  		}
82  
83  		count = extractCount( s, "select count(*) from Order o where o.orderee.id = 1" );
84  		assertEquals( 0, count );
85  		count = extractCount( s, "select count(*) from Order o where o.orderee.pk = 1" );
86  		assertEquals( 1, count );
87  		count = extractCount( s, "select count(*) from Order o where o.orderee.id = 123" );
88  		assertEquals( 1, count );
89  
90  		count = extractCount( s, "select count(*) from LineItem l where l.pk.order.id = 1" );
91  		assertEquals( 1, count );
92  		count = extractCount( s, "select count(*) from LineItem l where l.pk.order.number = 1" );
93  		assertEquals( 1, count );
94  		count = extractCount( s, "select count(*) from LineItem l where l.pk.order.orderee.pk = 1" );
95  		assertEquals( 1, count );
96  
97  		s.delete( o );
98  		s.delete( p );
99  		s.getTransaction().commit();
100 		s.close();
101 	}

103 	@Test
104 	public void testCriteriaIdPropertyReferences() {
105 		Session s = openSession();
106 		s.beginTransaction();
107 		Person p = new Person( new Long(1), "steve", 123 );
108 		s.save( p );
109 		Order o = new Order( new Long(1), p );
110 		LineItem l = new LineItem( o, "my-product", 2 );
111 		l.setId( "456" );
112 		s.save( o );
113 		s.getTransaction().commit();
114 		s.close();
115 
116 		s = openSession();
117 		s.beginTransaction();
118 
119 		Criteria crit = s.createCriteria( Person.class );
120 		crit.setProjection( Projections.rowCount() );
121 		crit.add( Restrictions.eq( "id", new Integer(123) ) );
122 		long count = extractCount( crit );
123 		assertEquals( "Person by id prop (non-identifier)", 1, count );
124 
125 		crit = s.createCriteria( Person.class );
126 		crit.setProjection( Projections.rowCount() );
127 		crit.add( Restrictions.eq( "pk", new Long(1) ) );
128 		count = extractCount( crit );
129 		assertEquals( "Person by pk prop (identifier)", 1, count );
130 
131 		crit = s.createCriteria( Order.class );
132 		crit.setProjection( Projections.rowCount() );
133 		crit.add(  Restrictions.eq( "number", new Long(1) ) );
134 		count = extractCount( crit );
135 		assertEquals( "Order by number prop (named identifier)", 1, count );
136 
137 		crit = s.createCriteria( Order.class );
138 		crit.setProjection( Projections.rowCount() );
139 		crit.add(  Restrictions.eq( "id", new Long(1) ) );
140 		count = extractCount( crit );
141 		assertEquals( "Order by id prop (virtual identifier)", 1, count );
142 
143 		crit = s.createCriteria( LineItem.class );
144 		crit.setProjection( Projections.rowCount() );
145 		crit.add(  Restrictions.eq( "id", "456" ) );
146 		count = extractCount( crit );
147 		assertEquals( "LineItem by id prop (non-identifier", 1, count );
148 
149 		if ( getDialect().supportsRowValueConstructorSyntax() ) {
150 			crit = s.createCriteria( LineItem.class );
151 			crit.setProjection( Projections.rowCount() );
152 			crit.add( Restrictions.eq( "pk", new LineItemPK( o, "my-product" ) ) );
153 			count = extractCount( crit );
154 			assertEquals( "LineItem by pk prop (named composite identifier)", 1, count );
155 		}
156 
157 		crit = s.createCriteria( Order.class );
158 		crit.setProjection( Projections.rowCount() );
159 		crit.createAlias( "orderee", "p" ).add( Restrictions.eq( "p.id", new Integer(1) ) );
160 		count = extractCount( crit );
161 		assertEquals( 0, count );
162 
163 		crit = s.createCriteria( Order.class );
164 		crit.setProjection( Projections.rowCount() );
165 		crit.createAlias( "orderee", "p" ).add( Restrictions.eq( "p.pk", new Long(1) ) );
166 		count = extractCount( crit );
167 		assertEquals( 1, count );
168 
169 		crit = s.createCriteria( Order.class );
170 		crit.setProjection( Projections.rowCount() );
171 		crit.createAlias( "orderee", "p" ).add( Restrictions.eq( "p.id", new Integer(123) ) );
172 		count = extractCount( crit );
173 		assertEquals( 1, count );
174 
175 		crit = s.createCriteria( LineItem.class );
176 		crit.setProjection( Projections.rowCount() );
177 		crit.add( Restrictions.eq( "pk.order.id", new Long(1) ) );
178 		count = extractCount( crit );
179 		assertEquals( 1, count );
180 
181 		crit = s.createCriteria( LineItem.class );
182 		crit.setProjection( Projections.rowCount() );
183 		crit.add( Restrictions.eq( "pk.order.number", new Long(1) ) );
184 		count = extractCount( crit );
185 		assertEquals( 1, count );
186 
187 		s.delete( o );
188 		s.delete( p );
189 		s.getTransaction().commit();
190 		s.close();
191 	}

193 	private long extractCount(Session s, String hql) {
194 		return extractCount( s.createQuery( hql ) );
195 	}
196 
197 	private long extractCount(Query query) {
198 		return ( ( Long ) query.list().get( 0 ) ).longValue();
199 	}
200 
201 	private long extractCount(Criteria crit) {
202 		return ( ( Long ) crit.list().get( 0 ) ).longValue();
203 	}
