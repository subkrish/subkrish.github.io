173 
174 		s = openSession();
175 		t = s.beginTransaction();

177 		foo2 = (FooProxy) s.load(Foo.class, id2);
178 		assertFalse( Hibernate.isInitialized( foo ) );
179 		Hibernate.initialize( foo2 );
180 		Hibernate.initialize(foo);
181 		assertTrue( foo.getComponent().getImportantDates().length==4 );
182 		assertTrue( foo2.getComponent().getImportantDates().length == 4 );
183 		t.commit();
184 		s.close();
185 
186 		s = openSession();
187 		t = s.beginTransaction();
188 		foo.setKey( "xyzid" );
189 		foo.setFloat( new Float( 1.2f ) );
190 		foo2.setKey( (String) id ); //intentionally id, not id2!
191 		foo2.setFloat( new Float( 1.3f ) );
192 		foo2.getDependent().setKey( null );
193 		foo2.getComponent().getSubcomponent().getFee().setKey(null);
194 		assertFalse( foo2.getKey().equals( id ) );
195 		s.save( foo );
196 		s.update( foo2 );
197 		assertEquals( foo2.getKey(), id );
198 		assertTrue( foo2.getInt() == 1234567 );
199 		assertEquals( foo.getKey(), "xyzid" );
200 		t.commit();
201 		s.close();
202 		
203 		s = openSession();
204 		t = s.beginTransaction();

206 		assertTrue( foo.getInt() == 1234567 );
207 		assertTrue( foo.getComponent().getImportantDates().length==4 );
208 		String feekey = foo.getDependent().getKey();
209 		String fookey = foo.getKey();
210 		s.delete( foo );
211 		s.delete( s.get(Foo.class, id2) );
212 		s.delete( s.get(Foo.class, "xyzid") );
213 // here is the issue (HHH-4092).  After the deletes above there are 2 Fees and a Glarch unexpectedly hanging around
214 		assertEquals( 2, doDelete( s, "from java.lang.Object" ) );
215 		t.commit();
216 		s.close();
217 		
218 		//to account for new id rollback shit
219 		foo.setKey(fookey);

221 		foo.getComponent().setGlarch( null );
222 		foo.getComponent().setSubcomponent(null);
223 		
224 		s = openSession();
225 		t = s.beginTransaction();
226 		//foo.getComponent().setGlarch(null); //no id property!
227 		s.replicate( foo, ReplicationMode.OVERWRITE );
228 		t.commit();
229 		s.close();
230

236 		t.commit();

242 		Session s = openSession();
243 		Transaction t = s.beginTransaction();
244 		Baz baz = new Baz();
245 		s.save(baz);
246 		baz.setDefaults();
247 		Map topGlarchez = new HashMap();
248 		baz.setTopGlarchez(topGlarchez);
249 		Glarch g1 = new Glarch();
250 		g1.setName("g1");
251 		s.save(g1);
252 		Glarch g2 = new Glarch();
253 		g2.setName("g2");
254 		s.save(g2);
255 		g1.setProxyArray( new GlarchProxy[] {g2} );
256 		topGlarchez.put( new Character('1'),g1 );
257 		topGlarchez.put( new Character('2'), g2);
258 		Foo foo1 = new Foo();
259 		Foo foo2 = new Foo();
260 		s.save(foo1);
261 		s.save(foo2);
262 		baz.getFooSet().add(foo1);
263 		baz.getFooSet().add(foo2);
264 		baz.setFooArray( new FooProxy[] { foo1 } );
265 
266 		LockMode lockMode = supportsLockingNullableSideOfJoin( getDialect() ) ? LockMode.UPGRADE : LockMode.READ;
267 
268 		Criteria crit = s.createCriteria(Baz.class);
269 		crit.createCriteria("topGlarchez")
270 			.add( Restrictions.isNotNull("name") )
271 			.createCriteria("proxyArray")
272 				.add( Restrictions.eqProperty("name", "name") )
273 				.add( Restrictions.eq("name", "g2") )
274 				.add( Restrictions.gt("x", new Integer(-666) ) );
275 		crit.createCriteria("fooSet")
276 			.add( Restrictions.isNull("null") )
277 			.add( Restrictions.eq("string", "a string") )
278 			.add( Restrictions.lt("integer", new Integer(-665) ) );
279 		crit.createCriteria("fooArray")
280 				// this is the bit causing the problems; creating the criteria on fooArray does not add it to FROM,
281 				// and so restriction below leads to an invalid reference.
282 			.add( Restrictions.eq("string", "a string") )
283 			.setLockMode(lockMode);
284 
285 		List list = crit.list();
286 		assertTrue( list.size()==2 );
287 		
288 		s.createCriteria(Glarch.class).setLockMode(LockMode.UPGRADE).list();
289 		s.createCriteria(Glarch.class).setLockMode(Criteria.ROOT_ALIAS, LockMode.UPGRADE).list();
290 		
291 		g2.setName(null);
292 		t.commit();
293 		s.close();
294 		
295 		s = openSession();
296 		t = s.beginTransaction();
297 		
298 		list = s.createCriteria(Baz.class).add( Restrictions.isEmpty("fooSet") ).list();
299 		assertEquals( list.size(), 0 );
300 
301 		list = s.createCriteria(Baz.class).add( Restrictions.isNotEmpty("fooSet") ).list();
302 		assertEquals( new HashSet(list).size(), 1 );
303 
304 		list = s.createCriteria(Baz.class).add( Restrictions.sizeEq("fooSet", 2) ).list();
305 		assertEquals( new HashSet(list).size(), 1 );
306 		
307 		t.commit();
308 		s.close();
309 
310 		s = openSession();
311 		t = s.beginTransaction();
312 
313 		crit = s.createCriteria(Baz.class)
314 			.setLockMode(lockMode);
315 		crit.createCriteria("topGlarchez")
316 			.add( Restrictions.gt( "x", new Integer(-666) ) );
317 		crit.createCriteria("fooSet")
318 			.add( Restrictions.isNull("null") );
319 		list = crit.list();
320 
321 		assertTrue( list.size()==4 );
322 		baz = (Baz) crit.uniqueResult();
323 		assertTrue( Hibernate.isInitialized(baz.getTopGlarchez()) ); //cos it is nonlazy
324 		assertTrue( !Hibernate.isInitialized(baz.getFooSet()) );
325 
326 		list = s.createCriteria(Baz.class)
327 			.createCriteria("fooSet")
328 				.createCriteria("foo")
329 					.createCriteria("component.glarch")
330 						.add( Restrictions.eq("name", "xxx") )
331 			.list();
332 		assertTrue( list.size()==0 );
333 
334 		list = s.createCriteria(Baz.class)
335 			.createAlias("fooSet", "foo")
336 			.createAlias("foo.foo", "foo2")
337 			.setLockMode("foo2", lockMode)
338 			.add( Restrictions.isNull("foo2.component.glarch") )
339 			.createCriteria("foo2.component.glarch")
340 				.add( Restrictions.eq("name", "xxx") )
341 			.list();
342 		assertTrue( list.size()==0 );
343 
344 		t.commit();
345 		s.close();
346 
347 		s = openSession();
348 		t = s.beginTransaction();
349 
350 		crit = s.createCriteria(Baz.class);
351 		crit.createCriteria("topGlarchez")
352 			.add( Restrictions.isNotNull("name") );
353 		crit.createCriteria("fooSet")
354 			.add( Restrictions.isNull("null") );
355 
356 		list = crit.list();
357 		assertTrue( list.size()==2 );
358 		baz = (Baz) crit.uniqueResult();
359 		assertTrue( Hibernate.isInitialized(baz.getTopGlarchez()) ); //cos it is nonlazy
360 		assertTrue( !Hibernate.isInitialized(baz.getFooSet()) );
361 		
362 		s.createCriteria(Child.class).setFetchMode("parent", FetchMode.JOIN).list();
363 
364 		doDelete( s, "from Glarch g" );
365 		s.delete( s.get(Foo.class, foo1.getKey() ) );
366 		s.delete( s.get(Foo.class, foo2.getKey() ) );
367 		s.delete(baz);
368 		t.commit();
369 		s.close();
370 	}
