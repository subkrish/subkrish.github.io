47  @RequiresDialect( Oracle8iDialect.class )
48  public class CursorFromCallableTest extends BaseCoreFunctionalTestCase {
49  	@Override
50  	protected Class<?>[] getAnnotatedClasses() {
51  		return new Class<?>[] { NumValue.class };
52  	}

54  	@Before
55  	public void createRefCursorFunction() {
56  		executeStatement( "CREATE OR REPLACE FUNCTION f_test_return_cursor RETURN SYS_REFCURSOR " +
57  				"IS " +
58  				"    l_Cursor SYS_REFCURSOR; " +
59  				"BEGIN " +
60  				"    OPEN l_Cursor FOR " +
61  				"      SELECT 1 AS BOT_NUM " +
62  				"           , 'Line 1' AS BOT_VALUE " +
63  				"        FROM DUAL " +
64  				"      UNION " +
65  				"      SELECT 2 AS BOT_NUM " +
66  				"           , 'Line 2' AS BOT_VALUE " +
67  				"        FROM DUAL; " +
68  				"    RETURN(l_Cursor); " +
69  				"END f_test_return_cursor;" );
70  	}
71  
72  	@After
73  	public void dropRefCursorFunction() {
74  		executeStatement( "DROP FUNCTION f_test_return_cursor" );
75  	}

77  	@Test
78  	@TestForIssue( jiraKey = "HHH-8022" )
79  	public void testReadResultSetFromRefCursor() {
80  		Session session = openSession();
81  		session.getTransaction().begin();
82  
83  		Assert.assertEquals(
84  				Arrays.asList( new NumValue( 1, "Line 1" ), new NumValue( 2, "Line 2" ) ),
85  				session.getNamedQuery( "NumValue.getSomeValues" ).list()
86  		);
87  
88  		session.getTransaction().commit();
89  		session.close();
90  	}
91  
92  	@Test
93  	@TestForIssue( jiraKey = "HHH-7984" )
94  	public void testStatementClosing() {
95  		Session session = openSession();
96  		session.getTransaction().begin();
97  		// Reading maximum number of opened cursors requires SYS privileges.
98  		// Verify statement closing with JdbcCoordinator#hasRegisteredResources() instead.
99  		// BigDecimal maxCursors = (BigDecimal) session.createSQLQuery( "SELECT value FROM v$parameter WHERE name = 'open_cursors'" ).uniqueResult();
100 		// for ( int i = 0; i < maxCursors + 10; ++i ) { named_query_execution }
101 		Assert.assertEquals(
102 				Arrays.asList( new NumValue( 1, "Line 1" ), new NumValue( 2, "Line 2" ) ),
103 				session.getNamedQuery( "NumValue.getSomeValues" ).list()
104 		);
105 		JdbcCoordinator jdbcCoordinator = ( (SessionImplementor) session ).getTransactionCoordinator().getJdbcCoordinator();
106 		Assert.assertFalse(
107 				"Prepared statement and result set should be released after query execution.",
108 				jdbcCoordinator.hasRegisteredResources()
109 		);
110 		session.getTransaction().commit();
111 		session.close();
112 	}
113 
114 	private void executeStatement(final String sql) {
115 		final Session session = openSession();
116 		session.getTransaction().begin();
117 
118 		session.doWork( new Work() {
119 			@Override
120 			public void execute(Connection connection) throws SQLException {
121 				final JdbcCoordinator jdbcCoordinator = ( (SessionImplementor) session ).getTransactionCoordinator().getJdbcCoordinator();
122 				final StatementPreparer statementPreparer = jdbcCoordinator.getStatementPreparer();
123 				final ResultSetReturn resultSetReturn = jdbcCoordinator.getResultSetReturn();
124 				PreparedStatement preparedStatement = null;
125 				try {
126 					preparedStatement = statementPreparer.prepareStatement( sql );
127 					resultSetReturn.execute( preparedStatement );
128 				}
129 				finally {
130 					if ( preparedStatement != null ) {
131 						try {
132 							jdbcCoordinator.release( preparedStatement );
133 						}
134 						catch ( Throwable ignore ) {
135 							// ignore...
136 						}
137 					}
138 				}
139 			}
140 		} );
141 
142 		session.getTransaction().commit();
143 		session.close();
144 	}
