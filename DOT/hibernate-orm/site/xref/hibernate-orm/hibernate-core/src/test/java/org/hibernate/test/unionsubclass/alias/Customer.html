24  package org.hibernate.test.unionsubclass.alias;
25  import java.io.Serializable;
26  
27  /**
28   * 
29   * @author Strong Liu <stliu@redhat.com>
30   */
31  public abstract class Customer implements Serializable {
32      private PersonID id;
33  
34      public PersonID getId() {
35          return id;
36      }
37  
38      public void setId( PersonID id ) {
39          this.id = id;
40      }
41  
42      public boolean equals( Object obj ) {
43          if ( obj == null )
44              return false;
45          if ( obj == this )
46              return true;
47          if ( !( obj instanceof Customer ) )
48              return false;
49          return ( (Customer) obj ).getId().equals( getId() );
50      }
51  
52      public int hashCode() {
53          return id.hashCode();
54      }
55  
56  }
