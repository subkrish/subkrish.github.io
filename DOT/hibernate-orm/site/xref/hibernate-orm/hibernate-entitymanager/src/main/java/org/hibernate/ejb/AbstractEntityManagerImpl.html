227 	private void setLockOptions(Map<String, Object> props, LockOptions options) {
228 		Object lockScope = props.get( AvailableSettings.LOCK_SCOPE );
229 		if ( lockScope instanceof String && PessimisticLockScope.valueOf( ( String ) lockScope ) == PessimisticLockScope.EXTENDED ) {
230 			options.setScope( true );
231 		}
232 		else if ( lockScope instanceof PessimisticLockScope ) {
233 			boolean extended = PessimisticLockScope.EXTENDED.equals( lockScope );
234 			options.setScope( extended );
235 		}
236 		else if ( lockScope != null ) {
237 			throw new PersistenceException( "Unable to parse " + AvailableSettings.LOCK_SCOPE + ": " + lockScope );
238 		}
239 
240 		Object lockTimeout = props.get( AvailableSettings.LOCK_TIMEOUT );
241 		int timeout = 0;
242 		boolean timeoutSet = false;
243 		if ( lockTimeout instanceof String ) {
244 			timeout = Integer.parseInt( ( String ) lockTimeout );
245 			timeoutSet = true;
246 		}
247 		else if ( lockTimeout instanceof Number ) {
248 			timeout = ( (Number) lockTimeout ).intValue();
249 			timeoutSet = true;
250 		}
251 		else if ( lockTimeout != null ) {
252 			throw new PersistenceException( "Unable to parse " + AvailableSettings.LOCK_TIMEOUT + ": " + lockTimeout );
253 		}
254 		if ( timeoutSet ) {
255 			if ( timeout < 0 ) {
256 				options.setTimeOut( LockOptions.WAIT_FOREVER );
257 			}
258 			else if ( timeout == 0 ) {
259 				options.setTimeOut( LockOptions.NO_WAIT );
260 			}
261 			else {
262 				options.setTimeOut( timeout );
263 			}
264 		}
265 	}

313 	public static class TupleBuilderTransformer extends BasicTransformerAdapter {
314 		private List<TupleElement<?>> tupleElements;
315 		private Map<String,HqlTupleElementImpl> tupleElementsByAlias;
316 
317 		public TupleBuilderTransformer(org.hibernate.Query hqlQuery) {
318 			final Type[] resultTypes = hqlQuery.getReturnTypes();
319 			final int tupleSize = resultTypes.length;
320 
321 			this.tupleElements = CollectionHelper.arrayList( tupleSize );
322 
323 			final String[] aliases = hqlQuery.getReturnAliases();
324 			final boolean hasAliases = aliases != null && aliases.length > 0;
325 			this.tupleElementsByAlias = hasAliases
326 					? CollectionHelper.<String, HqlTupleElementImpl>mapOfSize( tupleSize )
327 					: Collections.<String, HqlTupleElementImpl>emptyMap();
328 
329 			for ( int i = 0; i < tupleSize; i++ ) {
330 				final HqlTupleElementImpl tupleElement = new HqlTupleElementImpl(
331 						i,
332 						aliases == null ? null : aliases[i],
333 						resultTypes[i]
334 				);
335 				tupleElements.add( tupleElement );
336 				if ( hasAliases ) {
337 					final String alias = aliases[i];
338 					if ( alias != null ) {
339 						tupleElementsByAlias.put( alias, tupleElement );
340 					}
341 				}
342 			}
343 		}
344 
345 		@Override
346 		public Object transformTuple(Object[] tuple, String[] aliases) {
347 			if ( tuple.length != tupleElements.size() ) {
348 				throw new IllegalArgumentException(
349 						"Size mismatch between tuple result [" + tuple.length + "] and expected tuple elements [" +
350 								tupleElements.size() + "]"
351 				);
352 			}
353 			return new HqlTupleImpl( tuple );
354 		}

447 			String jpaqlString,
448 			Class<T> resultClass,
449 			Selection selection,
450 			Options options) {
451 		try {
452 			org.hibernate.Query hqlQuery = getSession().createQuery( jpaqlString );
453 
454 			if ( options.getValueHandlers() == null ) {
455 				options.getResultMetadataValidator().validate( hqlQuery.getReturnTypes() );
456 			}
457 
458 			// determine if we need a result transformer
459 			List tupleElements = Tuple.class.equals( resultClass )
460 					? ( ( CompoundSelectionImpl<Tuple> ) selection ).getCompoundSelectionItems()
461 					: null;
462 			if ( options.getValueHandlers() != null || tupleElements != null ) {
463 				hqlQuery.setResultTransformer(
464 						new CriteriaQueryTransformer( options.getValueHandlers(), tupleElements )
465 				);
466 			}
467 			return new QueryImpl<T>( hqlQuery, this, options.getNamedParameterExplicitTypes() );
468 		}
469 		catch ( HibernateException he ) {
470 			throw convert( he );
471 		}

474 	private static class CriteriaQueryTransformer extends BasicTransformerAdapter {
475 		private final List<ValueHandlerFactory.ValueHandler> valueHandlers;
476 		private final List tupleElements;
477 
478 		private CriteriaQueryTransformer(List<ValueHandlerFactory.ValueHandler> valueHandlers, List tupleElements) {
479 			// todo : should these 2 sizes match *always*?
480 			this.valueHandlers = valueHandlers;
481 			this.tupleElements = tupleElements;
482 		}
483 
484 		@Override
485 		public Object transformTuple(Object[] tuple, String[] aliases) {
486 			final Object[] valueHandlerResult;
487 			if ( valueHandlers == null ) {
488 				valueHandlerResult = tuple;
489 			}
490 			else {
491 				valueHandlerResult = new Object[tuple.length];
492 				for ( int i = 0; i < tuple.length; i++ ) {
493 					ValueHandlerFactory.ValueHandler valueHandler = valueHandlers.get( i );
494 					valueHandlerResult[i] = valueHandler == null
495 							? tuple[i]
496 							: valueHandler.convert( tuple[i] );
497 				}
498 			}
499 
500 			return tupleElements == null
501 					? valueHandlerResult.length == 1 ? valueHandlerResult[0] : valueHandlerResult
502 					: new TupleImpl( tuple );
503 
504 		}

506 		private class TupleImpl implements Tuple {
507 			private final Object[] tuples;
508 
509 			private TupleImpl(Object[] tuples) {
510 				if ( tuples.length != tupleElements.size() ) {
511 					throw new IllegalArgumentException(
512 							"Size mismatch between tuple result [" + tuples.length
513 									+ "] and expected tuple elements [" + tupleElements.size() + "]"
514 					);
515 				}
516 				this.tuples = tuples;
517 			}
518 
519 			public <X> X get(TupleElement<X> tupleElement) {
520 				int index = tupleElements.indexOf( tupleElement );
521 				if ( index < 0 ) {
522 					throw new IllegalArgumentException(
523 							"Requested tuple element did not correspond to element in the result tuple"
524 					);
525 				}
526 				// index should be "in range" by nature of size check in ctor
527 				return ( X ) tuples[index];
528 			}
529 
530 			public Object get(String alias) {
531 				int index = -1;
532 				if ( alias != null ) {
533 					alias = alias.trim();
534 					if ( alias.length() > 0 ) {
535 						int i = 0;
536 						for ( TupleElement selection : ( List<TupleElement> ) tupleElements ) {
537 							if ( alias.equals( selection.getAlias() ) ) {
538 								index = i;
539 								break;
540 							}
541 							i++;
542 						}
543 					}
544 				}
545 				if ( index < 0 ) {
546 					throw new IllegalArgumentException(
547 							"Given alias [" + alias + "] did not correspond to an element in the result tuple"
548 					);
549 				}
550 				// index should be "in range" by nature of size check in ctor
551 				return tuples[index];
552 			}
553 
554 			public <X> X get(String alias, Class<X> type) {
555 				return ( X ) get( alias );
556 			}
557 
558 			public Object get(int i) {
559 				if ( i >= tuples.length ) {
560 					throw new IllegalArgumentException(
561 							"Given index [" + i + "] was outside the range of result tuple size [" + tuples.length + "] "
562 					);
563 				}
564 				return tuples[i];
565 			}
566 
567 			public <X> X get(int i, Class<X> type) {
568 				return ( X ) get( i );
569 			}
570 
571 			public Object[] toArray() {
572 				return tuples;
573 			}
574 
575 			public List<TupleElement<?>> getElements() {
576 				return tupleElements;
577 			}
578 		}
579 	}

605 	public <T> TypedQuery<T> createNamedQuery(String name, Class<T> resultClass) {
606 		try {
607 			/*
608 			 * Get the named query.
609 			 * If the named query is a SQL query, get the expected returned type from the query definition
610 			 * or its associated result set mapping
611 			 * If the named query is a HQL query, use getReturnType()
612 			 */
613 			org.hibernate.Query namedQuery = getSession().getNamedQuery( name );
614 			//TODO clean this up to avoid downcasting
615 			final SessionFactoryImplementor factoryImplementor = entityManagerFactory.getSessionFactory();
616 			final NamedSQLQueryDefinition queryDefinition = factoryImplementor.getNamedSQLQuery( name );
617 			try {
618 				if ( queryDefinition != null ) {
619 					Class<?> actualReturnedClass;
620 
621 					final NativeSQLQueryReturn[] queryReturns;
622 					if ( queryDefinition.getQueryReturns() != null ) {
623 						queryReturns = queryDefinition.getQueryReturns();
624 					}
625 					else if ( queryDefinition.getResultSetRef() != null ) {
626 						final ResultSetMappingDefinition rsMapping = factoryImplementor.getResultSetMapping(
627 								queryDefinition.getResultSetRef()
628 						);
629 						queryReturns = rsMapping.getQueryReturns();
630 					}
631 					else {
632 						throw new AssertionFailure( "Unsupported named query model. Please report the bug in Hibernate EntityManager");
633 					}
634 					if ( queryReturns.length > 1 ) {
635 						throw new IllegalArgumentException( "Cannot create TypedQuery for query with more than one return" );
636 					}
637 					final NativeSQLQueryReturn nativeSQLQueryReturn = queryReturns[0];
638 					if ( nativeSQLQueryReturn instanceof NativeSQLQueryRootReturn ) {
639 						final String entityClassName = ( ( NativeSQLQueryRootReturn ) nativeSQLQueryReturn ).getReturnEntityName();
640 						try {
641 							actualReturnedClass = ReflectHelper.classForName( entityClassName, AbstractEntityManagerImpl.class );
642 						}
643 						catch ( ClassNotFoundException e ) {
644 							throw new AssertionFailure( "Unable to instantiate class declared on named native query: " + name + " " + entityClassName );
645 						}
646 						if ( !resultClass.isAssignableFrom( actualReturnedClass ) ) {
647 							throw buildIncompatibleException( resultClass, actualReturnedClass );
648 						}
649 					}
650 					else {
651 						//TODO support other NativeSQLQueryReturn type. For now let it go.
652 					}
653 				}
654 				else {
655 					resultClassChecking( resultClass, namedQuery );
656 
657 				}
658 				return new QueryImpl<T>( namedQuery, this );
659 			}
660 			catch ( HibernateException he ) {
661 				throw convert( he );
662 			}
663 		}
664 		catch ( MappingException e ) {
665 			throw new IllegalArgumentException( "Named query not found: " + name );
666 		}
667 	}

669 	private <T> void resultClassChecking(Class<T> resultClass, org.hibernate.Query query) {
670 		// make sure the query is a select -> HHH-7192
671 		final SessionImplementor session = unwrap( SessionImplementor.class );
672 		final HQLQueryPlan queryPlan = session.getFactory().getQueryPlanCache().getHQLQueryPlan(
673 				query.getQueryString(),
674 				false,
675 				session.getLoadQueryInfluencers().getEnabledFilters()
676 		);
677 		if ( queryPlan.getTranslators()[0].isManipulationStatement() ) {
678 			throw new IllegalArgumentException( "Update/delete queries cannot be typed" );
679 		}
680 
681 		// do some return type validation checking
682 		if ( Object[].class.equals( resultClass ) ) {
683 			// no validation needed
684 		}
685 		else if ( Tuple.class.equals( resultClass ) ) {
686 			TupleBuilderTransformer tupleTransformer = new TupleBuilderTransformer( query );
687 			query.setResultTransformer( tupleTransformer );
688 		}
689 		else {
690 			final Class dynamicInstantiationClass = queryPlan.getDynamicInstantiationResultType();
691 			if ( dynamicInstantiationClass != null ) {
692 				if ( ! resultClass.isAssignableFrom( dynamicInstantiationClass ) ) {
693 					throw new IllegalArgumentException(
694 							"Mismatch in requested result type [" + resultClass.getName() +
695 									"] and actual result type [" + dynamicInstantiationClass.getName() + "]"
696 					);
697 				}
698 			}
699 			else if ( query.getReturnTypes().length == 1 ) {
700 				// if we have only a single return expression, its java type should match with the requested type
701 				if ( !resultClass.isAssignableFrom( query.getReturnTypes()[0].getReturnedClass() ) ) {
702 					throw new IllegalArgumentException(
703 							"Type specified for TypedQuery [" +
704 									resultClass.getName() +
705 									"] is incompatible with query return type [" +
706 									query.getReturnTypes()[0].getReturnedClass() + "]"
707 					);
708 				}
709 			}
710 			else {
711 				throw new IllegalArgumentException(
712 						"Cannot create TypedQuery for query with more than one return using requested result type [" +
713 								resultClass.getName() + "]"
714 				);
715 			}
716 		}
717 	}

725 					);

793 	public <A> A find(Class<A> entityClass, Object primaryKey, LockModeType lockModeType, Map<String, Object> properties) {
794 		CacheMode previousCacheMode = getSession().getCacheMode();
795 		CacheMode cacheMode = determineAppropriateLocalCacheMode( properties );
796 		LockOptions lockOptions = null;
797 		try {
798 			getSession().setCacheMode( cacheMode );
799 			if ( lockModeType != null ) {
800 				lockOptions = getLockRequest( lockModeType, properties );
801 				return ( A ) getSession().get(
802 						entityClass, ( Serializable ) primaryKey, 
803 						lockOptions
804 				);
805 			}
806 			else {
807 				return ( A ) getSession().get( entityClass, ( Serializable ) primaryKey );
808 			}
809 		}
810 		catch ( EntityNotFoundException ignored ) {
811 			// DefaultLoadEventListener.returnNarrowedProxy may throw ENFE (see HHH-7861 for details),
812 			// which find() should not throw.  Find() should return null if the entity was not found.
813 			if ( LOG.isDebugEnabled() ) {
814 				String entityName = entityClass != null ? entityClass.getName(): null;
815 				String identifierValue = primaryKey != null ? primaryKey.toString() : null ;
816 				LOG.ignoringEntityNotFound( entityName, identifierValue );
817 			}
818 			return null;
819 		}
820 		catch ( ObjectDeletedException e ) {
821 			//the spec is silent about people doing remove() find() on the same PC
822 			return null;
823 		}
824 		catch ( ObjectNotFoundException e ) {
825 			//should not happen on the entity itself with get
826 			throw new IllegalArgumentException( e.getMessage(), e );
827 		}
828 		catch ( MappingException e ) {
829 			throw new IllegalArgumentException( e.getMessage(), e );
830 		}
831 		catch ( TypeMismatchException e ) {
832 			throw new IllegalArgumentException( e.getMessage(), e );
833 		}
834 		catch ( ClassCastException e ) {
835 			throw new IllegalArgumentException( e.getMessage(), e );
836 		}
837 		catch ( HibernateException he ) {
838 			throw convert( he, lockOptions );
839 		}
840 		finally {
841 			getSession().setCacheMode( previousCacheMode );
842 		}
843 	}

1199 	private void joinTransaction(boolean explicitRequest) {
1200 		if ( transactionType != PersistenceUnitTransactionType.JTA ) {
1201 			if ( explicitRequest ) {
1202 			    LOG.callingJoinTransactionOnNonJtaEntityManager();
1203 			}
1204 			return;
1205 		}
1206 
1207 		final SessionImplementor session = (SessionImplementor) getSession();
1208 		final TransactionCoordinator transactionCoordinator = session.getTransactionCoordinator();
1209 		final TransactionImplementor transaction = transactionCoordinator.getTransaction();
1210 
1211 		transaction.markForJoin();
1212 		transactionCoordinator.pulse();
1213 
1214 		LOG.debug( "Looking for a JTA transaction to join" );
1215 		if ( ! transactionCoordinator.isTransactionJoinable() ) {
1216 			if ( explicitRequest ) {
1217 				// if this is an explicit join request, log a warning so user can track underlying cause
1218 				// of subsequent exceptions/messages
1219 				LOG.unableToJoinTransaction(Environment.TRANSACTION_STRATEGY);
1220 			}
1221 		}
1222 
1223 		try {
1224 			if ( transaction.getJoinStatus() == JoinStatus.JOINED ) {
1225 				LOG.debug( "Transaction already joined" );
1226 				return; // noop
1227 			}
1228 
1229 			// join the transaction and then recheck the status
1230 			transaction.join();
1231 			if ( transaction.getJoinStatus() == JoinStatus.NOT_JOINED ) {
1232 				if ( explicitRequest ) {
1233 					throw new TransactionRequiredException( "No active JTA transaction on joinTransaction call" );
1234 				}
1235 				else {
1236 					LOG.debug( "Unable to join JTA transaction" );
1237 					return;
1238 				}
1239 			}
1240 			else if ( transaction.getJoinStatus() == JoinStatus.MARKED_FOR_JOINED ) {
1241 				throw new AssertionFailure( "Transaction MARKED_FOR_JOINED after isOpen() call" );
1242 			}
1243 
1244 			// register behavior changes
1245 			SynchronizationCallbackCoordinator callbackCoordinator = transactionCoordinator.getSynchronizationCallbackCoordinator();
1246 			callbackCoordinator.setManagedFlushChecker( new ManagedFlushCheckerImpl() );
1247 			callbackCoordinator.setExceptionMapper( new CallbackExceptionMapperImpl() );
1248 			callbackCoordinator.setAfterCompletionAction( new AfterCompletionActionImpl( session, transactionType ) );
1249 		}
1250 		catch ( HibernateException he ) {
1251 			throw convert( he );
1252 		}
1253 	}

1281 		if ( e instanceof LockTimeoutException ) {
1282 			return;
1283 		}
1284 		if ( e instanceof QueryTimeoutException ) {
1285 			return;
1286 		}
1287 
1288 		try {
1289 			markAsRollback();
1290 		}
1291 		catch ( Exception ne ) {
1292 			//we do not want the subsequent exception to swallow the original one
1293             LOG.unableToMarkForRollbackOnPersistenceException(ne);
1294 		}
1295 	}

1324 	/**
1325 	 * {@inheritDoc}
1326 	 */
1327 	public RuntimeException convert(HibernateException e, LockOptions lockOptions) {
1328 		if ( e instanceof StaleStateException ) {
1329 			PersistenceException converted = wrapStaleStateException( ( StaleStateException ) e );
1330 			handlePersistenceException( converted );
1331 			return converted;
1332 		}
1333 		else if ( e instanceof LockingStrategyException ) {
1334 			PersistenceException converted = wrapLockException( e, lockOptions );
1335 			handlePersistenceException( converted );
1336 			return converted;
1337 		}
1338 		else if ( e instanceof org.hibernate.exception.LockTimeoutException ) {
1339 			PersistenceException converted = wrapLockException( e, lockOptions );
1340 			handlePersistenceException( converted );
1341 			return converted;
1342 		}
1343 		else if ( e instanceof org.hibernate.PessimisticLockException ) {
1344 			PersistenceException converted = wrapLockException( e, lockOptions );
1345 			handlePersistenceException( converted );
1346 			return converted;
1347 		}
1348 		else if ( e instanceof org.hibernate.QueryTimeoutException ) {
1349 			QueryTimeoutException converted = new QueryTimeoutException(e.getMessage(), e);
1350 			handlePersistenceException( converted );
1351 			return converted;
1352 		}
1353 		else if ( e instanceof ObjectNotFoundException ) {
1354 			EntityNotFoundException converted = new EntityNotFoundException( e.getMessage() );
1355 			handlePersistenceException( converted );
1356 			return converted;
1357 		}
1358         else if ( e instanceof org.hibernate.NonUniqueObjectException ) {
1359             EntityExistsException converted = new EntityExistsException( e.getMessage() );
1360             handlePersistenceException( converted );
1361             return converted;
1362         }
1363 		else if ( e instanceof org.hibernate.NonUniqueResultException ) {
1364 			NonUniqueResultException converted = new NonUniqueResultException( e.getMessage() );
1365 			handlePersistenceException( converted );
1366 			return converted;
1367 		}
1368 		else if ( e instanceof UnresolvableObjectException ) {
1369 			EntityNotFoundException converted = new EntityNotFoundException( e.getMessage() );
1370 			handlePersistenceException( converted );
1371 			return converted;
1372 		}
1373 		else if ( e instanceof QueryException ) {
1374 			return new IllegalArgumentException( e );
1375 		}
1376 		else if ( e instanceof TransientObjectException ) {
1377 			try {
1378 				markAsRollback();
1379 			}
1380 			catch ( Exception ne ) {
1381 				//we do not want the subsequent exception to swallow the original one
1382                 LOG.unableToMarkForRollbackOnTransientObjectException(ne);
1383 			}
1384 			return new IllegalStateException( e ); //Spec 3.2.3 Synchronization rules
1385 		}
1386 		else {
1387 			PersistenceException converted = new PersistenceException( e );
1388 			handlePersistenceException( converted );
1389 			return converted;
1390 		}
1391 	}

1401 	 * {@inheritDoc}
1402 	 */
1403 	public PersistenceException wrapStaleStateException(StaleStateException e) {
1404 		PersistenceException pe;
1405 		if ( e instanceof StaleObjectStateException ) {
1406 			StaleObjectStateException sose = ( StaleObjectStateException ) e;
1407 			Serializable identifier = sose.getIdentifier();
1408 			if ( identifier != null ) {
1409 				try {
1410 					Object entity = getRawSession().load( sose.getEntityName(), identifier );
1411 					if ( entity instanceof Serializable ) {
1412 						//avoid some user errors regarding boundary crossing
1413 						pe = new OptimisticLockException( e.getMessage(), e, entity );
1414 					}
1415 					else {
1416 						pe = new OptimisticLockException( e.getMessage(), e );
1417 					}
1418 				}
1419 				catch ( EntityNotFoundException enfe ) {
1420 					pe = new OptimisticLockException( e.getMessage(), e );
1421 				}
1422 			}
1423 			else {
1424 				pe = new OptimisticLockException( e.getMessage(), e );
1425 			}
1426 		}
1427 		else {
1428 			pe = new OptimisticLockException( e.getMessage(), e );
1429 		}
1430 		return pe;
1431 	}

1433 	public PersistenceException wrapLockException(HibernateException e, LockOptions lockOptions) {
1434 		final PersistenceException pe;
1435 		if ( e instanceof OptimisticEntityLockException ) {
1436 			final OptimisticEntityLockException lockException = (OptimisticEntityLockException) e;
1437 			pe = new OptimisticLockException( lockException.getMessage(), lockException, lockException.getEntity() );
1438 		}
1439 		else if ( e instanceof org.hibernate.exception.LockTimeoutException ) {
1440 			pe = new LockTimeoutException( e.getMessage(), e, null );
1441 		}
1442 		else if ( e instanceof PessimisticEntityLockException ) {
1443 			PessimisticEntityLockException lockException = (PessimisticEntityLockException) e;
1444 			if ( lockOptions != null && lockOptions.getTimeOut() > -1 ) {
1445 				// assume lock timeout occurred if a timeout or NO WAIT was specified
1446 				pe = new LockTimeoutException( lockException.getMessage(), lockException, lockException.getEntity() );
1447 			}
1448 			else {
1449 				pe = new PessimisticLockException( lockException.getMessage(), lockException, lockException.getEntity() );
1450 			}
1451 		}
1452 		else if ( e instanceof org.hibernate.PessimisticLockException ) {
1453 			org.hibernate.PessimisticLockException jdbcLockException = ( org.hibernate.PessimisticLockException ) e;
1454 			if ( lockOptions != null && lockOptions.getTimeOut() > -1 ) {
1455 				// assume lock timeout occurred if a timeout or NO WAIT was specified
1456 				pe = new LockTimeoutException( jdbcLockException.getMessage(), jdbcLockException, null );
1457 			}
1458 			else {
1459 				pe = new PessimisticLockException( jdbcLockException.getMessage(), jdbcLockException, null );
1460 			}
1461 		}
1462 		else {
1463 			pe = new OptimisticLockException( e );
1464 		}
1465 		return pe;
1466 	}

1468 	private static class AfterCompletionActionImpl implements AfterCompletionAction {
1469 		private final SessionImplementor session;
1470 		private final PersistenceUnitTransactionType transactionType;
1471 
1472 		private AfterCompletionActionImpl(SessionImplementor session, PersistenceUnitTransactionType transactionType) {
1473 			this.session = session;
1474 			this.transactionType = transactionType;
1475 		}
1476 
1477 		@Override
1478 		public void doAction(TransactionCoordinator transactionCoordinator, int status) {
1479 			if ( session.isClosed() ) {
1480                 LOG.trace("Session was closed; nothing to do");
1481 				return;
1482 			}
1483 
1484 			final boolean successful = JtaStatusHelper.isCommitted( status );
1485 			if ( !successful && transactionType == PersistenceUnitTransactionType.JTA ) {
1486 				( (Session) session ).clear();
1487 			}
1488 			session.getTransactionCoordinator().resetJoinStatus();
1489 		}
1490 	}
