22  package org.hibernate.ejb;
23  import java.io.ByteArrayInputStream;
24  import java.io.ObjectInputStream;
25  import java.util.Hashtable;
26  import javax.naming.Context;
27  import javax.naming.Name;
28  import javax.naming.Reference;
29  import javax.naming.spi.ObjectFactory;
30  
31  /**
32   * @author Emmanuel Bernard
33   */
34  public class Ejb3ConfigurationObjectFactory implements ObjectFactory {
35  	public Object getObjectInstance(
36  			Object reference, Name name, Context nameCtx, Hashtable<?, ?> environment
37  	) throws Exception {
38  		byte[] serialized = (byte[]) ( (Reference) reference ).get(0).getContent();
39  		ByteArrayInputStream byteIn = new ByteArrayInputStream( serialized );
40  		ObjectInputStream in = new ObjectInputStream( byteIn );
41  		Ejb3Configuration cfg = (Ejb3Configuration) in.readObject();
42  		in.close();
43  		byteIn.close();
44  		return cfg;
45  	}
46  }
