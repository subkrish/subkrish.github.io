24  package org.hibernate.ejb.criteria;
25  import javax.persistence.criteria.Selection;
26  
27  /**
28   * Contract for query components capable of eirther being a parameter or containing parameters.
29   *
30   * @author Steve Ebersole
31   */
32  public interface ParameterContainer {
33  	/**
34  	 * Register any parameters contained within this query component with the given registry.
35  	 *
36  	 * @param registry The parameter registry with which to register.
37  	 */
38  	public void registerParameters(ParameterRegistry registry);
39  
40  	/**
41  	 * Helper to deal with potential parameter container nodes.
42  	 */
43  	public static class Helper {
44  		public static void possibleParameter(Selection selection, ParameterRegistry registry) {
45  			if ( ParameterContainer.class.isInstance( selection ) ) {
46  				( (ParameterContainer) selection ).registerParameters( registry );
47  			}
48  		}
49  	}
50  }
