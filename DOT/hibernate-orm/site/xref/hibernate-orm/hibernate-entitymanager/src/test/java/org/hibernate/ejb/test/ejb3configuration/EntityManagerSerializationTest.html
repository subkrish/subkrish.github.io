50  public class EntityManagerSerializationTest extends BaseEntityManagerFunctionalTestCase {
51  	@Test
52  	public void testSerialization() throws Exception {
53  		EntityManager em = entityManagerFactory().createEntityManager();
54  		//em.getTransaction().begin();
55  		//em.setFlushMode( FlushModeType.NEVER );
56  		Cat cat = new Cat();
57  		cat.setAge( 3 );
58  		cat.setDateOfBirth( new Date() );
59  		cat.setLength( 22 );
60  		cat.setName( "Kitty" );
61  		em.persist( cat );
62  		Item item = new Item();
63  		item.setName( "Train Ticket" );
64  		item.setDescr( "Paris-London" );
65  		em.persist( item );
66  		//em.getTransaction().commit();
67  		//em.getTransaction().begin();
68  		item.setDescr( "Paris-Bruxelles" );
69  		//em.getTransaction().commit();
70  
71  		//fake the in container work
72  		( (HibernateEntityManager) em ).getSession().disconnect();
73  		ByteArrayOutputStream stream = new ByteArrayOutputStream();
74  		ObjectOutput out = new ObjectOutputStream( stream );
75  		out.writeObject( em );
76  		out.close();
77  		byte[] serialized = stream.toByteArray();
78  		stream.close();
79  		ByteArrayInputStream byteIn = new ByteArrayInputStream( serialized );
80  		ObjectInputStream in = new ObjectInputStream( byteIn );
81  		em = (EntityManager) in.readObject();
82  		in.close();
83  		byteIn.close();
84  		//fake the in container work
85  		em.getTransaction().begin();
86  		item = em.find( Item.class, item.getName() );
87  		item.setDescr( item.getDescr() + "-Amsterdam" );
88  		cat = (Cat) em.createQuery( "select c from " + Cat.class.getName() + " c" ).getSingleResult();
89  		cat.setLength( 34 );
90  		em.flush();
91  		em.remove( item );
92  		em.remove( cat );
93  		em.flush();
94  		em.getTransaction().commit();
95  
96  		em.close();
97  	}
98  
99  	/**
100 	 * Add a non-serializable object to the EMF to ensure that the EM can be serialized even if its EMF is not serializable.
101 	 * This will ensure that the fix for HHH-6897 doesn't regress,
102 	 * @return
103 	 */
104 	@Override
105 	protected Map getConfig() {
106 		Map result = super.getConfig();
107 		result.put( "org.hibernate.ejb.test.BaseEntityManagerFunctionalTestCase.getConfig_addedNotSerializableObject", new NotSerializableClass());
108 		return result;
109 	}
