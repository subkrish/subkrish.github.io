27  /**
28   * JGroups' Auto Node Selector Test
29   * <p>
30   * Test if the selector choose a valid member
31   *
32   * @author Pedro Ruivo
33   * @since 4.3
34   */
35  @TestForIssue(jiraKey = "HSEARCH-1261")
36  public class AutoNodeSelectorTest {
37  
38  	private static final String NEGATIVE_HASH_CODE_INDEX_NAME = "test.Book";
39  	private static final String POSITIVE_HASH_CODE_INDEX_NAME = "test.Author";
40  	private static final String ZERO_HASH_CODE_INDEX_NAME = "";
41  
42  	private final AtomicInteger NEXT_ADDRESS_ID = new AtomicInteger( 0 );
43  	private final AtomicLong NEXT_VIEW_ID = new AtomicLong( 0 );
44  
45  	@Test
46  	public void testIndexNameUsed() {
47  		Assert.assertTrue( NEGATIVE_HASH_CODE_INDEX_NAME.hashCode() < 0 );
48  		Assert.assertTrue( ZERO_HASH_CODE_INDEX_NAME.hashCode() == 0 );
49  		Assert.assertTrue( POSITIVE_HASH_CODE_INDEX_NAME.hashCode() > 0 );
50  	}
51  
52  	@Test
53  	public void testNegativeHashCodeIndex() {
54  		performTest( NEGATIVE_HASH_CODE_INDEX_NAME );
55  	}
56  
57  	@Test
58  	public void testPositiveHashCodeIndex() {
59  		performTest( POSITIVE_HASH_CODE_INDEX_NAME );
60  	}
61  
62  	@Test
63  	public void testZeroHashCodeIndex() {
64  		performTest( ZERO_HASH_CODE_INDEX_NAME );
65  	}
66  
67  	private void performTest(String index) {
68  		AutoNodeSelector selector = new AutoNodeSelector( index );
69  
70  		/*
71  		 * there is 4 main cases:
72  		 * 1) when the view has 1 member;
73  		 * 2) when the view has 2 members;
74  		 * 3) when the view has 3 member
75  		 * 4) when the view has > 3 members;
76  		 *
77  		 * some view size can trigger the bug and other don't. test with a wide range
78  		 */
79  
80  		for ( int viewSize = 1; viewSize <= 50; ++viewSize ) {
81  			View view = createView( viewSize );
82  			Assert.assertEquals( view.getMembers().size(), viewSize );
83  			selector.viewAccepted( view );
84  		}
85  	}
86  
87  	private View createView(int size) {
88  		List<Address> addressList = new LinkedList<Address>();
89  		while ( size-- > 0 ) {
90  			addressList.add( new TestAddress( NEXT_ADDRESS_ID.incrementAndGet() ) );
91  		}
92  		return new View( addressList.get( 0 ), NEXT_VIEW_ID.incrementAndGet(), addressList );
93  	}
94  
95  	private static final class TestAddress implements Address {
96  
97  		private int addressId;
98  
99  		public TestAddress(int addressId) {
100 			this.addressId = addressId;
101 		}
102 
103 		@Override
104 		public int size() {
105 			return Global.INT_SIZE;
106 		}
107 
108 		@Override
109 		public int compareTo(Address o) {
110 			if ( o == null || !( o instanceof TestAddress ) ) {
111 				return -1;
112 			}
113 			return Integer.valueOf( addressId ).compareTo( ( (TestAddress) o ).addressId );
114 		}
115 
116 		@Override
117 		public void writeExternal(ObjectOutput out) throws IOException {
118 			out.writeInt( addressId );
119 		}
120 
121 		@Override
122 		public void readExternal(ObjectInput in) throws IOException, ClassNotFoundException {
123 			addressId = in.readInt();
124 		}
125 
126 		@Override
127 		public void writeTo(DataOutput dataOutput) throws Exception {
128 			dataOutput.writeInt( addressId );
129 		}
130 
131 		@Override
132 		public void readFrom(DataInput dataInput) throws Exception {
133 			addressId = dataInput.readInt();
134 		}
135 
136 		@Override
137 		public String toString() {
138 			return "TestAddress{" + "addressId=" + addressId + '}';
139 		}
140 	}
