103 	public static void addFacetingRequest(JsonBuilder.Object jsonQuery, FacetingRequest facetingRequest,
104 			String sourceFieldAbsoluteName, String facetRelativeName) {
105 		String aggregationFieldName = sourceFieldAbsoluteName + "." + facetRelativeName + FACET_FIELD_SUFFIX;
106 
107 		if ( facetingRequest instanceof DiscreteFacetRequest ) {
108 			JsonObject termsJsonQuery = JsonBuilder.object().add( "terms",
109 					JsonBuilder.object()
110 							.addProperty( "field", aggregationFieldName )
111 							.addProperty( "size", facetingRequest.getMaxNumberOfFacets() == -1 ? Integer.MAX_VALUE : facetingRequest.getMaxNumberOfFacets() )
112 							.add( "order", fromFacetSortOrder( facetingRequest.getSort() ) )
113 							.addProperty( "min_doc_count", facetingRequest.hasZeroCountsIncluded() ? 0 : 1 )
114 					).build();
115 
116 			if ( isNested( sourceFieldAbsoluteName ) ) {
117 				JsonBuilder.Object facetJsonQuery = JsonBuilder.object();
118 				facetJsonQuery.add( "nested", JsonBuilder.object()
119 								.addProperty( "path", FieldHelper.getEmbeddedFieldPath( sourceFieldAbsoluteName ) + "." + facetRelativeName + FACET_FIELD_SUFFIX ) );
120 				facetJsonQuery.add( "aggregations", JsonBuilder.object().add( aggregationFieldName, termsJsonQuery ) );
121 				jsonQuery.add( facetingRequest.getFacetingName(), facetJsonQuery );
122 			}
123 			else {
124 				jsonQuery.add( facetingRequest.getFacetingName(), termsJsonQuery );
125 			}
126 		}
127 		else if ( facetingRequest instanceof RangeFacetRequest<?> ) {
128 			RangeFacetRequest<?> rangeFacetingRequest = (RangeFacetRequest<?>) facetingRequest;
129 			for ( FacetRange<?> facetRange : rangeFacetingRequest.getFacetRangeList() ) {
130 				JsonBuilder.Object comparisonFragment = JsonBuilder.object();
131 				if ( facetRange.getMin() != null ) {
132 					comparisonFragment.addProperty( facetRange.isMinIncluded() ? "gte" : "gt", facetRange.getMin() );
133 				}
134 				if ( facetRange.getMax() != null ) {
135 					comparisonFragment.addProperty( facetRange.isMaxIncluded() ? "lte" : "lt", facetRange.getMax() );
136 				}
137 
138 				JsonObject rangeQuery = wrapQueryForNestedIfRequired( aggregationFieldName,
139 						JsonBuilder.object().add( "range",
140 								JsonBuilder.object().add( aggregationFieldName, comparisonFragment ) ).build() );
141 
142 				jsonQuery.add( facetingRequest.getFacetingName() + "-" + facetRange.getIdentifier(),
143 						JsonBuilder.object().add( "filter", rangeQuery ) );
144 			}
145 		}
146 		else {
147 			throw LOG.facetingRequestHasUnsupportedType( facetingRequest.getClass().getName() );
148 		}
149 	}

151 	private static JsonObject fromFacetSortOrder(FacetSortOrder sortOrder) {
152 		JsonObject sort = new JsonObject();
153 		switch ( sortOrder ) {
154 			case COUNT_ASC:
155 				sort.addProperty( "_count", "asc" );
156 				break;
157 			case COUNT_DESC:
158 				sort.addProperty( "_count", "desc" );
159 				break;
160 			case FIELD_VALUE:
161 				sort.addProperty( "_term", "asc" );
162 				break;
163 			case RANGE_DEFINITION_ORDER:
164 				throw LOG.cannotSendRangeDefinitionOrderToElasticsearchBackend();
165 		}
166 		return sort;
167 	}

170 		JsonObject jsonCondition;
171 		if ( conditions.size() == 1 ) {
172 			jsonCondition = conditions.get( 0 ).getAsJsonObject();
173 		}
174 		else {
175 			jsonCondition = JsonBuilder.object().add( "bool",
176 					JsonBuilder.object().add( operator, conditions ) ).build();
177 		}
178 		return jsonCondition;

181 	public static JsonObject fromLuceneQuery(Query query) {
182 		if ( query instanceof MatchAllDocsQuery ) {
183 			return convertMatchAllDocsQuery( (MatchAllDocsQuery) query );
184 		}
185 		else if ( query instanceof TermQuery ) {
186 			return convertTermQuery( (TermQuery) query );
187 		}
188 		else if ( query instanceof BooleanQuery ) {
189 			return convertBooleanQuery( (BooleanQuery) query );
190 		}
191 		else if ( query instanceof TermRangeQuery ) {
192 			return convertTermRangeQuery( (TermRangeQuery) query );
193 		}
194 		else if ( query instanceof NumericRangeQuery ) {
195 			return convertNumericRangeQuery( (NumericRangeQuery<?>) query );
196 		}
197 		else if ( query instanceof WildcardQuery ) {
198 			return convertWildcardQuery( (WildcardQuery) query );
199 		}
200 		else if ( query instanceof PrefixQuery ) {
201 			return convertPrefixQuery( (PrefixQuery) query );
202 		}
203 		else if ( query instanceof FuzzyQuery ) {
204 			return convertFuzzyQuery( (FuzzyQuery) query );
205 		}
206 		else if ( query instanceof RemotePhraseQuery ) {
207 			return convertRemotePhraseQuery( (RemotePhraseQuery) query );
208 		}
209 		else if ( query instanceof RemoteMatchQuery ) {
210 			return convertRemoteMatchQuery( (RemoteMatchQuery) query );
211 		}
212 		else if ( query instanceof ConstantScoreQuery ) {
213 			return convertConstantScoreQuery( (ConstantScoreQuery) query );
214 		}
215 		else if ( query instanceof FilteredQuery ) {
216 			return convertFilteredQuery( (FilteredQuery) query );
217 		}
218 		else if ( query instanceof Filter ) {
219 			return fromLuceneFilter( (Filter) query );
220 		}
221 		else if ( query instanceof PhraseQuery ) {
222 			return convertPhraseQuery( (PhraseQuery) query );
223 		}
224 
225 		throw LOG.cannotTransformLuceneQueryIntoEsQuery( query );
226 	}

236 	private static JsonObject convertBooleanQuery(BooleanQuery booleanQuery) {
237 		JsonArray musts = new JsonArray();
238 		JsonArray shoulds = new JsonArray();
239 		JsonArray mustNots = new JsonArray();
240 		JsonArray filters = new JsonArray();
241 
242 		for ( BooleanClause clause : booleanQuery.clauses() ) {
243 			switch ( clause.getOccur() ) {
244 				case MUST:
245 					musts.add( fromLuceneQuery( clause.getQuery() ) );
246 					break;
247 				case FILTER:
248 					filters.add( fromLuceneQuery( clause.getQuery() ) );
249 					break;
250 				case MUST_NOT:
251 					mustNots.add( fromLuceneQuery( clause.getQuery() ) );
252 					break;
253 				case SHOULD:
254 					shoulds.add( fromLuceneQuery( clause.getQuery() ) );
255 					break;
256 			}
257 		}
258 
259 		JsonObject clauses = new JsonObject();
260 
261 		if ( musts.size() > 1 ) {
262 			clauses.add( "must", musts );
263 		}
264 		else if ( musts.size() == 1 ) {
265 			clauses.add( "must", musts.iterator().next() );
266 		}
267 
268 		if ( shoulds.size() > 1 ) {
269 			clauses.add( "should", shoulds );
270 		}
271 		else if ( shoulds.size() == 1 ) {
272 			clauses.add( "should", shoulds.iterator().next() );
273 		}
274 
275 		if ( mustNots.size() > 1 ) {
276 			clauses.add( "must_not", mustNots );
277 		}
278 		else if ( mustNots.size() == 1 ) {
279 			clauses.add( "must_not", mustNots.iterator().next() );
280 		}
281 
282 		if ( filters.size() > 1 ) {
283 			clauses.add( "filter", filters );
284 		}
285 		else if ( filters.size() == 1 ) {
286 			clauses.add( "filter", filters.iterator().next() );
287 		}
288 
289 		JsonObject bool = new JsonObject();
290 		bool.add( "bool", clauses );
291 		return bool;
292 	}

356 	/**
357 	 * This is best effort only: the PhraseQuery may contain multiple terms at the same position
358 	 * (think synonyms) or gaps (think stopwords) and it's in this case impossible to translate
359 	 * it into a correct ElasticsearchQuery.
360 	 */
361 	private static JsonObject convertPhraseQuery(PhraseQuery query) {
362 		Term[] terms = query.getTerms();
363 
364 		if ( terms.length == 0 ) {
365 			throw LOG.cannotQueryOnEmptyPhraseQuery();
366 		}
367 
368 		String field = terms[0].field(); // phrase queries are only supporting one field
369 		StringBuilder phrase = new StringBuilder();
370 		for ( Term term : terms ) {
371 			phrase.append( " " ).append( term.text() );
372 		}
373 
374 		JsonObject phraseQuery = JsonBuilder.object()
375 				.add( "match_phrase",
376 						JsonBuilder.object().add( field,
377 								JsonBuilder.object()
378 										.addProperty( "query", phrase.toString().trim() )
379 										.append( slopAppender( query.getSlop() ) )
380 										.append( boostAppender( query ) )
381 								)
382 				).build();
383 
384 		return wrapQueryForNestedIfRequired( field, phraseQuery );
385 	}

421 	private static JsonObject convertTermRangeQuery(TermRangeQuery query) {
422 		JsonBuilder.Object interval = JsonBuilder.object();
423 
424 		if ( query.getLowerTerm() != null ) {
425 			interval.addProperty( query.includesLower() ? "gte" : "gt", query.getLowerTerm().utf8ToString() );
426 		}
427 		if ( query.getUpperTerm() != null ) {
428 			interval.addProperty( query.includesUpper() ? "lte" : "lt", query.getUpperTerm().utf8ToString() );
429 		}
430 
431 		interval.append( boostAppender( query ) );
432 
433 		JsonObject range = JsonBuilder.object().add( "range",
434 						JsonBuilder.object().add( query.getField(), interval ) )
435 				.build();
436 
437 		return wrapQueryForNestedIfRequired( query.getField(), range );
438 	}

482 	private static JsonObject convertDistanceFilter(DistanceFilter filter) {
483 		JsonObject distanceQuery = JsonBuilder.object()
484 				.add( "geo_distance",
485 						JsonBuilder.object()
486 								.addProperty( "distance", filter.getRadius() + "km" )
487 								.add( filter.getCoordinatesField(),
488 										JsonBuilder.object()
489 												.addProperty( "lat", filter.getCenter().getLatitude() )
490 												.addProperty( "lon", filter.getCenter().getLongitude() )
491 								)
492 				).build();
493 
494 		distanceQuery = wrapQueryForNestedIfRequired( filter.getCoordinatesField(), distanceQuery );
495 
496 		// we only implement the previous filter optimization when we use the hash method as Elasticsearch
497 		// automatically optimize the geo_distance query with a bounding box filter so we don't need to do it
498 		// ourselves when we use the range method.
499 		Filter previousFilter = filter.getPreviousFilter();
500 		if ( previousFilter instanceof SpatialHashFilter ) {
501 			distanceQuery = JsonBuilder.object()
502 					.add( "bool", JsonBuilder.object()
503 							.add( "must", distanceQuery )
504 							.add( "filter", convertSpatialHashFilter( (SpatialHashFilter) previousFilter ) )
505 					).build();
506 		}
507 
508 		return distanceQuery;
509 	}

511 	private static JsonObject convertSpatialHashFilter(SpatialHashFilter filter) {
512 		JsonArray cellsIdsJsonArray = new JsonArray();
513 		for ( String cellId : filter.getSpatialHashCellsIds() ) {
514 			cellsIdsJsonArray.add( cellId );
515 		}
516 
517 		JsonObject spatialHashFilter = JsonBuilder.object()
518 				.add( "terms", JsonBuilder.object()
519 						.add( filter.getFieldName(), cellsIdsJsonArray )
520 				).build();
521 
522 		return wrapQueryForNestedIfRequired( filter.getFieldName(), spatialHashFilter );
523 	}

546 			return NOOP_APPENDER;

567 	/**
568 	 * Appender that adds a "boost" property if necessary.
569 	 */
570 	private static JsonBuilder.JsonAppender<? super JsonBuilder.Object> boostAppender(Query query) {
571 		final float boost = query.getBoost();
572 		if ( boost != DEFAULT_BOOST ) { // We actually want to use float equality here
573 			return new JsonBuilder.JsonAppender<JsonBuilder.Object>() {
574 				@Override
575 				public void append(JsonBuilder.Object object) {
576 					object.addProperty( "boost", boost );

604 	public static JsonObject fromLuceneFilter(Filter luceneFilter) {
605 		if ( luceneFilter instanceof QueryWrapperFilter ) {
606 			Query query = ( (QueryWrapperFilter) luceneFilter ).getQuery();
607 			query.setBoost( luceneFilter.getBoost() * query.getBoost() );
608 			return fromLuceneQuery( query );
609 		}
610 		else if ( luceneFilter instanceof DistanceFilter ) {
611 			return convertDistanceFilter( (DistanceFilter) luceneFilter );
612 		}
613 		else if ( luceneFilter instanceof SpatialHashFilter ) {
614 			return convertSpatialHashFilter( (SpatialHashFilter) luceneFilter );
615 		}
616 		else if ( luceneFilter instanceof CachingWrapperFilter ) {
617 			return fromLuceneFilter( ( (CachingWrapperFilter) luceneFilter ).getCachedFilter() );
618 		}
619 		else if ( luceneFilter instanceof org.apache.lucene.search.CachingWrapperFilter ) {
620 			return fromLuceneFilter( ( (org.apache.lucene.search.CachingWrapperFilter) luceneFilter ).getFilter() );
621 		}
622 		throw LOG.cannotTransformLuceneFilterIntoEsQuery( luceneFilter );
623 	}

625 	/**
626 	 * Convert a Lucene {@link SortField} to an Elasticsearch {@link Sort}, trying to preserve
627 	 * the exact same meaning as the SortField would have in Lucene.
628 	 *
629 	 * <p>For instance, missing values on numeric fields are implicitly 0 in Lucene, so this
630 	 * method will add it explicitly on any numeric Elasticsearch sort.
631 	 *
632 	 * @param sortField The Lucene {@link SortField} to convert
633 	 * @return The equivalent Elasticsearch {@link Sort}
634 	 */
635 	public static Sort fromLuceneSortField(SortField sortField) {
636 		if ( sortField instanceof DistanceSortField ) {
637 			DistanceSortField distanceSortField = (DistanceSortField) sortField;
638 			return new DistanceSort( distanceSortField.getField(),
639 					distanceSortField.getCenter(),
640 					distanceSortField.getReverse() ? Sorting.DESC : Sorting.ASC );
641 		}
642 		else if ( sortField instanceof NativeSortField ) {
643 			NativeSortField nativeSortField = (NativeSortField) sortField;
644 			String sortFieldName = nativeSortField.getField();
645 			String sortDescriptionAsString = nativeSortField.getNativeSortDescription();
646 			JsonElement sortDescription = JSON_PARSER.parse( sortDescriptionAsString );
647 			return new ArbitrarySort( sortFieldName, sortDescription );
648 		}
649 		else {
650 			SortField.Type sortFieldType = sortField.getType();
651 
652 			String sortFieldName;
653 			if ( sortField.getField() == null ) {
654 				switch ( sortFieldType ) {
655 					case DOC:
656 						sortFieldName = "_uid";
657 						break;
658 					case SCORE:
659 						sortFieldName = "_score";
660 						break;
661 					default:
662 						throw LOG.cannotUseThisSortTypeWithNullSortFieldName( sortField.getType() );
663 				}
664 			}
665 			else {
666 				sortFieldName = sortField.getField();
667 			}
668 
669 			boolean reverse = sortField.getReverse();
670 			Sort sort = new Sort( sortFieldName, fromLuceneSortFieldOrder( sortFieldType, reverse ) );
671 			sort.setMissing( fromLuceneSortFieldMissing( sortFieldType, sortField.missingValue, reverse ) );
672 
673 			return sort;
674 		}
675 	}

677 	private static Sorting fromLuceneSortFieldOrder(Type sortFieldType, boolean reverse) {
678 		switch ( sortFieldType ) {
679 			case SCORE:
680 				return reverse ? Sorting.ASC : Sorting.DESC;
681 			default:
682 				return reverse ? Sorting.DESC : Sorting.ASC;
683 		}
684 	}

686 	private static Object fromLuceneSortFieldMissing(Type sortFieldType, Object luceneMissing, boolean reverse) {
687 		if ( luceneMissing == null ) {
688 			/*
689 			 * Simulate Lucene's behavior of assigning default missing values when none is explicitly provided.
690 			 */
691 			switch ( sortFieldType ) {
692 				case DOUBLE:
693 				case FLOAT:
694 				case INT:
695 				case LONG:
696 					luceneMissing = SORT_FIELD_SCALAR_DEFAULTS.get( sortFieldType );
697 					break;
698 				case STRING:
699 				case STRING_VAL:
700 					luceneMissing = reverse ? SortField.STRING_LAST : SortField.STRING_FIRST;
701 					break;
702 				default:
703 					break;
704 			}
705 		}
706 
707 		switch ( sortFieldType ) {
708 			case DOUBLE:
709 			case FLOAT:
710 			case INT:
711 			case LONG:
712 				// Use a more natural representation of the missing value, if possible
713 				if ( luceneMissing.equals( SORT_FIELD_SCALAR_MINIMUMS.get( sortFieldType ) ) ) {
714 					return reverse ? Missing.LAST : Missing.FIRST;
715 				}
716 				else if ( luceneMissing.equals( SORT_FIELD_SCALAR_MAXIMUMS.get( sortFieldType ) ) ) {
717 					return reverse ? Missing.FIRST : Missing.LAST;
718 				}
719 				else {
720 					// Make sure the correct type is used (and throw a ClassCastException if not, as Lucene does)
721 					return SORT_FIELD_SCALAR_TYPES.get( sortFieldType ).cast( luceneMissing );
722 				}
723 			case STRING:
724 			case STRING_VAL:
725 				if ( SortField.STRING_LAST.equals( luceneMissing ) ) {
726 					return Missing.LAST;
727 				}
728 				else if ( SortField.STRING_FIRST.equals( luceneMissing ) ) {
729 					return Missing.FIRST;
730 				}
731 				else if ( luceneMissing != null ) {
732 					throw new AssertionFailure( "Unexpected missing value specified on a String SortField: " + luceneMissing );
733 				}
734 				else {
735 					return null;
736 				}
737 			default:
738 				if ( luceneMissing != null ) {
739 					throw new AssertionFailure( "Missing value specified on a SortField which is not supposed to support it: " + sortFieldType );
740 				}
741 				else {
742 					return null;
743 				}
744 		}
