31  /**
32   * A helper classes dealing with the processing of annotation. It is there to share some annotation processing
33   * between the document builder and other metadata classes, eg {@code FieldMetadata}. In the long run
34   * this class might become obsolete.
35   *
36   * @author Hardy Ferentschik
37   */
38  public final class AnnotationProcessingHelper {
39  
40  	private AnnotationProcessingHelper() {
41  		//not allowed
42  	}
43  
44  	/**
45  	 * Using the passed field (or class bridge) settings determines the Lucene {@link org.apache.lucene.document.Field.Index}
46  	 *
47  	 * @param index is the field indexed or not
48  	 * @param analyze should the field be analyzed
49  	 * @param norms are norms to be added to index
50  	 * @return Returns the Lucene {@link org.apache.lucene.document.Field.Index} value for a given field
51  	 */
52  	public static Field.Index getIndex(Index index, Analyze analyze, Norms norms) {
53  		if ( Index.YES.equals( index ) ) {
54  			if ( Analyze.YES.equals( analyze ) ) {
55  				if ( Norms.YES.equals( norms ) ) {
56  					return Field.Index.ANALYZED;
57  				}
58  				else {
59  					return Field.Index.ANALYZED_NO_NORMS;
60  				}
61  			}
62  			else {
63  				if ( Norms.YES.equals( norms ) ) {
64  					return Field.Index.NOT_ANALYZED;
65  				}
66  				else {
67  					return Field.Index.NOT_ANALYZED_NO_NORMS;
68  				}
69  			}
70  		}
71  		else {
72  			return Field.Index.NO;
73  		}
74  	}
75  
76  	public static Float getBoost(XProperty member, Annotation fieldAnn) {
77  		float computedBoost = 1.0f;
78  		Boost boostAnn = member.getAnnotation( Boost.class );
79  		if ( boostAnn != null ) {
80  			computedBoost = boostAnn.value();
81  		}
82  		if ( fieldAnn != null ) {
83  			float boost;
84  			if ( fieldAnn instanceof org.hibernate.search.annotations.Field ) {
85  				boost = ( (org.hibernate.search.annotations.Field) fieldAnn ).boost().value();
86  			}
87  			else if ( fieldAnn instanceof Spatial ) {
88  				boost = ( (Spatial) fieldAnn ).boost().value();
89  			}
90  			else {
91  				raiseAssertionOnIncorrectAnnotation( fieldAnn );
92  				boost = 0; //never reached
93  			}
94  			computedBoost *= boost;
95  		}
96  		return computedBoost;
97  	}

99  	public static BoostStrategy getDynamicBoost(final XAnnotatedElement element) {
100 		if ( element == null ) {
101 			return DefaultBoostStrategy.INSTANCE;
102 		}
103 		DynamicBoost boostAnnotation = element.getAnnotation( DynamicBoost.class );
104 		if ( boostAnnotation == null ) {
105 			return DefaultBoostStrategy.INSTANCE;
106 		}
107 		Class<? extends BoostStrategy> boostStrategyClass = boostAnnotation.impl();
108 		return ClassLoaderHelper.instanceFromClass( BoostStrategy.class, boostStrategyClass, "boost strategy" );
109 	}

111 	public static Field.TermVector getTermVector(TermVector vector) {
112 		switch ( vector ) {
113 			case NO:
114 				return Field.TermVector.NO;
115 			case YES:
116 				return Field.TermVector.YES;
117 			case WITH_OFFSETS:
118 				return Field.TermVector.WITH_OFFSETS;
119 			case WITH_POSITIONS:
120 				return Field.TermVector.WITH_POSITIONS;
121 			case WITH_POSITION_OFFSETS:
122 				return Field.TermVector.WITH_POSITIONS_OFFSETS;
123 			default:
124 				throw new AssertionFailure( "Unexpected TermVector: " + vector );
125 		}
126 	}

128 	public static AnalyzerReference getAnalyzerReference(org.hibernate.search.annotations.Analyzer analyzerAnn, ConfigContext configContext,
129 			IndexManagerType indexManagerType) {
130 		MutableAnalyzerRegistry registry = indexManagerType == null ? null : configContext.getAnalyzerRegistry( indexManagerType );
131 		Class<?> analyzerClass = analyzerAnn == null ? void.class : analyzerAnn.impl();
132 		if ( analyzerClass != void.class ) {
133 			return registry.getOrCreateAnalyzerReference( analyzerClass );
134 		}
135 		else {
136 			String definition = analyzerAnn == null ? "" : analyzerAnn.definition();
137 			if ( StringHelper.isEmpty( definition ) ) {
138 				return null;
139 			}
140 			return registry.getOrCreateAnalyzerReference( definition );
141 		}
142 	}
143 
144 	public static Integer getPrecisionStep(NumericField numericFieldAnn) {
145 		return numericFieldAnn == null ? NumericField.PRECISION_STEP_DEFAULT : numericFieldAnn.precisionStep();
146 	}
147 
148 	public static String getFieldName(Annotation fieldAnn) {
149 		final String fieldName;
150 		if ( fieldAnn instanceof org.hibernate.search.annotations.Field ) {
151 			fieldName = ( (org.hibernate.search.annotations.Field) fieldAnn ).name();
152 		}
153 		else if ( fieldAnn instanceof Spatial ) {
154 			fieldName = ( (Spatial) fieldAnn ).name();
155 		}
156 		else if ( fieldAnn instanceof SortableField ) {
157 			fieldName = ( (SortableField) fieldAnn ).forField();
158 		}
159 		else if ( fieldAnn instanceof NumericField ) {
160 			fieldName = ( (NumericField) fieldAnn ).forField();
161 		}
162 		else {
163 			return raiseAssertionOnIncorrectAnnotation( fieldAnn );
164 		}
165 		return fieldName;
166 	}
167 
168 	private static String raiseAssertionOnIncorrectAnnotation(Annotation fieldAnn) {
169 		throw new AssertionFailure( "Cannot process instances other than @Field, @Spatial and @SortableField. Found: " + fieldAnn.getClass() );
170 	}
171 }
