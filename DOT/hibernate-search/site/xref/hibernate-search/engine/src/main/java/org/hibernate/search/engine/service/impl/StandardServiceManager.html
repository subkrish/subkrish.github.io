33  /**
34   * Default implementation of the {@code ServiceManager} interface.
35   *
36   * @author Emmanuel Bernard
37   * @author Sanne Grinovero
38   * @author Hardy Ferentschik
39   */
40  public class StandardServiceManager implements ServiceManager {
41  	private static final Log log = LoggerFactory.make();
42  
43  	private final Properties properties;
44  	private final BuildContext buildContext;
45  	private final ConcurrentHashMap<Class<?>, ServiceWrapper<?>> cachedServices = new ConcurrentHashMap<Class<?>, ServiceWrapper<?>>();
46  	private final Map<Class<? extends Service>, Object> providedServices;
47  	private final Map<Class<? extends Service>, String> defaultServices;
48  	private final ClassLoaderService classloaderService;
49  	private final boolean failOnUnreleasedService;
50  
51  	private volatile boolean allServicesReleased = false;
52  
53  	public StandardServiceManager(SearchConfiguration searchConfiguration, BuildContext buildContext) {
54  		this( searchConfiguration, buildContext, Collections.<Class<? extends Service>, String>emptyMap() );
55  	}
56  
57  	public StandardServiceManager(SearchConfiguration searchConfiguration,
58  			BuildContext buildContext,
59  			Map<Class<? extends Service>, String> defaultServices) {
60  		this.buildContext = buildContext;
61  		this.properties = searchConfiguration.getProperties();
62  		this.providedServices = createProvidedServices( searchConfiguration );
63  		this.defaultServices = defaultServices;
64  		this.classloaderService = searchConfiguration.getClassLoaderService();
65  		this.failOnUnreleasedService = Boolean.getBoolean( "org.hibernate.search.fail_on_unreleased_service" );
66  	}
67  
68  	@Override
69  	@SuppressWarnings("unchecked")
70  	public <S extends Service> S requestService(Class<S> serviceRole) {
71  		if ( serviceRole == null ) {
72  			throw new IllegalArgumentException( "'null' is not a valid service role" );
73  		}
74  
75  		if ( allServicesReleased ) {
76  			throw log.serviceRequestedAfterReleasedAllWasCalled();
77  		}
78  
79  		// provided services have priority over managed services
80  		final Object providedService = providedServices.get( serviceRole );
81  		if ( providedService != null ) {
82  			return (S) providedService;
83  		}
84  
85  		ServiceWrapper<S> wrapper = (ServiceWrapper<S>) cachedServices.get( serviceRole );
86  		if ( wrapper == null ) {
87  			wrapper = createAndCacheWrapper( serviceRole );
88  		}
89  		wrapper.startVirtual();
90  		return wrapper.getService();
91  	}
92  
93  	@Override
94  	public <S extends Service> ServiceReference<S> requestReference(Class<S> serviceRole) {
95  		return new ServiceReference<>( this, serviceRole );
96  	}
97  
98  	@Override
99  	public <S extends Service> void releaseService(Class<S> serviceRole) {
100 		if ( serviceRole == null ) {
101 			throw new IllegalArgumentException( "'null' is not a valid service role" );
102 		}
103 
104 		if ( providedServices.containsKey( serviceRole ) ) {
105 			return;
106 		}
107 
108 		ServiceWrapper wrapper = cachedServices.get( serviceRole );
109 		if ( wrapper != null ) {
110 			wrapper.stopVirtual();
111 		}
112 	}
113 
114 	@Override
115 	public synchronized void releaseAllServices() {
116 		for ( ServiceWrapper<?> wrapper : cachedServices.values() ) {
117 			/*
118 			 * Perform an additional stopVirtual, to remove the extra usage token granted at first initialization,
119 			 * which keeps the service to be really stopped when it's released by the service client, yet we're not shutting down
120 			 * the Search engine yet.
121 			 */
122 			wrapper.stopVirtual();
123 		}
124 
125 		/*
126 		 * If everything went well, the previous pass should have brought every
127 		 * user count to 0 for every service, so every service should have been stopped.
128 		 * This should also have chained-released services used by services, which
129 		 * ultimately should have stopped everything.
130 		 */
131 
132 		/*
133 		 * Second pass to check for still-running services and forcefully stop them.
134 		 */
135 		List<String> unreleasedServicesToReport = failOnUnreleasedService ? new ArrayList<String>() : null;
136 		for ( ServiceWrapper<?> wrapper : cachedServices.values() ) {
137 			synchronized ( wrapper ) {
138 				if ( wrapper.status != ServiceStatus.STOPPED ) {
139 					log.serviceProviderNotReleased( wrapper.serviceClass );
140 					wrapper.stopReal();
141 					if ( unreleasedServicesToReport != null ) {
142 						unreleasedServicesToReport.add( wrapper.serviceClass.getName() );
143 					}
144 				}
145 			}
146 		}
147 
148 		cachedServices.clear();
149 		allServicesReleased = true;
150 
151 		if ( failOnUnreleasedService && !unreleasedServicesToReport.isEmpty() ) {
152 			throw new AssertionFailure( "The following services have been used but not released: "
153 					+ unreleasedServicesToReport );
154 		}
155 	}

157 	private Map<Class<? extends Service>, Object> createProvidedServices(SearchConfiguration searchConfiguration) {
158 		Map<Class<? extends Service>, Object> tmpServices = new HashMap<Class<? extends Service>, Object>(
159 				searchConfiguration.getProvidedServices()
160 				);
161 
162 		if ( tmpServices.containsKey( ClassLoaderService.class ) ) {
163 			throw log.classLoaderServiceContainedInProvidedServicesException();
164 		}
165 		else {
166 			tmpServices.put( ClassLoaderService.class, searchConfiguration.getClassLoaderService() );
167 		}
168 
169 		return Collections.unmodifiableMap( tmpServices );
170 	}
171 
172 	/**
173 	 * The 'synchronized' is necessary to avoid loading the same service in parallel: enumerating service
174 	 * implementations is not threadsafe when delegating to Hibernate ORM's org.hibernate.boot.registry.classloading.spi.ClassLoaderService
175 	 */
176 	private synchronized <S extends Service> ServiceWrapper<S> createAndCacheWrapper(Class<S> serviceRole) {
177 		//Check again, for concurrent usage:
178 		ServiceWrapper<S> existingWrapper = (ServiceWrapper<S>) cachedServices.get( serviceRole );
179 		if ( existingWrapper != null ) {
180 			return existingWrapper;
181 		}
182 		Set<S> services = new HashSet<>();
183 		for ( S service : requestService( ClassLoaderService.class ).loadJavaServices( serviceRole ) ) {
184 			services.add( service );
185 		}
186 
187 		if ( services.size() == 0 ) {
188 			tryLoadingDefaultService( serviceRole, services );
189 		}
190 		else if ( services.size() > 1 ) {
191 			throw log.getMultipleServiceImplementationsException(
192 					serviceRole.toString(),
193 					StringHelper.join( services, "," )
194 			);
195 		}
196 		S service = services.iterator().next();
197 		ServiceWrapper<S> wrapper = new ServiceWrapper<S>( service, serviceRole, buildContext );
198 		@SuppressWarnings("unchecked")
199 		ServiceWrapper<S> previousWrapper = (ServiceWrapper<S>) cachedServices.putIfAbsent( serviceRole, wrapper );
200 		if ( previousWrapper != null ) {
201 			wrapper = previousWrapper;
202 		}
203 		else {
204 			//Initialize the service usage counter with an additional usage token, on top of the one granted by the service request:
205 			wrapper.startVirtual();
206 		}
207 		return wrapper;
208 	}
209 
210 	private <S extends Service> void tryLoadingDefaultService(Class<S> serviceRole, Set<S> services) {
211 		// there is no loadable service. Check whether we have a default one we can instantiate
212 		if ( defaultServices.containsKey( serviceRole ) ) {
213 			S service = ClassLoaderHelper.instanceFromName(
214 					serviceRole,
215 					defaultServices.get( serviceRole ),
216 					"default service",
217 					this
218 			);
219 			services.add( service );
220 		}
221 		else {
222 			throw log.getNoServiceImplementationFoundException( serviceRole.toString() );
223 		}
224 	}

226 	private class ServiceWrapper<S> {
227 		private final S service;
228 		private final BuildContext context;
229 		private final Class<S> serviceClass;
230 
231 		private int userCounter = 0;
232 		private ServiceStatus status = ServiceStatus.STOPPED;
233 
234 		ServiceWrapper(S service, Class<S> serviceClass, BuildContext context) {
235 			this.service = service;
236 			this.context = context;
237 			this.serviceClass = serviceClass;
238 		}
239 
240 		synchronized S getService() {
241 			if ( status != ServiceStatus.RUNNING ) {
242 				stateExpectedFailure();
243 			}
244 			return service;
245 		}
246 
247 		/**
248 		 * Virtual call to the start method: only actually starts the
249 		 * service when bumping up the counter of start requests from
250 		 * zero. Subsequent start requests will simply increment the
251 		 * counter, so that we can eventually tear down the services
252 		 * in reverse order to make dependency graphs happy.
253 		 *
254 		 * Make sure to invoke startVirtual() both on service request,
255 		 * and on creation of the wrapper so that the first request
256 		 * accounts for two usage tokens rather than one.
257 		 * The shutdown process will similarly invoke stopVirtual() an
258 		 * additional time; this is to prevent services from starting
259 		 * and stopping frequently at runtime.
260 		 */
261 		synchronized void startVirtual() {
262 			int previousValue = userCounter;
263 			userCounter++;
264 			if ( previousValue == 0 ) {
265 				if ( status != ServiceStatus.STOPPED ) {
266 					stateExpectedFailure();
267 				}
268 				startService( service );
269 			}
270 			if ( status != ServiceStatus.RUNNING ) {
271 				//Could happen on circular dependencies
272 				stateExpectedFailure();
273 			}
274 		}
275 
276 		synchronized void stopVirtual() {
277 			userCounter--;
278 			if ( userCounter == 0 ) {
279 				//Do not check for the expected status in this case: we don't want a previous service start failure to
280 				//prevent us to further attempt to stop services.
281 				stopReal();
282 			}
283 		}
284 
285 		synchronized void stopReal() {
286 			status = ServiceStatus.STOPPING;
287 			try {
288 				if ( service instanceof Stoppable ) {
289 					( (Stoppable) service ).stop();
290 				}
291 			}
292 			catch (Exception e) {
293 				log.stopServiceFailed( serviceClass, e );
294 			}
295 			finally {
296 				status = ServiceStatus.STOPPED;
297 			}
298 		}
299 
300 		private void startService(final S service) {
301 			status = ServiceStatus.STARTING;
302 			if ( service instanceof Startable ) {
303 				( (Startable) service ).start( properties, context );
304 			}
305 			status = ServiceStatus.RUNNING;
306 		}
307 
308 		private void stateExpectedFailure() {
309 			throw log.getUnexpectedServiceStatusException( status.name(), service.toString() );
310 		}
311 	}
312 
313 	private enum ServiceStatus {
314 		RUNNING, STOPPED, STARTING, STOPPING
315 	}

318 	public ClassLoaderService getClassLoaderService() {
319 		return classloaderService;
320 	}
