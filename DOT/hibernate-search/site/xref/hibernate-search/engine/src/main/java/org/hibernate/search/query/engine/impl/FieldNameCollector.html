7   package org.hibernate.search.query.engine.impl;
8   
9   import java.util.HashSet;
10  import java.util.Set;
11  
12  import org.apache.lucene.search.BooleanClause;
13  import org.apache.lucene.search.BooleanQuery;
14  import org.apache.lucene.search.ConstantScoreQuery;
15  import org.apache.lucene.search.DisjunctionMaxQuery;
16  import org.apache.lucene.search.MultiPhraseQuery;
17  import org.apache.lucene.search.MultiTermQuery;
18  import org.apache.lucene.search.NumericRangeQuery;
19  import org.apache.lucene.search.PhraseQuery;
20  import org.apache.lucene.search.Query;
21  import org.apache.lucene.search.TermQuery;
22  
23  /**
24   * Helper class to extract field names from the Lucene queries taking query types into account.
25   *
26   * @author Hardy Ferentschik
27   */
28  public class FieldNameCollector {
29  
30  	private FieldNameCollector() {
31  	}
32  
33  	public static FieldCollection extractFieldNames(Query query) {
34  		// first we need to find all composing queries since some query types are just containers
35  		Set<Query> composingQueries = new HashSet<>();
36  		collectComposingQueries( query, composingQueries );
37  
38  		FieldCollection fieldCollection = new FieldCollection();
39  		for ( Query composingQuery : composingQueries ) {
40  			if ( composingQuery instanceof NumericRangeQuery ) {
41  				fieldCollection.addNumericFieldName( ( (NumericRangeQuery) composingQuery ).getField() );
42  			}
43  			else if ( composingQuery instanceof MultiTermQuery ) {
44  				fieldCollection.addStringFieldName( ( (MultiTermQuery) composingQuery ).getField() );
45  			}
46  			else if ( composingQuery instanceof TermQuery ) {
47  				TermQuery termQuery = (TermQuery) composingQuery;
48  				fieldCollection.addStringFieldName( termQuery.getTerm().field() );
49  
50  			}
51  			else if ( composingQuery instanceof PhraseQuery ) {
52  				PhraseQuery phraseQuery = (PhraseQuery) composingQuery;
53  				// all terms must be against the same field, it's enough to look at the first
54  				fieldCollection.addStringFieldName( phraseQuery.getTerms()[0].field() );
55  			}
56  			else if ( composingQuery instanceof MultiPhraseQuery ) {
57  				MultiPhraseQuery phraseQuery = (MultiPhraseQuery) composingQuery;
58  				// all terms must be against the same field, it's enough to look at the first
59  				fieldCollection.addStringFieldName( phraseQuery.getTermArrays().get( 0 )[0].field() );
60  			}
61  		}
62  
63  		return fieldCollection;
64  	}
65  
66  	private static void collectComposingQueries(Query query, Set<Query> composingQueries) {
67  		if ( query instanceof BooleanQuery ) {
68  			BooleanQuery booleanQuery = (BooleanQuery) query;
69  			for ( BooleanClause clause : booleanQuery.getClauses() ) {
70  				collectComposingQueries( clause.getQuery(), composingQueries );
71  			}
72  		}
73  		else if ( query instanceof DisjunctionMaxQuery ) {
74  			DisjunctionMaxQuery disjunctionMaxQuery = (DisjunctionMaxQuery) query;
75  			for ( Query subQuery : disjunctionMaxQuery.getDisjuncts() ) {
76  				collectComposingQueries( subQuery, composingQueries );
77  			}
78  		}
79  		else if ( query instanceof ConstantScoreQuery ) {
80  			// this one is tricky, the ConstantScoreQuery can wrap a query or a filter.
81  			ConstantScoreQuery constantScoreQuery = (ConstantScoreQuery) query;
82  			if ( constantScoreQuery.getQuery() != null ) {
83  				collectComposingQueries( constantScoreQuery.getQuery(), composingQueries );
84  			}
85  		}
86  		else {
87  			composingQueries.add( query );
88  		}
89  	}
90  
91  	public static class FieldCollection {
92  		private final Set<String> numericFieldNames;
93  		private final Set<String> stringFieldNames;
94  
95  		public FieldCollection() {
96  			numericFieldNames = new HashSet<>();
97  			stringFieldNames = new HashSet<>();
98  		}
99  
100 		void addNumericFieldName(String fieldName) {
101 			numericFieldNames.add( fieldName );
102 		}
103 
104 		void addStringFieldName(String fieldName) {
105 			stringFieldNames.add( fieldName );
106 		}
107 
108 		public Set<String> getNumericFieldNames() {
109 			return numericFieldNames;
110 		}
111 
112 		public Set<String> getStringFieldNames() {
113 			return stringFieldNames;
114 		}
115 
116 		@Override
117 		public String toString() {
118 			return "FieldCollection{" +
119 					"numericFieldNames=" + numericFieldNames +
120 					", stringFieldNames=" + stringFieldNames +
121 					'}';
122 		}
123 	}
124 }
