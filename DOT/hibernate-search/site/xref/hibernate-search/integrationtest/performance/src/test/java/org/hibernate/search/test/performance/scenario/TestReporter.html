57  	public static void printReport(TestContext ctx) throws UnsupportedEncodingException, IOException {
58  		PrintStream outStream = createOutputStream( ctx.scenario.getClass().getSimpleName() );
59  		PrintWriter outWriter = new PrintWriter(
60  				new BufferedWriter( new OutputStreamWriter( outStream, "UTF-8" ) ), false );
61  
62  		outWriter.println( "==================================================================" );
63  		outWriter.println( "HIBERNATE SEARCH PERFORMANCE TEST REPORT" );
64  		printSummary( ctx, outWriter );
65  		printTaskInfo( ctx, outWriter );
66  		printEnvInfo( ctx, outWriter );
67  		outWriter.println( "==================================================================" );
68  		outWriter.flush();
69  
70  		CheckerLuceneIndex.printIndexReport( ctx, outStream );
71  		CheckerUncaughtExceptions.printUncaughtExceptions( ctx, outWriter );
72  
73  		outWriter.close();
74  		outStream.close();
75  	}

77  	private static void printSummary(TestContext ctx, PrintWriter out) {
78  		long freeMemory2 = -1;
79  		long totalMemory2 = -1;
80  		if ( MEASURE_MEMORY ) {
81  			runGarbageCollectorAndWait();
82  			freeMemory2 = Runtime.getRuntime().freeMemory();
83  			totalMemory2 = Runtime.getRuntime().totalMemory();
84  		}
85  
86  		long totalNanos = ctx.stopTime - ctx.startTime;
87  		long totalMilis = TimeUnit.MILLISECONDS.convert( totalNanos, TimeUnit.NANOSECONDS );
88  
89  		out.println( "" );
90  		out.println( "SUMMARY" );
91  		out.println( "    Name   : " + ctx.scenario.getClass().getSimpleName() );
92  		out.println( "    Date   : " + DateFormatUtils.format( new Date(), "yyyy-MM-dd HH:mm" ) );
93  		out.println( "" );
94  		out.println( "    Measured time (HH:mm:ss.SSS)" );
95  		out.println( "        MEASURED TASKS : " + DurationFormatUtils.formatDuration( totalMilis, "HH:mm:ss.SSS" ) );
96  		out.println( "        init database  : " + DurationFormatUtils.formatDuration( ctx.scenario.initDatabaseStopWatch.getTime(), "HH:mm:ss.SSS" ) );
97  		out.println( "        init index     : " + DurationFormatUtils.formatDuration( ctx.scenario.initIndexStopWatch.getTime(), "HH:mm:ss.SSS" ) );
98  		out.println( "        warmup phase   : " + DurationFormatUtils.formatDuration( ctx.scenario.warmupStopWatch.getTime(), "HH:mm:ss.SSS" ) );
99  
100 		if ( MEASURE_MEMORY ) {
101 			out.println( "" );
102 			out.println( "    Memory usage (total-free):" );
103 			out.println( "        before : " + toMB( ctx.totalMemory - ctx.freeMemory ) );
104 			out.println( "        after  : " + toMB( totalMemory2 - freeMemory2 ) );
105 		}
106 	}

108 	private static void printTaskInfo(TestContext ctx, PrintWriter out) {
109 		out.println( "" );
110 		out.println( "TASKS" );
111 		for ( AbstractTask task : ctx.tasks ) {
112 			String taskTotalTime = "n/a";
113 			String taskAverageTime = "n/a";
114 			if ( MEASURE_TASK_TIME ) {
115 				long taskTotalMilis = TimeUnit.MILLISECONDS.convert( task.getTimerValue(), TimeUnit.NANOSECONDS );
116 				long taskAverageMilis = taskTotalMilis / task.getCounterValue();
117 				taskTotalTime = DurationFormatUtils.formatDuration( taskTotalMilis, "mm:ss.SSS" );
118 				taskAverageTime = DurationFormatUtils.formatDuration( taskAverageMilis, "mm:ss.SSS" );
119 			}
120 			out.println( "    "
121 					+ leftPad( task.getCounterValue() + "x ", 5 )
122 					+ rightPad( task.getClass().getSimpleName(), 35 )
123 					+ " | sum " + taskTotalTime
124 					+ " | avg " + taskAverageTime );
125 		}
126 	}

159 	private static String toMB(long bytes) {
160 		long megabytes = bytes / 1000 / 1000;
161 		return megabytes + "MB";
162 	}

164 	private static PrintStream createOutputStream(String testScenarioName) {
165 		try {
166 			Path targetDir = getTargetDir();
167 			String reportFileName = "report-" + testScenarioName + "-" + DateFormatUtils.format( new Date(), "yyyy-MM-dd-HH'h'mm'm'" ) + ".txt";
168 			Path reportFile = targetDir.resolve( reportFileName );
169 
170 			final OutputStream std = System.out;
171 			final OutputStream file = new PrintStream( new FileOutputStream( reportFile.toFile() ), true, "UTF-8" );
172 			final OutputStream stream = new OutputStream() {
173 
174 				@Override
175 				public void write(int b) throws IOException {
176 					std.write( b );
177 					file.write( b );
178 				}
179 
180 				@Override
181 				public void flush() throws IOException {
182 					std.flush();
183 					file.flush();
184 				}
185 
186 				@Override
187 				public void close() throws IOException {
188 					file.close();
189 				}
190 			};
191 			return new PrintStream( stream, false, "UTF-8" );
192 		}
193 		catch (FileNotFoundException | UnsupportedEncodingException e) {
194 			throw new RuntimeException( e );
195 		}
196 	}
197 
198 	private static Path getTargetDir() {
199 		InputStream runnerPropertiesStream = TestReporter.class.getResourceAsStream( "/" + RUNNER_PROPERTIES );
200 		if ( runnerPropertiesStream != null ) {
201 			Properties runnerProperties = new Properties();
202 			try {
203 				runnerProperties.load( runnerPropertiesStream );
204 			}
205 			catch (IOException e) {
206 				throw new RuntimeException( e );
207 			}
208 			return Paths.get( runnerProperties.getProperty( TARGET_DIR_KEY ) );
209 		}
210 		else {
211 			return TargetDirHelper.getTargetDir();
212 		}
213 	}
