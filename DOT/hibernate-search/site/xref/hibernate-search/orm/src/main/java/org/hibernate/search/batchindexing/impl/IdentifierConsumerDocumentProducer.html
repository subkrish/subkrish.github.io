46  /**
47   * This {@code SessionAwareRunnable} is consuming entity identifiers and
48   * producing corresponding {@code AddLuceneWork} instances being forwarded
49   * to the index writing backend.
50   * It will finish when the queue it is consuming from will
51   * signal there are no more identifiers.
52   *
53   * @author Sanne Grinovero
54   */
55  public class IdentifierConsumerDocumentProducer implements Runnable {
56  
57  	private static final Log log = LoggerFactory.make();
58  
59  	private final ProducerConsumerQueue<List<Serializable>> source;
60  	private final SessionFactory sessionFactory;
61  	private final CacheMode cacheMode;
62  	private final Class<?> type;
63  	private final MassIndexerProgressMonitor monitor;
64  	private final Map<Class<?>, EntityIndexBinding> entityIndexBindings;
65  	private final String idName;
66  	private final ErrorHandler errorHandler;
67  	private final BatchBackend backend;
68  	private final CountDownLatch producerEndSignal;
69  	private final Integer transactionTimeout;
70  	private final String tenantId;
71  
72  	/**
73  	 * The JTA transaction manager or {@code null} if not in a JTA environment
74  	 */
75  	private final TransactionManager transactionManager;
76  
77  	public IdentifierConsumerDocumentProducer(
78  			ProducerConsumerQueue<List<Serializable>> fromIdentifierListToEntities,
79  			MassIndexerProgressMonitor monitor,
80  			SessionFactory sessionFactory,
81  			CountDownLatch producerEndSignal,
82  			CacheMode cacheMode, Class<?> type,
83  			ExtendedSearchIntegrator searchFactory,
84  			String idName, BatchBackend backend, ErrorHandler errorHandler,
85  			Integer transactionTimeout,
86  			String tenantId) {
87  		this.source = fromIdentifierListToEntities;
88  		this.monitor = monitor;
89  		this.sessionFactory = sessionFactory;
90  		this.cacheMode = cacheMode;
91  		this.type = type;
92  		this.idName = idName;
93  		this.backend = backend;
94  		this.errorHandler = errorHandler;
95  		this.producerEndSignal = producerEndSignal;
96  		this.entityIndexBindings = searchFactory.getIndexBindings();
97  		this.transactionTimeout = transactionTimeout;
98  		this.tenantId = tenantId;
99  		this.transactionManager = ( (SessionFactoryImplementor) sessionFactory )
100 				.getServiceRegistry()
101 				.getService( JtaPlatform.class )
102 				.retrieveTransactionManager();
103 
104 		log.trace( "created" );
105 	}
106 
107 	@Override
108 	public void run() {
109 		log.trace( "started" );
110 		SessionImplementor session = (SessionImplementor) sessionFactory
111 				.withOptions()
112 				.tenantIdentifier( tenantId )
113 				.openSession();
114 		session.setHibernateFlushMode( FlushMode.MANUAL );
115 		session.setCacheMode( cacheMode );
116 		session.setDefaultReadOnly( true );
117 		try {
118 			loadAllFromQueue( session );
119 		}
120 		catch (Exception exception) {
121 			errorHandler.handleException( log.massIndexerExceptionWhileTransformingIds(), exception );
122 		}
123 		finally {
124 			producerEndSignal.countDown();
125 			session.close();
126 		}
127 		log.trace( "finished" );
128 	}
129 
130 	private void loadAllFromQueue(SessionImplementor session) throws Exception {
131 		final InstanceInitializer sessionInitializer = new HibernateSessionLoadingInitializer( session );
132 		try {
133 			List<Serializable> idList;
134 			do {
135 				idList = source.take();
136 				if ( idList != null ) {
137 					log.tracef( "received list of ids %s", idList );
138 					loadList( idList, session, sessionInitializer );
139 				}
140 			}
141 			while ( idList != null );
142 		}
143 		catch (InterruptedException e) {
144 			// just quit
145 			Thread.currentThread().interrupt();
146 		}
147 	}
148 
149 	/**
150 	 * Loads a list of entities of defined type using their identifiers.
151 	 * The loaded objects are then transformed into Lucene Documents
152 	 * and forwarded to the indexing backend.
153 	 *
154 	 * @param listIds the list of entity identifiers (of type
155 	 * @param session the session to be used
156 	 * @param sessionInitializer the initilization strategies for entities and collections
157 	 *
158 	 * @throws InterruptedException
159 	 */
160 	private void loadList(List<Serializable> listIds, SessionImplementor session, InstanceInitializer sessionInitializer) throws Exception {
161 		try {
162 			beginTransaction( session );
163 
164 			Criteria criteria = new CriteriaImpl( type.getName(), session )
165 					.setCacheMode( cacheMode )
166 					.setLockMode( LockMode.NONE )
167 					.setCacheable( false )
168 					.setFlushMode( FlushMode.MANUAL )
169 					.setFetchSize( listIds.size() )
170 					.setResultTransformer( CriteriaSpecification.DISTINCT_ROOT_ENTITY )
171 					.add( Restrictions.in( idName, listIds ) );
172 			List<?> list = criteria.list();
173 			monitor.entitiesLoaded( list.size() );
174 			indexAllQueue( session, list, sessionInitializer );
175 			session.clear();
176 		}
177 		finally {
178 			// it's read-only, so no need to commit
179 			rollbackTransaction( session );
180 		}
181 	}
182 
183 	private void beginTransaction(Session session) throws Exception {
184 		if ( transactionManager != null ) {
185 			if ( transactionTimeout != null ) {
186 				transactionManager.setTransactionTimeout( transactionTimeout );
187 			}
188 
189 			transactionManager.begin();
190 		}
191 		else {
192 			session.beginTransaction();
193 		}
194 	}
195 
196 	private void rollbackTransaction(SessionImplementor session) throws Exception {
197 		try {
198 			if ( transactionManager != null ) {
199 				transactionManager.rollback();
200 			}
201 			else {
202 				session.accessTransaction().rollback();
203 			}
204 		}
205 		catch (Exception e) {
206 			log.errorRollingBackTransaction( e.getMessage(), e );
207 		}
208 	}
209 
210 	private void indexAllQueue(Session session, List<?> entities, InstanceInitializer sessionInitializer) throws InterruptedException {
211 		ConversionContext contextualBridge = new ContextualExceptionBridgeHelper();
212 
213 		if ( entities == null || entities.isEmpty() ) {
214 			return;
215 		}
216 		else {
217 			log.tracef( "received a list of objects to index: %s", entities );
218 			for ( Object object : entities ) {
219 				try {
220 					index( object, session, sessionInitializer, contextualBridge );
221 					monitor.documentsBuilt( 1 );
222 				}
223 				catch (RuntimeException e) {
224 					String errorMsg = log.massIndexerUnableToIndexInstance(
225 							object.getClass().getName(),
226 							object.toString()
227 					);
228 					errorHandler.handleException( errorMsg, e );
229 				}
230 			}
231 		}
232 	}

234 	@SuppressWarnings("unchecked")
235 	private void index(Object entity, Session session, InstanceInitializer sessionInitializer, ConversionContext conversionContext)
236 			throws InterruptedException {
237 
238 		// abort if the thread has been interrupted while not in wait(), I/O or similar which themselves would have
239 		// raised the InterruptedException
240 		if ( Thread.currentThread().isInterrupted() ) {
241 			throw new InterruptedException();
242 		}
243 
244 		Serializable id = session.getIdentifier( entity );
245 		Class<?> clazz = HibernateHelper.getClass( entity );
246 		EntityIndexBinding entityIndexBinding = entityIndexBindings.get( clazz );
247 		if ( entityIndexBinding == null ) {
248 			// it might be possible to receive not-indexes subclasses of the currently indexed type;
249 			// being not-indexed, we skip them.
250 			// FIXME for improved performance: avoid loading them in an early phase.
251 			return;
252 		}
253 
254 		EntityIndexingInterceptor interceptor = entityIndexBinding.getEntityIndexingInterceptor();
255 		if ( interceptor != null ) {
256 			IndexingOverride onAdd = interceptor.onAdd( entity );
257 			switch ( onAdd ) {
258 				case REMOVE:
259 				case SKIP:
260 					return;
261 			}
262 			//default: continue indexing this instance
263 		}
264 
265 		DocumentBuilderIndexedEntity docBuilder = entityIndexBinding.getDocumentBuilder();
266 		TwoWayFieldBridge idBridge = docBuilder.getIdBridge();
267 		String idPropertyName = docBuilder.getIdPropertyName();
268 		if ( idPropertyName != null ) {
269 			conversionContext.pushProperty( idPropertyName );
270 		}
271 		String idInString = null;
272 		try {
273 			idInString = conversionContext
274 					.setClass( clazz )
275 					.twoWayConversionContext( idBridge )
276 					.objectToString( id );
277 		}
278 		finally {
279 			if ( idPropertyName != null ) {
280 				conversionContext.popProperty();
281 			}
282 		}
283 		//depending on the complexity of the object graph going to be indexed it's possible
284 		//that we hit the database several times during work construction.
285 		AddLuceneWork addWork = docBuilder.createAddWork(
286 				tenantId,
287 				clazz,
288 				entity,
289 				id,
290 				idInString,
291 				sessionInitializer,
292 				conversionContext
293 		);
294 		backend.enqueueAsyncWork( addWork );
295 	}
