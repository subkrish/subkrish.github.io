8   package org.hibernate.search.test.bridge.provider;
9   
10  import org.hibernate.search.bridge.FieldBridge;
11  import org.hibernate.search.bridge.StringBridge;
12  import org.hibernate.search.bridge.spi.BridgeProvider;
13  import org.hibernate.search.bridge.util.impl.String2FieldBridgeAdaptor;
14  
15  /**
16   * @author Emmanuel Bernard
17   */
18  public class MovieBridgeProvider implements BridgeProvider {
19  	@Override
20  	public FieldBridge provideFieldBridge(BridgeProviderContext bridgeProviderContext) {
21  		if ( bridgeProviderContext.getReturnType().equals( Movie.class ) ) {
22  			return new String2FieldBridgeAdaptor( new StringBridge() {
23  
24  				@Override
25  				public String objectToString(Object object) {
26  					return ( (Movie) object ).toString();
27  				}
28  			} );
29  		}
30  		return null;
31  	}
32  }
