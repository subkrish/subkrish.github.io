65  /**
66   * Read/write operations on region and assignment information store in
67   * <code>hbase:meta</code>.
68   *
69   * Some of the methods of this class take ZooKeeperWatcher as a param. The only reason
70   * for this is because when used on client-side (like from HBaseAdmin), we want to use
71   * short-living connection (opened before each operation, closed right after), while
72   * when used on HM or HRS (like in AssignmentManager) we want permanent connection.
73   */
74  @InterfaceAudience.Private
75  public class MetaTableAccessor {
76  
77    /*
78     * HBASE-10070 adds a replicaId to HRI, meaning more than one HRI can be defined for the
79     * same table range (table, startKey, endKey). For every range, there will be at least one
80     * HRI defined which is called default replica.
81     *
82     * Meta layout (as of 0.98 + HBASE-10070) is like:
83     * For each table range, there is a single row, formatted like:
84     * <tableName>,<startKey>,<regionId>,<encodedRegionName>. This row corresponds to the regionName
85     * of the default region replica.
86     * Columns are:
87     * info:regioninfo         => contains serialized HRI for the default region replica
88     * info:server             => contains hostname:port (in string form) for the server hosting
89     *                            the default regionInfo replica
90     * info:server_<replicaId> => contains hostname:port (in string form) for the server hosting the
91     *                            regionInfo replica with replicaId
92     * info:serverstartcode    => contains server start code (in binary long form) for the server
93     *                            hosting the default regionInfo replica
94     * info:serverstartcode_<replicaId> => contains server start code (in binary long form) for the
95     *                                     server hosting the regionInfo replica with replicaId
96     * info:seqnumDuringOpen    => contains seqNum (in binary long form) for the region at the time
97     *                             the server opened the region with default replicaId
98     * info:seqnumDuringOpen_<replicaId> => contains seqNum (in binary long form) for the region at
99     *                             the time the server opened the region with replicaId
100    * info:splitA              => contains a serialized HRI for the first daughter region if the
101    *                             region is split
102    * info:splitB              => contains a serialized HRI for the second daughter region if the
103    *                             region is split
104    * info:mergeA              => contains a serialized HRI for the first parent region if the
105    *                             region is the result of a merge
106    * info:mergeB              => contains a serialized HRI for the second parent region if the
107    *                             region is the result of a merge
108    *
109    * The actual layout of meta should be encapsulated inside MetaTableAccessor methods,
110    * and should not leak out of it (through Result objects, etc)
111    */
112

173   /**
174    * Callers should call close on the returned {@link Table} instance.
175    * @param connection connection we're using to access Meta
176    * @return An {@link Table} for <code>hbase:meta</code>
177    * @throws IOException
178    */
179   static Table getMetaHTable(final Connection connection)
180   throws IOException {
181     // We used to pass whole CatalogTracker in here, now we just pass in Connection
182     if (connection == null) {
183       throw new NullPointerException("No connection");
184     } else if (connection.isClosed()) {
185       throw new IOException("connection is closed");
186     }
187     // If the passed in 'connection' is 'managed' -- i.e. every second test uses
188     // a Table or an HBaseAdmin with managed connections -- then doing
189     // connection.getTable will throw an exception saying you are NOT to use
190     // managed connections getting tables.  Leaving this as it is for now. Will
191     // revisit when inclined to change all tests.  User code probaby makes use of
192     // managed connections too so don't change it till post hbase 1.0.
193     //
194     // There should still be a way to use this method with an unmanaged connection.
195     if (connection instanceof ClusterConnection) {
196       if (((ClusterConnection) connection).isManaged()) {
197         return new HTable(TableName.META_TABLE_NAME, (ClusterConnection) connection);
198       }
199     }
200     return connection.getTable(TableName.META_TABLE_NAME);
201   }

233   /**
234    * Returns the HRegionLocation from meta for the given region
235    * @param connection connection we're using
236    * @param regionName region we're looking for
237    * @return HRegionLocation for the given region
238    * @throws IOException
239    */
240   public static HRegionLocation getRegionLocation(Connection connection,
241                                                   byte[] regionName) throws IOException {
242     byte[] row = regionName;
243     HRegionInfo parsedInfo = null;
244     try {
245       parsedInfo = parseRegionInfoFromRegionName(regionName);
246       row = getMetaKeyForRegion(parsedInfo);
247     } catch (Exception parseEx) {
248       // Ignore. This is used with tableName passed as regionName.
249     }
250     Get get = new Get(row);
251     get.addFamily(HConstants.CATALOG_FAMILY);
252     Result r = get(getMetaHTable(connection), get);
253     RegionLocations locations = getRegionLocations(r);
254     return locations == null
255       ? null
256       : locations.getRegionLocation(parsedInfo == null ? 0 : parsedInfo.getReplicaId());
257   }

280   /** Returns an HRI parsed from this regionName. Not all the fields of the HRI
281    * is stored in the name, so the returned object should only be used for the fields
282    * in the regionName.
283    */
284   protected static HRegionInfo parseRegionInfoFromRegionName(byte[] regionName)
285     throws IOException {
286     byte[][] fields = HRegionInfo.parseRegionName(regionName);
287     long regionId =  Long.parseLong(Bytes.toString(fields[2]));
288     int replicaId = fields.length > 3 ? Integer.parseInt(Bytes.toString(fields[3]), 16) : 0;
289     return new HRegionInfo(
290       TableName.valueOf(fields[0]), fields[1], fields[1], false, regionId, replicaId);
291   }

323   /**
324    * Checks if the specified table exists.  Looks at the hbase:meta table hosted on
325    * the specified server.
326    * @param connection connection we're using
327    * @param tableName table to check
328    * @return true if the table exists in meta, false if not
329    * @throws IOException
330    */
331   public static boolean tableExists(Connection connection,
332       final TableName tableName)
333   throws IOException {
334     if (tableName.equals(TableName.META_TABLE_NAME)) {
335       // Catalog tables always exist.
336       return true;
337     }
338     // Make a version of ResultCollectingVisitor that only collects the first
339     CollectingVisitor<HRegionInfo> visitor = new CollectingVisitor<HRegionInfo>() {
340       private HRegionInfo current = null;
341 
342       @Override
343       public boolean visit(Result r) throws IOException {
344         RegionLocations locations = getRegionLocations(r);
345         if (locations == null || locations.getRegionLocation().getRegionInfo() == null) {
346           LOG.warn("No serialized HRegionInfo in " + r);
347           return true;
348         }
349         this.current = locations.getRegionLocation().getRegionInfo();
350         if (this.current == null) {
351           LOG.warn("No serialized HRegionInfo in " + r);
352           return true;
353         }
354         if (!isInsideTable(this.current, tableName)) return false;
355         // Else call super and add this Result to the collection.
356         super.visit(r);
357         // Stop collecting regions from table after we get one.
358         return false;
359       }
360 
361       @Override
362       void add(Result r) {
363         // Add the current HRI.
364         this.results.add(this.current);
365       }
366     };
367     fullScan(connection, visitor, getTableStartRowForMeta(tableName));
368     // If visitor has results >= 1 then table exists.
369     return visitor.getResults().size() >= 1;
370   }

437   /**
438    * This method creates a Scan object that will only scan catalog rows that
439    * belong to the specified table. It doesn't specify any columns.
440    * This is a better alternative to just using a start row and scan until
441    * it hits a new table since that requires parsing the HRI to get the table
442    * name.
443    * @param tableName bytes of table's name
444    * @return configured Scan object
445    */
446   public static Scan getScanForTableName(TableName tableName) {
447     String strName = tableName.getNameAsString();
448     // Start key is just the table name with delimiters
449     byte[] startKey = Bytes.toBytes(strName + ",,");
450     // Stop key appends the smallest possible char to the table name
451     byte[] stopKey = Bytes.toBytes(strName + " ,,");
452 
453     Scan scan = new Scan(startKey);
454     scan.setStopRow(stopKey);
455     return scan;
456   }

472   /**
473    * @param zkw ZooKeeperWatcher instance we're using to get hbase:meta location
474    * @param connection connection we're using
475    * @param tableName table to work with
476    * @return Return list of regioninfos and server addresses.
477    * @throws IOException
478    */
479   public static List<Pair<HRegionInfo, ServerName>> getTableRegionsAndLocations(
480       ZooKeeperWatcher zkw, Connection connection, final TableName tableName,
481       final boolean excludeOfflinedSplitParents) throws IOException {
482 
483     if (tableName.equals(TableName.META_TABLE_NAME)) {
484       // If meta, do a bit of special handling.
485       ServerName serverName = new MetaTableLocator().getMetaRegionLocation(zkw);
486       List<Pair<HRegionInfo, ServerName>> list =
487         new ArrayList<Pair<HRegionInfo, ServerName>>();
488       list.add(new Pair<HRegionInfo, ServerName>(HRegionInfo.FIRST_META_REGIONINFO,
489         serverName));
490       return list;
491     }
492     // Make a version of CollectingVisitor that collects HRegionInfo and ServerAddress
493     CollectingVisitor<Pair<HRegionInfo, ServerName>> visitor =
494       new CollectingVisitor<Pair<HRegionInfo, ServerName>>() {
495         private RegionLocations current = null;
496 
497         @Override
498         public boolean visit(Result r) throws IOException {
499           current = getRegionLocations(r);
500           if (current == null || current.getRegionLocation().getRegionInfo() == null) {
501             LOG.warn("No serialized HRegionInfo in " + r);
502             return true;
503           }
504           HRegionInfo hri = current.getRegionLocation().getRegionInfo();
505           if (!isInsideTable(hri, tableName)) return false;
506           if (excludeOfflinedSplitParents && hri.isSplitParent()) return true;
507           // Else call super and add this Result to the collection.
508           return super.visit(r);
509         }
510 
511         @Override
512         void add(Result r) {
513           if (current == null) {
514             return;
515           }
516           for (HRegionLocation loc : current.getRegionLocations()) {
517             if (loc != null) {
518               this.results.add(new Pair<HRegionInfo, ServerName>(
519                 loc.getRegionInfo(), loc.getServerName()));
520             }
521           }
522         }
523       };
524     fullScan(connection, visitor, getTableStartRowForMeta(tableName));
525     return visitor.getResults();
526   }

528   /**
529    * @param connection connection we're using
530    * @param serverName server whose regions we're interested in
531    * @return List of user regions installed on this server (does not include
532    * catalog regions).
533    * @throws IOException
534    */
535   public static NavigableMap<HRegionInfo, Result>
536   getServerUserRegions(Connection connection, final ServerName serverName)
537     throws IOException {
538     final NavigableMap<HRegionInfo, Result> hris = new TreeMap<HRegionInfo, Result>();
539     // Fill the above hris map with entries from hbase:meta that have the passed
540     // servername.
541     CollectingVisitor<Result> v = new CollectingVisitor<Result>() {
542       @Override
543       void add(Result r) {
544         if (r == null || r.isEmpty()) return;
545         RegionLocations locations = getRegionLocations(r);
546         if (locations == null) return;
547         for (HRegionLocation loc : locations.getRegionLocations()) {
548           if (loc != null) {
549             if (loc.getServerName() != null && loc.getServerName().equals(serverName)) {
550               hris.put(loc.getRegionInfo(), r);
551             }
552           }
553         }
554       }
555     };
556     fullScan(connection, v);
557     return hris;
558   }

580   /**
581    * Performs a full scan of a catalog table.
582    * @param connection connection we're using
583    * @param visitor Visitor invoked against each row.
584    * @param startrow Where to start the scan. Pass null if want to begin scan
585    * at first row.
586    * <code>hbase:meta</code>, the default (pass false to scan hbase:meta)
587    * @throws IOException
588    */
589   public static void fullScan(Connection connection,
590     final Visitor visitor, final byte [] startrow)
591   throws IOException {
592     Scan scan = new Scan();
593     if (startrow != null) scan.setStartRow(startrow);
594     if (startrow == null) {
595       int caching = connection.getConfiguration()
596           .getInt(HConstants.HBASE_META_SCANNER_CACHING, 100);
597       scan.setCaching(caching);
598     }
599     scan.addFamily(HConstants.CATALOG_FAMILY);
600     Table metaTable = getMetaHTable(connection);
601     ResultScanner scanner = null;
602     try {
603       scanner = metaTable.getScanner(scan);
604       Result data;
605       while((data = scanner.next()) != null) {
606         if (data.isEmpty()) continue;
607         // Break if visit returns false.
608         if (!visitor.visit(data)) break;
609       }
610     } finally {
611       if (scanner != null) scanner.close();
612       metaTable.close();
613     }
614   }

671   /**
672    * Parses the replicaId from the server column qualifier. See top of the class javadoc
673    * for the actual meta layout
674    * @param serverColumn the column qualifier
675    * @return an int for the replicaId
676    */
677   @VisibleForTesting
678   static int parseReplicaIdFromServerColumn(byte[] serverColumn) {
679     String serverStr = Bytes.toString(serverColumn);
680 
681     Matcher matcher = SERVER_COLUMN_PATTERN.matcher(serverStr);
682     if (matcher.matches() && matcher.groupCount() > 0) {
683       String group = matcher.group(1);
684       if (group != null && group.length() > 0) {
685         return Integer.parseInt(group.substring(1), 16);
686       } else {
687         return 0;
688       }
689     }
690     return -1;
691   }

694    * Returns a {@link ServerName} from catalog table {@link Result}.
695    * @param r Result to pull from
696    * @return A ServerName instance or null if necessary fields not found or empty.
697    */
698   private static ServerName getServerName(final Result r, final int replicaId) {
699     byte[] serverColumn = getServerColumn(replicaId);
700     Cell cell = r.getColumnLatestCell(getFamily(), serverColumn);
701     if (cell == null || cell.getValueLength() == 0) return null;
702     String hostAndPort = Bytes.toString(
703       cell.getValueArray(), cell.getValueOffset(), cell.getValueLength());
704     byte[] startcodeColumn = getStartCodeColumn(replicaId);
705     cell = r.getColumnLatestCell(getFamily(), startcodeColumn);
706     if (cell == null || cell.getValueLength() == 0) return null;
707     return ServerName.valueOf(hostAndPort,
708       Bytes.toLong(cell.getValueArray(), cell.getValueOffset(), cell.getValueLength()));
709   }

723   /**
724    * Returns an HRegionLocationList extracted from the result.
725    * @return an HRegionLocationList containing all locations for the region range or null if
726    *  we can't deserialize the result.
727    */
728   public static RegionLocations getRegionLocations(final Result r) {
729     if (r == null) return null;
730     HRegionInfo regionInfo = getHRegionInfo(r, getRegionInfoColumn());
731     if (regionInfo == null) return null;
732 
733     List<HRegionLocation> locations = new ArrayList<HRegionLocation>(1);
734     NavigableMap<byte[],NavigableMap<byte[],byte[]>> familyMap = r.getNoVersionMap();
735 
736     locations.add(getRegionLocation(r, regionInfo, 0));
737 
738     NavigableMap<byte[], byte[]> infoMap = familyMap.get(getFamily());
739     if (infoMap == null) return new RegionLocations(locations);
740 
741     // iterate until all serverName columns are seen
742     int replicaId = 0;
743     byte[] serverColumn = getServerColumn(replicaId);
744     SortedMap<byte[], byte[]> serverMap = infoMap.tailMap(serverColumn, false);
745     if (serverMap.isEmpty()) return new RegionLocations(locations);
746 
747     for (Map.Entry<byte[], byte[]> entry : serverMap.entrySet()) {
748       replicaId = parseReplicaIdFromServerColumn(entry.getKey());
749       if (replicaId < 0) {
750         break;
751       }
752       HRegionLocation location = getRegionLocation(r, regionInfo, replicaId);
753       // In case the region replica is newly created, it's location might be null. We usually do not
754       // have HRL's in RegionLocations object with null ServerName. They are handled as null HRLs.
755       if (location == null || location.getServerName() == null) {
756         locations.add(null);
757       } else {
758         locations.add(location);
759       }
760     }
761 
762     return new RegionLocations(locations);
763   }

848   /**
849    * A {@link Visitor} that collects content out of passed {@link Result}.
850    */
851   static abstract class CollectingVisitor<T> implements Visitor {
852     final List<T> results = new ArrayList<T>();
853     @Override
854     public boolean visit(Result r) throws IOException {
855       if (r ==  null || r.isEmpty()) return true;
856       add(r);
857       return true;
858     }
859 
860     abstract void add(Result r);
861 
862     /**
863      * @return Collected results; wait till visits complete to collect all
864      * possible results
865      */
866     List<T> getResults() {
867       return this.results;
868     }
869   }

873    */

1193   /**
1194    * Merge the two regions into one in an atomic operation. Deletes the two
1195    * merging regions in hbase:meta and adds the merged region with the information of
1196    * two merging regions.
1197    * @param connection connection we're using
1198    * @param mergedRegion the merged region
1199    * @param regionA
1200    * @param regionB
1201    * @param sn the location of the region
1202    * @throws IOException
1203    */
1204   public static void mergeRegions(final Connection connection, HRegionInfo mergedRegion,
1205       HRegionInfo regionA, HRegionInfo regionB, ServerName sn, int regionReplication)
1206           throws IOException {
1207     Table meta = getMetaHTable(connection);
1208     try {
1209       HRegionInfo copyOfMerged = new HRegionInfo(mergedRegion);
1210 
1211       // Put for parent
1212       Put putOfMerged = makePutFromRegionInfo(copyOfMerged);
1213       putOfMerged.addImmutable(HConstants.CATALOG_FAMILY, HConstants.MERGEA_QUALIFIER,
1214         regionA.toByteArray());
1215       putOfMerged.addImmutable(HConstants.CATALOG_FAMILY, HConstants.MERGEB_QUALIFIER,
1216         regionB.toByteArray());
1217 
1218       // Deletes for merging regions
1219       Delete deleteA = makeDeleteFromRegionInfo(regionA);
1220       Delete deleteB = makeDeleteFromRegionInfo(regionB);
1221 
1222       // The merged is a new region, openSeqNum = 1 is fine.
1223       addLocation(putOfMerged, sn, 1, mergedRegion.getReplicaId());
1224 
1225       // Add empty locations for region replicas of the merged region so that number of replicas can
1226       // be cached whenever the primary region is looked up from meta
1227       for (int i = 1; i < regionReplication; i++) {
1228         addEmptyLocation(putOfMerged, i);
1229       }
1230 
1231       byte[] tableRow = Bytes.toBytes(mergedRegion.getRegionNameAsString()
1232         + HConstants.DELIMITER);
1233       multiMutate(meta, tableRow, putOfMerged, deleteA, deleteB);
1234     } finally {
1235       meta.close();
1236     }
1237   }

1249    */
1250   public static void splitRegion(final Connection connection,
1251                                  HRegionInfo parent, HRegionInfo splitA, HRegionInfo splitB,
1252                                  ServerName sn, int regionReplication) throws IOException {
1253     Table meta = getMetaHTable(connection);
1254     try {
1255       HRegionInfo copyOfParent = new HRegionInfo(parent);
1256       copyOfParent.setOffline(true);
1257       copyOfParent.setSplit(true);
1258 
1259       //Put for parent
1260       Put putParent = makePutFromRegionInfo(copyOfParent);
1261       addDaughtersToPut(putParent, splitA, splitB);
1262 
1263       //Puts for daughters
1264       Put putA = makePutFromRegionInfo(splitA);
1265       Put putB = makePutFromRegionInfo(splitB);
1266 
1267       addLocation(putA, sn, 1, splitA.getReplicaId()); //new regions, openSeqNum = 1 is fine.
1268       addLocation(putB, sn, 1, splitB.getReplicaId());
1269 
1270       // Add empty locations for region replicas of daughters so that number of replicas can be
1271       // cached whenever the primary region is looked up from meta
1272       for (int i = 1; i < regionReplication; i++) {
1273         addEmptyLocation(putA, i);
1274         addEmptyLocation(putB, i);
1275       }
1276 
1277       byte[] tableRow = Bytes.toBytes(parent.getRegionNameAsString() + HConstants.DELIMITER);
1278       multiMutate(meta, tableRow, putParent, putA, putB);
1279     } finally {
1280       meta.close();
1281     }
1282   }

1284   /**
1285    * Performs an atomic multi-Mutate operation against the given table.
1286    */
1287   private static void multiMutate(Table table, byte[] row, Mutation... mutations)
1288       throws IOException {
1289     CoprocessorRpcChannel channel = table.coprocessorService(row);
1290     MultiRowMutationProtos.MutateRowsRequest.Builder mmrBuilder
1291       = MultiRowMutationProtos.MutateRowsRequest.newBuilder();
1292     for (Mutation mutation : mutations) {
1293       if (mutation instanceof Put) {
1294         mmrBuilder.addMutationRequest(ProtobufUtil.toMutation(
1295           ClientProtos.MutationProto.MutationType.PUT, mutation));
1296       } else if (mutation instanceof Delete) {
1297         mmrBuilder.addMutationRequest(ProtobufUtil.toMutation(
1298           ClientProtos.MutationProto.MutationType.DELETE, mutation));
1299       } else {
1300         throw new DoNotRetryIOException("multi in MetaEditor doesn't support "
1301           + mutation.getClass().getName());
1302       }
1303     }
1304 
1305     MultiRowMutationProtos.MultiRowMutationService.BlockingInterface service =
1306       MultiRowMutationProtos.MultiRowMutationService.newBlockingStub(channel);
1307     try {
1308       service.mutateRows(null, mmrBuilder.build());
1309     } catch (ServiceException ex) {
1310       ProtobufUtil.toIOException(ex);
1311     }
1312   }

1386   /**
1387    * Adds and Removes the specified regions from hbase:meta
1388    * @param connection connection we're using
1389    * @param regionsToRemove list of regions to be deleted from META
1390    * @param regionsToAdd list of regions to be added to META
1391    * @throws IOException
1392    */
1393   public static void mutateRegions(Connection connection,
1394                                    final List<HRegionInfo> regionsToRemove,
1395                                    final List<HRegionInfo> regionsToAdd)
1396     throws IOException {
1397     List<Mutation> mutation = new ArrayList<Mutation>();
1398     if (regionsToRemove != null) {
1399       for (HRegionInfo hri: regionsToRemove) {
1400         mutation.add(new Delete(hri.getRegionName()));
1401       }
1402     }
1403     if (regionsToAdd != null) {
1404       for (HRegionInfo hri: regionsToAdd) {
1405         mutation.add(makePutFromRegionInfo(hri));
1406       }
1407     }
1408     mutateMetaTable(connection, mutation);
1409     if (regionsToRemove != null && regionsToRemove.size() > 0) {
1410       LOG.debug("Deleted " + regionsToRemove);
1411     }
1412     if (regionsToAdd != null && regionsToAdd.size() > 0) {
1413       LOG.debug("Added " + regionsToAdd);
1414     }
1415   }

1418    * Overwrites the specified regions from hbase:meta
1419    * @param connection connection we're using
1420    * @param regionInfos list of regions to be added to META
1421    * @throws IOException
1422    */
1423   public static void overwriteRegions(Connection connection,
1424       List<HRegionInfo> regionInfos, int regionReplication) throws IOException {
1425     deleteRegions(connection, regionInfos);
1426     // Why sleep? This is the easiest way to ensure that the previous deletes does not
1427     // eclipse the following puts, that might happen in the same ts from the server.
1428     // See HBASE-9906, and HBASE-9879. Once either HBASE-9879, HBASE-8770 is fixed,
1429     // or HBASE-9905 is fixed and meta uses seqIds, we do not need the sleep.
1430     Threads.sleep(20);
1431     addRegionsToMeta(connection, regionInfos, regionReplication);
1432     LOG.info("Overwritten " + regionInfos);
1433   }

1460   public static Put addLocation(final Put p, final ServerName sn, long openSeqNum, int replicaId){
1461     // using regionserver's local time as the timestamp of Put.
1462     // See: HBASE-11536
1463     long now = EnvironmentEdgeManager.currentTime();
1464     p.addImmutable(HConstants.CATALOG_FAMILY, getServerColumn(replicaId), now,
1465       Bytes.toBytes(sn.getHostAndPort()));
1466     p.addImmutable(HConstants.CATALOG_FAMILY, getStartCodeColumn(replicaId), now,
1467       Bytes.toBytes(sn.getStartcode()));
1468     p.addImmutable(HConstants.CATALOG_FAMILY, getSeqNumColumn(replicaId), now,
1469       Bytes.toBytes(openSeqNum));
1470     return p;
1471   }
