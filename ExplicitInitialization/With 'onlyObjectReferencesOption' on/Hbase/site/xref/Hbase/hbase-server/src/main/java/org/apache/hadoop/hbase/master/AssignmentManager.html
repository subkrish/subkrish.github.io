261   /**
262    * Constructs a new assignment manager.
263    *
264    * @param server instance of HMaster this AM running inside
265    * @param serverManager serverManager for associated HMaster
266    * @param balancer implementation of {@link LoadBalancer}
267    * @param service Executor service
268    * @param metricsMaster metrics manager
269    * @param tableLockManager TableLock manager
270    * @throws KeeperException
271    * @throws IOException
272    */
273   public AssignmentManager(Server server, ServerManager serverManager,
274       final LoadBalancer balancer,
275       final ExecutorService service, MetricsMaster metricsMaster,
276       final TableLockManager tableLockManager) throws KeeperException,
277         IOException, CoordinatedStateException {
278     super(server.getZooKeeper());
279     this.server = server;
280     this.serverManager = serverManager;
281     this.executorService = service;
282     this.regionStateStore = new RegionStateStore(server);
283     this.regionsToReopen = Collections.synchronizedMap
284                            (new HashMap<String, HRegionInfo> ());
285     Configuration conf = server.getConfiguration();
286     // Only read favored nodes if using the favored nodes load balancer.
287     this.shouldAssignRegionsWithFavoredNodes = conf.getClass(
288            HConstants.HBASE_MASTER_LOADBALANCER_CLASS, Object.class).equals(
289            FavoredNodeLoadBalancer.class);
290     try {
291       if (server.getCoordinatedStateManager() != null) {
292         this.tableStateManager = server.getCoordinatedStateManager().getTableStateManager();
293       } else {
294         this.tableStateManager = null;
295       }
296     } catch (InterruptedException e) {
297       throw new InterruptedIOException();
298     }
299     // This is the max attempts, not retries, so it should be at least 1.
300     this.maximumAttempts = Math.max(1,
301       this.server.getConfiguration().getInt("hbase.assignment.maximum.attempts", 10));
302     this.sleepTimeBeforeRetryingMetaAssignment = this.server.getConfiguration().getLong(
303         "hbase.meta.assignment.retry.sleeptime", 1000l);
304     this.balancer = balancer;
305     int maxThreads = conf.getInt("hbase.assignment.threads.max", 30);
306     this.threadPoolExecutorService = Threads.getBoundedCachedThreadPool(
307       maxThreads, 60L, TimeUnit.SECONDS, Threads.newDaemonThreadFactory("AM."));
308     this.regionStates = new RegionStates(
309       server, tableStateManager, serverManager, regionStateStore);
310 
311     this.bulkAssignWaitTillAllAssigned =
312       conf.getBoolean("hbase.bulk.assignment.waittillallassigned", false);
313     this.bulkAssignThresholdRegions = conf.getInt("hbase.bulk.assignment.threshold.regions", 7);
314     this.bulkAssignThresholdServers = conf.getInt("hbase.bulk.assignment.threshold.servers", 3);
315 
316     int workers = conf.getInt("hbase.assignment.zkevent.workers", 20);
317     ThreadFactory threadFactory = Threads.newDaemonThreadFactory("AM.ZK.Worker");
318     zkEventWorkers = Threads.getBoundedCachedThreadPool(workers, 60L,
319             TimeUnit.SECONDS, threadFactory);
320     this.tableLockManager = tableLockManager;
321 
322     this.metricsAssignmentManager = new MetricsAssignmentManager();
323     useZKForAssignment = ConfigUtil.useZKForAssignment(conf);
324   }

408    *
409    * @param tableName
410    * @return Pair indicating the status of the alter command
411    * @throws IOException
412    */
413   public Pair<Integer, Integer> getReopenStatus(TableName tableName)
414       throws IOException {
415     List<HRegionInfo> hris;
416     if (TableName.META_TABLE_NAME.equals(tableName)) {
417       hris = new MetaTableLocator().getMetaRegions(server.getZooKeeper());
418     } else {
419       hris = MetaTableAccessor.getTableRegions(server.getZooKeeper(),
420         server.getConnection(), tableName, true);
421     }
422 
423     Integer pending = 0;
424     for (HRegionInfo hri : hris) {
425       String name = hri.getEncodedName();
426       // no lock concurrent access ok: sequential consistency respected.
427       if (regionsToReopen.containsKey(name)
428           || regionStates.isRegionInTransition(name)) {
429         pending++;
430       }
431     }
432     return new Pair<Integer, Integer>(pending, hris.size());
433   }

461   /**
462    * Called on startup.
463    * Figures whether a fresh cluster start of we are joining extant running cluster.
464    * @throws IOException
465    * @throws KeeperException
466    * @throws InterruptedException
467    * @throws CoordinatedStateException
468    */
469   void joinCluster() throws IOException,
470       KeeperException, InterruptedException, CoordinatedStateException {
471     long startTime = System.currentTimeMillis();
472     // Concurrency note: In the below the accesses on regionsInTransition are
473     // outside of a synchronization block where usually all accesses to RIT are
474     // synchronized.  The presumption is that in this case it is safe since this
475     // method is being played by a single thread on startup.
476 
477     // TODO: Regions that have a null location and are not in regionsInTransitions
478     // need to be handled.
479 
480     // Scan hbase:meta to build list of existing regions, servers, and assignment
481     // Returns servers who have not checked in (assumed dead) that some regions
482     // were assigned to (according to the meta)
483     Set<ServerName> deadServers = rebuildUserRegions();
484 
485     // This method will assign all user regions if a clean server startup or
486     // it will reconstruct master state and cleanup any leftovers from
487     // previous master process.
488     boolean failover = processDeadServersAndRegionsInTransition(deadServers);
489 
490     if (!useZKForAssignment) {
491       // Not use ZK for assignment any more, remove the ZNode
492       ZKUtil.deleteNodeRecursively(watcher, watcher.assignmentZNode);
493     }
494     recoverTableInDisablingState();
495     recoverTableInEnablingState();
496     LOG.info("Joined the cluster in " + (System.currentTimeMillis()
497       - startTime) + "ms, failover=" + failover);
498   }

500   /**
501    * Process all regions that are in transition in zookeeper and also
502    * processes the list of dead servers by scanning the META.
503    * Used by master joining an cluster.  If we figure this is a clean cluster
504    * startup, will assign all user regions.
505    * @param deadServers
506    *          Map of dead servers and their regions. Can be null.
507    * @throws KeeperException
508    * @throws IOException
509    * @throws InterruptedException
510    */
511   boolean processDeadServersAndRegionsInTransition(
512       final Set<ServerName> deadServers) throws KeeperException,
513         IOException, InterruptedException, CoordinatedStateException {
514     List<String> nodes = ZKUtil.listChildrenNoWatch(watcher,
515       watcher.assignmentZNode);
516 
517     if (useZKForAssignment && nodes == null) {
518       String errorMessage = "Failed to get the children from ZK";
519       server.abort(errorMessage, new IOException(errorMessage));
520       return true; // Doesn't matter in this case
521     }
522 
523     boolean failover = !serverManager.getDeadServers().isEmpty();
524     if (failover) {
525       // This may not be a failover actually, especially if meta is on this master.
526       if (LOG.isDebugEnabled()) {
527         LOG.debug("Found dead servers out on cluster " + serverManager.getDeadServers());
528       }
529     } else {
530       // If any one region except meta is assigned, it's a failover.
531       Set<ServerName> onlineServers = serverManager.getOnlineServers().keySet();
532       for (Map.Entry<HRegionInfo, ServerName> en:
533           regionStates.getRegionAssignments().entrySet()) {
534         HRegionInfo hri = en.getKey();
535         if (!hri.isMetaTable()
536             && onlineServers.contains(en.getValue())) {
537           LOG.debug("Found " + hri + " out on cluster");
538           failover = true;
539           break;
540         }
541       }
542       if (!failover && nodes != null) {
543         // If any one region except meta is in transition, it's a failover.
544         for (String encodedName: nodes) {
545           RegionState regionState = regionStates.getRegionState(encodedName);
546           if (regionState != null && !regionState.getRegion().isMetaRegion()) {
547             LOG.debug("Found " + regionState + " in RITs");
548             failover = true;
549             break;
550           }
551         }
552       }
553     }
554     if (!failover && !useZKForAssignment) {
555       // If any region except meta is in transition on a live server, it's a failover.
556       Map<String, RegionState> regionsInTransition = regionStates.getRegionsInTransition();
557       if (!regionsInTransition.isEmpty()) {
558         Set<ServerName> onlineServers = serverManager.getOnlineServers().keySet();
559         for (RegionState regionState: regionsInTransition.values()) {
560           ServerName serverName = regionState.getServerName();
561           if (!regionState.getRegion().isMetaRegion()
562               && serverName != null && onlineServers.contains(serverName)) {
563             LOG.debug("Found " + regionState + " in RITs");
564             failover = true;
565             break;
566           }
567         }
568       }
569     }
570     if (!failover) {
571       // If we get here, we have a full cluster restart. It is a failover only
572       // if there are some WALs are not split yet. For meta WALs, they should have
573       // been split already, if any. We can walk through those queued dead servers,
574       // if they don't have any WALs, this restart should be considered as a clean one
575       Set<ServerName> queuedDeadServers = serverManager.getRequeuedDeadServers().keySet();
576       if (!queuedDeadServers.isEmpty()) {
577         Configuration conf = server.getConfiguration();
578         Path rootdir = FSUtils.getRootDir(conf);
579         FileSystem fs = rootdir.getFileSystem(conf);
580         for (ServerName serverName: queuedDeadServers) {
581           // In the case of a clean exit, the shutdown handler would have presplit any WALs and
582           // removed empty directories.
583           Path logDir = new Path(rootdir,
584               DefaultWALProvider.getWALDirectoryName(serverName.toString()));
585           Path splitDir = logDir.suffix(DefaultWALProvider.SPLITTING_EXT);
586           if (fs.exists(logDir) || fs.exists(splitDir)) {
587             LOG.debug("Found queued dead server " + serverName);
588             failover = true;
589             break;
590           }
591         }
592         if (!failover) {
593           // We figured that it's not a failover, so no need to
594           // work on these re-queued dead servers any more.
595           LOG.info("AM figured that it's not a failover and cleaned up "
596             + queuedDeadServers.size() + " queued dead servers");
597           serverManager.removeRequeuedDeadServers();
598         }
599       }
600     }
601 
602     Set<TableName> disabledOrDisablingOrEnabling = null;
603     Map<HRegionInfo, ServerName> allRegions = null;
604 
605     if (!failover) {
606       disabledOrDisablingOrEnabling = tableStateManager.getTablesInStates(
607         ZooKeeperProtos.Table.State.DISABLED, ZooKeeperProtos.Table.State.DISABLING,
608         ZooKeeperProtos.Table.State.ENABLING);
609 
610       // Clean re/start, mark all user regions closed before reassignment
611       allRegions = regionStates.closeAllUserRegions(
612         disabledOrDisablingOrEnabling);
613     }
614 
615     // Now region states are restored
616     regionStateStore.start();
617 
618     // If we found user regions out on cluster, its a failover.
619     if (failover) {
620       LOG.info("Found regions out on cluster or in RIT; presuming failover");
621       // Process list of dead servers and regions in RIT.
622       // See HBASE-4580 for more information.
623       processDeadServersAndRecoverLostRegions(deadServers);
624     }
625 
626     if (!failover && useZKForAssignment) {
627       // Cleanup any existing ZK nodes and start watching
628       ZKAssign.deleteAllNodes(watcher);
629       ZKUtil.listChildrenAndWatchForNewChildren(this.watcher,
630         this.watcher.assignmentZNode);
631     }
632 
633     // Now we can safely claim failover cleanup completed and enable
634     // ServerShutdownHandler for further processing. The nodes (below)
635     // in transition, if any, are for regions not related to those
636     // dead servers at all, and can be done in parallel to SSH.
637     failoverCleanupDone();
638     if (!failover) {
639       // Fresh cluster startup.
640       LOG.info("Clean cluster startup. Assigning user regions");
641       assignAllUserRegions(allRegions);
642     }
643     // unassign replicas of the split parents and the merged regions
644     // the daughter replicas are opened in assignAllUserRegions if it was
645     // not already opened.
646     for (HRegionInfo h : replicasToClose) {
647       unassign(h);
648     }
649     replicasToClose.clear();
650     return failover;
651   }

655    * the region is assigned and out of transition.  Used on startup for
656    * catalog regions.
657    * @param hri Region to look for.
658    * @return True if we processed a region in transition else false if region
659    * was not up in zk in transition.
660    * @throws InterruptedException
661    * @throws KeeperException

664   boolean processRegionInTransitionAndBlockUntilAssigned(final HRegionInfo hri)
665       throws InterruptedException, KeeperException, IOException {
666     String encodedRegionName = hri.getEncodedName();

669     }

676         // server. Doesn't help to block here any more. Caller need to
677         // verify the region is actually assigned.
678         break;
679       }
680       this.regionStates.waitForUpdate(100);
681     }
682     return true;

685   /**
686    * Process failover of new master for region <code>encodedRegionName</code>
687    * up in zookeeper.
688    * @param encodedRegionName Region to process failover for.
689    * @param regionInfo If null we'll go get it from meta table.
690    * @return True if we processed <code>regionInfo</code> as a RIT.
691    * @throws KeeperException
692    * @throws IOException
693    */
694   boolean processRegionInTransition(final String encodedRegionName,
695       final HRegionInfo regionInfo) throws KeeperException, IOException {
696     // We need a lock here to ensure that we will not put the same region twice
697     // It has no reason to be a lock shared with the other operations.
698     // We can do the lock on the region only, instead of a global lock: what we want to ensure
699     // is that we don't have two threads working on the same region.
700     Lock lock = locker.acquireLock(encodedRegionName);
701     try {
702       Stat stat = new Stat();
703       byte [] data = ZKAssign.getDataAndWatch(watcher, encodedRegionName, stat);
704       if (data == null) return false;
705       RegionTransition rt;
706       try {
707         rt = RegionTransition.parseFrom(data);
708       } catch (DeserializationException e) {
709         LOG.warn("Failed parse znode data", e);
710         return false;
711       }
712       HRegionInfo hri = regionInfo;
713       if (hri == null) {
714         // The region info is not passed in. We will try to find the region
715         // from region states map/meta based on the encoded region name. But we
716         // may not be able to find it. This is valid for online merge that
717         // the region may have not been created if the merge is not completed.
718         // Therefore, it is not in meta at master recovery time.
719         hri = regionStates.getRegionInfo(rt.getRegionName());
720         EventType et = rt.getEventType();
721         if (hri == null && et != EventType.RS_ZK_REGION_MERGING
722             && et != EventType.RS_ZK_REQUEST_REGION_MERGE) {
723           LOG.warn("Couldn't find the region in recovering " + rt);
724           return false;
725         }
726       }
727 
728       // TODO: This code is tied to ZK anyway, so for now leaving it as is,
729       // will refactor when whole region assignment will be abstracted from ZK
730       BaseCoordinatedStateManager cp =
731         (BaseCoordinatedStateManager) this.server.getCoordinatedStateManager();
732       OpenRegionCoordination openRegionCoordination = cp.getOpenRegionCoordination();
733 
734       ZkOpenRegionCoordination.ZkOpenRegionDetails zkOrd =
735         new ZkOpenRegionCoordination.ZkOpenRegionDetails();
736       zkOrd.setVersion(stat.getVersion());
737       zkOrd.setServerName(cp.getServer().getServerName());
738 
739       return processRegionsInTransition(
740         rt, hri, openRegionCoordination, zkOrd);
741     } finally {
742       lock.unlock();
743     }
744   }

746   /**
747    * This call is invoked only (1) master assign meta;
748    * (2) during failover mode startup, zk assignment node processing.
749    * The locker is set in the caller. It returns true if the region
750    * is in transition for sure, false otherwise.
751    *
752    * It should be private but it is used by some test too.
753    */
754   boolean processRegionsInTransition(
755       final RegionTransition rt, final HRegionInfo regionInfo,
756       OpenRegionCoordination coordination,
757       final OpenRegionCoordination.OpenRegionDetails ord) throws KeeperException {
758     EventType et = rt.getEventType();
759     // Get ServerName.  Could not be null.
760     final ServerName sn = rt.getServerName();
761     final byte[] regionName = rt.getRegionName();
762     final String encodedName = HRegionInfo.encodeRegionName(regionName);
763     final String prettyPrintedRegionName = HRegionInfo.prettyPrint(encodedName);
764     LOG.info("Processing " + prettyPrintedRegionName + " in state: " + et);
765 
766     if (regionStates.isRegionInTransition(encodedName)
767         && (regionInfo.isMetaRegion() || !useZKForAssignment)) {
768       LOG.info("Processed region " + prettyPrintedRegionName + " in state: "
769         + et + ", does nothing since the region is already in transition "
770         + regionStates.getRegionTransitionState(encodedName));
771       // Just return
772       return true;
773     }
774     if (!serverManager.isServerOnline(sn)) {
775       // It was transitioning on a dead server, so it's closed now.
776       // Force to OFFLINE and put it in transition, but not assign it
777       // since log splitting for the dead server is not done yet.
778       LOG.debug("RIT " + encodedName + " in state=" + rt.getEventType() +
779         " was on deadserver; forcing offline");
780       if (regionStates.isRegionOnline(regionInfo)) {
781         // Meta could still show the region is assigned to the previous
782         // server. If that server is online, when we reload the meta, the
783         // region is put back to online, we need to offline it.
784         regionStates.regionOffline(regionInfo);
785         sendRegionClosedNotification(regionInfo);
786       }
787       // Put it back in transition so that SSH can re-assign it
788       regionStates.updateRegionState(regionInfo, State.OFFLINE, sn);
789 
790       if (regionInfo.isMetaRegion()) {
791         // If it's meta region, reset the meta location.
792         // So that master knows the right meta region server.
793         MetaTableLocator.setMetaLocation(watcher, sn, State.OPEN);
794       } else {
795         // No matter the previous server is online or offline,
796         // we need to reset the last region server of the region.
797         regionStates.setLastRegionServerOfRegion(sn, encodedName);
798         // Make sure we know the server is dead.
799         if (!serverManager.isServerDead(sn)) {
800           serverManager.expireServer(sn);
801         }
802       }
803       return false;
804     }
805     switch (et) {
806       case M_ZK_REGION_CLOSING:
807         // Insert into RIT & resend the query to the region server: may be the previous master
808         // died before sending the query the first time.
809         final RegionState rsClosing = regionStates.updateRegionState(rt, State.CLOSING);
810         this.executorService.submit(
811           new EventHandler(server, EventType.M_MASTER_RECOVERY) {
812             @Override
813             public void process() throws IOException {
814               ReentrantLock lock = locker.acquireLock(regionInfo.getEncodedName());
815               try {
816                 final int expectedVersion = ((ZkOpenRegionCoordination.ZkOpenRegionDetails) ord)
817                   .getVersion();
818                 unassign(regionInfo, rsClosing, expectedVersion, null, useZKForAssignment, null);
819                 if (regionStates.isRegionOffline(regionInfo)) {
820                   assign(regionInfo, true);
821                 }
822               } finally {
823                 lock.unlock();
824               }
825             }
826           });
827         break;
828 
829       case RS_ZK_REGION_CLOSED:
830       case RS_ZK_REGION_FAILED_OPEN:
831         // Region is closed, insert into RIT and handle it
832         regionStates.updateRegionState(regionInfo, State.CLOSED, sn);
833         if (!replicasToClose.contains(regionInfo)) {
834           invokeAssign(regionInfo);
835         } else {
836           offlineDisabledRegion(regionInfo);
837         }
838         break;
839 
840       case M_ZK_REGION_OFFLINE:
841         // Insert in RIT and resend to the regionserver
842         regionStates.updateRegionState(rt, State.PENDING_OPEN);
843         final RegionState rsOffline = regionStates.getRegionState(regionInfo);
844         this.executorService.submit(
845           new EventHandler(server, EventType.M_MASTER_RECOVERY) {
846             @Override
847             public void process() throws IOException {
848               ReentrantLock lock = locker.acquireLock(regionInfo.getEncodedName());
849               try {
850                 RegionPlan plan = new RegionPlan(regionInfo, null, sn);
851                 addPlan(encodedName, plan);
852                 assign(rsOffline, false, false);
853               } finally {
854                 lock.unlock();
855               }
856             }
857           });
858         break;
859 
860       case RS_ZK_REGION_OPENING:
861         regionStates.updateRegionState(rt, State.OPENING);
862         break;
863 
864       case RS_ZK_REGION_OPENED:
865         // Region is opened, insert into RIT and handle it
866         // This could be done asynchronously, we would need then to acquire the lock in the
867         //  handler.
868         regionStates.updateRegionState(rt, State.OPEN);
869         new OpenedRegionHandler(server, this, regionInfo, coordination, ord).process();
870         break;
871       case RS_ZK_REQUEST_REGION_SPLIT:
872       case RS_ZK_REGION_SPLITTING:
873       case RS_ZK_REGION_SPLIT:
874         // Splitting region should be online. We could have skipped it during
875         // user region rebuilding since we may consider the split is completed.
876         // Put it in SPLITTING state to avoid complications.
877         regionStates.regionOnline(regionInfo, sn);
878         regionStates.updateRegionState(rt, State.SPLITTING);
879         if (!handleRegionSplitting(
880             rt, encodedName, prettyPrintedRegionName, sn)) {
881           deleteSplittingNode(encodedName, sn);
882         }
883         break;
884       case RS_ZK_REQUEST_REGION_MERGE:
885       case RS_ZK_REGION_MERGING:
886       case RS_ZK_REGION_MERGED:
887         if (!handleRegionMerging(
888             rt, encodedName, prettyPrintedRegionName, sn)) {
889           deleteMergingNode(encodedName, sn);
890         }
891         break;
892       default:
893         throw new IllegalStateException("Received region in state:" + et + " is not valid.");
894     }
895     LOG.info("Processed region " + prettyPrintedRegionName + " in state "
896       + et + ", on " + (serverManager.isServerOnline(sn) ? "" : "dead ")
897       + "server: " + sn);
898     return true;
899   }

911   /**
912    * Handles various states an unassigned node can be in.
913    * <p>
914    * Method is called when a state change is suspected for an unassigned node.
915    * <p>
916    * This deals with skipped transitions (we got a CLOSED but didn't see CLOSING
917    * yet).
918    * @param rt region transition
919    * @param coordination coordination for opening region
920    * @param ord details about opening region
921    */
922   void handleRegion(final RegionTransition rt, OpenRegionCoordination coordination,
923                     OpenRegionCoordination.OpenRegionDetails ord) {
924     if (rt == null) {
925       LOG.warn("Unexpected NULL input for RegionTransition rt");
926       return;
927     }
928     final ServerName sn = rt.getServerName();
929     // Check if this is a special HBCK transition
930     if (sn.equals(HBCK_CODE_SERVERNAME)) {
931       handleHBCK(rt);
932       return;
933     }
934     final long createTime = rt.getCreateTime();
935     final byte[] regionName = rt.getRegionName();
936     String encodedName = HRegionInfo.encodeRegionName(regionName);
937     String prettyPrintedRegionName = HRegionInfo.prettyPrint(encodedName);
938     // Verify this is a known server
939     if (!serverManager.isServerOnline(sn)
940       && !ignoreStatesRSOffline.contains(rt.getEventType())) {
941       LOG.warn("Attempted to handle region transition for server but " +
942         "it is not online: " + prettyPrintedRegionName + ", " + rt);
943       return;
944     }
945 
946     RegionState regionState =
947       regionStates.getRegionState(encodedName);
948     long startTime = System.currentTimeMillis();
949     if (LOG.isDebugEnabled()) {
950       boolean lateEvent = createTime < (startTime - 15000);
951       LOG.debug("Handling " + rt.getEventType() +
952         ", server=" + sn + ", region=" +
953         (prettyPrintedRegionName == null ? "null" : prettyPrintedRegionName) +
954         (lateEvent ? ", which is more than 15 seconds late" : "") +
955         ", current_state=" + regionState);
956     }
957     // We don't do anything for this event,
958     // so separate it out, no need to lock/unlock anything
959     if (rt.getEventType() == EventType.M_ZK_REGION_OFFLINE) {
960       return;
961     }
962 
963     // We need a lock on the region as we could update it
964     Lock lock = locker.acquireLock(encodedName);
965     try {
966       RegionState latestState =
967         regionStates.getRegionState(encodedName);
968       if ((regionState == null && latestState != null)
969           || (regionState != null && latestState == null)
970           || (regionState != null && latestState != null
971             && latestState.getState() != regionState.getState())) {
972         LOG.warn("Region state changed from " + regionState + " to "
973           + latestState + ", while acquiring lock");
974       }
975       long waitedTime = System.currentTimeMillis() - startTime;
976       if (waitedTime > 5000) {
977         LOG.warn("Took " + waitedTime + "ms to acquire the lock");
978       }
979       regionState = latestState;
980       switch (rt.getEventType()) {
981       case RS_ZK_REQUEST_REGION_SPLIT:
982       case RS_ZK_REGION_SPLITTING:
983       case RS_ZK_REGION_SPLIT:
984         if (!handleRegionSplitting(
985             rt, encodedName, prettyPrintedRegionName, sn)) {
986           deleteSplittingNode(encodedName, sn);
987         }
988         break;
989 
990       case RS_ZK_REQUEST_REGION_MERGE:
991       case RS_ZK_REGION_MERGING:
992       case RS_ZK_REGION_MERGED:
993         // Merged region is a new region, we can't find it in the region states now.
994         // However, the two merging regions are not new. They should be in state for merging.
995         if (!handleRegionMerging(
996             rt, encodedName, prettyPrintedRegionName, sn)) {
997           deleteMergingNode(encodedName, sn);
998         }
999         break;
1000 
1001       case M_ZK_REGION_CLOSING:
1002         // Should see CLOSING after we have asked it to CLOSE or additional
1003         // times after already being in state of CLOSING
1004         if (regionState == null
1005             || !regionState.isPendingCloseOrClosingOnServer(sn)) {
1006           LOG.warn("Received CLOSING for " + prettyPrintedRegionName
1007             + " from " + sn + " but the region isn't PENDING_CLOSE/CLOSING here: "
1008             + regionStates.getRegionState(encodedName));
1009           return;
1010         }
1011         // Transition to CLOSING (or update stamp if already CLOSING)
1012         regionStates.updateRegionState(rt, State.CLOSING);
1013         break;
1014 
1015       case RS_ZK_REGION_CLOSED:
1016         // Should see CLOSED after CLOSING but possible after PENDING_CLOSE
1017         if (regionState == null
1018             || !regionState.isPendingCloseOrClosingOnServer(sn)) {
1019           LOG.warn("Received CLOSED for " + prettyPrintedRegionName
1020             + " from " + sn + " but the region isn't PENDING_CLOSE/CLOSING here: "
1021             + regionStates.getRegionState(encodedName));
1022           return;
1023         }
1024         // Handle CLOSED by assigning elsewhere or stopping if a disable
1025         // If we got here all is good.  Need to update RegionState -- else
1026         // what follows will fail because not in expected state.
1027         new ClosedRegionHandler(server, this, regionState.getRegion()).process();
1028         updateClosedRegionHandlerTracker(regionState.getRegion());
1029         break;
1030 
1031         case RS_ZK_REGION_FAILED_OPEN:
1032           if (regionState == null
1033               || !regionState.isPendingOpenOrOpeningOnServer(sn)) {
1034             LOG.warn("Received FAILED_OPEN for " + prettyPrintedRegionName
1035               + " from " + sn + " but the region isn't PENDING_OPEN/OPENING here: "
1036               + regionStates.getRegionState(encodedName));
1037             return;
1038           }
1039           AtomicInteger failedOpenCount = failedOpenTracker.get(encodedName);
1040           if (failedOpenCount == null) {
1041             failedOpenCount = new AtomicInteger();
1042             // No need to use putIfAbsent, or extra synchronization since
1043             // this whole handleRegion block is locked on the encoded region
1044             // name, and failedOpenTracker is updated only in this block
1045             failedOpenTracker.put(encodedName, failedOpenCount);
1046           }
1047           if (failedOpenCount.incrementAndGet() >= maximumAttempts) {
1048             regionStates.updateRegionState(rt, State.FAILED_OPEN);
1049             // remove the tracking info to save memory, also reset
1050             // the count for next open initiative
1051             failedOpenTracker.remove(encodedName);
1052           } else {
1053             // Handle this the same as if it were opened and then closed.
1054             regionState = regionStates.updateRegionState(rt, State.CLOSED);
1055             if (regionState != null) {
1056               // When there are more than one region server a new RS is selected as the
1057               // destination and the same is updated in the regionplan. (HBASE-5546)
1058               try {
1059                 getRegionPlan(regionState.getRegion(), sn, true);
1060                 new ClosedRegionHandler(server, this, regionState.getRegion()).process();
1061               } catch (HBaseIOException e) {
1062                 LOG.warn("Failed to get region plan", e);
1063               }
1064             }
1065           }
1066           break;
1067 
1068         case RS_ZK_REGION_OPENING:
1069           // Should see OPENING after we have asked it to OPEN or additional
1070           // times after already being in state of OPENING
1071           if (regionState == null
1072               || !regionState.isPendingOpenOrOpeningOnServer(sn)) {
1073             LOG.warn("Received OPENING for " + prettyPrintedRegionName
1074               + " from " + sn + " but the region isn't PENDING_OPEN/OPENING here: "
1075               + regionStates.getRegionState(encodedName));
1076             return;
1077           }
1078           // Transition to OPENING (or update stamp if already OPENING)
1079           regionStates.updateRegionState(rt, State.OPENING);
1080           break;
1081 
1082         case RS_ZK_REGION_OPENED:
1083           // Should see OPENED after OPENING but possible after PENDING_OPEN.
1084           if (regionState == null
1085               || !regionState.isPendingOpenOrOpeningOnServer(sn)) {
1086             LOG.warn("Received OPENED for " + prettyPrintedRegionName
1087               + " from " + sn + " but the region isn't PENDING_OPEN/OPENING here: "
1088               + regionStates.getRegionState(encodedName));
1089 
1090             if (regionState != null) {
1091               // Close it without updating the internal region states,
1092               // so as not to create double assignments in unlucky scenarios
1093               // mentioned in OpenRegionHandler#process
1094               unassign(regionState.getRegion(), null, -1, null, false, sn);
1095             }
1096             return;
1097           }
1098           // Handle OPENED by removing from transition and deleted zk node
1099           regionState = 
1100               regionStates.transitionOpenFromPendingOpenOrOpeningOnServer(rt,regionState, sn);
1101           if (regionState != null) {
1102             failedOpenTracker.remove(encodedName); // reset the count, if any
1103             new OpenedRegionHandler(
1104               server, this, regionState.getRegion(), coordination, ord).process();
1105             updateOpenedRegionHandlerTracker(regionState.getRegion());
1106           }
1107           break;
1108 
1109         default:
1110           throw new IllegalStateException("Received event is not valid.");
1111       }
1112     } finally {
1113       lock.unlock();
1114     }
1115   }

1251   // We  don't want to have two events on the same region managed simultaneously.
1252   // For this reason, we need to wait if an event on the same region is currently in progress.
1253   // So we track the region names of the events in progress, and we keep a waiting list.
1254   private final Set<String> regionsInProgress = new HashSet<String>();
1255   // In a LinkedHashMultimap, the put order is kept when we retrieve the collection back. We need
1256   //  this as we want the events to be managed in the same order as we received them.
1257   private final LinkedHashMultimap <String, RegionRunnable>
1258       zkEventWorkerWaitingList = LinkedHashMultimap.create();

1260   /**
1261    * A specific runnable that works only on a region.
1262    */

1264     /**
1265      * @return - the name of the region it works on.
1266      */

1270   /**
1271    * Submit a task, ensuring that there is only one task at a time that working on a given region.
1272    * Order is respected.
1273    */
1274   protected void zkEventWorkersSubmit(final RegionRunnable regRunnable) {
1275 
1276     synchronized (regionsInProgress) {
1277       // If we're there is already a task with this region, we add it to the
1278       //  waiting list and return.
1279       if (regionsInProgress.contains(regRunnable.getRegionName())) {
1280         synchronized (zkEventWorkerWaitingList){
1281           zkEventWorkerWaitingList.put(regRunnable.getRegionName(), regRunnable);
1282         }
1283         return;
1284       }
1285 
1286       // No event in progress on this region => we can submit a new task immediately.
1287       regionsInProgress.add(regRunnable.getRegionName());
1288       zkEventWorkers.submit(new Runnable() {
1289         @Override
1290         public void run() {
1291           try {
1292             regRunnable.run();
1293           } finally {
1294             // now that we have finished, let's see if there is an event for the same region in the
1295             //  waiting list. If it's the case, we can now submit it to the pool.
1296             synchronized (regionsInProgress) {
1297               regionsInProgress.remove(regRunnable.getRegionName());
1298               synchronized (zkEventWorkerWaitingList) {
1299                 java.util.Set<RegionRunnable> waiting = zkEventWorkerWaitingList.get(
1300                     regRunnable.getRegionName());
1301                 if (!waiting.isEmpty()) {
1302                   // We want the first object only. The only way to get it is through an iterator.
1303                   RegionRunnable toSubmit = waiting.iterator().next();
1304                   zkEventWorkerWaitingList.remove(toSubmit.getRegionName(), toSubmit);
1305                   zkEventWorkersSubmit(toSubmit);
1306                 }
1307               }
1308             }
1309           }
1310         }
1311       });
1312     }
1313   }

1326         public void run() {
1327           Lock lock = locker.acquireLock(regionName);
1328           try {

1330             if (rs == null) {
1331               rs = regionStates.getRegionState(regionName);
1332               if (rs == null || !rs.isMergingNew()) {
1333                 // MergingNew is an offline state

1335               }
1336             }
1337

1341

1343                 ZooKeeperProtos.Table.State.DISABLED, ZooKeeperProtos.Table.State.DISABLING);
1344

1346             if (serverManager.isServerOnline(serverName)) {

1348                 synchronized (regionStates) {
1349                   regionOnline(regionInfo, serverName);

1356                       LOG.warn("Split daughter region not in transition " + hri_a);

1358                     if (!regionStates.isRegionInTransition(hri_b.getEncodedName())) {
1359                       LOG.warn("Split daughter region not in transition" + hri_b);

1367                     LOG.info("Opened " + regionNameStr
1368                         + "but this table is disabled, triggering close of region");
1369                     unassign(regionInfo);
1370                   }

1372               } else if (rs.isMergingNew()) {
1373                 synchronized (regionStates) {
1374                   String p = regionInfo.getEncodedName();

1376                   if (regions != null) {
1377                     onlineMergingRegion(disabled, regions.getFirst(), serverName);
1378                     onlineMergingRegion(disabled, regions.getSecond(), serverName);
1379                   }
1380                 }
1381               }
1382             }
1383           } finally {
1384             lock.unlock();
1385           }

1594   /**
1595    * Bulk assign regions to <code>destination</code>.
1596    * @param destination
1597    * @param regions Regions to assign.
1598    * @return true if successful
1599    */
1600   boolean assign(final ServerName destination, final List<HRegionInfo> regions)
1601     throws InterruptedException {
1602     long startTime = EnvironmentEdgeManager.currentTime();
1603     try {
1604       int regionCount = regions.size();
1605       if (regionCount == 0) {
1606         return true;
1607       }
1608       LOG.info("Assigning " + regionCount + " region(s) to " + destination.toString());
1609       Set<String> encodedNames = new HashSet<String>(regionCount);
1610       for (HRegionInfo region : regions) {
1611         encodedNames.add(region.getEncodedName());
1612       }
1613 
1614       List<HRegionInfo> failedToOpenRegions = new ArrayList<HRegionInfo>();
1615       Map<String, Lock> locks = locker.acquireLocks(encodedNames);
1616       try {
1617         AtomicInteger counter = new AtomicInteger(0);
1618         Map<String, Integer> offlineNodesVersions = new ConcurrentHashMap<String, Integer>();
1619         OfflineCallback cb = new OfflineCallback(
1620           watcher, destination, counter, offlineNodesVersions);
1621         Map<String, RegionPlan> plans = new HashMap<String, RegionPlan>(regions.size());
1622         List<RegionState> states = new ArrayList<RegionState>(regions.size());
1623         for (HRegionInfo region : regions) {
1624           String encodedName = region.getEncodedName();
1625           if (!isDisabledorDisablingRegionInRIT(region)) {
1626             RegionState state = forceRegionStateToOffline(region, false);
1627             boolean onDeadServer = false;
1628             if (state != null) {
1629               if (regionStates.wasRegionOnDeadServer(encodedName)) {
1630                 LOG.info("Skip assigning " + region.getRegionNameAsString()
1631                   + ", it's host " + regionStates.getLastRegionServerOfRegion(encodedName)
1632                   + " is dead but not processed yet");
1633                 onDeadServer = true;
1634               } else if (!useZKForAssignment
1635                   || asyncSetOfflineInZooKeeper(state, cb, destination)) {
1636                 RegionPlan plan = new RegionPlan(region, state.getServerName(), destination);
1637                 plans.put(encodedName, plan);
1638                 states.add(state);
1639                 continue;
1640               }
1641             }
1642             // Reassign if the region wasn't on a dead server
1643             if (!onDeadServer) {
1644               LOG.info("failed to force region state to offline or "
1645                 + "failed to set it offline in ZK, will reassign later: " + region);
1646               failedToOpenRegions.add(region); // assign individually later
1647             }
1648           }
1649           // Release the lock, this region is excluded from bulk assign because
1650           // we can't update its state, or set its znode to offline.
1651           Lock lock = locks.remove(encodedName);
1652           lock.unlock();
1653         }
1654 
1655         if (useZKForAssignment) {
1656           // Wait until all unassigned nodes have been put up and watchers set.
1657           int total = states.size();
1658           for (int oldCounter = 0; !server.isStopped();) {
1659             int count = counter.get();
1660             if (oldCounter != count) {
1661               LOG.debug(destination.toString() + " unassigned znodes=" + count +
1662                 " of total=" + total + "; oldCounter=" + oldCounter);
1663               oldCounter = count;
1664             }
1665             if (count >= total) break;
1666             Thread.sleep(5);
1667           }
1668         }
1669 
1670         if (server.isStopped()) {
1671           return false;
1672         }
1673 
1674         // Add region plans, so we can updateTimers when one region is opened so
1675         // that unnecessary timeout on RIT is reduced.
1676         this.addPlans(plans);
1677 
1678         List<Triple<HRegionInfo, Integer, List<ServerName>>> regionOpenInfos =
1679           new ArrayList<Triple<HRegionInfo, Integer, List<ServerName>>>(states.size());
1680         for (RegionState state: states) {
1681           HRegionInfo region = state.getRegion();
1682           String encodedRegionName = region.getEncodedName();
1683           Integer nodeVersion = offlineNodesVersions.get(encodedRegionName);
1684           if (useZKForAssignment && (nodeVersion == null || nodeVersion == -1)) {
1685             LOG.warn("failed to offline in zookeeper: " + region);
1686             failedToOpenRegions.add(region); // assign individually later
1687             Lock lock = locks.remove(encodedRegionName);
1688             lock.unlock();
1689           } else {
1690             regionStates.updateRegionState(
1691               region, State.PENDING_OPEN, destination);
1692             List<ServerName> favoredNodes = ServerName.EMPTY_SERVER_LIST;
1693             if (this.shouldAssignRegionsWithFavoredNodes) {
1694               favoredNodes = ((FavoredNodeLoadBalancer)this.balancer).getFavoredNodes(region);
1695             }
1696             regionOpenInfos.add(new Triple<HRegionInfo, Integer,  List<ServerName>>(
1697               region, nodeVersion, favoredNodes));
1698           }
1699         }
1700 
1701         // Move on to open regions.
1702         try {
1703           // Send OPEN RPC. If it fails on a IOE or RemoteException,
1704           // regions will be assigned individually.
1705           long maxWaitTime = System.currentTimeMillis() +
1706             this.server.getConfiguration().
1707               getLong("hbase.regionserver.rpc.startup.waittime", 60000);
1708           for (int i = 1; i <= maximumAttempts && !server.isStopped(); i++) {
1709             try {
1710               // regionOpenInfos is empty if all regions are in failedToOpenRegions list
1711               if (regionOpenInfos.isEmpty()) {
1712                 break;
1713               }
1714               List<RegionOpeningState> regionOpeningStateList = serverManager
1715                 .sendRegionOpen(destination, regionOpenInfos);
1716               if (regionOpeningStateList == null) {
1717                 // Failed getting RPC connection to this server
1718                 return false;
1719               }
1720               for (int k = 0, n = regionOpeningStateList.size(); k < n; k++) {
1721                 RegionOpeningState openingState = regionOpeningStateList.get(k);
1722                 if (openingState != RegionOpeningState.OPENED) {
1723                   HRegionInfo region = regionOpenInfos.get(k).getFirst();
1724                   if (openingState == RegionOpeningState.ALREADY_OPENED) {
1725                     processAlreadyOpenedRegion(region, destination);
1726                   } else if (openingState == RegionOpeningState.FAILED_OPENING) {
1727                     // Failed opening this region, reassign it later
1728                     failedToOpenRegions.add(region);
1729                   } else {
1730                     LOG.warn("THIS SHOULD NOT HAPPEN: unknown opening state "
1731                       + openingState + " in assigning region " + region);
1732                   }
1733                 }
1734               }
1735               break;
1736             } catch (IOException e) {
1737               if (e instanceof RemoteException) {
1738                 e = ((RemoteException)e).unwrapRemoteException();
1739               }
1740               if (e instanceof RegionServerStoppedException) {
1741                 LOG.warn("The region server was shut down, ", e);
1742                 // No need to retry, the region server is a goner.
1743                 return false;
1744               } else if (e instanceof ServerNotRunningYetException) {
1745                 long now = System.currentTimeMillis();
1746                 if (now < maxWaitTime) {
1747                   LOG.debug("Server is not yet up; waiting up to " +
1748                     (maxWaitTime - now) + "ms", e);
1749                   Thread.sleep(100);
1750                   i--; // reset the try count
1751                   continue;
1752                 }
1753               } else if (e instanceof java.net.SocketTimeoutException
1754                   && this.serverManager.isServerOnline(destination)) {
1755                 // In case socket is timed out and the region server is still online,
1756                 // the openRegion RPC could have been accepted by the server and
1757                 // just the response didn't go through.  So we will retry to
1758                 // open the region on the same server.
1759                 if (LOG.isDebugEnabled()) {
1760                   LOG.debug("Bulk assigner openRegion() to " + destination
1761                     + " has timed out, but the regions might"
1762                     + " already be opened on it.", e);
1763                 }
1764                 // wait and reset the re-try count, server might be just busy.
1765                 Thread.sleep(100);
1766                 i--;
1767                 continue;
1768               }
1769               throw e;
1770             }
1771           }
1772         } catch (IOException e) {
1773           // Can be a socket timeout, EOF, NoRouteToHost, etc
1774           LOG.info("Unable to communicate with " + destination
1775             + " in order to assign regions, ", e);
1776           return false;
1777         }
1778       } finally {
1779         for (Lock lock : locks.values()) {
1780           lock.unlock();
1781         }
1782       }
1783 
1784       if (!failedToOpenRegions.isEmpty()) {
1785         for (HRegionInfo region : failedToOpenRegions) {
1786           if (!regionStates.isRegionOnline(region)) {
1787             invokeAssign(region);
1788           }
1789         }
1790       }
1791       LOG.debug("Bulk assigning done for " + destination);
1792       return true;
1793     } finally {

1798   /**
1799    * Send CLOSE RPC if the server is online, otherwise, offline the region.
1800    *
1801    * The RPC will be sent only to the region sever found in the region state
1802    * if it is passed in, otherwise, to the src server specified. If region
1803    * state is not specified, we don't update region state at all, instead
1804    * we just send the RPC call. This is useful for some cleanup without
1805    * messing around the region states (see handleRegion, on region opened
1806    * on an unexpected server scenario, for an example)
1807    */
1808   private void unassign(final HRegionInfo region,
1809       final RegionState state, final int versionOfClosingNode,
1810       final ServerName dest, final boolean transitionInZK,
1811       final ServerName src) {
1812     ServerName server = src;
1813     if (state != null) {
1814       server = state.getServerName();
1815     }
1816     long maxWaitTime = -1;
1817     for (int i = 1; i <= this.maximumAttempts; i++) {
1818       if (this.server.isStopped() || this.server.isAborted()) {
1819         LOG.debug("Server stopped/aborted; skipping unassign of " + region);
1820         return;
1821       }
1822       // ClosedRegionhandler can remove the server from this.regions
1823       if (!serverManager.isServerOnline(server)) {
1824         LOG.debug("Offline " + region.getRegionNameAsString()
1825           + ", no need to unassign since it's on a dead server: " + server);
1826         if (transitionInZK) {
1827           // delete the node. if no node exists need not bother.
1828           deleteClosingOrClosedNode(region, server);
1829         }
1830         if (state != null) {
1831           regionOffline(region);
1832         }
1833         return;
1834       }
1835       try {
1836         // Send CLOSE RPC
1837         if (serverManager.sendRegionClose(server, region,
1838           versionOfClosingNode, dest, transitionInZK)) {
1839           LOG.debug("Sent CLOSE to " + server + " for region " +
1840             region.getRegionNameAsString());
1841           if (useZKForAssignment && !transitionInZK && state != null) {
1842             // Retry to make sure the region is
1843             // closed so as to avoid double assignment.
1844             unassign(region, state, versionOfClosingNode,
1845               dest, transitionInZK, src);
1846           }
1847           return;
1848         }
1849         // This never happens. Currently regionserver close always return true.
1850         // Todo; this can now happen (0.96) if there is an exception in a coprocessor
1851         LOG.warn("Server " + server + " region CLOSE RPC returned false for " +
1852           region.getRegionNameAsString());
1853       } catch (Throwable t) {
1854         if (t instanceof RemoteException) {
1855           t = ((RemoteException)t).unwrapRemoteException();
1856         }
1857         boolean logRetries = true;
1858         if (t instanceof NotServingRegionException
1859             || t instanceof RegionServerStoppedException
1860             || t instanceof ServerNotRunningYetException) {
1861           LOG.debug("Offline " + region.getRegionNameAsString()
1862             + ", it's not any more on " + server, t);
1863           if (transitionInZK) {
1864             deleteClosingOrClosedNode(region, server);
1865           }
1866           if (state != null) {
1867             regionOffline(region);
1868           }
1869           return;
1870         } else if ((t instanceof FailedServerException) || (state != null &&
1871             t instanceof RegionAlreadyInTransitionException)) {
1872           long sleepTime = 0;
1873           Configuration conf = this.server.getConfiguration();
1874           if(t instanceof FailedServerException) {
1875             sleepTime = 1 + conf.getInt(RpcClient.FAILED_SERVER_EXPIRY_KEY,
1876                   RpcClient.FAILED_SERVER_EXPIRY_DEFAULT);
1877           } else {
1878             // RS is already processing this region, only need to update the timestamp
1879             LOG.debug("update " + state + " the timestamp.");
1880             state.updateTimestampToNow();
1881             if (maxWaitTime < 0) {
1882               maxWaitTime =
1883                   EnvironmentEdgeManager.currentTime()
1884                       + conf.getLong(ALREADY_IN_TRANSITION_WAITTIME,
1885                         DEFAULT_ALREADY_IN_TRANSITION_WAITTIME);
1886             }
1887             long now = EnvironmentEdgeManager.currentTime();
1888             if (now < maxWaitTime) {
1889               LOG.debug("Region is already in transition; "
1890                 + "waiting up to " + (maxWaitTime - now) + "ms", t);
1891               sleepTime = 100;
1892               i--; // reset the try count
1893               logRetries = false;
1894             }
1895           }
1896           try {
1897             if (sleepTime > 0) {
1898               Thread.sleep(sleepTime);
1899             }
1900           } catch (InterruptedException ie) {
1901             LOG.warn("Failed to unassign "
1902               + region.getRegionNameAsString() + " since interrupted", ie);
1903             Thread.currentThread().interrupt();
1904             if (state != null) {
1905               regionStates.updateRegionState(region, State.FAILED_CLOSE);
1906             }
1907             return;
1908           }
1909         }
1910 
1911         if (logRetries) {
1912           LOG.info("Server " + server + " returned " + t + " for "
1913             + region.getRegionNameAsString() + ", try=" + i
1914             + " of " + this.maximumAttempts, t);
1915           // Presume retry or server will expire.
1916         }
1917       }
1918     }
1919     // Run out of attempts
1920     if (state != null) {
1921       regionStates.updateRegionState(region, State.FAILED_CLOSE);
1922     }
1923   }

1925   /**
1926    * Set region to OFFLINE unless it is opening and forceNewPlan is false.
1927    */
1928   private RegionState forceRegionStateToOffline(
1929       final HRegionInfo region, final boolean forceNewPlan) {
1930     RegionState state = regionStates.getRegionState(region);
1931     if (state == null) {
1932       LOG.warn("Assigning a region not in region states: " + region);
1933       state = regionStates.createRegionState(region);
1934     }
1935 
1936     ServerName sn = state.getServerName();
1937     if (forceNewPlan && LOG.isDebugEnabled()) {
1938       LOG.debug("Force region state offline " + state);
1939     }
1940 
1941     switch (state.getState()) {
1942     case OPEN:
1943     case OPENING:
1944     case PENDING_OPEN:
1945     case CLOSING:
1946     case PENDING_CLOSE:
1947       if (!forceNewPlan) {
1948         LOG.debug("Skip assigning " +
1949           region + ", it is already " + state);
1950         return null;
1951       }
1952     case FAILED_CLOSE:
1953     case FAILED_OPEN:
1954       unassign(region, state, -1, null, false, null);
1955       state = regionStates.getRegionState(region);
1956       if (state.isFailedClose()) {
1957         // If we can't close the region, we can't re-assign
1958         // it so as to avoid possible double assignment/data loss.
1959         LOG.info("Skip assigning " +
1960           region + ", we couldn't close it: " + state);
1961         return null;
1962       }
1963     case OFFLINE:
1964       // This region could have been open on this server
1965       // for a while. If the server is dead and not processed
1966       // yet, we can move on only if the meta shows the
1967       // region is not on this server actually, or on a server
1968       // not dead, or dead and processed already.
1969       // In case not using ZK, we don't need this check because
1970       // we have the latest info in memory, and the caller
1971       // will do another round checking any way.
1972       if (useZKForAssignment
1973           && regionStates.isServerDeadAndNotProcessed(sn)
1974           && wasRegionOnDeadServerByMeta(region, sn)) {
1975         if (!regionStates.isRegionInTransition(region)) {
1976           LOG.info("Updating the state to " + State.OFFLINE + " to allow to be reassigned by SSH");
1977           regionStates.updateRegionState(region, State.OFFLINE);
1978         }
1979         LOG.info("Skip assigning " + region.getRegionNameAsString()
1980             + ", it is on a dead but not processed yet server: " + sn);
1981         return null;
1982       }
1983     case CLOSED:
1984       break;
1985     default:
1986       LOG.error("Trying to assign region " + region
1987         + ", which is " + state);
1988       return null;
1989     }
1990     return state;
1991   }

2025    * @param state

2030       final boolean setOfflineInZK, final boolean forceNewPlan) {
2031     long startTime = EnvironmentEdgeManager.currentTime();
2032     try {
2033       Configuration conf = server.getConfiguration();

2035       int versionOfOfflineNode = -1;

2058

2063

2081               return;

2085             // try to set to ENABLED directly then client API may think table is enabled.
2086             // When we have a case such as all the regions are added directly into hbase:meta and we call

2091               ZooKeeperProtos.Table.State.ENABLED, ZooKeeperProtos.Table.State.ENABLING)) {

2094             }
2095           }
2096         }
2097         if (setOfflineInZK && versionOfOfflineNode == -1) {
2098           LOG.info("Unable to set offline in ZooKeeper to assign " + region);
2099           // Setting offline in ZK must have been failed due to ZK racing or some
2100           // exception which may make the server to abort. If it is ZK racing,
2101           // we should retry since we already reset the region state,
2102           // existing (re)assignment will fail anyway.
2103           if (!server.isAborted()) {
2104             continue;
2105           }
2106         }
2107         LOG.info("Assigning " + region.getRegionNameAsString() +
2108             " to " + plan.getDestination().toString());

2111           State.PENDING_OPEN, plan.getDestination());

2114         final String assignMsg = "Failed assignment of " + region.getRegionNameAsString() +

2117           List<ServerName> favoredNodes = ServerName.EMPTY_SERVER_LIST;
2118           if (this.shouldAssignRegionsWithFavoredNodes) {
2119             favoredNodes = ((FavoredNodeLoadBalancer)this.balancer).getFavoredNodes(region);
2120           }

2122               plan.getDestination(), region, versionOfOfflineNode, favoredNodes);
2123 
2124           if (regionOpenState == RegionOpeningState.FAILED_OPENING) {
2125             // Failed opening this region, looping again on a new server.
2126             needNewPlan = true;

2151           // the openRegion RPC could have been accepted by the server and
2152           // just the response didn't go through.  So we will retry to
2153           // open the region on the same server to avoid possible

2157 
2158

2162 
2163             if (maxWaitTime < 0) {
2164               if (t instanceof RegionAlreadyInTransitionException) {
2165                 maxWaitTime = EnvironmentEdgeManager.currentTime()
2166                   + this.server.getConfiguration().getLong(ALREADY_IN_TRANSITION_WAITTIME,
2167                     DEFAULT_ALREADY_IN_TRANSITION_WAITTIME);
2168               } else {
2169                 maxWaitTime = EnvironmentEdgeManager.currentTime()
2170                   + this.server.getConfiguration().getLong(
2171                     "hbase.regionserver.rpc.startup.waittime", 60000);
2172               }
2173             }
2174             try {
2175               needNewPlan = false;
2176               long now = EnvironmentEdgeManager.currentTime();
2177               if (now < maxWaitTime) {
2178                 LOG.debug("Server is not yet up or region is already in transition; "
2179                   + "waiting up to " + (maxWaitTime - now) + "ms", t);
2180                 Thread.sleep(100);
2181                 i--; // reset the try count
2182               } else if (!(t instanceof RegionAlreadyInTransitionException)) {
2183                 LOG.debug("Server is not up for a while; try a new one", t);
2184                 needNewPlan = true;
2185               }
2186             } catch (InterruptedException ie) {
2187               LOG.warn("Failed to assign "
2188                   + region.getRegionNameAsString() + " since interrupted", ie);
2189               regionStates.updateRegionState(region, State.FAILED_OPEN);
2190               Thread.currentThread().interrupt();
2191               return;
2192             }
2193           } else if (retry) {
2194             needNewPlan = false;
2195             i--; // we want to retry as many times as needed as long as the RS is not dead.
2196             LOG.warn(assignMsg + ", trying to assign to the same region server due ", t);
2197           } else {
2198             needNewPlan = true;
2199             LOG.warn(assignMsg + ", trying to assign elsewhere instead;" +
2200                 " try=" + i + " of " + this.maximumAttempts, t);
2201           }
2202         }
2203 
2204         if (i == this.maximumAttempts) {
2205           // For meta region, we have to keep retrying until succeeding
2206           if (region.isMetaRegion()) {
2207             i = 0; // re-set attempt count to 0 for at least 1 retry
2208             LOG.warn(assignMsg +
2209                 ", trying to assign a hbase:meta region reached to maximumAttempts (" +
2210                 this.maximumAttempts + ").  Reset attempt counts and continue retrying.");
2211             waitForRetryingMetaAssignment();
2212           }
2213           else {
2214             // Don't reset the region state or get a new plan any more.
2215             // This is the last try.
2216             continue;
2217           }
2218         }
2219 
2220         // If region opened on destination of present plan, reassigning to new
2221         // RS may cause double assignments. In case of RegionAlreadyInTransitionException
2222         // reassigning to same RS.
2223         if (needNewPlan) {
2224           // Force a new plan and reassign. Will return null if no servers.
2225           // The new plan could be the same as the existing plan since we don't
2226           // exclude the server of the original plan, which should not be
2227           // excluded since it could be the only server up now.
2228           RegionPlan newPlan = null;
2229           try {
2230             newPlan = getRegionPlan(region, true);
2231           } catch (HBaseIOException e) {
2232             LOG.warn("Failed to get region plan", e);
2233           }
2234           if (newPlan == null) {
2235             regionStates.updateRegionState(region, State.FAILED_OPEN);
2236             LOG.warn("Unable to find a viable location to assign region " +
2237                 region.getRegionNameAsString());
2238             return;
2239           }
2240 
2241           if (plan != newPlan && !plan.getDestination().equals(newPlan.getDestination())) {
2242             // Clean out plan we failed execute and one that doesn't look like it'll
2243             // succeed anyways; we need a new plan!
2244             // Transition back to OFFLINE
2245             currentState = regionStates.updateRegionState(region, State.OFFLINE);
2246             versionOfOfflineNode = -1;
2247             plan = newPlan;
2248           } else if(plan.getDestination().equals(newPlan.getDestination()) &&
2249               previousException instanceof FailedServerException) {
2250             try {
2251               LOG.info("Trying to re-assign " + region.getRegionNameAsString() +
2252                 " to the same failed server.");
2253               Thread.sleep(1 + conf.getInt(RpcClient.FAILED_SERVER_EXPIRY_KEY,
2254                 RpcClient.FAILED_SERVER_EXPIRY_DEFAULT));
2255             } catch (InterruptedException ie) {
2256               LOG.warn("Failed to assign "
2257                   + region.getRegionNameAsString() + " since interrupted", ie);
2258               regionStates.updateRegionState(region, State.FAILED_OPEN);
2259               Thread.currentThread().interrupt();
2260               return;
2261             }
2262           }
2263         }
2264       }
2265       // Run out of attempts
2266       regionStates.updateRegionState(region, State.FAILED_OPEN);
2267     } finally {
2268       metricsAssignmentManager.updateAssignmentTime(EnvironmentEdgeManager.currentTime() - startTime);
2269     }
2270   }

2350       serverManager.createDestinationServersList(serverToExclude);

2439   /**
2440    * Unassigns the specified region.
2441    * <p>
2442    * Updates the RegionState and sends the CLOSE RPC unless region is being
2443    * split by regionserver; then the unassign fails (silently) because we
2444    * presume the region being unassigned no longer exists (its been split out
2445    * of existence). TODO: What to do if split fails and is rolled back and
2446    * parent is revivified?
2447    * <p>
2448    * If a RegionPlan is already set, it will remain.
2449    *
2450    * @param region server to be unassigned
2451    * @param force if region should be closed even if already closing
2452    */
2453   public void unassign(HRegionInfo region, boolean force, ServerName dest) {
2454     // TODO: Method needs refactoring.  Ugly buried returns throughout.  Beware!
2455     LOG.debug("Starting unassign of " + region.getRegionNameAsString()
2456       + " (offlining), current state: " + regionStates.getRegionState(region));
2457 
2458     String encodedName = region.getEncodedName();
2459     // Grab the state of this region and synchronize on it
2460     int versionOfClosingNode = -1;
2461     // We need a lock here as we're going to do a put later and we don't want multiple states
2462     //  creation
2463     ReentrantLock lock = locker.acquireLock(encodedName);
2464     RegionState state = regionStates.getRegionTransitionState(encodedName);
2465     boolean reassign = true;
2466     try {
2467       if (state == null) {
2468         // Region is not in transition.
2469         // We can unassign it only if it's not SPLIT/MERGED.
2470         state = regionStates.getRegionState(encodedName);
2471         if (state != null && state.isUnassignable()) {
2472           LOG.info("Attempting to unassign " + state + ", ignored");
2473           // Offline region will be reassigned below
2474           return;
2475         }
2476         // Create the znode in CLOSING state
2477         try {
2478           if (state == null || state.getServerName() == null) {
2479             // We don't know where the region is, offline it.
2480             // No need to send CLOSE RPC
2481             LOG.warn("Attempting to unassign a region not in RegionStates"
2482               + region.getRegionNameAsString() + ", offlined");
2483             regionOffline(region);
2484             return;
2485           }
2486           if (useZKForAssignment) {
2487             versionOfClosingNode = ZKAssign.createNodeClosing(
2488               watcher, region, state.getServerName());
2489             if (versionOfClosingNode == -1) {
2490               LOG.info("Attempting to unassign " +
2491                 region.getRegionNameAsString() + " but ZK closing node "
2492                 + "can't be created.");
2493               reassign = false; // not unassigned at all
2494               return;
2495             }
2496           }
2497         } catch (KeeperException e) {
2498           if (e instanceof NodeExistsException) {
2499             // Handle race between master initiated close and regionserver
2500             // orchestrated splitting. See if existing node is in a
2501             // SPLITTING or SPLIT state.  If so, the regionserver started
2502             // an op on node before we could get our CLOSING in.  Deal.
2503             NodeExistsException nee = (NodeExistsException)e;
2504             String path = nee.getPath();
2505             try {
2506               if (isSplitOrSplittingOrMergedOrMerging(path)) {
2507                 LOG.debug(path + " is SPLIT or SPLITTING or MERGED or MERGING; " +
2508                   "skipping unassign because region no longer exists -- its split or merge");
2509                 reassign = false; // no need to reassign for split/merged region
2510                 return;
2511               }
2512             } catch (KeeperException.NoNodeException ke) {
2513               LOG.warn("Failed getData on SPLITTING/SPLIT at " + path +
2514                 "; presuming split and that the region to unassign, " +
2515                 encodedName + ", no longer exists -- confirm", ke);
2516               return;
2517             } catch (KeeperException ke) {
2518               LOG.error("Unexpected zk state", ke);
2519             } catch (DeserializationException de) {
2520               LOG.error("Failed parse", de);
2521             }
2522           }
2523           // If we get here, don't understand whats going on -- abort.
2524           server.abort("Unexpected ZK exception creating node CLOSING", e);
2525           reassign = false; // heading out already
2526           return;
2527         }
2528         state = regionStates.updateRegionState(region, State.PENDING_CLOSE);
2529       } else if (state.isFailedOpen()) {
2530         // The region is not open yet
2531         regionOffline(region);
2532         return;
2533       } else if (force && state.isPendingCloseOrClosing()) {
2534         LOG.debug("Attempting to unassign " + region.getRegionNameAsString() +
2535           " which is already " + state.getState()  +
2536           " but forcing to send a CLOSE RPC again ");
2537         if (state.isFailedClose()) {
2538           state = regionStates.updateRegionState(region, State.PENDING_CLOSE);
2539         }
2540         state.updateTimestampToNow();
2541       } else {
2542         LOG.debug("Attempting to unassign " +
2543           region.getRegionNameAsString() + " but it is " +
2544           "already in transition (" + state.getState() + ", force=" + force + ")");
2545         return;
2546       }
2547 
2548       unassign(region, state, versionOfClosingNode, dest, useZKForAssignment, null);
2549     } finally {
2550       lock.unlock();
2551 
2552       // Region is expected to be reassigned afterwards
2553       if (!replicasToClose.contains(region) && reassign && regionStates.isRegionOffline(region)) {
2554         assign(region, true);
2555       }
2556     }
2557   }

2837   /**
2838    * Rebuild the list of user regions and assignment information.
2839    * <p>
2840    * Returns a set of servers that are not found to be online that hosted
2841    * some regions.
2842    * @return set of servers not online that hosted some regions per meta
2843    * @throws IOException
2844    */
2845   Set<ServerName> rebuildUserRegions() throws
2846       IOException, KeeperException, CoordinatedStateException {
2847     Set<TableName> disabledOrEnablingTables = tableStateManager.getTablesInStates(
2848       ZooKeeperProtos.Table.State.DISABLED, ZooKeeperProtos.Table.State.ENABLING);
2849 
2850     Set<TableName> disabledOrDisablingOrEnabling = tableStateManager.getTablesInStates(
2851       ZooKeeperProtos.Table.State.DISABLED,
2852       ZooKeeperProtos.Table.State.DISABLING,
2853       ZooKeeperProtos.Table.State.ENABLING);
2854 
2855     // Region assignment from META
2856     List<Result> results = MetaTableAccessor.fullScanOfMeta(server.getConnection());
2857     // Get any new but slow to checkin region server that joined the cluster
2858     Set<ServerName> onlineServers = serverManager.getOnlineServers().keySet();
2859     // Set of offline servers to be returned
2860     Set<ServerName> offlineServers = new HashSet<ServerName>();
2861     // Iterate regions in META
2862     for (Result result : results) {
2863       if (result == null && LOG.isDebugEnabled()){
2864         LOG.debug("null result from meta - ignoring but this is strange.");
2865         continue;
2866       }
2867       // keep a track of replicas to close. These were the replicas of the originally
2868       // unmerged regions. The master might have closed them before but it mightn't
2869       // maybe because it crashed.
2870       PairOfSameType<HRegionInfo> p = MetaTableAccessor.getMergeRegions(result);
2871       if (p.getFirst() != null && p.getSecond() != null) {
2872         int numReplicas = ((MasterServices)server).getTableDescriptors().get(p.getFirst().
2873             getTable()).getRegionReplication();
2874         for (HRegionInfo merge : p) {
2875           for (int i = 1; i < numReplicas; i++) {
2876             replicasToClose.add(RegionReplicaUtil.getRegionInfoForReplica(merge, i));
2877           }
2878         }
2879       }
2880       RegionLocations rl =  MetaTableAccessor.getRegionLocations(result);
2881       if (rl == null) continue;
2882       HRegionLocation[] locations = rl.getRegionLocations();
2883       if (locations == null) continue;
2884       for (HRegionLocation hrl : locations) {
2885         HRegionInfo regionInfo = hrl.getRegionInfo();
2886         if (regionInfo == null) continue;
2887         int replicaId = regionInfo.getReplicaId();
2888         State state = RegionStateStore.getRegionState(result, replicaId);
2889         // keep a track of replicas to close. These were the replicas of the split parents
2890         // from the previous life of the master. The master should have closed them before
2891         // but it couldn't maybe because it crashed
2892         if (replicaId == 0 && state.equals(State.SPLIT)) {
2893           for (HRegionLocation h : locations) {
2894             replicasToClose.add(h.getRegionInfo());
2895           }
2896         }
2897         ServerName lastHost = hrl.getServerName();
2898         ServerName regionLocation = RegionStateStore.getRegionServer(result, replicaId);
2899         if (tableStateManager.isTableState(regionInfo.getTable(),
2900              ZooKeeperProtos.Table.State.DISABLED)) {
2901           // force region to forget it hosts for disabled/disabling tables.
2902           // see HBASE-13326
2903           lastHost = null;
2904           regionLocation = null;
2905         }
2906         regionStates.createRegionState(regionInfo, state, regionLocation, lastHost);
2907         if (!regionStates.isRegionInState(regionInfo, State.OPEN)) {
2908           // Region is not open (either offline or in transition), skip
2909           continue;
2910         }
2911         TableName tableName = regionInfo.getTable();
2912         if (!onlineServers.contains(regionLocation)) {
2913           // Region is located on a server that isn't online
2914           offlineServers.add(regionLocation);
2915           if (useZKForAssignment) {
2916             regionStates.regionOffline(regionInfo);
2917           }
2918         } else if (!disabledOrEnablingTables.contains(tableName)) {
2919           // Region is being served and on an active server
2920           // add only if region not in disabled or enabling table
2921           regionStates.regionOnline(regionInfo, regionLocation);
2922           balancer.regionOnline(regionInfo, regionLocation);
2923         } else if (useZKForAssignment) {
2924           regionStates.regionOffline(regionInfo);
2925         }
2926         // need to enable the table if not disabled or disabling or enabling
2927         // this will be used in rolling restarts
2928         if (!disabledOrDisablingOrEnabling.contains(tableName)
2929           && !getTableStateManager().isTableState(tableName,
2930             ZooKeeperProtos.Table.State.ENABLED)) {
2931           setEnabledTable(tableName);
2932         }
2933       }
2934     }
2935     return offlineServers;
2936   }

4012     // current replica assignments. If replica1 of parent is assigned to RS1,
4013     // the replica1s of daughters will be on the same machine

4020           " due to " + e.getMessage() + ". The assignment of daughter replicas " +

4036     }

4124   /**
4125    * Try to update some region states. If the state machine prevents
4126    * such update, an error message is returned to explain the reason.
4127    *
4128    * It's expected that in each transition there should have just one
4129    * region for opening/closing, 3 regions for splitting/merging.
4130    * These regions should be on the server that requested the change.
4131    *
4132    * Region state machine. Only these transitions
4133    * are expected to be triggered by a region server.
4134    *
4135    * On the state transition:
4136    *  (1) Open/Close should be initiated by master
4137    *      (a) Master sets the region to pending_open/pending_close
4138    *        in memory and hbase:meta after sending the request
4139    *        to the region server
4140    *      (b) Region server reports back to the master
4141    *        after open/close is done (either success/failure)
4142    *      (c) If region server has problem to report the status
4143    *        to master, it must be because the master is down or some
4144    *        temporary network issue. Otherwise, the region server should
4145    *        abort since it must be a bug. If the master is not accessible,
4146    *        the region server should keep trying until the server is
4147    *        stopped or till the status is reported to the (new) master
4148    *      (d) If region server dies in the middle of opening/closing
4149    *        a region, SSH picks it up and finishes it
4150    *      (e) If master dies in the middle, the new master recovers
4151    *        the state during initialization from hbase:meta. Region server
4152    *        can report any transition that has not been reported to
4153    *        the previous active master yet
4154    *  (2) Split/merge is initiated by region servers
4155    *      (a) To split a region, a region server sends a request
4156    *        to master to try to set a region to splitting, together with
4157    *        two daughters (to be created) to splitting new. If approved
4158    *        by the master, the splitting can then move ahead
4159    *      (b) To merge two regions, a region server sends a request to
4160    *        master to try to set the new merged region (to be created) to
4161    *        merging_new, together with two regions (to be merged) to merging.
4162    *        If it is ok with the master, the merge can then move ahead
4163    *      (c) Once the splitting/merging is done, the region server
4164    *        reports the status back to the master either success/failure.
4165    *      (d) Other scenarios should be handled similarly as for
4166    *        region open/close
4167    */
4168   protected String onRegionTransition(final ServerName serverName,
4169       final RegionStateTransition transition) {
4170     TransitionCode code = transition.getTransitionCode();
4171     HRegionInfo hri = HRegionInfo.convert(transition.getRegionInfo(0));
4172     RegionState current = regionStates.getRegionState(hri);
4173     if (LOG.isDebugEnabled()) {
4174       LOG.debug("Got transition " + code + " for "
4175         + (current != null ? current.toString() : hri.getShortNameToLog())
4176         + " from " + serverName);
4177     }
4178     String errorMsg = null;
4179     switch (code) {
4180     case OPENED:
4181       if (current != null && current.isOpened() && current.isOnServer(serverName)) {
4182         LOG.info("Region " + hri.getShortNameToLog() + " is already " + current.getState() + " on "
4183             + serverName);
4184         break;
4185       }
4186     case FAILED_OPEN:
4187       if (current == null
4188           || !current.isPendingOpenOrOpeningOnServer(serverName)) {
4189         errorMsg = hri.getShortNameToLog()
4190           + " is not pending open on " + serverName;
4191       } else if (code == TransitionCode.FAILED_OPEN) {
4192         onRegionFailedOpen(hri, serverName);
4193       } else {
4194         long openSeqNum = HConstants.NO_SEQNUM;
4195         if (transition.hasOpenSeqNum()) {
4196           openSeqNum = transition.getOpenSeqNum();
4197         }
4198         if (openSeqNum < 0) {
4199           errorMsg = "Newly opened region has invalid open seq num " + openSeqNum;
4200         } else {
4201           onRegionOpen(hri, serverName, openSeqNum);
4202         }
4203       }
4204       break;
4205 
4206     case CLOSED:
4207       if (current == null
4208           || !current.isPendingCloseOrClosingOnServer(serverName)) {
4209         errorMsg = hri.getShortNameToLog()
4210           + " is not pending close on " + serverName;
4211       } else {
4212         onRegionClosed(hri);
4213       }
4214       break;
4215 
4216     case READY_TO_SPLIT:
4217       try {
4218         regionStateListener.onRegionSplit(hri);
4219       } catch (IOException exp) {
4220         errorMsg = StringUtils.stringifyException(exp);
4221       }
4222     case SPLIT_PONR:
4223     case SPLIT:
4224     case SPLIT_REVERTED:
4225       errorMsg =
4226           onRegionSplit(serverName, code, hri, HRegionInfo.convert(transition.getRegionInfo(1)),
4227             HRegionInfo.convert(transition.getRegionInfo(2)));
4228       if (org.apache.commons.lang.StringUtils.isEmpty(errorMsg)) {
4229         try {
4230           regionStateListener.onRegionSplitReverted(hri);
4231         } catch (IOException exp) {
4232           LOG.warn(StringUtils.stringifyException(exp));
4233         }
4234       }
4235       break;
4236     case READY_TO_MERGE:
4237     case MERGE_PONR:
4238     case MERGED:
4239     case MERGE_REVERTED:
4240       errorMsg = onRegionMerge(serverName, code, hri,
4241         HRegionInfo.convert(transition.getRegionInfo(1)),
4242         HRegionInfo.convert(transition.getRegionInfo(2)));
4243       if (code == TransitionCode.MERGED && org.apache.commons.lang.StringUtils.isEmpty(errorMsg)) {
4244         try {
4245           regionStateListener.onRegionMerged(hri);
4246         } catch (IOException exp) {
4247           errorMsg = StringUtils.stringifyException(exp);
4248         }
4249       }
4250       break;
4251 
4252     default:
4253       errorMsg = "Unexpected transition code " + code;
4254     }
4255     if (errorMsg != null) {
4256       LOG.error("Failed to transtion region from " + current + " to "
4257         + code + " by " + serverName + ": " + errorMsg);
4258     }
4259     return errorMsg;
4260   }
