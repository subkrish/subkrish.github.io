131   /**
132    * Verify that you can stop a long-running compaction
133    * (used during RS shutdown)
134    * @throws Exception
135    */
136   @Test
137   public void testInterruptCompaction() throws Exception {
138     assertEquals(0, count());
139 
140     // lower the polling interval for this test
141     int origWI = HStore.closeCheckInterval;
142     HStore.closeCheckInterval = 10*1000; // 10 KB
143 
144     try {
145       // Create a couple store files w/ 15KB (over 10KB interval)
146       int jmax = (int) Math.ceil(15.0/compactionThreshold);
147       byte [] pad = new byte[1000]; // 1 KB chunk
148       for (int i = 0; i < compactionThreshold; i++) {
149         HRegionIncommon loader = new HRegionIncommon(r);
150         Put p = new Put(Bytes.add(STARTROW, Bytes.toBytes(i)));
151         p.setDurability(Durability.SKIP_WAL);
152         for (int j = 0; j < jmax; j++) {
153           p.add(COLUMN_FAMILY, Bytes.toBytes(j), pad);
154         }
155         HBaseTestCase.addContent(loader, Bytes.toString(COLUMN_FAMILY));
156         loader.put(p);
157         loader.flushcache();
158       }
159 
160       HRegion spyR = spy(r);
161       doAnswer(new Answer() {
162         public Object answer(InvocationOnMock invocation) throws Throwable {
163           r.writestate.writesEnabled = false;
164           return invocation.callRealMethod();
165         }
166       }).when(spyR).doRegionCompactionPrep();
167 
168       // force a minor compaction, but not before requesting a stop
169       spyR.compactStores();
170 
171       // ensure that the compaction stopped, all old files are intact,
172       Store s = r.stores.get(COLUMN_FAMILY);
173       assertEquals(compactionThreshold, s.getStorefilesCount());
174       assertTrue(s.getStorefilesSize() > 15*1000);
175       // and no new store files persisted past compactStores()
176       FileStatus[] ls = r.getFilesystem().listStatus(r.getRegionFileSystem().getTempDir());
177       assertEquals(0, ls.length);
178 
179     } finally {
180       // don't mess up future tests
181       r.writestate.writesEnabled = true;
182       HStore.closeCheckInterval = origWI;
183 
184       // Delete all Store information once done using
185       for (int i = 0; i < compactionThreshold; i++) {
186         Delete delete = new Delete(Bytes.add(STARTROW, Bytes.toBytes(i)));
187         byte [][] famAndQf = {COLUMN_FAMILY, null};
188         delete.deleteFamily(famAndQf[0]);
189         r.delete(delete);
190       }
191       r.flush(true);
192 
193       // Multiple versions allowed for an entry, so the delete isn't enough
194       // Lower TTL and expire to ensure that all our entries have been wiped
195       final int ttl = 1000;
196       for (Store hstore: this.r.stores.values()) {
197         HStore store = (HStore)hstore;
198         ScanInfo old = store.getScanInfo();
199         ScanInfo si = new ScanInfo(old.getFamily(),
200             old.getMinVersions(), old.getMaxVersions(), ttl,
201             old.getKeepDeletedCells(), 0, old.getComparator());
202         store.setScanInfo(si);
203       }
204       Thread.sleep(ttl);
205 
206       r.compact(true);
207       assertEquals(0, count());
208     }
209   }

211   private int count() throws IOException {
212     int count = 0;
213     for (StoreFile f: this.r.stores.
214         get(COLUMN_FAMILY_TEXT).getStorefiles()) {
215       HFileScanner scanner = f.getReader().getScanner(false, false);
216       if (!scanner.seekTo()) {
217         continue;
218       }
219       do {
220         count++;
221       } while(scanner.next());
222     }
223     return count;
224   }

236   @Test
237   public void testCompactionWithCorruptResult() throws Exception {
238     int nfiles = 10;
239     for (int i = 0; i < nfiles; i++) {
240       createStoreFile(r);
241     }
242     HStore store = (HStore) r.getStore(COLUMN_FAMILY);
243 
244     Collection<StoreFile> storeFiles = store.getStorefiles();
245     DefaultCompactor tool = (DefaultCompactor)store.storeEngine.getCompactor();
246     tool.compactForTesting(storeFiles, false);
247 
248     // Now lets corrupt the compacted file.
249     FileSystem fs = store.getFileSystem();
250     // default compaction policy created one and only one new compacted file
251     Path dstPath = store.getRegionFileSystem().createTempName();
252     FSDataOutputStream stream = fs.create(dstPath, null, true, 512, (short)3, (long)1024, null);
253     stream.writeChars("CORRUPT FILE!!!!");
254     stream.close();
255     Path origPath = store.getRegionFileSystem().commitStoreFile(
256       Bytes.toString(COLUMN_FAMILY), dstPath);
257 
258     try {
259       ((HStore)store).moveFileIntoPlace(origPath);
260     } catch (Exception e) {
261       // The complete compaction should fail and the corrupt file should remain
262       // in the 'tmp' directory;
263       assert (fs.exists(origPath));
264       assert (!fs.exists(dstPath));
265       System.out.println("testCompactionWithCorruptResult Passed");
266       return;
267     }
268     fail("testCompactionWithCorruptResult failed since no exception was" +
269         "thrown while completing a corrupt file");
270   }

277   public void testTrackingCompactionRequest() throws Exception {
278     // setup a compact/split thread on a mock server
279     HRegionServer mockServer = Mockito.mock(HRegionServer.class);
280     Mockito.when(mockServer.getConfiguration()).thenReturn(r.getBaseConf());
281     CompactSplitThread thread = new CompactSplitThread(mockServer);

288       createStoreFile(r);
289     }
290 
291     CountDownLatch latch = new CountDownLatch(1);
292     TrackableCompactionRequest request = new TrackableCompactionRequest(latch);

294     // wait for the latch to complete.
295     latch.await();
296

346     public class TestCompactionContext extends CompactionContext {
347       private List<StoreFile> selectedFiles;
348       public TestCompactionContext(List<StoreFile> selectedFiles) {
349         super();
350         this.selectedFiles = selectedFiles;
351       }
352 
353       @Override
354       public List<StoreFile> preSelect(List<StoreFile> filesCompacting) {
355         return new ArrayList<StoreFile>();
356       }
357 
358       @Override
359       public boolean select(List<StoreFile> filesCompacting, boolean isUserCompaction,
360           boolean mayUseOffPeak, boolean forceMajor) throws IOException {
361         this.request = new CompactionRequest(selectedFiles);
362         this.request.setPriority(getPriority());
363         return true;
364       }
365 
366       @Override
367       public List<Path> compact(CompactionThroughputController throughputController)
368           throws IOException {
369         finishCompaction(this.selectedFiles);
370         return new ArrayList<Path>();
371       }
372     }
373 
374     @Override
375     public synchronized CompactionContext selectCompaction() {
376       CompactionContext ctx = new TestCompactionContext(new ArrayList<StoreFile>(notCompacting));
377       compacting.addAll(notCompacting);
378       notCompacting.clear();
379       try {
380         ctx.select(null, false, false, false);
381       } catch (IOException ex) {
382         fail("Shouldn't happen");
383       }
384       return ctx;
385     }
386 
387     @Override
388     public synchronized void cancelCompaction(Object object) {
389       TestCompactionContext ctx = (TestCompactionContext)object;
390       compacting.removeAll(ctx.selectedFiles);
391       notCompacting.addAll(ctx.selectedFiles);
392     }
393 
394     public synchronized void finishCompaction(List<StoreFile> sfs) {
395       if (sfs.isEmpty()) return;
396       synchronized (results) {
397         results.add(sfs.size());
398       }
399       compacting.removeAll(sfs);
400     }
401 
402     @Override
403     public int getPriority() {
404       return 7 - compacting.size() - notCompacting.size();
405     }

411     public class BlockingCompactionContext extends CompactionContext {
412       public volatile boolean isInCompact = false;
413 
414       public void unblock() {
415         synchronized (this) { this.notifyAll(); }
416       }
417 
418       @Override
419       public List<Path> compact(CompactionThroughputController throughputController)
420           throws IOException {
421         try {
422           isInCompact = true;
423           synchronized (this) {
424             this.wait();
425           }
426         } catch (InterruptedException e) {
427           Assume.assumeNoException(e);
428         }
429         return new ArrayList<Path>();
430       }
431 
432       @Override
433       public List<StoreFile> preSelect(List<StoreFile> filesCompacting) {
434         return new ArrayList<StoreFile>();
435       }
436 
437       @Override
438       public boolean select(List<StoreFile> f, boolean i, boolean m, boolean e)
439           throws IOException {
440         this.request = new CompactionRequest(new ArrayList<StoreFile>());
441         return true;
442       }
443     }

446     public CompactionContext selectCompaction() {
447       this.blocked = new BlockingCompactionContext();
448       try {
449         this.blocked.select(null, false, false, false);
450       } catch (IOException ex) {
451         fail("Shouldn't happen");
452       }
453       return this.blocked;
454     }

463     public BlockingCompactionContext waitForBlocking() {
464       while (this.blocked == null || !this.blocked.isInCompact) {
465         Threads.sleepWithoutInterrupt(50);
466       }
467       BlockingCompactionContext ctx = this.blocked;
468       this.blocked = null;
469       return ctx;
470     }

485   /** Test compaction priority management and multiple compactions per store (HBASE-8665). */
486   @Test
487   public void testCompactionQueuePriorities() throws Exception {
488     // Setup a compact/split thread on a mock server.
489     final Configuration conf = HBaseConfiguration.create();
490     HRegionServer mockServer = mock(HRegionServer.class);
491     when(mockServer.isStopped()).thenReturn(false);
492     when(mockServer.getConfiguration()).thenReturn(conf);
493     when(mockServer.getChoreService()).thenReturn(new ChoreService("test"));
494     CompactSplitThread cst = new CompactSplitThread(mockServer);
495     when(mockServer.getCompactSplitThread()).thenReturn(cst);
496 
497     // Set up the region mock that redirects compactions.
498     HRegion r = mock(HRegion.class);
499     when(
500       r.compact(any(CompactionContext.class), any(Store.class),
501         any(CompactionThroughputController.class))).then(new Answer<Boolean>() {
502       public Boolean answer(InvocationOnMock invocation) throws Throwable {
503         invocation.getArgumentAt(0, CompactionContext.class).compact(
504           invocation.getArgumentAt(2, CompactionThroughputController.class));
505         return true;
506       }
507     });
508 
509     // Set up store mocks for 2 "real" stores and the one we use for blocking CST.
510     ArrayList<Integer> results = new ArrayList<Integer>();
511     StoreMockMaker sm = new StoreMockMaker(results), sm2 = new StoreMockMaker(results);
512     Store store = sm.createStoreMock("store1"), store2 = sm2.createStoreMock("store2");
513     BlockingStoreMockMaker blocker = new BlockingStoreMockMaker();
514 
515     // First, block the compaction thread so that we could muck with queue.
516     cst.requestSystemCompaction(r, blocker.createStoreMock(1, "b-pri1"), "b-pri1");
517     BlockingStoreMockMaker.BlockingCompactionContext currentBlock = blocker.waitForBlocking();
518 
519     // Add 4 files to store1, 3 to store2, and queue compactions; pri 3 and 4 respectively.
520     for (int i = 0; i < 4; ++i) {
521       sm.notCompacting.add(createFile());
522     }
523     cst.requestSystemCompaction(r, store, "s1-pri3");
524     for (int i = 0; i < 3; ++i) {
525       sm2.notCompacting.add(createFile());
526     }
527     cst.requestSystemCompaction(r, store2, "s2-pri4");
528     // Now add 2 more files to store1 and queue compaction - pri 1.
529     for (int i = 0; i < 2; ++i) {
530       sm.notCompacting.add(createFile());
531     }
532     cst.requestSystemCompaction(r, store, "s1-pri1");
533     // Finally add blocking compaction with priority 2.
534     cst.requestSystemCompaction(r, blocker.createStoreMock(2, "b-pri2"), "b-pri2");
535 
536     // Unblock the blocking compaction; we should run pri1 and become block again in pri2.
537     currentBlock.unblock();
538     currentBlock = blocker.waitForBlocking();
539     // Pri1 should have "compacted" all 6 files.
540     assertEquals(1, results.size());
541     assertEquals(6, results.get(0).intValue());
542     // Add 2 files to store 1 (it has 2 files now).
543     for (int i = 0; i < 2; ++i) {
544       sm.notCompacting.add(createFile());
545     }
546     // Now we have pri4 for store 2 in queue, and pri3 for store1; store1's current priority
547     // is 5, however, so it must not preempt store 2. Add blocking compaction at the end.
548     cst.requestSystemCompaction(r, blocker.createStoreMock(7, "b-pri7"), "b-pri7");
549     currentBlock.unblock();
550     currentBlock = blocker.waitForBlocking();
551     assertEquals(3, results.size());
552     assertEquals(3, results.get(1).intValue()); // 3 files should go before 2 files.
553     assertEquals(2, results.get(2).intValue());
554 
555     currentBlock.unblock();
556     cst.interruptIfNecessary();
557   }

568   /**
569    * Simple {@link CompactionRequest} on which you can wait until the requested compaction finishes.
570    */
571   public static class TrackableCompactionRequest extends CompactionRequest {
572     private CountDownLatch done;
573 
574     /**
575      * Constructor for a custom compaction. Uses the setXXX methods to update the state of the
576      * compaction before being used.
577      */
578     public TrackableCompactionRequest(CountDownLatch finished) {
579       super();
580       this.done = finished;
581     }
582 
583     @Override
584     public void afterExecute() {
585       super.afterExecute();
586       this.done.countDown();
587     }
