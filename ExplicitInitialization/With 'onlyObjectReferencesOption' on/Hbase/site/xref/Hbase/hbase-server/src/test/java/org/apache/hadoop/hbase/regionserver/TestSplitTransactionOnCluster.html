173    * Often region has not yet fully opened.  If we try to use it -- do a move for instance -- it
174    * will fail silently if the region is not yet opened.
175    * @param hri Region to check if in Regions In Transition... wait until out of transition before
176    * returning
177    * @return Passed in <code>hri</code>

180     // Close worked but we are going to open the region elsewhere.  Before going on, make sure
181     // this completes.
182     while (TESTING_UTIL.getHBaseCluster().getMaster().getAssignmentManager().

185         TESTING_UTIL.getHBaseCluster().getMaster().getAssignmentManager().getRegionStates().
186           getRegionTransitionState(hri));
187       Threads.sleep(10);
188     }
189     return hri;
190   }
