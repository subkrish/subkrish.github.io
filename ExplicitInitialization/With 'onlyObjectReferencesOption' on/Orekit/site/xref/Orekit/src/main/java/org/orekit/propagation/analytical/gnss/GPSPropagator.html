37   * This class aims at propagating a GPS orbit from {@link GPSOrbitalElements}.

45      // Constants
46      /** WGS 84 value of the earth's rotation rate in rad/s. */
47      private static final double GPS_AV = 7.2921151467e-5;
48  
49      /** Duration of the GPS cycle in seconds. */
50      private static final double GPS_CYCLE_DURATION = GPSOrbitalElements.GPS_WEEK_IN_SECONDS *
51                                                       GPSOrbitalElements.GPS_WEEK_NB;
52  
53      // Data used to solve Kepler's equation
54      /** First coefficient to compute Kepler equation solver starter. */
55      private static final double A;
56  
57      /** Second coefficient to compute Kepler equation solver starter. */
58      private static final double B;
59  
60      static {
61          final double k1 = 3 * FastMath.PI + 2;
62          final double k2 = FastMath.PI - 1;
63          final double k3 = 6 * FastMath.PI - 1;
64          A  = 3 * k2 * k2 / k1;
65          B  = k3 * k3 / (6 * k1);
66      }

195     /**
196      * Gets the PVCoordinates of the GPS SV in {@link #getECEF() ECEF frame}.
197      *
198      * <p>The algorithm is defined at Table 20-IV from IS-GPS-200 document,
199      * with automatic differentiation added to compute velocity and
200      * acceleration.</p>
201      *
202      * @param date the computation date
203      * @return the GPS SV PVCoordinates in {@link #getECEF() ECEF frame}
204      */
205     public PVCoordinates propagateInEcef(final AbsoluteDate date) {
206         // Duration from GPS ephemeris Reference date
207         final DerivativeStructure tk = new DerivativeStructure(1, 2, 0, getTk(date));
208         // Mean anomaly
209         final DerivativeStructure mk = tk.multiply(gpsOrbit.getMeanMotion()).add(gpsOrbit.getM0());
210         // Eccentric Anomaly
211         final DerivativeStructure ek = getEccentricAnomaly(mk);
212         // True Anomaly
213         final DerivativeStructure vk =  getTrueAnomaly(ek);
214         // Argument of Latitude
215         final DerivativeStructure phik    = vk.add(gpsOrbit.getPa());
216         final DerivativeStructure twoPhik = phik.multiply(2);
217         final DerivativeStructure c2phi   = twoPhik.cos();
218         final DerivativeStructure s2phi   = twoPhik.sin();
219         // Argument of Latitude Correction
220         final DerivativeStructure dphik = c2phi.multiply(gpsOrbit.getCuc()).add(s2phi.multiply(gpsOrbit.getCus()));
221         // Radius Correction
222         final DerivativeStructure drk = c2phi.multiply(gpsOrbit.getCrc()).add(s2phi.multiply(gpsOrbit.getCrs()));
223         // Inclination Correction
224         final DerivativeStructure dik = c2phi.multiply(gpsOrbit.getCic()).add(s2phi.multiply(gpsOrbit.getCis()));
225         // Corrected Argument of Latitude
226         final DerivativeStructure uk = phik.add(dphik);
227         // Corrected Radius
228         final DerivativeStructure rk = ek.cos().multiply(-gpsOrbit.getE()).add(1).multiply(gpsOrbit.getSma()).add(drk);
229         // Corrected Inclination
230         final DerivativeStructure ik  = tk.multiply(gpsOrbit.getIDot()).add(gpsOrbit.getI0()).add(dik);
231         final DerivativeStructure cik = ik.cos();
232         // Positions in orbital plane
233         final DerivativeStructure xk = uk.cos().multiply(rk);
234         final DerivativeStructure yk = uk.sin().multiply(rk);
235         // Corrected longitude of ascending node
236         final DerivativeStructure omk = tk.multiply(gpsOrbit.getOmegaDot() - GPS_AV).
237                                         add(gpsOrbit.getOmega0() - GPS_AV * gpsOrbit.getTime());
238         final DerivativeStructure comk = omk.cos();
239         final DerivativeStructure somk = omk.sin();
240         // returns the Earth-fixed coordinates
241         final FieldVector3D<DerivativeStructure> positionwithDerivatives =
242                         new FieldVector3D<DerivativeStructure>(xk.multiply(comk).subtract(yk.multiply(somk).multiply(cik)),
243                                                                xk.multiply(somk).add(yk.multiply(comk).multiply(cik)),
244                                                                yk.multiply(ik.sin()));
245         return new PVCoordinates(positionwithDerivatives);
246     }

248     /**
249      * Get the duration from GPS Reference epoch.
250      * <p>This takes the GPS week roll-over into account.</p>
251      *
252      * @param date the considered date
253      * @return the duration from GPS orbit Reference epoch (s)
254      */
255     private double getTk(final AbsoluteDate date) {
256         // Time from ephemeris reference epoch
257         double tk = date.durationFrom(gpsOrbit.getDate());
258         // Adjusts the time to take roll over week into account
259         while (tk > 0.5 * GPS_CYCLE_DURATION) {
260             tk -= GPS_CYCLE_DURATION;
261         }
262         while (tk < -0.5 * GPS_CYCLE_DURATION) {
263             tk += GPS_CYCLE_DURATION;
264         }
265         // Returns the time from ephemeris reference epoch
266         return tk;
267     }

269     /**
270      * Gets eccentric anomaly from mean anomaly.
271      * <p>The algorithm used to solve the Kepler equation has been published in:
272      * "Procedures for  solving Kepler's Equation", A. W. Odell and R. H. Gooding,
273      * Celestial Mechanics 38 (1986) 307-334</p>
274      * <p>It has been copied from the OREKIT library (KeplerianOrbit class).</p>
275      *
276      * @param mk the mean anomaly (rad)
277      * @return the eccentric anomaly (rad)
278      */
279     private DerivativeStructure getEccentricAnomaly(final DerivativeStructure mk) {
280 
281         // reduce M to [-PI PI] interval
282         final double[] mlDerivatives = mk.getAllDerivatives();
283         mlDerivatives[0] = MathUtils.normalizeAngle(mlDerivatives[0], 0.0);
284         final DerivativeStructure reducedM = new DerivativeStructure(mk.getFreeParameters(),
285                                                                      mk.getOrder(),
286                                                                      mlDerivatives);
287 
288         // compute start value according to A. W. Odell and R. H. Gooding S12 starter
289         DerivativeStructure ek;
290         if (FastMath.abs(reducedM.getValue()) < 1.0 / 6.0) {
291             ek = reducedM.add(reducedM.multiply(6).cbrt().subtract(reducedM).multiply(gpsOrbit.getE()));
292         } else {
293             if (reducedM.getValue() < 0) {
294                 final DerivativeStructure w = reducedM.add(FastMath.PI);
295                 ek = reducedM.add(w.multiply(-A).divide(w.subtract(B)).subtract(FastMath.PI).subtract(reducedM).multiply(gpsOrbit.getE()));
296             } else {
297                 final DerivativeStructure minusW = reducedM.subtract(FastMath.PI);
298                 ek = reducedM.add(minusW.multiply(A).divide(minusW.add(B)).add(FastMath.PI).subtract(reducedM).multiply(gpsOrbit.getE()));
299             }
300         }
301 
302         final double e1 = 1 - gpsOrbit.getE();
303         final boolean noCancellationRisk = (e1 + ek.getValue() * ek.getValue() / 6) >= 0.1;
304 
305         // perform two iterations, each consisting of one Halley step and one Newton-Raphson step
306         for (int j = 0; j < 2; ++j) {
307             final DerivativeStructure f;
308             DerivativeStructure fd;
309             final DerivativeStructure fdd  = ek.sin().multiply(gpsOrbit.getE());
310             final DerivativeStructure fddd = ek.cos().multiply(gpsOrbit.getE());
311             if (noCancellationRisk) {
312                 f  = ek.subtract(fdd).subtract(reducedM);
313                 fd = fddd.subtract(1).negate();
314             } else {
315                 f  = eMeSinE(ek).subtract(reducedM);
316                 final DerivativeStructure s = ek.multiply(0.5).sin();
317                 fd = s.multiply(s).multiply(2 * gpsOrbit.getE()).add(e1);
318             }
319             final DerivativeStructure dee = f.multiply(fd).divide(f.multiply(0.5).multiply(fdd).subtract(fd.multiply(fd)));
320 
321             // update eccentric anomaly, using expressions that limit underflow problems
322             final DerivativeStructure w = fd.add(dee.multiply(0.5).multiply(fdd.add(dee.multiply(fdd).divide(3))));
323             fd = fd.add(dee.multiply(fdd.add(dee.multiply(0.5).multiply(fdd))));
324             ek = ek.subtract(f.subtract(dee.multiply(fd.subtract(w))).divide(fd));
325         }
326 
327         // expand the result back to original range
328         ek = ek.add(mk.getValue() - reducedM.getValue());
329 
330         // Returns the eccentric anomaly
331         return ek;
332     }

334     /**
335      * Accurate computation of E - e sin(E).
336      *
337      * @param E eccentric anomaly
338      * @return E - e sin(E)
339      */
340     private DerivativeStructure eMeSinE(final DerivativeStructure E) {
341         DerivativeStructure x = E.sin().multiply(1 - gpsOrbit.getE());
342         final DerivativeStructure mE2 = E.negate().multiply(E);
343         DerivativeStructure term = E;
344         DerivativeStructure d    = E.getField().getZero();
345         // the inequality test below IS intentional and should NOT be replaced by a check with a small tolerance
346         for (DerivativeStructure x0 = d.add(Double.NaN); x.getValue() != x0.getValue();) {
347             d = d.add(2);
348             term = term.multiply(mE2.divide(d.multiply(d.add(1))));
349             x0 = x;
350             x = x.subtract(term);
351         }
352         return x;
353     }

355     /** Gets true anomaly from eccentric anomaly.
356      *
357      * @param ek the eccentric anomaly (rad)
358      * @return the true anomaly (rad)
359      */
360     private DerivativeStructure getTrueAnomaly(final DerivativeStructure ek) {
361         final DerivativeStructure svk = ek.sin().multiply(FastMath.sqrt(1. - gpsOrbit.getE() * gpsOrbit.getE()));
362         final DerivativeStructure cvk = ek.cos().subtract(gpsOrbit.getE());
363         return svk.atan2(cvk);
364     }

424     /** {@inheritDoc} */
425     protected Orbit propagateOrbit(final AbsoluteDate date) throws OrekitException {
426         // Gets the PVCoordinates in ECEF frame
427         final PVCoordinates pvaInECEF = propagateInEcef(date);
428         // Transforms the PVCoordinates to ECI frame
429         final PVCoordinates pvaInECI = ecef.getTransformTo(eci, date).transformPVCoordinates(pvaInECEF);
430         // Returns the Cartesian orbit
431         return new CartesianOrbit(pvaInECI, eci, date, GPSOrbitalElements.GPS_MU);
432     }
