96       * Parse the given URL as an antlib descriptor and return the
97       * content as something that can be turned into an Antlib task.
98       *
99       * <p>simply delegates to {@link #parseUnknownElement
100      * parseUnknownElement} if the resource provides an URL and throws
101      * an exception otherwise.</p>
102      *
103      * @since Ant 1.8.0
104      */
105     public UnknownElement parseAntlibDescriptor(Project containingProject,
106                                                 Resource resource) {
107         URLProvider up = resource.as(URLProvider.class);
108         if (up == null) {
109             throw new BuildException("Unsupported resource type: " + resource);
110         }
111         return parseUnknownElement(containingProject, up.getURL());
112     }

114     /**
115      * Parse an unknown element from a url
116      *
117      * @param project the current project
118      * @param source  the url containing the task
119      * @return a configured task
120      * @exception BuildException if an error occurs
121      */
122     public UnknownElement parseUnknownElement(Project project, URL source)
123         throws BuildException {
124         Target dummyTarget = new Target();
125         dummyTarget.setProject(project);
126 
127         AntXMLContext context = new AntXMLContext(project);
128         context.addTarget(dummyTarget);
129         context.setImplicitTarget(dummyTarget);
130 
131         parse(context.getProject(), source, new RootHandler(context, elementHandler));
132         Task[] tasks = dummyTarget.getTasks();
133         if (tasks.length != 1) {
134             throw new BuildException("No tasks defined");
135         }
136         return (UnknownElement) tasks[0];
137     }

139     /**
140      * Parse a source xml input.
141      *
142      * @param project the current project
143      * @param source  the xml source
144      * @exception BuildException if an error occurs
145      */
146     public void parse(Project project, Object source) throws BuildException {
147         getImportStack().addElement(source);
148         AntXMLContext context = null;
149         context = (AntXMLContext) project.getReference(REFID_CONTEXT);
150         if (context == null) {
151             context = new AntXMLContext(project);
152             project.addReference(REFID_CONTEXT, context);
153             project.addReference(REFID_TARGETS, context.getTargets());
154         }
155         if (getImportStack().size() > 1) {
156             // we are in an imported file.
157             context.setIgnoreProjectTag(true);
158             Target currentTarget = context.getCurrentTarget();
159             Target currentImplicit = context.getImplicitTarget();
160             Map<String, Target>    currentTargets = context.getCurrentTargets();
161             try {
162                 Target newCurrent = new Target();
163                 newCurrent.setProject(project);
164                 newCurrent.setName("");
165                 context.setCurrentTarget(newCurrent);
166                 context.setCurrentTargets(new HashMap<String, Target>());
167                 context.setImplicitTarget(newCurrent);
168                 parse(project, source, new RootHandler(context, mainHandler));
169                 newCurrent.execute();
170             } finally {
171                 context.setCurrentTarget(currentTarget);
172                 context.setImplicitTarget(currentImplicit);
173                 context.setCurrentTargets(currentTargets);
174             }
175         } else {
176             // top level file
177             context.setCurrentTargets(new HashMap<String, Target>());
178             parse(project, source, new RootHandler(context, mainHandler));
179             // Execute the top-level target
180             context.getImplicitTarget().execute();
181 
182             // resolve extensionOf attributes
183             resolveExtensionOfAttributes(project);
184         }
185     }

187     /**
188      * Parses the project file, configuring the project as it goes.
189      *
190      * @param project the current project
191      * @param source  the xml source
192      * @param handler the root handler to use (contains the current context)
193      * @exception BuildException if the configuration is invalid or cannot
194      *                           be read
195      */
196     public void parse(Project project, Object source, RootHandler handler) throws BuildException {
197 
198         AntXMLContext context = handler.context;
199 
200         File buildFile = null;
201         URL  url = null;
202         String buildFileName = null;
203 
204         if (source instanceof File) {
205             buildFile = (File) source;
206         } else if (source instanceof URL) {
207             url = (URL) source;
208         } else if (source instanceof Resource) {
209             FileProvider fp =
210                 ((Resource) source).as(FileProvider.class);
211             if (fp != null) {
212                 buildFile = fp.getFile();
213             } else {
214                 URLProvider up =
215                     ((Resource) source).as(URLProvider.class);
216                 if (up != null) {
217                     url = up.getURL();
218                 }
219             }
220         }
221         if (buildFile != null) {
222             buildFile = FILE_UTILS.normalize(buildFile.getAbsolutePath());
223             context.setBuildFile(buildFile);
224             buildFileName = buildFile.toString();
225         } else if (url != null) {
226             try {
227                 context.setBuildFile((File) null);
228                 context.setBuildFile(url);
229             } catch (java.net.MalformedURLException ex) {
230                 throw new BuildException(ex);
231             }
232             buildFileName = url.toString();
233         } else {
234             throw new BuildException("Source " + source.getClass().getName()
235                                      + " not supported by this plugin");
236         }
237         InputStream inputStream = null;
238         InputSource inputSource = null;
239         ZipFile zf = null;
240 
241         try {
242             /**
243              * SAX 2 style parser used to parse the given file.
244              */
245             XMLReader parser = JAXPUtils.getNamespaceXMLReader();
246 
247             String uri = null;
248             if (buildFile != null) {
249                 uri = FILE_UTILS.toURI(buildFile.getAbsolutePath());
250                 inputStream = new FileInputStream(buildFile);
251             } else {
252                 uri = url.toString();
253                 int pling = -1;
254                 if (uri.startsWith("jar:file")
255                     && (pling = uri.indexOf("!/")) > -1) {
256                     zf = new ZipFile(org.apache.tools.ant.launch.Locator
257                                      .fromJarURI(uri), "UTF-8");
258                     inputStream =
259                         zf.getInputStream(zf.getEntry(uri.substring(pling + 1)));
260                 } else {
261                     URLConnection conn = url.openConnection();
262                     conn.setUseCaches(false);
263                     inputStream = conn.getInputStream();
264                 }
265             }
266 
267             inputSource = new InputSource(inputStream);
268             if (uri != null) {
269                 inputSource.setSystemId(uri);
270             }
271             project.log("parsing buildfile " + buildFileName + " with URI = "
272                         + uri + (zf != null ? " from a zip file" : ""),
273                         Project.MSG_VERBOSE);
274 
275             DefaultHandler hb = handler;
276 
277             parser.setContentHandler(hb);
278             parser.setEntityResolver(hb);
279             parser.setErrorHandler(hb);
280             parser.setDTDHandler(hb);
281             parser.parse(inputSource);
282         } catch (SAXParseException exc) {
283             Location location = new Location(exc.getSystemId(), exc.getLineNumber(), exc
284                                              .getColumnNumber());
285 
286             Throwable t = exc.getException();
287             if (t instanceof BuildException) {
288                 BuildException be = (BuildException) t;
289                 if (be.getLocation() == Location.UNKNOWN_LOCATION) {
290                     be.setLocation(location);
291                 }
292                 throw be;
293             }
294             throw new BuildException(exc.getMessage(), t == null ? exc : t, location);
295         } catch (SAXException exc) {
296             Throwable t = exc.getException();
297             if (t instanceof BuildException) {
298                 throw (BuildException) t;
299             }
300             throw new BuildException(exc.getMessage(), t == null ? exc : t);
301         } catch (FileNotFoundException exc) {
302             throw new BuildException(exc);
303         } catch (UnsupportedEncodingException exc) {
304             throw new BuildException("Encoding of project file " + buildFileName + " is invalid.",
305                                      exc);
306         } catch (IOException exc) {
307             throw new BuildException("Error reading project file " + buildFileName + ": "
308                                      + exc.getMessage(), exc);
309         } finally {
310             FileUtils.close(inputStream);
311             ZipFile.closeQuietly(zf);
312         }
313     }

484     /**
485      * Handler for ant processing. Uses a stack of AntHandlers to
486      * implement each element ( the original parser used a recursive behavior,
487      * with the implicit execution stack )
488      */
489     public static class RootHandler extends DefaultHandler {
490         private Stack<AntHandler> antHandlers = new Stack<AntHandler>();
491         private AntHandler currentHandler = null;
492         private AntXMLContext context;
493 
494         /**
495          * Creates a new RootHandler instance.
496          *
497          * @param context The context for the handler.
498          * @param rootHandler The handler for the root element.
499          */
500         public RootHandler(AntXMLContext context, AntHandler rootHandler) {
501             currentHandler = rootHandler;
502             antHandlers.push(currentHandler);
503             this.context = context;
504         }

514         /**
515          * Resolves file: URIs relative to the build file.
516          *
517          * @param publicId The public identifier, or <code>null</code>
518          *                 if none is available. Ignored in this
519          *                 implementation.
520          * @param systemId The system identifier provided in the XML
521          *                 document. Will not be <code>null</code>.
522          * @return an inputsource for this identifier
523          */
524         public InputSource resolveEntity(String publicId, String systemId) {
525 
526             context.getProject().log("resolving systemId: " + systemId, Project.MSG_VERBOSE);
527 
528             if (systemId.startsWith("file:")) {
529                 String path = FILE_UTILS.fromURI(systemId);
530 
531                 File file = new File(path);
532                 if (!file.isAbsolute()) {
533                     file = FILE_UTILS.resolveFile(context.getBuildFileParent(), path);
534                     context.getProject().log(
535                                              "Warning: '" + systemId + "' in " + context.getBuildFile()
536                                              + " should be expressed simply as '" + path.replace('\\', '/')
537                                              + "' for compliance with other XML tools", Project.MSG_WARN);
538                 }
539                 context.getProject().log("file=" + file, Project.MSG_DEBUG);
540                 try {
541                     InputSource inputSource = new InputSource(new FileInputStream(file));
542                     inputSource.setSystemId(FILE_UTILS.toURI(file.getAbsolutePath()));
543                     return inputSource;
544                 } catch (FileNotFoundException fne) {
545                     context.getProject().log(file.getAbsolutePath() + " could not be found",
546                                              Project.MSG_WARN);
547                 }
548 
549             }
550             // use default if not file or file not found
551             context.getProject().log("could not resolve systemId", Project.MSG_DEBUG);
552             return null;
553         }

555         /**
556          * Handles the start of a project element. A project handler is created
557          * and initialised with the element name and attributes.
558          *
559          * @param uri The namespace uri for this element.
560          * @param tag The name of the element being started.
561          *            Will not be <code>null</code>.
562          * @param qname The qualified name for this element.
563          * @param attrs Attributes of the element being started.
564          *              Will not be <code>null</code>.
565          *
566          * @exception org.xml.sax.SAXParseException if the tag given is not
567          *                              <code>"project"</code>
568          */
569         public void startElement(String uri, String tag, String qname, Attributes attrs)
570             throws SAXParseException {
571             AntHandler next = currentHandler.onStartChild(uri, tag, qname, attrs, context);
572             antHandlers.push(currentHandler);
573             currentHandler = next;
574             currentHandler.onStartElement(uri, tag, qname, attrs, context);
575         }

583         public void setDocumentLocator(Locator locator) {
584             context.setLocator(locator);
585         }

587         /**
588          * Handles the end of an element. Any required clean-up is performed
589          * by the onEndElement() method and then the original handler is restored to the parser.
590          *
591          * @param uri  The namespace URI for this element.
592          * @param name The name of the element which is ending.
593          *             Will not be <code>null</code>.
594          * @param qName The qualified name for this element.
595          *
596          * @exception SAXException in case of error (not thrown in this implementation)
597          */
598         public void endElement(String uri, String name, String qName) throws SAXException {
599             currentHandler.onEndElement(uri, name, context);
600             AntHandler prev = (AntHandler) antHandlers.pop();
601             currentHandler = prev;
602             if (currentHandler != null) {
603                 currentHandler.onEndChild(uri, name, qName, context);
604             }
605         }
606 
607         /**
608          * Handle text within an element, calls currentHandler.characters.
609          *
610          * @param buf  A character array of the test.
611          * @param start The start offset in the array.
612          * @param count The number of characters to read.
613          * @exception SAXParseException if an error occurs
614          */
615         public void characters(char[] buf, int start, int count) throws SAXParseException {
616             currentHandler.characters(buf, start, count, context);
617         }

668                                         + "\" " + name, context.getLocator());
669         }

672     /**
673      * Handler for the top level "project" element.
674      */
675     public static class ProjectHandler extends AntHandler {
676 
677         /**
678          * Initialisation routine called after handler creation
679          * with the element name and attributes. The attributes which
680          * this handler can deal with are: <code>"default"</code>,
681          * <code>"name"</code>, <code>"id"</code> and <code>"basedir"</code>.
682          *
683          * @param uri The namespace URI for this element.
684          * @param tag Name of the element which caused this handler
685          *            to be created. Should not be <code>null</code>.
686          *            Ignored in this implementation.
687          * @param qname The qualified name for this element.
688          * @param attrs Attributes of the element which caused this
689          *              handler to be created. Must not be <code>null</code>.
690          * @param context The current context.
691          *
692          * @exception SAXParseException if an unexpected attribute is
693          *            encountered or if the <code>"default"</code> attribute
694          *            is missing.
695          */
696         public void onStartElement(String uri, String tag, String qname, Attributes attrs,
697                                    AntXMLContext context) throws SAXParseException {
698             String baseDir = null;
699             boolean nameAttributeSet = false;
700 
701             Project project = context.getProject();
702             // Set the location of the implicit target associated with the project tag
703             context.getImplicitTarget().setLocation(new Location(context.getLocator()));
704 
705             /** TODO I really don't like this - the XML processor is still
706              * too 'involved' in the processing. A better solution (IMO)
707              * would be to create UE for Project and Target too, and
708              * then process the tree and have Project/Target deal with
709              * its attributes ( similar with Description ).
710              *
711              * If we eventually switch to ( or add support for ) DOM,
712              * things will work smoothly - UE can be avoided almost completely
713              * ( it could still be created on demand, for backward compatibility )
714              */
715 
716             for (int i = 0; i < attrs.getLength(); i++) {
717                 String attrUri = attrs.getURI(i);
718                 if (attrUri != null && !attrUri.equals("") && !attrUri.equals(uri)) {
719                     continue; // Ignore attributes from unknown uris
720                 }
721                 String key = attrs.getLocalName(i);
722                 String value = attrs.getValue(i);
723 
724                 if (key.equals("default")) {
725                     if (value != null && !value.equals("")) {
726                         if (!context.isIgnoringProjectTag()) {
727                             project.setDefault(value);
728                         }
729                     }
730                 } else if (key.equals("name")) {
731                     if (value != null) {
732                         context.setCurrentProjectName(value);
733                         nameAttributeSet = true;
734                         if (!context.isIgnoringProjectTag()) {
735                             project.setName(value);
736                             project.addReference(value, project);
737                         } else if (isInIncludeMode()) {
738                             if (!"".equals(value) && getCurrentTargetPrefix()!= null && getCurrentTargetPrefix().endsWith(ProjectHelper.USE_PROJECT_NAME_AS_TARGET_PREFIX))  {
739                                 String newTargetPrefix = getCurrentTargetPrefix().replace(ProjectHelper.USE_PROJECT_NAME_AS_TARGET_PREFIX, value);
740                                 // help nested include tasks
741                                 setCurrentTargetPrefix(newTargetPrefix);
742                             }
743                         }
744                     }
745                 } else if (key.equals("id")) {
746                     if (value != null) {
747                         // What's the difference between id and name ?
748                         if (!context.isIgnoringProjectTag()) {
749                             project.addReference(value, project);
750                         }
751                     }
752                 } else if (key.equals("basedir")) {
753                     if (!context.isIgnoringProjectTag()) {
754                         baseDir = value;
755                     }
756                 } else {
757                     // TODO ignore attributes in a different NS ( maybe store them ? )
758                     throw new SAXParseException("Unexpected attribute \"" + attrs.getQName(i)
759                                                 + "\"", context.getLocator());
760                 }
761             }
762 
763             // TODO Move to Project ( so it is shared by all helpers )
764             String antFileProp =
765                 MagicNames.ANT_FILE + "." + context.getCurrentProjectName();
766             String dup = project.getProperty(antFileProp);
767             String typeProp =
768                 MagicNames.ANT_FILE_TYPE + "." + context.getCurrentProjectName();
769             String dupType = project.getProperty(typeProp);
770             if (dup != null && nameAttributeSet) {
771                 Object dupFile = null;
772                 Object contextFile = null;
773                 if (MagicNames.ANT_FILE_TYPE_URL.equals(dupType)) {
774                     try {
775                         dupFile = new URL(dup);
776                     } catch (java.net.MalformedURLException mue) {
777                         throw new BuildException("failed to parse "
778                                                  + dup + " as URL while looking"
779                                                  + " at a duplicate project"
780                                                  + " name.", mue);
781                     }
782                     contextFile = context.getBuildFileURL();
783                 } else {
784                     dupFile = new File(dup);
785                     contextFile = context.getBuildFile();
786                 }
787 
788                 if (context.isIgnoringProjectTag() && !dupFile.equals(contextFile)) {
789                     project.log("Duplicated project name in import. Project "
790                                 + context.getCurrentProjectName() + " defined first in " + dup
791                                 + " and again in " + contextFile, Project.MSG_WARN);
792                 }
793             }
794             if (nameAttributeSet) {
795                 if (context.getBuildFile() != null) {
796                     project.setUserProperty(antFileProp,
797                                             context.getBuildFile().toString());
798                     project.setUserProperty(typeProp,
799                                             MagicNames.ANT_FILE_TYPE_FILE);
800                 } else if (context.getBuildFileURL() != null) {
801                     project.setUserProperty(antFileProp,
802                                             context.getBuildFileURL().toString());
803                     project.setUserProperty(typeProp,
804                                             MagicNames.ANT_FILE_TYPE_URL);
805                 }
806             }
807             if (context.isIgnoringProjectTag()) {
808                 // no further processing
809                 return;
810             }
811             // set explicitly before starting ?
812             if (project.getProperty("basedir") != null) {
813                 project.setBasedir(project.getProperty("basedir"));
814             } else {
815                 // Default for baseDir is the location of the build file.
816                 if (baseDir == null) {
817                     project.setBasedir(context.getBuildFileParent().getAbsolutePath());
818                 } else {
819                     // check whether the user has specified an absolute path
820                     if ((new File(baseDir)).isAbsolute()) {
821                         project.setBasedir(baseDir);
822                     } else {
823                         project.setBaseDir(FILE_UTILS.resolveFile(context.getBuildFileParent(),
824                                                                   baseDir));
825                     }
826                 }
827             }
828             project.addTarget("", context.getImplicitTarget());
829             context.setCurrentTarget(context.getImplicitTarget());
830         }

833          * Handles the start of a top-level element within the project. An

835          * of the element.

838          * @param name The name of the element being started.
839          *            Will not be <code>null</code>.

841          * @param attrs Attributes of the element being started.
842          *              Will not be <code>null</code>.
843          * @param context The context for this element.
844          * @return a target or an element handler.
845          *
846          * @exception org.xml.sax.SAXParseException if the tag given is not

848          *            <code>"property"</code>, <code>"target"</code>,
849          *            <code>"extension-point"</code>
850          *            or a data type definition

852         public AntHandler onStartChild(String uri, String name, String qname, Attributes attrs,

865         /**
866          * Initialisation routine called after handler creation
867          * with the element name and attributes. The attributes which
868          * this handler can deal with are: <code>"name"</code>,
869          * <code>"depends"</code>, <code>"if"</code>,
870          * <code>"unless"</code>, <code>"id"</code> and
871          * <code>"description"</code>.
872          *
873          * @param uri The namespace URI for this element.
874          * @param tag Name of the element which caused this handler
875          *            to be created. Should not be <code>null</code>.
876          *            Ignored in this implementation.
877          * @param qname The qualified name for this element.
878          * @param attrs Attributes of the element which caused this
879          *              handler to be created. Must not be <code>null</code>.
880          * @param context The current context.
881          *
882          * @exception SAXParseException if an unexpected attribute is encountered
883          *            or if the <code>"name"</code> attribute is missing.
884          */
885         public void onStartElement(String uri, String tag, String qname, Attributes attrs,
886                                    AntXMLContext context) throws SAXParseException {
887             String name = null;
888             String depends = "";
889             String extensionPoint = null;
890             OnMissingExtensionPoint extensionPointMissing = null;
891 
892             Project project = context.getProject();
893             Target target = "target".equals(tag)
894                 ? new Target() : new ExtensionPoint();
895             target.setProject(project);
896             target.setLocation(new Location(context.getLocator()));
897             context.addTarget(target);
898 
899             for (int i = 0; i < attrs.getLength(); i++) {
900                 String attrUri = attrs.getURI(i);
901                 if (attrUri != null && !attrUri.equals("") && !attrUri.equals(uri)) {
902                     continue; // Ignore attributes from unknown uris
903                 }
904                 String key = attrs.getLocalName(i);
905                 String value = attrs.getValue(i);
906 
907                 if (key.equals("name")) {
908                     name = value;
909                     if ("".equals(name)) {
910                         throw new BuildException("name attribute must " + "not be empty");
911                     }
912                 } else if (key.equals("depends")) {
913                     depends = value;
914                 } else if (key.equals("if")) {
915                     target.setIf(value);
916                 } else if (key.equals("unless")) {
917                     target.setUnless(value);
918                 } else if (key.equals("id")) {
919                     if (value != null && !value.equals("")) {
920                         context.getProject().addReference(value, target);
921                     }
922                 } else if (key.equals("description")) {
923                     target.setDescription(value);
924                 } else if (key.equals("extensionOf")) {
925                     extensionPoint = value;
926                 } else if (key.equals("onMissingExtensionPoint")) {
927                     try {
928                         extensionPointMissing = OnMissingExtensionPoint.valueOf(value);
929                     } catch (IllegalArgumentException e) {
930                         throw new BuildException("Invalid onMissingExtensionPoint " + value);
931                     }
932                 } else {
933                     throw new SAXParseException("Unexpected attribute \"" + key + "\"", context
934                                                 .getLocator());
935                 }
936             }
937 
938             if (name == null) {
939                 throw new SAXParseException("target element appears without a name attribute",
940                                             context.getLocator());
941             }
942 
943             String prefix = null;
944             boolean isInIncludeMode =
945                 context.isIgnoringProjectTag() && isInIncludeMode();
946             String sep = getCurrentPrefixSeparator();
947 
948             if (isInIncludeMode) {
949                 prefix = getTargetPrefix(context);
950                 if (prefix == null) {
951                     throw new BuildException("can't include build file "
952                                              + context.getBuildFileURL()
953                                              + ", no as attribute has been given"
954                                              + " and the project tag doesn't"
955                                              + " specify a name attribute");
956                 }
957                 name = prefix + sep + name;
958             }
959 
960             // Check if this target is in the current build file
961             if (context.getCurrentTargets().get(name) != null) {
962                 throw new BuildException("Duplicate target '" + name + "'",
963                                          target.getLocation());
964             }
965             Hashtable<String, Target> projectTargets = project.getTargets();
966             boolean   usedTarget = false;
967             // If the name has not already been defined define it
968             if (projectTargets.containsKey(name)) {
969                 project.log("Already defined in main or a previous import, ignore " + name,
970                             Project.MSG_VERBOSE);
971             } else {
972                 target.setName(name);
973                 context.getCurrentTargets().put(name, target);
974                 project.addOrReplaceTarget(name, target);
975                 usedTarget = true;
976             }
977 
978             if (depends.length() > 0) {
979                 if (!isInIncludeMode) {
980                     target.setDepends(depends);
981                 } else {
982                     for (String string : Target.parseDepends(depends, name, "depends")) {
983                         target.addDependency(prefix + sep + string);
984                    }
985                 }
986             }
987             if (!isInIncludeMode && context.isIgnoringProjectTag()
988                 && (prefix = getTargetPrefix(context)) != null) {
989                 // In an imported file (and not completely
990                 // ignoring the project tag or having a preconfigured prefix)
991                 String newName = prefix + sep + name;
992                 Target newTarget = target;
993                 if (usedTarget) {
994                     newTarget = "target".equals(tag)
995                             ? new Target(target) : new ExtensionPoint(target);
996                 }
997                 newTarget.setName(newName);
998                 context.getCurrentTargets().put(newName, newTarget);
999                 project.addOrReplaceTarget(newName, newTarget);
1000             }
1001             if (extensionPointMissing != null && extensionPoint == null) {
1002                 throw new BuildException("onMissingExtensionPoint attribute cannot " +
1003                                          "be specified unless extensionOf is specified",
1004                                          target.getLocation());
1005 
1006             }
1007             if (extensionPoint != null) {
1008                 ProjectHelper helper =
1009                     (ProjectHelper) context.getProject().
1010                     getReference(ProjectHelper.PROJECTHELPER_REFERENCE);
1011                 for (String extPointName : Target.parseDepends(extensionPoint, name, "extensionOf")) {
1012                     if (extensionPointMissing == null) {
1013                         extensionPointMissing = OnMissingExtensionPoint.FAIL;
1014                     }
1015                     // defer extensionpoint resolution until the full
1016                     // import stack has been processed
1017                     if (isInIncludeMode()) {
1018                         // if in include mode, provide prefix we're including by
1019                         // so that we can try and resolve extension point from
1020                         // the local file first
1021                         helper.getExtensionStack().add(
1022                                 new String[] {extPointName, target.getName(),
1023                                         extensionPointMissing.name(), prefix + sep});
1024                     } else {
1025                         helper.getExtensionStack().add(
1026                                 new String[] {extPointName, target.getName(),
1027                                         extensionPointMissing.name()});
1028                     }
1029                 }
1030             }
1031         }

1033         private String getTargetPrefix(AntXMLContext context) {
1034             String configuredValue = getCurrentTargetPrefix();
1035             if (configuredValue != null && configuredValue.length() == 0) {
1036                 configuredValue = null;
1037             }
1038             if (configuredValue != null) {
1039                 return configuredValue;
1040             }
1041 
1042             String projectName = context.getCurrentProjectName();
1043             if ("".equals(projectName)) {
1044                 projectName = null;
1045             }
1046 
1047             return projectName;
1048         }

1089          * Constructor.
1090          */
1091         public ElementHandler() {

1094         /**
1095          * Initialisation routine called after handler creation
1096          * with the element name and attributes. This configures
1097          * the element with its attributes and sets it up with
1098          * its parent container (if any). Nested elements are then
1099          * added later as the parser encounters them.
1100          *
1101          * @param uri The namespace URI for this element.
1102          * @param tag Name of the element which caused this handler
1103          *            to be created. Must not be <code>null</code>.
1104          * @param qname The qualified name for this element.
1105          * @param attrs Attributes of the element which caused this
1106          *              handler to be created. Must not be <code>null</code>.
1107          * @param context The current context.
1108          *
1109          * @exception SAXParseException in case of error (not thrown in
1110          *                              this implementation)
1111          */
1112         public void onStartElement(String uri, String tag, String qname, Attributes attrs,
1113                                    AntXMLContext context) throws SAXParseException {
1114             RuntimeConfigurable parentWrapper = context.currentWrapper();
1115             Object parent = null;
1116 
1117             if (parentWrapper != null) {
1118                 parent = parentWrapper.getProxy();
1119             }
1120 
1121             /* UnknownElement is used for tasks and data types - with
1122                delayed eval */
1123             UnknownElement task = new UnknownElement(tag);
1124             task.setProject(context.getProject());
1125             task.setNamespace(uri);
1126             task.setQName(qname);
1127             task.setTaskType(ProjectHelper.genComponentName(task.getNamespace(), tag));
1128             task.setTaskName(qname);
1129 
1130             Location location = new Location(context.getLocator().getSystemId(), context
1131                                              .getLocator().getLineNumber(), context.getLocator().getColumnNumber());
1132             task.setLocation(location);
1133             task.setOwningTarget(context.getCurrentTarget());
1134 
1135             if (parent != null) {
1136                 // Nested element
1137                 ((UnknownElement) parent).addChild(task);
1138             }  else {
1139                 // Task included in a target ( including the default one ).
1140                 context.getCurrentTarget().addTask(task);
1141             }
1142 
1143             context.configureId(task, attrs);
1144 
1145             // container.addTask(task);
1146             // This is a nop in UE: task.init();
1147 
1148             RuntimeConfigurable wrapper = new RuntimeConfigurable(task, task.getTaskName());
1149 
1150             for (int i = 0; i < attrs.getLength(); i++) {
1151                 String name = attrs.getLocalName(i);
1152                 String attrUri = attrs.getURI(i);
1153                 if (attrUri != null && !attrUri.equals("") && !attrUri.equals(uri)) {
1154                     name = attrUri + ":" + attrs.getQName(i);
1155                 }
1156                 String value = attrs.getValue(i);
1157                 // PR: Hack for ant-type value
1158                 //  an ant-type is a component name which can
1159                 // be namespaced, need to extract the name
1160                 // and convert from qualified name to uri/name
1161                 if (ANT_TYPE.equals(name)
1162                     || (ANT_CORE_URI.equals(attrUri)
1163                         && ANT_TYPE.equals(attrs.getLocalName(i)))) {
1164                     name = ANT_TYPE;
1165                     int index = value.indexOf(":");
1166                     if (index >= 0) {
1167                         String prefix = value.substring(0, index);
1168                         String mappedUri = context.getPrefixMapping(prefix);
1169                         if (mappedUri == null) {
1170                             throw new BuildException("Unable to find XML NS prefix \"" + prefix
1171                                                      + "\"");
1172                         }
1173                         value = ProjectHelper.genComponentName(mappedUri, value
1174                                                                .substring(index + 1));
1175                     }
1176                 }
1177                 wrapper.setAttribute(name, value);
1178             }
1179             if (parentWrapper != null) {
1180                 parentWrapper.addChild(wrapper);
1181             }
1182             context.pushWrapper(wrapper);
1183         }
