50  /**
51   * Unzip a file.
52   *
53   * @since Ant 1.1
54   *
55   * @ant.task category="packaging"
56   *           name="unzip"
57   *           name="unjar"
58   *           name="unwar"
59   */
60  public class Expand extends Task {
61      private static final int BUFFER_SIZE = 1024;
62      private File dest; //req
63      private File source; // req
64      private boolean overwrite = true;
65      private Mapper mapperElement = null;
66      private Vector<PatternSet> patternsets = new Vector<PatternSet>();
67      private Union resources = new Union();
68      private boolean resourcesSpecified = false;
69      private boolean failOnEmptyArchive = false;
70      private boolean stripAbsolutePathSpec = false;
71      private boolean scanForUnicodeExtraFields = true;

99      /**
100      * Do the work.
101      *
102      * @exception BuildException Thrown in unrecoverable error.
103      */
104     public void execute() throws BuildException {
105         if ("expand".equals(getTaskType())) {
106             log("!! expand is deprecated. Use unzip instead. !!");
107         }
108 
109         if (source == null && !resourcesSpecified) {
110             throw new BuildException("src attribute and/or resources must be "
111                                      + "specified");
112         }
113 
114         if (dest == null) {
115             throw new BuildException(
116                 "Dest attribute must be specified");
117         }
118 
119         if (dest.exists() && !dest.isDirectory()) {
120             throw new BuildException("Dest must be a directory.", getLocation());
121         }
122 
123         if (source != null) {
124             if (source.isDirectory()) {
125                 throw new BuildException("Src must not be a directory."
126                     + " Use nested filesets instead.", getLocation());
127             } else if (!source.exists()) {
128                 throw new BuildException("src '" + source + "' doesn't exist.");
129             } else if (!source.canRead()) {
130                 throw new BuildException("src '" + source + "' cannot be read.");
131             } else {
132                 expandFile(FILE_UTILS, source, dest);
133             }
134         }
135         for (Resource r : resources) {
136             if (!r.isExists()) {
137                 log("Skipping '" + r.getName() + "' because it doesn't exist.");
138                 continue;
139             }
140 
141             FileProvider fp = r.as(FileProvider.class);
142             if (fp != null) {
143                 expandFile(FILE_UTILS, fp.getFile(), dest);
144             } else {
145                 expandResource(r, dest);
146             }
147         }
148     }

150     /**
151      * This method is to be overridden by extending unarchival tasks.
152      *
153      * @param fileUtils the fileUtils
154      * @param srcF      the source file
155      * @param dir       the destination directory
156      */
157     protected void expandFile(FileUtils fileUtils, File srcF, File dir) {
158         log("Expanding: " + srcF + " into " + dir, Project.MSG_INFO);
159         ZipFile zf = null;
160         FileNameMapper mapper = getMapper();
161         if (!srcF.exists()) {
162             throw new BuildException("Unable to expand "
163                     + srcF
164                     + " as the file does not exist",
165                     getLocation());
166         }
167         try {
168             zf = new ZipFile(srcF, encoding, scanForUnicodeExtraFields);
169             boolean empty = true;
170             Enumeration<ZipEntry> e = zf.getEntries();
171             while (e.hasMoreElements()) {
172                 empty = false;
173                 ZipEntry ze = e.nextElement();
174                 InputStream is = null;
175                 log("extracting " + ze.getName(), Project.MSG_DEBUG);
176                 try {
177                     extractFile(fileUtils, srcF, dir,
178                                 is = zf.getInputStream(ze),
179                                 ze.getName(), new Date(ze.getTime()),
180                                 ze.isDirectory(), mapper);
181                 } finally {
182                     FileUtils.close(is);
183                 }
184             }
185             if (empty && getFailOnEmptyArchive()) {
186                 throw new BuildException("archive '" + srcF + "' is empty");
187             }
188             log("expand complete", Project.MSG_VERBOSE);
189         } catch (IOException ioe) {
190             throw new BuildException(
191                 "Error while expanding " + srcF.getPath()
192                 + "\n" + ioe.toString(),
193                 ioe);
194         } finally {
195             ZipFile.closeQuietly(zf);
196         }
197     }

210     /**
211      * get a mapper for a file
212      * @return a filenamemapper for a file
213      */
214     protected FileNameMapper getMapper() {
215         FileNameMapper mapper = null;
216         if (mapperElement != null) {
217             mapper = mapperElement.getImplementation();
218         } else {
219             mapper = new IdentityMapper();
220         }
221         return mapper;
222     }

225     /**
226      * extract a file to a directory
227      * @param fileUtils             a fileUtils object
228      * @param srcF                  the source file
229      * @param dir                   the destination directory
230      * @param compressedInputStream the input stream
231      * @param entryName             the name of the entry
232      * @param entryDate             the date of the entry
233      * @param isDirectory           if this is true the entry is a directory
234      * @param mapper                the filename mapper to use
235      * @throws IOException on error
236      */
237     protected void extractFile(FileUtils fileUtils, File srcF, File dir,
238                                InputStream compressedInputStream,
239                                String entryName, Date entryDate,
240                                boolean isDirectory, FileNameMapper mapper)
241                                throws IOException {
242 
243         if (stripAbsolutePathSpec && entryName.length() > 0
244             && (entryName.charAt(0) == File.separatorChar
245                 || entryName.charAt(0) == '/'
246                 || entryName.charAt(0) == '\\')) {
247             log("stripped absolute path spec from " + entryName,
248                 Project.MSG_VERBOSE);
249             entryName = entryName.substring(1);
250         }
251 
252         if (patternsets != null && patternsets.size() > 0) {
253             String name = entryName.replace('/', File.separatorChar)
254                 .replace('\\', File.separatorChar);
255 
256             boolean included = false;
257             Set<String> includePatterns = new HashSet<String>();
258             Set<String> excludePatterns = new HashSet<String>();
259             final int size = patternsets.size();
260             for (int v = 0; v < size; v++) {
261                 PatternSet p = patternsets.elementAt(v);
262                 String[] incls = p.getIncludePatterns(getProject());
263                 if (incls == null || incls.length == 0) {
264                     // no include pattern implicitly means includes="**"
265                     incls = new String[] {"**"};
266                 }
267 
268                 for (int w = 0; w < incls.length; w++) {
269                     String pattern = incls[w].replace('/', File.separatorChar)
270                         .replace('\\', File.separatorChar);
271                     if (pattern.endsWith(File.separator)) {
272                         pattern += "**";
273                     }
274                     includePatterns.add(pattern);
275                 }
276 
277                 String[] excls = p.getExcludePatterns(getProject());
278                 if (excls != null) {
279                     for (int w = 0; w < excls.length; w++) {
280                         String pattern = excls[w]
281                             .replace('/', File.separatorChar)
282                             .replace('\\', File.separatorChar);
283                         if (pattern.endsWith(File.separator)) {
284                             pattern += "**";
285                         }
286                         excludePatterns.add(pattern);
287                     }
288                 }
289             }
290 
291             for (Iterator<String> iter = includePatterns.iterator();
292                  !included && iter.hasNext();) {
293                 String pattern = iter.next();
294                 included = SelectorUtils.matchPath(pattern, name);
295             }
296 
297             for (Iterator<String> iter = excludePatterns.iterator();
298                  included && iter.hasNext();) {
299                 String pattern = iter.next();
300                 included = !SelectorUtils.matchPath(pattern, name);
301             }
302 
303             if (!included) {
304                 //Do not process this file
305                 log("skipping " + entryName
306                     + " as it is excluded or not included.",
307                     Project.MSG_VERBOSE);
308                 return;
309             }
310         }
311         String[] mappedNames = mapper.mapFileName(entryName);
312         if (mappedNames == null || mappedNames.length == 0) {
313             mappedNames = new String[] {entryName};
314         }
315         File f = fileUtils.resolveFile(dir, mappedNames[0]);
316         try {
317             if (!overwrite && f.exists()
318                 && f.lastModified() >= entryDate.getTime()) {
319                 log("Skipping " + f + " as it is up-to-date",
320                     Project.MSG_DEBUG);
321                 return;
322             }
323 
324             log("expanding " + entryName + " to " + f,
325                 Project.MSG_VERBOSE);
326             // create intermediary directories - sometimes zip don't add them
327             File dirF = f.getParentFile();
328             if (dirF != null) {
329                 dirF.mkdirs();
330             }
331 
332             if (isDirectory) {
333                 f.mkdirs();
334             } else {
335                 byte[] buffer = new byte[BUFFER_SIZE];
336                 int length = 0;
337                 FileOutputStream fos = null;
338                 try {
339                     fos = new FileOutputStream(f);
340 
341                     while ((length =
342                             compressedInputStream.read(buffer)) >= 0) {
343                         fos.write(buffer, 0, length);
344                     }
345 
346                     fos.close();
347                     fos = null;
348                 } finally {
349                     FileUtils.close(fos);
350                 }
351             }
352 
353             fileUtils.setFileLastModified(f, entryDate.getTime());
354         } catch (FileNotFoundException ex) {
355             log("Unable to expand to file " + f.getPath(),
356                     ex,
357                     Project.MSG_WARN);
358         }
359 
360     }

417     /**
418      * Defines the mapper to map source entries to destination files.
419      * @return a mapper to be configured
420      * @exception BuildException if more than one mapper is defined
421      * @since Ant1.7
422      */
423     public Mapper createMapper() throws BuildException {
424         if (mapperElement != null) {
425             throw new BuildException(ERROR_MULTIPLE_MAPPERS,
426                                      getLocation());
427         }
428         mapperElement = new Mapper(getProject());
429         return mapperElement;
430     }
