44  /**
45   * Compiles Java source files. This task can take the following
46   * arguments:
47   * <ul>
48   * <li>sourcedir
49   * <li>destdir
50   * <li>deprecation
51   * <li>classpath
52   * <li>bootclasspath
53   * <li>extdirs
54   * <li>optimize
55   * <li>debug
56   * <li>encoding
57   * <li>target
58   * <li>depend
59   * <li>verbose
60   * <li>failonerror
61   * <li>includeantruntime
62   * <li>includejavaruntime
63   * <li>source
64   * <li>compiler
65   * </ul>
66   * Of these arguments, the <b>sourcedir</b> and <b>destdir</b> are required.
67   * <p>
68   * When this task executes, it will recursively scan the sourcedir and
69   * destdir looking for Java source files to compile. This task makes its
70   * compile decision based on timestamp.
71   *
72   *
73   * @since Ant 1.1
74   *
75   * @ant.task category="java"
76   */

145     private String assumedJavaVersion() {
146         if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_4)) {
147             return JAVAC14;
148         } else if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_5)) {
149             return JAVAC15;
150         } else if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_6)) {
151             return JAVAC16;
152         } else if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_7)) {
153             return JAVAC17;
154         } else if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_8)) {
155             return JAVAC18;
156         } else if (JavaEnvUtils.isJavaVersion(JavaEnvUtils.JAVA_1_9)) {
157             return JAVAC19;
158         } else {
159             return CLASSIC;
160         }
161     }

745     /**
746      * Get the additional implementation specific command line arguments.
747      * @return array of command line arguments, guaranteed to be non-null.
748      */
749     public String[] getCurrentCompilerArgs() {
750         String chosen = facade.getExplicitChoice();
751         try {
752             // make sure facade knows about magic properties and fork setting
753             String appliedCompiler = getCompiler();
754             facade.setImplementation(appliedCompiler);
755 
756             String[] result = facade.getArgs();
757 
758             String altCompilerName = getAltCompilerName(facade.getImplementation());
759 
760             if (result.length == 0 && altCompilerName != null) {
761                 facade.setImplementation(altCompilerName);
762                 result = facade.getArgs();
763             }
764 
765             return result;
766 
767         } finally {
768             facade.setImplementation(chosen);
769         }
770     }

772     private String getAltCompilerName(String anImplementation) {
773         if (JAVAC19.equalsIgnoreCase(anImplementation)
774                 || JAVAC18.equalsIgnoreCase(anImplementation)
775                 || JAVAC17.equalsIgnoreCase(anImplementation)
776                 || JAVAC16.equalsIgnoreCase(anImplementation)
777                 || JAVAC15.equalsIgnoreCase(anImplementation)
778                 || JAVAC14.equalsIgnoreCase(anImplementation)
779                 || JAVAC13.equalsIgnoreCase(anImplementation)) {
780             return MODERN;
781         }
782         if (JAVAC12.equalsIgnoreCase(anImplementation)
783                 || JAVAC11.equalsIgnoreCase(anImplementation)) {
784             return CLASSIC;
785         }
786         if (MODERN.equalsIgnoreCase(anImplementation)) {
787             String nextSelected = assumedJavaVersion();
788             if (JAVAC19.equalsIgnoreCase(nextSelected)
789                     || JAVAC18.equalsIgnoreCase(nextSelected)
790                     || JAVAC17.equalsIgnoreCase(nextSelected)
791                     || JAVAC16.equalsIgnoreCase(nextSelected)
792                     || JAVAC15.equalsIgnoreCase(nextSelected)
793                     || JAVAC14.equalsIgnoreCase(nextSelected)
794                     || JAVAC13.equalsIgnoreCase(nextSelected)) {
795                 return nextSelected;
796             }
797         }
798         if (CLASSIC.equalsIgnoreCase(anImplementation)) {
799             return assumedJavaVersion();
800         }
801         if (EXTJAVAC.equalsIgnoreCase(anImplementation)) {
802             return assumedJavaVersion();
803         }
804         return null;
805     }

910     /**
911      * Executes the task.
912      * @exception BuildException if an error occurs
913      */
914     public void execute() throws BuildException {
915         checkParameters();
916         resetFileLists();
917 
918         // scan source directories and dest directory to build up
919         // compile lists
920         String[] list = src.list();
921         for (int i = 0; i < list.length; i++) {
922             File srcDir = getProject().resolveFile(list[i]);
923             if (!srcDir.exists()) {
924                 throw new BuildException("srcdir \""
925                                          + srcDir.getPath()
926                                          + "\" does not exist!", getLocation());
927             }
928 
929             DirectoryScanner ds = this.getDirectoryScanner(srcDir);
930             String[] files = ds.getIncludedFiles();
931 
932             scanDir(srcDir, destDir != null ? destDir : srcDir, files);
933         }
934 
935         compile();
936         if (updatedProperty != null
937             && taskSuccess
938             && compileList.length != 0) {
939             getProject().setNewProperty(updatedProperty, "true");
940         }
941     }

951     /**
952      * Scans the directory looking for source files to be compiled.
953      * The results are returned in the class variable compileList
954      *
955      * @param srcDir   The source directory
956      * @param destDir  The destination directory
957      * @param files    An array of filenames
958      */
959     protected void scanDir(File srcDir, File destDir, String[] files) {
960         GlobPatternMapper m = new GlobPatternMapper();
961         String[] extensions = findSupportedFileExtensions();
962 
963         for (int i = 0; i < extensions.length; i++) {
964             m.setFrom(extensions[i]);
965             m.setTo("*.class");
966             SourceFileScanner sfs = new SourceFileScanner(this);
967             File[] newFiles = sfs.restrictAsFiles(files, srcDir, destDir, m);
968 
969             if (newFiles.length > 0) {
970                 lookForPackageInfos(srcDir, newFiles);
971                 File[] newCompileList
972                     = new File[compileList.length + newFiles.length];
973                 System.arraycopy(compileList, 0, newCompileList, 0,
974                                  compileList.length);
975                 System.arraycopy(newFiles, 0, newCompileList,
976                                  compileList.length, newFiles.length);
977                 compileList = newCompileList;
978             }
979         }
980     }

982     private String[] findSupportedFileExtensions() {
983         String compilerImpl = getCompiler();
984         CompilerAdapter adapter =
985             nestedAdapter != null ? nestedAdapter :
986             CompilerAdapterFactory.getCompiler(compilerImpl, this,
987                                                createCompilerClasspath());
988         String[] extensions = null;
989         if (adapter instanceof CompilerAdapterExtension) {
990             extensions =
991                 ((CompilerAdapterExtension) adapter).getSupportedFileExtensions();
992         }
993 
994         if (extensions == null) {
995             extensions = new String[] { "java" };
996         }
997 
998         // now process the extensions to ensure that they are the
999         // right format
1000         for (int i = 0; i < extensions.length; i++) {
1001             if (!extensions[i].startsWith("*.")) {
1002                 extensions[i] = "*." + extensions[i];
1003             }
1004         }
1005         return extensions;
1006     }

1130     /**
1131      * Perform the compilation.
1132      *
1133      * @since Ant 1.5
1134      */
1135     protected void compile() {
1136         String compilerImpl = getCompiler();
1137 
1138         if (compileList.length > 0) {
1139             log("Compiling " + compileList.length + " source file"
1140                 + (compileList.length == 1 ? "" : "s")
1141                 + (destDir != null ? " to " + destDir : ""));
1142 
1143             if (listFiles) {
1144                 for (int i = 0; i < compileList.length; i++) {
1145                   String filename = compileList[i].getAbsolutePath();
1146                   log(filename);
1147                 }
1148             }
1149 
1150             CompilerAdapter adapter =
1151                 nestedAdapter != null ? nestedAdapter :
1152                 CompilerAdapterFactory.getCompiler(compilerImpl, this,
1153                                                    createCompilerClasspath());
1154 
1155             // now we need to populate the compiler adapter
1156             adapter.setJavac(this);
1157 
1158             // finally, lets execute the compiler!!
1159             if (adapter.execute()) {
1160                 // Success
1161                 if (createMissingPackageInfoClass) {
1162                     try {
1163                         generateMissingPackageInfoClasses(destDir != null
1164                                                           ? destDir
1165                                                           : getProject()
1166                                                           .resolveFile(src.list()[0]));
1167                     } catch (IOException x) {
1168                         // Should this be made a nonfatal warning?
1169                         throw new BuildException(x, getLocation());
1170                     }
1171                 }
1172             } else {
1173                 // Fail path
1174                 this.taskSuccess = false;
1175                 if (errorProperty != null) {
1176                     getProject().setNewProperty(
1177                         errorProperty, "true");
1178                 }
1179                 if (failOnError) {
1180                     throw new BuildException(FAIL_MSG, getLocation());
1181                 } else {
1182                     log(FAIL_MSG, Project.MSG_ERR);
1183                 }
1184             }
1185         }
1186     }

1204     private void lookForPackageInfos(File srcDir, File[] newFiles) {
1205         for (int i = 0; i < newFiles.length; i++) {
1206             File f = newFiles[i];
1207             if (!f.getName().equals("package-info.java")) {
1208                 continue;
1209             }
1210             String path = FILE_UTILS.removeLeadingPath(srcDir, f).
1211                     replace(File.separatorChar, '/');
1212             String suffix = "/package-info.java";
1213             if (!path.endsWith(suffix)) {
1214                 log("anomalous package-info.java path: " + path, Project.MSG_WARN);
1215                 continue;
1216             }
1217             String pkg = path.substring(0, path.length() - suffix.length());
1218             packageInfos.put(pkg, new Long(f.lastModified()));
1219         }
1220     }

1222     /**
1223      * Ensure that every {@code package-info.java} produced a {@code package-info.class}.
1224      * Otherwise this task's up-to-date tracking mechanisms do not work.
1225      * @see <a href="https://issues.apache.org/bugzilla/show_bug.cgi?id=43114">Bug #43114</a>
1226      */
1227     private void generateMissingPackageInfoClasses(File dest) throws IOException {
1228         for (Entry<String, Long> entry : packageInfos.entrySet()) {
1229             String pkg = entry.getKey();
1230             Long sourceLastMod = entry.getValue();
1231             File pkgBinDir = new File(dest, pkg.replace('/', File.separatorChar));
1232             pkgBinDir.mkdirs();
1233             File pkgInfoClass = new File(pkgBinDir, "package-info.class");
1234             if (pkgInfoClass.isFile() && pkgInfoClass.lastModified() >= sourceLastMod.longValue()) {
1235                 continue;
1236             }
1237             log("Creating empty " + pkgInfoClass);
1238             OutputStream os = new FileOutputStream(pkgInfoClass);
1239             try {
1240                 os.write(PACKAGE_INFO_CLASS_HEADER);
1241                 byte[] name = pkg.getBytes("UTF-8");
1242                 int length = name.length + /* "/package-info" */ 13;
1243                 os.write((byte) length / 256);
1244                 os.write((byte) length % 256);
1245                 os.write(name);
1246                 os.write(PACKAGE_INFO_CLASS_FOOTER);
1247             } finally {
1248                 os.close();
1249             }
1250         }
1251     }
