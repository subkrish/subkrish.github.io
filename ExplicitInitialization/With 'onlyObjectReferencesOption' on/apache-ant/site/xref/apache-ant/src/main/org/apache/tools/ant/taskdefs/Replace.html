137         /**
138          * Validate the filter's configuration.
139          * @throws BuildException if any part is invalid.
140          */
141         public void validate() throws BuildException {
142             //Validate mandatory attributes
143             if (token == null) {
144                 String message = "token is a mandatory for replacefilter.";
145                 throw new BuildException(message);
146             }
147 
148             if ("".equals(token.getText())) {
149                 String message = "The token must not be an empty "
150                     + "string.";
151                 throw new BuildException(message);
152             }
153 
154             //value and property are mutually exclusive attributes
155             if ((value != null) && (property != null)) {
156                 String message = "Either value or property "
157                     + "can be specified, but a replacefilter "
158                     + "element cannot have both.";
159                 throw new BuildException(message);
160             }
161 
162             if ((property != null)) {
163                 //the property attribute must have access to a property file
164                 if (propertyResource == null) {
165                     String message = "The replacefilter's property attribute "
166                         + "can only be used with the replacetask's "
167                         + "propertyFile/Resource attribute.";
168                     throw new BuildException(message);
169                 }
170 
171                 //Make sure property exists in property file
172                 if (properties == null
173                     || properties.getProperty(property) == null) {
174                     String message = "property \"" + property
175                         + "\" was not found in " + propertyResource.getName();
176                     throw new BuildException(message);
177                 }
178             }
179 
180             replaceValue = getReplaceValue();
181         }
182 
183         /**
184          * Get the replacement value for this filter token.
185          * @return the replacement value
186          */
187         public String getReplaceValue() {
188             if (property != null) {
189                 return properties.getProperty(property);
190             } else if (value != null) {
191                 return value.getText();
192             } else if (Replace.this.value != null) {
193                 return Replace.this.value.getText();
194             } else {
195                 //Default is empty string
196                 return "";
197             }
198         }

275         /**
276          * Retrieves the output buffer of this filter. The filter guarantees
277          * that data is only appended to the end of this StringBuffer.
278          * @return The StringBuffer containing the output of this filter.
279          */
280         StringBuffer getOutputBuffer() {
281             return outputBuffer;
282         }
283 
284         /**
285          * Sets the input buffer for this filter.
286          * The filter expects from the component providing the input that data
287          * is only added by that component to the end of this StringBuffer.
288          * This StringBuffer will be modified by this filter, and expects that
289          * another component will only apped to this StringBuffer.
290          * @param input The input for this filter.
291          */
292         void setInputBuffer(StringBuffer input) {
293             inputBuffer = input;
294         }
295 
296         /**
297          * Processes the buffer as far as possible. Takes into account that
298          * appended data may make it possible to replace the end of the already
299          * received data, when the token is split over the "old" and the "new"
300          * part.
301          * @return true if some data has been made available in the
302          *         output buffer.
303          */
304         boolean process() {
305             String t = getToken();
306             if (inputBuffer.length() > t.length()) {
307                 int pos = replace();
308                 pos = Math.max((inputBuffer.length() - t.length()), pos);
309                 outputBuffer.append(inputBuffer.substring(0, pos));
310                 inputBuffer.delete(0, pos);
311                 return true;
312             }
313             return false;
314         }
315 
316         /**
317          * Processes the buffer to the end. Does not take into account that
318          * appended data may make it possible to replace the end of the already
319          * received data.
320          */
321         void flush() {
322             replace();
323             outputBuffer.append(inputBuffer);
324             inputBuffer.delete(0, inputBuffer.length());
325         }
326 
327         /**
328          * Performs the replace operation.
329          * @return The position of the last character that was inserted as
330          *         replacement.
331          */
332         private int replace() {
333             String t = getToken();
334             int found = inputBuffer.indexOf(t);
335             int pos = -1;
336             final int tokenLength = t.length();
337             final int replaceValueLength = replaceValue.length();
338             while (found >= 0) {
339                 inputBuffer.replace(found, found + tokenLength, replaceValue);
340                 pos = found + replaceValueLength;
341                 found = inputBuffer.indexOf(t, pos);
342                 ++replaceCount;
343             }
344             return pos;
345         }
346     }

350      * a StringBuffer. Compatible with the Replacefilter.
351      * @since 1.7
352      */
353     private class FileInput /* JDK 5: implements Closeable */ {
354         private StringBuffer outputBuffer;
355         private final InputStream is;
356         private Reader reader;
357         private char[] buffer;
358         private static final int BUFF_SIZE = 4096;
359 
360         /**
361          * Constructs the input component. Opens the file for reading.
362          * @param source The file to read from.
363          * @throws IOException When the file cannot be read from.
364          */
365         FileInput(File source) throws IOException {
366             outputBuffer = new StringBuffer();
367             buffer = new char[BUFF_SIZE];
368             is = new FileInputStream(source);
369             try {
370                 reader = new BufferedReader(encoding != null ? new InputStreamReader(is, encoding) : new InputStreamReader(is));
371             } finally {
372                 if (reader == null) {
373                     is.close();
374                 }
375             }
376         }
377 
378         /**
379          * Retrieves the output buffer of this filter. The component guarantees
380          * that data is only appended to the end of this StringBuffer.
381          * @return The StringBuffer containing the output of this filter.
382          */
383         StringBuffer getOutputBuffer() {
384             return outputBuffer;
385         }
386 
387         /**
388          * Reads some data from the file.
389          * @return true when the end of the file has not been reached.
390          * @throws IOException When the file cannot be read from.
391          */
392         boolean readChunk() throws IOException {
393             int bufferLength = 0;
394             bufferLength = reader.read(buffer);
395             if (bufferLength < 0) {
396                 return false;
397             }
398             outputBuffer.append(new String(buffer, 0, bufferLength));
399             return true;
400         }
401 
402         /**
403          * Closes the file.
404          * @throws IOException When the file cannot be closed.
405          */
406         public void close() throws IOException {
407             is.close();
408         }

482     /**
483      * Do the execution.
484      * @throws BuildException if we cant build
485      */
486     public void execute() throws BuildException {
487 
488         ArrayList savedFilters = (ArrayList) replacefilters.clone();
489         Properties savedProperties =
490             properties == null ? null : (Properties) properties.clone();
491 
492         if (token != null) {
493             // line separators in values and tokens are "\n"
494             // in order to compare with the file contents, replace them
495             // as needed
496             StringBuffer val = new StringBuffer(value.getText());
497             stringReplace(val, "\r\n", "\n");
498             stringReplace(val, "\n", StringUtils.LINE_SEP);
499             StringBuffer tok = new StringBuffer(token.getText());
500             stringReplace(tok, "\r\n", "\n");
501             stringReplace(tok, "\n", StringUtils.LINE_SEP);
502             Replacefilter firstFilter = createPrimaryfilter();
503             firstFilter.setToken(tok.toString());
504             firstFilter.setValue(val.toString());
505         }
506 
507         try {
508             if (replaceFilterResource != null) {
509                 Properties props = getProperties(replaceFilterResource);
510                 Iterator e = props.keySet().iterator();
511                 while (e.hasNext()) {
512                     String tok =  e.next().toString();
513                     Replacefilter replaceFilter = createReplacefilter();
514                     replaceFilter.setToken(tok);
515                     replaceFilter.setValue(props.getProperty(tok));
516                 }
517             }
518 
519             validateAttributes();
520 
521             if (propertyResource != null) {
522                 properties = getProperties(propertyResource);
523             }
524 
525             validateReplacefilters();
526             fileCount = 0;
527             replaceCount = 0;
528 
529             if (sourceFile != null) {
530                 processFile(sourceFile);
531             }
532 
533             if (dir != null) {
534                 DirectoryScanner ds = super.getDirectoryScanner(dir);
535                 String[] srcs = ds.getIncludedFiles();
536 
537                 for (int i = 0; i < srcs.length; i++) {
538                     File file = new File(dir, srcs[i]);
539                     processFile(file);
540                 }
541             }
542 
543             if (resources != null) {
544                 for (Resource r : resources) {
545                     FileProvider fp =
546                     r.as(FileProvider.class);
547                     processFile(fp.getFile());
548                 }
549             }
550 
551             if (summary) {
552                 log("Replaced " + replaceCount + " occurrences in "
553                     + fileCount + " files.", Project.MSG_INFO);
554             }
555             if (failOnNoReplacements && replaceCount == 0) {
556                 throw new BuildException("didn't replace anything");
557             }
558         } finally {
559             replacefilters = savedFilters;
560             properties = savedProperties;
561         } // end of finally
562 
563     }

566      * Validate attributes provided for this task in .xml build file.
567      *
568      * @exception BuildException if any supplied attribute is invalid or any
569      * mandatory attribute is missing.
570      */
571     public void validateAttributes() throws BuildException {
572         if (sourceFile == null && dir == null && resources == null) {
573             String message = "Either the file or the dir attribute "
574                 + "or nested resources must be specified";
575             throw new BuildException(message, getLocation());
576         }
577         if (propertyResource != null && !propertyResource.isExists()) {
578             String message = "Property file " + propertyResource.getName()
579                 + " does not exist.";
580             throw new BuildException(message, getLocation());
581         }
582         if (token == null && replacefilters.size() == 0) {
583             String message = "Either token or a nested replacefilter "
584                 + "must be specified";
585             throw new BuildException(message, getLocation());
586         }
587         if (token != null && "".equals(token.getText())) {
588             String message = "The token attribute must not be an empty string.";
589             throw new BuildException(message, getLocation());
590         }
591     }

645     /**
646      * Perform the replacement on the given file.
647      *
648      * The replacement is performed on a temporary file which then
649      * replaces the original file.
650      *
651      * @param src the source <code>File</code>.
652      */
653     private void processFile(File src) throws BuildException {
654         if (!src.exists()) {
655             throw new BuildException("Replace: source file " + src.getPath()
656                                      + " doesn't exist", getLocation());
657         }
658 
659         int repCountStart = replaceCount;
660         logFilterChain(src.getPath());
661 
662         try {
663             File temp = FILE_UTILS.createTempFile("rep", ".tmp",
664                     src.getParentFile(), false, true);
665             try {
666                 FileInput in = new FileInput(src);
667                 try {
668                     FileOutput out = new FileOutput(temp);
669                     try {
670                         out.setInputBuffer(buildFilterChain(in.getOutputBuffer()));
671 
672                         while (in.readChunk()) {
673                             if (processFilterChain()) {
674                                 out.process();
675                             }
676                         }
677 
678                         flushFilterChain();
679 
680                         out.flush();
681                     } finally {
682                         out.close();
683                     }
684                 } finally {
685                     in.close();
686                 }
687                 boolean changes = (replaceCount != repCountStart);
688                 if (changes) {
689                     fileCount++;
690                     long origLastModified = src.lastModified();
691                     FILE_UTILS.rename(temp, src);
692                     if (preserveLastModified) {
693                         FILE_UTILS.setFileLastModified(src, origLastModified);
694                     }
695                 }
696             } finally {
697                 if (temp.isFile() && !temp.delete()) {
698                     temp.deleteOnExit();
699                 }
700             }
701         } catch (IOException ioe) {
702             throw new BuildException("IOException in " + src + " - "
703                     + ioe.getClass().getName() + ":"
704                     + ioe.getMessage(), ioe, getLocation());
705         }
706     }

708     /**
709      * Flushes all filters.
710      */
711     private void flushFilterChain() {
712         final int size = replacefilters.size();
713         for (int i = 0; i < size; i++) {
714             Replacefilter filter = (Replacefilter) replacefilters.get(i);
715             filter.flush();
716         }
717     }

719     /**
720      * Performs the normal processing of the filters.
721      * @return true if the filter chain produced new output.
722      */
723     private boolean processFilterChain() {
724         final int size = replacefilters.size();
725         for (int i = 0; i < size; i++) {
726             Replacefilter filter = (Replacefilter) replacefilters.get(i);
727             if (!filter.process()) {
728                 return false;
729             }
730         }
731         return true;
732     }

734     /**
735      * Creates the chain of filters to operate.
736      * @param inputBuffer <code>StringBuffer</code> containing the input for the
737      *                    first filter.
738      * @return <code>StringBuffer</code> containing the output of the last filter.
739      */
740     private StringBuffer buildFilterChain(StringBuffer inputBuffer) {
741         StringBuffer buf = inputBuffer;
742         final int size = replacefilters.size();
743         for (int i = 0; i < size; i++) {
744             Replacefilter filter = (Replacefilter) replacefilters.get(i);
745             filter.setInputBuffer(buf);
746             buf = filter.getOutputBuffer();
747         }
748         return buf;
749     }

751     /**
752      * Logs the chain of filters to operate on the file.
753      * @param filename <code>String</code>.
754      */
755     private void logFilterChain(String filename) {
756         final int size = replacefilters.size();
757         for (int i = 0; i < size; i++) {
758             Replacefilter filter = (Replacefilter) replacefilters.get(i);
759             log("Replacing in " + filename + ": " + filter.getToken()

761         }

941     /**
942      * Replace occurrences of str1 in StringBuffer str with str2.
943      */
944     private void stringReplace(StringBuffer str, String str1, String str2) {
945         int found = str.indexOf(str1);
946         final int str1Length = str1.length();
947         final int str2Length = str2.length();
948         while (found >= 0) {
949             str.replace(found, found + str1Length, str2);
950             found = str.indexOf(str1, found + str2Length);
951         }
952     }
