45  /**
46   * Touch a file and/or fileset(s) and/or filelist(s);
47   * corresponds to the Unix touch command.
48   *
49   * <p>If the file to touch doesn't exist, an empty one is created.</p>
50   *
51   * @since Ant 1.1
52   *
53   * @ant.task category="filesystem"
54   */
55  public class Touch extends Task {

57      public interface DateFormatFactory {
58          DateFormat getPrimaryFormat();
59          DateFormat getFallbackFormat();
60      }
61  
62      public static final DateFormatFactory DEFAULT_DF_FACTORY
63          = new DateFormatFactory() {
64          /*
65           * The initial version used DateFormat.SHORT for the
66           * time format, which ignores seconds.  If we want
67           * seconds as well, we need DateFormat.MEDIUM, which
68           * in turn would break all old build files.
69           *
70           * First try to parse with DateFormat.SHORT and if
71           * that fails with MEDIUM - throw an exception if both
72           * fail.
73           */
74          public DateFormat getPrimaryFormat() {
75              return DateFormat.getDateTimeInstance(DateFormat.SHORT,
76                  DateFormat.SHORT, Locale.US);
77          }
78          public DateFormat getFallbackFormat() {
79              return DateFormat.getDateTimeInstance(DateFormat.SHORT,
80                  DateFormat.MEDIUM, Locale.US);
81          }
82      };

221     /**
222      * Check that this task has been configured properly.
223      * @throws BuildException if configuration errors are detected.
224      * @since Ant 1.6.3
225      */
226     protected synchronized void checkConfiguration() throws BuildException {
227         if (file == null && resources == null) {
228             throw new BuildException("Specify at least one source"
229                                    + "--a file or resource collection.");
230         }
231         if (file != null && file.exists() && file.isDirectory()) {
232             throw new BuildException("Use a resource collection to touch directories.");
233         }
234         if (dateTime != null && !dateTimeConfigured) {
235             long workmillis = millis;
236             if ("now".equalsIgnoreCase(dateTime)) {
237                 workmillis = System.currentTimeMillis();
238             } else {
239                 DateFormat df = dfFactory.getPrimaryFormat();
240                 ParseException pe = null;
241                 try {
242                     workmillis = df.parse(dateTime).getTime();
243                 } catch (ParseException peOne) {
244                     df = dfFactory.getFallbackFormat();
245                     if (df == null) {
246                         pe = peOne;
247                     } else {
248                         try {
249                             workmillis = df.parse(dateTime).getTime();
250                         } catch (ParseException peTwo) {
251                             pe = peTwo;
252                         }
253                     }
254                 }
255                 if (pe != null) {
256                     throw new BuildException(pe.getMessage(), pe, getLocation());
257                 }
258                 if (workmillis < 0) {
259                     throw new BuildException("Date of " + dateTime
260                             + " results in negative " + "milliseconds value "
261                             + "relative to epoch " + "(January 1, 1970, "
262                             + "00:00:00 GMT).");
263                 }
264             }
265             log("Setting millis to " + workmillis + " from datetime attribute",
266                     ((millis < 0) ? Project.MSG_DEBUG : Project.MSG_VERBOSE));
267             setMillis(workmillis);
268             // only set if successful to this point:
269             dateTimeConfigured = true;
270         }
271     }

284     /**
285      * Does the actual work; assumes everything has been checked by now.
286      * @throws BuildException if an error occurs.
287      */
288     protected void touch() throws BuildException {
289         long defaultTimestamp = getTimestamp();
290 
291         if (file != null) {
292             touch(new FileResource(file.getParentFile(), file.getName()),
293                   defaultTimestamp);
294         }
295         if (resources == null) {
296             return;
297         }
298         // deal with the resource collections
299         for (Resource r : resources) {
300             Touchable t = r.as(Touchable.class);
301             if (t == null) {
302                 throw new BuildException("Can't touch " + r);
303             }
304             touch(r, defaultTimestamp);
305         }
306 
307         // deal with filesets in a special way since the task
308         // originally also used the directories and Union won't return
309         // them.
310         final int size = filesets.size();
311         for (int i = 0; i < size; i++) {
312             FileSet fs = (FileSet) filesets.elementAt(i);
313             DirectoryScanner ds = fs.getDirectoryScanner(getProject());
314             File fromDir = fs.getDir(getProject());
315 
316             String[] srcDirs = ds.getIncludedDirectories();
317 
318             for (int j = 0; j < srcDirs.length; j++) {
319                 touch(new FileResource(fromDir, srcDirs[j]), defaultTimestamp);
320             }
321         }
322     }

324     /**
325      * Touch a single file with the current timestamp (this.millis). This method
326      * does not interact with any nested mappers and remains for reasons of
327      * backwards-compatibility only.
328      * @param file file to touch
329      * @throws BuildException on error
330      * @deprecated since 1.6.x.
331      */
332     protected void touch(File file) {
333         touch(file, getTimestamp());
334     }
335 
336     private long getTimestamp() {
337         return (millis < 0) ? System.currentTimeMillis() : millis;
338     }
339 
340     private void touch(Resource r, long defaultTimestamp) {
341         if (fileNameMapper == null) {
342             FileProvider fp = r.as(FileProvider.class);
343             if (fp != null) {
344                 // use this to create file and deal with non-writable files
345                 touch(fp.getFile(), defaultTimestamp);
346             } else {
347                 r.as(Touchable.class).touch(defaultTimestamp);
348             }
349         } else {
350             String[] mapped = fileNameMapper.mapFileName(r.getName());
351             if (mapped != null && mapped.length > 0) {
352                 long modTime = defaultTimestamp;
353                 if (millis < 0 && r.isExists()) {
354                     modTime = r.getLastModified();
355                 }
356                 for (int i = 0; i < mapped.length; i++) {
357                     touch(getProject().resolveFile(mapped[i]), modTime);
358                 }
359             }
360         }
361     }
362 
363     private void touch(File file, long modTime) {
364         if (!file.exists()) {
365             log("Creating " + file,
366                 ((verbose) ? Project.MSG_INFO : Project.MSG_VERBOSE));
367             try {
368                 FILE_UTILS.createNewFile(file, mkdirs);
369             } catch (IOException ioe) {
370                 throw new BuildException("Could not create " + file, ioe,
371                                          getLocation());
372             }
373         }
374         if (!file.canWrite()) {
375             throw new BuildException("Can not change modification date of "
376                                      + "read-only file " + file);
377         }
378         FILE_UTILS.setFileLastModified(file, modTime);
379     }
