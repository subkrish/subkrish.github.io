274     /**
275      * Execute the command on the remote host.
276      *
277      * @exception BuildException  Most likely a network error or bad parameter.
278      */
279     public void execute() throws BuildException {
280 
281         if (getHost() == null) {
282             throw new BuildException("Host is required.");
283         }
284         if (getUserInfo().getName() == null) {
285             throw new BuildException("Username is required.");
286         }
287         if (getUserInfo().getKeyfile() == null
288             && getUserInfo().getPassword() == null) {
289             throw new BuildException("Password or Keyfile is required.");
290         }
291         if (command == null && commandResource == null) {
292             throw new BuildException("Command or commandResource is required.");
293         }
294 
295         int numberOfInputs = (inputFile != null ? 1 : 0)
296             + (inputProperty != null ? 1 : 0)
297             + (inputString != null ? 1 : 0);
298         if (numberOfInputs > 1) {
299             throw new BuildException("You can't specify more than one of"
300                                      + " inputFile, inputProperty and"
301                                      + " inputString.");
302         }
303         if (inputFile != null && !inputFile.exists()) {
304             throw new BuildException("The input file "
305                                      + inputFile.getAbsolutePath()
306                                      + " does not exist.");
307         }
308 
309         Session session = null;
310         StringBuffer output = new StringBuffer();
311         try {
312             session = openSession();
313             /* called once */
314             if (command != null) {
315                 log("cmd : " + command, Project.MSG_INFO);
316                 executeCommand(session, command, output);
317             } else { // read command resource and execute for each command
318                 try {
319                     BufferedReader br = new BufferedReader(
320                             new InputStreamReader(commandResource.getInputStream()));
321                     String cmd;
322                     while ((cmd = br.readLine()) != null) {
323                         log("cmd : " + cmd, Project.MSG_INFO);
324                         output.append(cmd).append(" : ");
325                         executeCommand(session, cmd, output);
326                         output.append("\n");
327                     }
328                     FileUtils.close(br);
329                 } catch (IOException e) {
330                     if (getFailonerror()) {
331                         throw new BuildException(e);
332                     } else {
333                         log("Caught exception: " + e.getMessage(),
334                             Project.MSG_ERR);
335                     }
336                 }
337             }
338         } catch (JSchException e) {
339             if (getFailonerror()) {
340                 throw new BuildException(e);
341             } else {
342                 log("Caught exception: " + e.getMessage(), Project.MSG_ERR);
343             }
344         } finally {
345             if (outputProperty != null) {
346                 getProject().setNewProperty(outputProperty, output.toString());
347             }
348             if (session != null && session.isConnected()) {
349                 session.disconnect();
350             }
351         }
352     }

354     private void executeCommand(Session session, String cmd, StringBuffer sb)
355         throws BuildException {
356         ByteArrayOutputStream out = new ByteArrayOutputStream();
357         ByteArrayOutputStream errout = new ByteArrayOutputStream();
358         OutputStream teeErr = suppressSystemErr ? errout : new TeeOutputStream(errout, KeepAliveOutputStream.wrapSystemErr());
359         OutputStream tee = suppressSystemOut ? out : new TeeOutputStream(out, KeepAliveOutputStream.wrapSystemOut());
360 
361         InputStream istream = null ;
362         if (inputFile != null) {
363             try {
364                 istream = new FileInputStream(inputFile) ;
365             } catch (IOException e) {
366                 // because we checked the existence before, this one
367                 // shouldn't happen What if the file exists, but there
368                 // are no read permissions?
369                 log("Failed to read " + inputFile + " because of: "
370                     + e.getMessage(), Project.MSG_WARN);
371             }
372         }
373         if (inputProperty != null) {
374             String inputData = getProject().getProperty(inputProperty) ;
375             if (inputData != null) {
376                 istream = new ByteArrayInputStream(inputData.getBytes()) ;
377             }
378         }
379         if (inputString != null) {
380             istream = new ByteArrayInputStream(inputString.getBytes());
381         }
382 
383         if (useSystemIn) {
384             istream = KeepAliveInputStream.wrapSystemIn();
385         }
386 
387         try {
388             final ChannelExec channel;
389             session.setTimeout((int) maxwait);
390             /* execute the command */
391             channel = (ChannelExec) session.openChannel("exec");
392             channel.setCommand(cmd);
393             channel.setOutputStream(tee);
394             channel.setExtOutputStream(tee);
395             channel.setErrStream(teeErr);
396             if (istream != null) {
397                 channel.setInputStream(istream);
398             }
399             channel.setPty(usePty);
400             channel.connect();
401             // wait for it to finish
402             thread =
403                 new Thread() {
404                     public void run() {
405                         while (!channel.isClosed()) {
406                             if (thread == null) {
407                                 return;
408                             }
409                             try {
410                                 sleep(RETRY_INTERVAL);
411                             } catch (Exception e) {
412                                 // ignored
413                             }
414                         }
415                     }
416                 };
417 
418             thread.start();
419             thread.join(maxwait);
420 
421             if (thread.isAlive()) {
422                 // ran out of time
423                 thread = null;
424                 if (getFailonerror()) {
425                     throw new BuildException(TIMEOUT_MESSAGE);
426                 } else {
427                     log(TIMEOUT_MESSAGE, Project.MSG_ERR);
428                 }
429             } else {
430                 // stdout to outputFile
431                 if (outputFile != null) {
432                     writeToFile(out.toString(), append, outputFile);
433                 }
434                 // set errorProperty
435                 if (errorProperty != null) {
436                     getProject().setNewProperty(errorProperty, errout.toString());
437                 }
438                 // stderr to errorFile
439                 if (errorFile != null) {
440                     writeToFile(errout.toString(), appenderr, errorFile);
441                 }
442                 // this is the wrong test if the remote OS is OpenVMS,
443                 // but there doesn't seem to be a way to detect it.
444                 int ec = channel.getExitStatus();
445                 // set resultproperty
446                 if (resultProperty != null) {
447                     getProject().setNewProperty(resultProperty, Integer.toString(ec));
448                 }
449                 if (ec != 0) {
450                     String msg = "Remote command failed with exit status " + ec;
451                     if (getFailonerror()) {
452                         throw new BuildException(msg);
453                     } else {
454                         log(msg, Project.MSG_ERR);
455                     }
456                 }
457             }
458         } catch (BuildException e) {
459             throw e;
460         } catch (JSchException e) {
461             if (e.getMessage().indexOf("session is down") >= 0) {
462                 if (getFailonerror()) {
463                     throw new BuildException(TIMEOUT_MESSAGE, e);
464                 } else {
465                     log(TIMEOUT_MESSAGE, Project.MSG_ERR);
466                 }
467             } else {
468                 if (getFailonerror()) {
469                     throw new BuildException(e);
470                 } else {
471                     log("Caught exception: " + e.getMessage(),
472                         Project.MSG_ERR);
473                 }
474             }
475         } catch (Exception e) {
476             if (getFailonerror()) {
477                 throw new BuildException(e);
478             } else {
479                 log("Caught exception: " + e.getMessage(), Project.MSG_ERR);
480             }
481         } finally {
482             sb.append(out.toString());
483             FileUtils.close(istream);
484         }
485     }

487     /**
488      * Writes a string to a file. If destination file exists, it may be
489      * overwritten depending on the "append" value.
490      *
491      * @param from           string to write
492      * @param to             file to write to
493      * @param append         if true, append to existing file, else overwrite
494      * @exception Exception  most likely an IOException
495      */
496     private void writeToFile(String from, boolean append, File to)
497         throws IOException {
498         FileWriter out = null;
499         try {
500             out = new FileWriter(to.getAbsolutePath(), append);
501             StringReader in = new StringReader(from);
502             char[] buffer = new char[BUFFER_SIZE];
503             int bytesRead;
504             while (true) {
505                 bytesRead = in.read(buffer);
506                 if (bytesRead == -1) {
507                     break;
508                 }
509                 out.write(buffer, 0, bytesRead);
510             }
511             out.flush();
512         } finally {
513             if (out != null) {
514                 out.close();
515             }
516         }
517     }
