19  package org.apache.tools.ant.types.selectors;
20  
21  import java.io.BufferedReader;
22  import java.io.File;
23  import java.io.IOException;
24  import java.io.InputStreamReader;
25  
26  import org.apache.tools.ant.BuildException;
27  import org.apache.tools.ant.Project;
28  import org.apache.tools.ant.types.Parameter;
29  import org.apache.tools.ant.types.Resource;
30  import org.apache.tools.ant.types.resources.FileResource;
31  import org.apache.tools.ant.types.resources.selectors.ResourceSelector;
32  import org.apache.tools.ant.util.FileUtils;
33  
34  /**
35   * Selector that filters files/resources based on whether they contain a
36   * particular string.
37   *
38   * @since 1.5
39   */
40  public class ContainsSelector extends BaseExtendSelector implements ResourceSelector {
41  
42      private String contains = null;
43      private boolean casesensitive = true;
44      private boolean ignorewhitespace = false;
45      private String encoding = null;
46      /** Key to used for parameterized custom selector */
47      public static final String EXPRESSION_KEY = "expression";
48      /** Used for parameterized custom selector */
49      public static final String CONTAINS_KEY = "text";
50      /** Used for parameterized custom selector */
51      public static final String CASE_KEY = "casesensitive";
52      /** Used for parameterized custom selector */
53      public static final String WHITESPACE_KEY = "ignorewhitespace";
54  
55  
56      /**
57       * Creates a new <code>ContainsSelector</code> instance.
58       *
59       */
60      public ContainsSelector() {
61      }
62  
63      /**
64       * @return a string describing this object
65       */
66      public String toString() {
67          StringBuilder buf = new StringBuilder("{containsselector text: ");
68          buf.append('"').append(contains).append('"');
69          buf.append(" casesensitive: ");
70          buf.append(casesensitive ? "true" : "false");
71          buf.append(" ignorewhitespace: ");
72          buf.append(ignorewhitespace ? "true" : "false");
73          buf.append("}");
74          return buf.toString();
75      }
76  
77      /**
78       * The string to search for within a file.
79       *
80       * @param contains the string that a file must contain to be selected.
81       */
82      public void setText(String contains) {
83          this.contains = contains;
84      }
85  
86      /**
87       * The encoding of the resources processed
88       * @since Ant 1.9.0
89       * @param encoding encoding of the resources processed
90       */
91      public void setEncoding(String encoding) {
92          this.encoding = encoding;
93      }
94  
95      /**
96       * Whether to ignore case in the string being searched.
97       *
98       * @param casesensitive whether to pay attention to case sensitivity
99       */
100     public void setCasesensitive(boolean casesensitive) {
101         this.casesensitive = casesensitive;
102     }
103 
104     /**
105      * Whether to ignore whitespace in the string being searched.
106      *
107      * @param ignorewhitespace whether to ignore any whitespace
108      *        (spaces, tabs, etc.) in the searchstring
109      */
110     public void setIgnorewhitespace(boolean ignorewhitespace) {
111         this.ignorewhitespace = ignorewhitespace;
112     }
113 
114     /**
115      * When using this as a custom selector, this method will be called.
116      * It translates each parameter into the appropriate setXXX() call.
117      *
118      * @param parameters the complete set of parameters for this selector
119      */
120     public void setParameters(Parameter[] parameters) {
121         super.setParameters(parameters);
122         if (parameters != null) {
123             for (int i = 0; i < parameters.length; i++) {
124                 String paramname = parameters[i].getName();
125                 if (CONTAINS_KEY.equalsIgnoreCase(paramname)) {
126                     setText(parameters[i].getValue());
127                 } else if (CASE_KEY.equalsIgnoreCase(paramname)) {
128                     setCasesensitive(Project.toBoolean(
129                             parameters[i].getValue()));
130                 } else if (WHITESPACE_KEY.equalsIgnoreCase(paramname)) {
131                     setIgnorewhitespace(Project.toBoolean(
132                             parameters[i].getValue()));
133                 } else {
134                     setError("Invalid parameter " + paramname);
135                 }
136             }
137         }
138     }
139 
140     /**
141      * Checks to make sure all settings are kosher. In this case, it
142      * means that the pattern attribute has been set.
143      *
144      */
145     public void verifySettings() {
146         if (contains == null) {
147             setError("The text attribute is required");
148         }
149     }
150 
151     /**
152      * The heart of the matter. This is where the selector gets to decide
153      * on the inclusion of a file in a particular fileset.
154      *
155      * @param basedir the base directory the scan is being done from
156      * @param filename is the name of the file to check
157      * @param file is a java.io.File object the selector can use
158      * @return whether the file should be selected or not
159      */
160     public boolean isSelected(File basedir, String filename, File file) {
161         return isSelected(new FileResource(file));
162     }
163 
164     /**
165      * The heart of the matter. This is where the selector gets to decide
166      * on the inclusion of a Resource.
167      *
168      * @param r the Resource to check.
169      * @return whether the Resource is selected.
170      */
171     public boolean isSelected(Resource r) {
172 
173         // throw BuildException on error
174         validate();
175 
176         if (r.isDirectory() || contains.length() == 0) {
177             return true;
178         }
179 
180         String userstr = contains;
181         if (!casesensitive) {
182             userstr = contains.toLowerCase();
183         }
184         if (ignorewhitespace) {
185             userstr = SelectorUtils.removeWhitespace(userstr);
186         }
187         BufferedReader in = null;
188         try {
189             if (encoding != null) {
190                 in = new BufferedReader(new InputStreamReader(r.getInputStream(), encoding));
191             }   else {
192                 in = new BufferedReader(new InputStreamReader(r.getInputStream()));
193             }
194         } catch (Exception e) {
195             throw new BuildException("Could not get InputStream from "
196                     + r.toLongString(), e);
197         }
198         try {
199             String teststr = in.readLine();
200             while (teststr != null) {
201                 if (!casesensitive) {
202                     teststr = teststr.toLowerCase();
203                 }
204                 if (ignorewhitespace) {
205                     teststr = SelectorUtils.removeWhitespace(teststr);
206                 }
207                 if (teststr.indexOf(userstr) > -1) {
208                     return true;
209                 }
210                 teststr = in.readLine();
211             }
212             return false;
213         } catch (IOException ioe) {
214             throw new BuildException("Could not read " + r.toLongString());
215         } finally {
216             FileUtils.close(in);
217         }
218     }
219 
220 }
