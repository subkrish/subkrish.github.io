44  /**
45   * <!-- START SNIPPET: javadoc -->
46   * <p>
47   * Default action mapper implementation, using the standard *.[ext] (where ext
48   * usually "action") pattern. The extension is looked up from the Struts
49   * configuration key <b>struts.action.extension</b>.
50   * </p>
51   *
52   * <p>
53   * To help with dealing with buttons and other related requirements, this
54   * mapper (and other {@link ActionMapper}s, we hope) has the ability to name a
55   * button with some predefined prefix and have that button name alter the
56   * execution behaviour. The four prefixes are:
57   * </p>
58   *
59   * <ul>
60   * <li>Method prefix - <i>method:default</i></li>
61   * <li>Action prefix - <i>action:dashboard</i></li>
62   * </ul>
63   *
64   * <p>
65   * In addition to these four prefixes, this mapper also understands the
66   * action naming pattern of <i>foo!bar</i> in either the extension form (eg:
67   * foo!bar.action) or in the prefix form (eg: action:foo!bar). This syntax tells
68   * this mapper to map to the action named <i>foo</i> and the method <i>bar</i>.
69   * </p>
70   * <!-- END SNIPPET: javadoc -->
71   * <b>Method Prefix</b>
72   * <!-- START SNIPPET: method -->
73   * <p>
74   * With method-prefix, instead of calling baz action's execute() method (by
75   * default if it isn't overriden in struts.xml to be something else), the baz
76   * action's anotherMethod() will be called. A very elegant way determine which
77   * button is clicked. Alternatively, one would have submit button set a
78   * particular value on the action when clicked, and the execute() method decides
79   * on what to do with the setted value depending on which button is clicked.
80   * </p>
81   * <!-- END SNIPPET: method -->
82   *
83   * <pre>
84   *  &lt;!-- START SNIPPET: method-example --&gt;
85   *  &lt;s:form action=&quot;baz&quot;&gt;
86   *      &lt;s:textfield label=&quot;Enter your name&quot; name=&quot;person.name&quot;/&gt;
87   *      &lt;s:submit value=&quot;Create person&quot;/&gt;
88   *      &lt;s:submit method=&quot;anotherMethod&quot; value=&quot;Cancel&quot;/&gt;
89   *  &lt;/s:form&gt;
90   *  &lt;!-- END SNIPPET: method-example --&gt;
91   * </pre>
92   * <b>Action prefix</b>
93   * <!-- START SNIPPET: action -->
94   * <p>
95   * With action-prefix, instead of executing baz action's execute() method (by
96   * default if it isn't overridden in struts.xml to be something else), the
97   * anotherAction action's execute() method (assuming again if it isn't overridden
98   * with something else in struts.xml) will be executed.
99   * </p>
100  * <!-- END SNIPPET: action -->
101  *
102  * <pre>
103  *  &lt;!-- START SNIPPET: action-example --&gt;
104  *  &lt;s:form action=&quot;baz&quot;&gt;
105  *      &lt;s:textfield label=&quot;Enter your name&quot; name=&quot;person.name&quot;/&gt;
106  *      &lt;s:submit value=&quot;Create person&quot;/&gt;
107  *      &lt;s:submit action=&quot;anotherAction&quot; value=&quot;Cancel&quot;/&gt;
108  *  &lt;/s:form&gt;
109  *  &lt;!-- END SNIPPET: action-example --&gt;
110  * </pre>
111  */
112 public class DefaultActionMapper implements ActionMapper {

114     private static final Logger LOG = LogManager.getLogger(DefaultActionMapper.class);

116     protected static final String METHOD_PREFIX = "method:";
117     protected static final String ACTION_PREFIX = "action:";

119     protected boolean allowDynamicMethodCalls = false;
120     protected boolean allowSlashesInActionNames = false;
121     protected boolean alwaysSelectFullNamespace = false;
122     protected PrefixTrie prefixTrie = null;
123     protected Pattern allowedActionNames = Pattern.compile("[a-zA-Z0-9._!/\\-]*");
124     protected String defaultActionName = "index";
125 
126     protected Pattern allowedMethodNames = Pattern.compile("[a-zA-Z_]*[0-9]*");
127     protected String defaultMethodName = "execute";

129     private boolean allowActionPrefix = false;
130     private boolean allowActionCrossNamespaceAccess = false;

132     protected List<String> extensions = new ArrayList<String>() {{
133         add("action");
134         add("");

139     public DefaultActionMapper() {
140         prefixTrie = new PrefixTrie() {
141             {
142                 put(METHOD_PREFIX, new ParameterAction() {
143                     public void execute(String key, ActionMapping mapping) {
144                         if (allowDynamicMethodCalls) {
145                             mapping.setMethod(cleanupMethodName(key.substring(METHOD_PREFIX.length())));
146                         }
147                     }
148                 });
149 
150                 put(ACTION_PREFIX, new ParameterAction() {
151                     public void execute(final String key, ActionMapping mapping) {
152                         if (allowActionPrefix) {
153                             String name = key.substring(ACTION_PREFIX.length());
154                             if (allowDynamicMethodCalls) {
155                                 int bang = name.indexOf('!');
156                                 if (bang != -1) {
157                                     String method = cleanupMethodName(name.substring(bang + 1));
158                                     mapping.setMethod(method);
159                                     name = name.substring(0, bang);
160                                 }
161                             }
162                             String actionName = cleanupActionName(name);
163                             if (allowSlashesInActionNames && !allowActionCrossNamespaceAccess) {
164                                 if (actionName.startsWith("/")) {
165                                     actionName = actionName.substring(1);
166                                 }
167                             }
168                             if (!allowSlashesInActionNames && !allowActionCrossNamespaceAccess) {
169                                 if (actionName.lastIndexOf("/") != -1) {
170                                     actionName = actionName.substring(actionName.lastIndexOf("/") + 1);
171                                 }
172                             }
173                             mapping.setName(actionName);
174                         }
175                     }
176                 });
177 
178             }
179         };
180     }

243     @Inject(StrutsConstants.STRUTS_ACTION_EXTENSION)
244     public void setExtensions(String extensions) {
245         if (StringUtils.isNotEmpty(extensions)) {
246             List<String> list = new ArrayList<>();
247             String[] tokens = extensions.split(",");
248             Collections.addAll(list, tokens);
249             if (extensions.endsWith(",")) {
250                 list.add("");
251             }
252             this.extensions = Collections.unmodifiableList(list);
253         } else {
254             this.extensions = null;
255         }
256     }

268     /*
269      * (non-Javadoc)
270      *
271      * @see org.apache.struts2.dispatcher.mapper.ActionMapper#getMapping(javax.servlet.http.HttpServletRequest)
272      */
273     public ActionMapping getMapping(HttpServletRequest request, ConfigurationManager configManager) {
274         ActionMapping mapping = new ActionMapping();
275         String uri = RequestUtils.getUri(request);
276 
277         int indexOfSemicolon = uri.indexOf(";");
278         uri = (indexOfSemicolon > -1) ? uri.substring(0, indexOfSemicolon) : uri;
279 
280         uri = dropExtension(uri, mapping);
281         if (uri == null) {
282             return null;
283         }
284 
285         parseNameAndNamespace(uri, mapping, configManager);
286         handleSpecialParameters(request, mapping);
287         return parseActionName(mapping);
288     }
289 
290     protected ActionMapping parseActionName(ActionMapping mapping) {
291         if (mapping.getName() == null) {
292             return null;
293         }
294         if (allowDynamicMethodCalls) {
295             // handle "name!method" convention.
296             String name = mapping.getName();
297             int exclamation = name.lastIndexOf("!");
298             if (exclamation != -1) {
299                 mapping.setName(name.substring(0, exclamation));
300 
301                 mapping.setMethod(name.substring(exclamation + 1));
302             }
303         }
304         return mapping;
305     }

307     /**
308      * Special parameters, as described in the class-level comment, are searched
309      * for and handled.
310      *
311      * @param request The request
312      * @param mapping The action mapping
313      */
314     public void handleSpecialParameters(HttpServletRequest request, ActionMapping mapping) {
315         // handle special parameter prefixes.
316         Set<String> uniqueParameters = new HashSet<>();
317         Map parameterMap = request.getParameterMap();
318         for (Object o : parameterMap.keySet()) {
319             String key = (String) o;
320 
321             // Strip off the image button location info, if found
322             if (key.endsWith(".x") || key.endsWith(".y")) {
323                 key = key.substring(0, key.length() - 2);
324             }
325 
326             // Ensure a parameter doesn't get processed twice
327             if (!uniqueParameters.contains(key)) {
328                 ParameterAction parameterAction = (ParameterAction) prefixTrie.get(key);
329                 if (parameterAction != null) {
330                     parameterAction.execute(key, mapping);
331                     uniqueParameters.add(key);
332                     break;
333                 }
334             }
335         }
336     }

338     /**
339      * Parses the name and namespace from the uri
340      *
341      * @param uri     The uri
342      * @param mapping The action mapping to populate
343      * @param configManager configuration manager
344      */
345     protected void parseNameAndNamespace(String uri, ActionMapping mapping, ConfigurationManager configManager) {
346         String namespace, name;
347         int lastSlash = uri.lastIndexOf("/");
348         if (lastSlash == -1) {
349             namespace = "";
350             name = uri;
351         } else if (lastSlash == 0) {
352             // ww-1046, assume it is the root namespace, it will fallback to
353             // default
354             // namespace anyway if not found in root namespace.
355             namespace = "/";
356             name = uri.substring(lastSlash + 1);
357         } else if (alwaysSelectFullNamespace) {
358             // Simply select the namespace as everything before the last slash
359             namespace = uri.substring(0, lastSlash);
360             name = uri.substring(lastSlash + 1);
361         } else {
362             // Try to find the namespace in those defined, defaulting to ""
363             Configuration config = configManager.getConfiguration();
364             String prefix = uri.substring(0, lastSlash);
365             namespace = "";
366             boolean rootAvailable = false;
367             // Find the longest matching namespace, defaulting to the default
368             for (PackageConfig cfg : config.getPackageConfigs().values()) {
369                 String ns = cfg.getNamespace();
370                 if (ns != null && prefix.startsWith(ns) && (prefix.length() == ns.length() || prefix.charAt(ns.length()) == '/')) {
371                     if (ns.length() > namespace.length()) {
372                         namespace = ns;
373                     }
374                 }
375                 if ("/".equals(ns)) {
376                     rootAvailable = true;
377                 }
378             }
379 
380             name = uri.substring(namespace.length() + 1);
381 
382             // Still none found, use root namespace if found
383             if (rootAvailable && "".equals(namespace)) {
384                 namespace = "/";
385             }
386         }
387 
388         if (!allowSlashesInActionNames) {
389             int pos = name.lastIndexOf('/');
390             if (pos > -1 && pos < name.length() - 1) {
391                 name = name.substring(pos + 1);
392             }
393         }
394 
395         mapping.setNamespace(namespace);
396         mapping.setName(cleanupActionName(name));
397     }

399     /**
400      * Checks action name against allowed pattern if not matched returns default action name
401      *
402      * @param rawActionName action name extracted from URI
403      * @return safe action name
404      */
405     protected String cleanupActionName(final String rawActionName) {
406         if (allowedActionNames.matcher(rawActionName).matches()) {
407             return rawActionName;
408         } else {
409             LOG.warn("{} did not match allowed action names {} - default action {} will be used!", rawActionName, allowedActionNames, defaultActionName);
410             return defaultActionName;
411         }
412     }

417      * @param rawMethodName method name extracted from URI

423         } else {

425             return defaultMethodName;

429     /**
430      * Drops the extension from the action name, storing it in the mapping for later use
431      *
432      * @param name    The action name
433      * @param mapping The action mapping to store the extension in
434      * @return The action name without its extension
435      */
436     protected String dropExtension(String name, ActionMapping mapping) {
437         if (extensions == null) {
438             return name;
439         }
440         for (String ext : extensions) {
441             if ("".equals(ext)) {
442                 // This should also handle cases such as /foo/bar-1.0/description. It is tricky to
443                 // distinquish /foo/bar-1.0 but perhaps adding a numeric check in the future could
444                 // work
445                 int index = name.lastIndexOf('.');
446                 if (index == -1 || name.indexOf('/', index) >= 0) {
447                     return name;
448                 }
449             } else {
450                 String extension = "." + ext;
451                 if (name.endsWith(extension)) {
452                     name = name.substring(0, name.length() - extension.length());
453                     mapping.setExtension(ext);
454                     return name;
455                 }
456             }
457         }
458         return null;
459     }

461     /**
462      * @return  null if no extension is specified.
463      */
464     protected String getDefaultExtension() {
465         if (extensions == null) {
466             return null;
467         } else {
468             return extensions.get(0);
469         }
470     }

472     /*
473      * (non-Javadoc)
474      *
475      * @see org.apache.struts2.dispatcher.mapper.ActionMapper#getUriFromActionMapping(org.apache.struts2.dispatcher.mapper.ActionMapping)
476      */
477     public String getUriFromActionMapping(ActionMapping mapping) {
478         StringBuilder uri = new StringBuilder();
479 
480         handleNamespace(mapping, uri);
481         handleName(mapping, uri);
482         handleDynamicMethod(mapping, uri);
483         handleExtension(mapping, uri);
484         handleParams(mapping, uri);
485 
486         return uri.toString();
487     }
488 
489     protected void handleNamespace(ActionMapping mapping, StringBuilder uri) {
490         if (mapping.getNamespace() != null) {
491             uri.append(mapping.getNamespace());
492             if (!"/".equals(mapping.getNamespace())) {
493                 uri.append("/");
494             }
495         }
496     }
497 
498     protected void handleName(ActionMapping mapping, StringBuilder uri) {
499         String name = mapping.getName();
500         if (name.indexOf('?') != -1) {
501             name = name.substring(0, name.indexOf('?'));
502         }
503         uri.append(name);
504     }
505 
506     protected void handleDynamicMethod(ActionMapping mapping, StringBuilder uri) {
507         // See WW-3965
508         if (StringUtils.isNotEmpty(mapping.getMethod())) {
509             if (allowDynamicMethodCalls) {
510                 // handle "name!method" convention.
511                 String name = mapping.getName();
512                 if (!name.contains("!")) {
513                     // Append the method as no bang found
514                     uri.append("!").append(mapping.getMethod());
515                 }
516             } else {
517                 uri.append("!").append(mapping.getMethod());
518             }
519         }
520     }
521 
522     protected void handleExtension(ActionMapping mapping, StringBuilder uri) {
523         String extension = lookupExtension(mapping.getExtension());
524 
525         if (extension != null) {
526             if (extension.length() == 0 || (extension.length() > 0 && uri.indexOf('.' + extension) == -1)) {
527                 if (extension.length() > 0) {
528                     uri.append(".").append(extension);
529                 }
530             }
531         }
532     }
533 
534     protected String lookupExtension(String extension) {
535         if (extension == null) {
536             // Look for the current extension, if available
537             ActionContext context = ActionContext.getContext();
538             if (context != null) {
539                 ActionMapping orig = (ActionMapping) context.get(ServletActionContext.ACTION_MAPPING);
540                 if (orig != null) {
541                     extension = orig.getExtension();
542                 }
543             }
544             if (extension == null) {
545                 extension = getDefaultExtension();
546             }
547         }
548         return extension;
549     }
550 
551     protected void handleParams(ActionMapping mapping, StringBuilder uri) {
552         String name = mapping.getName();
553         String params = "";
554         if (name.indexOf('?') != -1) {
555             params = name.substring(name.indexOf('?'));
556         }
557         if (params.length() > 0) {
558             uri.append(params);
559         }
560     }
