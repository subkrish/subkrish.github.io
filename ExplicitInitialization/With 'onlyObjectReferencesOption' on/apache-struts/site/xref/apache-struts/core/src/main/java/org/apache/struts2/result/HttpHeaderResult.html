22  package org.apache.struts2.result;
23  
24  import com.opensymphony.xwork2.ActionContext;
25  import com.opensymphony.xwork2.ActionInvocation;
26  import com.opensymphony.xwork2.Result;
27  import com.opensymphony.xwork2.util.TextParseUtil;
28  import com.opensymphony.xwork2.util.ValueStack;
29  import org.apache.logging.log4j.LogManager;
30  import org.apache.logging.log4j.Logger;
31  import org.apache.struts2.ServletActionContext;
32  
33  import javax.servlet.http.HttpServletResponse;
34  import java.util.HashMap;
35  import java.util.Map;
36  
37  
38  /**
39   * <!-- START SNIPPET: description -->
40   * <p>
41   * A custom Result type for setting HTTP headers and status by optionally evaluating against the ValueStack.
42   * This result can also be used to send an error to the client. All the parameters can be evaluated against the ValueStack.
43   * </p>
44   *
45   * <!-- END SNIPPET: description -->
46   * <p>
47   * <b>This result type takes the following parameters:</b>
48   * </p>
49   * <!-- START SNIPPET: params -->
50   *
51   * <ul>
52   *
53   * <li><b>status</b> - the http servlet response status code that should be set on a response.</li>
54   *
55   * <li><b>parse</b> - true by default. If set to false, the headers param will not be parsed for Ognl expressions.</li>
56   *
57   * <li><b>headers</b> - header values.</li>
58   *
59   * <li><b>error</b> - the http servlet response error code that should be set on a response.</li>
60   *
61   * <li><b>errorMessage</b> - error message to be set on response if 'error' is set.</li>
62   * </ul>
63   *
64   * <!-- END SNIPPET: params -->
65   * <p>
66   * <b>Example:</b>
67   * </p>
68   * <pre><!-- START SNIPPET: example -->
69   * &lt;result name="success" type="httpheader"&gt;
70   *   &lt;param name="status"&gt;204&lt;/param&gt;
71   *   &lt;param name="headers.a"&gt;a custom header value&lt;/param&gt;
72   *   &lt;param name="headers.b"&gt;another custom header value&lt;/param&gt;
73   * &lt;/result&gt;
74   *
75   * &lt;result name="proxyRequired" type="httpheader"&gt;
76   *   &lt;param name="error"&gt;305&lt;/param&gt;
77   *   &lt;param name="errorMessage"&gt;this action must be accessed through a proxy&lt;/param&gt;
78   * &lt;/result&gt;
79   * <!-- END SNIPPET: example --></pre>
80   */
81  public class HttpHeaderResult implements Result {
82  
83      private static final long serialVersionUID = 195648957144219214L;
84      private static final Logger LOG = LogManager.getLogger(HttpHeaderResult.class);
85  
86      /**
87       * This result type doesn't have a default param, null is ok to reduce noise in logs
88       */
89      public static final String DEFAULT_PARAM = null;
90  
91      private boolean parse = true;
92      private Map<String, String> headers;
93      private int status = -1;
94      private String error = null;
95      private String errorMessage;
96  
97      public HttpHeaderResult() {
98          super();
99          headers = new HashMap<>();
100     }
101 
102     public HttpHeaderResult(int status) {
103         this();
104         this.status = status;
105         this.parse = false;
106     }
107 
108     /**
109      * Sets the http servlet error code that should be set on the response
110      *
111      * @param error the Http error code
112      * @see javax.servlet.http.HttpServletResponse#sendError(int)
113      */
114     public void setError(String error) {
115         this.error = error;
116     }
117 
118     /**
119      * Sets the error message that should be set on the reponse
120      *
121      * @param errorMessage error message send to the client
122      * @see javax.servlet.http.HttpServletResponse#sendError(int, String)
123      */
124     public void setErrorMessage(String errorMessage) {
125         this.errorMessage = errorMessage;
126     }
127 
128     /**
129      * Returns a Map of all HTTP headers.
130      *
131      * @return a Map of all HTTP headers.
132      */
133     public Map<String, String> getHeaders() {
134         return headers;
135     }
136 
137     /**
138      * Sets whether or not the HTTP header values should be evaluated against the ValueStack (by default they are).
139      *
140      * @param parse <tt>true</tt> if HTTP header values should be evaluated against the ValueStack, <tt>false</tt>
141      *              otherwise.
142      */
143     public void setParse(boolean parse) {
144         this.parse = parse;
145     }
146 
147     /**
148      * Sets the http servlet response status code that should be set on a response.
149      *
150      * @param status the Http status code
151      * @see javax.servlet.http.HttpServletResponse#setStatus(int)
152      */
153     public void setStatus(int status) {
154         this.status = status;
155     }
156 
157     public int getStatus() {
158         return status;
159     }
160 
161     /**
162      * Adds an HTTP header to the response
163      *
164      * @param name  header name
165      * @param value header value
166      */
167     public void addHeader(String name, String value) {
168         headers.put(name, value);
169     }
170 
171     /**
172      * Sets the optional HTTP response status code and also re-sets HTTP headers after they've
173      * been optionally evaluated against the ValueStack.
174      *
175      * @param invocation an encapsulation of the action execution state.
176      * @throws Exception if an error occurs when re-setting the headers.
177      */
178     public void execute(ActionInvocation invocation) throws Exception {
179         HttpServletResponse response = ServletActionContext.getResponse();
180         ValueStack stack = ActionContext.getContext().getValueStack();
181 
182         if (status != -1) {
183             response.setStatus(status);
184         }
185 
186         if (headers != null) {
187             for (Map.Entry<String, String> entry : headers.entrySet()) {
188                 String value = entry.getValue();
189                 String finalValue = parse ? TextParseUtil.translateVariables(value, stack) : value;
190                 response.addHeader(entry.getKey(), finalValue);
191             }
192         }
193 
194         if (status == -1 && error != null) {
195             int errorCode = -1;
196             try {
197                 errorCode = Integer.parseInt(parse ? TextParseUtil.translateVariables(error, stack) : error);
198             } catch (Exception e) {
199                 LOG.error("Cannot parse errorCode [{}] value as Integer!", error, e);
200             }
201             if (errorCode != -1) {
202                 if (errorMessage != null) {
203                     String finalMessage = parse ? TextParseUtil.translateVariables(errorMessage, stack) : errorMessage;
204                     response.sendError(errorCode, finalMessage);
205                 } else {
206                     response.sendError(errorCode);
207                 }
208             }
209         }
210     }
211 }
