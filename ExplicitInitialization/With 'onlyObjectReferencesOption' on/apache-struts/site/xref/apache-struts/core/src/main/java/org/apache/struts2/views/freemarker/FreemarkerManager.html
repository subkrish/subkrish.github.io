58  /**
59   * <p>
60   * Static Configuration Manager for the FreemarkerResult's configuration
61   * </p>
62   *
63   * <p>
64   * Possible extension points are:
65   *
66   * <ul>
67   *   <li>createConfiguration method</li>
68   *   <li>loadSettings method</li>
69   *   <li>createTemplateLoader method</li>
70   *   <li>populateContext method</li>
71   * </ul>
72   *
73   * <p>
74   * <b> createConfiguration method </b><br>
75   * Create a freemarker Configuration.
76   * </p>
77   *
78   * <p>
79   * <b> loadSettings method </b><br>
80   * Load freemarker settings, default to freemarker.properties (if found in classpath)
81   * </p>
82   *
83   * <p>
84   * <b> createTemplateLoader method</b><br>
85   * create a freemarker TemplateLoader that loads freemarker template in the following order:
86   * </p>
87   *
88   * <ol>
89   *   <li>path defined in ServletContext init parameter named 'templatePath' or 'TemplatePath' (must be an absolute path)</li>
90   *   <li>webapp classpath</li>
91   *   <li>struts's static folder (under [STRUT2_SOURCE]/org/apache/struts2/static/</li>
92   * </ol>
93   *
94   * <p>
95   * <b> populateContext method</b><br>
96   * populate the created model.
97   * </p>
98   */
99  public class FreemarkerManager {

250     public synchronized Configuration getConfiguration(ServletContext servletContext) {
251         if (config == null) {
252             try {
253                 init(servletContext);
254             } catch (TemplateException e) {
255                 LOG.error("Cannot load freemarker configuration: ", e);
256             }
257             // store this configuration in the servlet context
258             servletContext.setAttribute(CONFIG_SERVLET_CONTEXT_KEY, config);
259         }
260         return config;
261     }
262 
263     public void init(ServletContext servletContext) throws TemplateException {
264         config = createConfiguration(servletContext);
265 
266         // Set defaults:
267         config.setTemplateExceptionHandler(TemplateExceptionHandler.HTML_DEBUG_HANDLER);
268         contentType = DEFAULT_CONTENT_TYPE;
269 
270         // Process object_wrapper init-param out of order:
271         wrapper = createObjectWrapper(servletContext);
272         LOG.debug("Using object wrapper of class {}", wrapper.getClass().getName());
273         config.setObjectWrapper(wrapper);
274 
275         // Process TemplatePath init-param out of order:
276         templatePath = servletContext.getInitParameter(INITPARAM_TEMPLATE_PATH);
277         if (templatePath == null) {
278             templatePath = servletContext.getInitParameter("templatePath");
279         }
280 
281         configureTemplateLoader(createTemplateLoader(servletContext, templatePath));
282 
283         loadSettings(servletContext);
284     }

299     /**
300      * Create the instance of the freemarker Configuration object.
301      * <p>
302      * this implementation
303      * </p>
304      *
305      * <ul>
306      * <li>obtains the default configuration from Configuration.getDefaultConfiguration()
307      * <li>sets up template loading from a ClassTemplateLoader and a WebappTemplateLoader
308      * <li>sets up the object wrapper to be the BeansWrapper
309      * <li>loads settings from the classpath file /freemarker.properties
310      * </ul>
311      *
312      * @param servletContext the servlet context
313      * @return a instance of the freemarker configuration object
314      * @throws TemplateException in case of errors during creating the configuration
315      */
316     protected Configuration createConfiguration(ServletContext servletContext) throws TemplateException {
317         Configuration configuration = new Configuration(Configuration.VERSION_2_3_0);
318 
319         configuration.setTemplateExceptionHandler(TemplateExceptionHandler.HTML_DEBUG_HANDLER);
320 
321         if (mruMaxStrongSize > 0) {
322             LOG.debug("Sets Configuration.CACHE_STORAGE_KEY to strong:{}", mruMaxStrongSize);
323             configuration.setSetting(Configuration.CACHE_STORAGE_KEY, "strong:" + mruMaxStrongSize);
324         }
325         if (templateUpdateDelay != null) {
326             LOG.debug("Sets Configuration.TEMPLATE_UPDATE_DELAY_KEY to {}", templateUpdateDelay);
327             configuration.setSetting(Configuration.TEMPLATE_UPDATE_DELAY_KEY, templateUpdateDelay);
328         }
329         if (encoding != null) {
330             LOG.debug("Sets DefaultEncoding to {}", encoding);
331             configuration.setDefaultEncoding(encoding);
332         }
333         LOG.debug("Disabled localized lookups");
334         configuration.setLocalizedLookup(false);
335         LOG.debug("Enabled whitespace stripping");
336         configuration.setWhitespaceStripping(true);
337         LOG.debug("Sets NewBuiltinClassResolver to TemplateClassResolver.SAFER_RESOLVER");
338         configuration.setNewBuiltinClassResolver(TemplateClassResolver.SAFER_RESOLVER);
339 
340         return configuration;
341     }

344     protected ScopesHashModel buildScopesHashModel(ServletContext servletContext, HttpServletRequest request, HttpServletResponse response, ObjectWrapper wrapper, ValueStack stack) {
345         ScopesHashModel model = new ScopesHashModel(wrapper, servletContext, request, stack);
346 
347         // Create hash model wrapper for servlet context (the application). We need one thread, once per servlet context
348         synchronized (servletContext) {
349             ServletContextHashModel servletContextModel = (ServletContextHashModel) servletContext.getAttribute(ATTR_APPLICATION_MODEL);
350             if (servletContextModel == null) {
351                 // first try a JSP support servlet.  If it fails, default to the servlet.
352                 GenericServlet servlet = JspSupportServlet.jspSupportServlet;
353                 if (servlet != null) {
354                     servletContextModel = new ServletContextHashModel(servlet, wrapper);
355                     servletContext.setAttribute(ATTR_APPLICATION_MODEL, servletContextModel);
356                 } else {
357                     servletContextModel = new ServletContextHashModel(servletContext, wrapper);
358                     servletContext.setAttribute(ATTR_APPLICATION_MODEL, servletContextModel);
359                 }
360                 TaglibFactory taglibs = new TaglibFactory(servletContext);
361                 taglibs.setObjectWrapper(wrapper);
362                 servletContext.setAttribute(ATTR_JSP_TAGLIBS_MODEL, taglibs);
363             }
364             model.put(KEY_APPLICATION, servletContextModel);
365             model.putUnlistedModel(KEY_APPLICATION_PRIVATE, servletContextModel);
366         }
367         model.put(KEY_JSP_TAGLIBS, (TemplateModel) servletContext.getAttribute(ATTR_JSP_TAGLIBS_MODEL));
368 
369         // Create hash model wrapper for session
370         HttpSession session = request.getSession(false);
371         if (session != null) {
372             model.put(KEY_SESSION, new HttpSessionHashModel(session, wrapper));
373         }
374 
375         // Create hash model wrapper for the request attributes
376         HttpRequestHashModel requestModel = (HttpRequestHashModel) request.getAttribute(ATTR_REQUEST_MODEL);
377 
378         if ((requestModel == null) || (requestModel.getRequest() != request)) {
379             requestModel = new HttpRequestHashModel(request, response, wrapper);
380             request.setAttribute(ATTR_REQUEST_MODEL, requestModel);
381         }
382 
383         model.put(KEY_REQUEST, requestModel);
384 
385 
386         // Create hash model wrapper for request parameters
387         HttpRequestParametersHashModel reqParametersModel = (HttpRequestParametersHashModel) request.getAttribute(ATTR_REQUEST_PARAMETERS_MODEL);
388         if (reqParametersModel == null || requestModel.getRequest() != request) {
389             reqParametersModel = new HttpRequestParametersHashModel(request);
390             request.setAttribute(ATTR_REQUEST_PARAMETERS_MODEL, reqParametersModel);
391         }
392         model.put(ATTR_REQUEST_PARAMETERS_MODEL, reqParametersModel);
393         model.put(KEY_REQUEST_PARAMETERS_STRUTS,reqParametersModel);
394 
395         return model;
396     }

405      /**
406      * Create the template loader. The default implementation will create a
407      * {@link ClassTemplateLoader} if the template path starts with "class://",
408      * a {@link FileTemplateLoader} if the template path starts with "file://",
409      * and a {@link WebappTemplateLoader} otherwise.
410      *
411      * @param servletContext the servlet path
412      * @param templatePath the template path to create a loader for
413      * @return a newly created template loader
414      */
415     protected TemplateLoader createTemplateLoader(ServletContext servletContext, String templatePath) {
416         TemplateLoader templatePathLoader = null;
417 
418          try {
419              if(templatePath!=null){
420                  if (templatePath.startsWith("class://")) {
421                      // substring(7) is intentional as we "reuse" the last slash
422                      templatePathLoader = new ClassTemplateLoader(getClass(), templatePath.substring(7));
423                  } else if (templatePath.startsWith("file://")) {
424                      templatePathLoader = new FileTemplateLoader(new File(templatePath.substring(7)));
425                  }
426              }
427          } catch (IOException e) {
428              LOG.error("Invalid template path specified: {}", e.getMessage(), e);
429          }
430 
431          // presume that most apps will require the class and webapp template loader
432          // if people wish to
433          return templatePathLoader != null ?
434                  new MultiTemplateLoader(new TemplateLoader[]{
435                          templatePathLoader,
436                          new WebappTemplateLoader(servletContext),
437                          new StrutsClassTemplateLoader()
438                  })
439                  : new MultiTemplateLoader(new TemplateLoader[]{
440                  new WebappTemplateLoader(servletContext),
441                  new StrutsClassTemplateLoader()
442          });
443      }

446     /**
447      * Load the settings from the /freemarker.properties file on the classpath
448      *
449      * @param servletContext the servlet context
450      *
451      * @see freemarker.template.Configuration#setSettings for the definition of valid settings
452      */
453     protected void loadSettings(ServletContext servletContext) {
454         try (InputStream in = fileManager.loadFile(ClassLoaderUtil.getResource("freemarker.properties", getClass()))){
455             if (in != null) {
456                 Properties p = new Properties();
457                 p.load(in);
458 
459                 for (Object o : p.keySet()) {
460                     String name = (String) o;
461                     String value = (String) p.get(name);
462 
463                     if (name == null) {
464                         throw new IOException(
465                                 "init-param without param-name.  Maybe the freemarker.properties is not well-formed?");
466                     }
467                     if (value == null) {
468                         throw new IOException(
469                                 "init-param without param-value.  Maybe the freemarker.properties is not well-formed?");
470                     }
471                     addSetting(name, value);
472                 }
473             }
474         } catch (IOException e) {
475             LOG.error("Error while loading freemarker settings from /freemarker.properties", e);
476         } catch (TemplateException e) {
477             LOG.error("Error while loading freemarker settings from /freemarker.properties", e);
478         }
479     }

481     public void addSetting(String name, String value) throws TemplateException {
482         // Process all other init-params:
483         if (name.equals(INITPARAM_NOCACHE)) {
484             nocache = StringUtil.getYesNo(value);
485         } else if (name.equals(INITPARAM_DEBUG)) {
486             debug = StringUtil.getYesNo(value);
487         } else if (name.equals(INITPARAM_CONTENT_TYPE)) {
488             contentType = value;
489         } else {
490             config.setSetting(name, value);
491         }
492 
493         if (contentType != null && !contentTypeEvaluated) {
494             int i = contentType.toLowerCase().indexOf("charset=");
495             contentTypeEvaluated = true;
496             if (i != -1) {
497                 char c = ' ';
498                 i--;
499                 while (i >= 0) {
500                     c = contentType.charAt(i);
501                     if (!Character.isWhitespace(c)) break;
502                     i--;
503                 }
504                 if (i == -1 || c == ';') {
505                     noCharsetInContentType = false;
506                 }
507             }
508         }
509     }

513     public ScopesHashModel buildTemplateModel(ValueStack stack, Object action, ServletContext servletContext, HttpServletRequest request, HttpServletResponse response, ObjectWrapper wrapper) {
514         ScopesHashModel model = buildScopesHashModel(servletContext, request, response, wrapper, stack);
515         populateContext(model, stack, action, request, response);
516         if (tagLibraries != null) {
517             for (String prefix : tagLibraries.keySet()) {
518                 model.put(prefix, tagLibraries.get(prefix).getModels(stack, request, response));
519             }
520         }
521 
522         //place the model in the request using the special parameter.  This can be retrieved for freemarker and velocity.
523         request.setAttribute(ATTR_TEMPLATE_MODEL, model);
524 
525         return model;
526     }

529     protected void populateContext(ScopesHashModel model, ValueStack stack, Object action, HttpServletRequest request, HttpServletResponse response) {
530         // put the same objects into the context that the velocity result uses
531         Map standard = ContextUtil.getStandardContext(stack, request, response);
532         model.putAll(standard);
533 
534         // support for JSP exception pages, exposing the servlet or JSP exception
535         Throwable exception = (Throwable) request.getAttribute("javax.servlet.error.exception");
536 
537         if (exception == null) {
538             exception = (Throwable) request.getAttribute("javax.servlet.error.JspException");
539         }
540 
541         if (exception != null) {
542             model.put(KEY_EXCEPTION, exception);
543         }
544     }
