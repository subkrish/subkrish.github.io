64      private double getVersion(TreeNode webApp) {
65          String v = webApp.findAttribute("version");
66          if (v != null) {
67              try {
68                  return Double.parseDouble(v);
69              } catch (NumberFormatException e) {
70              }
71          }
72          return 2.3;
73      }

75      private void processWebDotXml(ServletContext ctxt) throws JasperException {
76  
77          InputStream is = null;
78  
79          try {
80              URL uri = ctxt.getResource(WEB_XML);
81              if (uri == null) {
82                  // no web.xml
83                  return;
84              }
85  
86              is = uri.openStream();
87              InputSource ip = new InputSource(is);
88              ip.setSystemId(uri.toExternalForm()); 
89  
90              ParserUtils pu = new ParserUtils();
91              TreeNode webApp = pu.parseXMLDocument(WEB_XML, ip);
92  
93              if (webApp == null
94                      || getVersion(webApp) < 2.4) {
95                  defaultIsELIgnored = "true";
96                  return;
97              }
98              TreeNode jspConfig = webApp.findChild("jsp-config");
99              if (jspConfig == null) {
100                 return;
101             }
102 
103             jspProperties = new Vector();
104             Iterator jspPropertyList = jspConfig.findChildren("jsp-property-group");
105             while (jspPropertyList.hasNext()) {
106 
107                 TreeNode element = (TreeNode) jspPropertyList.next();
108                 Iterator list = element.findChildren();
109 
110                 Vector urlPatterns = new Vector();
111                 String pageEncoding = null;
112                 String scriptingInvalid = null;
113                 String elIgnored = null;
114                 String isXml = null;
115                 Vector includePrelude = new Vector();
116                 Vector includeCoda = new Vector();
117                 String deferredSyntaxAllowedAsLiteral = null;
118                 String trimDirectiveWhitespaces = null;
119 
120                 while (list.hasNext()) {
121 
122                     element = (TreeNode) list.next();
123                     String tname = element.getName();
124 
125                     if ("url-pattern".equals(tname))
126                         urlPatterns.addElement( element.getBody() );
127                     else if ("page-encoding".equals(tname))
128                         pageEncoding = element.getBody();
129                     else if ("is-xml".equals(tname))
130                         isXml = element.getBody();
131                     else if ("el-ignored".equals(tname))
132                         elIgnored = element.getBody();
133                     else if ("scripting-invalid".equals(tname))
134                         scriptingInvalid = element.getBody();
135                     else if ("include-prelude".equals(tname))
136                         includePrelude.addElement(element.getBody());
137                     else if ("include-coda".equals(tname))
138                         includeCoda.addElement(element.getBody());
139                     else if ("deferred-syntax-allowed-as-literal".equals(tname))
140                         deferredSyntaxAllowedAsLiteral = element.getBody();
141                     else if ("trim-directive-whitespaces".equals(tname))
142                         trimDirectiveWhitespaces = element.getBody();
143                 }
144 
145                 if (urlPatterns.size() == 0) {
146                     continue;
147                 }
148 
149                 // Add one JspPropertyGroup for each URL Pattern.  This makes
150                 // the matching logic easier.
151                 for( int p = 0; p < urlPatterns.size(); p++ ) {
152                     String urlPattern = (String)urlPatterns.elementAt( p );
153                     String path = null;
154                     String extension = null;
155 
156                     if (urlPattern.indexOf('*') < 0) {
157                         // Exact match
158                         path = urlPattern;
159                     } else {
160                         int i = urlPattern.lastIndexOf('/');
161                         String file;
162                         if (i >= 0) {
163                             path = urlPattern.substring(0,i+1);
164                             file = urlPattern.substring(i+1);
165                         } else {
166                             file = urlPattern;
167                         }
168 
169                         // pattern must be "*", or of the form "*.jsp"
170                         if (file.equals("*")) {
171                             extension = "*";
172                         } else if (file.startsWith("*.")) {
173                             extension = file.substring(file.indexOf('.')+1);
174                         }
175 
176                         // The url patterns are reconstructed as the follwoing:
177                         // path != null, extension == null:  / or /foo/bar.ext
178                         // path == null, extension != null:  *.ext
179                         // path != null, extension == "*":   /foo/*
180                         boolean isStar = "*".equals(extension);
181                         if ((path == null && (extension == null || isStar))
182                                 || (path != null && !isStar)) {
183                             if (log.isWarnEnabled()) {
184                                 log.warn(Localizer.getMessage(
185                                         "jsp.warning.bad.urlpattern.propertygroup",
186                                         urlPattern));
187                             }
188                             continue;
189                         }
190                     }
191 
192                     JspProperty property = new JspProperty(isXml,
193                             elIgnored,
194                             scriptingInvalid,
195                             pageEncoding,
196                             includePrelude,
197                             includeCoda,
198                             deferredSyntaxAllowedAsLiteral,
199                             trimDirectiveWhitespaces);
200                     JspPropertyGroup propertyGroup =
201                         new JspPropertyGroup(path, extension, property);
202 
203                     jspProperties.addElement(propertyGroup);
204                 }
205             }
206         } catch (Exception ex) {
207             throw new JasperException(ex);
208         } finally {
209             if (is != null) {
210                 try {
211                     is.close();
212                 } catch (Throwable t) {}
213             }
214         }
215     }
216 
217     private void init() throws JasperException {
218 
219         if (!initialized) {
220             processWebDotXml(ctxt);
221             defaultJspProperty = new JspProperty(defaultIsXml,
222                     defaultIsELIgnored,
223                     defaultIsScriptingInvalid,
224                     null, null, null, defaultDeferedSyntaxAllowedAsLiteral, 
225                     defaultTrimDirectiveWhitespaces);
226             initialized = true;
227         }
228     }

230     /**
231      * Select the property group that has more restrictive url-pattern.
232      * In case of tie, select the first.
233      */
234     private JspPropertyGroup selectProperty(JspPropertyGroup prev,
235             JspPropertyGroup curr) {
236         if (prev == null) {
237             return curr;
238         }
239         if (prev.getExtension() == null) {
240             // exact match
241             return prev;
242         }
243         if (curr.getExtension() == null) {
244             // exact match
245             return curr;
246         }
247         String prevPath = prev.getPath();
248         String currPath = curr.getPath();
249         if (prevPath == null && currPath == null) {
250             // Both specifies a *.ext, keep the first one
251             return prev;
252         }
253         if (prevPath == null && currPath != null) {
254             return curr;
255         }
256         if (prevPath != null && currPath == null) {
257             return prev;
258         }
259         if (prevPath.length() >= currPath.length()) {
260             return prev;
261         }
262         return curr;
263     }

266     /**
267      * Find a property that best matches the supplied resource.
268      * @param uri the resource supplied.
269      * @return a JspProperty indicating the best match, or some default.
270      * @throws JasperException in case of Jasper errors
271      */
272     public JspProperty findJspProperty(String uri) throws JasperException {
273 
274         init();
275 
276         // JSP Configuration settings do not apply to tag files	    
277         if (jspProperties == null || uri.endsWith(".tag")
278                 || uri.endsWith(".tagx")) {
279             return defaultJspProperty;
280         }
281 
282         String uriPath = null;
283         int index = uri.lastIndexOf('/');
284         if (index >=0 ) {
285             uriPath = uri.substring(0, index+1);
286         }
287         String uriExtension = null;
288         index = uri.lastIndexOf('.');
289         if (index >=0) {
290             uriExtension = uri.substring(index+1);
291         }
292 
293         Vector includePreludes = new Vector();
294         Vector includeCodas = new Vector();
295 
296         JspPropertyGroup isXmlMatch = null;
297         JspPropertyGroup elIgnoredMatch = null;
298         JspPropertyGroup scriptingInvalidMatch = null;
299         JspPropertyGroup pageEncodingMatch = null;
300         JspPropertyGroup deferedSyntaxAllowedAsLiteralMatch = null;
301         JspPropertyGroup trimDirectiveWhitespacesMatch = null;
302 
303         Iterator iter = jspProperties.iterator();
304         while (iter.hasNext()) {
305 
306             JspPropertyGroup jpg = (JspPropertyGroup) iter.next();
307             JspProperty jp = jpg.getJspProperty();
308 
309             // (arrays will be the same length)
310             String extension = jpg.getExtension();
311             String path = jpg.getPath();
312 
313             if (extension == null) {
314                 // exact match pattern: /a/foo.jsp
315                 if (!uri.equals(path)) {
316                     // not matched;
317                     continue;
318                 }
319             } else {
320                 // Matching patterns *.ext or /p/*
321                 if (path != null && uriPath != null &&
322                         ! uriPath.startsWith(path)) {
323                     // not matched
324                     continue;
325                 }
326                 if (!extension.equals("*") &&
327                         !extension.equals(uriExtension)) {
328                     // not matched
329                     continue;
330                 }
331             }
332             // We have a match
333             // Add include-preludes and include-codas
334             if (jp.getIncludePrelude() != null) {
335                 includePreludes.addAll(jp.getIncludePrelude());
336             }
337             if (jp.getIncludeCoda() != null) {
338                 includeCodas.addAll(jp.getIncludeCoda());
339             }
340 
341             // If there is a previous match for the same property, remember
342             // the one that is more restrictive.
343             if (jp.isXml() != null) {
344                 isXmlMatch = selectProperty(isXmlMatch, jpg);
345             }
346             if (jp.isELIgnored() != null) {
347                 elIgnoredMatch = selectProperty(elIgnoredMatch, jpg);
348             }
349             if (jp.isScriptingInvalid() != null) {
350                 scriptingInvalidMatch =
351                     selectProperty(scriptingInvalidMatch, jpg);
352             }
353             if (jp.getPageEncoding() != null) {
354                 pageEncodingMatch = selectProperty(pageEncodingMatch, jpg);
355             }
356             if (jp.isDeferedSyntaxAllowedAsLiteral() != null) {
357                 deferedSyntaxAllowedAsLiteralMatch =
358                     selectProperty(deferedSyntaxAllowedAsLiteralMatch, jpg);
359             }
360             if (jp.isTrimDirectiveWhitespaces() != null) {
361                 trimDirectiveWhitespacesMatch =
362                     selectProperty(trimDirectiveWhitespacesMatch, jpg);
363             }
364         }
365 
366 
367         String isXml = defaultIsXml;
368         String isELIgnored = defaultIsELIgnored;
369         String isScriptingInvalid = defaultIsScriptingInvalid;
370         String pageEncoding = null;
371         String isDeferedSyntaxAllowedAsLiteral = defaultDeferedSyntaxAllowedAsLiteral;
372         String isTrimDirectiveWhitespaces = defaultTrimDirectiveWhitespaces;
373 
374         if (isXmlMatch != null) {
375             isXml = isXmlMatch.getJspProperty().isXml();
376         }
377         if (elIgnoredMatch != null) {
378             isELIgnored = elIgnoredMatch.getJspProperty().isELIgnored();
379         }
380         if (scriptingInvalidMatch != null) {
381             isScriptingInvalid =
382                 scriptingInvalidMatch.getJspProperty().isScriptingInvalid();
383         }
384         if (pageEncodingMatch != null) {
385             pageEncoding = pageEncodingMatch.getJspProperty().getPageEncoding();
386         }
387         if (deferedSyntaxAllowedAsLiteralMatch != null) {
388             isDeferedSyntaxAllowedAsLiteral =
389                 deferedSyntaxAllowedAsLiteralMatch.getJspProperty().isDeferedSyntaxAllowedAsLiteral();
390         }
391         if (trimDirectiveWhitespacesMatch != null) {
392             isTrimDirectiveWhitespaces =
393                 trimDirectiveWhitespacesMatch.getJspProperty().isTrimDirectiveWhitespaces();
394         }
395 
396         return new JspProperty(isXml, isELIgnored, isScriptingInvalid,
397                 pageEncoding, includePreludes, includeCodas, 
398                 isDeferedSyntaxAllowedAsLiteral, isTrimDirectiveWhitespaces);
399     }

401     /**
402      * To find out if an uri matches an url pattern in jsp config.  If so,
403      * then the uri is a JSP page.  This is used primarily for jspc.
404      *
405      * @param uri the uri
406      * @return true if uri is a JSP page
407      * @throws JasperException in case of Jasper errors
408      */
409     public boolean isJspPage(String uri) throws JasperException {
410 
411         init();
412         if (jspProperties == null) {
413             return false;
414         }
415 
416         String uriPath = null;
417         int index = uri.lastIndexOf('/');
418         if (index >=0 ) {
419             uriPath = uri.substring(0, index+1);
420         }
421         String uriExtension = null;
422         index = uri.lastIndexOf('.');
423         if (index >=0) {
424             uriExtension = uri.substring(index+1);
425         }
426 
427         for (Object jspProperty : jspProperties) {
428 
429             JspPropertyGroup jpg = (JspPropertyGroup) jspProperty;
430             JspProperty jp = jpg.getJspProperty();
431 
432             String extension = jpg.getExtension();
433             String path = jpg.getPath();
434 
435             if (extension == null) {
436                 if (uri.equals(path)) {
437                     // There is an exact match
438                     return true;
439                 }
440             } else {
441                 if ((path == null || path.equals(uriPath)) &&
442                         (extension.equals("*") || extension.equals(uriExtension))) {
443                     // Matches *, *.ext, /p/*, or /p/*.ext
444                     return true;
445                 }
446             }
447         }
448         return false;
449     }

451     static class JspPropertyGroup {
452         private String path;
453         private String extension;
454         private JspProperty jspProperty;
455 
456         JspPropertyGroup(String path, String extension,
457                 JspProperty jspProperty) {
458             this.path = path;
459             this.extension = extension;
460             this.jspProperty = jspProperty;
461         }
462 
463         public String getPath() {
464             return path;
465         }
466 
467         public String getExtension() {
468             return extension;
469         }
470 
471         public JspProperty getJspProperty() {
472             return jspProperty;
473         }
474     }

488                 String scriptingInvalid, String pageEncoding,
489                 Vector includePrelude, Vector includeCoda,

491                 String trimDirectiveWhitespaces) {

494             this.elIgnored = elIgnored;
495             this.scriptingInvalid = scriptingInvalid;
496             this.pageEncoding = pageEncoding;
497             this.includePrelude = includePrelude;
498             this.includeCoda = includeCoda;
499             this.deferedSyntaxAllowedAsLiteral = deferedSyntaxAllowedAsLiteral;
500             this.trimDirectiveWhitespaces = trimDirectiveWhitespaces;
501         }
502

504             return isXml;
505         }
506 
507         public String isELIgnored() {
508             return elIgnored;
509         }
510 
511         public String isScriptingInvalid() {
512             return scriptingInvalid;
513         }
514 
515         public String getPageEncoding() {
516             return pageEncoding;
517         }
518 
519         public Vector getIncludePrelude() {
520             return includePrelude;
521         }
522 
523         public Vector getIncludeCoda() {
524             return includeCoda;

526         
527         public String isDeferedSyntaxAllowedAsLiteral() {
