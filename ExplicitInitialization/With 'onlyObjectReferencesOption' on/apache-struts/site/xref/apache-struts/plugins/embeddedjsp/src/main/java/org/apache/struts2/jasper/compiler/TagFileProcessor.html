149         public void visit(Node.TagDirective n) throws JasperException {
150 
151             JspUtil.checkAttributes("Tag directive", n, tagDirectiveAttrs, err);
152 
153             bodycontent = checkConflict(n, bodycontent, "body-content");
154             if (bodycontent != null
155                     && !bodycontent
156                             .equalsIgnoreCase(TagInfo.BODY_CONTENT_EMPTY)
157                     && !bodycontent
158                             .equalsIgnoreCase(TagInfo.BODY_CONTENT_TAG_DEPENDENT)
159                     && !bodycontent
160                             .equalsIgnoreCase(TagInfo.BODY_CONTENT_SCRIPTLESS)) {
161                 err.jspError(n, "jsp.error.tagdirective.badbodycontent",
162                         bodycontent);
163             }
164             dynamicAttrsMapName = checkConflict(n, dynamicAttrsMapName,
165                     "dynamic-attributes");
166             if (dynamicAttrsMapName != null) {
167                 checkUniqueName(dynamicAttrsMapName, TAG_DYNAMIC, n);
168             }
169             smallIcon = checkConflict(n, smallIcon, "small-icon");
170             largeIcon = checkConflict(n, largeIcon, "large-icon");
171             description = checkConflict(n, description, "description");
172             displayName = checkConflict(n, displayName, "display-name");
173             example = checkConflict(n, example, "example");
174         }
175 
176         private String checkConflict(Node n, String oldAttrValue, String attr)
177                 throws JasperException {
178 
179             String result = oldAttrValue;
180             String attrValue = n.getAttributeValue(attr);
181             if (attrValue != null) {
182                 if (oldAttrValue != null && !oldAttrValue.equals(attrValue)) {
183                     err.jspError(n, "jsp.error.tag.conflict.attr", attr,
184                             oldAttrValue, attrValue);
185                 }
186                 result = attrValue;
187             }
188             return result;
189         }

191         public void visit(Node.AttributeDirective n) throws JasperException {
192 
193             JspUtil.checkAttributes("Attribute directive", n,
194                     attributeDirectiveAttrs, err);
195 
196             // JSP 2.1 Table JSP.8-3
197             // handle deferredValue and deferredValueType
198             boolean deferredValue = false;
199             boolean deferredValueSpecified = false;
200             String deferredValueString = n.getAttributeValue("deferredValue");
201             if (deferredValueString != null) {
202                 deferredValueSpecified = true;
203                 deferredValue = JspUtil.booleanValue(deferredValueString);
204             }
205             String deferredValueType = n.getAttributeValue("deferredValueType");
206             if (deferredValueType != null) {
207                 if (deferredValueSpecified && !deferredValue) {
208                     err.jspError(n, "jsp.error.deferredvaluetypewithoutdeferredvalue");
209                 } else {
210                     deferredValue = true;
211                 }
212             } else if (deferredValue) {
213                 deferredValueType = "java.lang.Object";
214             } else {
215                 deferredValueType = "java.lang.String";
216             }
217 
218             // JSP 2.1 Table JSP.8-3
219             // handle deferredMethod and deferredMethodSignature
220             boolean deferredMethod = false;
221             boolean deferredMethodSpecified = false;
222             String deferredMethodString = n.getAttributeValue("deferredMethod");
223             if (deferredMethodString != null) {
224                 deferredMethodSpecified = true;
225                 deferredMethod = JspUtil.booleanValue(deferredMethodString);
226             }
227             String deferredMethodSignature = n
228                     .getAttributeValue("deferredMethodSignature");
229             if (deferredMethodSignature != null) {
230                 if (deferredMethodSpecified && !deferredMethod) {
231                     err.jspError(n, "jsp.error.deferredmethodsignaturewithoutdeferredmethod");
232                 } else {
233                     deferredMethod = true;
234                 }
235             } else if (deferredMethod) {
236                 deferredMethodSignature = "void methodname()";
237             }
238 
239             if (deferredMethod && deferredValue) {
240                 err.jspError(n, "jsp.error.deferredmethodandvalue");
241             }
242             
243             String attrName = n.getAttributeValue("name");
244             boolean required = JspUtil.booleanValue(n
245                     .getAttributeValue("required"));
246             boolean rtexprvalue = true;
247             String rtexprvalueString = n.getAttributeValue("rtexprvalue");
248             if (rtexprvalueString != null) {
249                 rtexprvalue = JspUtil.booleanValue(rtexprvalueString);
250             }
251             boolean fragment = JspUtil.booleanValue(n
252                     .getAttributeValue("fragment"));
253             String type = n.getAttributeValue("type");
254             if (fragment) {
255                 // type is fixed to "JspFragment" and a translation error
256                 // must occur if specified.
257                 if (type != null) {
258                     err.jspError(n, "jsp.error.fragmentwithtype");
259                 }
260                 // rtexprvalue is fixed to "true" and a translation error
261                 // must occur if specified.
262                 rtexprvalue = true;
263                 if (rtexprvalueString != null) {
264                     err.jspError(n, "jsp.error.frgmentwithrtexprvalue");
265                 }
266             } else {
267                 if (type == null)
268                     type = "java.lang.String";
269                 
270                 if (deferredValue) {
271                     type = ValueExpression.class.getName();
272                 } else if (deferredMethod) {
273                     type = MethodExpression.class.getName();
274                 }
275             }
276 
277             if (("2.0".equals(tagLibInfo.getRequiredVersion()) || ("1.2".equals(tagLibInfo.getRequiredVersion())))
278                     && (deferredMethodSpecified || deferredMethod
279                             || deferredValueSpecified || deferredValue)) {
280                 err.jspError("jsp.error.invalid.version", path);
281             }
282             
283             TagAttributeInfo tagAttributeInfo = new TagAttributeInfo(attrName,
284                     required, type, rtexprvalue, fragment, null, deferredValue,
285                     deferredMethod, deferredValueType, deferredMethodSignature);
286             attributeVector.addElement(tagAttributeInfo);
287             checkUniqueName(attrName, ATTR_NAME, n, tagAttributeInfo);
288         }

290         public void visit(Node.VariableDirective n) throws JasperException {
291 
292             JspUtil.checkAttributes("Variable directive", n,
293                     variableDirectiveAttrs, err);
294 
295             String nameGiven = n.getAttributeValue("name-given");
296             String nameFromAttribute = n
297                     .getAttributeValue("name-from-attribute");
298             if (nameGiven == null && nameFromAttribute == null) {
299                 err.jspError("jsp.error.variable.either.name");
300             }
301 
302             if (nameGiven != null && nameFromAttribute != null) {
303                 err.jspError("jsp.error.variable.both.name");
304             }
305 
306             String alias = n.getAttributeValue("alias");
307             if (nameFromAttribute != null && alias == null
308                     || nameFromAttribute == null && alias != null) {
309                 err.jspError("jsp.error.variable.alias");
310             }
311 
312             String className = n.getAttributeValue("variable-class");
313             if (className == null)
314                 className = "java.lang.String";
315 
316             String declareStr = n.getAttributeValue("declare");
317             boolean declare = true;
318             if (declareStr != null)
319                 declare = JspUtil.booleanValue(declareStr);
320 
321             int scope = VariableInfo.NESTED;
322             String scopeStr = n.getAttributeValue("scope");
323             if (scopeStr != null) {
324                 if ("NESTED".equals(scopeStr)) {
325                     // Already the default
326                 } else if ("AT_BEGIN".equals(scopeStr)) {
327                     scope = VariableInfo.AT_BEGIN;
328                 } else if ("AT_END".equals(scopeStr)) {
329                     scope = VariableInfo.AT_END;
330                 }
331             }
332 
333             if (nameFromAttribute != null) {
334                 /*
335                  * An alias has been specified. We use 'nameGiven' to hold the
336                  * value of the alias, and 'nameFromAttribute' to hold the name
337                  * of the attribute whose value (at invocation-time) denotes the
338                  * name of the variable that is being aliased
339                  */
340                 nameGiven = alias;
341                 checkUniqueName(nameFromAttribute, VAR_NAME_FROM, n);
342                 checkUniqueName(alias, VAR_ALIAS, n);
343             } else {
344                 // name-given specified
345                 checkUniqueName(nameGiven, VAR_NAME_GIVEN, n);
346             }
347 
348             variableVector.addElement(new TagVariableInfo(nameGiven,
349                     nameFromAttribute, className, declare, scope));
350         }

374         public TagInfo getTagInfo() throws JasperException {
375 
376             if (name == null) {
377                 // XXX Get it from tag file name
378             }
379 
380             if (bodycontent == null) {
381                 bodycontent = TagInfo.BODY_CONTENT_SCRIPTLESS;
382             }
383 
384             String tagClassName = JspUtil.getTagHandlerClassName(
385                     path, tagLibInfo.getReliableURN(), err);
386 
387             TagVariableInfo[] tagVariableInfos = new TagVariableInfo[variableVector
388                     .size()];
389             variableVector.copyInto(tagVariableInfos);
390 
391             TagAttributeInfo[] tagAttributeInfo = new TagAttributeInfo[attributeVector
392                     .size()];
393             attributeVector.copyInto(tagAttributeInfo);
394 
395             return new JasperTagInfo(name, tagClassName, bodycontent,
396                     description, tagLibInfo, tei, tagAttributeInfo,
397                     displayName, smallIcon, largeIcon, tagVariableInfos,
398                     dynamicAttrsMapName);
399         }

427         /**
428          * Reports a translation error if names specified in attributes of
429          * directives are not unique in this translation unit.
430          * 
431          * The value of the following attributes must be unique. 1. 'name'
432          * attribute of an attribute directive 2. 'name-given' attribute of a
433          * variable directive 3. 'alias' attribute of variable directive 4.
434          * 'dynamic-attributes' of a tag directive except that
435          * 'dynamic-attributes' can (and must) have the same value when it
436          * appears in multiple tag directives.
437          * 
438          * Also, 'name-from' attribute of a variable directive cannot have the
439          * same value as that from another variable directive.
440          */
441         private void checkUniqueName(String name, String type, Node n)
442                 throws JasperException {
443             checkUniqueName(name, type, n, null);
444         }
445 
446         private void checkUniqueName(String name, String type, Node n,
447                 TagAttributeInfo attr) throws JasperException {
448 
449             HashMap table = (type == VAR_NAME_FROM) ? nameFromTable : nameTable;
450             NameEntry nameEntry = (NameEntry) table.get(name);
451             if (nameEntry != null) {
452                 if (type != TAG_DYNAMIC || nameEntry.getType() != TAG_DYNAMIC) {
453                     int line = nameEntry.getNode().getStart().getLineNumber();
454                     err.jspError(n, "jsp.error.tagfile.nameNotUnique", type,
455                             nameEntry.getType(), Integer.toString(line));
456                 }
457             } else {
458                 table.put(name, new NameEntry(type, n, attr));
459             }
460         }

462         /**
463          * Perform miscellean checks after the nodes are visited.
464          */
465         void postCheck() throws JasperException {
466             // Check that var.name-from-attributes has valid values.
467             Iterator iter = nameFromTable.keySet().iterator();
468             while (iter.hasNext()) {
469                 String nameFrom = (String) iter.next();
470                 NameEntry nameEntry = (NameEntry) nameTable.get(nameFrom);
471                 NameEntry nameFromEntry = (NameEntry) nameFromTable
472                         .get(nameFrom);
473                 Node nameFromNode = nameFromEntry.getNode();
474                 if (nameEntry == null) {
475                     err.jspError(nameFromNode,
476                             "jsp.error.tagfile.nameFrom.noAttribute", nameFrom);
477                 } else {
478                     Node node = nameEntry.getNode();
479                     TagAttributeInfo tagAttr = nameEntry.getTagAttributeInfo();
480                     if (!"java.lang.String".equals(tagAttr.getTypeName())
481                             || !tagAttr.isRequired()
482                             || tagAttr.canBeRequestTime()) {
483                         err.jspError(nameFromNode,
484                                 "jsp.error.tagfile.nameFrom.badAttribute",
485                                 nameFrom, Integer.toString(node.getStart()
486                                         .getLineNumber()));
487                     }
488                 }
489             }
490         }

520     /**
521      * Parses the tag file, and collects information on the directives included
522      * in it. The method is used to obtain the info on the tag file, when the
523      * handler that it represents is referenced. The tag file is not compiled
524      * here.
525      * 
526      * @param pc
527      *            the current ParserController used in this compilation
528      * @param name
529      *            the tag name as specified in the TLD
530      * @param tagfile
531      *            the path for the tagfile
532      * @param tagFileJarUrl
533      *            the url for the Jar containign the tag file 
534      * @param tagLibInfo
535      *            the TagLibraryInfo object associated with this TagInfo
536      * @return a TagInfo object assembled from the directives in the tag file.
537      */
538     public static TagInfo parseTagFileDirectives(ParserController pc,
539             String name, String path, URL tagFileJarUrl, TagLibraryInfo tagLibInfo)
540             throws JasperException {
541 
542         ErrorDispatcher err = pc.getCompiler().getErrorDispatcher();
543 
544         Node.Nodes page = null;
545         try {
546             page = pc.parseTagFileDirectives(path, tagFileJarUrl);
547         } catch (FileNotFoundException e) {
548             err.jspError("jsp.error.file.not.found", path);
549         } catch (IOException e) {
550             err.jspError("jsp.error.file.not.found", path);
551         }
552 
553         TagFileDirectiveVisitor tagFileVisitor = new TagFileDirectiveVisitor(pc
554                 .getCompiler(), tagLibInfo, name, path);
555         page.visit(tagFileVisitor);
556         tagFileVisitor.postCheck();
557 
558         return tagFileVisitor.getTagInfo();
559     }

561     /**
562      * Compiles and loads a tagfile.
563      */
564     private Class loadTagFile(Compiler compiler, String tagFilePath,
565             TagInfo tagInfo, PageInfo parentPageInfo) throws JasperException {
566 
567         URL tagFileJarUrl = null;
568         if (tagFilePath.startsWith("/META-INF/")) {
569             try { 
570                 tagFileJarUrl = new URL("jar:" +
571                         compiler.getCompilationContext().getTldLocation(
572                         tagInfo.getTagLibrary().getURI())[0] + "!/");
573             } catch (MalformedURLException e) {
574                 // Ignore - tagFileJarUrl will be null
575             }
576         }
577         String tagFileJarPath;
578         if (tagFileJarUrl == null) {
579             tagFileJarPath = "";
580         } else {
581             tagFileJarPath = tagFileJarUrl.toString();
582         }
583 
584         JspCompilationContext ctxt = compiler.getCompilationContext();
585         JspRuntimeContext rctxt = ctxt.getRuntimeContext();
586         JspServletWrapper wrapper = rctxt.getWrapper(tagFileJarPath + tagFilePath);
587 
588         synchronized (rctxt) {
589             if (wrapper == null) {
590                 wrapper = new JspServletWrapper(ctxt.getServletContext(), ctxt
591                         .getOptions(), tagFilePath, tagInfo, ctxt
592                         .getRuntimeContext(), tagFileJarUrl);
593                 rctxt.addWrapper(tagFileJarPath + tagFilePath, wrapper);
594 
595                 // Use same classloader and classpath for compiling tag files
596                 wrapper.getJspEngineContext().setClassLoader(
597                         (URLClassLoader) ctxt.getClassLoader());
598                 wrapper.getJspEngineContext().setClassPath(ctxt.getClassPath());
599             } else {
600                 // Make sure that JspCompilationContext gets the latest TagInfo
601                 // for the tag file. TagInfo instance was created the last
602                 // time the tag file was scanned for directives, and the tag
603                 // file may have been modified since then.
604                 wrapper.getJspEngineContext().setTagInfo(tagInfo);
605             }
606 
607             Class tagClazz;
608             int tripCount = wrapper.incTripCount();
609             try {
610                 if (tripCount > 0) {
611                     // When tripCount is greater than zero, a circular
612                     // dependency exists. The circularily dependant tag
613                     // file is compiled in prototype mode, to avoid infinite
614                     // recursion.
615 
616                     JspServletWrapper tempWrapper = new JspServletWrapper(ctxt
617                             .getServletContext(), ctxt.getOptions(),
618                             tagFilePath, tagInfo, ctxt.getRuntimeContext(),
619                             ctxt.getTagFileJarUrl(tagFilePath));
620                     tagClazz = tempWrapper.loadTagFilePrototype();
621                     tempVector.add(tempWrapper.getJspEngineContext()
622                             .getCompiler());
623                 } else {
624                     tagClazz = wrapper.loadTagFile();
625                 }
626             } finally {
627                 wrapper.decTripCount();
628             }
629 
630             // Add the dependants for this tag file to its parent's
631             // dependant list. The only reliable dependency information
632             // can only be obtained from the tag instance.
633             try {
634                 Object tagIns = tagClazz.newInstance();
635                 if (tagIns instanceof JspSourceDependent) {
636                     Iterator iter = ((List) ((JspSourceDependent) tagIns)
637                             .getDependants()).iterator();
638                     while (iter.hasNext()) {
639                         parentPageInfo.addDependant((String) iter.next());
640                     }
641                 }
642             } catch (Exception e) {
643                 // ignore errors
644             }
645 
646             return tagClazz;
647         }
648     }

650     /*
651      * Visitor which scans the page and looks for tag handlers that are tag
652      * files, compiling (if necessary) and loading them.
653      */
654     private class TagFileLoaderVisitor extends Node.Visitor {
655 
656         private Compiler compiler;
657 
658         private PageInfo pageInfo;
659 
660         TagFileLoaderVisitor(Compiler compiler) {
661 
662             this.compiler = compiler;
663             this.pageInfo = compiler.getPageInfo();
664         }
665 
666         public void visit(Node.CustomTag n) throws JasperException {
667             TagFileInfo tagFileInfo = n.getTagFileInfo();
668             if (tagFileInfo != null) {
669                 String tagFilePath = tagFileInfo.getPath();
670                 if (tagFilePath.startsWith("/META-INF/")) {
671                     // For tags in JARs, add the TLD and the tag as a dependency
672                     String[] location =
673                         compiler.getCompilationContext().getTldLocation(
674                             tagFileInfo.getTagInfo().getTagLibrary().getURI());
675                     // Add TLD
676                     pageInfo.addDependant("jar:" + location[0] + "!/" +
677                             location[1]);
678                     // Add Tag
679                     pageInfo.addDependant("jar:" + location[0] + "!" +
680                             tagFilePath);
681                 } else {
682                     pageInfo.addDependant(tagFilePath);
683                 }
684                 Class c = loadTagFile(compiler, tagFilePath, n.getTagInfo(),
685                         pageInfo);
686                 n.setTagHandlerClass(c);
687             }
688             visitBody(n);
689         }
690     }

692     /**
693      * Implements a phase of the translation that compiles (if necessary) the
694      * tag files used in a JSP files. The directives in the tag files are
695      * assumed to have been proccessed and encapsulated as TagFileInfo in the
696      * CustomTag nodes.
697      */
698     public void loadTagFiles(Compiler compiler, Node.Nodes page)
699             throws JasperException {
700 
701         tempVector = new Vector();
702         page.visit(new TagFileLoaderVisitor(compiler));
703     }
704 
705     /**
706      * Removed the java and class files for the tag prototype generated from the
707      * current compilation.
708      * 
709      * @param classFileName
710      *            If non-null, remove only the class file with with this name.
711      */
712     public void removeProtoTypeFiles(String classFileName) {
713         Iterator iter = tempVector.iterator();
714         while (iter.hasNext()) {
715             Compiler c = (Compiler) iter.next();
716             if (classFileName == null) {
717                 c.removeGeneratedClassFiles();
718             } else if (classFileName.equals(c.getCompilationContext()
719                     .getClassFileName())) {
720                 c.removeGeneratedClassFiles();
721                 tempVector.remove(c);
722                 return;
723             }
724         }
725     }
