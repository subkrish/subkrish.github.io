190     /**
191      * Initialize the portlet with the init parameters from <tt>portlet.xml</tt>
192      *
193      * @param cfg portlet configuration
194      * @throws PortletException in case of errors
195      */
196     public void init(PortletConfig cfg) throws PortletException {
197         super.init(cfg);
198         LOG.debug("Initializing portlet {}", getPortletName());
199 
200         Map<String, String> params = new HashMap<String, String>();
201         for (Enumeration e = cfg.getInitParameterNames(); e.hasMoreElements(); ) {
202             String name = (String) e.nextElement();
203             String value = cfg.getInitParameter(name);
204             params.put(name, value);
205         }
206 
207         servletContext = new PortletServletContext(cfg.getPortletContext());
208         dispatcherUtils = new Dispatcher(servletContext, params);
209         dispatcherUtils.init();
210 
211         // For testability
212         if (factory == null) {
213             factory = dispatcherUtils.getContainer().getInstance(ActionProxyFactory.class);
214         }
215         portletNamespace = cfg.getInitParameter("portletNamespace");
216         LOG.debug("PortletNamespace: {}", portletNamespace);
217 
218         parseModeConfig(actionMap, cfg, PortletMode.VIEW, "viewNamespace",
219                 "defaultViewAction");
220         parseModeConfig(actionMap, cfg, PortletMode.EDIT, "editNamespace",
221                 "defaultEditAction");
222         parseModeConfig(actionMap, cfg, PortletMode.HELP, "helpNamespace",
223                 "defaultHelpAction");
224         parseModeConfig(actionMap, cfg, new PortletMode("config"), "configNamespace",
225                 "defaultConfigAction");
226         parseModeConfig(actionMap, cfg, new PortletMode("about"), "aboutNamespace",
227                 "defaultAboutAction");
228         parseModeConfig(actionMap, cfg, new PortletMode("print"), "printNamespace",
229                 "defaultPrintAction");
230         parseModeConfig(actionMap, cfg, new PortletMode("preview"), "previewNamespace",
231                 "defaultPreviewAction");
232         parseModeConfig(actionMap, cfg, new PortletMode("edit_defaults"),
233                 "editDefaultsNamespace", "defaultEditDefaultsAction");
234         if (StringUtils.isEmpty(portletNamespace)) {
235             portletNamespace = "";
236         }
237 
238         container = dispatcherUtils.getContainer();
239         actionMapper = container.getInstance(ActionMapper.class);
240     }

242     /**
243      * Parse the mode to namespace mappings configured in portlet.xml
244      * @param actionMap The map with mode <-> default action mapping.
245      * @param portletConfig The PortletConfig.
246      * @param portletMode The PortletMode.
247      * @param nameSpaceParam Name of the init parameter where the namespace for the mode
248      * is configured.
249      * @param defaultActionParam Name of the init parameter where the default action to
250      * execute for the mode is configured.
251      */
252     void parseModeConfig(Map<PortletMode, ActionMapping> actionMap, PortletConfig portletConfig,
253                          PortletMode portletMode, String nameSpaceParam,
254                          String defaultActionParam) {
255         String namespace = portletConfig.getInitParameter(nameSpaceParam);
256         if (StringUtils.isEmpty(namespace)) {
257             namespace = "";
258         }
259         modeMap.put(portletMode, namespace);
260         String defaultAction = portletConfig.getInitParameter(defaultActionParam);
261         String method = null;
262         if (StringUtils.isEmpty(defaultAction)) {
263             defaultAction = DEFAULT_ACTION_NAME;
264         }
265         if (defaultAction.indexOf('!') >= 0) {
266             method = defaultAction.substring(defaultAction.indexOf('!') + 1);
267             defaultAction = defaultAction.substring(0, defaultAction.indexOf('!'));
268         }
269         StringBuilder fullPath = new StringBuilder();
270         if (StringUtils.isNotEmpty(portletNamespace)) {
271             fullPath.append(portletNamespace);
272         }
273         if (StringUtils.isNotEmpty(namespace)) {
274             fullPath.append(namespace).append("/");
275         } else {
276             fullPath.append("/");
277         }
278         fullPath.append(defaultAction);
279         ActionMapping mapping = new ActionMapping();
280         mapping.setName(getActionName(fullPath.toString()));
281         mapping.setNamespace(getNamespace(fullPath.toString()));
282         if (method != null) {
283             mapping.setMethod(method);
284         }
285         actionMap.put(portletMode, mapping);
286     }

288     /**
289      * Service an action from the <tt>event</tt> phase.
290      *
291      * @param request action request
292      * @param response  action response
293      *
294      * @throws PortletException in case of errors
295      * @throws IOException in case of IO errors
296      *
297      * @see javax.portlet.Portlet#processAction(javax.portlet.ActionRequest,
298      *      javax.portlet.ActionResponse)
299      */
300     public void processAction(ActionRequest request, ActionResponse response)
301             throws PortletException, IOException {
302         if (LOG.isDebugEnabled()) {
303             LOG.debug("Entering processAction in mode ", request.getPortletMode().toString());
304         }
305         resetActionContext();
306         try {
307             serviceAction(request, response, getRequestMap(request), getParameterMap(request),
308                     getSessionMap(request), getApplicationMap(),
309                     portletNamespace, PortletPhase.ACTION_PHASE);
310             if (LOG.isDebugEnabled()) LOG.debug("Leaving processAction");
311         } finally {
312             ActionContext.setContext(null);
313         }
314     }

316     /**
317      * Service an action from the <tt>render</tt> phase.
318      *
319      * @param request render request
320      * @param response  render response
321      *
322      * @throws PortletException in case of errors
323      * @throws IOException in case of IO errors
324      *
325      * @see javax.portlet.Portlet#render(javax.portlet.RenderRequest,
326      *      javax.portlet.RenderResponse)
327      */
328     public void render(RenderRequest request, RenderResponse response)
329             throws PortletException, IOException {
330 
331         if (LOG.isDebugEnabled()) {
332             LOG.debug("Entering render in mode ", request.getPortletMode().toString());
333         }
334         resetActionContext();
335         response.setTitle(getTitle(request));
336         if (!request.getWindowState().equals(WindowState.MINIMIZED)) {
337             try {
338                 // Check to see if an event set the render to be included directly
339                 serviceAction(request, response, getRequestMap(request), getParameterMap(request),
340                         getSessionMap(request), getApplicationMap(),
341                         portletNamespace, PortletPhase.RENDER_PHASE);
342                 if (LOG.isDebugEnabled()) LOG.debug("Leaving render");
343             } finally {
344                 resetActionContext();
345             }
346         }
347     }

356     /**
357      * Merges all application and portlet attributes into a single
358      * <tt>HashMap</tt> to represent the entire <tt>Action</tt> context.
359      *
360      * @param requestMap a Map of all request attributes.
361      * @param parameterMap a Map of all request parameters.
362      * @param sessionMap a Map of all session attributes.
363      * @param applicationMap a Map of all servlet context attributes.
364      * @param request the PortletRequest object.
365      * @param response the PortletResponse object.
366      * @param servletRequest the HttpServletRequest object.
367      * @param servletResponse the HttpServletResponse object.
368      * @param servletContext the ServletContext object.
369      * @param portletConfig the PortletConfig object.
370      * @param phase The portlet phase (render or action, see
371      *        {@link PortletConstants})
372      * @return a HashMap representing the <tt>Action</tt> context.
373      *
374      * @throws IOException in case of IO errors
375      */
376     public HashMap<String, Object> createContextMap(Map<String, Object> requestMap, Map<String, String[]> parameterMap,
377                                                     Map<String, Object> sessionMap, Map<String, Object> applicationMap,
378                                                     PortletRequest request, PortletResponse response, HttpServletRequest servletRequest,
379                                                     HttpServletResponse servletResponse, ServletContext servletContext,
380                                                     PortletConfig portletConfig, PortletPhase phase) throws IOException {
381 
382         // TODO Must put http request/response objects into map for use with
383         container.inject(servletRequest);
384 
385         // ServletActionContext
386         HashMap<String, Object> extraContext = new HashMap<String, Object>();
387         // The dummy servlet objects. Eases reuse of existing interceptors that uses the servlet objects.
388         extraContext.put(StrutsStatics.HTTP_REQUEST, servletRequest);
389         extraContext.put(StrutsStatics.HTTP_RESPONSE, servletResponse);
390         extraContext.put(StrutsStatics.SERVLET_CONTEXT, servletContext);
391         // End dummy servlet objects
392         extraContext.put(ActionContext.PARAMETERS, HttpParameters.create(parameterMap).build());
393         extraContext.put(ActionContext.SESSION, sessionMap);
394         extraContext.put(ActionContext.APPLICATION, applicationMap);
395 
396         String defaultLocale = container.getInstance(String.class, StrutsConstants.STRUTS_LOCALE);
397         Locale locale;
398         if (defaultLocale != null) {
399             locale = DefaultLocalizedTextProvider.localeFromString(defaultLocale, request.getLocale());
400         } else {
401             locale = request.getLocale();
402         }
403         extraContext.put(ActionContext.LOCALE, locale);
404 
405         extraContext.put(StrutsStatics.STRUTS_PORTLET_CONTEXT, getPortletContext());
406         extraContext.put(REQUEST, request);
407         extraContext.put(RESPONSE, response);
408         extraContext.put(PORTLET_CONFIG, portletConfig);
409         extraContext.put(PORTLET_NAMESPACE, portletNamespace);
410         extraContext.put(DEFAULT_ACTION_FOR_MODE, actionMap.get(request.getPortletMode()));
411         // helpers to get access to request/session/application scope
412         extraContext.put("request", requestMap);
413         extraContext.put("session", sessionMap);
414         extraContext.put("application", applicationMap);
415         extraContext.put("parameters", parameterMap);
416         extraContext.put(MODE_NAMESPACE_MAP, modeMap);
417         extraContext.put(PortletConstants.DEFAULT_ACTION_MAP, actionMap);
418 
419         extraContext.put(PortletConstants.PHASE, phase);
420 
421         AttributeMap attrMap = new AttributeMap(extraContext);
422         extraContext.put("attr", attrMap);
423 
424         return extraContext;
425     }

427     /**
428      * Loads the action and executes it. This method first creates the action
429      * context from the given parameters then loads an <tt>ActionProxy</tt>
430      * from the given action name and namespace. After that, the action is
431      * executed and output channels throught the response object.
432      *
433      * @param request the HttpServletRequest object.
434      * @param response the HttpServletResponse object.
435      * @param requestMap a Map of request attributes.
436      * @param parameterMap a Map of request parameters.
437      * @param sessionMap a Map of all session attributes.
438      * @param applicationMap a Map of all application attributes.
439      * @param portletNamespace the namespace or context of the action.
440      * @param phase The portlet phase (render or action, see {@link PortletConstants})
441      *
442      * @throws PortletException in case of errors
443      */
444     public void serviceAction(PortletRequest request, PortletResponse response, Map<String, Object> requestMap, Map<String, String[]> parameterMap,
445                               Map<String, Object> sessionMap, Map<String, Object> applicationMap, String portletNamespace,
446                               PortletPhase phase) throws PortletException {
447         if (LOG.isDebugEnabled()) LOG.debug("serviceAction");
448         Dispatcher.setInstance(dispatcherUtils);
449         String actionName = null;
450         String namespace;
451         try {
452             HttpServletRequest servletRequest = new PortletServletRequest(request, getPortletContext());
453             HttpServletResponse servletResponse = createPortletServletResponse(response);
454             if (phase.isAction()) {
455                 servletRequest = dispatcherUtils.wrapRequest(servletRequest);
456                 if (servletRequest instanceof MultiPartRequestWrapper) {
457                     // Multipart request. Request parameters are encoded in the multipart data,
458                     // so we need to manually add them to the parameter map.
459                     parameterMap.putAll(servletRequest.getParameterMap());
460                 }
461             }
462             container.inject(servletRequest);
463             ActionMapping mapping = getActionMapping(request, servletRequest);
464             actionName = mapping.getName();
465             if ("renderDirect".equals(actionName)) {
466                 namespace = request.getParameter(PortletConstants.RENDER_DIRECT_NAMESPACE);
467             } else {
468                 namespace = mapping.getNamespace();
469             }
470             HashMap<String, Object> extraContext = createContextMap(requestMap, parameterMap,
471                     sessionMap, applicationMap, request, response, servletRequest, servletResponse,
472                     servletContext, getPortletConfig(), phase);
473             extraContext.put(PortletConstants.ACTION_MAPPING, mapping);
474             if (LOG.isDebugEnabled()) {
475                 LOG.debug("Creating action proxy for name = " + actionName + ", namespace = " + namespace);
476             }
477             ActionProxy proxy = factory.createActionProxy(namespace, actionName, mapping.getMethod(), extraContext);
478             request.setAttribute("struts.valueStack", proxy.getInvocation().getStack());
479             proxy.execute();
480         } catch (ConfigurationException e) {
481             if (LOG.isErrorEnabled()) {
482                 LOG.error("Could not find action", e);
483             }
484             throw new PortletException("Could not find action " + actionName, e);
485         } catch (Exception e) {
486             if (LOG.isErrorEnabled()) {
487                 LOG.error("Could not execute action", e);
488             }
489             throw new PortletException("Error executing action " + actionName, e);
490         } finally {
491             Dispatcher.setInstance(null);
492         }
493     }

505     /**
506      * Gets the namespace of the action from the request. The namespace is the
507      * same as the portlet mode. E.g, view mode is mapped to namespace
508      * <code>view</code>, and edit mode is mapped to the namespace
509      * <code>edit</code>
510      *
511      * @param portletRequest the PortletRequest object.
512      * @param servletRequest the ServletRequest to use
513      *
514      * @return the namespace of the action.
515      */
516     protected ActionMapping getActionMapping(final PortletRequest portletRequest, final HttpServletRequest servletRequest) {
517         ActionMapping mapping;
518         String actionPath = getDefaultActionPath(portletRequest);
519         if (resetAction(portletRequest)) {
520             mapping = actionMap.get(portletRequest.getPortletMode());
521         } else {
522             actionPath = servletRequest.getParameter(ACTION_PARAM);
523             if (StringUtils.isEmpty(actionPath)) {
524                 mapping = actionMap.get(portletRequest.getPortletMode());
525             } else {
526 
527                 // Use the usual action mapper, but it is expecting an action extension
528                 // on the uri, so we add the default one, which should be ok as the
529                 // portlet is a portlet first, a servlet second
530                 mapping = actionMapper.getMapping(servletRequest, dispatcherUtils.getConfigurationManager());
531             }
532         }
533 
534         if (mapping == null) {
535             throw new StrutsException("Unable to locate action mapping for request, probably due to an invalid action path: " + actionPath);
536         }
537         return mapping;
538     }

544     /**
545      * Get the namespace part of the action path.
546      * @param actionPath Full path to action
547      * @return The namespace part.
548      */
549     String getNamespace(String actionPath) {
550         int idx = actionPath.lastIndexOf('/');
551         String namespace = "";
552         if (idx >= 0) {
553             namespace = actionPath.substring(0, idx);
554         }
555         return namespace;
556     }

558     /**
559      * Get the action name part of the action path.
560      * @param actionPath Full path to action
561      * @return The action name.
562      */
563     String getActionName(String actionPath) {
564         int idx = actionPath.lastIndexOf('/');
565         String action = actionPath;
566         if (idx >= 0) {
567             action = actionPath.substring(idx + 1);
568         }
569         return action;
570     }

617     /**
618      * Check to see if the action parameter is valid for the current portlet mode. If the portlet
619      * mode has been changed with the portal widgets, the action name is invalid, since the
620      * action name belongs to the previous executing portlet mode. If this method evaluates to
621      * <code>true</code> the <code>default&lt;Mode&gt;Action</code> is used instead.
622      * @param request The portlet request.
623      * @return <code>true</code> if the action should be reset.
624      */
625     private boolean resetAction(PortletRequest request) {
626         boolean reset = false;
627         Map paramMap = request.getParameterMap();
628         String[] modeParam = (String[]) paramMap.get(MODE_PARAM);
629         if (modeParam != null && modeParam.length == 1) {
630             String originatingMode = modeParam[0];
631             String currentMode = request.getPortletMode().toString();
632             if (!currentMode.equals(originatingMode)) {
633                 reset = true;
634             }
635         }
636         if (reset) {
637             request.setAttribute(ACTION_RESET, Boolean.TRUE);
638         } else {
639             request.setAttribute(ACTION_RESET, Boolean.FALSE);
640         }
641         return reset;
642     }
