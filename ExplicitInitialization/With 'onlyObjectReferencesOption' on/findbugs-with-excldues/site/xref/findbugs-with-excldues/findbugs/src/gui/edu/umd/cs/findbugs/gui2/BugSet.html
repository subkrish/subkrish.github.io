39  /**
40   * BugSet is what we use instead of SortedBugCollections. BugSet is somewhat
41   * poorly named, in that its actually a HashList of bugs, not a Set of them. (It
42   * can't be a set because we need to be able to sort it, also, HashList is great
43   * for doing contains and indexOf, its just slow for removing which we never
44   * need to do) The power of BugSet is in query. You can query a BugSet with a
45   * BugAspects, a list of StringPairs like <priority,high>,
46   * <designation,unclassified> and you will get out a new BugSet containing all
47   * of the bugs that are both high priority and unclassified. Also, after the
48   * first time a query is made, the results will come back instantly on future
49   * calls because the old queries are cached. Note that this caching can also
50   * lead to issues, problems with the BugTreeModel and the JTree getting out of
51   * sync, if there comes a time when the model and tree are out of sync but come
52   * back into sync if the tree is rebuilt, say by sorting the column headers, it
53   * probably means that resetData needs to be called on the model after doing one
54   * of its operations.
55   *
56   * @author Dan
57   *
58   */

129     static boolean suppress(BugLeafNode p) {
130         return !MainFrame.getInstance().shouldDisplayIssue(p.getBug());
131     }

133     /**
134      * we cache all values of each sortable that appear in the BugSet as we
135      * create it using cacheSortables, this makes it possible to only show
136      * branches that actually have bugs in them, and makes it faster by caching
137      * the results.
138      */
139     void cacheSortables() {
140         sortablesToStrings = new HashMap<Sortables, String[]>();
141     }
142 
143     String[] getDistinctValues(Sortables key) {
144         String[] list = sortablesToStrings.get(key);
145         if (list == null) {
146             list = computeDistinctValues(key);
147             sortablesToStrings.put(key, list);
148         }
149         return list;
150     }
151 
152     private static final String[] EMPTY_STRING_ARRAY = new String[0];
153 
154     String[] computeDistinctValues(Sortables key) {
155 
156         if (key == Sortables.DIVIDER) {
157             return EMPTY_STRING_ARRAY;
158         }
159 
160         Collection<String> list = new HashSet<String>();
161 
162         for (BugLeafNode p : mainList) {
163             if (suppress(p)) {
164                 continue;
165             }
166             BugInstance bug = p.getBug();
167 
168             String value = key.getFrom(bug);
169             list.add(value);
170 
171         }
172         String result[] = list.toArray(new String[list.size()]);
173         Collections.sort(Arrays.asList(result), new SortableStringComparator(key));
174         return result;
175 
176     }

206     /**
207      * A String pair has a key and a value. The key is the general category ie:
208      * Type The value is the value ie: Malicious Code.
209      *
210      * Query looks through a BugLeafNode set with a keyValuePair to see which
211      * BugLeafNodes inside match the value under the category of key.
212      *
213      * passing in a key of Abbrev and a value of MS should return a new BugSet
214      * with all the Mutable Static bugs in the current set Note also: This query
215      * will only be performed once, and then stored and reused if the same query
216      * is used again.
217      */
218     BugSet query(SortableValue keyValuePair) {
219         if (doneMap.containsKey(keyValuePair)) {
220             return doneMap.get(keyValuePair);
221         }
222         ArrayList<BugLeafNode> bugs = new ArrayList<BugLeafNode>();
223 
224         for (BugLeafNode b : mainList) {
225             if (b.matches(keyValuePair)) {
226                 bugs.add(b);
227             }
228         }
229 
230         BugSet temp = new BugSet(bugs);
231         doneMap.put(keyValuePair, temp);
232         return temp;
233     }

235     /*
236      * Sort the contents of the list by the Sortables in the order after the
237      * divider, if any.
238      */
239     void sortList() {
240 
241         final List<Sortables> order = MainFrame.getInstance().getSorter().getOrderAfterDivider();
242 
243         Comparator<BugLeafNode> comparator = new Comparator<BugLeafNode>() {
244             int compare(int one, int two) {
245                 if (one > two) {
246                     return 1;
247                 } else if (one < two) {
248                     return -1;
249                 }
250                 return 0;
251             }
252 
253             @Override
254             public int compare(BugLeafNode one, BugLeafNode two) {
255                 if (one == two) {
256                     return 0;
257                 }
258                 int result;
259                 for (Sortables i : order) {
260                     result = i.getBugLeafNodeComparator().compare(one, two);
261                     if (result != 0) {
262                         return result;
263                     }
264                 }
265                 BugInstance bugOne = one.getBug();
266                 BugInstance bugTwo = two.getBug();
267                 result = bugOne.getPrimaryClass().getClassName().compareTo(bugTwo.getPrimaryClass().getClassName());
268                 if (result != 0) {
269                     return result;
270                 }
271                 SourceLineAnnotation oneSource = bugOne.getPrimarySourceLineAnnotation();
272                 SourceLineAnnotation twoSource = bugTwo.getPrimarySourceLineAnnotation();
273                 result = oneSource.getClassName().compareTo(twoSource.getClassName());
274                 if (result != 0) {
275                     return result;
276                 }
277                 result = compare(oneSource.getStartLine(), twoSource.getStartLine());
278                 if (result != 0) {
279                     return result;
280                 }
281                 result = compare(oneSource.getEndLine(), twoSource.getEndLine());
282                 if (result != 0) {
283                     return result;
284                 }
285                 result = compare(oneSource.getStartBytecode(), twoSource.getStartBytecode());
286                 if (result != 0) {
287                     return result;
288                 }
289                 result = compare(oneSource.getEndBytecode(), twoSource.getEndBytecode());
290                 return result;
291 
292             }
293         };
294         ArrayList<BugLeafNode> copy = new ArrayList<BugLeafNode>(mainList);
295         Collections.sort(copy, comparator);
296         mainList = copy;
297 
298         if (SystemProperties.ASSERTIONS_ENABLED) {
299             for(int i = 0; i < mainList.size(); i++) {
300                 BugLeafNode nodeI = mainList.get(i);
301 
302                 for(int j = i+1; j < mainList.size(); j++) {
303                     BugLeafNode nodeJ = mainList.get(j);
304                     if (comparator.compare(nodeI, nodeJ) > 0) {
305                         throw new AssertionError(
306                                 String.format("bug list isn't consistently sorted (%d:%s) vs. (%d:%s)",
307                                         i, nodeI.getBug().getInstanceHash(), j, nodeJ.getBug().getInstanceHash()));
308                     }
309                 }}
310         }
311 
312 
313 
314     }

316     /**
317      *
318      * Contains takes a key/value pair
319      *
320      * @param keyValuePair
321      * @return true if a bug leaf from filterNoCache() matches the pair
322      */
323     public boolean contains(SortableValue keyValuePair) {
324         if (doneContainsMap.containsKey(keyValuePair)) {
325             return doneContainsMap.get(keyValuePair);
326         }
327 
328         for (BugLeafNode p : filteredBugsCached().mainList) {
329             if (p.matches(keyValuePair)) {
330                 doneContainsMap.put(keyValuePair, true);
331                 return true;
332             }
333         }
334         doneContainsMap.put(keyValuePair, false);
335         return false;
336     }

338     /**
339      * Gives you back the BugSet containing all bugs that match your query
340      */
341     public BugSet query(BugAspects a) {
342         BugSet result = this;
343         for (SortableValue sp : a) {
344             result = result.query(sp);
345         }
346 
347         return result;
348     }

378     private BugSet filteredBugsNoCache() {
379 
380         ArrayList<BugLeafNode> people = new ArrayList<BugLeafNode>();
381         for (BugLeafNode p : mainList) {
382             if (!suppress(p)) {
383                 people.add(p);
384             }
385         }
386         return new BugSet(people, false);
387     }
388 
389     BugSet cache = null;
390 
391     public void clearCache() {
392         cache = null;
393     }
394 
395     private BugSet filteredBugsCached() {
396         if (cache == null) {
397             cache = filteredBugsNoCache();
398         }
399         return cache;
400     }
401 
402     public BugSet getBugsMatchingFilter(Matcher m) {
403         ArrayList<BugLeafNode> people = new ArrayList<BugLeafNode>();
404         for (BugLeafNode p : mainList) {
405             if (!(m.match(p.getBug()))) {
406                 people.add(p);
407             }
408         }
409         return new BugSet(people, false);
410     }
