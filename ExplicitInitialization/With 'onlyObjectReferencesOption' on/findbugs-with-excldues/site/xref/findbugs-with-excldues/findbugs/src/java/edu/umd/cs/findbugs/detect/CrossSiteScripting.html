49      public CrossSiteScripting(BugReporter bugReporter) {
50          this.bugReporter = bugReporter;
51          this.accumulator = new BugAccumulator(bugReporter);
52          StringPassthruDatabase database = Global.getAnalysisCache().getDatabase(StringPassthruDatabase.class);
53          allFileNameStringMethods = database.getFileNameStringMethods();
54      }

58      OpcodeStack.Item top = null;

60      Pattern xmlSafe = Pattern.compile("\\p{Alnum}+");
61  
62      @Override
63      public void visit(Code code) {
64          isPlainText = false;
65          super.visit(code);
66          map.clear();
67          accumulator.reportAccumulatedBugs();
68      }

70      private void annotateAndReport(BugInstance bug, OpcodeStack.Item item) {
71          assert item.isServletParameterTainted();
72          String s = item.getHttpParameterName();
73          int pc = item.getInjectionPC();
74          if (s != null && xmlSafe.matcher(s).matches()) {
75              bug.addString(s).describe(StringAnnotation.PARAMETER_NAME_ROLE);
76          }
77          SourceLineAnnotation thisLine = SourceLineAnnotation.fromVisitedInstruction(this);
78          if (pc >= 0) {
79              SourceLineAnnotation source = SourceLineAnnotation.fromVisitedInstruction(this, pc);
80              if (thisLine.getStartLine() != source.getStartLine()) {
81                  bug.add(source).describe(SourceLineAnnotation.ROLE_GENERATED_AT);
82              }
83          }
84  
85          bug.addOptionalLocalVariable(this, item);
86          accumulator.accumulateBug(bug, this);
87      }

89      OpcodeStack.Item replaceTop = null;
90  
91      boolean isPlainText;

93      @Override
94      public void sawOpcode(int seen) {
95          if (replaceTop != null) {
96              stack.replaceTop(replaceTop);
97              replaceTop = null;
98          }
99  
100         OpcodeStack.Item oldTop = top;
101         top = null;
102         if (seen == INVOKESPECIAL || seen == INVOKESTATIC || seen == INVOKEINTERFACE || seen == INVOKEVIRTUAL) {
103             int[] params = allFileNameStringMethods.get(getMethodDescriptorOperand());
104             if(params != null) {
105                 int numArgs = getNumberArguments(getSigConstantOperand());
106                 for(int param : params) {
107                     OpcodeStack.Item path = stack.getStackItem(numArgs-1-param);
108                     if (isTainted(path)) {
109                         String bugPattern = taintPriority(path) == Priorities.HIGH_PRIORITY ? "PT_ABSOLUTE_PATH_TRAVERSAL"
110                                 : "PT_RELATIVE_PATH_TRAVERSAL";
111                         annotateAndReport(new BugInstance(this, bugPattern, Priorities.NORMAL_PRIORITY).addClassAndMethod(this)
112                                 .addCalledMethod(this), path);
113                     }
114                 }
115             }
116         }
117         if (seen == INVOKESPECIAL) {
118             String calledClassName = getClassConstantOperand();
119             String calledMethodName = getNameConstantOperand();
120             String calledMethodSig = getSigConstantOperand();
121 
122             if ("javax/servlet/http/Cookie".equals(calledClassName) && "<init>".equals(calledMethodName)
123                     && "(Ljava/lang/String;Ljava/lang/String;)V".equals(calledMethodSig)) {
124                 OpcodeStack.Item value = stack.getStackItem(0);
125                 OpcodeStack.Item name = stack.getStackItem(1);
126                 if (value.isServletParameterTainted() || name.isServletParameterTainted()) {
127                     int priority = Math.min(taintPriority(value), taintPriority(name));
128                     annotateAndReport(new BugInstance(this, "HRS_REQUEST_PARAMETER_TO_COOKIE", priority).addClassAndMethod(this),
129                             value.isServletParameterTainted() ? value : name);
130                 }
131 
132             }
133 
134         } else if (seen == INVOKEINTERFACE) {
135             String calledClassName = getClassConstantOperand();
136             String calledMethodName = getNameConstantOperand();
137             String calledMethodSig = getSigConstantOperand();
138             if ("javax/servlet/http/HttpServletResponse".equals(calledClassName) && "setContentType".equals(calledMethodName)) {
139                 OpcodeStack.Item writing = stack.getStackItem(0);
140                 if ("text/plain".equals(writing.getConstant())) {
141                     isPlainText = true;
142                 }
143             } else if ("javax/servlet/http/HttpSession".equals(calledClassName) && "setAttribute".equals(calledMethodName)) {
144 
145                 OpcodeStack.Item value = stack.getStackItem(0);
146                 OpcodeStack.Item name = stack.getStackItem(1);
147                 Object nameConstant = name.getConstant();
148                 if (nameConstant instanceof String) {
149                     map.put((String) nameConstant, value);
150                 }
151             } else if ("javax/servlet/http/HttpSession".equals(calledClassName) && "getAttribute".equals(calledMethodName)) {
152                 OpcodeStack.Item name = stack.getStackItem(0);
153                 Object nameConstant = name.getConstant();
154                 if (nameConstant instanceof String) {
155                     top = map.get(nameConstant);
156 
157                     if (isTainted(top)) {
158                         replaceTop = top;
159                     }
160                 }
161             } else if ("javax/servlet/http/HttpServletResponse".equals(calledClassName)
162                     && (calledMethodName.startsWith("send") || calledMethodName.endsWith("Header"))
163                     && calledMethodSig.endsWith("Ljava/lang/String;)V")) {
164 
165                 OpcodeStack.Item writing = stack.getStackItem(0);
166                 if (isTainted(writing)) {
167                     if ("sendError".equals(calledMethodName)) {
168                         annotateAndReport(
169                                 new BugInstance(this, "XSS_REQUEST_PARAMETER_TO_SEND_ERROR", taintPriority(writing))
170                                 .addClassAndMethod(this),
171                                 writing);
172                     } else {
173                         annotateAndReport(
174                                 new BugInstance(this, "HRS_REQUEST_PARAMETER_TO_HTTP_HEADER", taintPriority(writing))
175                                 .addClassAndMethod(this),
176                                 writing);
177                     }
178                 }
179             }
180 
181         } else if (seen == INVOKEVIRTUAL && !isPlainText) {
182             String calledClassName = getClassConstantOperand();
183             String calledMethodName = getNameConstantOperand();
184             String calledMethodSig = getSigConstantOperand();
185 
186             if ((calledMethodName.startsWith("print") || "write".equals(calledMethodName))
187                     && "javax/servlet/jsp/JspWriter".equals(calledClassName)
188                     && ("(Ljava/lang/Object;)V".equals(calledMethodSig) || "(Ljava/lang/String;)V".equals(calledMethodSig))) {
189                 OpcodeStack.Item writing = stack.getStackItem(0);
190                 // System.out.println(SourceLineAnnotation.fromVisitedInstruction(this)
191                 // + " writing " + writing);
192                 if (isTainted(writing)) {
193                     annotateAndReport(
194                             new BugInstance(this, "XSS_REQUEST_PARAMETER_TO_JSP_WRITER", taintPriority(writing))
195                             .addClassAndMethod(this),
196                             writing);
197                 } else if (isTainted(oldTop)) {
198                     annotateAndReport(
199                             new BugInstance(this, "XSS_REQUEST_PARAMETER_TO_JSP_WRITER", Priorities.NORMAL_PRIORITY)
200                             .addClassAndMethod(this),
201                             oldTop);
202                 }
203             } else if (calledClassName.startsWith("java/io/") && calledClassName.endsWith("Writer")
204                     && (calledMethodName.startsWith("print") || calledMethodName.startsWith("write"))
205                     && ("(Ljava/lang/Object;)V".equals(calledMethodSig) || "(Ljava/lang/String;)V".equals(calledMethodSig))) {
206                 OpcodeStack.Item writing = stack.getStackItem(0);
207                 OpcodeStack.Item writingTo = stack.getStackItem(1);
208                 if (isTainted(writing) && writingTo.isServletWriter()) {
209                     annotateAndReport(
210                             new BugInstance(this, "XSS_REQUEST_PARAMETER_TO_SERVLET_WRITER", taintPriority(writing))
211                             .addClassAndMethod(this),
212                             writing);
213                 } else if (isTainted(oldTop) && writingTo.isServletWriter()) {
214                     annotateAndReport(
215                             new BugInstance(this, "XSS_REQUEST_PARAMETER_TO_SERVLET_WRITER", Priorities.NORMAL_PRIORITY)
216                             .addClassAndMethod(this),
217                             writing);
218                 }
219 
220             }
221         }
222     }

224     private boolean isTainted(OpcodeStack.Item writing) {
225         if (writing == null) {
226             return false;
227         }
228         return writing.isServletParameterTainted();
229     }

247     private int taintPriority(OpcodeStack.Item writing) {
248         if (writing == null) {
249             return Priorities.NORMAL_PRIORITY;
250         }
251         XMethod method = writing.getReturnValueOf();
252         if (method != null && "getParameter".equals(method.getName())
253                 && "javax.servlet.http.HttpServletRequest".equals(method.getClassName())) {
254             return Priorities.HIGH_PRIORITY;
255         }
256         return Priorities.NORMAL_PRIORITY;
257     }
