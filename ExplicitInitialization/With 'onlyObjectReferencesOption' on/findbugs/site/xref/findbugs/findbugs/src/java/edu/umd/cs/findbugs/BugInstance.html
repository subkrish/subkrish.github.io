94  /**
95   * An instance of a bug pattern. A BugInstance consists of several parts:
96   * <p/>
97   * <ul>
98   * <li>the type, which is a string indicating what kind of bug it is; used as a
99   * key for the FindBugsMessages resource bundle
100  * <li>the priority; how likely this instance is to actually be a bug
101  * <li>a list of <em>annotations</em>
102  * </ul>
103  * <p/>
104  * The annotations describe classes, methods, fields, source locations, and
105  * other relevant context information about the bug instance. Every BugInstance
106  * must have at least one ClassAnnotation, which describes the class in which
107  * the instance was found. This is the "primary class annotation".
108  * <p/>
109  * <p>
110  * BugInstance objects are built up by calling a string of <code>add</code>
111  * methods. (These methods all "return this", so they can be chained). Some of
112  * the add methods are specialized to get information automatically from a
113  * BetterVisitor or DismantleBytecode object.
114  *
115  * @author David Hovemeyer
116  * @see BugAnnotation
117  */
118 public class BugInstance implements Comparable<BugInstance>, XMLWriteable, Serializable, Cloneable {

193     public BugInstance(String type, int priority) {
194         this.type = type.intern();
195         this.priority = priority;
196         lastVersion = -1;
197         xmlProps = new AtomicReference<XmlProps>();
198         annotationList = new ArrayList<BugAnnotation>(4);
199         cachedHashCode = INVALID_HASH_CODE;
200 
201         BugPattern p = DetectorFactoryCollection.instance().lookupBugPattern(type);
202         if (p == null) {
203             if ( missingBugTypes.add(type)) {
204                 String msg = "Can't find definition of bug type " + type;
205                 AnalysisContext.logError(msg, new NoSuchBugPattern(type));
206             }
207         } else {
208             this.priority += p.getPriorityAdjustment();
209         }
210         if (adjustExperimental && isExperimental()) {
211             this.priority = Priorities.EXP_PRIORITY;
212         }
213         boundPriority();
214     }

230     @Override
231     public Object clone() {
232         BugInstance dup;
233         try {
234             dup = (BugInstance) super.clone();
235 
236             // Do deep copying of mutable objects
237             for (int i = 0; i < dup.annotationList.size(); ++i) {
238                 dup.annotationList.set(i, (BugAnnotation) dup.annotationList.get(i).clone());
239             }
240             dup.propertyListHead = dup.propertyListTail = null;
241             for (Iterator<BugProperty> i = propertyIterator(); i.hasNext();) {
242                 dup.addProperty((BugProperty) i.next().clone());
243             }
244 
245             return dup;
246         } catch (CloneNotSupportedException e) {
247             throw new AssertionError(e);
248         }
249     }

275     public void adjustForDetector(String detectorName) {
276         DetectorFactory factory = DetectorFactoryCollection.instance().getFactoryByClassName(detectorName);
277         detectorFactory = factory;
278         if (factory != null) {
279             this.priority += factory.getPriorityAdjustment();
280             boundPriority();
281             BugPattern bugPattern = getBugPattern();
282             if (SystemProperties.ASSERTIONS_ENABLED && !"EXPERIMENTAL".equals(bugPattern.getCategory())
283                     && !factory.getReportedBugPatterns().contains(bugPattern)) {
284                 AnalysisContext.logError(factory.getShortName() + " doesn't note that it reports "
285                         + bugPattern + " in category " + bugPattern.getCategory());
286             }
287         }
288     }

485     /**
486      * Find the first BugAnnotation in the list of annotations that is the same
487      * type or a subtype as the given Class parameter.
488      *
489      * @param cls
490      *            the Class parameter
491      * @return the first matching BugAnnotation of the given type, or null if
492      *         there is no such BugAnnotation
493      */
494     @CheckForNull
495     private <T extends BugAnnotation> T findPrimaryAnnotationOfType(Class<T> cls) {
496         T firstMatch = null;
497         for (Iterator<BugAnnotation> i = annotationIterator(); i.hasNext();) {
498             BugAnnotation annotation = i.next();
499             if (cls.isAssignableFrom(annotation.getClass())) {
500                 if (annotation.getDescription().endsWith("DEFAULT")) {
501                     return cls.cast(annotation);
502                 } else if (firstMatch == null) {
503                     firstMatch = cls.cast(annotation);
504                 }
505             }
506         }
507         return firstMatch;
508     }

519     /**
520      * Get the primary source line annotation. There is guaranteed to be one
521      * (unless some Detector constructed an invalid BugInstance).
522      *
523      * @return the source line annotation
524      */
525     @Nonnull
526     public SourceLineAnnotation getPrimarySourceLineAnnotation() {
527         // Highest priority: return the first top level source line annotation
528         for (BugAnnotation annotation : annotationList) {
529             if (annotation instanceof SourceLineAnnotation
530                     && SourceLineAnnotation.DEFAULT_ROLE.equals(annotation.getDescription())
531                     && !((SourceLineAnnotation) annotation).isUnknown()) {
532                 return (SourceLineAnnotation) annotation;
533             }
534         }
535 
536         for (BugAnnotation annotation : annotationList) {
537             if (annotation instanceof SourceLineAnnotation && !((SourceLineAnnotation) annotation).isUnknown()) {
538                 return (SourceLineAnnotation) annotation;
539             }
540         }
541         // Next: Try primary method, primary field, primary class
542         SourceLineAnnotation srcLine;
543         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryMethod())) != null) {
544             return srcLine;
545         }
546         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryField())) != null) {
547             return srcLine;
548         }
549         if ((srcLine = inspectPackageMemberSourceLines(getPrimaryClass())) != null) {
550             return srcLine;
551         }
552 
553         // Last resort: throw exception
554         throw new IllegalStateException("BugInstance for " + getType()
555                 + " must contain at least one class, method, or field annotation");
556     }

867     private class BugPropertyIterator implements Iterator<BugProperty> {
868         private BugProperty prev, cur;
869 
870         private boolean removed;
871 
872         @Override
873         public boolean hasNext() {
874             return findNext() != null;
875         }
876 
877         @Override
878         public BugProperty next() {
879             BugProperty next = findNext();
880             if (next == null) {
881                 throw new NoSuchElementException();
882             }
883             prev = cur;
884             cur = next;
885             removed = false;
886             return cur;
887         }
888 
889         @Override
890         public void remove() {
891             if (cur == null || removed) {
892                 throw new IllegalStateException();
893             }
894             if (prev == null) {
895                 propertyListHead = cur.getNext();
896             } else {
897                 prev.setNext(cur.getNext());
898             }
899             if (cur == propertyListTail) {
900                 propertyListTail = prev;
901             }
902             removed = true;
903         }
904 
905         private BugProperty findNext() {
906             return cur == null ? propertyListHead : cur.getNext();
907         }
908     }

971      *

973      *            name of the property to look for
974      * @return the BugProperty with the given name, or null if the property has
975      *         not been set
976      */
977     public BugProperty lookupProperty(String name) {
978         BugProperty prop = propertyListHead;
979 
980         while (prop != null) {
981             if (prop.getName().equals(name)) {
982                 break;
983             }
984             prop = prop.getNext();
985         }
986 
987         return prop;

990     /**
991      * Delete property with given name.
992      *
993      * @param name
994      *            name of the property to delete
995      * @return true if a property with that name was deleted, or false if there
996      *         is no such property
997      */
998     public boolean deleteProperty(String name) {
999         BugProperty prev = null;
1000         BugProperty prop = propertyListHead;
1001 
1002         while (prop != null) {
1003             if (prop.getName().equals(name)) {
1004                 break;
1005             }
1006             prev = prop;
1007             prop = prop.getNext();
1008         }
1009 
1010         if (prop != null) {
1011             if (prev != null) {
1012                 // Deleted node in interior or at tail of list
1013                 prev.setNext(prop.getNext());
1014             } else {
1015                 // Deleted node at head of list
1016                 propertyListHead = prop.getNext();
1017             }
1018 
1019             if (prop.getNext() == null) {
1020                 // Deleted node at end of list
1021                 propertyListTail = prev;
1022             }
1023 
1024             return true;
1025         } else {
1026             // No such property
1027             return false;
1028         }
1029     }

1031     private void addProperty(BugProperty prop) {
1032         if (propertyListTail != null) {
1033             propertyListTail.setNext(prop);
1034             propertyListTail = prop;
1035         } else {
1036             propertyListHead = propertyListTail = prop;
1037         }
1038         prop.setNext(null);
1039     }
