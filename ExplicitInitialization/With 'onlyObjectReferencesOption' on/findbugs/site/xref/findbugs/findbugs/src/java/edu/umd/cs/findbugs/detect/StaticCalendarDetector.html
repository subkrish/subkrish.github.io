53  /**
54   * Detector for static fields of type {@link java.util.Calendar} or
55   * {@link java.text.DateFormat} and their subclasses. Because
56   * {@link java.util.Calendar} is unsafe for multithreaded use, static fields
57   * look suspicous. To work correctly, all access would need to be synchronized
58   * by the client which cannot be guaranteed.
59   *
60   * @author Daniel Schneller
61   */
62  public class StaticCalendarDetector extends OpcodeStackDetector {
63  
64      /** External Debug flag set? */
65      private static final boolean DEBUG = Boolean.getBoolean("debug.staticcal");
66  
67      /**
68       * External flag to determine whether to skip the test for synchronized
69       * blocks (default: if a call on a static Calendar or DateFormat is detected
70       * inside a synchronizationb block, it will not be reported). Setting this
71       * to <code>true</code> will report method calls on static fields if they
72       * are in a synchronized block. As the check currently does not take into
73       * account the lock's mutex it may be useful to switch allow
74       */
75      private static final String PROP_SKIP_SYNCHRONIZED_CHECK = "staticcal.skipsynccheck";
76  
77      /** The reporter to report to */
78      final private BugReporter reporter;
79  
80      final private BugAccumulator bugAccumulator;
81  
82      /** Name of the class being inspected */
83      private String currentClass;
84  
85      /**
86       * {@link org.apache.bcel.generic.ObjectType} for {@link java.util.Calendar}
87       */
88      private final ClassDescriptor calendarType = DescriptorFactory.createClassDescriptor(java.util.Calendar.class);
89  
90      /**
91       * {@link org.apache.bcel.generic.ObjectType} for
92       * {@link java.text.DateFormat}
93       */
94      private final ClassDescriptor dateFormatType = DescriptorFactory.createClassDescriptor(java.text.DateFormat.class);
95  
96      /** Stores the current method */
97      private Method currentMethod = null;
98  
99      /** Stores current Control Flow Graph */
100     private CFG currentCFG;
101 
102     /** Stores current LDF */
103     private LockDataflow currentLockDataFlow;

105     private final Map<XField, BugInstance> pendingBugs = new HashMap<XField, BugInstance>();

107     /**
108      * Creates a new instance of this Detector.
109      *
110      * @param aReporter
111      *            {@link BugReporter} instance to report found problems to.
112      */
113     public StaticCalendarDetector(BugReporter aReporter) {
114         reporter = aReporter;
115         bugAccumulator = new BugAccumulator(reporter);
116     }

118     Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();

120     private boolean sawDateClass;

122     /**
123      * Remembers the class name and resets temporary fields.
124      */
125     @Override
126     public void visit(JavaClass someObj) {
127         currentClass = someObj.getClassName();
128         currentMethod = null;
129         currentCFG = null;
130         currentLockDataFlow = null;
131         sawDateClass = false;
132 
133     }

135     @Override
136     public void visit(ConstantPool pool) {
137         for (Constant constant : pool.getConstantPool()) {
138             if (constant instanceof ConstantClass) {
139                 ConstantClass cc = (ConstantClass) constant;
140                 @SlashedClassName String className = cc.getBytes(pool);
141                 if ("java/util/Calendar".equals(className) || "java/text/DateFormat".equals(className)) {
142                     sawDateClass = true;
143                     break;
144                 }
145                 try {
146                     ClassDescriptor cDesc = DescriptorFactory.createClassDescriptor(className);
147 
148                     if (subtypes2.isSubtype(cDesc, calendarType) || subtypes2.isSubtype(cDesc, dateFormatType)) {
149                         sawDateClass = true;
150                         break;
151                     }
152                 } catch (ClassNotFoundException e) {
153                     reporter.reportMissingClass(e);
154                 }
155 
156 
157             }
158         }
159     }

161     /**
162      * Checks if the visited field is of type {@link java.util.Calendar} or
163      * {@link java.text.DateFormat} or a subclass of either one. If so and the
164      * field is static and non-private it is suspicious and will be reported.
165      */
166     @Override
167     public void visit(Field aField) {
168         if (aField.isPrivate()) {
169             /*
170              * private fields are harmless, as long as they are used correctly
171              * inside their own class. This should be something the rest of this
172              * detector can find out, so do not report them, they might be false
173              * positives.
174              */
175             return;
176         }
177         String superclassName = getSuperclassName();
178         if (!aField.isStatic() && !"java/lang/Enum".equals(superclassName)) {
179             return;
180         }
181         if (!aField.isPublic() && !aField.isProtected()) {
182             return;
183         }
184         ClassDescriptor classOfField = DescriptorFactory.createClassDescriptorFromFieldSignature(aField.getSignature());
185         String tBugType = null;
186         int priority = aField.isPublic() && aField.isFinal() && aField.getName().equals(aField.getName().toUpperCase())
187                 && getThisClass().isPublic() ? HIGH_PRIORITY : NORMAL_PRIORITY;
188         if (classOfField != null) {
189             try {
190                 if (subtypes2.isSubtype(classOfField, calendarType)) {
191                     tBugType = "STCAL_STATIC_CALENDAR_INSTANCE";
192                     priority++;
193                 } else if (subtypes2.isSubtype(classOfField, dateFormatType)) {
194                     tBugType = "STCAL_STATIC_SIMPLE_DATE_FORMAT_INSTANCE";
195                 }
196                 if (getClassContext().getXClass().usesConcurrency()) {
197                     priority--;
198                 }
199                 if (tBugType != null) {
200 
201                     pendingBugs.put(getXField(), new BugInstance(this, tBugType, priority).addClass(currentClass).addField(this));
202                 }
203             } catch (ClassNotFoundException e) {
204                 AnalysisContext.reportMissingClass(e);
205             }
206         }
207 
208     }

210     /*
211      * (non-Javadoc)
212      *
213      * @see
214      * edu.umd.cs.findbugs.visitclass.BetterVisitor#visitMethod(org.apache.bcel
215      * .classfile.Method)
216      */
217     @Override
218     public void visitMethod(Method obj) {
219         if (sawDateClass) {
220             try {
221                 super.visitMethod(obj);
222                 currentMethod = obj;
223                 currentLockDataFlow = getClassContext().getLockDataflow(currentMethod);
224                 currentCFG = getClassContext().getCFG(currentMethod);
225             } catch (CFGBuilderException e) {
226                 reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
227             } catch (DataflowAnalysisException e) {
228                 reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
229             }
230         }
231     }

233     @Override
234     public void visit(Code obj) {
235         if (sawDateClass) {
236             super.visit(obj);
237             bugAccumulator.reportAccumulatedBugs();
238         }

241     /**
242      * Checks for method invocations (
243      * {@link org.apache.bcel.generic.INVOKEVIRTUAL}) call on a static
244      * {@link java.util.Calendar} or {@link java.text.DateFormat} fields. The
245      * {@link OpcodeStack} is used to determine if an invocation is done on such
246      * a static field.
247      *
248      * @param seen
249      *            An opcode to be analyzed
250      * @see edu.umd.cs.findbugs.visitclass.DismantleBytecode#sawOpcode(int)
251      */
252     @Override
253     public void sawOpcode(int seen) {
254 
255         if (seen == GETSTATIC) {
256             XField f = getXFieldOperand();
257             if (pendingBugs.containsKey(f)) {
258                 if (!isLocked()) {
259                     reporter.reportBug(pendingBugs.remove(f));
260 
261                 }
262 
263             }
264         }
265         // we are only interested in method calls
266         if (seen != INVOKEVIRTUAL) {
267             return;
268         }
269 
270         try {
271             @SlashedClassName String className = getClassConstantOperand();
272 
273             if (className.startsWith("[")) {
274                 // Ignore array classes
275                 return;
276             }
277             ClassDescriptor cDesc = DescriptorFactory.createClassDescriptor(className);
278 
279             // if it is not compatible with Calendar or DateFormat, we are not
280             // interested anymore
281             boolean isCalendar = subtypes2.isSubtype(cDesc, calendarType);
282             boolean isDateFormat = subtypes2.isSubtype(cDesc, dateFormatType);
283             if (!isCalendar && !isDateFormat) {
284                 return;
285             }
286 
287             // determine the number of arguments the method expects
288             int numArguments = getNumberArguments(getSigConstantOperand());
289             // go back on the stack to find what the receiver of the method is
290             OpcodeStack.Item invokedOn = stack.getStackItem(numArguments);
291             XField field = invokedOn.getXField();
292             // find out, if the field is static. if not, we are not interested
293             // anymore
294             if (field == null || !field.isStatic()) {
295                 return;
296             }
297 
298             if ("<clinit>".equals(getMethodName()) && field.getClassName().equals(getDottedClassName())) {
299                 return;
300             }
301             String invokedName = getNameConstantOperand();
302             if (invokedName.startsWith("get")) {
303                 return;
304             }
305             if ("equals".equals(invokedName) && numArguments == 1) {
306                 OpcodeStack.Item passedAsArgument = stack.getStackItem(0);
307                 field = passedAsArgument.getXField();
308                 if (field == null || !field.isStatic()) {
309                     return;
310                 }
311             }
312 
313             if (!SystemProperties.getBoolean(PROP_SKIP_SYNCHRONIZED_CHECK)) {
314                 // check synchronization
315                 if (isLocked()) {
316                     return;
317                 }
318             }
319 
320             // if we get here, we want to generate a report, depending on the
321             // type
322             String tBugType;
323             if (isCalendar) {
324                 tBugType = "STCAL_INVOKE_ON_STATIC_CALENDAR_INSTANCE";
325             } else if (isDateFormat) {
326                 tBugType = "STCAL_INVOKE_ON_STATIC_DATE_FORMAT_INSTANCE";
327             } else {
328                 throw new IllegalStateException("Not possible");
329             }
330             int priority;
331             if (amVisitingMainMethod()) {
332                 priority = LOW_PRIORITY;
333             } else {
334                 if (getClassContext().getXClass().usesConcurrency()) {
335                     priority = NORMAL_PRIORITY;
336                 } else {
337                     priority = LOW_PRIORITY;
338                 }
339                 if (invokedName.startsWith("set") || "format".equals(invokedName) || "add".equals(invokedName)
340                         || "clear".equals(invokedName) || "parse".equals(invokedName) || "applyPattern".equals(invokedName)) {
341                     priority--;
342                 }
343             }
344             bugAccumulator.accumulateBug(new BugInstance(this, tBugType, priority).addClassAndMethod(this).addCalledMethod(this)
345                     .addOptionalField(field), this);
346 
347         } catch (ClassNotFoundException e) {
348             AnalysisContext.reportMissingClass(e);
349         }
350     }
351 
352     private boolean isLocked() {
353         try {
354             if (currentMethod != null && currentLockDataFlow != null && currentCFG != null) {
355                 Collection<Location> tLocations = currentCFG.getLocationsContainingInstructionWithOffset(getPC());
356                 for (Location tLoc : tLocations) {
357                     LockSet lockSet = currentLockDataFlow.getFactAtLocation(tLoc);
358                     if (lockSet.getNumLockedObjects() > 0) {
359                         // within a synchronized block
360                         return true;
361                     }
362                 }
363             }
364         } catch (DataflowAnalysisException e) {
365             reporter.logError("Synchronization check in Static Calendar Detector caught an error.", e);
366         }
367         return false;
368     }
369 
370     @Override
371     public void report() {
372         // Report any fields that were not accessed, but are potentially
373         // accessible
374         for (BugInstance bug : pendingBugs.values()) {
375             reporter.reportBug(bug);
376         }
377     }
