15  package com.google.common.util.concurrent;
16  
17  import static com.google.common.collect.Sets.newConcurrentHashSet;
18  import static java.util.concurrent.atomic.AtomicIntegerFieldUpdater.newUpdater;
19  import static java.util.concurrent.atomic.AtomicReferenceFieldUpdater.newUpdater;
20  
21  import com.google.common.annotations.GwtCompatible;
22  import java.util.Set;
23  import java.util.concurrent.atomic.AtomicIntegerFieldUpdater;
24  import java.util.concurrent.atomic.AtomicReferenceFieldUpdater;
25  import java.util.logging.Level;
26  import java.util.logging.Logger;
27  
28  /**
29   * A helper which does some thread-safe operations for aggregate futures, which must be implemented
30   * differently in GWT. Namely:
31   * <ul>
32   * <li>Lazily initializes a set of seen exceptions
33   * <li>Decrements a counter atomically
34   * </ul>
35   */
36  @GwtCompatible(emulated = true)
37  abstract class AggregateFutureState {
38    // Lazily initialized the first time we see an exception; not released until all the input futures
39    // & this future completes. Released when the future releases the reference to the running state
40    private volatile Set<Throwable> seenExceptions = null;
41  
42    private volatile int remaining;
43  
44    private static final AtomicHelper ATOMIC_HELPER;
45  
46    private static final Logger log = Logger.getLogger(AggregateFutureState.class.getName());
47  
48    static {
49      AtomicHelper helper;
50      try {
51        helper =
52            new SafeAtomicHelper(
53                newUpdater(AggregateFutureState.class, (Class) Set.class, "seenExceptions"),
54                newUpdater(AggregateFutureState.class, "remaining"));
55      } catch (Throwable reflectionFailure) {
56        // Some Android 5.0.x Samsung devices have bugs in JDK reflection APIs that cause
57        // getDeclaredField to throw a NoSuchFieldException when the field is definitely there.
58        // For these users fallback to a suboptimal implementation, based on synchronized. This will
59        // be a definite performance hit to those users.
60        log.log(Level.SEVERE, "SafeAtomicHelper is broken!", reflectionFailure);
61        helper = new SynchronizedAtomicHelper();
62      }
63      ATOMIC_HELPER = helper;
64    }
65  
66    AggregateFutureState(int remainingFutures) {
67      this.remaining = remainingFutures;
68    }
69  
70    final Set<Throwable> getOrInitSeenExceptions() {
71      /*
72       * The initialization of seenExceptions has to be more complicated than we'd like. The simple
73       * approach would be for each caller CAS it from null to a Set populated with its exception. But
74       * there's another race: If the first thread fails with an exception and a second thread
75       * immediately fails with the same exception:
76       *
77       * Thread1: calls setException(), which returns true, context switch before it can CAS
78       * seenExceptions to its exception
79       *
80       * Thread2: calls setException(), which returns false, CASes seenExceptions to its exception,
81       * and wrongly believes that its exception is new (leading it to logging it when it shouldn't)
82       *
83       * Our solution is for threads to CAS seenExceptions from null to a Set population with _the
84       * initial exception_, no matter which thread does the work. This ensures that seenExceptions
85       * always contains not just the current thread's exception but also the initial thread's.
86       */
87      Set<Throwable> seenExceptionsLocal = seenExceptions;
88      if (seenExceptionsLocal == null) {
89        seenExceptionsLocal = newConcurrentHashSet();
90        /*
91         * Other handleException() callers may see this as soon as we publish it. We need to populate
92         * it with the initial failure before we do, or else they may think that the initial failure
93         * has never been seen before.
94         */
95        addInitialException(seenExceptionsLocal);
96  
97        ATOMIC_HELPER.compareAndSetSeenExceptions(this, null, seenExceptionsLocal);
98        /*
99         * If another handleException() caller created the set, we need to use that copy in case yet
100        * other callers have added to it.
101        *
102        * This read is guaranteed to get us the right value because we only set this once (here).
103        */
104       seenExceptionsLocal = seenExceptions;
105     }
106     return seenExceptionsLocal;
107   }
108 
109   /** Populates {@code seen} with the exception that was passed to {@code setException}. */
110   abstract void addInitialException(Set<Throwable> seen);
111 
112   final int decrementRemainingAndGet() {
113     return ATOMIC_HELPER.decrementAndGetRemainingCount(this);
114   }
115 
116   private abstract static class AtomicHelper {
117     /** Atomic compare-and-set of the {@link AggregateFutureState#seenExceptions} field. */
118     abstract void compareAndSetSeenExceptions(
119         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update);
120 
121     /** Atomic decrement-and-get of the {@link AggregateFutureState#remaining} field. */
122     abstract int decrementAndGetRemainingCount(AggregateFutureState state);
123   }
124 
125   private static final class SafeAtomicHelper extends AtomicHelper {
126     final AtomicReferenceFieldUpdater<AggregateFutureState, Set<Throwable>> seenExceptionsUpdater;
127 
128     final AtomicIntegerFieldUpdater<AggregateFutureState> remainingCountUpdater;
129 
130     SafeAtomicHelper(
131         AtomicReferenceFieldUpdater seenExceptionsUpdater,
132         AtomicIntegerFieldUpdater remainingCountUpdater) {
133       this.seenExceptionsUpdater = seenExceptionsUpdater;
134       this.remainingCountUpdater = remainingCountUpdater;
135     }
136 
137     @Override
138     void compareAndSetSeenExceptions(
139         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update) {
140       seenExceptionsUpdater.compareAndSet(state, expect, update);
141     }
142 
143     @Override
144     int decrementAndGetRemainingCount(AggregateFutureState state) {
145       return remainingCountUpdater.decrementAndGet(state);
146     }
147   }
148 
149   private static final class SynchronizedAtomicHelper extends AtomicHelper {
150     @Override
151     void compareAndSetSeenExceptions(
152         AggregateFutureState state, Set<Throwable> expect, Set<Throwable> update) {
153       synchronized (state) {
154         if (state.seenExceptions == expect) {
155           state.seenExceptions = update;
156         }
157       }
158     }
159 
160     @Override
161     int decrementAndGetRemainingCount(AggregateFutureState state) {
162       synchronized (state) {
163         state.remaining--;
164         return state.remaining;
165       }
166     }
167   }
