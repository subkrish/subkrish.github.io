64  	/**
65  	 * make sure user didn't mangle the id
66  	 */
67  	public void checkId(Object object, EntityPersister persister, Serializable id, SessionImplementor session)
68  			throws HibernateException {
69  
70  		if ( id != null && id instanceof DelayedPostInsertIdentifier ) {
71  			// this is a situation where the entity id is assigned by a post-insert generator
72  			// and was saved outside the transaction forcing it to be delayed
73  			return;
74  		}
75  
76  		if ( persister.canExtractIdOutOfEntity() ) {
77  
78  			Serializable oid = persister.getIdentifier( object, session );
79  			if (id==null) {
80  				throw new AssertionFailure("null id in " + persister.getEntityName() + " entry (don't flush the Session after an exception occurs)");
81  			}
82  			if ( !persister.getIdentifierType().isEqual( id, oid, session.getFactory() ) ) {
83  				throw new HibernateException(
84  						"identifier of an instance of " +
85  						persister.getEntityName() +
86  						" was altered from " + id +
87  						" to " + oid
88  					);
89  			}
90  		}
91  
92  	}

94  	private void checkNaturalId(
95  			EntityPersister persister,
96  	        EntityEntry entry,
97  	        Object[] current,
98  	        Object[] loaded,
99  	        SessionImplementor session) {
100 		if ( persister.hasNaturalIdentifier() && entry.getStatus() != Status.READ_ONLY ) {
101 			if ( ! persister.getEntityMetamodel().hasImmutableNaturalId() ) {
102 				// SHORT-CUT: if the natural id is mutable (!immutable), no need to do the below checks
103 				// EARLY EXIT!!!
104 				return;
105 			}
106 
107 			final int[] naturalIdentifierPropertiesIndexes = persister.getNaturalIdentifierProperties();
108 			final Type[] propertyTypes = persister.getPropertyTypes();
109 			final boolean[] propertyUpdateability = persister.getPropertyUpdateability();
110 
111 			final Object[] snapshot = loaded == null
112 					? session.getPersistenceContext().getNaturalIdSnapshot( entry.getId(), persister )
113 					: session.getPersistenceContext().getNaturalIdHelper().extractNaturalIdValues( loaded, persister );
114 
115 			for ( int i=0; i<naturalIdentifierPropertiesIndexes.length; i++ ) {
116 				final int naturalIdentifierPropertyIndex = naturalIdentifierPropertiesIndexes[i];
117 				if ( propertyUpdateability[ naturalIdentifierPropertyIndex ] ) {
118 					// if the given natural id property is updatable (mutable), there is nothing to check
119 					continue;
120 				}
121 
122 				final Type propertyType = propertyTypes[naturalIdentifierPropertyIndex];
123 				if ( ! propertyType.isEqual( current[naturalIdentifierPropertyIndex], snapshot[i] ) ) {
124 					throw new HibernateException(
125 							String.format(
126 									"An immutable natural identifier of entity %s was altered from %s to %s",
127 									persister.getEntityName(),
128 									propertyTypes[naturalIdentifierPropertyIndex].toLoggableString(
129 											snapshot[i],
130 											session.getFactory()
131 									),
132 									propertyTypes[naturalIdentifierPropertyIndex].toLoggableString(
133 											current[naturalIdentifierPropertyIndex],
134 											session.getFactory()
135 									)
136 							)
137 					);
138 			   }
139 			}
140 		}
141 	}

143 	/**
144 	 * Flushes a single entity's state to the database, by scheduling
145 	 * an update action, if necessary
146 	 */
147 	public void onFlushEntity(FlushEntityEvent event) throws HibernateException {
148 		final Object entity = event.getEntity();
149 		final EntityEntry entry = event.getEntityEntry();
150 		final EventSource session = event.getSession();
151 		final EntityPersister persister = entry.getPersister();
152 		final Status status = entry.getStatus();
153 		final Type[] types = persister.getPropertyTypes();
154 
155 		final boolean mightBeDirty = entry.requiresDirtyCheck(entity);
156 
157 		final Object[] values = getValues( entity, entry, mightBeDirty, session );
158 
159 		event.setPropertyValues(values);
160 
161 		//TODO: avoid this for non-new instances where mightBeDirty==false
162 		boolean substitute = wrapCollections( session, persister, types, values);
163 
164 		if ( isUpdateNecessary( event, mightBeDirty ) ) {
165 			substitute = scheduleUpdate( event ) || substitute;
166 		}
167 
168 		if ( status != Status.DELETED ) {
169 			// now update the object .. has to be outside the main if block above (because of collections)
170 			if (substitute) persister.setPropertyValues( entity, values );
171 
172 			// Search for collections by reachability, updating their role.
173 			// We don't want to touch collections reachable from a deleted object
174 			if ( persister.hasCollections() ) {
175 				new FlushVisitor(session, entity).processEntityPropertyValues(values, types);
176 			}
177 		}
178 
179 	}

181 	private Object[] getValues(Object entity, EntityEntry entry, boolean mightBeDirty, SessionImplementor session) {
182 		final Object[] loadedState = entry.getLoadedState();
183 		final Status status = entry.getStatus();
184 		final EntityPersister persister = entry.getPersister();
185 
186 		final Object[] values;
187 		if ( status == Status.DELETED ) {
188 			//grab its state saved at deletion
189 			values = entry.getDeletedState();
190 		}
191 		else if ( !mightBeDirty && loadedState!=null ) {
192 			values = loadedState;
193 		}
194 		else {
195 			checkId( entity, persister, entry.getId(), session );
196 
197 			// grab its current state
198 			values = persister.getPropertyValues( entity );
199 
200 			checkNaturalId( persister, entry, values, loadedState, session );
201 		}
202 		return values;
203 	}

205 	private boolean wrapCollections(
206 			EventSource session,
207 			EntityPersister persister,
208 			Type[] types,
209 			Object[] values
210 	) {
211 		if ( persister.hasCollections() ) {
212 
213 			// wrap up any new collections directly referenced by the object
214 			// or its components
215 
216 			// NOTE: we need to do the wrap here even if its not "dirty",
217 			// because collections need wrapping but changes to _them_
218 			// don't dirty the container. Also, for versioned data, we
219 			// need to wrap before calling searchForDirtyCollections
220 
221 			WrapVisitor visitor = new WrapVisitor(session);
222 			// substitutes into values by side-effect
223 			visitor.processEntityPropertyValues(values, types);
224 			return visitor.isSubstitutionRequired();
225 		}
226 		else {
227 			return false;
228 		}
229 	}

231 	private boolean isUpdateNecessary(final FlushEntityEvent event, final boolean mightBeDirty) {
232 		final Status status = event.getEntityEntry().getStatus();
233 		if ( mightBeDirty || status==Status.DELETED ) {
234 			// compare to cached state (ignoring collections unless versioned)
235 			dirtyCheck(event);
236 			if ( isUpdateNecessary(event) ) {
237 				return true;
238 			}
239 			else {
240 				if ( event.getEntityEntry().getPersister().getInstrumentationMetadata().isInstrumented() ) {
241 					event.getEntityEntry()
242 							.getPersister()
243 							.getInstrumentationMetadata()
244 							.extractInterceptor( event.getEntity() )
245 							.clearDirty();
246 				}
247 				event.getSession()
248 						.getFactory()
249 						.getCustomEntityDirtinessStrategy()
250 						.resetDirty( event.getEntity(), event.getEntityEntry().getPersister(), event.getSession() );
251 				return false;
252 			}
253 		}
254 		else {
255 			return hasDirtyCollections( event, event.getEntityEntry().getPersister(), status );
256 		}
257 	}

259 	private boolean scheduleUpdate(final FlushEntityEvent event) {
260 		final EntityEntry entry = event.getEntityEntry();
261 		final EventSource session = event.getSession();
262 		final Object entity = event.getEntity();
263 		final Status status = entry.getStatus();
264 		final EntityPersister persister = entry.getPersister();
265 		final Object[] values = event.getPropertyValues();
266 
267 		if ( LOG.isTraceEnabled() ) {
268 			if ( status == Status.DELETED ) {
269 				if ( !persister.isMutable() ) {
270 					LOG.tracev( "Updating immutable, deleted entity: {0}",
271 							MessageHelper.infoString( persister, entry.getId(), session.getFactory() ) );
272 				}
273 				else if ( !entry.isModifiableEntity() )
274 					LOG.tracev( "Updating non-modifiable, deleted entity: {0}",
275 							MessageHelper.infoString( persister, entry.getId(), session.getFactory() ) );
276 				else
277 					LOG.tracev( "Updating deleted entity: ",
278 							MessageHelper.infoString( persister, entry.getId(), session.getFactory() ) );
279 			}
280 			else
281 				LOG.tracev( "Updating entity: {0}",
282 						MessageHelper.infoString( persister, entry.getId(), session.getFactory() ) );
283 		}
284 
285 		final boolean intercepted = !entry.isBeingReplicated() && handleInterception( event );
286 
287 		// increment the version number (if necessary)
288 		final Object nextVersion = getNextVersion(event);
289 
290 		// if it was dirtied by a collection only
291 		int[] dirtyProperties = event.getDirtyProperties();
292 		if ( event.isDirtyCheckPossible() && dirtyProperties == null ) {
293 			if ( ! intercepted && !event.hasDirtyCollection() ) {
294 				throw new AssertionFailure( "dirty, but no dirty properties" );
295 			}
296 			dirtyProperties = ArrayHelper.EMPTY_INT_ARRAY;
297 		}
298 
299 		// check nullability but do not doAfterTransactionCompletion command execute
300 		// we'll use scheduled updates for that.
301 		new Nullability(session).checkNullability( values, persister, true );
302 
303 		// schedule the update
304 		// note that we intentionally do _not_ pass in currentPersistentState!
305 		session.getActionQueue().addAction(
306 				new EntityUpdateAction(
307 						entry.getId(),
308 						values,
309 						dirtyProperties,
310 						event.hasDirtyCollection(),
311 						( status == Status.DELETED && ! entry.isModifiableEntity() ?
312 								persister.getPropertyValues( entity ) :
313 								entry.getLoadedState() ),
314 						entry.getVersion(),
315 						nextVersion,
316 						entity,
317 						entry.getRowId(),
318 						persister,
319 						session
320 					)
321 			);
322 
323 		return intercepted;
324 	}

326 	protected boolean handleInterception(FlushEntityEvent event) {
327 		SessionImplementor session = event.getSession();
328 		EntityEntry entry = event.getEntityEntry();
329 		EntityPersister persister = entry.getPersister();
330 		Object entity = event.getEntity();
331 
332 		//give the Interceptor a chance to modify property values
333 		final Object[] values = event.getPropertyValues();
334 		final boolean intercepted = invokeInterceptor( session, entity, entry, values, persister );
335 
336 		//now we might need to recalculate the dirtyProperties array
337 		if ( intercepted && event.isDirtyCheckPossible() && !event.isDirtyCheckHandledByInterceptor() ) {
338 			int[] dirtyProperties;
339 			if ( event.hasDatabaseSnapshot() ) {
340 				dirtyProperties = persister.findModified( event.getDatabaseSnapshot(), values, entity, session );
341 			}
342 			else {
343 				dirtyProperties = persister.findDirty( values, entry.getLoadedState(), entity, session );
344 			}
345 			event.setDirtyProperties(dirtyProperties);
346 		}
347 
348 		return intercepted;
349 	}

351 	protected boolean invokeInterceptor(
352 			SessionImplementor session,
353 			Object entity,
354 			EntityEntry entry,
355 			final Object[] values,
356 			EntityPersister persister) {
357 		return session.getInterceptor().onFlushDirty(
358 				entity,
359 				entry.getId(),
360 				values,
361 				entry.getLoadedState(),
362 				persister.getPropertyNames(),
363 				persister.getPropertyTypes()
364 		);
365 	}

367 	/**
368 	 * Convience method to retreive an entities next version value
369 	 */
370 	private Object getNextVersion(FlushEntityEvent event) throws HibernateException {
371 
372 		EntityEntry entry = event.getEntityEntry();
373 		EntityPersister persister = entry.getPersister();
374 		if ( persister.isVersioned() ) {
375 
376 			Object[] values = event.getPropertyValues();
377 
378 			if ( entry.isBeingReplicated() ) {
379 				return Versioning.getVersion(values, persister);
380 			}
381 			else {
382 				int[] dirtyProperties = event.getDirtyProperties();
383 
384 				final boolean isVersionIncrementRequired = isVersionIncrementRequired(
385 						event,
386 						entry,
387 						persister,
388 						dirtyProperties
389 					);
390 
391 				final Object nextVersion = isVersionIncrementRequired ?
392 						Versioning.increment( entry.getVersion(), persister.getVersionType(), event.getSession() ) :
393 						entry.getVersion(); //use the current version
394 
395 				Versioning.setVersion(values, nextVersion, persister);
396 
397 				return nextVersion;
398 			}
399 		}
400 		else {
401 			return null;
402 		}
403 
404 	}

406 	private boolean isVersionIncrementRequired(
407 			FlushEntityEvent event,
408 			EntityEntry entry,
409 			EntityPersister persister,
410 			int[] dirtyProperties
411 	) {
412 		final boolean isVersionIncrementRequired = entry.getStatus()!=Status.DELETED && (
413 				dirtyProperties==null ||
414 				Versioning.isVersionIncrementRequired(
415 						dirtyProperties,
416 						event.hasDirtyCollection(),
417 						persister.getPropertyVersionability()
418 				)
419 			);
420 		return isVersionIncrementRequired;
421 	}

423 	/**
424 	 * Performs all necessary checking to determine if an entity needs an SQL update
425 	 * to synchronize its state to the database. Modifies the event by side-effect!
426 	 * Note: this method is quite slow, avoid calling if possible!
427 	 */
428 	protected final boolean isUpdateNecessary(FlushEntityEvent event) throws HibernateException {
429 
430 		EntityPersister persister = event.getEntityEntry().getPersister();
431 		Status status = event.getEntityEntry().getStatus();
432 
433 		if ( !event.isDirtyCheckPossible() ) {
434 			return true;
435 		}
436 		else {
437 
438 			int[] dirtyProperties = event.getDirtyProperties();
439 			if ( dirtyProperties!=null && dirtyProperties.length!=0 ) {
440 				return true; //TODO: suck into event class
441 			}
442 			else {
443 				return hasDirtyCollections( event, persister, status );
444 			}
445 
446 		}
447 	}

449 	private boolean hasDirtyCollections(FlushEntityEvent event, EntityPersister persister, Status status) {
450 		if ( isCollectionDirtyCheckNecessary(persister, status ) ) {
451 			DirtyCollectionSearchVisitor visitor = new DirtyCollectionSearchVisitor(
452 					event.getSession(),
453 					persister.getPropertyVersionability()
454 				);
455 			visitor.processEntityPropertyValues( event.getPropertyValues(), persister.getPropertyTypes() );
456 			boolean hasDirtyCollections = visitor.wasDirtyCollectionFound();
457 			event.setHasDirtyCollection(hasDirtyCollections);
458 			return hasDirtyCollections;
459 		}
460 		else {
461 			return false;
462 		}
463 	}

465 	private boolean isCollectionDirtyCheckNecessary(EntityPersister persister, Status status) {
466 		return ( status == Status.MANAGED || status == Status.READ_ONLY ) &&
467 				persister.isVersioned() &&
468 				persister.hasCollections();
469 	}

471 	/**
472 	 * Perform a dirty check, and attach the results to the event
473 	 */
474 	protected void dirtyCheck(final FlushEntityEvent event) throws HibernateException {
475 
476 		final Object entity = event.getEntity();
477 		final Object[] values = event.getPropertyValues();
478 		final SessionImplementor session = event.getSession();
479 		final EntityEntry entry = event.getEntityEntry();
480 		final EntityPersister persister = entry.getPersister();
481 		final Serializable id = entry.getId();
482 		final Object[] loadedState = entry.getLoadedState();
483 
484 		int[] dirtyProperties = session.getInterceptor().findDirty(
485 				entity,
486 				id,
487 				values,
488 				loadedState,
489 				persister.getPropertyNames(),
490 				persister.getPropertyTypes()
491 		);
492 
493 		if ( dirtyProperties == null ) {
494             if(entity instanceof SelfDirtinessTracker) {
495                 if(((SelfDirtinessTracker) entity).$$_hibernate_hasDirtyAttributes()) {
496                    dirtyProperties = persister.resolveAttributeIndexes(((SelfDirtinessTracker) entity).$$_hibernate_getDirtyAttributes());
497                 }
498             }
499             else {
500                 // see if the custom dirtiness strategy can tell us...
501                 class DirtyCheckContextImpl implements CustomEntityDirtinessStrategy.DirtyCheckContext {
502                     int[] found = null;
503                     @Override
504                     public void doDirtyChecking(CustomEntityDirtinessStrategy.AttributeChecker attributeChecker) {
505                         found = new DirtyCheckAttributeInfoImpl( event ).visitAttributes( attributeChecker );
506                         if ( found != null && found.length == 0 ) {
507                             found = null;
508                         }
509                     }
510                 }
511                 DirtyCheckContextImpl context = new DirtyCheckContextImpl();
512                 session.getFactory().getCustomEntityDirtinessStrategy().findDirty(
513                         entity,
514                         persister,
515                         (Session) session,
516                         context
517                 );
518                 dirtyProperties = context.found;
519             }
520 		}
521 
522 		event.setDatabaseSnapshot(null);
523 
524 		final boolean interceptorHandledDirtyCheck;
525 		boolean cannotDirtyCheck;
526 
527 		if ( dirtyProperties==null ) {
528 			// Interceptor returned null, so do the dirtycheck ourself, if possible
529 			try {
530 				session.getEventListenerManager().dirtyCalculationStart();
531 
532 				interceptorHandledDirtyCheck = false;
533 				// object loaded by update()
534 				cannotDirtyCheck = loadedState==null;
535 				if ( !cannotDirtyCheck ) {
536 					// dirty check against the usual snapshot of the entity
537 					dirtyProperties = persister.findDirty( values, loadedState, entity, session );
538 				}
539 				else if ( entry.getStatus() == Status.DELETED && ! event.getEntityEntry().isModifiableEntity() ) {
540 					// A non-modifiable (e.g., read-only or immutable) entity needs to be have
541 					// references to transient entities set to null before being deleted. No other
542 					// fields should be updated.
543 					if ( values != entry.getDeletedState() ) {
544 						throw new IllegalStateException(
545 								"Entity has status Status.DELETED but values != entry.getDeletedState"
546 						);
547 					}
548 					// Even if loadedState == null, we can dirty-check by comparing currentState and
549 					// entry.getDeletedState() because the only fields to be updated are those that
550 					// refer to transient entities that are being set to null.
551 					// - currentState contains the entity's current property values.
552 					// - entry.getDeletedState() contains the entity's current property values with
553 					//   references to transient entities set to null.
554 					// - dirtyProperties will only contain properties that refer to transient entities
555 					final Object[] currentState = persister.getPropertyValues( event.getEntity() );
556 					dirtyProperties = persister.findDirty( entry.getDeletedState(), currentState, entity, session );
557 					cannotDirtyCheck = false;
558 				}
559 				else {
560 					// dirty check against the database snapshot, if possible/necessary
561 					final Object[] databaseSnapshot = getDatabaseSnapshot(session, persister, id);
562 					if ( databaseSnapshot != null ) {
563 						dirtyProperties = persister.findModified(databaseSnapshot, values, entity, session);
564 						cannotDirtyCheck = false;
565 						event.setDatabaseSnapshot(databaseSnapshot);
566 					}
567 				}
568 			}
569 			finally {
570 				session.getEventListenerManager().dirtyCalculationEnd( dirtyProperties != null );
571 			}
572 		}
573 		else {
574 			// the Interceptor handled the dirty checking
575 			cannotDirtyCheck = false;
576 			interceptorHandledDirtyCheck = true;
577 		}
578 
579 		logDirtyProperties( id, dirtyProperties, persister );
580 
581 		event.setDirtyProperties(dirtyProperties);
582 		event.setDirtyCheckHandledByInterceptor(interceptorHandledDirtyCheck);
583 		event.setDirtyCheckPossible(!cannotDirtyCheck);
584 
585 	}

587 	private class DirtyCheckAttributeInfoImpl implements CustomEntityDirtinessStrategy.AttributeInformation {
588 		private final FlushEntityEvent event;
589 		private final EntityPersister persister;
590 		private final int numberOfAttributes;
591 		private int index = 0;
592 
593 		private DirtyCheckAttributeInfoImpl(FlushEntityEvent event) {
594 			this.event = event;
595 			this.persister = event.getEntityEntry().getPersister();
596 			this.numberOfAttributes = persister.getPropertyNames().length;
597 		}
598 
599 		@Override
600 		public EntityPersister getContainingPersister() {
601 			return persister;
602 		}
603 
604 		@Override
605 		public int getAttributeIndex() {
606 			return index;
607 		}
608 
609 		@Override
610 		public String getName() {
611 			return persister.getPropertyNames()[ index ];
612 		}
613 
614 		@Override
615 		public Type getType() {
616 			return persister.getPropertyTypes()[ index ];
617 		}
618 
619 		@Override
620 		public Object getCurrentValue() {
621 			return event.getPropertyValues()[ index ];
622 		}
623 
624 		Object[] databaseSnapshot;
625 
626 		@Override
627 		public Object getLoadedValue() {
628 			if ( databaseSnapshot == null ) {
629 				databaseSnapshot = getDatabaseSnapshot( event.getSession(), persister, event.getEntityEntry().getId() );
630 			}
631 			return databaseSnapshot[ index ];
632 		}
633 
634 		public int[] visitAttributes(CustomEntityDirtinessStrategy.AttributeChecker attributeChecker) {
635 			databaseSnapshot = null;
636 			index = 0;
637 
638 			final int[] indexes = new int[ numberOfAttributes ];
639 			int count = 0;
640 			for ( ; index < numberOfAttributes; index++ ) {
641 				if ( attributeChecker.isDirty( this ) ) {
642 					indexes[ count++ ] = index;
643 				}
644 			}
645 			return Arrays.copyOf( indexes, count );
646 		}
647 	}

649 	private void logDirtyProperties(Serializable id, int[] dirtyProperties, EntityPersister persister) {
650 		if ( dirtyProperties != null && dirtyProperties.length > 0 && LOG.isTraceEnabled() ) {
651 			final String[] allPropertyNames = persister.getPropertyNames();
652 			final String[] dirtyPropertyNames = new String[ dirtyProperties.length ];
653 			for ( int i = 0; i < dirtyProperties.length; i++ ) {
654 				dirtyPropertyNames[i] = allPropertyNames[ dirtyProperties[i]];
655 			}
656 			LOG.tracev( "Found dirty properties [{0}] : {1}",
657 					MessageHelper.infoString( persister.getEntityName(), id ),
658 					dirtyPropertyNames );
659 		}
660 	}

662 	private Object[] getDatabaseSnapshot(SessionImplementor session, EntityPersister persister, Serializable id) {
663 		if ( persister.isSelectBeforeUpdateRequired() ) {
664 			Object[] snapshot = session.getPersistenceContext()
665 					.getDatabaseSnapshot(id, persister);
666 			if (snapshot==null) {
667 				//do we even really need this? the update will fail anyway....
668 				if ( session.getFactory().getStatistics().isStatisticsEnabled() ) {
669 					session.getFactory().getStatisticsImplementor()
670 							.optimisticFailure( persister.getEntityName() );
671 				}
672 				throw new StaleObjectStateException( persister.getEntityName(), id );
673 			}
674 			return snapshot;
675 		}
676 		// TODO: optimize away this lookup for entities w/o unsaved-value="undefined"
677 		final EntityKey entityKey = session.generateEntityKey( id, persister );
678 		return session.getPersistenceContext().getCachedDatabaseSnapshot( entityKey );
679 	}
