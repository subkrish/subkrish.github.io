24  package org.hibernate.test.annotations.naturalid;
25  
26  import java.util.HashSet;
27  import java.util.Set;
28  import javax.persistence.Column;
29  import javax.persistence.Entity;
30  import javax.persistence.GeneratedValue;
31  import javax.persistence.GenerationType;
32  import javax.persistence.Id;
33  import javax.persistence.Version;
34  
35  import org.hibernate.annotations.Cache;
36  import org.hibernate.annotations.CacheConcurrencyStrategy;
37  import org.hibernate.annotations.NaturalId;
38  import org.hibernate.annotations.NaturalIdCache;
39  
40  /**
41   * @author Guenther Demetz
42   */
43  @Entity
44  @Cache(usage = CacheConcurrencyStrategy.READ_WRITE)
45  @NaturalIdCache
46  public class A {
47  
48  	@Id
49  	@GeneratedValue(strategy = GenerationType.TABLE)
50  	private long oid;
51  
52  	@Version
53  	private int version;
54  
55  	@Column
56  	@NaturalId(mutable = false)
57  	private String name;
58  
59  	@Cache(usage = CacheConcurrencyStrategy.READ_WRITE)
60  	@org.hibernate.annotations.OptimisticLock(excluded = true)
61  	@javax.persistence.OneToMany(mappedBy = "a")
62  	private Set<D> ds = new HashSet<D>();
63  
64  	@javax.persistence.OneToOne
65  	private D singleD = null;
66  
67  	public String getName() {
68  		return name;
69  	}
70  
71  	public void setName(String name) {
72  		this.name = name;
73  	}
74  
75  	public Set<D> getDs() {
76  		return ds;
77  	}
78  
79  	public void setDs(Set<D> ds) {
80  		this.ds = ds;
81  	}
82  
83  	public D getSingleD() {
84  		return singleD;
85  	}
86  
87  	public void setSingleD(D singleD) {
88  		this.singleD = singleD;
89  	}
90  
91  }
