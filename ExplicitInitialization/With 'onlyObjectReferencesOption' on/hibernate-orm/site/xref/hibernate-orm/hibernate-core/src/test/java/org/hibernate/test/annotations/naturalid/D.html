24  package org.hibernate.test.annotations.naturalid;
25  
26  import javax.persistence.Entity;
27  import javax.persistence.FetchType;
28  import javax.persistence.GeneratedValue;
29  import javax.persistence.GenerationType;
30  import javax.persistence.Id;
31  import javax.persistence.Version;
32  
33  import org.hibernate.annotations.Cache;
34  import org.hibernate.annotations.CacheConcurrencyStrategy;
35  
36  /**
37   * @author Guenther Demetz
38   */
39  @Entity
40  @Cache(usage = CacheConcurrencyStrategy.TRANSACTIONAL)
41  public class D {
42  	@Id
43  	@GeneratedValue(strategy = GenerationType.TABLE)
44  	public long oid;
45  
46  	@Version
47  	private int version;
48  
49  	@javax.persistence.ManyToOne(fetch = FetchType.LAZY)
50  	private A a = null;
51  
52  	@javax.persistence.OneToOne(mappedBy = "singleD")
53  	private A singleA = null;
54  
55  	public A getA() {
56  		return a;
57  	}
58  
59  	public void setA(A a) {
60  		this.a = a;
61  	}
62  
63  	public A getSingleA() {
64  		return singleA;
65  	}
66  
67  	public void setSingleA(A singleA) {
68  		this.singleA = singleA;
69  	}
70  }
