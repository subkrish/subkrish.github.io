19  @TestForIssue(jiraKey = "HHH-7003")
20  public class ColumnScalePrecisionTest extends BaseEnversJPAFunctionalTestCase {
21      private Table auditTable = null;
22      private Table originalTable = null;
23      private Long id = null;
24  
25      public void configure(Ejb3Configuration cfg) {
26          cfg.addAnnotatedClass(ScalePrecisionEntity.class);
27      }
28  
29      @Test
30      @Priority(10)
31      public void initData() {
32          EntityManager em = getEntityManager();
33  
34          // Revision 1
35          em.getTransaction().begin();
36          ScalePrecisionEntity entity = new ScalePrecisionEntity(13.0);
37          em.persist(entity);
38          em.getTransaction().commit();
39  
40          id = entity.getId();
41          auditTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.basic.ScalePrecisionEntity_AUD").getTable();
42          originalTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.basic.ScalePrecisionEntity").getTable();
43      }
44  
45      @Test
46      public void testColumnScalePrecision() {
47          Column testColumn = new Column("wholeNumber");
48          Column scalePrecisionAuditColumn = auditTable.getColumn(testColumn);
49          Column scalePrecisionColumn = originalTable.getColumn(testColumn);
50  
51          Assert.assertNotNull(scalePrecisionAuditColumn);
52          Assert.assertEquals(scalePrecisionColumn.getPrecision(), scalePrecisionAuditColumn.getPrecision());
53          Assert.assertEquals(scalePrecisionColumn.getScale(), scalePrecisionAuditColumn.getScale());
54      }
55  
56      @Test
57      public void testRevisionsCounts() {
58          assert Arrays.asList(1).equals(getAuditReader().getRevisions(ScalePrecisionEntity.class, id));
59      }
60  
61      @Test
62      public void testHistoryOfScalePrecisionEntity() {
63          ScalePrecisionEntity ver1 = new ScalePrecisionEntity(13.0, id);
64  
65          Assert.assertEquals(ver1, getAuditReader().find(ScalePrecisionEntity.class, id, 1));
66      }
67  }
