15  @TestForIssue(jiraKey = "HHH-7690")
16  public class RelationInsideEmbeddableTest extends BaseEnversJPAFunctionalTestCase {
17      private Integer orderId = null;
18      private ItemId itemId = null;
19  
20      @Override
21      protected Class<?>[] getAnnotatedClasses() {
22          return new Class<?>[]{PurchaseOrder.class, Item.class, ItemId.class, Producer.class};
23      }
24  
25      @Test
26      @Priority(10)
27      public void initData() {
28          EntityManager em = getEntityManager();
29  
30          // Revision 1
31          em.getTransaction().begin();
32          Producer producer = new Producer(1, "Sony");
33          ItemId sonyId = new ItemId("TV", 1, producer);
34          Item item = new Item(sonyId, 100.50);
35          PurchaseOrder order = new PurchaseOrder(item, null);
36          em.persist(producer);
37          em.persist(item);
38          em.persist(order);
39          em.getTransaction().commit();
40  
41          // Revision 2
42          em.getTransaction().begin();
43          order = em.find(PurchaseOrder.class, order.getId());
44          order.setComment("fragile");
45          order = em.merge(order);
46          em.getTransaction().commit();
47  
48          // Revision 3
49          em.getTransaction().begin();
50          item = em.find(Item.class, sonyId);
51          item.setPrice(110.00);
52          em.getTransaction().commit();
53  
54          orderId = order.getId();
55          itemId = sonyId;
56  
57          em.close();
58      }
59  
60      @Test
61      public void testRevisionsCounts() throws Exception {
62          Assert.assertEquals(Arrays.asList(1, 2), getAuditReader().getRevisions(PurchaseOrder.class, orderId));
63          Assert.assertEquals(Arrays.asList(1, 3), getAuditReader().getRevisions(Item.class, itemId));
64      }
65  
66      @Test
67      public void testHistoryOfPurchaseOrder() {
68          PurchaseOrder ver1 = new PurchaseOrder(orderId, new Item(new ItemId("TV", 1, new Producer(1, "Sony")), 100.50), null);
69          PurchaseOrder ver2 = new PurchaseOrder(orderId, new Item(new ItemId("TV", 1, new Producer(1, "Sony")), 100.50), "fragile");
70  
71          Assert.assertEquals(ver1, getAuditReader().find(PurchaseOrder.class, orderId, 1));
72          Assert.assertEquals(ver2, getAuditReader().find(PurchaseOrder.class, orderId, 2));
73      }
74  
75      @Test
76      public void testHistoryOfItem() {
77          Item ver1 = new Item(itemId, 100.50);
78          Item ver2 = new Item(itemId, 110.00);
79  
80          Assert.assertEquals(ver1, getAuditReader().find(Item.class, itemId, 1));
81          Assert.assertEquals(ver2, getAuditReader().find(Item.class, itemId, 3));
82      }
83  }
