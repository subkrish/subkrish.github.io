21  @TestForIssue( jiraKey = "HHH-6613" )
22  public class HasChangedComponentCollection extends AbstractModifiedFlagsEntityTest {
23  	private Integer ele1_id = null;
24  
25  	private final Component4 c4_1 = new Component4( "c41", "c41_value", "c41_description" );
26  	private final Component4 c4_2 = new Component4( "c42", "c42_value2", "c42_description" );
27  	private final Component3 c3_1 = new Component3( "c31", c4_1, c4_2 );
28  	private final Component3 c3_2 = new Component3( "c32", c4_1, c4_2 );
29  
30  	@Override
31  	protected Class<?>[] getAnnotatedClasses() {
32  		return new Class[] { EmbeddableListEntity1.class };
33  	}
34  
35  	@Test
36  	@Priority(10)
37  	public void initData() {
38  		EntityManager em = getEntityManager();
39  
40  		// Revision 1 (ele1: initially 1 element in both collections)
41  		em.getTransaction().begin();
42  		EmbeddableListEntity1 ele1 = new EmbeddableListEntity1();
43  		ele1.setOtherData( "data" );
44  		ele1.getComponentList().add( c3_1 );
45  		em.persist( ele1 );
46  		em.getTransaction().commit();
47  
48  		// Revision (still 1) (ele1: removing non-existing element)
49  		em.getTransaction().begin();
50  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
51  		ele1.getComponentList().remove( c3_2 );
52  		em.getTransaction().commit();
53  
54  		// Revision 2 (ele1: updating singular property and removing non-existing element)
55  		em.getTransaction().begin();
56  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
57  		ele1.setOtherData( "modified" );
58  		ele1.getComponentList().remove( c3_2 );
59  		ele1 = em.merge( ele1 );
60  		em.getTransaction().commit();
61  
62  		// Revision 3 (ele1: adding one element)
63  		em.getTransaction().begin();
64  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
65  		ele1.getComponentList().add( c3_2 );
66  		em.getTransaction().commit();
67  
68  		// Revision 4 (ele1: adding one existing element)
69  		em.getTransaction().begin();
70  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
71  		ele1.getComponentList().add( c3_1 );
72  		em.getTransaction().commit();
73  
74  		// Revision 5 (ele1: removing one existing element)
75  		em.getTransaction().begin();
76  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
77  		ele1.getComponentList().remove( c3_2 );
78  		em.getTransaction().commit();
79  
80  		// Revision 6 (ele1: changing singular property only)
81  		em.getTransaction().begin();
82  		ele1 = em.find( EmbeddableListEntity1.class, ele1.getId() );
83  		ele1.setOtherData( "another modification" );
84  		ele1 = em.merge( ele1 );
85  		em.getTransaction().commit();
86  
87  		ele1_id = ele1.getId();
88  
89  		em.close();
90  	}
91  
92  	@Test
93  	public void testHasChangedEle() {
94  		List list = queryForPropertyHasChanged( EmbeddableListEntity1.class, ele1_id, "componentList" );
95  		assertEquals( 4, list.size() );
96  		assertEquals( makeList( 1, 3, 4, 5 ), extractRevisionNumbers( list ) );
97  
98  		list = queryForPropertyHasChanged( EmbeddableListEntity1.class, ele1_id, "otherData" );
99  		assertEquals( 3, list.size() );
100 		assertEquals( makeList( 1, 2, 6 ), extractRevisionNumbers( list ) );
101 	}
102 }
