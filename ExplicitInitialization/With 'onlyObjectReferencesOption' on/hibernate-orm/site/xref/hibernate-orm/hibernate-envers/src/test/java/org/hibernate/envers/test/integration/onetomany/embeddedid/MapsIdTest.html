30      @Test
31      @Priority(10)
32      public void initData() {
33          EntityManager em = getEntityManager();
34  
35          // Revision 1
36          em.getTransaction().begin();
37          Person personA = new Person("Peter");
38          Person personB = new Person("Mary");
39          em.persist(personA);
40          em.persist(personB);
41          Constant cons = new Constant("USD", "US Dollar");
42          em.persist(cons);
43          PersonTuple tuple1 = new PersonTuple(true, personA, personB, cons);
44          em.persist(tuple1);
45          em.getTransaction().commit();
46  
47          tuple1Ver1 = new PersonTuple(tuple1.isHelloWorld(), tuple1.getPersonA(), tuple1.getPersonB(), tuple1.getConstant());
48  
49          // Revision 2
50          em.getTransaction().begin();
51          cons = em.find(Constant.class, cons.getId());
52          Person personC1 = new Person("Lukasz");
53          em.persist(personC1);
54          PersonTuple tuple2 = new PersonTuple(true, personA, personC1, cons);
55          em.persist(tuple2);
56          em.getTransaction().commit();
57  
58          tuple2Ver1 = new PersonTuple(tuple2.isHelloWorld(), tuple2.getPersonA(), tuple2.getPersonB(), tuple2.getConstant());
59          personCVer1 = new Person(personC1.getId(), personC1.getName());
60          personCVer1.getPersonBTuples().add(tuple2Ver1);
61  
62          // Revision 3
63          em.getTransaction().begin();
64          tuple2 = em.find(PersonTuple.class, tuple2.getPersonTupleId());
65          tuple2.setHelloWorld(false);
66          em.merge(tuple2);
67          em.getTransaction().commit();
68  
69          tuple2Ver2 = new PersonTuple(tuple2.isHelloWorld(), tuple2.getPersonA(), tuple2.getPersonB(), tuple2.getConstant());
70  
71          // Revision 4
72          em.getTransaction().begin();
73          Person personC2 = em.find(Person.class, personC1.getId());
74          personC2.setName("Robert");
75          em.merge(personC2);
76          em.getTransaction().commit();
77  
78          personCVer2 = new Person(personC2.getId(), personC2.getName());
79          personCVer2.getPersonBTuples().add(tuple2Ver1);
80  
81          em.close();
82      }
83  
84      @Test
85      public void testRevisionsCounts() {
86          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(PersonTuple.class, tuple1Ver1.getPersonTupleId()));
87          Assert.assertEquals(Arrays.asList(2, 3), getAuditReader().getRevisions(PersonTuple.class, tuple2Ver1.getPersonTupleId()));
88          Assert.assertEquals(Arrays.asList(2, 4), getAuditReader().getRevisions(Person.class, personCVer1.getId()));
89      }
90  
91      @Test
92      public void testHistoryOfTuple1() {
93          PersonTuple tuple = getAuditReader().find(PersonTuple.class, tuple1Ver1.getPersonTupleId(), 1);
94  
95          Assert.assertEquals(tuple1Ver1, tuple);
96          Assert.assertEquals(tuple1Ver1.isHelloWorld(), tuple.isHelloWorld());
97          Assert.assertEquals(tuple1Ver1.getPersonA().getId(), tuple.getPersonA().getId());
98          Assert.assertEquals(tuple1Ver1.getPersonB().getId(), tuple.getPersonB().getId());
99      }
