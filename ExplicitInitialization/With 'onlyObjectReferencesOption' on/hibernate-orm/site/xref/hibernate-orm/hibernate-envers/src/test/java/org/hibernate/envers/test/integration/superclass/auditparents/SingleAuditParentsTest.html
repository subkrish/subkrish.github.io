19  /**
20   * Tests mapping of child entity that declares one of its ancestors as audited with {@link Audited#auditParents()} property.
21   * All supperclasses are marked with {@link MappedSuperclass} annotation but not {@link Audited}.
22   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
23   */
24  public class SingleAuditParentsTest extends BaseEnversJPAFunctionalTestCase {
25      private long childSingleId = 1L;
26      private Integer siteSingleId = null;
27  
28      @Override
29      protected Class<?>[] getAnnotatedClasses() {
30          return new Class[]{MappedGrandparentEntity.class, MappedParentEntity.class, ChildSingleParentEntity.class, StrIntTestEntity.class};
31      }
32  
33      @Test
34      @Priority(10)
35      public void initData() {
36          EntityManager em = getEntityManager();
37          // Revision 1
38          em.getTransaction().begin();
39          StrIntTestEntity siteSingle = new StrIntTestEntity("data 1", 1);
40          em.persist(siteSingle);
41          em.persist(new ChildSingleParentEntity(childSingleId, "grandparent 1", "notAudited 1", "parent 1", "child 1", siteSingle));
42          em.getTransaction().commit();
43          siteSingleId = siteSingle.getId();
44          em.close();
45      }
46  
47      @Test
48      public void testCreatedAuditTable() {
49          Set<String> expectedColumns = TestTools.makeSet("child", "grandparent", "id");
50          Set<String> unexpectedColumns = TestTools.makeSet("parent", "relation_id", "notAudited");
51  
52          Table table = getCfg().getClassMapping("org.hibernate.envers.test.integration.superclass.auditparents.ChildSingleParentEntity_AUD").getTable();
53  
54          for (String columnName : expectedColumns) {
55              // Check whether expected column exists.
56              Assert.assertNotNull(table.getColumn(new Column(columnName)));
57          }
58          for (String columnName : unexpectedColumns) {
59              // Check whether unexpected column does not exist.
60              Assert.assertNull(table.getColumn(new Column(columnName)));
61          }
62      }
63  
64      @Test
65      public void testSingleAuditParent() {
66          // expectedSingleChild.parent, expectedSingleChild.relation and expectedSingleChild.notAudited shall be null, because they are not audited.
67          ChildSingleParentEntity expectedSingleChild = new ChildSingleParentEntity(childSingleId, "grandparent 1", null, null, "child 1", null);
68          ChildSingleParentEntity child = getAuditReader().find(ChildSingleParentEntity.class, childSingleId, 1);
69          Assert.assertEquals(expectedSingleChild, child);
70          Assert.assertNull(child.getRelation());
71      }
72  }
