7   package org.hibernate.search.backend.jms.impl;
8   
9   import java.util.Collections;
10  import java.util.List;
11  import java.util.Properties;
12  
13  import javax.jms.JMSException;
14  import javax.jms.Queue;
15  import javax.jms.QueueConnection;
16  import javax.jms.QueueConnectionFactory;
17  
18  import org.hibernate.search.backend.IndexingMonitor;
19  import org.hibernate.search.backend.LuceneWork;
20  import org.hibernate.search.backend.spi.BackendQueueProcessor;
21  import org.hibernate.search.cfg.Environment;
22  import org.hibernate.search.indexes.spi.IndexManager;
23  import org.hibernate.search.spi.SearchIntegrator;
24  import org.hibernate.search.spi.WorkerBuildContext;
25  import org.hibernate.search.util.logging.impl.Log;
26  import org.hibernate.search.util.logging.impl.LoggerFactory;
27  
28  /**
29   * Backend sending the workload via JMS.
30   * Optionally support transactions: the JMS messages can be sent as part of the current transaction.
31   *
32   * @author Emmanuel Bernard
33   * @author Hardy Ferentschik
34   * @author Sanne Grinovero (C) 2011 Red Hat Inc.
35   * @author Yoann Gendre
36   */
37  public abstract class JmsBackendQueueProcessor implements BackendQueueProcessor, BackendQueueProcessor.Transactional {
38  
39  	private String jmsQueueName;
40  	protected static final String JNDI_PREFIX = Environment.WORKER_PREFIX + "jndi.";
41  	private Queue jmsQueue;
42  	private QueueConnectionFactory factory;
43  	private String indexName;
44  	private SearchIntegrator integrator;
45  	private QueueConnection connection;
46  
47  	public static final String JMS_CONNECTION_FACTORY = Environment.WORKER_PREFIX + "jms.connection_factory";
48  	public static final String JMS_QUEUE = Environment.WORKER_PREFIX + "jms.queue";
49  	public static final String JMS_CONNECTION_LOGIN = Environment.WORKER_PREFIX + "jms.login";
50  	public static final String JMS_CONNECTION_PASSWORD = Environment.WORKER_PREFIX + "jms.password";
51  
52  	private static final Log log = LoggerFactory.make();
53  
54  	private Properties props = null;
55  	private boolean isTransactional;
56  
57  	@Override
58  	public void initialize(Properties props, WorkerBuildContext context, IndexManager indexManager) {
59  		this.props = props;
60  		this.isTransactional = context.enlistWorkerInTransaction();
61  		this.jmsQueueName = props.getProperty( JMS_QUEUE );
62  		this.indexName = indexManager.getIndexName();
63  		this.integrator = context.getUninitializedSearchIntegrator();
64  		this.factory = initializeJMSQueueConnectionFactory( props );
65  
66  		if ( ! isTransactional ) {
67  			// if we are not transactional, we can eagerly initialize the queue and connection
68  			this.jmsQueue = initializeJMSQueue( factory, props );
69  			this.connection = initializeJMSConnection( factory, props );
70  		}
71  	}
72  
73  	public Queue getJmsQueue() {
74  		if ( isTransactional ) {
75  			// return jmsQueue;
76  			return initializeJMSQueue( factory, props );
77  		}
78  		else {
79  			return jmsQueue;
80  		}
81  	}
82  
83  	public String getJmsQueueName() {
84  		return jmsQueueName;
85  	}
86  
87  	public String getIndexName() {
88  		return indexName;
89  	}
90  
91  	public SearchIntegrator getSearchIntegrator() {
92  		return integrator;
93  	}
94  
95  	@Override
96  	public void applyWork(List<LuceneWork> workList, IndexingMonitor monitor) {
97  		if ( workList == null ) {
98  			throw new IllegalArgumentException( "workList should not be null" );
99  		}
100 
101 		Runnable operation = new JmsBackendQueueTask( indexName, workList, this, integrator.getWorkSerializer() );
102 		operation.run();
103 	}
104 
105 	@Override
106 	public void applyStreamWork(LuceneWork singleOperation, IndexingMonitor monitor) {
107 		applyWork( Collections.singletonList( singleOperation ), monitor );
108 	}
109 
110 	public QueueConnection getJMSConnection() {
111 		// return connection;
112 		if ( isTransactional ) {
113 			return initializeJMSConnection( factory, props );
114 		}
115 		else {
116 			return connection;
117 		}
118 	}
119 
120 	public void releaseJMSConnection(QueueConnection queueConnection) {
121 		if ( isTransactional ) {
122 			// we create the connection each time
123 			// let's close it
124 			try {
125 				if ( queueConnection != null ) {
126 					queueConnection.close();
127 				}
128 			}
129 			catch (JMSException e) {
130 				log.unableToCloseJmsConnection( jmsQueueName, e );
131 			}
132 		}
133 		else {
134 			// we share the connection accross calls
135 			// noop
136 		}
137 	}
138 
139 	public boolean isTransactional() {
140 		return isTransactional;
141 	}
142 
143 	@Override
144 	public void close() {
145 		try {
146 			if ( connection != null ) {
147 				connection.close();
148 			}
149 		}
150 		catch (JMSException e) {
151 			log.unableToCloseJmsConnection( jmsQueueName, e );
152 		}
153 	}
154 
155 	/**
156 	 * Initialises the JMS QueueConnectionFactory to be used for sending Lucene work operations to the master node.
157 	 *
158 	 * @return the initialized {@link javax.jms.QueueConnectionFactory}
159 	 * @param props a {@link java.util.Properties} object.
160 	 */
161 	protected abstract QueueConnectionFactory initializeJMSQueueConnectionFactory(Properties props);
162 
163 	/**
164 	 * Initialises the JMS queue to be used for sending Lucene work operations to the master node.
165 	 * Invoked after {@link #initializeJMSQueueConnectionFactory(Properties)}
166 	 *
167 	 * @return the initialized {@link javax.jms.Queue}
168 	 * @param factory a {@link javax.jms.QueueConnectionFactory} object.
169 	 * @param props a {@link java.util.Properties} object.
170 	 */
171 	protected abstract Queue initializeJMSQueue(QueueConnectionFactory factory, Properties props);
172 
173 	/**
174 	 * Initialises the JMS QueueConnection to be used for sending Lucene work operations to the master node.
175 	 * This is invoked after {@link #initializeJMSQueue(QueueConnectionFactory, Properties)}.
176 	 *
177 	 * @return the initialized {@link javax.jms.QueueConnection}
178 	 * @param factory a {@link javax.jms.QueueConnectionFactory} object.
179 	 * @param props a {@link java.util.Properties} object.
180 	 */
181 	protected abstract QueueConnection initializeJMSConnection(QueueConnectionFactory factory, Properties props);
182 
183 }
