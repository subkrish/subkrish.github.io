7   package org.hibernate.search.bridge.builtin;
8   
9   import org.hibernate.search.util.StringHelper;
10  import org.hibernate.search.bridge.AppliedOnTypeAwareBridge;
11  import org.hibernate.search.bridge.TwoWayStringBridge;
12  
13  /**
14   * Bridge an {@link Enum} to a {@link String} using  {@link Enum#name()}.
15   *
16   * @author Sylvain Vieujot
17   */
18  public class EnumBridge implements TwoWayStringBridge, AppliedOnTypeAwareBridge {
19  
20  	private Class<? extends Enum> clazz = null;
21  
22  	@Override
23  	public Enum<? extends Enum> stringToObject(String stringValue) {
24  		if ( StringHelper.isEmpty( stringValue ) ) {
25  			return null;
26  		}
27  		return Enum.valueOf( clazz, stringValue );
28  	}
29  
30  	@Override
31  	public String objectToString(Object object) {
32  		Enum e = (Enum) object;
33  		return e != null ? e.name() : null;
34  	}
35  
36  	@Override
37  	public void setAppliedOnType(Class<?> returnType) {
38  		@SuppressWarnings("unchecked") //only called for an enum
39  		Class<? extends Enum> enumReturnType = (Class<? extends Enum>) returnType;
40  		this.clazz = enumReturnType;
41  	}
42  }
