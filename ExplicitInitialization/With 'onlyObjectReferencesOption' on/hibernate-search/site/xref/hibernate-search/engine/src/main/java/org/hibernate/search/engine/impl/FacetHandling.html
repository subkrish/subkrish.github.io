7   package org.hibernate.search.engine.impl;
8   
9   import java.io.IOException;
10  
11  import org.apache.lucene.document.Document;
12  import org.apache.lucene.facet.FacetsConfig;
13  import org.hibernate.search.util.logging.impl.Log;
14  import org.hibernate.search.util.logging.impl.LoggerFactory;
15  
16  /**
17   * Helper class to manage a {@literal FacetsConfig} instance:
18   * this needs to be lazily allocated as it's a quite heavy object.
19   * Ideally it should be reused across threads, but since field options
20   * are index specific that doesn't suit our current model.
21   */
22  public final class FacetHandling {
23  
24  	private static final Log log = LoggerFactory.make();
25  
26  	private FacetsConfig facetConfig = null;
27  
28  	public Document build(Document doc) {
29  		if ( facetConfig == null ) {
30  			return doc;
31  		}
32  		else {
33  			try {
34  				return facetConfig.build( doc );
35  			}
36  			catch (IOException e) {
37  				throw log.errorDuringFacetingIndexing( e );
38  			}
39  		}
40  	}
41  
42  	public void setMultiValued(String facetName) {
43  		facetConfig.setMultiValued( facetName, true );
44  	}
45  
46  	public void enableFacetProcessing() {
47  		if ( facetConfig == null ) {
48  			facetConfig = new FacetsConfig();
49  		}
50  	}
51  
52  }
