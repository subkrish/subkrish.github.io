156 	public ScoreDoc scoreDoc(int index) throws IOException {
157 		if ( index >= totalHits ) {
158 			throw new SearchException( "Not a valid ScoreDoc index: " + index );
159 		}
160 
161 		// TODO - Is there a better way to get more TopDocs? Get more or less?
162 		if ( index >= topDocs.scoreDocs.length ) {
163 			updateTopDocs( 2 * index );
164 		}
165 		//if the refresh timed out, raise an exception
166 		if ( timeoutManager.isTimedOut() && index >= topDocs.scoreDocs.length ) {
167 			throw timeoutExceptionFactory.createTimeoutException(
168 					"Timeout period exceeded. Cannot load document: " + index,
169 					searcher.describeQuery()
170 			);
171 		}
172 		return topDocs.scoreDocs[index];
173 	}

211 	/**
212 	 * @param n the number of {@code TopDoc}s to retrieve. The actual retrieved number of {@code TopDoc}s is n or the
213 	 * total number of documents if {@code n > maxDoc}
214 	 *
215 	 * @throws IOException in case a search exception occurs
216 	 */
217 	private void updateTopDocs(int n) throws IOException {
218 		final int totalMaxDocs = searcher.maxDoc();
219 		final int maxDocs = Math.min( n, totalMaxDocs );
220 
221 		final TopDocsCollector<?> topDocCollector;
222 		final TotalHitCountCollector hitCountCollector;
223 		Collector collector;
224 		if ( maxDocs != 0 ) {
225 			topDocCollector = createTopDocCollector( maxDocs );
226 			hitCountCollector = null;
227 			collector = topDocCollector;
228 			collector = optionallyEnableFacetingCollector( collector );
229 			collector = optionallyEnableDistanceCollector( collector, maxDocs );
230 		}
231 		else {
232 			topDocCollector = null;
233 			hitCountCollector = new TotalHitCountCollector();
234 			collector = hitCountCollector;
235 		}
236 		collector = decorateWithTimeOutCollector( collector );
237 
238 		boolean timeoutNow = isImmediateTimeout();
239 		if ( !timeoutNow ) {
240 			try {
241 				searcher.search( filter, collector );
242 			}
243 			catch (TimeLimitingCollector.TimeExceededException e) {
244 				//we have reached the time limit and stopped before the end
245 				//TimeoutManager.isTimedOut should be above that limit but set if for safety
246 				timeoutManager.forceTimedOut();
247 			}
248 		}
249 
250 		// update top docs and totalHits
251 		if ( maxDocs != 0 ) {
252 			this.topDocs = topDocCollector.topDocs();
253 			this.totalHits = topDocs.totalHits;
254 			// if we were collecting facet data we have to update our instance state
255 			if ( facetsCollector != null ) {
256 				updateFacets();
257 			}
258 		}
259 		else {
260 			this.topDocs = null;
261 			this.totalHits = hitCountCollector.getTotalHits();
262 		}
263 		timeoutManager.isTimedOut();
264 	}

266 	private void updateFacets() throws IOException {
267 		facetMap = new HashMap<>();
268 		for ( FacetingRequest facetRequest : facetRequests.values() ) {
269 			ArrayList<Facet> facets;
270 			if ( facetRequest instanceof DiscreteFacetRequest ) {
271 				facets = updateStringFacets( (DiscreteFacetRequest) facetRequest );
272 			}
273 			else {
274 				facets = updateRangeFacets( (RangeFacetRequest<?>) facetRequest );
275 			}
276 
277 			// sort if necessary
278 			if ( !facetRequest.getSort().equals( FacetSortOrder.RANGE_DEFINITION_ORDER ) ) {
279 				Collections.sort( facets, FacetComparators.get( facetRequest.getSort() ) );
280 			}
281 
282 			// trim to the expected size
283 			int maxNumberOfExpectedFacets = facetRequest.getMaxNumberOfFacets();
284 			if ( maxNumberOfExpectedFacets > 0 && facets.size() > maxNumberOfExpectedFacets ) {
285 				facets = new ArrayList<>( facets.subList( 0, facetRequest.getMaxNumberOfFacets() ) );
286 			}
287 
288 			facetMap.put( facetRequest.getFacetingName(), facets );
289 		}
290 	}

292 	private ArrayList<Facet> updateRangeFacets(RangeFacetRequest<?> facetRequest) throws IOException {
293 		ArrayList<Facet> facets;
294 		if ( ReflectionHelper.isIntegerType( facetRequest.getFacetValueType() )
295 				|| Date.class.isAssignableFrom( facetRequest.getFacetValueType() ) ) {
296 			FacetResult facetResult = getFacetResultForLongRange( facetRequest );
297 			facets = new ArrayList<>( facetResult.labelValues.length );
298 			for ( LabelAndValue labelAndValue : facetResult.labelValues ) {
299 				if ( !facetRequest.hasZeroCountsIncluded() && (int) labelAndValue.value == 0 ) {
300 					continue;
301 				}
302 
303 				Facet facet = facetRequest.createFacet( labelAndValue.label, (int) labelAndValue.value );
304 				facets.add( facet );
305 			}
306 		}
307 		else if ( ReflectionHelper.isFloatingPointType( facetRequest.getFacetValueType() ) ) {
308 			FacetResult facetResult = getFacetResultForFloatingPointRange( facetRequest );
309 			facets = new ArrayList<>( facetResult.labelValues.length );
310 			for ( LabelAndValue labelAndValue : facetResult.labelValues ) {
311 				if ( !facetRequest.hasZeroCountsIncluded() && (int) labelAndValue.value == 0 ) {
312 					continue;
313 				}
314 
315 				Facet facet = facetRequest.createFacet( labelAndValue.label, (int) labelAndValue.value );
316 				facets.add( facet );
317 			}
318 		}
319 		else {
320 			throw log.unsupportedFacetRangeParameter( facetRequest.getFacetValueType().getName() );
321 		}
322 
323 		return facets;
324 	}

326 	private FacetResult getFacetResultForFloatingPointRange(RangeFacetRequest<?> facetRequest) throws IOException {
327 		List<? extends FacetRange<?>> facetRanges = facetRequest.getFacetRangeList();
328 		DoubleRange[] ranges = new DoubleRange[facetRanges.size()];
329 
330 		int i = 0;
331 		for ( FacetRange<?> facetRange : facetRanges ) {
332 			ranges[i] = new DoubleRange(
333 					facetRange.getRangeString(),
334 					facetRange.getMin() == null ? Double.MIN_VALUE : ( (Number) facetRange.getMin() ).doubleValue(),
335 					facetRange.isMinIncluded(),
336 					facetRange.getMax() == null ? Double.MAX_VALUE : ( (Number) facetRange.getMax() ).doubleValue(),
337 					facetRange.isMaxIncluded()
338 			);
339 			i++;
340 		}
341 
342 		DoubleRangeFacetCounts facetCount = new DoubleRangeFacetCounts(
343 				facetRequest.getFieldName(),
344 				facetsCollector,
345 				ranges
346 		);
347 		return facetCount.getTopChildren(
348 				facetRequest.getMaxNumberOfFacets(),
349 				facetRequest.getFieldName()
350 		);
351 	}

353 	private FacetResult getFacetResultForLongRange(RangeFacetRequest<?> facetRequest)
354 			throws IOException {
355 		List<? extends FacetRange<?>> facetRanges = facetRequest.getFacetRangeList();
356 		LongRange[] ranges = new LongRange[facetRanges.size()];
357 
358 		int i = 0;
359 		for ( FacetRange<?> facetRange : facetRanges ) {
360 			long min;
361 			long max;
362 
363 			if ( ReflectionHelper.isIntegerType( facetRequest.getFacetValueType() ) ) {
364 				min = facetRange.getMin() == null ? Long.MIN_VALUE : ( (Number) facetRange.getMin() ).longValue();
365 				max = facetRange.getMax() == null ? Long.MAX_VALUE : ( (Number) facetRange.getMax() ).longValue();
366 			}
367 			else {
368 				min = facetRange.getMin() == null ? Long.MIN_VALUE : ( (Date) facetRange.getMin() ).getTime();
369 				max = facetRange.getMax() == null ? Long.MAX_VALUE : ( (Date) facetRange.getMax() ).getTime();
370 			}
371 
372 			ranges[i] = new LongRange(
373 					facetRange.getRangeString(),
374 					min,
375 					facetRange.isMinIncluded(),
376 					max,
377 					facetRange.isMaxIncluded()
378 			);
379 			i++;
380 		}
381 
382 		LongRangeFacetCounts facetCount = new LongRangeFacetCounts(
383 				facetRequest.getFieldName(),
384 				facetsCollector,
385 				ranges
386 		);
387 		return facetCount.getTopChildren(
388 				facetRequest.getMaxNumberOfFacets(),
389 				facetRequest.getFieldName()
390 		);
391 	}

393 	private ArrayList<Facet> updateStringFacets(DiscreteFacetRequest facetRequest) throws IOException {
394 		SortedSetDocValuesReaderState docValuesReaderState;
395 		try {
396 			docValuesReaderState = new DefaultSortedSetDocValuesReaderState( searcher.getIndexReader() );
397 		}
398 		catch (IllegalArgumentException e) {
399 			// happens in case there are no facets at all configured for the matching documents
400 			throw log.unknownFieldNameForFaceting( facetRequest.getFacetingName(), facetRequest.getFieldName() );
401 		}
402 		SortedSetDocValuesFacetCounts facetCounts = new SortedSetDocValuesFacetCounts(
403 				docValuesReaderState, facetsCollector
404 		);
405 
406 		Set<String> termValues = Collections.emptySet();
407 		if ( facetRequest.hasZeroCountsIncluded() ) {
408 			termValues = findAllTermsForField( facetRequest.getFieldName(), searcher.getIndexReader() );
409 		}
410 
411 		int maxFacetCount = facetRequest.getMaxNumberOfFacets() < 0 ? DEFAULT_FACET_RETRIEVAL_SIZE : facetRequest.getMaxNumberOfFacets();
412 		final FacetResult facetResult;
413 		try {
414 			// This might return null!
415 			facetResult = facetCounts.getTopChildren( maxFacetCount, facetRequest.getFieldName() );
416 		}
417 		catch (IllegalArgumentException e) {
418 			// happens in case there are facets in general, but not for this specific field
419 			throw log.unknownFieldNameForFaceting( facetRequest.getFacetingName(), facetRequest.getFieldName() );
420 		}
421 		ArrayList<Facet> facets = new ArrayList<>();
422 		if ( facetResult != null ) {
423 			for ( LabelAndValue labelAndValue : facetResult.labelValues ) {
424 				Facet facet = facetRequest.createFacet( labelAndValue.label, (int) labelAndValue.value );
425 				facets.add( facet );
426 				termValues.remove( labelAndValue.label );
427 			}
428 		}
429 		for ( String termValue : termValues ) {
430 			Facet facet = facetRequest.createFacet( termValue, 0 );
431 			facets.add( 0, facet );
432 		}
433 		return facets;
434 	}

436 	/**
437 	 * Returns a set of all possible indexed values for a given field name.
438 	 *
439 	 * Per default Lucene's native discrete faceting won't return values where
440 	 * the count would be 0 for a given query. To still include these "zero count values"
441 	 * we need to go through the index and find all potential values for a field and compare them
442 	 * against the values which had a count. This might potentially be expensive, so returning
443 	 * zero count values is per default disabled.
444 	 *
445 	 * @param fieldName the document field name
446 	 * @param reader the index reader
447 	 * @return a set of all possible indexed values for a given field name
448 	 * @throws IOException
449 	 */
450 	private Set<String> findAllTermsForField(String fieldName, IndexReader reader) throws IOException {
451 		Set<String> termValues = new HashSet<>();
452 		for ( LeafReaderContext leaf : reader.leaves() ) {
453 			final LeafReader atomicReader = leaf.reader();
454 			Terms terms = atomicReader.terms( fieldName );
455 			if ( terms == null ) {
456 				continue;
457 			}
458 			final TermsEnum iterator = terms.iterator();
459 			BytesRef byteRef;
460 			while ( ( byteRef = iterator.next() ) != null ) {
461 				termValues.add( byteRef.utf8ToString() );
462 			}
463 		}
464 		return termValues;
465 	}

467 	private Collector optionallyEnableFacetingCollector(Collector collector) {
468 		if ( facetRequests == null || facetRequests.isEmpty() ) {
469 			return collector;
470 		}
471 		facetsCollector = new FacetsCollector();
472 		return MultiCollector.wrap( facetsCollector, collector );
473 	}
474 
475 	private Collector optionallyEnableDistanceCollector(Collector collector, int maxDocs) {
476 		if ( spatialFieldName == null || spatialFieldName.isEmpty() || spatialSearchCenter == null ) {
477 			return collector;
478 		}
479 		distanceCollector = new DistanceCollector( spatialSearchCenter, maxDocs, spatialFieldName );
480 
481 		return MultiCollector.wrap( distanceCollector, collector );
482 	}

484 	private boolean isImmediateTimeout() {
485 		boolean timeoutAt0 = false;
486 		if ( timeoutManager.getType() == TimeoutManager.Type.LIMIT ) {
487 			final Long timeoutLeft = timeoutManager.getTimeoutLeftInMilliseconds();
488 			if ( timeoutLeft != null ) {
489 				if ( timeoutLeft == 0l ) {
490 					if ( timeoutManager.isTimedOut() ) {
491 						timeoutManager.forceTimedOut();
492 						timeoutAt0 = true;
493 					}
494 				}
495 			}
496 			else {
497 				if ( timeoutManager.isTimedOut() ) {
498 					timeoutManager.forceTimedOut();
499 				}
500 			}
501 		}
502 		return timeoutAt0;
503 	}
504 
505 	private Collector decorateWithTimeOutCollector(Collector collector) {
506 		Collector maybeTimeLimitingCollector = collector;
507 		if ( timeoutManager.getType() == TimeoutManager.Type.LIMIT ) {
508 			final Long timeoutLeft = timeoutManager.getTimeoutLeftInMilliseconds();
509 			if ( timeoutLeft != null ) {
510 				Counter counter = timeoutManager.getLuceneTimeoutCounter();
511 				maybeTimeLimitingCollector = new TimeLimitingCollector( collector, counter, timeoutLeft );
512 			}
513 		}
514 		return maybeTimeLimitingCollector;
515 	}

517 	private TopDocsCollector<?> createTopDocCollector(int maxDocs) throws IOException {
518 		TopDocsCollector<?> topCollector;
519 		if ( sort == null ) {
520 			topCollector = TopScoreDocCollector.create( maxDocs );
521 		}
522 		else {
523 			boolean fillFields = true;
524 			topCollector = TopFieldCollector.create(
525 					sort,
526 					maxDocs,
527 					fillFields,
528 					searcher.isFieldSortDoTrackScores(),
529 					searcher.isFieldSortDoMaxScore()
530 			);
531 		}
532 		return topCollector;
533 	}
