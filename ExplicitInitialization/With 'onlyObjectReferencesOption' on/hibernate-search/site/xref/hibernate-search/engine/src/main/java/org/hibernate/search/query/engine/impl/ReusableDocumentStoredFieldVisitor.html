7   package org.hibernate.search.query.engine.impl;
8   
9   import java.io.IOException;
10  import java.nio.charset.StandardCharsets;
11  import java.util.Set;
12  
13  import org.apache.lucene.document.Document;
14  import org.apache.lucene.document.Field;
15  import org.apache.lucene.document.FieldType;
16  import org.apache.lucene.document.StoredField;
17  import org.apache.lucene.document.TextField;
18  import org.apache.lucene.index.FieldInfo;
19  import org.apache.lucene.index.StoredFieldVisitor;
20  
21  /**
22   * Inspired by Lucene's DocumentStoredFieldVisitor, with the difference that we want
23   * to reuse the same FieldVisitor to load multiple Document instances.
24   * Also the accepted fields are kept in a linked-list like structure, to optimize
25   * for small amounts of accepted fields.
26   *
27   * A ReusableDocumentStoredFieldVisitor is NOT threadsafe: in case you need one
28   * for multiple threads make a clone for each thread.
29   *
30   * @see org.apache.lucene.document.DocumentStoredFieldVisitor
31   * @author Sanne Grinovero
32   */
33  public final class ReusableDocumentStoredFieldVisitor extends StoredFieldVisitor {
34  
35  	private static final FieldAcceptor NOT_ACCEPT = new DenyingFieldAcceptor();
36  
37  	private final FieldAcceptor rootAcceptor;
38  	private final int totalFields;
39  
40  	//The Lucene Document which will be returned. Lazily initialized.
41  	private Document doc = null;
42  
43  	//Counts the number of fields which have not been loaded yet (counting down from the known set of needed fields)
44  	//This field needs to be reset to the value of totalFields when the doc field is changed.
45  	private int missingFields;
46  
47  	public ReusableDocumentStoredFieldVisitor(Set<String> fieldsToLoad) {
48  		FieldAcceptor previous = NOT_ACCEPT;
49  		for ( String fieldName : fieldsToLoad ) {
50  			previous = new ChainedFieldAcceptor( previous, fieldName );
51  		}
52  		this.rootAcceptor = previous;
53  		this.totalFields = fieldsToLoad.size();
54  		this.missingFields = totalFields;
55  	}
56  
57  	@Override
58  	public void binaryField(FieldInfo fieldInfo, byte[] value) throws IOException {
59  		getDocument().add( new StoredField( fieldInfo.name, value ) );
60  	}
61  
62  	@Override
63  	public void stringField(FieldInfo fieldInfo, byte[] value) throws IOException {
64  		final FieldType ft = new FieldType( TextField.TYPE_STORED );
65  		ft.setStoreTermVectors( fieldInfo.hasVectors() );
66  		ft.setOmitNorms( fieldInfo.omitsNorms() );
67  		ft.setIndexOptions( fieldInfo.getIndexOptions() );
68  		getDocument().add( new Field( fieldInfo.name, new String( value, StandardCharsets.UTF_8 ), ft ) );
69  	}
70  
71  	@Override
72  	public void intField(FieldInfo fieldInfo, int value) {
73  		getDocument().add( new StoredField( fieldInfo.name, value ) );
74  	}
75  
76  	@Override
77  	public void longField(FieldInfo fieldInfo, long value) {
78  		getDocument().add( new StoredField( fieldInfo.name, value ) );
79  	}
80  
81  	@Override
82  	public void floatField(FieldInfo fieldInfo, float value) {
83  		getDocument().add( new StoredField( fieldInfo.name, value ) );
84  	}
85  
86  	@Override
87  	public void doubleField(FieldInfo fieldInfo, double value) {
88  		getDocument().add( new StoredField( fieldInfo.name, value ) );
89  	}
90  
91  	@Override
92  	public Status needsField(FieldInfo fieldInfo) throws IOException {
93  		if ( missingFields == 0 ) {
94  			// An aggressive STOP could prevent unnecessary I/O !
95  			return Status.STOP;
96  		}
97  		final Status s = rootAcceptor.acceptField( fieldInfo.name );
98  		if ( s == Status.YES ) {
99  			missingFields--;
100 		}
101 		return s;
102 	}
103 
104 	/**
105 	 * Useful for tests
106 	 * @return the amount of accepted fields
107 	 */
108 	public int countAcceptedFields() {
109 		FieldAcceptor acceptor = rootAcceptor;
110 		int count = 0;
111 		while ( acceptor != NOT_ACCEPT ) {
112 			//If it's not negative it has to be positive:
113 			ChainedFieldAcceptor positiveAcceptor = (ChainedFieldAcceptor) acceptor;
114 			acceptor = positiveAcceptor.next;
115 			count++;
116 		}
117 		return count;
118 	}
119 
120 	/**
121 	 * Retrieve the visited document, and resets the instance to be reused by creating a new Document
122 	 * internally.
123 	 *
124 	 * @return Document populated with stored fields.
125 	 */
126 	public Document getDocumentAndReset() {
127 		final Document localDoc = this.doc;
128 		if ( localDoc == null ) {
129 			return new Document();
130 		}
131 		else {
132 			this.doc = null;
133 			this.missingFields = totalFields;
134 			return localDoc;
135 		}
136 	}
137 
138 	private Document getDocument() {
139 		Document localDoc = this.doc;
140 		if ( localDoc == null ) {
141 			localDoc = new Document();
142 			this.doc = localDoc;
143 		}
144 		return localDoc;
145 	}
146 
147 	/* The structure below shapes a chain of accepted field names:
148 	 * you could think of it as a linked list. */
149 
150 	private interface FieldAcceptor {
151 		Status acceptField(String fieldName);
152 	}
153 
154 	private static final class DenyingFieldAcceptor implements FieldAcceptor {
155 		@Override
156 		public Status acceptField(final String fieldName) {
157 			return Status.NO;
158 		}
159 	}
160 
161 	private static final class ChainedFieldAcceptor implements FieldAcceptor {
162 		final FieldAcceptor next;
163 		final String acceptedFieldName;
164 		ChainedFieldAcceptor(FieldAcceptor next, String acceptedFieldName) {
165 			this.next = next;
166 			this.acceptedFieldName = acceptedFieldName;
167 		}
168 		@Override
169 		public Status acceptField(final String fieldName) {
170 			if ( acceptedFieldName.equals( fieldName ) ) {
171 				return Status.YES;
172 			}
173 			else {
174 				return next.acceptField( fieldName );
175 			}
176 		}
177 	}
178 
179 }
