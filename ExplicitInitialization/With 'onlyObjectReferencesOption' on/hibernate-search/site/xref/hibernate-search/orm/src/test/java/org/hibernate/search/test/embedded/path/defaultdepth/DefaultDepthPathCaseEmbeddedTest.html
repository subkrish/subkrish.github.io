8   package org.hibernate.search.test.embedded.path.defaultdepth;
9   
10  import java.util.List;
11  
12  import org.apache.lucene.search.Query;
13  
14  import org.hibernate.Session;
15  import org.hibernate.Transaction;
16  
17  import org.hibernate.search.FullTextSession;
18  import org.hibernate.search.Search;
19  import org.hibernate.search.exception.SearchException;
20  import org.hibernate.search.query.dsl.QueryBuilder;
21  import org.hibernate.search.test.SearchTestBase;
22  import org.junit.After;
23  import org.junit.Assert;
24  import org.junit.Before;
25  import org.junit.Test;
26  
27  import static org.junit.Assert.fail;
28  
29  /**
30   * @author Davide D'Alto
31   */
32  public class DefaultDepthPathCaseEmbeddedTest extends SearchTestBase {
33  
34  	private Session s = null;
35  	private EntityA entityA = null;
36  
37  	@Override
38  	@Before
39  	public void setUp() throws Exception {
40  		super.setUp();
41  		EntityC indexedC = new EntityC( "indexed" );
42  		EntityC skippedC = new EntityC( "indexed" );
43  
44  		EntityB indexedB = new EntityB( indexedC, skippedC );
45  
46  		entityA = new EntityA( indexedB );
47  		s = openSession();
48  		persistEntity( s, indexedC, skippedC, indexedB, entityA );
49  	}
50  
51  	@Override
52  	@After
53  	public void tearDown() throws Exception {
54  		s.clear();
55  
56  		deleteAll( s, EntityA.class, EntityB.class, EntityC.class );
57  		s.close();
58  		super.tearDown();
59  	}
60  
61  	@Test
62  	public void testFieldIsIndexedIfInPath() throws Exception {
63  		List<EntityA> result = search( s, "b.indexed.field", "indexed" );
64  
65  		Assert.assertEquals( 1, result.size() );
66  		Assert.assertEquals( entityA.id, result.get( 0 ).id );
67  	}
68  
69  	@Test
70  	public void testEmbeddedNotIndexedIfNotInPath() throws Exception {
71  		try {
72  			search( s, "b.skipped.indexed", "indexed" );
73  			fail( "Should not index embedded property if not in path and not in depth limit" );
74  		}
75  		catch (SearchException e) {
76  		}
77  	}
78  
79  	@Test
80  	public void testFieldNotIndexedIfNotInPath() throws Exception {
81  		try {
82  			search( s, "b.indexed.skipped", "skipped" );
83  			fail( "Should not index embedded property if not in path and not in depth limit" );
84  		}
85  		catch (SearchException e) {
86  		}
87  	}
88  
89  	private List<EntityA> search(Session s, String field, String value) {
90  		FullTextSession session = Search.getFullTextSession( s );
91  		QueryBuilder queryBuilder = session.getSearchFactory().buildQueryBuilder().forEntity( EntityA.class ).get();
92  		Query query = queryBuilder.keyword().onField( field ).matching( value ).createQuery();
93  		@SuppressWarnings("unchecked")
94  		List<EntityA> result = session.createFullTextQuery( query ).list();
95  		return result;
96  	}
97  
98  	private void deleteAll(Session s, Class<?>... classes) {
99  		Transaction tx = s.beginTransaction();
100 		for ( Class<?> each : classes ) {
101 			List<?> list = s.createCriteria( each ).list();
102 			for ( Object object : list ) {
103 				s.delete( object );
104 			}
105 		}
106 		tx.commit();
107 	}
108 
109 	private void persistEntity(Session s, Object... entities) {
110 		Transaction tx = s.beginTransaction();
111 		for ( Object entity : entities ) {
112 			s.persist( entity );
113 		}
114 		tx.commit();
115 	}
116 
117 	@Override
118 	public Class<?>[] getAnnotatedClasses() {
119 		return new Class<?>[] { EntityA.class, EntityB.class, EntityC.class };
120 	}
121 }
