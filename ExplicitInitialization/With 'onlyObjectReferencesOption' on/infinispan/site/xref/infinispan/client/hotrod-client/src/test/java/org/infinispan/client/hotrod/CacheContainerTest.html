19  @Test(groups = "functional", testName = "client.hotrod.CacheContainerTest")
20  public class CacheContainerTest extends SingleCacheManagerTest {
21  
22     private static final String CACHE_NAME = "someName";
23  
24     EmbeddedCacheManager cacheManager = null;
25     HotRodServer hotrodServer = null;
26     RemoteCacheManager remoteCacheManager;
27  
28     @Override
29     protected EmbeddedCacheManager createCacheManager() throws Exception {
30        cacheManager = TestCacheManagerFactory.createCacheManager(
31              hotRodCacheConfiguration());
32        cacheManager.defineConfiguration(CACHE_NAME, hotRodCacheConfiguration().build());
33        hotrodServer = HotRodClientTestingUtil.startHotRodServer(cacheManager);
34        remoteCacheManager = new RemoteCacheManager("localhost:" + hotrodServer.getPort(), true);
35        return cacheManager;
36     }
37  
38     @AfterTest
39     public void release() {
40        killCacheManagers(cacheManager);
41        killRemoteCacheManager(remoteCacheManager);
42        killServers(hotrodServer);
43     }
44  
45     public void testObtainingSameInstanceMultipleTimes() {
46        RemoteCache<Object, Object> objectCache = remoteCacheManager.getCache();
47        RemoteCache<Object, Object> objectCache2 = remoteCacheManager.getCache();
48        assert objectCache == objectCache2;
49     }
50  
51     public void testObtainingSameInstanceMultipleTimes2() {
52        RemoteCache<Object, Object> objectCache = remoteCacheManager.getCache(CACHE_NAME);
53        RemoteCache<Object, Object> objectCache2 = remoteCacheManager.getCache(CACHE_NAME);
54        assert objectCache == objectCache2;
55     }
56  }
