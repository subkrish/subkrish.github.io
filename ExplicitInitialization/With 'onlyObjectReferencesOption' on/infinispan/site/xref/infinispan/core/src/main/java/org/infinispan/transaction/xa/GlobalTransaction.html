15  /**
16   * Uniquely identifies a transaction that spans all JVMs in a cluster. This is used when replicating all modifications
17   * in a transaction; the PREPARE and COMMIT (or ROLLBACK) messages have to have a unique identifier to associate the
18   * changes with<br>. GlobalTransaction should be instantiated thorough {@link TransactionFactory} class,
19   * as their type depends on the runtime configuration.
20   *
21   * @author <a href="mailto:bela@jboss.org">Bela Ban</a> Apr 12, 2003
22   * @author <a href="mailto:manik@jboss.org">Manik Surtani (manik@jboss.org)</a>
23   * @author Mircea.Markus@jboss.com
24   * @since 4.0
25   */
26  public class GlobalTransaction implements Cloneable {
27  
28     private static final AtomicLong sid = new AtomicLong(0);
29  
30     protected long id = -1;
31  
32     protected Address addr = null;
33     private int hash_code = -1;  // in the worst case, hashCode() returns 0, then increases, so we're safe here
34     private boolean remote = false;
35  
36     /**
37      * empty ctor used by externalization.
38      */
39     protected GlobalTransaction() {
40     }
41  
42     protected GlobalTransaction(Address addr, boolean remote) {
43        this.id = sid.incrementAndGet();
44        this.addr = addr;
45        this.remote = remote;
46     }
47  
48     public Address getAddress() {
49        return addr;
50     }
51  
52     public long getId() {
53        return id;
54     }
55  
56     public boolean isRemote() {
57        return remote;
58     }
59  
60     public void setRemote(boolean remote) {
61        this.remote = remote;
62     }
63  
64     @Override
65     public int hashCode() {
66        if (hash_code == -1) {
67           hash_code = (addr != null ? addr.hashCode() : 0) + (int) id;
68        }
69        return hash_code;
70     }
71  
72     @Override
73     public boolean equals(Object other) {
74        if (this == other)
75           return true;
76        if (!(other instanceof GlobalTransaction))
77           return false;
78  
79        GlobalTransaction otherGtx = (GlobalTransaction) other;
80        boolean aeq = (addr == null) ? (otherGtx.addr == null) : addr.equals(otherGtx.addr);
81        return aeq && (id == otherGtx.id);
82     }
83  
84     @Override
85     public String toString() {
86        StringBuilder sb = new StringBuilder();
87        sb.append("GlobalTransaction:<").append(addr).append(">:").append(id).append(isRemote() ? ":remote" : ":local");
88        return sb.toString();
89     }
90  
91     /**
92      * Returns a simplified representation of the transaction.
93      */
94     public final String globalId() {
95        return getAddress() + ":" + getId();
96     }
97  
98     public void setId(long id) {
99        this.id = id;
100    }
101 
102    public void setAddress(Address address) {
103       this.addr = address;
104    }
105 
106    @Override
107    public Object clone() {
108       try {
109          return super.clone();
110       } catch (CloneNotSupportedException e) {
111          throw new IllegalStateException("Impossible!");
112       }
113    }
114 
115    protected abstract static class AbstractGlobalTxExternalizer<T extends GlobalTransaction> extends AbstractExternalizer<T> {
116       @Override
117       public void writeObject(ObjectOutput output, T gtx) throws IOException {
118          output.writeLong(gtx.id);
119          output.writeObject(gtx.addr);
120       }
121 
122       /**
123        * Factory method for GlobalTransactions
124        * @return a newly constructed instance of GlobalTransaction or one of its subclasses
125        **/
126       protected abstract T createGlobalTransaction();
127 
128       @Override
129       public T readObject(ObjectInput input) throws IOException, ClassNotFoundException {
130          T gtx = createGlobalTransaction();
131          gtx.id = input.readLong();
132          gtx.addr = (Address) input.readObject();
133          return gtx;
134       }
135    }
136 
137    public static class Externalizer extends AbstractGlobalTxExternalizer<GlobalTransaction> {
138       @Override
139       @SuppressWarnings("unchecked")
140       public Set<Class<? extends GlobalTransaction>> getTypeClasses() {
141          return Util.<Class<? extends GlobalTransaction>>asSet(GlobalTransaction.class);
142       }
143 
144       @Override
145       public Integer getId() {
146          return Ids.GLOBAL_TRANSACTION;
147       }
148 
149       @Override
150       protected GlobalTransaction createGlobalTransaction() {
151          return TransactionFactory.TxFactoryEnum.NODLD_NORECOVERY_XA.newGlobalTransaction();
152       }
153    }
