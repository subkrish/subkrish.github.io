35  /**
36   * Tests for HotRod client and its RemoteCache API. Subclasses must provide a
37   * way to get the list of remote HotRod servers and to assert the cache is
38   * empty.
39   * <p/>
40   * Subclasses may be used in Client-Server mode or Hybrid mode where HotRod
41   * server runs as a library deployed in an application server.
42   *
43   * @author Richard Achmatowicz
44   * @author Martin Gencur
45   * @author Jozef Vilkolak

57      @Before
58      public void initialize() {
59          if (remoteCacheManager == null) {
60              Configuration config = createRemoteCacheManagerConfiguration();
61              remoteCacheManager = new RemoteCacheManager(config, true);

64          assertCacheEmpty();

74      private Configuration createRemoteCacheManagerConfiguration() {
75          ConfigurationBuilder config = new ConfigurationBuilder();
76          for (RemoteInfinispanServer server : getServers()) {
77              config.addServer().host(server.getHotrodEndpoint().getInetAddress().getHostName())
78                      .port(server.getHotrodEndpoint().getPort());
79          }
80          config.balancingStrategy("org.infinispan.server.test.client.hotrod.HotRodTestRequestBalancingStrategy")
81                  // load balancing
82                  .balancingStrategy("org.infinispan.client.hotrod.impl.transport.tcp.RoundRobinBalancingStrategy")
83                          // list of HotRod servers available to connect to
84                          //.addServers(hotRodServerList)
85                  .forceReturnValues(false)
86                          // TCP stuff
87                  .tcpNoDelay(true)
88                  .pingOnStartup(true)
89                  .transportFactory("org.infinispan.client.hotrod.impl.transport.tcp.TcpTransportFactory")
90                          // marshalling
91                  .marshaller("org.infinispan.commons.marshall.jboss.GenericJBossMarshaller")
92                          // executors
93                  .asyncExecutorFactory().factoryClass("org.infinispan.client.hotrod.impl.async.DefaultAsyncExecutorFactory")
94                  .addExecutorProperty("infinispan.client.hotrod.default_executor_factory.pool_size", "10")
95                  .addExecutorProperty("infinispan.client.hotrod.default_executor_factory.queue_size", "100000")
96                          //hashing
97                  .keySizeEstimate(64)
98                  .valueSizeEstimate(512);
99  
100         if (isDistributedMode()) {
101             config.consistentHashImpl(1, "org.infinispan.client.hotrod.impl.consistenthash.ConsistentHashV1");
102         } else {
103             config.consistentHashImpl(2, "org.infinispan.client.hotrod.impl.consistenthash.ConsistentHashV2");
104         }
105 
106         return config.build();
107     }

989     public static <K> void expectSingleEvent(K key, EventLogListener eventListener, ClientEvent.Type type) {
990         switch (type) {
991             case CLIENT_CACHE_ENTRY_CREATED:
992                 ClientCacheEntryCreatedEvent createdEvent = eventListener.pollEvent(type);
993                 assertEquals(key, createdEvent.getKey());
994                 break;
995             case CLIENT_CACHE_ENTRY_MODIFIED:
996                 ClientCacheEntryModifiedEvent modifiedEvent = eventListener.pollEvent(type);
997                 assertEquals(key, modifiedEvent.getKey());
998                 break;
999             case CLIENT_CACHE_ENTRY_REMOVED:
1000                 ClientCacheEntryRemovedEvent removedEvent = eventListener.pollEvent(type);
1001                 assertEquals(key, removedEvent.getKey());
1002                 break;
1003         }
1004         assertEquals(0, eventListener.queue(type).size());
1005     }
