167 	private static Context createNewContext() {
168 		Context c = new Context();
169 		// Older javacs such as the 1.6 of apple has DefaultFileManager. Newer ones have JavacFileManager.
170 		// As javac6 might be on the classpath, JavacFileManager will probably exist but its initialization will fail.
171 		// Initializing both is as far as I know not an issue. -ReinierZ
172 		try {
173 			Method m = Class.forName("com.sun.tools.javac.util.DefaultFileManager").getDeclaredMethod("preRegister", Context.class);
174 			m.invoke(null, c);
175 		} catch (Throwable t) {
176 			// intentional do nothing
177 		}
178 		try {
179 			Method m = Class.forName("com.sun.tools.javac.util.JavacFileManager").getDeclaredMethod("preRegister", Context.class);
180 			m.invoke(null, c);
181 		} catch (Throwable t) {
182 			// intentional do nothing
183 		}
184 		// DefaultFileManager.preRegister(c);
185 		// JavacFileManager.preRegister(c);
186 		return c;
187 	}

207 	private JCTree toTree(Node node) {
208 		if (node == null) return null;
209 		JcTreeBuilder builder = create();
210 		node.accept(builder.visitor);
211 		try {
212 			return builder.get();
213 		} catch (RuntimeException e) {
214 			System.err.printf("Node '%s' (%s) did not produce any results\n", node, node.getClass().getSimpleName());
215 			throw e;
216 		}
217 	}

227 	private <T extends JCTree> List<T> toList(Class<T> type, StrictListAccessor<?, ?> accessor) {
228 		List<T> result = List.nil();
229 		for (Node node : accessor) {
230 			JcTreeBuilder builder = create();
231 			node.accept(builder.visitor);
232 			
233 			List<? extends JCTree> values;
234 			
235 			try {
236 				values = builder.getAll();
237 				if (values.size() == 0) throw new RuntimeException();
238 			} catch (RuntimeException e) {
239 				System.err.printf("Node '%s' (%s) did not produce any results\n", node, node.getClass().getSimpleName());
240 				throw e;
241 			}
242 			
243 			for (JCTree value : values) {
244 				if (value != null && !type.isInstance(value)) {
245 					throw new ClassCastException(value.getClass().getName() + " cannot be cast to " + type.getName());
246 				}
247 				result = result.append(type.cast(value));
248 			}
249 		}
250 		return result;
251 	}

253 	private <T extends JCTree> List<T> toList(Class<T> type, Node node) {
254 		if (node == null) return List.nil();
255 		JcTreeBuilder builder = create();
256 		node.accept(builder.visitor);
257 		@SuppressWarnings("unchecked")
258 		List<T> all = (List<T>)builder.getAll();
259 		return List.<T>nil().appendList(all);
260 	}

277 	private boolean set(Node node, JCTree value) {
278 		if (result != null) throw new IllegalStateException("result is already set");
279 		JCTree actualValue = value;
280 		if (node instanceof Expression) {
281 			for (int i = 0; i < ((Expression)node).getIntendedParens(); i++) {
282 				actualValue = treeMaker.Parens((JCExpression)actualValue);
283 				posParen(node, i, ((Expression)node).astParensPositions(), actualValue);
284 			}
285 		}
286 		result = List.of(actualValue);
287 		return true;
288 	}

290 	private void posParen(Node node, int iteration, java.util.List<Position> parenPositions, JCTree jcTree) {
291 		Position p = null;
292 		if (parenPositions.size() > iteration) p = parenPositions.get(iteration);
293 		int start = (p == null || p.isUnplaced() || p.getStart() < 0) ? node.getPosition().getStart() - 1 - iteration : p.getStart();
294 		int end = (p == null || p.isUnplaced() || p.getEnd() < 0) ? node.getPosition().getEnd() + 1 + iteration : p.getEnd();
295 		setPos(start, end, jcTree);
296 	}

298 	private boolean set(List<? extends JCTree> values) {
299 		if (result != null) throw new IllegalStateException("result is already set");
300 		result = values;
301 		return true;
302 	}

314 		public boolean visitNode(Node node) {

316 		}

318 		@Override
319 		public boolean visitCompilationUnit(CompilationUnit node) {
320 			List<JCTree> preamble = toList(JCTree.class, node.astPackageDeclaration());
321 			List<JCTree> imports = toList(JCTree.class, node.astImportDeclarations());
322 			List<JCTree> types = toList(JCTree.class, node.astTypeDeclarations());
323 			
324 			List<JCAnnotation> annotations = List.nil();
325 			JCExpression pid = null;
326 			
327 			for (JCTree elem : preamble) {
328 				if (elem instanceof JCAnnotation) {
329 					annotations = annotations.append((JCAnnotation)elem);
330 				} else if (elem instanceof JCExpression && pid == null) {
331 					pid = (JCExpression) elem;
332 				} else {
333 					throw new RuntimeException("Unexpected element in preamble: " + elem);
334 				}
335 			}
336 			
337 			JCCompilationUnit topLevel = treeMaker.TopLevel(annotations, pid, imports.appendList(types));
338 			topLevel.endPositions = endPosTable;
339 			if (hasConversionStructureInfo(node, "converted")) return posSet(node, topLevel);
340 			
341 			int start = Integer.MAX_VALUE;
342 			int end = node.getPosition().getEnd();
343 			if (node.astPackageDeclaration() != null) start = Math.min(start, node.astPackageDeclaration().getPosition().getStart());
344 			if (!node.astImportDeclarations().isEmpty()) start = Math.min(start, node.rawImportDeclarations().first().getPosition().getStart());
345 			if (!node.astTypeDeclarations().isEmpty()) start = Math.min(start, node.rawTypeDeclarations().first().getPosition().getStart());
346 			if (start == Integer.MAX_VALUE) start = node.getPosition().getStart();
347 			return set(node, setPos(start, end, topLevel));
348 		}

350 		@Override
351 		public boolean visitPackageDeclaration(PackageDeclaration node) {
352 			List<JCTree> defs = List.nil();
353 			
354 			for (Annotation annotation : node.astAnnotations()) {
355 				defs = defs.append(toTree(annotation));
356 			}
357 			
358 			//Actual package declaration
359 			defs = defs.append(chain(node.astParts()));
360 			
361 			set(defs);
362 			return true;
363 		}

365 		@Override
366 		public boolean visitImportDeclaration(ImportDeclaration node) {
367 			JCExpression name = chain(node.astParts());
368 			if (node.astStarImport()) {
369 				int start, end;
370 				Position jcDotStarPos = getConversionPositionInfo(node, ".*");
371 				start = jcDotStarPos == null ? posOfStructure(node, ".", true) : jcDotStarPos.getStart();
372 				end = jcDotStarPos == null ? posOfStructure(node, "*", false) : jcDotStarPos.getEnd();
373 				name = setPos(start, end, treeMaker.Select(name, table.asterisk));
374 			}
375 			return posSet(node, treeMaker.Import(name, node.astStaticImport()));
376 		}

437 		private static final long ENUM_CONSTANT_FLAGS = Flags.PUBLIC | Flags.STATIC | Flags.FINAL | Flags.ENUM;
438 		
439 		@Override
440 		public boolean visitEnumConstant(EnumConstant node) {
441 			JCIdent parentType1 = treeMaker.Ident(toName(((EnumDeclaration)node.getParent().getParent()).astName()));
442 			JCIdent parentType2 = treeMaker.Ident(toName(((EnumDeclaration)node.getParent().getParent()).astName()));
443 			JCClassDecl body = (JCClassDecl) toTree(node.astBody());
444 			if (body != null) body.mods.flags |= Flags.STATIC | Flags.ENUM;
445 			JCNewClass newClass = treeMaker.NewClass(
446 					null, 
447 					List.<JCExpression>nil(),
448 					parentType1,
449 					toList(JCExpression.class, node.astArguments()),
450 					body
451 			);
452 			
453 			int start, end;
454 			Position jcNewClassPos = getConversionPositionInfo(node, "newClass");
455 			start = jcNewClassPos == null ? posOfStructure(node, "(", true) : jcNewClassPos.getStart();
456 			end = jcNewClassPos == null ? (body != null ? node.getPosition().getEnd() : posOfStructure(node, ")", false)) : jcNewClassPos.getEnd();
457 			boolean posIsSet = false;
458 			
459 			if (jcNewClassPos == null && start > node.astName().getPosition().getStart()) {
460 				setPos(start, end, newClass);
461 				posIsSet = true;
462 			}
463 			
464 			if (jcNewClassPos != null && start != node.getPosition().getStart()) {
465 				setPos(start, end, newClass);
466 				posIsSet = true;
467 			}
468 			
469 			if (body != null) body.pos = node.getPosition().getStart();
470 			if (!posIsSet && body != null) setPos(node.astBody(), newClass);
471 			
472 			JCModifiers mods = treeMaker.Modifiers(ENUM_CONSTANT_FLAGS, toList(JCAnnotation.class, node.astAnnotations()));
473 			if (!node.astAnnotations().isEmpty()) {
474 				int modStart = node.astAnnotations().first().getPosition().getStart();
475 				int modEnd = node.astAnnotations().last().getPosition().getEnd();
476 				setPos(modStart, modEnd, mods);
477 			}
478 			return posSet(node, treeMaker.VarDef(mods, toName(node.astName()), parentType2, newClass));
479 		}

719 		private boolean tryStringCombine(BinaryExpression node) {
720 			if (node.getParens() > 0) {
721 				;
722 			} else if (node.getParent() instanceof BinaryExpression) {
723 				try {
724 					if (!((BinaryExpression)node.getParent()).astOperator().isAssignment()) return false;
725 				} catch (AstException ignore) {
726 					return false;
727 				}
728 			} else if (node.getParent() instanceof InstanceOf) {
729 				return false;
730 			}
731 			
732 			java.util.List<String> buffer = Lists.newArrayList();
733 			BinaryExpression current = node;
734 			int start = Integer.MAX_VALUE;
735 			while (true) {
736 				start = Math.min(start, posOfStructure(current, "+", true));
737 				if (current.rawRight() instanceof StringLiteral && current.astRight().getParens() == 0) {
738 					buffer.add(((StringLiteral)current.rawRight()).astValue());
739 				} else {
740 					return false;
741 				}
742 				
743 				if (current.rawLeft() instanceof BinaryExpression) {
744 					current = (BinaryExpression) current.rawLeft();
745 					try {
746 						if (current.astOperator() != BinaryOperator.PLUS || current.getParens() > 0) return false;
747 					} catch (AstException e) {
748 						return false;
749 					}
750 				} else if (current.rawLeft() instanceof StringLiteral && current.astLeft().getParens() == 0) {
751 					buffer.add(((StringLiteral)current.rawLeft()).astValue());
752 					break;
753 				} else {
754 					return false;
755 				}
756 			}
757 			
758 			StringBuilder out = new StringBuilder();
759 			for (int i = buffer.size() - 1; i >= 0; i--) out.append(buffer.get(i));
760 			int end = node.getPosition().getEnd();
761 			return set(node, setPos(start, end, treeMaker.Literal(TypeTags.CLASS, out.toString())));
762 		}
