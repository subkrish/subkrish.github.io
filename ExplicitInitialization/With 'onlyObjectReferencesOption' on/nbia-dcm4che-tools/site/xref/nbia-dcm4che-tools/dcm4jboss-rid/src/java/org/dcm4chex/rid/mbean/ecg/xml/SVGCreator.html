119 	public SVGCreator(WaveformGroup wfgrp, WaveformInfo wfInfo, Float width, Float height) {
120 		waveForms = wfgrp;
121 		info = wfInfo;
122 		if ( width != null ) {
123 			setPageWidth( width.floatValue() );
124 		} else {
125 			float prefWidth = PIX_PER_SEC / wfgrp.getSampleFreq() * (float) wfgrp.getNrOfSamples(); //default 25mm/sec
126 			prefWidth *= 2;//two columns!
127 			prefWidth += 110;//space for calPulse
128 			if ( log.isDebugEnabled() ) log.debug("calculated width:"+prefWidth );
129 			if ( prefWidth > this.viewBoxWidth ) { //if prefWidth < default -> dont change default width!
130 				setPageWidth( prefWidth / 100 );//in cm
131 			}
132 		}
133 		if ( height != null ) {
134 			this.height = cmFormatter.format( height );
135 			viewBoxHeight = height.floatValue()*100;
136 			graphicHeightCm = height.intValue();//round to cm
137 			graphicHeight = graphicHeightCm*100;
138 		}
139 	}

239     private void toXML() throws SAXException {
240 		if ( log.isDebugEnabled() ) log.debug("toXML called!");
241         util = new XMLUtil( th, nameSpace );
242         addSVGStart();
243         	addTitleAndDesc();
244         	if ( info != null ) {
245         		addTextSegment();
246 	        	addHeaderSeparator();
247 	        	addShortmeasSegment();
248 	        	addSeveritySegment();
249 	        	addInterpContinue();
250 	        	addLeftstatementSegment();
251 	        	addRightstatementSegment();
252 	        	addInterpSeparator();
253         	}
254     		wfTemplate = WaveformTemplateFactory.getInstance( waveForms, (graphicWidthCm-1)*100, (graphicHeightCm-1)*100 );
255         	addFooter();
256         	addG("scaling", "translate("+this.graphicXOffset*scaling+","+(this.graphicYOffset+100)*scaling+"), scale("+scaling +")",null,null,null);
257         	addDefs();
258         	addGrid( graphicWidthCm, graphicHeightCm );
259     		if ( log.isDebugEnabled() ) log.debug("grid added!");
260     		addGraphic( graphicWidthCm, graphicHeightCm );
261     		if ( log.isDebugEnabled() ) log.debug("graph added!");
262         	util.endElement("g");
263         util.endElement( "svg" );
264

422 	private void addWaveform(WaveformArea area) throws SAXException {
423 		int lead=area.getWaveformIndex();
424 		WaveFormChannel channel = waveForms.getChannel( lead );
425 		WaveformScalingInfo scalingInfo = prepareScalingInfo( channel, area );
426 		float topPos=area.getTopY();
427 		float baseLineY = topPos + area.getHeight()*scalingInfo.getZeroLine();
428 		if ( log.isDebugEnabled() ) log.debug("topPos:"+topPos+", baseLineY:"+baseLineY+", scalingInfo:"+scalingInfo);
429 		channel.reset();
430 		addG( "lead"+lead, "translate("+(graphicXOffset+area.getLeftX())+","+baseLineY+")",null, null, null );
431 			addPath( "lead"+lead, "fill:none;stroke:"+WF_COLORS[lead%WF_COLORS.length], "5", getWaveFormString( channel, area, scalingInfo ));
432 		util.endElement("g");
433 		addG( "lead"+lead, "translate("+(graphicXOffset+area.getLeftX())+","+(topPos + area.getHeight()*0.5)+")",null, null, null );
434 			addText( "0", "-100", "30", "green", null, channel.getChSource());
435 		    addPath( "waveseparator", "fill:none;stroke:green", "5" ,"M 0 -90 L 0 -20 M 0 90 L 0 20 ");
436 		    if ( scalingInfo.getYScaleDesc() != null ) {
437 		    	addText( "0", "-150", "30", "green", null, scalingInfo.getYScaleDesc());
438 		    }
439 		    if ( scalingInfo.getXScaleDesc() != null ) {
440 		    	addText( Float.toString(area.getWidth()-300), Float.toString( topPos+area.getHeight()-150), "30", "green", null, scalingInfo.getXScaleDesc());
441 		    }
442 	   util.endElement("g");
443 	}

445 	/**
446 	 * @param channel
447 	 * @param scalingInfo
448 	 * @return
449 	 */
450 	private String getWaveFormString(WaveFormChannel channel, WaveformArea area, WaveformScalingInfo scalingInfo ) {
451 		float width = area.getWidth();
452 		StringBuffer sb = new StringBuffer();
453 		float xDelta = scalingInfo.getPixPerSec() / waveForms.getSampleFreq();
454 		float yDelta = scalingInfo.getPixPerUnit() * -1;
455 		
456 		int len = waveForms.getNrOfSamples();
457 		if ( log.isDebugEnabled() ) log.debug("NrOfSamples:"+len);
458 		if ( len * xDelta > width ) {
459 			if ( log.isDebugEnabled() ) log.debug("correction: (len*xdelta):"+(len*xDelta)+">"+width);
460 			len = new Float( width / xDelta).intValue();
461 			if ( log.isDebugEnabled() ) log.debug("xDelta:"+xDelta+" --> width/xDelta:"+( width / xDelta)+" len:"+len);
462 		}
463 		sb.append("M 0 0 L ");
464 		float currX = 0f;
465 		for ( int i = 0 ; i < len ; i++ ) {
466 			sb.append( currX ).append( " " ).append( channel.getValue() * yDelta ).append(" ");
467 			currX += xDelta;
468 		}
469 		return sb.toString();
470 	}

472 	/**
473 	 * @param channel
474 	 * @param scalingInfo
475 	 * @return
476 	 */
477 	private WaveformScalingInfo prepareScalingInfo(WaveFormChannel channel, WaveformArea area) {
478 		float pixPerSec = -1;
479 		float pixPerUnit = -1f;
480 		float zeroLine = 0.5f;
481 		String xScaleDesc = null;
482 		String yScaleDesc = null;
483 		WaveformScalingInfo scalingInfo = area.getScalingInfo();
484 		if ( scalingInfo != null ) {
485 			if ( (pixPerSec = scalingInfo.getPixPerSec()) > 0 && (pixPerUnit = scalingInfo.getPixPerUnit()) > 0 ) {
486 				channel.applyAreaScaling(scalingInfo.getYScaleDesc());
487 				return scalingInfo;
488 			}
489 			xScaleDesc = scalingInfo.getXScaleDesc();
490 			yScaleDesc = scalingInfo.getYScaleDesc();
491 		}
492 		//OK! We need scaling information
493 		if ( pixPerSec <= 0f ) {
494 			int nrOfSamples = channel.getWaveformGroup().getNrOfSamples();
495 			pixPerSec = area.getWidth() / nrOfSamples * channel.getWaveformGroup().getSampleFreq();
496 			xScaleDesc = (pixPerSec/10)+"mm/sec";
497 		}
498 		if ( pixPerUnit <= 0f ) {
499 			Integer min = channel.getMinValue();
500 			Integer max = channel.getMaxValue();
501 			float[] minmax;
502 			if ( min != null && max != null ) {
503 				float sens = channel.getSensitivity();
504 				minmax = new float[] { min.floatValue()*sens, 
505 									   max.floatValue()*sens };
506 			} else {
507 				minmax = channel.calcMinMax(0,channel.getWaveformGroup().getNrOfSamples());
508 			}
509 			float unitRange = minmax[1]-minmax[0];
510 			if ( unitRange == 0 ) unitRange = 1;
511 			pixPerUnit = area.getHeight() / unitRange;
512 			yScaleDesc = channel.getUnit()+" ("+(pixPerUnit/10)+"mm/"+channel.getUnit()+")";
513 			if ( minmax[0] > 0 && minmax[1] > 0 ) { 
514 				zeroLine = minmax[1] / unitRange;
515 			} else if ( minmax[0] < 0 && minmax[1] < 0) {
516 				zeroLine = minmax[0] / unitRange;
517 			} else {
518 				zeroLine = (unitRange + minmax[0])/unitRange;// unitrange - abs(min)
519 			}
520 			
521 		}
522 		scalingInfo = new WaveformScalingInfo(pixPerSec,xScaleDesc,pixPerUnit,yScaleDesc);
523 		scalingInfo.setZeroLine( zeroLine );
524 		return scalingInfo;
525 	}
