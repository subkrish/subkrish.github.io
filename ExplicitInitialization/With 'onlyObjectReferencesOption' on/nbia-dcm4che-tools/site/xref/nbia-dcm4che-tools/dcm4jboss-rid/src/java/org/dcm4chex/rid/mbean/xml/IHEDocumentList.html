123     public boolean add( Dataset ds ) {
124         if ( iuids.contains(ds.getString(Tags.SOPInstanceUID))) {
125             return false;
126         }
127         Date date = ds.getDateTime( Tags.ContentDate, Tags.ContentTime );
128         if (date != null) {
129             long ms = date.getTime();
130             if (ms < lowerDateTime || ms > upperDateTime) {
131                 return false;
132             }
133         }
134         iuids.add(ds.getString(Tags.SOPInstanceUID));
135         return datasets.add(ds);
136     }

142     private void applyMostRecentResults() {
143         Collections.sort( datasets, new DatasetDateComparator() );
144         if ( mostRecentResults > 0 && datasets.size() > mostRecentResults ) {
145             datasets.subList( mostRecentResults, datasets.size() ).clear();//Remains mostRecentResults items in list; removes all older dataset
146         }
147     }

284     public void toXML( OutputStream out ) throws TransformerConfigurationException, SAXException {
285         SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
286 
287         applyMostRecentResults();//sorts the list and ( if mostRecentResults > 0 ) shrink the list.
288 
289         if (xslt != null) {
290             try {
291                 th = tf.newTransformerHandler(new StreamSource(xslt.openStream(),
292                         xslt.toExternalForm()));
293             } catch ( IOException x ) {
294                 log.error("Cant open xsl file:"+xslt, x );
295             }
296         } else {
297             th = tf.newTransformerHandler();
298             th.getTransformer().setOutputProperty(OutputKeys.INDENT, "yes");
299         }
300         th.setResult( new StreamResult(out) );
301         th.startDocument();
302         if ( xslFile != null ) {
303             th.processingInstruction("xml-stylesheet", "href='"+xslFile+"' type='text/xsl'");
304         }
305 
306         toXML();
307 
308         th.endDocument();
309         try {
310             out.flush();
311             out.close();
312         } catch (Exception e) {
313             // TODO Auto-generated catch block
314             log.error("Could not close stream.", e);
315         }
316     }
317     private void toXML() throws SAXException {
318         util = new XMLUtil( th );
319 
320         util.startElement("IHEDocumentList", EMPTY_ATTRIBUTES );
321         addDocCode( );
322         addActivityTime( );
323         Dataset ds;
324         if ( datasets.size() > 0 ) {
325             ds = (Dataset) datasets.get( 0 );
326         } else {
327             if ( queryDS != null ) {
328                 ds = queryDS;
329             } else {
330                 ds = DcmObjectFactory.getInstance().newDataset();
331                 ds.putLO(Tags.PatientID, "");
332             }
333         }
334         addRecordTarget( ds );
335         addAuthor();
336         addDocuments();
337         util.endElement("IHEDocumentList");
338 
339 
340     }

403     private void addPatientPatient( Dataset ds ) throws SAXException {
404         String familyName = "";
405         String givenName = "";
406         String genderCode = "121103"; // Code Value for Patient's Sex 'O' (Undetermined sex)
407         String birthDate = "";
408         try {
409             PersonName pn = ds.getPersonName(Tags.PatientName );
410             if ( pn != null ) {
411                 familyName = pn.get( PersonName.FAMILY );
412                 givenName = pn.get( PersonName.GIVEN );
413                 if ( givenName == null ) givenName = "";
414             }
415             String s = ds.getString( Tags.PatientSex );
416             if ( "M".equals(s) || "F".equals(s) ) genderCode = s;
417             Date date = ds.getDate( Tags.PatientBirthDate  );
418             if ( date != null )
419                 birthDate = DATE_FORMATTER.format( date );
420 
421         } catch ( Exception x ) {
422             log.warn("Exception getting person informations:", x);
423         }
424         util.startElement("patientPatient", EMPTY_ATTRIBUTES );
425         //Names
426         util.startElement("name", EMPTY_ATTRIBUTES );
427         util.startElement("family", EMPTY_ATTRIBUTES );
428         th.characters(familyName.toCharArray(),0,familyName.length());
429         util.endElement("family" );       
430         util.startElement("given", EMPTY_ATTRIBUTES );
431         th.characters(givenName.toCharArray(),0,givenName.length());
432         util.endElement("given" );       
433         util.endElement("name" );
434         //genderCode
435         AttributesImpl attr = new AttributesImpl();
436         util.addAttribute( attr, "code", genderCode );
437         util.addAttribute( attr, "codeSystem", "1.2.840.10008.2.16.4" );//??
438         util.startElement("administrativeGenderCode", attr );
439         util.endElement("administrativeGenderCode" );
440         //birth
441         AttributesImpl attrBirth = new AttributesImpl();
442         util.addAttribute( attrBirth, "value", birthDate );
443         util.startElement("birthTime", attrBirth );
444         util.endElement("birthTime" );
445         util.endElement("patientPatient" );
446     }

487     private void addComponent( Dataset ds ) throws SAXException {
488         String uid = ds.getString( Tags.SOPInstanceUID );
489         if ( uid == null ) uid = "---";
490         Date date = ds.getDateTime( Tags.ContentDate, Tags.ContentTime );
491         if ( date == null ) {
492             date = ds.getDate( Tags.AcquisitionDatetime );
493         }
494         String acquisTime = date == null ? "" : DATETIME_FORMATTER.format( date );
495         String title = "DocumentTitle";
496         String mime = ds.getString(Tags.MIMETypeOfEncapsulatedDocument,"application/pdf");
497         String link = docRIDUrl+"/IHERetrieveDocument?requestType=DOCUMENT&documentUID="+
498         uid + "&preferredContentType="+mime;
499         util.startElement("component", EMPTY_ATTRIBUTES );
500         util.startElement("documentInformation", EMPTY_ATTRIBUTES );
501         //id
502         AttributesImpl attrID = new AttributesImpl();
503         util.addAttribute( attrID, "root", uid );
504         util.startElement("id", attrID );
505         util.endElement("id" );
506         //component code (SUMMARY, SUMMARY_RADIOLOGY,..)
507         addComponentCode( ds );        
508         //title
509         util.startElement("title", EMPTY_ATTRIBUTES );
510         th.characters(title.toCharArray(), 0, title.length() );
511         util.endElement("title" );
512         //text
513         util.startElement("text", EMPTY_ATTRIBUTES );
514         AttributesImpl attrTxt = new AttributesImpl();
515         util.addAttribute( attrTxt, "value", link );
516         util.startElement("reference", attrTxt );
517         util.endElement("reference" );
518         util.endElement("text" );
519         //statusCode 
520         addComponentStatusCode( ds );
521         //effective	time
522         AttributesImpl attrEff = new AttributesImpl();
523         util.addAttribute( attrEff, "value", acquisTime );
524         util.startElement("effectiveTime", attrEff );
525         util.endElement("effectiveTime" );
526 
527         util.endElement("documentInformation" );
528         util.endElement("component" );
529 
530     }

551     private void addComponentCode(Dataset ds) throws SAXException {
552         //code SR: aus (0008,1032)	ProcedureCodeSequence Code ?; ECG: from SOP Class UID
553         String code = "";
554         String codeSystem = "";
555         String displayname = "";
556         String cuid = ds.getString( Tags.SOPClassUID );
557         if ( "SR".equals(ds.getString( Tags.Modality ) ) || ( this.docSopCuids != null && docSopCuids.contains(cuid)) ) {
558             DcmElement elem = ds.get( Tags.ConceptNameCodeSeq );
559             if ( elem != null && elem.countItems() > 0 ) {
560                 Dataset ds1 = elem.getItem(0);
561                 code = ds1.getString(Tags.CodeValue);
562                 codeSystem = ds1.getString(Tags.CodingSchemeDesignator);
563                 displayname = ds1.getString(Tags.CodeMeaning);
564             } else {
565                 displayname = ds.getString( Tags.DocumentTitle );
566             }
567             if ( displayname == null ) displayname = ds.getString( Tags.StudyDescription );
568         } else { //ECG
569             if ( UIDs.TwelveLeadECGWaveformStorage.equals( cuid ) )
570                 displayname = "12-lead ECG";
571             else if ( UIDs.GeneralECGWaveformStorage.equals( cuid ) )
572                 displayname = "General ECG";
573             else if ( UIDs.AmbulatoryECGWaveformStorage.equals( cuid ) )
574                 displayname = "Ambulatory ECG";
575             else if ( UIDs.HemodynamicWaveformStorage.equals( cuid ) )
576                 displayname = "Hemodynamic";
577             else if ( UIDs.CardiacElectrophysiologyWaveformStorage.equals( cuid ) )
578                 displayname = "Cardiac Electrophysiology";
579         }
580         AttributesImpl attrCode = new AttributesImpl();
581         util.addAttribute( attrCode, "code", code );
582         util.addAttribute( attrCode, "codeSystem", codeSystem );
583         util.addAttribute( attrCode, "displayName", displayname );
584         util.startElement("code", attrCode );
585         util.endElement("code" );
586     }

595         /**
596          * Compares the modification time of two File objects.
597          * <p>
598          * Compares its two arguments for order. Returns a negative integer, zero, or a positive integer 
599          * as the first argument is less than, equal to, or greater than the second.
600          * <p>
601          * Throws an Exception if one of the arguments is null or not a Dataset object.
602          *  
603          * @param arg0 	First argument
604          * @param arg1	Second argument
605          * 
606          * @return <0 if arg0<arg1, 0 if equal and >0 if arg0>arg1
607          */
608         public int compare( Dataset arg0, Dataset arg1 ) {
609             Date d1 = getDateFromDS( (Dataset) arg0 );
610             if ( d1 == null) return 1;
611             Date d2 = getDateFromDS( (Dataset) arg1 );
612             return d2 == null ? -1 : d2.compareTo( d1 );
613         }
