75          super( request );

85      }

126     /* (non-Javadoc)
127      * @see org.dcm4chex.rid.common.BasicRequestObject#checkRequest()
128      */
129     public int checkRequest() {
130         if ( getRequestType() == null || patientID == null || mostRecentResults == null ) 
131             return RIDRequestObject.INVALID_RID_URL; //required param missing!
132 
133         if ( ! getRequestType().startsWith( "SUMMARY") && ! getRequestType().startsWith( "LIST") )
134             return RIDRequestObject.INVALID_RID_URL;//RID Information requestmust start either with SUMMARY or LIST
135 
136         if ( ! getSupportedInformationTypes().contains( getRequestType() ) )
137             return RIDRequestObject.RID_REQUEST_NOT_SUPPORTED;
138 
139         try {
140             Integer.parseInt( this.mostRecentResults );
141         } catch ( Exception x ) {
142             return RIDRequestObject.INVALID_RID_URL;//not an integer string
143         }
144 
145         if ( this.lowerDateTime != null ) {
146             if ( lowerDateTime.trim().length() > 0 ) {
147                 try {
148                     new ISO8601DateFormat().parse( lowerDateTime );			
149                 } catch ( Exception x ) {
150                     return RIDRequestObject.INVALID_RID_URL;//invalid date/time string
151                 }
152             } else {
153                 lowerDateTime = null;
154             }
155         }
156 
157         if ( this.upperDateTime != null ) {
158             if ( upperDateTime.trim().length() > 0 ) {
159                 try {
160                     new ISO8601DateFormat().parse( upperDateTime );			
161                 } catch ( Exception x ) {
162                     return RIDRequestObject.INVALID_RID_URL;//invalid date/time string
163                 }
164             } else {
165                 upperDateTime = null;
166             }
167         }
168 
169         return RIDRequestObject.SUMMERY_INFO;
170     }
171     /**
172      * @return
173      */
174     public List getSupportedInformationTypes() {
175         if ( supportedInfoTypes == null ) {
176             supportedInfoTypes = new ArrayList();
177             Field[] fields = RIDSupport.class.getFields();
178             for ( Field f : fields ) {
179                 if ( f.getName().startsWith("SUMMARY") ) {
180                     try {
181                         supportedInfoTypes.add( f.get(null));
182                         if (log.isDebugEnabled()) log.debug("----- "+f.get(null)+" added to SupportedInformationTypes!");
183                     } catch (Exception e) {
184                         if (log.isDebugEnabled()) log.error("Can not add "+f.getName()+" to SupportedInformationTypes!", e);
185                     }
186                 }
187             }
188 
189         }
190         return supportedInfoTypes;
191     }
