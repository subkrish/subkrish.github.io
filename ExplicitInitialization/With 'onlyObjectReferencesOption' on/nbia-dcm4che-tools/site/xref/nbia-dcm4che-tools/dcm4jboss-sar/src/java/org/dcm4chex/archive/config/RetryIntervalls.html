43  import java.util.HashMap;
44  import java.util.StringTokenizer;

52  public final class RetryIntervalls {
53  
54      private static final long MS_PER_MIN = 60000L;
55  
56      private static final long MS_PER_HOUR = 60 * MS_PER_MIN;
57  
58      private static final long MS_PER_DAY = 24 * MS_PER_HOUR;
59  
60      private static final long MS_PER_WEEK = 7 * MS_PER_DAY;
61      
62      private static final String NEWLINE = System.getProperty("line.separator", "\n");
63      
64      private final int[] counts;
65  
66      private final long[] intervalls;
67  
68      public RetryIntervalls() {
69          counts = new int[0];
70          intervalls = new long[0];
71      }
72  
73      public RetryIntervalls(String text) {
74          if (text.trim().equalsIgnoreCase("NEVER")) {
75              counts = new int[0];
76              intervalls = new long[0];
77          } else
78              try {
79                  StringTokenizer stk = new StringTokenizer(text, ", \t\n\r");
80                  counts = new int[stk.countTokens()];
81                  intervalls = new long[counts.length];
82                  for (int i = 0; i < counts.length; i++) {
83                      init(i, stk.nextToken());
84                  }
85              } catch (IllegalArgumentException e) {
86                  throw new IllegalArgumentException(text);
87              }
88      }
89  
90      private void init(int i, String item) {
91          final int x = item.indexOf('x');
92          intervalls[i] = parseInterval(x != -1 ? item.substring(x + 1)
93                  : item);
94          counts[i] = x != -1 ? Math.max(1, Integer
95                  .parseInt(item.substring(0, x))) : 1;
96      }
97  
98      public String toString() {
99          if (counts.length == 0) { return "NEVER"; }
100         StringBuffer sb = new StringBuffer();
101         for (int i = 0; i < counts.length; i++) {
102             sb.append(counts[i]).append('x');
103             formatInterval(intervalls[i], sb).append(',');
104         }
105         sb.setLength(sb.length() - 1);
106         return sb.toString();
107     }
108     
109     public static String formatIntervalZeroAsNever(long ms) {
110         return ms == 0L ? "NEVER" : formatInterval(ms);
111     }
112 
113     public static String formatInterval(long ms) {
114         return formatInterval(ms, new StringBuffer()).toString();
115     }
116 
117     private static StringBuffer formatInterval(long ms, StringBuffer sb) {
118         if (ms % MS_PER_WEEK == 0)
119             sb.append(ms / MS_PER_WEEK).append("w");
120         else if (ms % MS_PER_DAY == 0)
121             sb.append(ms / MS_PER_DAY).append("d");
122         else if (ms % MS_PER_HOUR == 0)
123             sb.append(ms / MS_PER_HOUR).append("h");
124         else if (ms % MS_PER_MIN == 0)
125             sb.append(ms / MS_PER_MIN).append("m");
126         else
127             sb.append(ms / 1000).append("s");
128         return sb;
129     }
130 
131     public static long parseIntervalOrNever(String text) {
132         return ("NEVER".equalsIgnoreCase(text)) ? 0L
133                 : parseInterval(text);
134     }
135     
136     public static long parseInterval(String text) {
137         int len = text.length();
138         long ms = Long.parseLong(text.substring(0, len - 1));
139         switch (text.charAt(len - 1)) {
140         case 'w':
141             ms *= 7;
142         case 'd':
143             ms *= 24;
144         case 'h':
145             ms *= 60;
146         case 'm':
147             ms *= 60;
148         case 's':
149             ms *= 1000;
150             break;
151             default:
152                 throw new IllegalArgumentException("interval: " + text);
153         }
154         return ms;
155     }
156 
157     public long getIntervall(int failureCount) {
158         int countDown = failureCount;
159         for (int i = 0; i < counts.length; ++i) {
160             if ((countDown -= counts[i]) <= 0) { return intervalls[i]; }
161         }
162         return -1L;
163     }
164 
165     public static final class Map {
166 
167         private final ArrayList<String> keys = new ArrayList<String>();
168         private final ArrayList<RetryIntervalls> values =
169                 new ArrayList<RetryIntervalls>();
170         private RetryIntervalls defval = null;
171 
172         public Map(String s) {
173             StringTokenizer st = new StringTokenizer(s,";\r\n");
174             String token;
175             int pos;
176             while (st.hasMoreTokens()) {
177                 token = st.nextToken().trim();
178                 pos = token.indexOf(']');
179                 if ( pos > 0 ) {
180                     keys.add(token.substring(1,pos));
181                     values.add(new RetryIntervalls(token.substring(pos+1)));
182                 } else {
183                     defval = new RetryIntervalls(token);
184                 }
185             }
186             keys.trimToSize();
187             values.trimToSize();
188         }
189 
190         public String toString() {
191             StringBuilder sb = new StringBuilder();
192             int size = keys.size();
193             for (int i = 0; i < size; i++)
194                 sb.append('[').append(keys.get(i)).append(']')
195                         .append(values.get(i)).append(NEWLINE);
196             if (defval != null)
197                 sb.append(defval).append(NEWLINE);
198             return sb.toString();
199         }
200 
201         public long getIntervall(String key, int failureCount) {
202             int index = keys.indexOf(key);
203             RetryIntervalls val = index >= 0 ? values.get(index) : defval;
204             return val != null ? val.getIntervall(failureCount) : -1L;
205         }
206     }
