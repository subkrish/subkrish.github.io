191     private final NotificationListener callingAETChangeListener = new NotificationListener() {
192         public void handleNotification(Notification notif, Object handback) {
193             try {
194                 CallingAetChanged userData = (CallingAetChanged) notif
195                         .getUserData();
196                 if (areCalledAETsAffected(userData.getAffectedCalledAETs())) {
197                     String[] newCallingAets = userData.getNewCallingAETs();
198                     String newCallingAETs = newCallingAets == null ? ANY
199                             : newCallingAets.length == 0 ? CONFIGURED_AETS
200                                     : StringUtils
201                                             .toString(newCallingAets, '\\');
202                     log.debug("newCallingAETs:" + newCallingAETs);
203                     server.setAttribute(serviceName, new Attribute(
204                             "CallingAETitles", newCallingAETs));
205                 }
206             } catch (Throwable th) {
207                 log.warn("Failed to process callingAET change notification: ",
208                         th);
209             }
210         }
211 
212         private boolean areCalledAETsAffected(String[] affectedCalledAETs) {
213             if (calledAETs == null)
214                 return true;
215             if (affectedCalledAETs != null) {
216                 for (int i = 0; i < affectedCalledAETs.length; i++) {
217                     for (int j = 0; j < calledAETs.length; j++) {
218                         if (affectedCalledAETs[i].equals(calledAETs[j]))
219                             return true;
220                     }
221                 }
222             }
223             return false;
224         }
225     };

311     public final void setGeneratePatientID(String pattern) {
312         if (pattern.equalsIgnoreCase(NONE)) {
313             this.generatePatientID = null;
314             return;
315         }
316         int pl = pattern.indexOf('#');
317         boolean pnameHash = pl != -1;
318         int pr = pnameHash ? pattern.lastIndexOf('#') : -1;
319         int sl = pattern.indexOf('$');
320         int sr = sl != -1 ? pattern.lastIndexOf('$') : -1;
321         if (!pnameHash && sl == -1) {
322             this.generatePatientID = new String[] { pattern };
323         } else if (pnameHash && sl != -1) {
324             this.generatePatientID = pl < sl 
325                     ? split(pattern, pl, pr, sl, sr)
326                     : split(pattern, sl, sr, pl, pr);
327         } else {
328             this.generatePatientID = pnameHash 
329                     ? split(pattern, pl, pr)
330                     : split(pattern, sl, sr);
331         }
332         pnameHashInGeneratePatientID =  pnameHash;
333     }

335     private static String[] split(String pattern, int l1, int r1) {
336         return new String[] { pattern.substring(0, l1),
337                 pattern.substring(l1, r1 + 1), pattern.substring(r1 + 1), };
338     }

340     private static String[] split(String pattern, int l1, int r1, int l2, int r2) {
341         if (r1 > l2) {
342             throw new IllegalArgumentException(pattern);
343         }
344         return new String[] { pattern.substring(0, l1),
345                 pattern.substring(l1, r1 + 1), pattern.substring(r1 + 1, l2),
346                 pattern.substring(l2, r2 + 1), pattern.substring(r2 + 1) };
347     }

496     protected boolean enableService() {
497         if (dcmHandler == null)
498             return false;
499         boolean changed = false;
500         String[] callingAETs = getCallingAETsForPolicy();
501         AcceptorPolicy policy = dcmHandler.getAcceptorPolicy();
502         for (int i = 0; i < calledAETs.length; ++i) {
503             AcceptorPolicy policy1 = policy
504                     .getPolicyForCalledAET(calledAETs[i]);
505             if (policy1 == null) {
506                 policy1 = AssociationFactory.getInstance().newAcceptorPolicy();
507                 policy1.setCallingAETs(callingAETs);
508                 policy1.setUserIdentityNegotiator(userIdentityNegotiator);
509                 policy.putPolicyForCalledAET(calledAETs[i], policy1);
510                 policy.addCalledAET(calledAETs[i]);
511                 changed = true;
512             } else {
513                 String[] aets = policy1.getCallingAETs();
514                 if (aets.length == 0) {
515                     if (callingAETs != null) {
516                         policy1.setCallingAETs(callingAETs);
517                         changed = true;
518                     }
519                 } else {
520                     if (!haveSameItems(aets, callingAETs)) {
521                         policy1.setCallingAETs(callingAETs);
522                         changed = true;
523                     }
524                 }
525             }
526             policy1.setMaxPDULength(maxPDULength);
527             policy1.setAsyncOpsWindow(maxOpsInvoked, maxOpsPerformed);
528             enablePresContexts(policy1);
529         }
530         return changed;
531     }

533     @SuppressWarnings("unchecked")
534     private String[] getCallingAETsForPolicy() {
535         if (callingAETs == null)
536             return null;
537         if (callingAETs.length != 0)
538             return callingAETs;
539         log.debug("Use 'CONFIGURED_AETS' for list of calling AETs");
540         try {
541             Collection<AEDTO> l = aeMgr().findAll();
542             if (l.size() == 0) {
543                 log.warn("No AETs configured! No calling AET is allowed!");
544                 return callingAETs;
545             }
546             List<String> dicomAEs = new ArrayList<String>(l.size());
547             String aet;
548             for (Iterator<AEDTO> iter = l.iterator(); iter.hasNext();) {
549                 aet = iter.next().getTitle();
550                 if (aet.indexOf('^') == -1) {// filter 'HL7' AETs
551                     dicomAEs.add(aet);
552                 }
553             }
554             log
555                     .debug("Use 'CONFIGURED_AETS'. Current list of configured (dicom) AETs"
556                             + dicomAEs);
557             String[] sa = new String[dicomAEs.size()];
558             return dicomAEs.toArray(sa);
559         } catch (Exception e) {
560             log
561                     .error(
562                             "Failed to query configured AETs! No calling AET is allowed!",
563                             e);
564             return callingAETs;
565         }
566     }

568     // Only check if all items in o1 are also in o2! (and same length)
569     // e.g. {"a","a","d"}, {"a","d","d"} will also return true!
570     private boolean haveSameItems(Object[] o1, Object[] o2) {
571         if (o1 == null || o2 == null || o1.length != o2.length)
572             return false;
573         if (o1.length == 1)
574             return o1[0].equals(o2[0]);
575         iloop: for (int i = 0, len = o1.length; i < len; i++) {
576             for (int j = 0; j < len; j++) {
577                 if (o1[i].equals(o2[j]))
578                     continue iloop;
579             }
580             return false;
581         }
582         return true;
583     }

585     private void disableService() {
586         if (dcmHandler == null)
587             return;
588         AcceptorPolicy policy = dcmHandler.getAcceptorPolicy();
589         for (int i = 0; i < calledAETs.length; ++i) {
590             AcceptorPolicy policy1 = policy
591                     .getPolicyForCalledAET(calledAETs[i]);
592             if (policy1 != null) {
593                 disablePresContexts(policy1);
594                 if (policy1.listPresContext().isEmpty()) {
595                     policy.putPolicyForCalledAET(calledAETs[i], null);
596                     policy.removeCalledAET(calledAETs[i]);
597                 }
598             }
599         }
600     }

624     protected void updateAcceptedSOPClass(Map<String, String> cuidMap,
625             String newval, DcmService scp) {
626         Map<String, String> tmp = parseUIDs(newval);
627         if (cuidMap.keySet().equals(tmp.keySet()))
628             return;
629         disableService();
630         if (scp != null)
631             unbindAll(valuesToStringArray(cuidMap));
632         cuidMap.clear();
633         cuidMap.putAll(tmp);
634         if (scp != null)
635             bindAll(valuesToStringArray(cuidMap), scp);
636         enableService();
637     }

643     protected void bindAll(String[] cuids, DcmService scp) {
644         if (dcmHandler == null)
645             return; // nothing to do!

691     protected String toString(Map<String, String> uids) {
692         if (uids == null || uids.isEmpty())
693             return "";
694         String nl = System.getProperty("line.separator", "\n");
695         StringBuffer sb = new StringBuffer();
696         Iterator<String> iter = uids.keySet().iterator();
697         while (iter.hasNext()) {
698             sb.append(iter.next()).append(nl);
699         }
700         return sb.toString();
701     }

703     protected static Map<String, String> parseUIDs(String uids) {
704         StringTokenizer st = new StringTokenizer(uids, " \t\r\n;");
705         String uid, name;
706         Map<String, String> map = new LinkedHashMap<String, String>();
707         while (st.hasMoreTokens()) {
708             uid = st.nextToken().trim();
709             name = uid;
710 
711             if (isDigit(uid.charAt(0))) {
712                 if (!UIDs.isValid(uid))
713                     throw new IllegalArgumentException("UID " + uid
714                             + " isn't a valid UID!");
715             } else {
716                 uid = UIDs.forName(name);
717             }
718             map.put(name, uid);
719         }
720         return map;
721     }

1259             origAttrItem.putCS(Tags.ReasonForTheAttributeModification, COERCE);

1269         if (generatePatientID == null) {
1270             return;
1271         }
1272         String pid = pat.getString(Tags.PatientID);
1273         if (pid != null) {
1274             return;
1275         }
1276         String pname = pat.getString(Tags.PatientName);
1277         String issuer = issuerOfGeneratedPatientID(calledAET);
1278         if (generatePatientID.length == 1) {
1279             pid = generatePatientID[0];
1280         } else {
1281             String suid = sty != null ? sty.getString(Tags.StudyInstanceUID)
1282                     : null;
1283             int suidHash = suid != null ? suid.hashCode() : ++sequenceInt;
1284             // generate different Patient IDs for different studies
1285             // if no Patient Name
1286             int pnameHash = pname == null ? suidHash : pname.hashCode() * 37
1287                     + pat.getString(Tags.PatientBirthDate, "").hashCode()
1288                     + pat.getString(Tags.PatientSex, "").hashCode();
1289             
1290             pid = generatePatientID(pnameHash, suidHash);
1291             if (pnameHashInGeneratePatientID && pname != null) {
1292                 try {
1293                     Storage s = getStorage();
1294                     while (!pname.equals(
1295                             s.getPatientByIDWithIssuer(pid, issuer)
1296                             .getString(Tags.PatientName)))
1297                         pid = generatePatientID(++pnameHash, suidHash);
1298                 } catch (ObjectNotFoundException e) {
1299                 } catch (Exception e) {
1300                     log.error("Failed to query DB for patient with pid="
1301                             + pid + ", issuer=" + issuer);
1302                     throw new DcmServiceException(Status.ProcessingFailure, e);
1303                 }
1304             }
1305         }
1306         pat.putLO(Tags.PatientID, pid);
1307         pat.putLO(Tags.IssuerOfPatientID, issuer);
1308         if (log.isInfoEnabled())
1309             log.info("Generate Patient ID: " + pid + "^^^" + issuer
1310                     + " for Patient: " + pname);
1311     }

1313     private String generatePatientID(int pnameHash, int suidHash) {
1314         StringBuffer sb = new StringBuffer();
1315         for (int i = 0; i < generatePatientID.length; i++) {
1316             String s = generatePatientID[i];
1317             int l = s.length();
1318             if (l == 0)
1319                 continue;
1320             char c = s.charAt(0);
1321             if (c != '#' && c != '$') {
1322                 sb.append(s);
1323                 continue;
1324             }
1325             String v = Long
1326                     .toString((c == '#' ? pnameHash : suidHash) & 0xffffffffL);
1327             for (int j = v.length() - l; j < 0; j++) {
1328                 sb.append('0');
1329             }
1330             sb.append(v);
1331         }
1332         return sb.toString();
1333     }

1335     private String issuerOfGeneratedPatientID(String calledAET) {
1336         try {
1337             AEDTO ae = aeMgr().findByAET(calledAET);
1338             String issuer = ae.getIssuerOfPatientID();
1339             if (issuer != null)
1340                 return issuer;
1341         } catch (Exception ignore) {}
1342         return issuerOfGeneratedPatientID;
1343     }

1345     public boolean ignorePatientIDForUnscheduled(Dataset ds,
1346             int requestAttrsSeqTag, String callingAET) {
1347         String pid = ds.getString(Tags.PatientID);
1348         Dataset requestAttrs = ds.getItem(requestAttrsSeqTag);
1349         if (pid != null
1350                 && (requestAttrs == null
1351                         || !requestAttrs.containsValue(Tags.SPSID))
1352                 && isGeneratePatientIDForUnscheduledFromAET(callingAET)) {
1353             String issuer = ds.getString(Tags.IssuerOfPatientID);
1354             ds.putLO(Tags.PatientID);
1355             ds.remove(Tags.IssuerOfPatientID);
1356             if (log.isInfoEnabled()) {
1357                 StringBuffer prompt = new StringBuffer("Ignore Patient ID: ");
1358                 prompt.append(pid);
1359                 if (issuer != null) {
1360                     prompt.append("^^^").append(issuer);
1361                 }
1362                 prompt.append(" for Patient: ")
1363                     .append(ds.getString(Tags.PatientName));
1364                 log.info(prompt.toString());
1365             }
1366             return true;
1367         }
1368         return false;
1369     }
