122     private String codes2String(Map codes) {
123         if (codes == null || codes.isEmpty())
124             return "";
125         StringBuffer sb = new StringBuffer();
126         Dataset ds;
127         String design;
128         for (Iterator iter = codes.values().iterator(); iter.hasNext();) {
129             ds = (Dataset) iter.next();
130             design = ds.getString(Tags.CodingSchemeDesignator);
131             sb.append(ds.getString(Tags.CodeValue)).append("^");
132             if (design != null)
133                 sb.append(design).append("^");
134             sb.append(ds.getString(Tags.CodeMeaning)).append(",");
135         }
136 
137         return sb.substring(0, sb.length() - 1);
138     }
139 
140     private Map string2Codes(String codes, String defaultDesign) {
141         StringTokenizer st = new StringTokenizer(codes, ",");
142         Map map = new HashMap();
143         int nrOfTokens;
144         StringTokenizer stCode;
145         Dataset ds;
146         String codeValue;
147         while (st.hasMoreTokens()) {
148             stCode = new StringTokenizer(st.nextToken(), "^");
149             nrOfTokens = stCode.countTokens();
150             if (nrOfTokens < 2) {
151                 throw new IllegalArgumentException(
152                         "Wrong format of human performer configuration! (<codeValue>[^<designator>]^<meaning>)");
153             }
154             ds = dof.newDataset();
155             codeValue = stCode.nextToken();
156             ds.putSH(Tags.CodeValue, codeValue);
157             if (nrOfTokens > 2) {
158                 ds.putSH(Tags.CodingSchemeDesignator, stCode.nextToken());
159             } else if (defaultDesign != null) {
160                 ds.putSH(Tags.CodingSchemeDesignator, defaultDesign);
161             }
162             ds.putLO(Tags.CodeMeaning, stCode.nextToken());
163             map.put(codeValue, ds);
164         }
165         return map;
166     }

168     public List listTemplates() {
169         if (templates == null) {
170             File tmplPath = FileUtils.resolve(templatePath);
171             File[] files = tmplPath.listFiles();
172             templates = new ArrayList();
173             String fn;
174             for (int i = 0; i < files.length; i++) {
175                 fn = files[i].getName();
176                 if (fn.endsWith(".xml")) {
177                     templates.add(fn.substring(0, fn.length() - 4));
178                 }
179             }
180         }
181         log.info("Template List:" + templates);
182         return templates;
183     }
184 
185     public void clearTemplateList() {
186         templates = null;
187     }

189     public void addWorklistItem(Long studyPk, String templateFile,
190             String humanPerformerCode, Long scheduleDate) throws Exception {
191         String uri = FileUtils.resolve(
192                 new File(templatePath, templateFile + ".xml")).toURI()
193                 .toString();
194         if (log.isDebugEnabled())
195             log.debug("load template file: " + uri);
196         Dataset ds = DatasetUtils.fromXML(new InputSource(uri));
197 
198         ContentManager cm = getContentManager();
199         // patient
200         Dataset patDS = cm.getPatientForStudy(studyPk.longValue());
201         if (log.isDebugEnabled()) {
202             log.debug("Patient Dataset:");
203             log.debug(patDS);
204         }
205 
206         ds.putAll(patDS.subSet(PAT_ATTR_TAGS));
207         //
208         Dataset sopInstRef = cm.getSOPInstanceRefMacro(studyPk.longValue(),
209                 false);
210         String studyIUID = sopInstRef.getString(Tags.StudyInstanceUID);
211         ds.putUI(Tags.SOPInstanceUID, UIDGenerator.getInstance().createUID());
212         ds.putUI(Tags.StudyInstanceUID, studyIUID);
213         DcmElement inSq = ds.putSQ(Tags.InputInformationSeq);
214         inSq.addItem(sopInstRef);
215 
216         // Scheduled Human Performer Seq
217         DcmElement schedHPSq = ds.putSQ(Tags.ScheduledHumanPerformersSeq);
218         Dataset item = schedHPSq.addNewItem();
219         DcmElement hpCodeSq = item.putSQ(Tags.HumanPerformerCodeSeq);
220         Dataset dsCode = (Dataset) this.humanPerformer.get(humanPerformerCode);
221         log.info(dsCode);
222         if (dsCode != null) {
223             hpCodeSq.addItem(dsCode);
224             item.putPN(Tags.HumanPerformerName, dsCode
225                     .getString(Tags.CodeMeaning));
226         }
227 
228         // Scheduled Procedure Step Start Date and Time
229         ds.putDT(Tags.SPSStartDateAndTime, new Date(scheduleDate.longValue()));
230 
231         if (log.isDebugEnabled()) {
232             log.debug("GPSPS Dataset:");
233             log.debug(ds);
234         }
235 
236         addWorklistItem(ds);
237     }

239     private void addWorklistItem(Dataset ds) {
240         if (ds == null)
241             return;
242         try {
243             getGPWLManager().addWorklistItem(ds);
244         } catch (Exception e) {
245             log.error("Failed to add Worklist Item:", e);
246         }
247     }
