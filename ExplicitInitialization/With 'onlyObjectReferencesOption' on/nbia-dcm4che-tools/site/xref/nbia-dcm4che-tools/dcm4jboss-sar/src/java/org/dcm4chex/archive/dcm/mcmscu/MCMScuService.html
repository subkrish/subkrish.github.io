631     /**
632      * Collect studies to media for media creation. (offline storage)
633      * <p>
634      * Search for instances that are older than <code>daysBefore</code> and
635      * are not already assigned to a media.<br>
636      * Collect this instances to studies to ensure that all instances of a study
637      * are on one media.<br>
638      * Assign instances to media for best media usage using maxMediaUsage as
639      * limit.
640      * 
641      * @return Number of instances
642      */
643     public int scheduleMedia() {
644         synchronized(this) {
645             if (isRunningScheduleMedia) {
646                 log.info("RunningScheduleMedia is already running!");
647                 return -1;
648             }
649             isRunningScheduleMedia = true;
650         }
651         try {
652             log.info("Check for studies for scheduling on media");
653             long l1 = System.currentTimeMillis();
654             MediaComposer mc = null;
655             try {
656                 mc = this.lookupMediaComposer();
657                 Collection studies = mc.getStudiesReceivedBefore(getSearchDate());
658                 if ( studies.isEmpty() ) {
659                 	log.info("No Studies found for scheduling on media!");
660                 	return 0;
661                 }
662                 List mediaPool = null;
663                 String prefix = getFileSetIdPrefix();
664                 log.info(studies.size() + " are selected for scheduling on media!");
665                 for (Iterator iter = studies.iterator(); iter.hasNext();) {
666                     mediaPool = mc.assignStudyToMedia((StudyLocal) iter.next(),
667                             mediaPool, maxMediaUsage, prefix);
668                 }
669                 log.info("Schedule of " + studies.size()
670                         + " studies completed. Number of collecting media:"
671                         + mediaPool.size() + ". time:"
672                         + (System.currentTimeMillis() - l1) + " ms!");
673                 return studies.size();
674             } catch (Exception x) {
675                 log.error("Can not create MediaComposer!", x);
676                 return -1;
677             }
678         } finally {
679             isRunningScheduleMedia = false;
680         }
681     }

694     /**
695      * Start listening to the JMS queue deined in <code>QUEUE</code>
696      * <p>
697      * This queue is used to receive media creation request from scheduler or
698      * web interface.
699      */
700     protected void startService() throws Exception {
701         scheduleMediaListenerID = scheduler.startScheduler(
702                 timerIDCheckForStudiesToSchedule, scheduleMediaInterval,
703                 scheduleMediaListener);
704         updateMediaStatusListenerID = scheduler.startScheduler(
705                 timerIDCheckMediaStatus, updateMediaStatusInterval,
706                 updateMediaStatusListener);
707         burnMediaListenerID = scheduler.startScheduler(
708                 timerIDCheckForMediaToBurn, burnMediaInterval,
709                 burnMediaListener);
710         jmsDelegate.startListening(queueName, this, concurrency);
711     }
712 
713     /**
714      * Stop listening to the JMS queue deined in <code>QUEUE</code>
715      * 
716      */
717     protected void stopService() throws Exception {
718         scheduler.stopScheduler(timerIDCheckForStudiesToSchedule,
719                 scheduleMediaListenerID, scheduleMediaListener);
720         scheduler.stopScheduler(timerIDCheckMediaStatus,
721                 updateMediaStatusListenerID, updateMediaStatusListener);
722         scheduler.stopScheduler(timerIDCheckForMediaToBurn,
723                 burnMediaListenerID, burnMediaListener);
724         jmsDelegate.stopListening(queueName);
725         super.stopService();
726     }

732     /**
733      * Handles a JMS message.
734      * 
735      * @param message
736      *            The JMS message to handle.
737      */
738     public void onMessage(Message message) {
739         ObjectMessage om = (ObjectMessage) message;
740         MediaDTO mediaDTO = null;
741         try {
742             mediaDTO = (MediaDTO) om.getObject();
743             log.info("Start processing " + mediaDTO);
744             process(mediaDTO);
745         } catch (JMSException e) {
746             handleError(mediaDTO, "jms error during processing message: "
747                     + message, e);
748         } catch (Throwable e) {
749             handleError(mediaDTO,
750                     "unexpected error during processing message: " + message, e);
751         }
752     }

756      * <p>
757      * <DL>
758      * <DD>1) Set media status to PROCESSING.</DD>
759      * <DD>2) Move instances to <code>destAET</code> using
760      * <code>retrieveAET</code>.</DD>
761      * <DD>3) Send media creation request to <code>mcMScpAET</code>. </DD>
762      * <DD>4) Send N-Action command for media creation to
763      * <code>mcMScpAET</code>. </DD>
764      * </DL>
765      * 
766      * @param mediaDTO
767      *            The MediaDTO object for the media to create.
768      * 
769      * @throws RemoteException
770      * @throws FinderException
771      * @throws HomeFactoryException
772      * @throws CreateException
773      */
774     private void process(MediaDTO mediaDTO) throws RemoteException,
775             FinderException, HomeFactoryException, CreateException {
776         this.lookupMediaComposer().setMediaCreationRequestIuid(
777                 mediaDTO.getPk(), null);
778         this.lookupMediaComposer().setMediaStatus(mediaDTO.getPk(),
779                 MediaDTO.TRANSFERING, "");
780         if (processMove(mediaDTO)) {
781             log.info("Move instances of " + mediaDTO.getFilesetId() + " done!");
782             this.lookupMediaComposer().setMediaStatus(mediaDTO.getPk(),
783                     MediaDTO.BURNING, "");
784             if (processMediaCreation(mediaDTO)) {
785                 log.info("Sending media creation request of "
786                         + mediaDTO.getFilesetId() + " done!");
787                 return;
788             }
789         }
790         log.error("processing " + mediaDTO.getFilesetId() + " failed!!!");
791     }

793     /**
794      * Process the move command on study level.
795      * <p>
796      * Move instances of all studies defined for given media to
797      * <code>destAET</code>.<br>
798      * Use <code>retrieveAET</code> to perform the move command.
799      * <p>
800      * Moves all instances of the studies regardless if all instances assigned
801      * to this media.<br>
802      * (The normal case is that all instances of a study is assigned to one
803      * media!)<br>
804      * This has no effect for the creted media because the media creation
805      * request contains only instances assigned to this media!<br>
806      * The unassigned instances are deleted after a timeout by the
807      * <code>mcMScpAET</code>.
808      * 
809      * @param mediaDTO
810      *            The MediaDTO to process.
811      * 
812      * @return true if move was sucessful, false if move failed.
813      */
814     private boolean processMove(MediaDTO mediaDTO) {
815         ActiveAssociation aa;
816         try {
817             aa = openAssociation(retrieveAET,
818                             UIDs.StudyRootQueryRetrieveInformationModelMOVE);
819         } catch (Exception e) {
820                 handleError(mediaDTO, "processMove failed for "
821                         + mediaDTO.getFilesetId(), e);
822                 return false;       
823         }
824         try {
825             Association as = aa.getAssociation();
826             DcmObjectFactory oFact = DcmObjectFactory.getInstance();
827             Command cmd = oFact.newCommand();
828             String[] studyUIDs = getStudyUids(mediaDTO);
829             Dataset ds = oFact.newDataset();
830 
831             ds.putCS(Tags.QueryRetrieveLevel, "STUDY");
832             for (int i = 0, len = studyUIDs.length; i < len; i++) {
833                 cmd.initCMoveRQ(as.nextMsgID(),
834                         UIDs.StudyRootQueryRetrieveInformationModelMOVE,
835                         priority, this.getMoveDestinationAET());
836                 ds.putUI(Tags.StudyInstanceUID, studyUIDs[i]);
837                 Dimse moveRQ = AssociationFactory.getInstance().newDimse(1,
838                         cmd, ds);
839                 FutureRSP rsp = aa.invoke(moveRQ);
840                 Dimse dimse = rsp.get();
841                 if (!checkResponse(dimse, mediaDTO, "for study:" + studyUIDs[i])) {
842                     return false;
843                 }
844             }
845         } catch (Exception e) {
846             handleError(mediaDTO, "processMove failed for "
847                     + mediaDTO.getFilesetId() + "! Reason: unexpected error", e);
848             return false;
849         } finally {
850             try {
851                 aa.release(true);
852             } catch (Exception e) {
853                 log.warn("Failed to release association " + aa.getAssociation(),
854                         e);
855             }
856         }
857         return true;
858     }

860     /**
861      * Process the media creation request.
862      * <DL>
863      * <DD>1) Send media creation request to <code>mcMScpAET</code>. </DD>
864      * <DD>2) Send N-Action command for media creation to
865      * <code>mcMScpAET</code>. </DD>
866      * </DL>
867      * <p>
868      * 
869      * @param mediaDTO
870      *            The MediaDTO object to process.
871      */
872     private boolean processMediaCreation(MediaDTO mediaDTO) {
873         ActiveAssociation aa;
874         try {
875             aa = openAssociation(mcmScpAET,
876                             UIDs.MediaCreationManagementSOPClass);
877         } catch (Exception e) {
878                 handleError(mediaDTO, "processMove failed for "
879                         + mediaDTO.getFilesetId(), e);
880                 return false;       
881         }
882         try {
883             DcmObjectFactory oFact = DcmObjectFactory.getInstance();
884             Command cmd = oFact.newCommand();
885             cmd.initNCreateRQ(1, UIDs.MediaCreationManagementSOPClass, null);
886             Dataset ds = getMediaCreationReqDS(mediaDTO);
887             Dimse mcRQ = AssociationFactory.getInstance().newDimse(1, cmd, ds);
888             FutureRSP rsp = aa.invoke(mcRQ);
889             Dimse dimse = rsp.get();
890             if (!checkResponse(dimse, mediaDTO, "")) {
891                 return false;
892             }
893             String iuid = dimse.getCommand().getAffectedSOPInstanceUID();
894             Command cmdRsp = dimse.getCommand();
895             Dataset dataRsp = dimse.getDataset();
896             // send action
897             FutureRSP futureRsp = aa.invoke(AssociationFactory.getInstance()
898                     .newDimse(1, oFact.newCommand().initNActionRQ(3,
899                                     UIDs.MediaCreationManagementSOPClass, iuid,
900                                     INITIATE_MEDIA_CREATION),
901                             getMediaCreationActionDS()));
902             dimse = futureRsp.get();
903             this.lookupMediaComposer().setMediaCreationRequestIuid(
904                     mediaDTO.getPk(), iuid);
905             if (!checkResponse(dimse, mediaDTO, "")) {
906                 return false;
907             }
908 
909         } catch (Exception e) {
910             handleError(mediaDTO, "processMediaCreation failed for "
911                     + mediaDTO.getFilesetId() + "! Reason: unexpected error", e);
912             return false;
913         } finally {
914             try {
915                 aa.release(true);
916             } catch (Exception e) {
917                 log.warn("Failed to release association " + aa.getAssociation(),
918                         e);
919             }
920         }
921         return true;
922     }

992     /**
993      * Check a response DICOM message for return status.
994      * <p>
995      * Log a warning if status is <code>Status.AttributeValueOutOfRange</code>.
996      * <p>
997      * Call <code>handleError</code> with status info and error comment if
998      * message indicates a failure.
999      * <p>
1000      * This method is generic for the usage in this class because the status
1001      * values of move, create and action doesnt conflict!
1002      * 
1003      * @param mediaDTO
1004      *            The effected media
1005      * @param dimse
1006      *            The DICOM message to check.
1007      * 
1008      * @return true if status is OK, false if message indicates a failure.
1009      */
1010     private boolean checkResponse(Dimse rsp, MediaDTO mediaDTO, String msg) {
1011         Command cmdRsp = rsp.getCommand();
1012         Dataset dataRsp = null;
1013         try {
1014             dataRsp = rsp.getDataset();
1015         } catch (IOException e) {
1016             log.error("Cant get Dataset from response message!", e);
1017         }
1018         int status = cmdRsp.getStatus();
1019         switch (status) {
1020         case Status.AttributeValueOutOfRange:
1021             log.warn("Warning: Attribute Value Out Of Range: "
1022                     + cmdRsp.getString(Tags.ErrorComment, "") + dataRsp);
1023         case Status.Success:
1024             return true;
1025         }
1026         log.error("Media creation failed! " + " [" + msg + "]"
1027                 + "Failure Status " + Integer.toHexString(status) + ": "
1028                 + cmdRsp.getString(Tags.ErrorComment, "") + dataRsp);
1029         handleError(mediaDTO, "Media creation failed! return status:"
1030                 + Integer.toHexString(status) + " Reason:"
1031                 + cmdRsp.getString(Tags.ErrorComment, "") + " [" + msg + "]",
1032                 null);
1033         return false;
1034     }

1057     public String updateMediaStatus() {
1058         synchronized(this) {
1059             if (isRunningUpdateMediaStatus) {
1060                 String msg = "UpdateMediaStatus is already running!"; 
1061                 log.info(msg);
1062                 return msg;
1063             }
1064             isRunningUpdateMediaStatus = true;
1065         }
1066         try {
1067             List procList;
1068             try {
1069                 procList = this.lookupMediaComposer().getWithStatus(
1070                         MediaDTO.BURNING);
1071                 if (procList.isEmpty())
1072                     return "No Media in processing status.";
1073             } catch (Exception e) {
1074                 log.error("Check for pending media creation processes fails!", e);
1075                 return "Error: Check for pending media creation processes fails!";
1076             }
1077             ActiveAssociation aa;
1078             try {
1079                 aa = openAssociation(mcmScpAET,
1080                                 UIDs.MediaCreationManagementSOPClass);
1081             } catch (Exception e) {
1082                 log.error("Cant get media creation status!", e);
1083                 return "Error: could not open association!";
1084             }
1085             try {
1086                 Association as = aa.getAssociation();
1087                 String iuid = null;
1088                 MediaDTO mediaDTO = null;
1089                 int[] getAttrs = new int[] { Tags.ExecutionStatus,
1090                         Tags.ExecutionStatusInfo, Tags.FailedSOPSeq };
1091                 int mediaStatus;
1092                 int mediaWithAction = 0;
1093                 int mediaDone = 0;
1094                 int mediaFailed = 0;
1095                 for (Iterator iter = procList.iterator(); iter.hasNext();) {
1096                     mediaDTO = (MediaDTO) iter.next();
1097                     iuid = mediaDTO.getMediaCreationRequestIuid();
1098                     mediaStatus = mediaDTO.getMediaStatus();
1099                     if (iuid != null && iuid.length() > 0) {
1100                         mediaWithAction++;
1101                         Command cmdRq = DcmObjectFactory.getInstance().newCommand();
1102                         cmdRq.initNGetRQ(as.nextMsgID(),
1103                                 UIDs.MediaCreationManagementSOPClass, iuid,
1104                                 getAttrs);
1105                         FutureRSP futureRsp = aa.invoke(AssociationFactory
1106                                 .getInstance().newDimse(1, cmdRq));
1107                         Dimse rsp = futureRsp.get();
1108                         Command cmdRsp = rsp.getCommand();
1109                         Dataset dataRsp = rsp.getDataset();
1110                         int status = cmdRsp.getStatus();
1111                         if (status != 0) {
1112                             log
1113                                     .error("Cant get media creation status! Failure Status:"
1114                                             + Integer.toHexString(status)
1115                                             + ": "
1116                                             + cmdRsp.getString(Tags.ErrorComment,
1117                                                     "")
1118                                             + (dataRsp == null ? ""
1119                                                     : ("\n" + dataRsp)));
1120                         } else {
1121                             if (log.isDebugEnabled())
1122                                 log.debug("Received Attributes:\n" + dataRsp);
1123                             String execStatus = dataRsp
1124                                     .getString(Tags.ExecutionStatus);
1125                             if ("DONE".equals(execStatus)) {
1126                                 mediaDone++;
1127                                 mediaComposer.setMediaStatus(mediaDTO.getPk(),
1128                                         MediaDTO.COMPLETED,
1129                                         "successfully completed");
1130                                 if (log.isInfoEnabled())
1131                                     log.info("Media " + mediaDTO.getFilesetId()
1132                                             + " successfully created!");
1133                             } else if ("FAILURE".equals(execStatus)) {
1134                                 mediaFailed++;
1135                                 String info = dataRsp
1136                                         .getString(Tags.ExecutionStatusInfo);
1137                                 if ("NO_INSTANCE".equals(info)) {
1138                                     info = info + "("
1139                                             + dataRsp.vm(Tags.FailedSOPSeq)
1140                                             + " number of instances missing)";
1141                                 }
1142                                 log.error("Cant create media "
1143                                         + mediaDTO.getFilesetId() + "! Reason:"
1144                                         + info);
1145                                 mediaComposer.setMediaStatus(mediaDTO.getPk(),
1146                                         MediaDTO.ERROR, info);
1147                             } else {
1148                                 mediaComposer.setMediaStatus(mediaDTO.getPk(),
1149                                         mediaStatus, execStatus);
1150                             }
1151                         }
1152                     }// end if iuid
1153                 }// end for
1154                 return "Media creation status:" + mediaFailed + " media FAILED, "
1155                         + mediaDone + " media done! Total: " + procList.size()
1156                         + " media processing / " + mediaWithAction
1157                         + " with N-ACTION";
1158             } catch (Exception x) {
1159                 log.error("Cant get media creation status! Reason: unexpected error.",
1160                                 x);
1161                 return "Cant get media create status: Unexpected error"
1162                         + x.getMessage();
1163             } finally {
1164                 try {
1165                     aa.release(true);
1166                 } catch (Exception e) {
1167                     log.warn("Failed to release association " + aa.getAssociation(),
1168                             e);
1169                 }
1170             }
1171         } finally {
1172             isRunningUpdateMediaStatus = false;
1173         }
1174     }

1176     /**
1177      * Initiate creation of Media with studies older than MaxStudyAge.
1178      * 
1179      * @return Number of media creations initiated.
1180      */
1181     public int burnMedia() {
1182         synchronized(this) {
1183             if (isRunningBurnMedia) {
1184                 log.info("BurnMedia is already running!");
1185                 return -1;
1186             }
1187             isRunningBurnMedia = true;
1188         }
1189         log.info("Check for scheduled Media to burn");
1190         try {
1191             Collection c = lookupMediaComposer().getWithStatus(MediaDTO.OPEN);
1192             long maxAgeDate = System.currentTimeMillis()
1193                     - (maxStudyAge - minStudyAge);// media is created after
1194                                                     // minStudyAge -> max media
1195                                                     // age is
1196                                                     // maxStudyAge-minStudyAge
1197             List mediaToBurn = new ArrayList();
1198             MediaDTO mediaDTO;
1199             for (Iterator iter = c.iterator(); iter.hasNext();) {
1200                 mediaDTO = (MediaDTO) iter.next();
1201                 if (mediaDTO.getCreatedTime().getTime() < maxAgeDate) {
1202                     if (this.isAutomaticMediaCreation()) {
1203                         process(mediaDTO);
1204                     }
1205                     mediaToBurn.add(mediaDTO);
1206                 }
1207             }
1208             if (!this.isAutomaticMediaCreation()) {
1209                 notifyMediaToBurn(mediaToBurn);
1210             }
1211             return mediaToBurn.size();
1212         } catch (Exception e) {
1213             log.error("Failed to initiate media creation:", e);
1214             return -1;
1215         } finally {
1216             isRunningBurnMedia = false;
1217         }
1218     }
