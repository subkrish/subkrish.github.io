40  package org.dcm4chex.archive.web.maverick;
41  
42  import java.text.ParseException;
43  import java.text.SimpleDateFormat;
44  import java.util.Date;
45  import java.util.List;
46  
47  import javax.servlet.http.HttpServletRequest;
48  import org.slf4j.Logger;
49  import org.slf4j.LoggerFactory;
50  
51  
52  /**
53   * @author franz.willer@gwi-ag.com
54   * @version $Revision: 14081 $ $Date: 2010-10-05 21:39:30 +0200 (Di, 05 Okt 2010) $
55   * @since 5.10.2004
56   *
57   */
58  public class AddWorklistCtrl extends Dcm4cheeFormController {
59      private static Logger log = LoggerFactory.getLogger(AddWorklistCtrl.class);
60  
61      /** Popup message */
62      private String popupMsg = null;
63  
64      private long studyPk;
65      private static GPWLFeedDelegate delegate = null;;
66  
67      private String template;
68      private String humanPerformer;
69      private long scheduleDate = System.currentTimeMillis()+60000;
70  
71      private static final SimpleDateFormat formatter = new SimpleDateFormat("yyyy/MM/dd HH:mm");
72  
73      public final long getStudyPk() {
74          return studyPk;
75      }
76  
77      public final void setStudyPk(long pk) {
78          this.studyPk = pk;
79      }
80  
81      /**
82       * @return Returns the humanPerformer.
83       */
84      public String getHumanPerformer() {
85          return humanPerformer;
86      }
87      /**
88       * @param humanPerformer The humanPerformer to set.
89       */
90      public void setHumanPerformer(String humanPerformer) {
91          this.humanPerformer = humanPerformer;
92      }
93      /**
94       * @return Returns the scheduleDate.
95       */
96      public String getScheduleDate() {
97          return formatter.format( new Date(scheduleDate) );
98      }
99      /**
100      * @param scheduleDate The scheduleDate to set.
101      * @throws ParseException
102      */
103     public void setScheduleDate(String scheduleDate) throws ParseException {
104         this.scheduleDate = formatter.parse(scheduleDate).getTime();
105     }
106     /**
107      * @return Returns the template.
108      */
109     public String getTemplate() {
110         return template;
111     }
112     /**
113      * @param template The template to set.
114      */
115     public void setTemplate(String template) {
116         this.template = template;
117     }
118     public List getHumanPerformerList() {
119         return delegate.getHumanPerformerList();
120     }
121     public List getTemplateList() {
122         return delegate.getTemplateList();
123     }
124 
125     /**
126      * @return Returns the popupMsg.
127      */
128     public String getPopupMsg() {
129         return popupMsg;
130     }
131 
132     protected void init() {
133         popupMsg = null;
134         if ( delegate  == null ) {
135             delegate = new GPWLFeedDelegate();
136             try {
137                 delegate.init( getCtx() );
138             } catch( Exception x ) {
139                 log.warn("Could not initialize new delegate {}", delegate, x);
140             }
141         }
142     }
143 
144     protected String perform() throws Exception {
145         init();
146         HttpServletRequest rq = getCtx().getRequest();
147         if (rq.getParameter("add") != null
148                 || rq.getParameter("add.x") != null) { return addWorklistItem(); }
149         if (rq.getParameter("cancel") != null
150                 || rq.getParameter("cancel.x") != null) { return "cancel"; }
151         return SUCCESS;
152     }
153 
154     /**
155      * @return
156      * @throws ParseException
157      */
158     private String addWorklistItem() throws ParseException {
159         if ( delegate.addWorklistItem( studyPk, template, humanPerformer, scheduleDate ) ) {
160             //popupMsg = "New worklist item added!"; (see maverick.xml view type folder in command addWorklist)
161             return "folder";
162         } else {
163             popupMsg = "Failed to add new worklist item!";
164             return SUCCESS;//to open AddWorklistCtrl!
165         }
166     }
167 
168 }
