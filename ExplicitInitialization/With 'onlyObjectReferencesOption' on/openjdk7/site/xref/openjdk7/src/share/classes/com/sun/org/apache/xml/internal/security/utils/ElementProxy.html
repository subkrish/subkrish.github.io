81      * Constructor ElementProxy
82      *
83      * @param doc
84      */
85     public ElementProxy(Document doc) {
86        if (doc == null) {
87           throw new RuntimeException("Document is null");
88        }
89  
90        this._doc = doc;
91        this._constructionElement = createElementForFamilyLocal(this._doc,
92                    this.getBaseNamespace(), this.getBaseLocalName());
93     }
94     protected Element createElementForFamilyLocal(Document doc, String namespace,
95             String localName) {
96                    Element result = null;
97                if (namespace == null) {
98                   result = doc.createElementNS(null, localName);
99                } else {
100                   String baseName=this.getBaseNamespace();
101                   String prefix=ElementProxy.getDefaultPrefix(baseName);
102                  if ((prefix == null) || (prefix.length() == 0)) {
103                     result = doc.createElementNS(namespace, localName);
104 
105                     result.setAttributeNS(Constants.NamespaceSpecNS, "xmlns",
106                                           namespace);
107                  } else {
108                          String tagName=null;
109                          String defaultPrefixNaming=ElementProxy.getDefaultPrefixBindings(baseName);
110                          StringBuffer sb=new StringBuffer(prefix);
111                          sb.append(':');
112                          sb.append(localName);
113                          tagName=sb.toString();
114                     result = doc.createElementNS(namespace, tagName );
115 
116                     result.setAttributeNS(Constants.NamespaceSpecNS,  defaultPrefixNaming,
117                                           namespace);
118                  }
119               }
120               return result;
121 }

124    /**
125     * This method creates an Element in a given namespace with a given localname.
126     * It uses the {@link ElementProxy#getDefaultPrefix} method to decide whether
127     * a particular prefix is bound to that namespace.
128     * <BR />
129     * This method was refactored out of the constructor.
130     *
131     * @param doc
132     * @param namespace
133     * @param localName
134     * @return The element created.
135     */
136    public static Element createElementForFamily(Document doc, String namespace,
137            String localName) {
138        //Element nscontext = XMLUtils.createDSctx(doc, "x", namespace);
139       Element result = null;
140       String prefix = ElementProxy.getDefaultPrefix(namespace);
141 
142       if (namespace == null) {
143          result = doc.createElementNS(null, localName);
144       } else {
145          if ((prefix == null) || (prefix.length() == 0)) {
146             result = doc.createElementNS(namespace, localName);
147 
148             result.setAttributeNS(Constants.NamespaceSpecNS, "xmlns",
149                                   namespace);
150          } else {
151             result = doc.createElementNS(namespace, prefix + ":" + localName);
152 
153             result.setAttributeNS(Constants.NamespaceSpecNS,  ElementProxy.getDefaultPrefixBindings(namespace),
154                                   namespace);
155          }
156       }
157 
158       return result;
159    }

168    public void setElement(Element element, String BaseURI)
169            throws XMLSecurityException {
170 
171       if (element == null) {
172          throw new XMLSecurityException("ElementProxy.nullElement");
173       }
174 
175       if (log.isLoggable(java.util.logging.Level.FINE)) {
176         log.log(java.util.logging.Level.FINE, "setElement(" + element.getTagName() + ", \"" + BaseURI + "\"");
177       }
178 
179       this._doc = element.getOwnerDocument();
180       this._constructionElement = element;
181       this._baseURI = BaseURI;
182    }

191    public ElementProxy(Element element, String BaseURI)
192            throws XMLSecurityException {
193       if (element == null) {
194          throw new XMLSecurityException("ElementProxy.nullElement");

197       if (log.isLoggable(java.util.logging.Level.FINE)) {
198         log.log(java.util.logging.Level.FINE, "setElement(\"" + element.getTagName() + "\", \"" + BaseURI
199                 + "\")");
200       }
201 
202       this._doc = element.getOwnerDocument();
203       this._constructionElement = element;
204       this._baseURI = BaseURI;

426    /**
427     * Method length
428     *
429     * @param namespace
430     * @param localname
431     * @return the number of elements {namespace}:localname under this element
432     */
433    public int length(String namespace, String localname) {
434             int number=0;
435             Node sibling=this._constructionElement.getFirstChild();
436             while (sibling!=null) {
437                 if (localname.equals(sibling.getLocalName())
438                                 &&
439                                         namespace==sibling.getNamespaceURI() ) {
440                         number++;
441                 }
442                 sibling=sibling.getNextSibling();
443             }
444             return number;
445      }

447    /**
448     * Adds an xmlns: definition to the Element. This can be called as follows:
449     *
450     * <PRE>
451     * // set namespace with ds prefix
452     * xpathContainer.setXPathNamespaceContext("ds", "http://www.w3.org/2000/09/xmldsig#");
453     * xpathContainer.setXPathNamespaceContext("xmlns:ds", "http://www.w3.org/2000/09/xmldsig#");
454     * </PRE>
455     *
456     * @param prefix
457     * @param uri
458     * @throws XMLSecurityException
459     */
460    public void setXPathNamespaceContext(String prefix, String uri)
461            throws XMLSecurityException {
462 
463       String ns;
464 
465       if ((prefix == null) || (prefix.length() == 0)) {
466        throw new XMLSecurityException("defaultNamespaceCannotBeSetHere");
467       } else if (prefix.equals("xmlns")) {
468         throw new XMLSecurityException("defaultNamespaceCannotBeSetHere");
469       } else if (prefix.startsWith("xmlns:")) {
470          ns = prefix;//"xmlns:" + prefix.substring("xmlns:".length());
471       } else {
472          ns = "xmlns:" + prefix;
473       }
474 
475 
476 
477       Attr a = this._constructionElement.getAttributeNodeNS(Constants.NamespaceSpecNS, ns);
478 
479       if (a != null) {
480        if (!a.getNodeValue().equals(uri)) {
481          Object exArgs[] = { ns,
482                              this._constructionElement.getAttributeNS(null,
483                                                                       ns) };
484 
485          throw new XMLSecurityException("namespacePrefixAlreadyUsedByOtherURI",
486                                         exArgs);
487        }
488        return;
489       }
490 
491       this._constructionElement.setAttributeNS(Constants.NamespaceSpecNS, ns,
492                                                uri);
493    }

499     /**
500      * Method setDefaultPrefix
501      *
502      * @param namespace
503      * @param prefix
504      * @throws XMLSecurityException
505      */
506     public static void setDefaultPrefix(String namespace, String prefix)
507         throws XMLSecurityException {
508 
509         if (ElementProxy._prefixMappings.containsValue(prefix)) {
510 
511             Object storedNamespace=ElementProxy._prefixMappings.get(namespace);
512             if (!storedNamespace.equals(prefix)) {
513                 Object exArgs[] = { prefix, namespace, storedNamespace };
514 
515                 throw new XMLSecurityException("prefix.AlreadyAssigned", exArgs);
516             }
517         }
518         if (Constants.SignatureSpecNS.equals(namespace)) {
519             XMLUtils.dsPrefix=prefix;
520         }
521         ElementProxy._prefixMappings.put(namespace, prefix.intern());
522         if (prefix.length() == 0) {
523             ElementProxy._prefixMappingsBindings.put(namespace, "xmlns");
524         } else {
525             ElementProxy._prefixMappingsBindings.put(namespace, ("xmlns:"+prefix).intern());
526         }
527    }
