39  /*
40   * Implementation notes:
41   *
42   * (1)"Standard based" (ASCII, JIS_X_0201 and JIS_X_0208) ISO2022-JP charset
43   * is provided by the base implementation of this class.
44   *
45   * Three Microsoft ISO2022-JP variants, MS50220, MS50221 and MSISO2022JP
46   * are provided via subclasses.
47   *
48   * (2)MS50220 and MS50221 are assumed to work the same way as Microsoft
49   * CP50220 and CP50221's 7-bit implementation works by using CP5022X
50   * specific JIS0208 and JIS0212 mapping tables (generated via Microsoft's
51   * MultiByteToWideChar/WideCharToMultiByte APIs). The only difference
52   * between these 2 classes is that MS50220 does not support singlebyte
53   * halfwidth kana (Uff61-Uff9f) shiftin mechanism when "encoding", instead
54   * these halfwidth kana characters are converted to their fullwidth JIS0208
55   * counterparts.
56   *
57   * The difference between the standard JIS_X_0208 and JIS_X_0212 mappings
58   * and the CP50220/50221 specific are
59   *
60   * 0208 mapping:
61   *              1)0x213d <-> U2015 (compared to U2014)
62   *              2)One way mappings for 5 characters below
63   *                u2225 (ms) -> 0x2142 <-> u2016 (jis)
64   *                uff0d (ms) -> 0x215d <-> u2212 (jis)
65   *                uffe0 (ms) -> 0x2171 <-> u00a2 (jis)
66   *                uffe1 (ms) -> 0x2172 <-> u00a3 (jis)
67   *                uffe2 (ms) -> 0x224c <-> u00ac (jis)
68   *                //should consider 0xff5e -> 0x2141 <-> U301c?
69   *              3)NEC Row13 0x2d21-0x2d79
70   *              4)85-94 ku <-> UE000,UE3AB (includes NEC selected
71   *                IBM kanji in 89-92ku)
72   *              5)UFF61-UFF9f -> Fullwidth 0208 KANA
73   *
74   * 0212 mapping:
75   *              1)0x2237 <-> UFF5E (Fullwidth Tilde)
76   *              2)0x2271 <-> U2116 (Numero Sign)
77   *              3)85-94 ku <-> UE3AC - UE757
78   *
79   * (3)MSISO2022JP uses a JIS0208 mapping generated from MS932DB.b2c
80   * and MS932DB.c2b by converting the SJIS codepoints back to their
81   * JIS0208 counterparts. With the exception of
82   *
83   * (a)Codepoints with a resulting JIS0208 codepoints beyond 0x7e00 are
84   *    dropped (this includs the IBM Extended Kanji/Non-kanji from 0x9321
85   *    to 0x972c)
86   * (b)The Unicode codepoints that the IBM Extended Kanji/Non-kanji are
87   *    mapped to (in MS932) are mapped back to NEC selected IBM Kanji/
88   *    Non-kanji area at 0x7921-0x7c7e.
89   *
90   * Compared to JIS_X_0208 mapping, this MS932 based mapping has
91  
92   * (a)different mappings for 7 JIS codepoints
93   *        0x213d <-> U2015
94   *        0x2141 <-> UFF5E
95   *        0x2142 <-> U2225
96   *        0x215d <-> Uff0d
97   *        0x2171 <-> Uffe0
98   *        0x2172 <-> Uffe1
99   *        0x224c <-> Uffe2
100  * (b)added one-way c2b mappings for
101  *        U00b8 -> 0x2124
102  *        U00b7 -> 0x2126
103  *        U00af -> 0x2131
104  *        U00ab -> 0x2263
105  *        U00bb -> 0x2264
106  *        U3094 -> 0x2574
107  *        U00b5 -> 0x264c
108  * (c)NEC Row 13
109  * (d)NEC selected IBM extended Kanji/Non-kanji
110  *    These codepoints are mapped to the same Unicode codepoints as
111  *    the MS932 does, while MS50220/50221 maps them to the Unicode
112  *    private area.
113  *
114  * # There is also an interesting difference when compared to MS5022X
115  *   0208 mapping for JIS codepoint "0x2D60", MS932 maps it to U301d
116  *   but MS5022X maps it to U301e, obvious MS5022X is wrong, but...
117  */
118

229         private CoderResult decodeArrayLoop(ByteBuffer src,
230                                             CharBuffer dst)
231         {
232             int inputSize = 0;
233             int b1 = 0, b2 = 0, b3 = 0, b4 = 0;
234             char c = REPLACE_CHAR;
235             byte[] sa = src.array();
236             int sp = src.arrayOffset() + src.position();
237             int sl = src.arrayOffset() + src.limit();
238             assert (sp <= sl);
239             sp = (sp <= sl ? sp : sl);
240 
241             char[] da = dst.array();
242             int dp = dst.arrayOffset() + dst.position();
243             int dl = dst.arrayOffset() + dst.limit();
244             assert (dp <= dl);
245             dp = (dp <= dl ? dp : dl);
246 
247             try {
248                 while (sp < sl) {
249                     b1 = sa[sp] & 0xff;
250                     inputSize = 1;
251                     if ((b1 & 0x80) != 0) {
252                         return CoderResult.malformedForLength(inputSize);
253                     }
254                     if (b1 == ESC || b1 == SO || b1 == SI) {
255                         if (b1 == ESC) {
256                             if (sp + inputSize + 2 > sl)
257                                 return CoderResult.UNDERFLOW;
258                             b2 = sa[sp + inputSize++] & 0xff;
259                             if (b2 == '(') {
260                                 b3 = sa[sp + inputSize++] & 0xff;
261                                 if (b3 == 'B'){
262                                     currentState = ASCII;
263                                 } else if (b3 == 'J'){
264                                     currentState = JISX0201_1976;
265                                 } else if (b3 == 'I'){
266                                     currentState = JISX0201_1976_KANA;
267                                 } else {
268                                     return CoderResult.malformedForLength(inputSize);
269                                 }
270                             } else if (b2 == '$'){
271                                 b3 = sa[sp + inputSize++] & 0xff;
272                                 if (b3 == '@'){
273                                     currentState = JISX0208_1978;
274                                 } else if (b3 == 'B'){
275                                     currentState = JISX0208_1983;
276                                 } else if (b3 == '(' && decoder0212 != null) {
277                                     if (sp + inputSize + 1 > sl)
278                                         return CoderResult.UNDERFLOW;
279                                     b4 = sa[sp + inputSize++] & 0xff;
280                                     if (b4 == 'D') {
281                                         currentState = JISX0212_1990;
282                                     } else {
283                                         return CoderResult.malformedForLength(inputSize);
284                                     }
285                                 } else {
286                                     return CoderResult.malformedForLength(inputSize);
287                                 }
288                             } else {
289                                 return CoderResult.malformedForLength(inputSize);
290                             }
291                         } else if (b1 == SO) {
292                             previousState = currentState;
293                             currentState = SHIFTOUT;
294                         } else if (b1 == SI) {
295                             currentState = previousState;
296                         }
297                         sp += inputSize;
298                         continue;
299                     }
300                     if (dp + 1 > dl)
301                         return CoderResult.OVERFLOW;
302 
303                     switch (currentState){
304                         case ASCII:
305                             da[dp++] = (char)(b1 & 0xff);
306                             break;
307                         case JISX0201_1976:
308                           switch (b1) {
309                               case 0x5c:  // Yen/tilde substitution
310                                 da[dp++] = '\u00a5';
311                                 break;
312                               case 0x7e:
313                                 da[dp++] = '\u203e';
314                                 break;
315                               default:
316                                 da[dp++] = (char)b1;
317                                 break;
318                             }
319                             break;
320                         case JISX0208_1978:
321                         case JISX0208_1983:
322                             if (sp + inputSize + 1 > sl)
323                                 return CoderResult.UNDERFLOW;
324                             b2 = sa[sp + inputSize++] & 0xff;
325                             c = decodeDouble(b1,b2);
326                             if (c == REPLACE_CHAR)
327                                 return CoderResult.unmappableForLength(inputSize);
328                             da[dp++] = c;
329                             break;
330                         case JISX0212_1990:
331                             if (sp + inputSize + 1 > sl)
332                                 return CoderResult.UNDERFLOW;
333                             b2 = sa[sp + inputSize++] & 0xff;
334                             c = decoder0212.decodeDouble(b1,b2);
335                             if (c == REPLACE_CHAR)
336                                 return CoderResult.unmappableForLength(inputSize);
337                             da[dp++] = c;
338                             break;
339                         case JISX0201_1976_KANA:
340                         case SHIFTOUT:
341                             if (b1 > 0x60) {
342                                 return CoderResult.malformedForLength(inputSize);
343                             }
344                             da[dp++] = (char)(b1 + 0xff40);
345                             break;
346                     }
347                     sp += inputSize;
348                 }
349                 return CoderResult.UNDERFLOW;
350             } finally {
351                 src.position(sp - src.arrayOffset());
352                 dst.position(dp - dst.arrayOffset());
353             }
354         }

356         private CoderResult decodeBufferLoop(ByteBuffer src,
357                                              CharBuffer dst)
358         {
359             int mark = src.position();
360             int b1 = 0, b2 = 0, b3 = 0, b4=0;
361             char c = REPLACE_CHAR;
362             int inputSize = 0;
363             try {
364                 while (src.hasRemaining()) {
365                     b1 = src.get() & 0xff;
366                     inputSize = 1;
367                     if ((b1 & 0x80) != 0)
368                         return CoderResult.malformedForLength(inputSize);
369                     if (b1 == ESC || b1 == SO || b1 == SI) {
370                         if (b1 == ESC) {  // ESC
371                             if (src.remaining() < 2)
372                                 return CoderResult.UNDERFLOW;
373                             b2 = src.get() & 0xff;
374                             inputSize++;
375                             if (b2 == '(') {
376                                 b3 = src.get() & 0xff;
377                                 inputSize++;
378                                 if (b3 == 'B'){
379                                     currentState = ASCII;
380                                 } else if (b3 == 'J'){
381                                     currentState = JISX0201_1976;
382                                 } else if (b3 == 'I'){
383                                     currentState = JISX0201_1976_KANA;
384                                 } else {
385                                    return CoderResult.malformedForLength(inputSize);
386                                 }
387                             } else if (b2 == '$'){
388                                 b3 = src.get() & 0xff;
389                                 inputSize++;
390                                 if (b3 == '@'){
391                                     currentState = JISX0208_1978;
392                                 } else if (b3 == 'B'){
393                                     currentState = JISX0208_1983;
394                                 } else if (b3 == '(' && decoder0212 != null) {
395                                     if (!src.hasRemaining())
396                                         return CoderResult.UNDERFLOW;
397                                     b4 = src.get() & 0xff;
398                                     inputSize++;
399                                     if (b4 == 'D') {
400                                         currentState = JISX0212_1990;
401                                     } else {
402                                         return CoderResult.malformedForLength(inputSize);
403                                     }
404                                 } else {
405                                     return CoderResult.malformedForLength(inputSize);
406                                 }
407                             } else {
408                                 return CoderResult.malformedForLength(inputSize);
409                             }
410                         } else if (b1 == SO) {
411                             previousState = currentState;
412                             currentState = SHIFTOUT;
413                         } else if (b1 == SI) { // shift back in
414                             currentState = previousState;
415                         }
416                         mark += inputSize;
417                         continue;
418                     }
419                     if (!dst.hasRemaining())
420                         return CoderResult.OVERFLOW;
421 
422                     switch (currentState){
423                         case ASCII:
424                             dst.put((char)(b1 & 0xff));
425                             break;
426                         case JISX0201_1976:
427                             switch (b1) {
428                               case 0x5c:  // Yen/tilde substitution
429                                 dst.put('\u00a5');
430                                 break;
431                               case 0x7e:
432                                 dst.put('\u203e');
433                                 break;
434                               default:
435                                 dst.put((char)b1);
436                                 break;
437                             }
438                             break;
439                         case JISX0208_1978:
440                         case JISX0208_1983:
441                             if (!src.hasRemaining())
442                                 return CoderResult.UNDERFLOW;
443                             b2 = src.get() & 0xff;
444                             inputSize++;
445                             c = decodeDouble(b1,b2);
446                             if (c == REPLACE_CHAR)
447                                 return CoderResult.unmappableForLength(inputSize);
448                             dst.put(c);
449                             break;
450                         case JISX0212_1990:
451                             if (!src.hasRemaining())
452                                 return CoderResult.UNDERFLOW;
453                             b2 = src.get() & 0xff;
454                             inputSize++;
455                             c = decoder0212.decodeDouble(b1,b2);
456                             if (c == REPLACE_CHAR)
457                                 return CoderResult.unmappableForLength(inputSize);
458                             dst.put(c);
459                             break;
460                         case JISX0201_1976_KANA:
461                         case SHIFTOUT:
462                             if (b1 > 0x60) {
463                                 return CoderResult.malformedForLength(inputSize);
464                             }
465                             dst.put((char)(b1 + 0xff40));
466                             break;
467                     }
468                     mark += inputSize;
469                 }
470                 return CoderResult.UNDERFLOW;
471             } finally {
472                 src.position(mark);
473             }
474         }

489     private static class Encoder extends DoubleByteEncoder {
490         private static byte[] repl = { (byte)0x21, (byte)0x29 };
491         private int currentMode = ASCII;
492         private int replaceMode = JISX0208_1983;
493         private DoubleByteEncoder encoder0212 = null;
494         private boolean doSBKANA;

496         private Encoder(Charset cs,
497                         short[] index1,
498                         String[] index2,
499                         DoubleByteEncoder encoder0212,
500                         boolean doSBKANA) {
501             super(cs,
502                   index1,
503                   index2,
504                   repl,
505                   4.0f,
506                   (encoder0212 != null)? 9.0f : 8.0f);
507             this.encoder0212 = encoder0212;
508             this.doSBKANA = doSBKANA;
509         }

511         protected int encodeSingle(char inputChar) {
512             return -1;
513         }
514 
515         protected void implReset() {
516             currentMode = ASCII;
517         }
518 
519         protected void implReplaceWith(byte[] newReplacement) {
520             /* It's almost impossible to decide which charset they belong
521                to. The best thing we can do here is to "guess" based on
522                the length of newReplacement.
523              */
524             if (newReplacement.length == 1) {
525                 replaceMode = ASCII;
526             } else if (newReplacement.length == 2) {
527                 replaceMode = JISX0208_1983;
528             }
529         }
530 
531         protected CoderResult implFlush(ByteBuffer out) {
532             if (currentMode != ASCII) {
533                 if (out.remaining() < 3)
534                     return CoderResult.OVERFLOW;
535                 out.put((byte)0x1b);
536                 out.put((byte)0x28);
537                 out.put((byte)0x42);
538                 currentMode = ASCII;
539             }
540             return CoderResult.UNDERFLOW;
541         }

543         public boolean canEncode(char c) {
544             return ((c <= '\u007F') ||
545                     (c >= 0xFF61 && c <= 0xFF9F) ||
546                     (c == '\u00A5') ||
547                     (c == '\u203E') ||
548                     super.canEncode(c) ||
549                     (encoder0212!=null && encoder0212.canEncode(c)));
550         }

552         private final Surrogate.Parser sgp = new Surrogate.Parser();
553 
554         private CoderResult encodeArrayLoop(CharBuffer src,
555                                             ByteBuffer dst)
556         {
557             char[] sa = src.array();
558             int sp = src.arrayOffset() + src.position();
559             int sl = src.arrayOffset() + src.limit();
560             assert (sp <= sl);
561             sp = (sp <= sl ? sp : sl);
562             byte[] da = dst.array();
563             int dp = dst.arrayOffset() + dst.position();
564             int dl = dst.arrayOffset() + dst.limit();
565             assert (dp <= dl);
566             dp = (dp <= dl ? dp : dl);
567 
568             try {
569                 while (sp < sl) {
570                     char c = sa[sp];
571                     if (c <= '\u007F') {
572                         if (currentMode != ASCII) {
573                             if (dl - dp < 3)
574                                 return CoderResult.OVERFLOW;
575                             da[dp++] = (byte)0x1b;
576                             da[dp++] = (byte)0x28;
577                             da[dp++] = (byte)0x42;
578                             currentMode = ASCII;
579                         }
580                         if (dl - dp < 1)
581                             return CoderResult.OVERFLOW;
582                         da[dp++] = (byte)c;
583                     } else if (c >= 0xff61 && c <= 0xff9f && doSBKANA) {
584                         //a single byte kana
585                         if (currentMode != JISX0201_1976_KANA) {
586                             if (dl - dp < 3)
587                                 return CoderResult.OVERFLOW;
588                             da[dp++] = (byte)0x1b;
589                             da[dp++] = (byte)0x28;
590                             da[dp++] = (byte)0x49;
591                             currentMode = JISX0201_1976_KANA;
592                         }
593                         if (dl - dp < 1)
594                             return CoderResult.OVERFLOW;
595                         da[dp++] = (byte)(c - 0xff40);
596                     } else if (c == '\u00A5' || c == '\u203E') {
597                         //backslash or tilde
598                         if (currentMode != JISX0201_1976) {
599                             if (dl - dp < 3)
600                                 return CoderResult.OVERFLOW;
601                             da[dp++] = (byte)0x1b;
602                             da[dp++] = (byte)0x28;
603                             da[dp++] = (byte)0x4a;
604                             currentMode = JISX0201_1976;
605                         }
606                         if (dl - dp < 1)
607                             return CoderResult.OVERFLOW;
608                         da[dp++] = (c == '\u00A5')?(byte)0x5C:(byte)0x7e;
609                     } else {
610                         int index = encodeDouble(c);
611                         if (index != 0) {
612                             if (currentMode != JISX0208_1983) {
613                                 if (dl - dp < 3)
614                                     return CoderResult.OVERFLOW;
615                                 da[dp++] = (byte)0x1b;
616                                 da[dp++] = (byte)0x24;
617                                 da[dp++] = (byte)0x42;
618                                 currentMode = JISX0208_1983;
619                             }
620                             if (dl - dp < 2)
621                                 return CoderResult.OVERFLOW;
622                             da[dp++] = (byte)(index >> 8);
623                             da[dp++] = (byte)(index & 0xff);
624                         } else if (encoder0212 != null &&
625                                    (index = encoder0212.encodeDouble(c)) != 0) {
626                             if (currentMode != JISX0212_1990) {
627                                 if (dl - dp < 4)
628                                     return CoderResult.OVERFLOW;
629                                 da[dp++] = (byte)0x1b;
630                                 da[dp++] = (byte)0x24;
631                                 da[dp++] = (byte)0x28;
632                                 da[dp++] = (byte)0x44;
633                                 currentMode = JISX0212_1990;
634                             }
635                             if (dl - dp < 2)
636                                 return CoderResult.OVERFLOW;
637                             da[dp++] = (byte)(index >> 8);
638                             da[dp++] = (byte)(index & 0xff);
639                         } else {
640                             if (Character.isSurrogate(c) && sgp.parse(c, sa, sp, sl) < 0)
641                                 return sgp.error();
642                             if (unmappableCharacterAction()
643                                 == CodingErrorAction.REPLACE
644                                 && currentMode != replaceMode) {
645                                 if (dl - dp < 3)
646                                     return CoderResult.OVERFLOW;
647                                 if (replaceMode == ASCII) {
648                                     da[dp++] = (byte)0x1b;
649                                     da[dp++] = (byte)0x28;
650                                     da[dp++] = (byte)0x42;
651                                 } else {
652                                     da[dp++] = (byte)0x1b;
653                                     da[dp++] = (byte)0x24;
654                                     da[dp++] = (byte)0x42;
655                                 }
656                                 currentMode = replaceMode;
657                             }
658                             if (Character.isSurrogate(c))
659                                 return sgp.unmappableResult();
660                             return CoderResult.unmappableForLength(1);
661                         }
662                     }
663                     sp++;
664                 }
665                 return CoderResult.UNDERFLOW;
666             } finally {
667                 src.position(sp - src.arrayOffset());
668                 dst.position(dp - dst.arrayOffset());
669             }
670         }

672         private CoderResult encodeBufferLoop(CharBuffer src,
673                                              ByteBuffer dst)
674         {
675             int mark = src.position();
676             try {
677                 while (src.hasRemaining()) {
678                     char c = src.get();
679 
680                     if (c <= '\u007F') {
681                         if (currentMode != ASCII) {
682                             if (dst.remaining() < 3)
683                                 return CoderResult.OVERFLOW;
684                             dst.put((byte)0x1b);
685                             dst.put((byte)0x28);
686                             dst.put((byte)0x42);
687                             currentMode = ASCII;
688                         }
689                         if (dst.remaining() < 1)
690                             return CoderResult.OVERFLOW;
691                         dst.put((byte)c);
692                     } else if (c >= 0xff61 && c <= 0xff9f && doSBKANA) {
693                         //Is it a single byte kana?
694                         if (currentMode != JISX0201_1976_KANA) {
695                             if (dst.remaining() < 3)
696                                 return CoderResult.OVERFLOW;
697                             dst.put((byte)0x1b);
698                             dst.put((byte)0x28);
699                             dst.put((byte)0x49);
700                             currentMode = JISX0201_1976_KANA;
701                         }
702                         if (dst.remaining() < 1)
703                             return CoderResult.OVERFLOW;
704                         dst.put((byte)(c - 0xff40));
705                     } else if (c == '\u00a5' || c == '\u203E') {
706                         if (currentMode != JISX0201_1976) {
707                             if (dst.remaining() < 3)
708                                 return CoderResult.OVERFLOW;
709                             dst.put((byte)0x1b);
710                             dst.put((byte)0x28);
711                             dst.put((byte)0x4a);
712                             currentMode = JISX0201_1976;
713                         }
714                         if (dst.remaining() < 1)
715                             return CoderResult.OVERFLOW;
716                         dst.put((c == '\u00A5')?(byte)0x5C:(byte)0x7e);
717                     } else {
718                         int index = encodeDouble(c);
719                         if (index != 0) {
720                             if (currentMode != JISX0208_1983) {
721                                 if (dst.remaining() < 3)
722                                     return CoderResult.OVERFLOW;
723                                 dst.put((byte)0x1b);
724                                 dst.put((byte)0x24);
725                                 dst.put((byte)0x42);
726                                 currentMode = JISX0208_1983;
727                             }
728                             if (dst.remaining() < 2)
729                                 return CoderResult.OVERFLOW;
730                             dst.put((byte)(index >> 8));
731                             dst.put((byte)(index & 0xff));
732                         } else if (encoder0212 != null &&
733                                    (index = encoder0212.encodeDouble(c)) != 0) {
734                             if (currentMode != JISX0212_1990) {
735                                 if (dst.remaining() < 4)
736                                     return CoderResult.OVERFLOW;
737                                 dst.put((byte)0x1b);
738                                 dst.put((byte)0x24);
739                                 dst.put((byte)0x28);
740                                 dst.put((byte)0x44);
741                                 currentMode = JISX0212_1990;
742                             }
743                             if (dst.remaining() < 2)
744                                 return CoderResult.OVERFLOW;
745                             dst.put((byte)(index >> 8));
746                             dst.put((byte)(index & 0xff));
747                         } else {
748                             if (Character.isSurrogate(c) && sgp.parse(c, src) < 0)
749                                 return sgp.error();
750                             if (unmappableCharacterAction() == CodingErrorAction.REPLACE
751                                 && currentMode != replaceMode) {
752                                 if (dst.remaining() < 3)
753                                     return CoderResult.OVERFLOW;
754                                 if (replaceMode == ASCII) {
755                                     dst.put((byte)0x1b);
756                                     dst.put((byte)0x28);
757                                     dst.put((byte)0x42);
758                                 } else {
759                                     dst.put((byte)0x1b);
760                                     dst.put((byte)0x24);
761                                     dst.put((byte)0x42);
762                                 }
763                                 currentMode = replaceMode;
764                             }
765                             if (Character.isSurrogate(c))
766                                 return sgp.unmappableResult();
767                             return CoderResult.unmappableForLength(1);
768                         }
769                     }
770                     mark++;
771                 }
772                 return CoderResult.UNDERFLOW;
773               } finally {
774                 src.position(mark);
775             }
776         }

778         protected CoderResult encodeLoop(CharBuffer src,
779                                          ByteBuffer dst)
780         {
781             if (src.hasArray() && dst.hasArray())
782                 return encodeArrayLoop(src, dst);
783             else
784                 return encodeBufferLoop(src, dst);
785         }
