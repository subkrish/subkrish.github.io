147     static int getBSDCommandIndex() {
148         String command  = "/usr/sbin/lpc status";
149         String[] names = execCmd(command);
150 
151         if ((names == null) || (names.length == 0)) {
152             return BSD_LPD_NG;
153         }
154 
155         for (int i=0; i<names.length; i++) {
156             if (names[i].indexOf('@') != -1) {
157                 return BSD_LPD_NG;
158             }
159         }
160 
161         return BSD_LPD;
162     }

180     public synchronized PrintService[] getPrintServices() {
181         SecurityManager security = System.getSecurityManager();
182         if (security != null) {
183             security.checkPrintJobAccess();
184         }
185 
186         if (printServices == null || !pollServices) {
187             refreshServices();
188         }
189         if (printServices == null) {
190             return new PrintService[0];
191         } else {
192             return printServices;
193         }
194     }

197     // refreshes "printServices"
198     public synchronized void refreshServices() {
199         /* excludes the default printer */
200         String[] printers = null; // array of printer names
201         String[] printerURIs = null; //array of printer URIs
202 
203         getDefaultPrintService();
204         if (CUPSPrinter.isCupsRunning()) {
205             printerURIs = CUPSPrinter.getAllPrinters();
206             if ((printerURIs != null) && (printerURIs.length > 0)) {
207                 printers = new String[printerURIs.length];
208                 for (int i=0; i<printerURIs.length; i++) {
209                     int lastIndex = printerURIs[i].lastIndexOf("/");
210                     printers[i] = printerURIs[i].substring(lastIndex+1);
211                 }
212             }
213         } else {
214             if (isSysV()) {
215                 printers = getAllPrinterNamesSysV();
216             } else { //BSD
217                 printers = getAllPrinterNamesBSD();
218             }
219         }
220 
221         if (printers == null) {
222             if (defaultPrintService != null) {
223                 printServices = new PrintService[1];
224                 printServices[0] = defaultPrintService;
225             } else {
226                 printServices = null;
227             }
228             return;
229         }
230 
231         ArrayList printerList = new ArrayList();
232         int defaultIndex = -1;
233         for (int p=0; p<printers.length; p++) {
234             if (printers[p] == null) {
235                 continue;
236             }
237             if ((defaultPrintService != null)
238                 && printers[p].equals(defaultPrintService.getName())) {
239                 printerList.add(defaultPrintService);
240                 defaultIndex = printerList.size() - 1;
241             } else {
242                 if (printServices == null) {
243                     IPPPrintService.debug_println(debugPrefix+
244                                                   "total# of printers = "+printers.length);
245 
246                     if (CUPSPrinter.isCupsRunning()) {
247                         try {
248                             printerList.add(new IPPPrintService(printers[p],
249                                                                 printerURIs[p],
250                                                                 true));
251                         } catch (Exception e) {
252                             IPPPrintService.debug_println(debugPrefix+
253                                                           " getAllPrinters Exception "+
254                                                           e);
255 
256                         }
257                     } else {
258                         printerList.add(new UnixPrintService(printers[p]));
259                     }
260                 } else {
261                     int j;
262                     for (j=0; j<printServices.length; j++) {
263                         if ((printServices[j] != null) &&
264                             (printers[p].equals(printServices[j].getName()))) {
265                             printerList.add(printServices[j]);
266                             printServices[j] = null;
267                             break;
268                         }
269                     }
270 
271                     if (j == printServices.length) {      // not found?
272                         if (CUPSPrinter.isCupsRunning()) {
273                             try {
274                                 printerList.add(new IPPPrintService(
275                                                                printers[p],
276                                                                printerURIs[p],
277                                                                true));
278                             } catch (Exception e) {
279                                 IPPPrintService.debug_println(debugPrefix+
280                                                               " getAllPrinters Exception "+
281                                                               e);
282 
283                             }
284                         } else {
285                             printerList.add(new UnixPrintService(printers[p]));
286                         }
287                     }
288                 }
289             }
290         }
291 
292         // Look for deleted services and invalidate these
293         if (printServices != null) {
294             for (int j=0; j < printServices.length; j++) {
295                 if ((printServices[j] instanceof UnixPrintService) &&
296                     (!printServices[j].equals(defaultPrintService))) {
297                     ((UnixPrintService)printServices[j]).invalidateService();
298                 }
299             }
300         }
301 
302         //if defaultService is not found in printerList
303         if (defaultIndex == -1 && defaultPrintService != null) {
304             //add default to the list
305             printerList.add(defaultPrintService);
306             defaultIndex = printerList.size() - 1;
307         }
308 
309         printServices = (PrintService[])printerList.toArray(
310                                       new PrintService[] {});
311 
312         // swap default with the first in the list
313         if (defaultIndex > 0) {
314             PrintService saveService = printServices[0];
315             printServices[0] = printServices[defaultIndex];
316             printServices[defaultIndex] = saveService;
317         }
318     }

320     private boolean matchesAttributes(PrintService service,
321                                       PrintServiceAttributeSet attributes) {
322 
323         Attribute [] attrs =  attributes.toArray();
324         Attribute serviceAttr;
325         for (int i=0; i<attrs.length; i++) {
326             serviceAttr
327                 = service.getAttribute((Class<PrintServiceAttribute>)attrs[i].getCategory());
328             if (serviceAttr == null || !serviceAttr.equals(attrs[i])) {
329                 return false;
330             }
331         }
332         return true;
333     }

336        * parameter to a shell command.

338       private boolean checkPrinterName(String s) {
339         char c;
340 
341         for (int i=0; i < s.length(); i++) {
342           c = s.charAt(i);
343           if (Character.isLetterOrDigit(c) ||
344               c == '-' || c == '_' || c == '.' || c == '/') {
345             continue;
346           } else {
347             return false;
348           }
349         }
350         return true;
351       }

353     /* On a network with many (hundreds) of network printers, it
354      * can save several seconds if you know all you want is a particular
355      * printer, to ask for that printer rather than retrieving all printers.
356      */
357     private PrintService getServiceByName(PrinterName nameAttr) {
358         String name = nameAttr.getValue();
359         PrintService printer = null;
360         if (name == null || name.equals("") || !checkPrinterName(name)) {
361             return null;
362         }
363         if (isSysV()) {
364             printer = getNamedPrinterNameSysV(name);
365         } else {
366             printer = getNamedPrinterNameBSD(name);
367         }
368         return printer;
369     }

371     private PrintService[]
372         getPrintServices(PrintServiceAttributeSet serviceSet) {
373 
374         if (serviceSet == null || serviceSet.isEmpty()) {
375             return getPrintServices();
376         }
377 
378         /* Typically expect that if a service attribute is specified that
379          * its a printer name and there ought to be only one match.
380          * Directly retrieve that service and confirm
381          * that it meets the other requirements.
382          * If printer name isn't mentioned then go a slow path checking
383          * all printers if they meet the reqiremements.
384          */
385         PrintService[] services;
386         PrinterName name = (PrinterName)serviceSet.get(PrinterName.class);
387         PrintService defService;
388         if (name != null && (defService = getDefaultPrintService()) != null) {
389             /* To avoid execing a unix command  see if the client is asking
390              * for the default printer by name, since we already have that
391              * initialised.
392              */
393 
394             PrinterName defName =
395                 (PrinterName)defService.getAttribute(PrinterName.class);
396 
397             if (defName != null && name.equals(defName)) {
398                 if (matchesAttributes(defService, serviceSet)) {
399                     services = new PrintService[1];
400                     services[0] = defService;
401                     return services;
402                 } else {
403                     return new PrintService[0];
404                 }
405             } else {
406                 /* Its not the default service */
407                 PrintService service = getServiceByName(name);
408                 if (service != null &&
409                     matchesAttributes(service, serviceSet)) {
410                     services = new PrintService[1];
411                     services[0] = service;
412                     return services;
413                 } else {
414                     return new PrintService[0];
415                 }
416             }
417         } else {
418             /* specified service attributes don't include a name.*/
419             Vector matchedServices = new Vector();
420             services = getPrintServices();
421             for (int i = 0; i< services.length; i++) {
422                 if (matchesAttributes(services[i], serviceSet)) {
423                     matchedServices.add(services[i]);
424                 }
425             }
426             services = new PrintService[matchedServices.size()];
427             for (int i = 0; i< services.length; i++) {
428                 services[i] = (PrintService)matchedServices.elementAt(i);
429             }
430             return services;
431         }
432     }

434     /*
435      * If service attributes are specified then there must be additional
436      * filtering.
437      */
438     public PrintService[] getPrintServices(DocFlavor flavor,
439                                            AttributeSet attributes) {
440         SecurityManager security = System.getSecurityManager();
441         if (security != null) {
442           security.checkPrintJobAccess();
443         }
444         PrintRequestAttributeSet requestSet = null;
445         PrintServiceAttributeSet serviceSet = null;
446 
447         if (attributes != null && !attributes.isEmpty()) {
448 
449             requestSet = new HashPrintRequestAttributeSet();
450             serviceSet = new HashPrintServiceAttributeSet();
451 
452             Attribute[] attrs = attributes.toArray();
453             for (int i=0; i<attrs.length; i++) {
454                 if (attrs[i] instanceof PrintRequestAttribute) {
455                     requestSet.add(attrs[i]);
456                 } else if (attrs[i] instanceof PrintServiceAttribute) {
457                     serviceSet.add(attrs[i]);
458                 }
459             }
460         }
461 
462         PrintService[] services = getPrintServices(serviceSet);
463         if (services.length == 0) {
464             return services;
465         }
466 
467         if (CUPSPrinter.isCupsRunning()) {
468             ArrayList matchingServices = new ArrayList();
469             for (int i=0; i<services.length; i++) {
470                 try {
471                     if (services[i].
472                         getUnsupportedAttributes(flavor, requestSet) == null) {
473                         matchingServices.add(services[i]);
474                     }
475                 } catch (IllegalArgumentException e) {
476                 }
477             }
478             services = new PrintService[matchingServices.size()];
479             return (PrintService[])matchingServices.toArray(services);
480 
481         } else {
482             // We only need to compare 1 PrintService because all
483             // UnixPrintServices are the same anyway.  We will not use
484             // default PrintService because it might be null.
485             PrintService service = services[0];
486             if ((flavor == null ||
487                  service.isDocFlavorSupported(flavor)) &&
488                  service.getUnsupportedAttributes(flavor, requestSet) == null)
489             {
490                 return services;
491             } else {
492                 return new PrintService[0];
493             }
494         }
495     }

511     public synchronized PrintService getDefaultPrintService() {
512         SecurityManager security = System.getSecurityManager();
513         if (security != null) {
514           security.checkPrintJobAccess();
515         }
516 
517         // clear defaultPrintService
518         defaultPrintService = null;
519 
520         IPPPrintService.debug_println("isRunning ? "+
521                                       (CUPSPrinter.isCupsRunning()));
522         if (CUPSPrinter.isCupsRunning()) {
523             defaultPrinter = CUPSPrinter.getDefaultPrinter();
524         } else {
525             if (isSysV()) {
526                 defaultPrinter = getDefaultPrinterNameSysV();
527             } else {
528                 defaultPrinter = getDefaultPrinterNameBSD();
529             }
530         }
531         if (defaultPrinter == null) {
532             return null;
533         }
534         defaultPrintService = null;
535         if (printServices != null) {
536             for (int j=0; j<printServices.length; j++) {
537                 if (defaultPrinter.equals(printServices[j].getName())) {
538                     defaultPrintService = printServices[j];
539                     break;
540                 }
541             }
542         }
543         if (defaultPrintService == null) {
544             if (CUPSPrinter.isCupsRunning()) {
545                 try {
546                     PrintService defaultPS =
547                         new IPPPrintService(defaultPrinter,
548                                             new URL("http://"+
549                                                     CUPSPrinter.getServer()+":"+
550                                                     CUPSPrinter.getPort()+"/"+
551                                                     defaultPrinter));
552                     defaultPrintService = defaultPS;
553                 } catch (Exception e) {
554                 }
555             } else {
556                 defaultPrintService = new UnixPrintService(defaultPrinter);
557             }
558         }
559 
560         return defaultPrintService;
561     }

564         getServicesInbackground(BackgroundLookupListener listener) {
565         if (printServices != null) {
566             listener.notifyServices(printServices);
567         } else {
568             if (lookupListeners == null) {
569                 lookupListeners = new Vector();
570                 lookupListeners.add(listener);

572                 lookupThread.start();
573             } else {
574                 lookupListeners.add(listener);
575             }
576         }

583     private PrintService[] copyOf(PrintService[] inArr) {

586         } else {
587             PrintService []outArr = new PrintService[inArr.length];

593     public void run() {
594         PrintService[] services = getPrintServices();
595         synchronized (this) {
596             BackgroundLookupListener listener;
597             for (int i=0; i<lookupListeners.size(); i++) {
598                 listener =
599                     (BackgroundLookupListener)lookupListeners.elementAt(i);
600                 listener.notifyServices(copyOf(services));
601             }
602             lookupListeners = null;
603         }

646     private String getDefaultPrinterNameSysV() {
647         String defaultPrinter = "lp";
648         String command = "/usr/bin/lpstat -d";
649 
650         String [] names = execCmd(command);
651         if (names == null || names.length == 0) {
652             return defaultPrinter;
653         } else {
654             int index = names[0].indexOf(":");
655             if (index == -1  || (names[0].length() <= index+1)) {
656                 return null;
657             } else {
658                 String name = names[0].substring(index+1).trim();
659                 if (name.length() == 0) {
660                     return null;
661                 } else {
662                     return name;
663                 }
664             }
665         }
666     }

680     private String[] getAllPrinterNamesSysV() {
681         String defaultPrinter = "lp";
682         String command = "/usr/bin/lpstat -v|/usr/bin/expand|/usr/bin/cut -f3 -d' ' |/usr/bin/cut -f1 -d':' | /usr/bin/sort";
683 
684         String [] names = execCmd(command);
685         ArrayList printerNames = new ArrayList();
686         for (int i=0; i < names.length; i++) {
687             if (!names[i].equals("_default") &&
688                 !names[i].equals(defaultPrinter) &&

690                 printerNames.add(names[i]);
691             }
692         }
693         return (String[])printerNames.toArray(new String[printerNames.size()]);
694     }

696     static String[] execCmd(final String command) {
697         ArrayList results = null;
698         try {
699             final String[] cmd = new String[3];
700             if (isSysV()) {
701                 cmd[0] = "/usr/bin/sh";
702                 cmd[1] = "-c";
703                 cmd[2] = "env LC_ALL=C " + command;
704             } else {
705                 cmd[0] = "/bin/sh";
706                 cmd[1] = "-c";
707                 cmd[2] = "LC_ALL=C " + command;
708             }
709 
710             results = (ArrayList)AccessController.doPrivileged(
711                 new PrivilegedExceptionAction() {
712                     public Object run() throws IOException {
713 
714                         Process proc;
715                         BufferedReader bufferedReader = null;
716                         File f = File.createTempFile("prn","xc");
717                         cmd[2] = cmd[2]+">"+f.getAbsolutePath();
718 
719                         proc = Runtime.getRuntime().exec(cmd);
720                         try {
721                             boolean done = false; // in case of interrupt.
722                             while (!done) {
723                                 try {
724                                     proc.waitFor();
725                                     done = true;
726                                 } catch (InterruptedException e) {
727                                 }
728                             }
729 
730                             if (proc.exitValue() == 0) {
731                                 FileReader reader = new FileReader(f);
732                                 bufferedReader = new BufferedReader(reader);
733                                 String line;
734                                 ArrayList results = new ArrayList();
735                                 while ((line = bufferedReader.readLine())
736                                        != null) {
737                                     results.add(line);
738                                 }
739                                 return results;
740                             }
741                         } finally {
742                             f.delete();
743                             // promptly close all streams.
744                             if (bufferedReader != null) {
745                                 bufferedReader.close();
746                             }
747                             proc.getInputStream().close();
748                             proc.getErrorStream().close();
749                             proc.getOutputStream().close();
750                         }
751                         return null;
752                     }
753                 });
754         } catch (PrivilegedActionException e) {
755         }
756         if (results == null) {
757             return new String[0];
758         } else {
759             return (String[])results.toArray(new String[results.size()]);
760         }
761     }

763     private class PrinterChangeListener extends Thread {
764 
765         public void run() {
766             int refreshSecs;
767             while (true) {
768                 try {
769                     refreshServices();
770                 } catch (Exception se) {
771                     IPPPrintService.debug_println(debugPrefix+"Exception in refresh thread.");
772                     break;
773                 }
774 
775                 if ((printServices != null) &&
776                     (printServices.length > minRefreshTime)) {
777                     // compute new refresh time 1 printer = 1 sec
778                     refreshSecs = printServices.length;
779                 } else {
780                     refreshSecs = minRefreshTime;
781                 }
782                 try {
783                     sleep(refreshSecs * 1000);
784                 } catch (InterruptedException e) {
785                     break;
786                 }
787             }
788         }
