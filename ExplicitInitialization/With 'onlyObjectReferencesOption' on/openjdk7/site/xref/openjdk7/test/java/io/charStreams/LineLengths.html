24  /* @test
25     @summary Check line lengths using all recognized line-termination sequences
26   */
27  
28  import java.io.*;
29  
30  
31  public class LineLengths {
32  
33      static String enc = "UTF8";
34      static String inEnc = null;
35      static String outEnc = null;
36  
37      public static void main(String[] args) throws Exception {
38          PrintWriter log
39              = new PrintWriter(new OutputStreamWriter(System.err), true);
40  
41          if (inEnc == null)
42              inEnc = enc;
43          if (outEnc == null)
44              outEnc = enc;
45  
46          PipedOutputStream uo = new PipedOutputStream();
47          PipedInputStream ui = new PipedInputStream(uo);
48          PipedOutputStream co = new PipedOutputStream();
49          PipedInputStream ci = new PipedInputStream(co);
50  
51          BufferedWriter w = new BufferedWriter(new OutputStreamWriter(co, outEnc));
52          BufferedReader r = new BufferedReader(new InputStreamReader(ci, inEnc));
53  
54          Thread t1 = new Thread(new LineLengthsSource(uo, w, log));
55          Thread t2 = new Thread(new LineSink(ui, r, log));
56          t1.start();
57          t2.start();
58          t1.join();
59          t2.join();
60      }
61  
62  }
