44  /**
45   * A KDC server.
46   * <p>
47   * Features:
48   * <ol>
49   * <li> Supports TCP and UDP
50   * <li> Supports AS-REQ and TGS-REQ
51   * <li> Principal db and other settings hard coded in application
52   * <li> Options, say, request preauth or not
53   * </ol>
54   * Side effects:
55   * <ol>
56   * <li> The Sun-internal class <code>sun.security.krb5.Config</code> is a
57   * singleton and initialized according to Kerberos settings (krb5.conf and
58   * java.security.krb5.* system properties). This means once it's initialized
59   * it will not automatically notice any changes to these settings (or file
60   * changes of krb5.conf). The KDC class normally does not touch these
61   * settings (except for the <code>writeKtab()</code> method). However, to make
62   * sure nothing ever goes wrong, if you want to make any changes to these
63   * settings after calling a KDC method, call <code>Config.refresh()</code> to
64   * make sure your changes are reflected in the <code>Config</code> object.
65   * </ol>
66   * System properties recognized:
67   * <ul>
68   * <li>test.kdc.save.ccache
69   * </ul>
70   * Support policies:
71   * <ul>
72   * <li>ok-as-delegate
73   * </ul>
74   * Issues and TODOs:
75   * <ol>
76   * <li> Generates krb5.conf to be used on another machine, currently the kdc is
77   * always localhost
78   * <li> More options to KDC, say, error output, say, response nonce !=
79   * request nonce
80   * </ol>
81   * Note: This program uses internal krb5 classes (including reflection to
82   * access private fields and methods).
83   * <p>
84   * Usages:
85   * <p>
86   * 1. Init and start the KDC:
87   * <pre>
88   * KDC kdc = KDC.create("REALM.NAME", port, isDaemon);
89   * KDC kdc = KDC.create("REALM.NAME");
90   * </pre>
91   * Here, <code>port</code> is the UDP and TCP port number the KDC server
92   * listens on. If zero, a random port is chosen, which you can use getPort()
93   * later to retrieve the value.
94   * <p>
95   * If <code>isDaemon</code> is true, the KDC worker threads will be daemons.
96   * <p>
97   * The shortcut <code>KDC.create("REALM.NAME")</code> has port=0 and
98   * isDaemon=false, and is commonly used in an embedded KDC.
99   * <p>
100  * 2. Adding users:
101  * <pre>
102  * kdc.addPrincipal(String principal_name, char[] password);
103  * kdc.addPrincipalRandKey(String principal_name);
104  * </pre>
105  * A service principal's name should look like "host/f.q.d.n". The second form
106  * generates a random key. To expose this key, call <code>writeKtab()</code> to
107  * save the keys into a keytab file.
108  * <p>
109  * Note that you need to add the principal name krbtgt/REALM.NAME yourself.
110  * <p>
111  * Note that you can safely add a principal at any time after the KDC is
112  * started and before a user requests info on this principal.
113  * <p>
114  * 3. Other public methods:
115  * <ul>
116  * <li> <code>getPort</code>: Returns the port number the KDC uses
117  * <li> <code>getRealm</code>: Returns the realm name
118  * <li> <code>writeKtab</code>: Writes all principals' keys into a keytab file
119  * <li> <code>saveConfig</code>: Saves a krb5.conf file to access this KDC
120  * <li> <code>setOption</code>: Sets various options
121  * </ul>
122  * Read the javadoc for details. Lazy developer can use <code>OneKDC</code>
123  * directly.
124  */

230     /**
231      * Writes or appends KDC keys into a keytab. See doc for writeMultiKtab.
232      * @param append true if append, otherwise, overwrite.
233      */
234     private static void writeKtab0(String tab, boolean append, KDC... kdcs)
235             throws IOException, KrbException {
236         KeyTab ktab = append ? KeyTab.getInstance(tab) : KeyTab.create(tab);
237         for (KDC kdc: kdcs) {
238             for (String name : kdc.passwords.keySet()) {
239                 char[] pass = kdc.passwords.get(name);
240                 int kvno = 0;
241                 if (Character.isDigit(pass[pass.length-1])) {
242                     kvno = pass[pass.length-1] - '0';
243                 }
244                 ktab.addEntry(new PrincipalName(name,
245                         name.indexOf('/') < 0 ?
246                             PrincipalName.KRB_NT_UNKNOWN :
247                             PrincipalName.KRB_NT_SRV_HST),
248                             pass,
249                             kvno,
250                             true);
251             }
252         }
253         ktab.save();
254     }

345     /**
346      * Writes a krb5.conf for one or more KDC that includes KDC locations for
347      * each realm and the default realm name. You can also add extra strings
348      * into the file. The method should be called like:
349      * <pre>
350      *   KDC.saveConfig("krb5.conf", kdc1, kdc2, ..., line1, line2, ...);
351      * </pre>
352      * Here you can provide one or more kdc# and zero or more line# arguments.
353      * The line# will be put after [libdefaults] and before [realms]. Therefore
354      * you can append new lines into [libdefaults] and/or create your new
355      * stanzas as well. Note that a newline character will be appended to
356      * each line# argument.
357      * <p>
358      * For example:
359      * <pre>
360      * KDC.saveConfig("krb5.conf", this);
361      * </pre>
362      * generates:
363      * <pre>
364      * [libdefaults]
365      * default_realm = REALM.NAME
366      *
367      * [realms]
368      *   REALM.NAME = {
369      *     kdc = host:port_number
370      *   }
371      * </pre>
372      *
373      * Another example:
374      * <pre>
375      * KDC.saveConfig("krb5.conf", kdc1, kdc2, "forwardable = true", "",
376      *         "[domain_realm]",
377      *         ".kdc1.com = KDC1.NAME");
378      * </pre>
379      * generates:
380      * <pre>
381      * [libdefaults]
382      * default_realm = KDC1.NAME
383      * forwardable = true
384      *
385      * [domain_realm]
386      * .kdc1.com = KDC1.NAME
387      *
388      * [realms]
389      *   KDC1.NAME = {
390      *     kdc = host:port1
391      *   }
392      *   KDC2.NAME = {
393      *     kdc = host:port2
394      *   }
395      * </pre>
396      * @param file the name of the file to write into
397      * @param kdc the first (and default) KDC
398      * @param more more KDCs or extra lines (in their appearing order) to
399      * insert into the krb5.conf file. This method reads each argument's type
400      * to determine what it's for. This argument can be empty.
401      * @throws java.io.IOException for any file output error
402      */
403     public static void saveConfig(String file, KDC kdc, Object... more)
404             throws IOException {
405         File f = new File(file);
406         StringBuffer sb = new StringBuffer();
407         sb.append("[libdefaults]\ndefault_realm = ");
408         sb.append(kdc.realm);
409         sb.append("\n");
410         for (Object o: more) {
411             if (o instanceof String) {
412                 sb.append(o);
413                 sb.append("\n");
414             }
415         }
416         sb.append("\n[realms]\n");
417         sb.append(realmLineForKDC(kdc));
418         for (Object o: more) {
419             if (o instanceof KDC) {
420                 sb.append(realmLineForKDC((KDC)o));
421             }
422         }
423         FileOutputStream fos = new FileOutputStream(f);
424         fos.write(sb.toString().getBytes());
425         fos.close();
426     }

455     /**
456      * Generates a 32-char random password
457      * @return the password
458      */
459     private static char[] randomPassword() {
460         char[] pass = new char[32];
461         for (int i=0; i<31; i++)
462             pass[i] = (char)secureRandom.nextInt();
463         // The last char cannot be a number, otherwise, keyForUser()
464         // believes it's a sign of kvno
465         pass[31] = 'Z';
466         return pass;
467     }

469     /**
470      * Generates a random key for the given encryption type.
471      * @param eType the encryption type
472      * @return the generated key
473      * @throws sun.security.krb5.KrbException for unknown/unsupported etype
474      */
475     private static EncryptionKey generateRandomKey(int eType)
476             throws KrbException  {
477         // Is 32 enough for AES256? I should have generated the keys directly
478         // but different cryptos have different rules on what keys are valid.
479         char[] pass = randomPassword();
480         String algo;
481         switch (eType) {
482             case EncryptedData.ETYPE_DES_CBC_MD5: algo = "DES"; break;
483             case EncryptedData.ETYPE_DES3_CBC_HMAC_SHA1_KD: algo = "DESede"; break;
484             case EncryptedData.ETYPE_AES128_CTS_HMAC_SHA1_96: algo = "AES128"; break;
485             case EncryptedData.ETYPE_ARCFOUR_HMAC: algo = "ArcFourHMAC"; break;
486             case EncryptedData.ETYPE_AES256_CTS_HMAC_SHA1_96: algo = "AES256"; break;
487             default: algo = "DES"; break;
488         }
489         return new EncryptionKey(pass, "NOTHING", algo);    // Silly
490     }

492     /**
493      * Returns the password for a given principal
494      * @param p principal
495      * @return the password
496      * @throws sun.security.krb5.KrbException when the principal is not inside
497      *         the database.
498      */
499     private char[] getPassword(PrincipalName p, boolean server)
500             throws KrbException {
501         String pn = p.toString();
502         if (p.getRealmString() == null) {
503             pn = pn + "@" + getRealm();
504         }
505         char[] pass = passwords.get(pn);
506         if (pass == null) {
507             throw new KrbException(server?
508                 Krb5.KDC_ERR_S_PRINCIPAL_UNKNOWN:
509                 Krb5.KDC_ERR_C_PRINCIPAL_UNKNOWN);
510         }
511         return pass;
512     }

514     /**
515      * Returns the salt string for the principal.
516      * @param p principal
517      * @return the salt
518      */
519     private String getSalt(PrincipalName p) {
520         String pn = p.toString();
521         if (p.getRealmString() == null) {
522             pn = pn + "@" + getRealm();
523         }
524         if (passwords.containsKey(pn)) {
525             try {
526                 // Find the principal name with correct case.
527                 p = new PrincipalName(passwords.ceilingEntry(pn).getKey());
528             } catch (RealmException re) {
529                 // Won't happen
530             }
531         }
532         String s = p.getRealmString();
533         if (s == null) s = getRealm();
534         for (String n: p.getNameStrings()) {
535             s += n;
536         }
537         return s;
538     }

540     /**
541      * Returns the key for a given principal of the given encryption type
542      * @param p the principal
543      * @param etype the encryption type
544      * @param server looking for a server principal?
545      * @return the key
546      * @throws sun.security.krb5.KrbException for unknown/unsupported etype
547      */
548     private EncryptionKey keyForUser(PrincipalName p, int etype, boolean server)
549             throws KrbException {
550         try {
551             // Do not call EncryptionKey.acquireSecretKeys(), otherwise
552             // the krb5.conf config file would be loaded.
553             Integer kvno = null;
554             // For service whose password ending with a number, use it as kvno.
555             // Kvno must be postive.
556             if (p.toString().indexOf('/') > 0) {
557                 char[] pass = getPassword(p, server);
558                 if (Character.isDigit(pass[pass.length-1])) {
559                     kvno = pass[pass.length-1] - '0';
560                 }
561             }
562             return new EncryptionKey(EncryptionKeyDotStringToKey(
563                     getPassword(p, server), getSalt(p), null, etype),
564                     etype, kvno);
565         } catch (KrbException ke) {
566             throw ke;
567         } catch (Exception e) {
568             throw new RuntimeException(e);  // should not happen
569         }
570     }

581     /**
582      * If the provided client/server pair matches a rule
583      *
584      * A system property named test.kdc.policy.RULE will be consulted.
585      * If it's unset, returns false. If its value is "", any pair is
586      * matched. Otherwise, it should contains the server name matched.
587      *
588      * TODO: client name is not used currently.
589      *
590      * @param c client name
591      * @param s server name
592      * @param rule rule name
593      * @return if a match is found
594      */
595     private boolean configMatch(String c, String s, String rule) {
596         String policy = policies.get(rule);
597         boolean result = false;
598         if (policy == null) {
599             result = false;
600         } else if (policy.length() == 0) {
601             result = true;
602         } else {
603             String[] names = policy.split("\\s+");
604             for (String name: names) {
605                 if (name.equals(s)) {
606                     result = true;
607                     break;
608                 }
609             }
610         }
611         if (result) {
612             System.out.printf(">>>> Policy match result (%s vs %s on %s) %b\n",
613                     c, s, rule, result);
614         }
615         return result;
616     }

632     /**
633      * Processes a TGS_REQ and generates a TGS_REP (or KRB_ERROR)
634      * @param in the request
635      * @return the response
636      * @throws java.lang.Exception for various errors
637      */
638     private byte[] processTgsReq(byte[] in) throws Exception {
639         TGSReq tgsReq = new TGSReq(in);
640         try {
641             System.out.println(realm + "> " + tgsReq.reqBody.cname +
642                     " sends TGS-REQ for " +
643                     tgsReq.reqBody.sname);
644             KDCReqBody body = tgsReq.reqBody;
645             int[] eTypes = KDCReqBodyDotEType(body);
646             int e2 = eTypes[0];     // etype for outgoing session key
647             int e3 = eTypes[0];     // etype for outgoing ticket
648 
649             PAData[] pas = kDCReqDotPAData(tgsReq);
650 
651             Ticket tkt = null;
652             EncTicketPart etp = null;
653             if (pas == null || pas.length == 0) {
654                 throw new KrbException(Krb5.KDC_ERR_PADATA_TYPE_NOSUPP);
655             } else {
656                 for (PAData pa: pas) {
657                     if (pa.getType() == Krb5.PA_TGS_REQ) {
658                         APReq apReq = new APReq(pa.getValue());
659                         EncryptedData ed = apReq.authenticator;
660                         tkt = apReq.ticket;
661                         int te = tkt.encPart.getEType();
662                         tkt.sname.setRealm(tkt.realm);
663                         EncryptionKey kkey = keyForUser(tkt.sname, te, true);

1047      * Start the KDC service. This server listens on both UDP and TCP using
1048      * the same port number. It uses three threads to deal with requests.
1049      * They can be set to daemon threads if requested.
1050      * @param port the port number to listen to. If zero, a random available
1051      *  port no less than 8000 will be chosen and used.
1052      * @param asDaemon true if the KDC threads should be daemons
1053      * @throws java.io.IOException for any communication error
1054      */
1055     protected void startServer(int port, boolean asDaemon) throws IOException {
1056         if (port > 0) {
1057             u1 = new DatagramSocket(port, InetAddress.getByName("127.0.0.1"));
1058             t1 = new ServerSocket(port);
1059         } else {
1060             while (true) {
1061                 // Try to find a port number that's both TCP and UDP free
1062                 try {
1063                     port = 8000 + new java.util.Random().nextInt(10000);
1064                     u1 = null;
1065                     u1 = new DatagramSocket(port, InetAddress.getByName("127.0.0.1"));
1066                     t1 = new ServerSocket(port);
1067                     break;
1068                 } catch (Exception e) {
1069                     if (u1 != null) u1.close();
1070                 }
1071             }
1072         }
1073         final DatagramSocket udp = u1;
1074         final ServerSocket tcp = t1;
1075         System.out.println("Start KDC on " + port);
1076 
1077         this.port = port;
1078 
1079         // The UDP consumer
1080         thread1 = new Thread() {
1081             public void run() {
1082                 while (true) {
1083                     try {
1084                         byte[] inbuf = new byte[8192];
1085                         DatagramPacket p = new DatagramPacket(inbuf, inbuf.length);
1086                         udp.receive(p);
1087                         System.out.println("-----------------------------------------------");
1088                         System.out.println(">>>>> UDP packet received");
1089                         q.put(new Job(processMessage(Arrays.copyOf(inbuf, p.getLength())), udp, p));
1090                     } catch (Exception e) {
1091                         e.printStackTrace();
1092                     }
1093                 }
1094             }
1095         };
1096         thread1.setDaemon(asDaemon);
1097         thread1.start();
1098 
1099         // The TCP consumer
1100         thread2 = new Thread() {
1101             public void run() {
1102                 while (true) {
1103                     try {
1104                         Socket socket = tcp.accept();
1105                         System.out.println("-----------------------------------------------");
1106                         System.out.println(">>>>> TCP connection established");
1107                         DataInputStream in = new DataInputStream(socket.getInputStream());
1108                         DataOutputStream out = new DataOutputStream(socket.getOutputStream());
1109                         byte[] token = new byte[in.readInt()];
1110                         in.readFully(token);
1111                         q.put(new Job(processMessage(token), socket, out));
1112                     } catch (Exception e) {
1113                         e.printStackTrace();
1114                     }
1115                 }
1116             }
1117         };
1118         thread2.setDaemon(asDaemon);
1119         thread2.start();
1120 
1121         // The dispatcher
1122         thread3 = new Thread() {
1123             public void run() {
1124                 while (true) {
1125                     try {
1126                         q.take().send();
1127                     } catch (Exception e) {
1128                     }
1129                 }
1130             }
1131         };
1132         thread3.setDaemon(true);
1133         thread3.start();
1134     }

1164             this.out = out;

1172             this.dp = dp;
1173         }

1176         void send() {
1177             try {
1178                 if (useTCP) {
1179                     System.out.println(">>>>> TCP request honored");
1180                     out.writeInt(token.length);
1181                     out.write(token);
1182                     s.close();
1183                 } else {
1184                     System.out.println(">>>>> UDP request honored");
1185                     s2.send(new DatagramPacket(token, token.length, dp.getAddress(), dp.getPort()));
1186                 }
1187             } catch (Exception e) {
1188                 e.printStackTrace();
1189             }
1190         }
