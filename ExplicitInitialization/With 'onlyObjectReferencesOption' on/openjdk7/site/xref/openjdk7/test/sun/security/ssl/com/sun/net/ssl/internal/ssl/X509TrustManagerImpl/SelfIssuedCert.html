153     /*
154      * Define the server side of the test.
155      *
156      * If the server prematurely exits, serverReady will be set to true
157      * to avoid infinite hangs.
158      */
159     void doServerSide() throws Exception {
160         SSLContext context = getSSLContext(null, targetCertStr,
161                                             targetPrivateKey);
162         SSLServerSocketFactory sslssf = context.getServerSocketFactory();
163 
164         SSLServerSocket sslServerSocket =
165             (SSLServerSocket)sslssf.createServerSocket(serverPort);
166         serverPort = sslServerSocket.getLocalPort();
167 
168         /*
169          * Signal Client, we're ready for his connect.
170          */
171         serverReady = true;
172 
173         SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
174         sslSocket.setNeedClientAuth(false);
175 
176         InputStream sslIS = sslSocket.getInputStream();
177         OutputStream sslOS = sslSocket.getOutputStream();
178 
179         sslIS.read();
180         sslOS.write(85);
181         sslOS.flush();
182 
183         sslSocket.close();
184 
185     }

187     /*
188      * Define the client side of the test.
189      *
190      * If the server prematurely exits, serverReady will be set to true
191      * to avoid infinite hangs.
192      */
193     void doClientSide() throws Exception {
194         /*
195          * Wait for server to get started.
196          */
197         while (!serverReady) {
198             Thread.sleep(50);
199         }
200 
201         SSLContext context = getSSLContext(trusedCertStr, null, null);
202         SSLSocketFactory sslsf = context.getSocketFactory();
203 
204         SSLSocket sslSocket =
205             (SSLSocket)sslsf.createSocket("localhost", serverPort);
206 
207         InputStream sslIS = sslSocket.getInputStream();
208         OutputStream sslOS = sslSocket.getOutputStream();
209 
210         sslOS.write(280);
211         sslOS.flush();
212         sslIS.read();
213 
214         sslSocket.close();
215     }

217     // get the ssl context
218     private static SSLContext getSSLContext(String trusedCertStr,
219             String keyCertStr, String keySpecStr) throws Exception {
220 
221         // generate certificate from cert string
222         CertificateFactory cf = CertificateFactory.getInstance("X.509");
223 
224         // create a key store
225         KeyStore ks = KeyStore.getInstance("JKS");
226         ks.load(null, null);
227 
228         // import the trused cert
229         Certificate trusedCert = null;
230         ByteArrayInputStream is = null;
231         if (trusedCertStr != null) {
232             is = new ByteArrayInputStream(trusedCertStr.getBytes());
233             trusedCert = cf.generateCertificate(is);
234             is.close();
235 
236             ks.setCertificateEntry("RSA Export Signer", trusedCert);
237         }
238 
239         if (keyCertStr != null) {
240             // generate the private key.
241             PKCS8EncodedKeySpec priKeySpec = new PKCS8EncodedKeySpec(
242                                 new BASE64Decoder().decodeBuffer(keySpecStr));
243             KeyFactory kf = KeyFactory.getInstance("RSA");
244             RSAPrivateKey priKey =
245                     (RSAPrivateKey)kf.generatePrivate(priKeySpec);
246 
247             // generate certificate chain
248             is = new ByteArrayInputStream(keyCertStr.getBytes());
249             Certificate keyCert = cf.generateCertificate(is);
250             is.close();
251 
252             Certificate[] chain = null;
253             if (trusedCert != null) {
254                 chain = new Certificate[2];
255                 chain[0] = keyCert;
256                 chain[1] = trusedCert;
257             } else {
258                 chain = new Certificate[1];
259                 chain[0] = keyCert;
260             }
261 
262             // import the key entry.
263             ks.setKeyEntry("Whatever", priKey, passphrase, chain);
264         }
265 
266         // create SSL context
267         TrustManagerFactory tmf = TrustManagerFactory.getInstance(tmAlgorithm);
268         tmf.init(ks);
269 
270         SSLContext ctx = SSLContext.getInstance("TLS");
271         if (keyCertStr != null && !keyCertStr.isEmpty()) {
272             KeyManagerFactory kmf = KeyManagerFactory.getInstance("NewSunX509");
273             kmf.init(ks, passphrase);
274 
275             ctx.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
276             ks = null;
277         } else {
278             ctx.init(null, tmf.getTrustManagers(), null);
279         }
280 
281         return ctx;
282     }

317     Thread clientThread = null;
318     Thread serverThread = null;
319     /*
320      * Primary constructor, used to drive remainder of the test.
321      *
322      * Fork off the other side, then do your work.
323      */
324     SelfIssuedCert() throws Exception {
325         if (separateServerThread) {
326             startServer(true);
327             startClient(false);
328         } else {
329             startClient(true);
330             startServer(false);
331         }
332 
333         /*
334          * Wait for other side to close down.
335          */
336         if (separateServerThread) {
337             serverThread.join();
338         } else {
339             clientThread.join();
340         }
341 
342         /*
343          * When we get here, the test is pretty much over.
344          *
345          * If the main thread excepted, that propagates back
346          * immediately.  If the other thread threw an exception, we
347          * should report back.
348          */
349         if (serverException != null)
350             throw serverException;
351         if (clientException != null)
352             throw clientException;
353     }
354 
355     void startServer(boolean newThread) throws Exception {
356         if (newThread) {
357             serverThread = new Thread() {
358                 public void run() {
359                     try {
360                         doServerSide();
361                     } catch (Exception e) {
362                         /*
363                          * Our server thread just died.
364                          *
365                          * Release the client, if not active already...
366                          */
367                         System.err.println("Server died...");
368                         serverReady = true;
369                         serverException = e;
370                     }
371                 }
372             };
373             serverThread.start();
374         } else {
375             doServerSide();
376         }
377     }
378 
379     void startClient(boolean newThread) throws Exception {
380         if (newThread) {
381             clientThread = new Thread() {
382                 public void run() {
383                     try {
384                         doClientSide();
385                     } catch (Exception e) {
386                         /*
387                          * Our client thread just died.
388                          */
389                         System.err.println("Client died...");
390                         clientException = e;
391                     }
392                 }
393             };
394             clientThread.start();
395         } else {
396             doClientSide();
397         }
398     }
