146     public void setBounds(int x, int y, int width, int height, int op) {
147         // Should set paintPending before reahape to prevent
148         // thread race between paint events
149         // Native components do redraw after resize
150         paintPending = (width != oldWidth) || (height != oldHeight);
151 
152         if ( (op & NO_EMBEDDED_CHECK) != 0 ) {
153             reshapeNoCheck(x, y, width, height);
154         } else {
155             reshape(x, y, width, height);
156         }
157         if ((width != oldWidth) || (height != oldHeight)) {
158             // Only recreate surfaceData if this setBounds is called
159             // for a resize; a simple move should not trigger a recreation
160             try {
161                 replaceSurfaceData();
162             } catch (InvalidPipeException e) {
163                 // REMIND : what do we do if our surface creation failed?
164             }
165             oldWidth = width;
166             oldHeight = height;
167         }
168 
169         serialNum++;
170     }

172     /*
173      * Called from native code (on Toolkit thread) in order to
174      * dynamically layout the Container during resizing
175      */
176     void dynamicallyLayoutContainer() {
177         // If we got the WM_SIZING, this must be a Container, right?
178         // In fact, it must be the top-level Container.
179         if (log.isLoggable(PlatformLogger.Level.FINE)) {
180             Container parent = WToolkit.getNativeContainer((Component)target);
181             if (parent != null) {
182                 log.fine("Assertion (parent == null) failed");
183             }
184         }
185         final Container cont = (Container)target;
186 
187         WToolkit.executeOnEventHandlerThread(cont, new Runnable() {
188             public void run() {
189                 // Discarding old paint events doesn't seem to be necessary.
190                 cont.invalidate();
191                 cont.validate();
192 
193                 if (surfaceData instanceof D3DSurfaceData.D3DWindowSurfaceData ||
194                     surfaceData instanceof OGLSurfaceData)
195                 {
196                     // When OGL or D3D is enabled, it is necessary to
197                     // replace the SurfaceData for each dynamic layout
198                     // request so that the viewport stays in sync
199                     // with the window bounds.
200                     try {
201                         replaceSurfaceData();
202                     } catch (InvalidPipeException e) {
203                         // REMIND: this is unlikely to occur for OGL, but
204                         // what do we do if surface creation fails?
205                     }
206                 }
207 
208                 // Forcing a paint here doesn't seem to be necessary.
209                 // paintDamagedAreaImmediately();
210             }
211         });
212     }

238     private static final double BANDING_DIVISOR = 4.0;
239     private native int[] createPrintedPixels(int srcX, int srcY,
240                                              int srcW, int srcH,
241                                              int alpha);
242     public void print(Graphics g) {
243 
244         Component comp = (Component)target;
245 
246         // To conserve memory usage, we will band the image.
247 
248         int totalW = comp.getWidth();
249         int totalH = comp.getHeight();
250 
251         int hInc = (int)(totalH / BANDING_DIVISOR);
252         if (hInc == 0) {
253             hInc = totalH;
254         }
255 
256         for (int startY = 0; startY < totalH; startY += hInc) {
257             int endY = startY + hInc - 1;
258             if (endY >= totalH) {
259                 endY = totalH - 1;
260             }
261             int h = endY - startY + 1;
262 
263             Color bgColor = comp.getBackground();
264             int[] pix = createPrintedPixels(0, startY, totalW, h,
265                                             bgColor == null ? 255 : bgColor.getAlpha());
266             if (pix != null) {
267                 BufferedImage bim = new BufferedImage(totalW, h,
268                                               BufferedImage.TYPE_INT_ARGB);
269                 bim.setRGB(0, 0, totalW, h, pix, 0, totalW);
270                 g.drawImage(bim, 0, startY, null);
271                 bim.flush();
272             }
273         }
274 
275         comp.print(g);
276     }

278     public void coalescePaintEvent(PaintEvent e) {
279         Rectangle r = e.getUpdateRect();
280         if (!(e instanceof IgnorePaintEvent)) {
281             paintArea.add(r, e.getID());
282         }
283 
284         if (log.isLoggable(PlatformLogger.Level.FINEST)) {
285             switch(e.getID()) {
286             case PaintEvent.UPDATE:
287                 log.finest("coalescePaintEvent: UPDATE: add: x = " +
288                     r.x + ", y = " + r.y + ", width = " + r.width + ", height = " + r.height);
289                 return;
290             case PaintEvent.PAINT:
291                 log.finest("coalescePaintEvent: PAINT: add: x = " +
292                     r.x + ", y = " + r.y + ", width = " + r.width + ", height = " + r.height);
293                 return;
294             }
295         }
296     }

305     public void handleJavaMouseEvent(MouseEvent e) {
306         switch (e.getID()) {
307           case MouseEvent.MOUSE_PRESSED:
308               // Note that Swing requests focus in its own mouse event handler.
309               if (target == e.getSource() &&
310                   !((Component)target).isFocusOwner() &&
311                   WKeyboardFocusManagerPeer.shouldFocusOnClick((Component)target))
312               {
313                   WKeyboardFocusManagerPeer.requestFocusFor((Component)target,
314                                                             CausedFocusEvent.Cause.MOUSE_EVENT);
315               }
316               break;
317         }
318     }

322     @SuppressWarnings("fallthrough")
323     public void handleEvent(AWTEvent e) {
324         int id = e.getID();
325 
326         if ((e instanceof InputEvent) && !((InputEvent)e).isConsumed() &&
327             ((Component)target).isEnabled())
328         {
329             if (e instanceof MouseEvent && !(e instanceof MouseWheelEvent)) {
330                 handleJavaMouseEvent((MouseEvent) e);
331             } else if (e instanceof KeyEvent) {
332                 if (handleJavaKeyEvent((KeyEvent)e)) {
333                     return;
334                 }
335             }
336         }
337 
338         switch(id) {
339             case PaintEvent.PAINT:
340                 // Got native painting
341                 paintPending = false;
342                 // Fallthrough to next statement
343             case PaintEvent.UPDATE:
344                 // Skip all painting while layouting and all UPDATEs
345                 // while waiting for native paint
346                 if (!isLayouting && ! paintPending) {
347                     paintArea.paint(target,shouldClearRectBeforePaint());
348                 }
349                 return;
350             case FocusEvent.FOCUS_LOST:
351             case FocusEvent.FOCUS_GAINED:
352                 handleJavaFocusEvent((FocusEvent)e);
353             default:
354             break;
355         }
356 
357         // Call the native code
358         nativeHandleEvent(e);
359     }

430     /**
431      * Multi-buffer version of replaceSurfaceData.  This version is called
432      * by createBuffers(), which needs to acquire the same locks in the same
433      * order, but also needs to perform additional functions inside the
434      * locks.
435      */
436     public void replaceSurfaceData(int newNumBackBuffers,
437                                    BufferCapabilities caps)
438     {
439         SurfaceData oldData = null;
440         VolatileImage oldBB = null;
441         synchronized(((Component)target).getTreeLock()) {
442             synchronized(this) {
443                 if (pData == 0) {
444                     return;
445                 }
446                 numBackBuffers = newNumBackBuffers;
447                 ScreenUpdateManager mgr = ScreenUpdateManager.getInstance();
448                 oldData = surfaceData;
449                 mgr.dropScreenSurface(oldData);
450                 createScreenSurface(true);
451                 if (oldData != null) {
452                     oldData.invalidate();
453                 }
454 
455                 oldBB = backBuffer;
456                 if (numBackBuffers > 0) {
457                     // set the caps first, they're used when creating the bb
458                     backBufferCaps = caps;
459                     Win32GraphicsConfig gc =
460                         (Win32GraphicsConfig)getGraphicsConfiguration();
461                     backBuffer = gc.createBackBuffer(this);
462                 } else if (backBuffer != null) {
463                     backBufferCaps = null;
464                     backBuffer = null;
465                 }
466             }
467         }
468         // it would be better to do this before we create new ones,
469         // but then we'd run into deadlock issues
470         if (oldData != null) {
471             oldData.flush();
472             // null out the old data to make it collected faster
473             oldData = null;
474         }
475         if (oldBB != null) {
476             oldBB.flush();
477             // null out the old data to make it collected faster
478             oldData = null;
479         }
480     }

483         Runnable r = new Runnable() {
484             public void run() {
485                 // Shouldn't do anything if object is disposed in meanwhile
486                 // No need for sync as disposeAction in Window is performed
487                 // on EDT
488                 if (!isDisposed()) {
489                     try {
490                         replaceSurfaceData();
491                     } catch (InvalidPipeException e) {
492                         // REMIND : what do we do if our surface creation failed?
493                     }
494                 }
495             }
496         };
497         Component c = (Component)target;
498         // Fix 6255371.
499         if (!PaintEventDispatcher.getPaintEventDispatcher().queueSurfaceDataReplacing(c, r)) {
500             postEvent(new InvocationEvent(c, r));
501         }
502     }

552     @SuppressWarnings("deprecation")
553     public Graphics getGraphics() {
554         if (isDisposed()) {
555             return null;
556         }
557 
558         Component target = (Component)getTarget();
559         Window window = SunToolkit.getContainingWindow(target);
560         if (window != null) {
561             Graphics g =
562                 ((WWindowPeer)window.getPeer()).getTranslucentGraphics();
563             // getTranslucentGraphics() returns non-null value for non-opaque windows only
564             if (g != null) {
565                 // Non-opaque windows do not support heavyweight children.
566                 // Redirect all painting to the Window's Graphics instead.
567                 // The caller is responsible for calling the
568                 // WindowPeer.updateWindow() after painting has finished.
569                 int x = 0, y = 0;
570                 for (Component c = target; c != window; c = c.getParent()) {
571                     x += c.getX();
572                     y += c.getY();
573                 }
574 
575                 g.translate(x, y);
576                 g.clipRect(0, 0, target.getWidth(), target.getHeight());
577 
578                 return g;
579             }
580         }
581 
582         SurfaceData surfaceData = this.surfaceData;
583         if (surfaceData != null) {
584             /* Fix for bug 4746122. Color and Font shouldn't be null */
585             Color bgColor = background;
586             if (bgColor == null) {
587                 bgColor = SystemColor.window;
588             }
589             Color fgColor = foreground;
590             if (fgColor == null) {
591                 fgColor = SystemColor.windowText;
592             }
593             Font font = this.font;
594             if (font == null) {
595                 font = defaultFont;
596             }
597             ScreenUpdateManager mgr =
598                 ScreenUpdateManager.getInstance();
599             return mgr.createGraphics(surfaceData, this, fgColor,
600                                       bgColor, font);
601         }
602         return null;
603     }

659     // TODO: consider moving it to KeyboardFocusManagerPeerImpl
660     @SuppressWarnings("deprecation")
661     public boolean requestFocus(Component lightweightChild, boolean temporary,
662                                 boolean focusedWindowChangeAllowed, long time,
663                                 CausedFocusEvent.Cause cause)
664     {
665         if (WKeyboardFocusManagerPeer.
666             processSynchronousLightweightTransfer((Component)target, lightweightChild, temporary,
667                                                   focusedWindowChangeAllowed, time))
668         {
669             return true;
670         }
671 
672         int result = WKeyboardFocusManagerPeer
673             .shouldNativelyFocusHeavyweight((Component)target, lightweightChild,
674                                             temporary, focusedWindowChangeAllowed,
675                                             time, cause);
676 
677         switch (result) {
678           case WKeyboardFocusManagerPeer.SNFH_FAILURE:
679               return false;
680           case WKeyboardFocusManagerPeer.SNFH_SUCCESS_PROCEED:
681               if (focusLog.isLoggable(PlatformLogger.Level.FINER)) {
682                   focusLog.finer("Proceeding with request to " + lightweightChild + " in " + target);
683               }
684               Window parentWindow = SunToolkit.getContainingWindow((Component)target);
685               if (parentWindow == null) {
686                   return rejectFocusRequestHelper("WARNING: Parent window is null");
687               }
688               WWindowPeer wpeer = (WWindowPeer)parentWindow.getPeer();
689               if (wpeer == null) {
690                   return rejectFocusRequestHelper("WARNING: Parent window's peer is null");
691               }
692               boolean res = wpeer.requestWindowFocus(cause);
693 
694               if (focusLog.isLoggable(PlatformLogger.Level.FINER)) {
695                   focusLog.finer("Requested window focus: " + res);
696               }
697               // If parent window can be made focused and has been made focused(synchronously)
698               // then we can proceed with children, otherwise we retreat.
699               if (!(res && parentWindow.isFocused())) {
700                   return rejectFocusRequestHelper("Waiting for asynchronous processing of the request");
701               }
702               return WKeyboardFocusManagerPeer.deliverFocus(lightweightChild,
703                                                             (Component)target,
704                                                             temporary,
705                                                             focusedWindowChangeAllowed,
706                                                             time, cause);
707 
708           case WKeyboardFocusManagerPeer.SNFH_SUCCESS_HANDLED:
709               // Either lightweight or excessive request - all events are generated.
710               return true;
711         }
712         return false;
713     }

796     void initialize() {
797         if (((Component)target).isVisible()) {
798             show();  // the wnd starts hidden
799         }
800         Color fg = ((Component)target).getForeground();
801         if (fg != null) {
802             setForeground(fg);
803         }
804         // Set background color in C++, to avoid inheriting a parent's color.
805         Font  f = ((Component)target).getFont();
806         if (f != null) {
807             setFont(f);
808         }
809         if (! ((Component)target).isEnabled()) {
810             disable();
811         }
812         Rectangle r = ((Component)target).getBounds();
813         setBounds(r.x, r.y, r.width, r.height, SET_BOUNDS);
814     }

844     private void postPaintIfNecessary(int x, int y, int w, int h) {
845         if ( !AWTAccessor.getComponentAccessor().getIgnoreRepaint( (Component) target) ) {
846             PaintEvent event = PaintEventDispatcher.getPaintEventDispatcher().
847                 createPaintEvent((Component)target, x, y, w, h);
848             if (event != null) {
849                 postEvent(event);
850             }
851         }
852     }

871         if(!paintArea.isEmpty() && !paintPending &&
872             !((Component)target).getIgnoreRepaint()) {

1020     /**
1021      * Returns whether this component is capable of being hw accelerated.
1022      * More specifically, whether rendering to this component or a
1023      * BufferStrategy's back-buffer for this component can be hw accelerated.
1024      *
1025      * Conditions which could prevent hw acceleration include the toplevel
1026      * window containing this component being
1027      * {@link GraphicsDevice.WindowTranslucency#PERPIXEL_TRANSLUCENT
1028      * PERPIXEL_TRANSLUCENT}.
1029      *
1030      * Another condition is if Xor paint mode was detected when rendering
1031      * to an on-screen accelerated surface associated with this peer.
1032      * in this case both on- and off-screen acceleration for this peer is
1033      * disabled.
1034      *
1035      * @return {@code true} if this component is capable of being hw
1036      * accelerated, {@code false} otherwise
1037      * @see GraphicsDevice.WindowTranslucency#PERPIXEL_TRANSLUCENT
1038      */
1039     public boolean isAccelCapable() {
1040         if (!isAccelCapable ||
1041             !isContainingTopLevelAccelCapable((Component)target))
1042         {
1043             return false;
1044         }
1045 
1046         boolean isTranslucent =
1047             SunToolkit.isContainingTopLevelTranslucent((Component)target);
1048         // D3D/OGL and translucent windows interacted poorly in Windows XP;
1049         // these problems are no longer present in Vista
1050         return !isTranslucent || Win32GraphicsEnvironment.isVistaOS();
1051     }

1054      * Disables acceleration for this peer.
