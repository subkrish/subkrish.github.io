185     // Notification handler
186     public void handleNotification(Notification notification,
187                                    Object handback) {
188         MonitorNotification n = (MonitorNotification) notification;
189         echo("\tInside handleNotification...");
190         String type = n.getType();
191         try {
192             if (type.equals(MonitorNotification.
193                             THRESHOLD_VALUE_EXCEEDED)) {
194                 echo("\t\t" + n.getObservedAttribute() +
195                      " has reached or exceeded the threshold");
196                 echo("\t\tDerived Gauge = " + n.getDerivedGauge());
197                 echo("\t\tTrigger = " + n.getTrigger());
198                 counterMessageReceived = true;
199             } else if (type.equals(MonitorNotification.
200                                    THRESHOLD_HIGH_VALUE_EXCEEDED)) {
201                 echo("\t\t" + n.getObservedAttribute() +
202                      " has reached or exceeded the high threshold");
203                 echo("\t\tDerived Gauge = " + n.getDerivedGauge());
204                 echo("\t\tTrigger = " + n.getTrigger());
205                 gaugeMessageReceived = true;
206             } else if (type.equals(MonitorNotification.
207                                    STRING_TO_COMPARE_VALUE_MATCHED)) {
208                 echo("\t\t" + n.getObservedAttribute() +
209                      " matches the string-to-compare value");
210                 echo("\t\tDerived Gauge = " + n.getDerivedGauge());
211                 echo("\t\tTrigger = " + n.getTrigger());
212                 stringMessageReceived = true;
213             } else {
214                 echo("\t\tSkipping notification of type: " + type);
215             }
216         } catch (Exception e) {
217             echo("\tError in handleNotification!");
218             e.printStackTrace(System.out);
219         }
220     }

242

244             counterMonitor.addNotificationListener(this, null, null);

263                     break;
264                 case 3:
265                     ca.setArrayAttribute(new Integer[0]);
266                     obsObj.setComplexAttribute(ca);
267                     break;
268             }
269             server.registerMBean(obsObj, obsObjName);
270 
271             echo(">>> SET the attributes of the CounterMonitor:");
272 
273             counterMonitor.addObservedObject(obsObjName);
274             echo("\tATTRIBUTE \"ObservedObject\"    = " + obsObjName);

276             switch (testCase) {

730     public static Object elementFromComplex(Object complex, String element)
731     throws AttributeNotFoundException {
732         try {
733             if (complex.getClass().isArray() && element.equals("length")) {
734                 return Array.getLength(complex);
735             } else if (complex instanceof CompositeData) {
736                 return ((CompositeData) complex).get(element);
737             } else {
738                 // Java Beans introspection
739                 //
740                 BeanInfo bi = java.beans.Introspector.getBeanInfo(complex.getClass());
741                 PropertyDescriptor[] pds = bi.getPropertyDescriptors();
742                 System.out.println("PDs: " + pds.length);
743                 for (PropertyDescriptor pd : pds) {
744                     System.out.println("Property: " + pd.getName());
745                     if (pd.getName().equals(element))
746                         return pd.getReadMethod().invoke(complex);
747                 }
748                 throw new AttributeNotFoundException(
749                     "Could not find the getter method for the property " +
750                     element + " using the Java Beans introspector");
751             }
752         } catch (InvocationTargetException e) {
753             throw new IllegalArgumentException(e);
754         } catch (AttributeNotFoundException e) {
755             throw e;
756         } catch (Exception e) {
757             AttributeNotFoundException anfe =
758                 new AttributeNotFoundException(e.getMessage());
759             anfe.initCause(e);
760             throw anfe;
761         }
762     }
