134     @SuppressWarnings("AssignmentToForLoopParameter")
135     public static void main(String[] args) throws Exception {
136         if (args.length != 0) {
137             String currentArg = null;
138             try {
139                 for (int i = 0; i < args.length; i++) {
140                     currentArg = args[i];
141                     switch (currentArg) {
142                     case "-draft":
143                         String draftDataType = args[++i];
144                         try {
145                             DraftType.setDefault(draftDataType);
146                         } catch (NullPointerException e) {
147                             severe("Error: incorrect draft value: %s%n", draftDataType);
148                             System.exit(1);
149                         }
150                         info("Using the specified data type: %s%n", draftDataType);
151                         break;
152 
153                     case "-base":
154                         // base directory for input files
155                         CLDR_BASE = args[++i];
156                         if (!CLDR_BASE.endsWith("/")) {
157                             CLDR_BASE += "/";
158                         }
159                         break;
160 
161                     case "-baselocales":
162                         // base locales
163                         setupBaseLocales(args[++i]);
164                         break;
165 
166                     case "-basemodule":
167                         // indicates java.base module resource generation
168                         isBaseModule = true;
169                         break;
170 
171                     case "-o":
172                         // output directory
173                         DESTINATION_DIR = args[++i];
174                         break;
175 
176                     case "-utf8":
177                         USE_UTF8 = true;
178                         break;
179 
180                     case "-verbose":
181                         verbose = true;
182                         break;
183 
184                     case "-help":
185                         usage();
186                         System.exit(0);
187                         break;
188 
189                     default:
190                         throw new RuntimeException();
191                     }
192                 }
193             } catch (RuntimeException e) {
194                 severe("unknown or imcomplete arg(s): " + currentArg);
195                 usage();
196                 System.exit(1);
197             }
198         }
199 
200         // Set up path names
201         LOCAL_LDML_DTD = CLDR_BASE + "/dtd/ldml.dtd";
202         LOCAL_SPPL_LDML_DTD = CLDR_BASE + "/dtd/ldmlSupplemental.dtd";
203         SOURCE_FILE_DIR = CLDR_BASE + "/main";
204         SPPL_SOURCE_FILE = CLDR_BASE + "/supplemental/supplementalData.xml";
205         LIKELYSUBTAGS_SOURCE_FILE = CLDR_BASE + "/supplemental/likelySubtags.xml";
206         NUMBERING_SOURCE_FILE = CLDR_BASE + "/supplemental/numberingSystems.xml";
207         METAZONES_SOURCE_FILE = CLDR_BASE + "/supplemental/metaZones.xml";
208 
209         if (BASE_LOCALES.isEmpty()) {
210             setupBaseLocales("en-US");
211         }
212 
213         bundleGenerator = new ResourceBundleGenerator();
214 
215         // Parse data independent of locales
216         parseSupplemental();
217 
218         List<Bundle> bundles = readBundleList();
219         convertBundles(bundles);
220     }

287     private static List<Bundle> readBundleList() throws Exception {
288         List<Bundle> retList = new ArrayList<>();
289         Path path = FileSystems.getDefault().getPath(SOURCE_FILE_DIR);
290         try (DirectoryStream<Path> dirStr = Files.newDirectoryStream(path)) {
291             for (Path entry : dirStr) {
292                 String fileName = entry.getFileName().toString();
293                 if (fileName.endsWith(".xml")) {
294                     String id = fileName.substring(0, fileName.indexOf('.'));
295                     Locale cldrLoc = Locale.forLanguageTag(toLanguageTag(id));
296                     List<Locale> candList = applyParentLocales("", defCon.getCandidateLocales("", cldrLoc));
297                     StringBuilder sb = new StringBuilder();
298                     for (Locale loc : candList) {
299                         if (!loc.equals(Locale.ROOT)) {
300                             sb.append(toLocaleName(loc.toLanguageTag()));
301                             sb.append(",");
302                         }
303                     }
304                     if (sb.indexOf("root") == -1) {
305                         sb.append("root");
306                     }
307                     Bundle b = new Bundle(id, sb.toString(), null, null);
308                     // Insert the bundle for root at the top so that it will get
309                     // processed first.
310                     if ("root".equals(id)) {
311                         retList.add(0, b);
312                     } else {
313                         retList.add(b);
314                     }
315                 }
316             }
317         }
318         return retList;
319     }

323     static Map<String, Object> getCLDRBundle(String id) throws Exception {
324         Map<String, Object> bundle = cldrBundles.get(id);
325         if (bundle != null) {
326             return bundle;
327         }
328         SAXParserFactory factory = SAXParserFactory.newInstance();
329         factory.setValidating(true);
330         SAXParser parser = factory.newSAXParser();
331         enableFileAccess(parser);
332         LDMLParseHandler handler = new LDMLParseHandler(id);
333         File file = new File(SOURCE_FILE_DIR + File.separator + id + ".xml");
334         if (!file.exists()) {
335             // Skip if the file doesn't exist.
336             return Collections.emptyMap();
337         }
338 
339         info("..... main directory .....");
340         info("Reading file " + file);
341         parser.parse(file, handler);
342 
343         bundle = handler.getData();
344         cldrBundles.put(id, bundle);
345         String country = getCountryCode(id);
346         if (country != null) {
347             bundle = handlerSuppl.getData(country);
348             if (bundle != null) {
349                 //merge two maps into one map
350                 Map<String, Object> temp = cldrBundles.remove(id);
351                 bundle.putAll(temp);
352                 cldrBundles.put(id, bundle);
353             }
354         }
355         return bundle;
356     }

358     // Parsers for data in "supplemental" directory
359     //
360     private static void parseSupplemental() throws Exception {
361         // Parse SupplementalData file and store the information in the HashMap
362         // Calendar information such as firstDay and minDay are stored in
363         // supplementalData.xml as of CLDR1.4. Individual territory is listed
364         // with its ISO 3166 country code while default is listed using UNM49
365         // region and composition numerical code (001 for World.)
366         //
367         // SupplementalData file also provides the "parent" locales which
368         // are othrwise not to be fallen back. Process them here as well.
369         //
370         info("..... Parsing supplementalData.xml .....");
371         SAXParserFactory factorySuppl = SAXParserFactory.newInstance();
372         factorySuppl.setValidating(true);
373         SAXParser parserSuppl = factorySuppl.newSAXParser();
374         enableFileAccess(parserSuppl);
375         handlerSuppl = new SupplementDataParseHandler();
376         File fileSupply = new File(SPPL_SOURCE_FILE);
377         parserSuppl.parse(fileSupply, handlerSuppl);
378         Map<String, Object> parentData = handlerSuppl.getData("root");
379         parentData.keySet().forEach(key -> {
380                 parentLocalesMap.put(key, new TreeSet(
381                     Arrays.asList(((String)parentData.get(key)).split(" "))));
382             });
383 
384         // Parse numberingSystems to get digit zero character information.
385         SAXParserFactory numberingParser = SAXParserFactory.newInstance();
386         numberingParser.setValidating(true);
387         SAXParser parserNumbering = numberingParser.newSAXParser();
388         enableFileAccess(parserNumbering);
389         handlerNumbering = new NumberingSystemsParseHandler();
390         File fileNumbering = new File(NUMBERING_SOURCE_FILE);
391         parserNumbering.parse(fileNumbering, handlerNumbering);
392 
393         // Parse metaZones to create mappings between Olson tzids and CLDR meta zone names
394         info("..... Parsing metaZones.xml .....");
395         SAXParserFactory metazonesParser = SAXParserFactory.newInstance();
396         metazonesParser.setValidating(true);
397         SAXParser parserMetaZones = metazonesParser.newSAXParser();
398         enableFileAccess(parserMetaZones);
399         handlerMetaZones = new MetaZonesParseHandler();
400         File fileMetaZones = new File(METAZONES_SOURCE_FILE);
401         parserMetaZones.parse(fileMetaZones, handlerMetaZones);
402 
403         // Parse likelySubtags
404         info("..... Parsing likelySubtags.xml .....");
405         SAXParserFactory likelySubtagsParser = SAXParserFactory.newInstance();
406         likelySubtagsParser.setValidating(true);
407         SAXParser parserLikelySubtags = likelySubtagsParser.newSAXParser();
408         enableFileAccess(parserLikelySubtags);
409         handlerLikelySubtags = new LikelySubtagsParseHandler();
410         File fileLikelySubtags = new File(LIKELYSUBTAGS_SOURCE_FILE);
411         parserLikelySubtags.parse(fileLikelySubtags, handlerLikelySubtags);
412     }

414     private static void convertBundles(List<Bundle> bundles) throws Exception {
415         // For generating information on supported locales.
416         Map<String, SortedSet<String>> metaInfo = new HashMap<>();
417         metaInfo.put("LocaleNames", new TreeSet<>());
418         metaInfo.put("CurrencyNames", new TreeSet<>());
419         metaInfo.put("TimeZoneNames", new TreeSet<>());
420         metaInfo.put("CalendarData", new TreeSet<>());
421         metaInfo.put("FormatData", new TreeSet<>());
422         metaInfo.put("AvailableLocales", new TreeSet<>());
423 
424         // parent locales map. The mappings are put in base metaInfo file
425         // for now.
426         if (isBaseModule) {
427             metaInfo.putAll(parentLocalesMap);
428         }
429 
430         for (Bundle bundle : bundles) {
431             // Get the target map, which contains all the data that should be
432             // visible for the bundle's locale
433 
434             Map<String, Object> targetMap = bundle.getTargetMap();
435 
436             EnumSet<Bundle.Type> bundleTypes = bundle.getBundleTypes();
437 
438             if (bundle.isRoot()) {
439                 // Add DateTimePatternChars because CLDR no longer supports localized patterns.
440                 targetMap.put("DateTimePatternChars", "GyMdkHmsSEDFwWahKzZ");
441             }
442 
443             // Now the map contains just the entries that need to be in the resources bundles.
444             // Go ahead and generate them.
445             if (bundleTypes.contains(Bundle.Type.LOCALENAMES)) {
446                 Map<String, Object> localeNamesMap = extractLocaleNames(targetMap, bundle.getID());
447                 if (!localeNamesMap.isEmpty() || bundle.isRoot()) {
448                     metaInfo.get("LocaleNames").add(toLanguageTag(bundle.getID()));
449                     addLikelySubtags(metaInfo, "LocaleNames", bundle.getID());
450                     bundleGenerator.generateBundle("util", "LocaleNames", bundle.getJavaID(), true, localeNamesMap, BundleType.OPEN);
451                 }
452             }
453             if (bundleTypes.contains(Bundle.Type.CURRENCYNAMES)) {
454                 Map<String, Object> currencyNamesMap = extractCurrencyNames(targetMap, bundle.getID(), bundle.getCurrencies());
455                 if (!currencyNamesMap.isEmpty() || bundle.isRoot()) {
456                     metaInfo.get("CurrencyNames").add(toLanguageTag(bundle.getID()));
457                     addLikelySubtags(metaInfo, "CurrencyNames", bundle.getID());
458                     bundleGenerator.generateBundle("util", "CurrencyNames", bundle.getJavaID(), true, currencyNamesMap, BundleType.OPEN);
459                 }
460             }
461             if (bundleTypes.contains(Bundle.Type.TIMEZONENAMES)) {
462                 Map<String, Object> zoneNamesMap = extractZoneNames(targetMap, bundle.getID());
463                 if (!zoneNamesMap.isEmpty() || bundle.isRoot()) {
464                     metaInfo.get("TimeZoneNames").add(toLanguageTag(bundle.getID()));
465                     addLikelySubtags(metaInfo, "TimeZoneNames", bundle.getID());
466                     bundleGenerator.generateBundle("util", "TimeZoneNames", bundle.getJavaID(), true, zoneNamesMap, BundleType.TIMEZONE);
467                 }
468             }
469             if (bundleTypes.contains(Bundle.Type.CALENDARDATA)) {
470                 Map<String, Object> calendarDataMap = extractCalendarData(targetMap, bundle.getID());
471                 if (!calendarDataMap.isEmpty() || bundle.isRoot()) {
472                     metaInfo.get("CalendarData").add(toLanguageTag(bundle.getID()));
473                     addLikelySubtags(metaInfo, "CalendarData", bundle.getID());
474                     bundleGenerator.generateBundle("util", "CalendarData", bundle.getJavaID(), true, calendarDataMap, BundleType.PLAIN);
475                 }
476             }
477             if (bundleTypes.contains(Bundle.Type.FORMATDATA)) {
478                 Map<String, Object> formatDataMap = extractFormatData(targetMap, bundle.getID());
479                 if (!formatDataMap.isEmpty() || bundle.isRoot()) {
480                     metaInfo.get("FormatData").add(toLanguageTag(bundle.getID()));
481                     addLikelySubtags(metaInfo, "FormatData", bundle.getID());
482                     bundleGenerator.generateBundle("text", "FormatData", bundle.getJavaID(), true, formatDataMap, BundleType.PLAIN);
483                 }
484             }
485 
486             // For AvailableLocales
487             metaInfo.get("AvailableLocales").add(toLanguageTag(bundle.getID()));
488             addLikelySubtags(metaInfo, "AvailableLocales", bundle.getID());
489         }
490         addCldrImplicitLocales(metaInfo);
491         bundleGenerator.generateMetaInfo(metaInfo);
492     }

529     /**
530      * Translate the aliases into the real entries in the bundle map.
531      */
532     static void handleAliases(Map<String, Object> bundleMap) {
533         Set bundleKeys = bundleMap.keySet();
534         try {
535             for (String key : aliases.keySet()) {
536                 String targetKey = aliases.get(key);
537                 if (bundleKeys.contains(targetKey)) {
538                     bundleMap.putIfAbsent(key, bundleMap.get(targetKey));
539                 }
540             }
541         } catch (Exception ex) {
542             Logger.getLogger(CLDRConverter.class.getName()).log(Level.SEVERE, null, ex);
543         }
544     }

547      * Returns the language portion of the given id.
548      * If id is "root", "" is returned.

550     static String getLanguageCode(String id) {
551         return "root".equals(id) ? "" : Locale.forLanguageTag(id.replaceAll("_", "-")).getLanguage();
552     }

554     /**
555      * Examine if the id includes the country (territory) code. If it does, it returns
556      * the country code.
557      * Otherwise, it returns null. eg. when the id is "zh_Hans_SG", it return "SG".
558      * It does NOT return UN M.49 code, e.g., '001', as those three digit numbers cannot
559      * be translated into package names.
560      */
561     static String getCountryCode(String id) {
562         String rgn = getRegionCode(id);
563         return rgn.length() == 2 ? rgn: null;
564     }
565 
566     /**
567      * Examine if the id includes the region code. If it does, it returns
568      * the region code.
569      * Otherwise, it returns null. eg. when the id is "zh_Hans_SG", it return "SG".
570      * It DOES return UN M.49 code, e.g., '001', as well as ISO 3166 two letter country codes.
571      */
572     static String getRegionCode(String id) {
573         return Locale.forLanguageTag(id.replaceAll("_", "-")).getCountry();
574     }

576     private static class KeyComparator implements Comparator<String> {
577         static KeyComparator INSTANCE = new KeyComparator();
578 
579         private KeyComparator() {
580         }
581 
582         @Override
583         public int compare(String o1, String o2) {
584             int len1 = o1.length();
585             int len2 = o2.length();
586             if (!isDigit(o1.charAt(0)) && !isDigit(o2.charAt(0))) {
587                 // Shorter string comes first unless either starts with a digit.
588                 if (len1 < len2) {
589                     return -1;
590                 }
591                 if (len1 > len2) {
592                     return 1;
593                 }
594             }
595             return o1.compareTo(o2);
596         }
597 
598         private boolean isDigit(char c) {
599             return c >= '0' && c <= '9';
600         }
601     }

603     private static Map<String, Object> extractLocaleNames(Map<String, Object> map, String id) {
604         Map<String, Object> localeNames = new TreeMap<>(KeyComparator.INSTANCE);
605         for (String key : map.keySet()) {
606             if (key.startsWith(LOCALE_NAME_PREFIX)) {
607                 localeNames.put(key.substring(LOCALE_NAME_PREFIX.length()), map.get(key));
608             }
609         }
610         return localeNames;
611     }

627     private static Map<String, Object> extractZoneNames(Map<String, Object> map, String id) {
628         Map<String, Object> names = new HashMap<>();
629 
630         // Copy over missing time zone ids from JRE for English locale
631         if (id.equals("en")) {
632             Map<String[], String> jreMetaMap = new HashMap<>();
633             jreTimeZoneNames.stream().forEach(e -> {
634                 String tzid = (String)e[0];
635                 String[] data = (String[])e[1];
636 
637                 if (map.get(TIMEZONE_ID_PREFIX + tzid) == null &&
638                     handlerMetaZones.get(tzid) == null ||
639                     handlerMetaZones.get(tzid) != null &&
640                     map.get(METAZONE_ID_PREFIX + handlerMetaZones.get(tzid)) == null) {
641                     // First, check the CLDR meta key
642                     Optional<Map.Entry<String, String>> cldrMeta =
643                         handlerMetaZones.getData().entrySet().stream()
644                             .filter(me ->
645                                 Arrays.deepEquals(data,
646                                     (String[])map.get(METAZONE_ID_PREFIX + me.getValue())))
647                             .findAny();
648                     if (cldrMeta.isPresent()) {
649                         names.put(tzid, cldrMeta.get().getValue());
650                     } else {
651                         // check the JRE meta key, add if there is not.
652                         Optional<Map.Entry<String[], String>> jreMeta =
653                             jreMetaMap.entrySet().stream()
654                                 .filter(jm -> Arrays.deepEquals(data, jm.getKey()))
655                                 .findAny();
656                         if (jreMeta.isPresent()) {
657                             names.put(tzid, jreMeta.get().getValue());
658                         } else {
659                             String metaName = "JRE_" + tzid.replaceAll("[/-]", "_");
660                             names.put(METAZONE_ID_PREFIX + metaName, data);
661                             names.put(tzid, metaName);
662                             jreMetaMap.put(data, metaName);
663                         }
664                     }
665                 }
666             });
667         }
668 
669         for (String tzid : handlerMetaZones.keySet()) {
670             String tzKey = TIMEZONE_ID_PREFIX + tzid;
671             Object data = map.get(tzKey);
672             if (data instanceof String[]) {
673                 names.put(tzid, data);
674             } else {
675                 String meta = handlerMetaZones.get(tzid);
676                 if (meta != null) {
677                     String metaKey = METAZONE_ID_PREFIX + meta;
678                     data = map.get(metaKey);
679                     if (data instanceof String[]) {
680                         // Keep the metazone prefix here.
681                         names.put(metaKey, data);
682                         names.put(tzid, meta);
683                     }
684                 }
685             }
686         }
687         return names;
688     }

740     private static Map<String, Object> extractFormatData(Map<String, Object> map, String id) {
741         Map<String, Object> formatData = new LinkedHashMap<>();
742         for (CalendarType calendarType : CalendarType.values()) {
743             if (calendarType == CalendarType.GENERIC) {
744                 continue;
745             }
746             String prefix = calendarType.keyElementName();
747             for (String element : FORMAT_DATA_ELEMENTS) {
748                 String key = prefix + element;
749                 copyIfPresent(map, "java.time." + key, formatData);
750                 copyIfPresent(map, key, formatData);
751             }
752         }
753 
754         for (String key : map.keySet()) {
755         // Copy available calendar names
756             if (key.startsWith(CLDRConverter.CALENDAR_NAME_PREFIX)) {
757                 String type = key.substring(CLDRConverter.CALENDAR_NAME_PREFIX.length());
758                 for (CalendarType calendarType : CalendarType.values()) {
759                     if (calendarType == CalendarType.GENERIC) {
760                         continue;
761                     }
762                     if (type.equals(calendarType.lname())) {
763                         Object value = map.get(key);
764                         formatData.put(key, value);
765                         String ukey = CLDRConverter.CALENDAR_NAME_PREFIX + calendarType.uname();
766                         if (!key.equals(ukey)) {
767                             formatData.put(ukey, value);
768                         }
769                     }
770                 }
771             }
772         }
773 
774         copyIfPresent(map, "DefaultNumberingSystem", formatData);
775 
776         @SuppressWarnings("unchecked")
777         List<String> numberingScripts = (List<String>) map.remove("numberingScripts");
778         if (numberingScripts != null) {
779             for (String script : numberingScripts) {
780                 copyIfPresent(map, script + "." + "NumberElements", formatData);
781             }
782         } else {
783             copyIfPresent(map, "NumberElements", formatData);
784         }
785         copyIfPresent(map, "NumberPatterns", formatData);
786         return formatData;
787     }

793         }

797     private static final String specialSaveCharsJava = "\"";
798     private static final String specialSaveCharsProperties = "=: \t\r\n\f#!";

800     /*
801      * Converts unicodes to encoded &#92;uxxxx
802      * and writes out any of the characters in specialSaveChars
803      * with a preceding slash
804      */
805     static String saveConvert(String theString, boolean useJava) {
806         if (theString == null) {
807             return "";
808         }
809 
810         String specialSaveChars;
811         if (useJava) {
812             specialSaveChars = specialSaveCharsJava;
813         } else {
814             specialSaveChars = specialSaveCharsProperties;
815         }
816         boolean escapeSpace = false;
817 
818         int len = theString.length();
819         StringBuilder outBuffer = new StringBuilder(len * 2);
820         Formatter formatter = new Formatter(outBuffer, Locale.ROOT);
821 
822         for (int x = 0; x < len; x++) {
823             char aChar = theString.charAt(x);
824             switch (aChar) {
825             case ' ':
826                 if (x == 0 || escapeSpace) {
827                     outBuffer.append('\\');
828                 }
829                 outBuffer.append(' ');
830                 break;
831             case '\\':
832                 outBuffer.append('\\');
833                 outBuffer.append('\\');
834                 break;
835             case '\t':
836                 outBuffer.append('\\');
837                 outBuffer.append('t');
838                 break;
839             case '\n':
840                 outBuffer.append('\\');
841                 outBuffer.append('n');
842                 break;
843             case '\r':
844                 outBuffer.append('\\');
845                 outBuffer.append('r');
846                 break;
847             case '\f':
848                 outBuffer.append('\\');
849                 outBuffer.append('f');
850                 break;
851             default:
852                 if (aChar < 0x0020 || (!USE_UTF8 && aChar > 0x007e)) {
853                     formatter.format("\\u%04x", (int)aChar);
854                 } else {
855                     if (specialSaveChars.indexOf(aChar) != -1) {
856                         outBuffer.append('\\');
857                     }
858                     outBuffer.append(aChar);
859                 }
860             }
861         }
862         return outBuffer.toString();
863     }

897     // applying parent locale rules to the passed candidates list
898     // This has to match with the one in sun.util.cldr.CLDRLocaleProviderAdapter
899     private static Map<Locale, Locale> childToParentLocaleMap = null;
900     private static List<Locale> applyParentLocales(String baseName, List<Locale> candidates) {
901         if (Objects.isNull(childToParentLocaleMap)) {
902             childToParentLocaleMap = new HashMap<>();
903             parentLocalesMap.keySet().forEach(key -> {
904                 String parent = key.substring(PARENT_LOCALE_PREFIX.length()).replaceAll("_", "-");
905                 parentLocalesMap.get(key).stream().forEach(child -> {
906                     childToParentLocaleMap.put(Locale.forLanguageTag(child),
907                         "root".equals(parent) ? Locale.ROOT : Locale.forLanguageTag(parent));
908                 });
909             });
910         }
911 
912         // check irregular parents
913         for (int i = 0; i < candidates.size(); i++) {
914             Locale l = candidates.get(i);
915             Locale p = childToParentLocaleMap.get(l);
916             if (!l.equals(Locale.ROOT) &&
917                 Objects.nonNull(p) &&
918                 !candidates.get(i+1).equals(p)) {
919                 List<Locale> applied = candidates.subList(0, i+1);
920                 applied.addAll(applyParentLocales(baseName, defCon.getCandidateLocales(baseName, p)));
921                 return applied;
922             }
923         }
924 
925         return candidates;
926     }
