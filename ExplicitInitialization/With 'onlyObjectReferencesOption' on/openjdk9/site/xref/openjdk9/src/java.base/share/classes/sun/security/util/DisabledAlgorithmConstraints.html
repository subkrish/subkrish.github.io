161     // Check if a string is contained inside the property
162     public boolean checkProperty(String param) {
163         param = param.toLowerCase(Locale.ENGLISH);
164         for (String block : disabledAlgorithms) {
165             if (block.toLowerCase(Locale.ENGLISH).indexOf(param) >= 0) {
166                 return true;
167             }
168         }
169         return false;
170     }

172     // Check algorithm constraints with key and algorithm
173     private boolean checkConstraints(Set<CryptoPrimitive> primitives,
174             String algorithm, Key key, AlgorithmParameters parameters) {
175 
176         // check the key parameter, it cannot be null.
177         if (key == null) {
178             throw new IllegalArgumentException("The key cannot be null");
179         }
180 
181         // check the signature algorithm with parameters
182         if (algorithm != null && algorithm.length() != 0) {
183             if (!permits(primitives, algorithm, parameters)) {
184                 return false;
185             }
186         }
187 
188         // check the key algorithm
189         if (!permits(primitives, key.getAlgorithm(), null)) {
190             return false;
191         }
192 
193         // check the key constraints
194         return algorithmConstraints.permits(key);
195     }

198     /**
199      * Key and Certificate Constraints
200      *
201      * The complete disabling of an algorithm is not handled by Constraints or
202      * Constraint classes.  That is addressed with
203      *   permit(Set<CryptoPrimitive>, String, AlgorithmParameters)
204      *
205      * When passing a Key to permit(), the boolean return values follow the
206      * same as the interface class AlgorithmConstraints.permit().  This is to
207      * maintain compatibility:
208      * 'true' means the operation is allowed.
209      * 'false' means it failed the constraints and is disallowed.
210      *
211      * When passing ConstraintsParameters through permit(), an exception
212      * will be thrown on a failure to better identify why the operation was
213      * disallowed.
214      */
215 
216     private static class Constraints {
217         private Map<String, List<Constraint>> constraintsMap = new HashMap<>();
218 
219         private static class Holder {
220             private static final Pattern DENY_AFTER_PATTERN = Pattern.compile(
221                     "denyAfter\\s+(\\d{4})-(\\d{2})-(\\d{2})");
222         }
223 
224         public Constraints(String[] constraintArray) {
225             for (String constraintEntry : constraintArray) {
226                 if (constraintEntry == null || constraintEntry.isEmpty()) {
227                     continue;
228                 }
229 
230                 constraintEntry = constraintEntry.trim();
231                 if (debug != null) {
232                     debug.println("Constraints: " + constraintEntry);
233                 }
234 
235                 // Check if constraint is a complete disabling of an
236                 // algorithm or has conditions.
237                 int space = constraintEntry.indexOf(' ');
238                 String algorithm = AlgorithmDecomposer.hashName(
239                         ((space > 0 ? constraintEntry.substring(0, space) :
240                                 constraintEntry).
241                                 toUpperCase(Locale.ENGLISH)));
242                 List<Constraint> constraintList =
243                         constraintsMap.getOrDefault(algorithm,
244                                 new ArrayList<>(1));
245                 constraintsMap.putIfAbsent(algorithm, constraintList);
246                 if (space <= 0) {
247                     constraintList.add(new DisabledConstraint(algorithm));
248                     continue;
249                 }
250 
251                 String policy = constraintEntry.substring(space + 1);
252 
253                 // Convert constraint conditions into Constraint classes
254                 Constraint c, lastConstraint = null;
255                 // Allow only one jdkCA entry per constraint entry
256                 boolean jdkCALimit = false;
257                 // Allow only one denyAfter entry per constraint entry
258                 boolean denyAfterLimit = false;
259 
260                 for (String entry : policy.split("&")) {
261                     entry = entry.trim();
262 
263                     Matcher matcher;
264                     if (entry.startsWith("keySize")) {
265                         if (debug != null) {
266                             debug.println("Constraints set to keySize: " +
267                                     entry);
268                         }
269                         StringTokenizer tokens = new StringTokenizer(entry);
270                         if (!"keySize".equals(tokens.nextToken())) {
271                             throw new IllegalArgumentException("Error in " +
272                                     "security property. Constraint unknown: " +
273                                     entry);
274                         }
275                         c = new KeySizeConstraint(algorithm,
276                                 KeySizeConstraint.Operator.of(tokens.nextToken()),
277                                 Integer.parseInt(tokens.nextToken()));
278 
279                     } else if (entry.equalsIgnoreCase("jdkCA")) {
280                         if (debug != null) {
281                             debug.println("Constraints set to jdkCA.");
282                         }
283                         if (jdkCALimit) {
284                             throw new IllegalArgumentException("Only one " +
285                                     "jdkCA entry allowed in property. " +
286                                     "Constraint: " + constraintEntry);
287                         }
288                         c = new jdkCAConstraint(algorithm);
289                         jdkCALimit = true;
290 
291                     } else if (entry.startsWith("denyAfter") &&
292                             (matcher = Holder.DENY_AFTER_PATTERN.matcher(entry))
293                                     .matches()) {
294                         if (debug != null) {
295                             debug.println("Constraints set to denyAfter");
296                         }
297                         if (denyAfterLimit) {
298                             throw new IllegalArgumentException("Only one " +
299                                     "denyAfter entry allowed in property. " +
300                                     "Constraint: " + constraintEntry);
301                         }
302                         int year = Integer.parseInt(matcher.group(1));
303                         int month = Integer.parseInt(matcher.group(2));
304                         int day = Integer.parseInt(matcher.group(3));
305                         c = new DenyAfterConstraint(algorithm, year, month,
306                                 day);
307                         denyAfterLimit = true;
308                     } else if (entry.startsWith("usage")) {
309                         String s[] = (entry.substring(5)).trim().split(" ");
310                         c = new UsageConstraint(algorithm, s);
311                         if (debug != null) {
312                             debug.println("Constraints usage length is " + s.length);
313                         }
314                     } else {
315                         throw new IllegalArgumentException("Error in security" +
316                                 " property. Constraint unknown: " + entry);
317                     }
318 
319                     // Link multiple conditions for a single constraint
320                     // into a linked list.
321                     if (lastConstraint == null) {
322                         constraintList.add(c);
323                     } else {
324                         lastConstraint.nextConstraint = c;
325                     }
326                     lastConstraint = c;
327                 }
328             }
329         }
330 
331         // Get applicable constraints based off the signature algorithm
332         private List<Constraint> getConstraints(String algorithm) {
333             return constraintsMap.get(algorithm);
334         }
335 
336         // Check if KeySizeConstraints permit the specified key
337         public boolean permits(Key key) {
338             List<Constraint> list = getConstraints(key.getAlgorithm());
339             if (list == null) {
340                 return true;
341             }
342             for (Constraint constraint : list) {
343                 if (!constraint.permits(key)) {
344                     if (debug != null) {
345                         debug.println("keySizeConstraint: failed key " +
346                                 "constraint check " + KeyUtil.getKeySize(key));
347                     }
348                     return false;
349                 }
350             }
351             return true;
352         }
353 
354         // Check if constraints permit this cert.
355         public void permits(String algorithm, ConstraintsParameters cp)
356                 throws CertPathValidatorException {
357             X509Certificate cert = cp.getCertificate();
358 
359             if (debug != null) {
360                 debug.println("Constraints.permits(): " + algorithm +
361                         " Variant: " + cp.getVariant());
362             }
363 
364             // Get all signature algorithms to check for constraints
365             Set<String> algorithms = new HashSet<>();
366             if (algorithm != null) {
367                 algorithms.addAll(AlgorithmDecomposer.decomposeOneHash(algorithm));
368             }
369 
370             // Attempt to add the public key algorithm if cert provided
371             if (cert != null) {
372                 algorithms.add(cert.getPublicKey().getAlgorithm());
373             }
374             if (cp.getPublicKey() != null) {
375                 algorithms.add(cp.getPublicKey().getAlgorithm());
376             }
377             // Check all applicable constraints
378             for (String alg : algorithms) {
379                 List<Constraint> list = getConstraints(alg);
380                 if (list == null) {
381                     continue;
382                 }
383                 for (Constraint constraint : list) {
384                     constraint.permits(cp);
385                 }
386             }
387         }

390     /**
391      * This abstract Constraint class for algorithm-based checking
392      * may contain one or more constraints.  If the '&' on the {@Security}
393      * property is used, multiple constraints have been grouped together
394      * requiring all the constraints to fail for the check to be disallowed.
395      *
396      * If the class contains multiple constraints, the next constraint
397      * is stored in {@code nextConstraint} in linked-list fashion.
398      */
399     private abstract static class Constraint {
400         String algorithm;
401         Constraint nextConstraint = null;
402 
403         // operator
404         enum Operator {
405             EQ,         // "=="
406             NE,         // "!="
407             LT,         // "<"
408             LE,         // "<="
409             GT,         // ">"
410             GE;         // ">="
411 
412             static Operator of(String s) {
413                 switch (s) {
414                     case "==":
415                         return EQ;
416                     case "!=":
417                         return NE;
418                     case "<":
419                         return LT;
420                     case "<=":
421                         return LE;
422                     case ">":
423                         return GT;
424                     case ">=":
425                         return GE;
426                 }
427 
428                 throw new IllegalArgumentException("Error in security " +
429                         "property. " + s + " is not a legal Operator");
430             }
431         }
432 
433         /**
434          * Check if an algorithm constraint is permitted with a given key.
435          *
436          * If the check inside of {@code permit()} fails, it must call
437          * {@code next()} with the same {@code Key} parameter passed if
438          * multiple constraints need to be checked.
439          *
440          * @param key Public key
441          * @return 'true' if constraint is allowed, 'false' if disallowed.
442          */
443         public boolean permits(Key key) {
444             return true;
445         }
446 
447         /**
448          * Check if an algorithm constraint is permitted with a given
449          * ConstraintsParameters.
450          *
451          * If the check inside of {@code permits()} fails, it must call
452          * {@code next()} with the same {@code ConstraintsParameters}
453          * parameter passed if multiple constraints need to be checked.
454          *
455          * @param cp CertConstraintParameter containing certificate info
456          * @throws CertPathValidatorException if constraint disallows.
457          *
458          */
459         public abstract void permits(ConstraintsParameters cp)
460                 throws CertPathValidatorException;
461 
462         /**
463          * Recursively check if the constraints are allowed.
464          *
465          * If {@code nextConstraint} is non-null, this method will
466          * call {@code nextConstraint}'s {@code permits()} to check if the
467          * constraint is allowed or denied.  If the constraint's
468          * {@code permits()} is allowed, this method will exit this and any
469          * recursive next() calls, returning 'true'.  If the constraints called
470          * were disallowed, the last constraint will throw
471          * {@code CertPathValidatorException}.
472          *
473          * @param cp ConstraintsParameters
474          * @return 'true' if constraint allows the operation, 'false' if
475          * we are at the end of the constraint list or,
476          * {@code nextConstraint} is null.
477          */
478         boolean next(ConstraintsParameters cp)
479                 throws CertPathValidatorException {
480             if (nextConstraint != null) {
481                 nextConstraint.permits(cp);
482                 return true;
483             }
484             return false;
485         }
486 
487         /**
488          * Recursively check if this constraint is allowed,
489          *
490          * If {@code nextConstraint} is non-null, this method will
491          * call {@code nextConstraint}'s {@code permit()} to check if the
492          * constraint is allowed or denied.  If the constraint's
493          * {@code permit()} is allowed, this method will exit this and any
494          * recursive next() calls, returning 'true'.  If the constraints
495          * called were disallowed the check will exit with 'false'.
496          *
497          * @param key Public key
498          * @return 'true' if constraint allows the operation, 'false' if
499          * the constraint denies the operation.
500          */
501         boolean next(Key key) {
502             if (nextConstraint != null && nextConstraint.permits(key)) {
503                 return true;
504             }
505             return false;
506         }
507 
508         String extendedMsg(ConstraintsParameters cp) {
509             return (cp.getCertificate() == null ? "." :
510                     " used with certificate: " +
511                             cp.getCertificate().getSubjectX500Principal() +
512                     (cp.getVariant() != Validator.VAR_GENERIC ?
513                             ".  Usage was " + cp.getVariant() : "."));
514         }
515     }

518      * This class contains constraints dealing with the certificate chain

526         /*
527          * Check if ConstraintsParameters has a trusted match, if it does
528          * call next() for any following constraints. If it does not, exit
529          * as this constraint(s) does not restrict the operation.
530          */
531         public void permits(ConstraintsParameters cp)
532                 throws CertPathValidatorException {
533             if (debug != null) {
534                 debug.println("jdkCAConstraints.permits(): " + algorithm);
535             }
536 
537             // Check chain has a trust anchor in cacerts
538             if (cp.isTrustedMatch()) {
539                 if (next(cp)) {
540                     return;
541                 }
542                 throw new CertPathValidatorException(
543                         "Algorithm constraints check failed on certificate " +
544                         "anchor limits. " + algorithm + extendedMsg(cp),
545                         null, null, -1, BasicReason.ALGORITHM_CONSTRAINED);
546             }
547         }

550     /*
551      * This class handles the denyAfter constraint.  The date is in the UTC/GMT
552      * timezone.
553      */
554      private static class DenyAfterConstraint extends Constraint {
555          private Date denyAfterDate;
556          private static final SimpleDateFormat dateFormat =
557                  new SimpleDateFormat("EEE, MMM d HH:mm:ss z yyyy");
558 
559          DenyAfterConstraint(String algo, int year, int month, int day) {
560              Calendar c;
561 
562              algorithm = algo;
563 
564              if (debug != null) {
565                  debug.println("DenyAfterConstraint read in as:  year " +
566                          year + ", month = " + month + ", day = " + day);
567              }
568 
569              c = new Calendar.Builder().setTimeZone(TimeZone.getTimeZone("GMT"))
570                      .setDate(year, month - 1, day).build();
571 
572              if (year > c.getActualMaximum(Calendar.YEAR) ||
573                      year < c.getActualMinimum(Calendar.YEAR)) {
574                  throw new IllegalArgumentException(
575                          "Invalid year given in constraint: " + year);
576              }
577              if ((month - 1) > c.getActualMaximum(Calendar.MONTH) ||
578                      (month - 1) < c.getActualMinimum(Calendar.MONTH)) {
579                  throw new IllegalArgumentException(
580                          "Invalid month given in constraint: " + month);
581              }
582              if (day > c.getActualMaximum(Calendar.DAY_OF_MONTH) ||
583                      day < c.getActualMinimum(Calendar.DAY_OF_MONTH)) {
584                  throw new IllegalArgumentException(
585                          "Invalid Day of Month given in constraint: " + day);
586              }
587 
588              denyAfterDate = c.getTime();
589              if (debug != null) {
590                  debug.println("DenyAfterConstraint date set to: " +
591                          dateFormat.format(denyAfterDate));
592              }
593          }
594 
595          /*
596           * Checking that the provided date is not beyond the constraint date.
597           * The provided date can be the PKIXParameter date if given,
598           * otherwise it is the current date.
599           *
600           * If the constraint disallows, call next() for any following
601           * constraints. Throw an exception if this is the last constraint.
602           */
603          @Override
604          public void permits(ConstraintsParameters cp)
605                  throws CertPathValidatorException {
606              Date currentDate;
607              String errmsg;
608 
609              if (cp.getJARTimestamp() != null) {
610                  currentDate = cp.getJARTimestamp().getTimestamp();
611                  errmsg = "JAR Timestamp date: ";
612              } else if (cp.getPKIXParamDate() != null) {
613                  currentDate = cp.getPKIXParamDate();
614                  errmsg = "PKIXParameter date: ";
615              } else {
616                  currentDate = new Date();
617                  errmsg = "Current date: ";
618              }
619 
620              if (!denyAfterDate.after(currentDate)) {
621                  if (next(cp)) {
622                      return;
623                  }
624                  throw new CertPathValidatorException(
625                          "denyAfter constraint check failed: " + algorithm +
626                          " used with Constraint date: " +
627                          dateFormat.format(denyAfterDate) + "; " + errmsg +
628                          dateFormat.format(currentDate) + extendedMsg(cp),
629                          null, null, -1, BasicReason.ALGORITHM_CONSTRAINED);
630              }
631          }
632 
633          /*
634           * Return result if the constraint's date is beyond the current date
635           * in UTC timezone.
636           */
637          public boolean permits(Key key) {
638              if (next(key)) {
639                  return true;
640              }
641              if (debug != null) {
642                  debug.println("DenyAfterConstraints.permits(): " + algorithm);
643              }
644 
645              return denyAfterDate.after(new Date());
646          }
647      }

649     /*
650      * The usage constraint is for the "usage" keyword.  It checks against the
651      * variant value in ConstraintsParameters.
652      */
653     private static class UsageConstraint extends Constraint {
654         String[] usages;
655 
656         UsageConstraint(String algorithm, String[] usages) {
657             this.algorithm = algorithm;
658             this.usages = usages;
659         }
660 
661         public void permits(ConstraintsParameters cp)
662                 throws CertPathValidatorException {
663             for (String usage : usages) {
664 
665                 String v = null;
666                 if (usage.compareToIgnoreCase("TLSServer") == 0) {
667                     v = Validator.VAR_TLS_SERVER;
668                 } else if (usage.compareToIgnoreCase("TLSClient") == 0) {
669                     v = Validator.VAR_TLS_CLIENT;
670                 } else if (usage.compareToIgnoreCase("SignedJAR") == 0) {
671                     v = Validator.VAR_PLUGIN_CODE_SIGNING;
672                 }
673 
674                 if (debug != null) {
675                     debug.println("Checking if usage constraint \"" + v +
676                             "\" matches \"" + cp.getVariant() + "\"");
677                     // Because usage checking can come from many places
678                     // a stack trace is very helpful.
679                     ByteArrayOutputStream ba = new ByteArrayOutputStream();
680                     PrintStream ps = new PrintStream(ba);
681                     (new Exception()).printStackTrace(ps);
682                     debug.println(ba.toString());
683                 }
684                 if (cp.getVariant().compareTo(v) == 0) {
685                     if (next(cp)) {
686                         return;
687                     }
688                     throw new CertPathValidatorException("Usage constraint " +
689                             usage + " check failed: " + algorithm +
690                             extendedMsg(cp),
691                             null, null, -1, BasicReason.ALGORITHM_CONSTRAINED);
692                 }
693             }
694         }
695     }

697     /*
698      * This class contains constraints dealing with the key size
699      * support limits per algorithm.   e.g.  "keySize <= 1024"
700      */
701     private static class KeySizeConstraint extends Constraint {
702 
703         private int minSize;            // the minimal available key size
704         private int maxSize;            // the maximal available key size
705         private int prohibitedSize = -1;    // unavailable key sizes
706         private int size;
707 
708         public KeySizeConstraint(String algo, Operator operator, int length) {
709             algorithm = algo;
710             switch (operator) {
711                 case EQ:      // an unavailable key size
712                     this.minSize = 0;
713                     this.maxSize = Integer.MAX_VALUE;
714                     prohibitedSize = length;
715                     break;
716                 case NE:
717                     this.minSize = length;
718                     this.maxSize = length;
719                     break;
720                 case LT:
721                     this.minSize = length;
722                     this.maxSize = Integer.MAX_VALUE;
723                     break;
724                 case LE:
725                     this.minSize = length + 1;
726                     this.maxSize = Integer.MAX_VALUE;
727                     break;
728                 case GT:
729                     this.minSize = 0;
730                     this.maxSize = length;
731                     break;
732                 case GE:
733                     this.minSize = 0;
734                     this.maxSize = length > 1 ? (length - 1) : 0;
735                     break;
736                 default:
737                     // unlikely to happen
738                     this.minSize = Integer.MAX_VALUE;
739                     this.maxSize = -1;
740             }
741         }
742 
743         /*
744          * If we are passed a certificate, extract the public key and use it.
745          *
746          * Check if each constraint fails and check if there is a linked
747          * constraint  Any permitted constraint will exit the linked list
748          * to allow the operation.
749          */
750         public void permits(ConstraintsParameters cp)
751                 throws CertPathValidatorException {
752             Key key = null;
753             if (cp.getPublicKey() != null) {
754                 key = cp.getPublicKey();
755             } else if (cp.getCertificate() != null) {
756                 key = cp.getCertificate().getPublicKey();
757             }
758             if (key != null && !permitsImpl(key)) {
759                 if (nextConstraint != null) {
760                     nextConstraint.permits(cp);
761                     return;
762                 }
763                 throw new CertPathValidatorException(
764                         "Algorithm constraints check failed on keysize limits. "
765                         + algorithm + " " + size + "bit key" + extendedMsg(cp),
766                         null, null, -1, BasicReason.ALGORITHM_CONSTRAINED);
767             }
768         }
769 
770 
771         // Check if key constraint disable the specified key
772         // Uses old style permit()
773         public boolean permits(Key key) {
774             // If we recursively find a constraint that permits us to use
775             // this key, return true and skip any other constraint checks.
776             if (nextConstraint != null && nextConstraint.permits(key)) {
777                 return true;
778             }
779             if (debug != null) {
780                 debug.println("KeySizeConstraints.permits(): " + algorithm);
781             }
782 
783             return permitsImpl(key);
784         }
785 
786         private boolean permitsImpl(Key key) {
787             // Verify this constraint is for this public key algorithm
788             if (algorithm.compareToIgnoreCase(key.getAlgorithm()) != 0) {
789                 return true;
790             }
791 
792             size = KeyUtil.getKeySize(key);
793             if (size == 0) {
794                 return false;    // we don't allow any key of size 0.
795             } else if (size > 0) {
796                 return !((size < minSize) || (size > maxSize) ||
797                     (prohibitedSize == size));
798             }   // Otherwise, the key size is not accessible. Conservatively,
799                 // please don't disable such keys.
800 
801             return true;
802         }
803     }

805     /*
806      * This constraint is used for the complete disabling of the algorithm.
807      */
808     private static class DisabledConstraint extends Constraint {
809         DisabledConstraint(String algo) {
810             algorithm = algo;
811         }

813         public void permits(ConstraintsParameters cp)
814                 throws CertPathValidatorException {

816                     "Algorithm constraints check failed on disabled " +
817                             "algorithm: " + algorithm + extendedMsg(cp),
818                     null, null, -1, BasicReason.ALGORITHM_CONSTRAINED);
819         }

821         public boolean permits(Key key) {
822             return false;
823         }
