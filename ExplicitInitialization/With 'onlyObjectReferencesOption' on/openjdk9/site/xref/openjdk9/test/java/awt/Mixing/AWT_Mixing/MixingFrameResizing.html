26  import java.awt.Point;

28  import java.awt.event.InputEvent;
29  import javax.swing.JFrame;

31  import javax.swing.SwingUtilities;
32  import test.java.awt.regtesthelpers.Util;

34  /**
35   * AWT/Swing overlapping test.
36   * <p>This test puts heavyweight component into JFrame and verifies that it's being drawn correctly after resizing the frame.
37   * <p>See base class for test info.
38   */
39  /*
40   * @test
41   * @key headful
42   * @bug 6777370
43   * @summary Issues when resizing the JFrame with HW components
44   * @author sergey.grinev@oracle.com: area=awt.mixing
45   * @library /java/awt/patchlib  ../../regtesthelpers
46   * @modules java.desktop/sun.awt
47   *          java.desktop/java.awt.peer
48   * @build java.desktop/java.awt.Helper
49   * @build Util
50   * @run main MixingFrameResizing
51   */
52  public class MixingFrameResizing extends OverlappingTestBase {
53  
54      {testEmbeddedFrame = true;}
55  
56      private JFrame frame = null;
57      private Point lLoc;
58      private Point lLoc2;
59      private Dimension size;
60  
61      protected void prepareControls() {
62          if(frame != null) {
63              frame.setVisible(false);
64          }
65          frame = new JFrame("Mixing : Frame Resizing test");
66          frame.setLayout(new SpringLayout());
67          frame.setSize(50, 50);
68          frame.setVisible(true);
69          propagateAWTControls(frame);
70          Util.waitTillShown(frame);
71      }
72  
73      @Override
74      protected boolean performTest() {
75          int BORDER_SHIFT = frameBorderCounter();
76          BORDER_SHIFT = Math.abs(BORDER_SHIFT) == 1 ? BORDER_SHIFT : (BORDER_SHIFT / 2);
77          try {
78              SwingUtilities.invokeAndWait(new Runnable() {
79                  public void run() {
80                      lLoc = frame.getLocationOnScreen();
81                      size = frame.getSize();
82                      lLoc2 = frame.getContentPane().getLocationOnScreen();
83                  }
84              });
85          } catch (Exception e) {
86              e.printStackTrace();
87              throw new RuntimeException("Where is frame?");
88          }
89          Robot robot = Util.createRobot();
90          robot.setAutoDelay(ROBOT_DELAY/2);
91  
92          // resize window
93          robot.mouseMove(lLoc.x + size.width / 2 + BORDER_SHIFT, lLoc.y + size.height + BORDER_SHIFT);
94          Util.waitForIdle(robot);
95          robot.mousePress(InputEvent.BUTTON1_MASK);
96          for (int i = 0; i < 10; i++) {
97              robot.mouseMove(lLoc.x + size.width / 2 + BORDER_SHIFT, lLoc.y + size.height + BORDER_SHIFT + 20 * i);
98          }
99          robot.mouseRelease(InputEvent.BUTTON1_MASK);
100 
101         robot.mouseMove(lLoc.x + size.width + BORDER_SHIFT, lLoc.y + size.height + BORDER_SHIFT);
102         Util.waitForIdle(robot);
103         robot.mousePress(InputEvent.BUTTON1_MASK);
104         for (int i = 0; i < 10; i++) {
105             robot.mouseMove(lLoc.x + size.width + BORDER_SHIFT + 20 * i, lLoc.y + size.height + BORDER_SHIFT);
106         }
107         robot.mouseRelease(InputEvent.BUTTON1_MASK);
108 
109         Util.waitForIdle(robot);
110         // check if component is visible on the opened space
111         try {
112             Thread.sleep(300); //some more wait for Solaris (for some reason)
113         }catch(Exception ex) {}
114         lLoc2.translate(75, 75);
115         Color c = robot.getPixelColor(lLoc2.x, lLoc2.y);
116         System.out.println("Actual: "+c+", expected: "+AWT_VERIFY_COLOR);
117 
118         if (!c.equals(AWT_VERIFY_COLOR)) {
119             fail("HW component is not visible after resizing");
120         }
121 
122         return true;
123     }
124 
125     // this strange plumbing stuff is required due to "Standard Test Machinery" in base class
126     public static void main(String args[]) throws InterruptedException {
127         if (System.getProperty("os.name").toLowerCase().contains("os x")) {
128             System.out.println("Aqua L&F ignores setting color to component. Test passes on Mac OS X.");
129             return;
130         }
131         instance = new MixingFrameResizing();
132         OverlappingTestBase.doMain(args);
133     }
134 }
