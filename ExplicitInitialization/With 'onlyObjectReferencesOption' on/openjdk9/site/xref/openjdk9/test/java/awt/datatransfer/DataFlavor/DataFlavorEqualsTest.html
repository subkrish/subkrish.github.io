25    @test
26    @bug 4175341
27    @summary DataFlavor.equals throws NullPointerException
28    @author prs@sparc.spb.su: area=
29    @modules java.datatransfer
30    @run main DataFlavorEqualsTest
31  */

33  import java.awt.datatransfer.DataFlavor;
34  
35  public class DataFlavorEqualsTest {
36       public static boolean finished = false;
37       static boolean noexc = true;
38       static boolean eq = false;
39       static DataFlavor df = null;
40  
41       public static void main(String[] args) {
42           df = new DataFlavor();
43           try {
44               eq = df.equals((Object)new DataFlavor());
45               if (!eq) noexc = false;
46               eq = df.equals(new DataFlavor());
47               if (!eq) noexc = false;
48               eq = df.equals("application/postscript;class=java.awt.datatransfer.DataFlavor");
49               if (eq) noexc = false;
50           } catch (NullPointerException e1) {
51               noexc = false;
52           }
53           finished = true;
54           if (!noexc)
55               throw new RuntimeException("Test FAILED");
56  
57       }
58  
59  }
