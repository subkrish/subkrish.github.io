126 /**
127  * This is an integration test borrowed from goraci, written by Keith Turner,
128  * which is in turn inspired by the Accumulo test called continous ingest (ci).
129  * The original source code can be found here:
130  * https://github.com/keith-turner/goraci
131  * https://github.com/enis/goraci/
132  *
133  * Apache Accumulo [0] has a simple test suite that verifies that data is not
134  * lost at scale. This test suite is called continuous ingest. This test runs
135  * many ingest clients that continually create linked lists containing 25
136  * million nodes. At some point the clients are stopped and a map reduce job is
137  * run to ensure no linked list has a hole. A hole indicates data was lost.··
138  *
139  * The nodes in the linked list are random. This causes each linked list to
140  * spread across the table. Therefore if one part of a table loses data, then it
141  * will be detected by references in another part of the table.
142  *
143  * THE ANATOMY OF THE TEST
144  *
145  * Below is rough sketch of how data is written. For specific details look at
146  * the Generator code.
147  *
148  * 1 Write out 1 million nodes· 2 Flush the client· 3 Write out 1 million that
149  * reference previous million· 4 If this is the 25th set of 1 million nodes,
150  * then update 1st set of million to point to last· 5 goto 1
151  *
152  * The key is that nodes only reference flushed nodes. Therefore a node should
153  * never reference a missing node, even if the ingest client is killed at any
154  * point in time.
155  *
156  * When running this test suite w/ Accumulo there is a script running in
157  * parallel called the Aggitator that randomly and continuously kills server
158  * processes.·· The outcome was that many data loss bugs were found in Accumulo
159  * by doing this.· This test suite can also help find bugs that impact uptime
160  * and stability when· run for days or weeks.··
161  *
162  * This test suite consists the following· - a few Java programs· - a little
163  * helper script to run the java programs - a maven script to build it.··
164  *
165  * When generating data, its best to have each map task generate a multiple of
166  * 25 million. The reason for this is that circular linked list are generated
167  * every 25M. Not generating a multiple in 25M will result in some nodes in the
168  * linked list not having references. The loss of an unreferenced node can not
169  * be detected.
170  *
171  *
172  * Below is a description of the Java programs
173  *
174  * Generator - A map only job that generates data. As stated previously,·
175  * its best to generate data in multiples of 25M.
176  *
177  * Verify - A map reduce job that looks for holes. Look at the counts after running. REFERENCED and
178  * UNREFERENCED are· ok, any UNDEFINED counts are bad. Do not run at the· same
179  * time as the Generator.
180  *
181  * Walker - A standalone program that start following a linked list· and emits timing info.··
182  *
183  * Print - A standalone program that prints nodes in the linked list
184  *
185  * Delete - A standalone program that deletes a single node
186  *
187  * This class can be run as a unit test, as an integration test, or from the command line
188  */

375      * </pre>
376      */
377     static class GeneratorMapper
378       extends Mapper<BytesWritable, NullWritable, NullWritable, NullWritable> {
379 
380       byte[][] first = null;
381       byte[][] prev = null;
382       byte[][] current = null;
383       byte[] id;
384       long count = 0;
385       int i;
386       BufferedMutator mutator;
387       Connection connection;
388       long numNodes;
389       long wrap;
390       int width;
391       boolean multipleUnevenColumnFamilies;
392       byte[] tinyValue = new byte[] { 't' };
393       byte[] bigValue = null;
394 
395       @Override
396       protected void setup(Context context) throws IOException, InterruptedException {
397         id = Bytes.toBytes("Job: "+context.getJobID() + " Task: " + context.getTaskAttemptID());
398         Configuration conf = context.getConfiguration();
399         connection = ConnectionFactory.createConnection(conf);
400         instantiateHTable();
401         this.width = context.getConfiguration().getInt(GENERATOR_WIDTH_KEY, WIDTH_DEFAULT);
402         current = new byte[this.width][];
403         int wrapMultiplier = context.getConfiguration().getInt(GENERATOR_WRAP_KEY, WRAP_DEFAULT);
404         this.wrap = (long)wrapMultiplier * width;
405         this.numNodes = context.getConfiguration().getLong(
406             GENERATOR_NUM_ROWS_PER_MAP_KEY, (long)WIDTH_DEFAULT * WRAP_DEFAULT);
407         if (this.numNodes < this.wrap) {
408           this.wrap = this.numNodes;
409         }
410         this.multipleUnevenColumnFamilies =
411             context.getConfiguration().getBoolean(MULTIPLE_UNEVEN_COLUMNFAMILIES_KEY, false);
412       }
413 
414       protected void instantiateHTable() throws IOException {
415         mutator = connection.getBufferedMutator(
416             new BufferedMutatorParams(getTableName(connection.getConfiguration()))
417                 .writeBufferSize(4 * 1024 * 1024));
418       }
419 
420       @Override
421       protected void cleanup(Context context) throws IOException ,InterruptedException {
422         mutator.close();
423         connection.close();
424       }
425 
426       @Override
427       protected void map(BytesWritable key, NullWritable value, Context output) throws IOException {
428         current[i] = new byte[key.getLength()];
429         System.arraycopy(key.getBytes(), 0, current[i], 0, key.getLength());
430         if (++i == current.length) {
431           LOG.info("Persisting current.length=" + current.length + ", count=" + count + ", id=" +
432             Bytes.toStringBinary(id) + ", current=" + Bytes.toStringBinary(current[0]) +
433             ", i=" + i);
434           persist(output, count, prev, current, id);
435           i = 0;
436 
437           if (first == null) {
438             first = current;
439           }
440           prev = current;
441           current = new byte[this.width][];
442 
443           count += current.length;
444           output.setStatus("Count " + count);
445 
446           if (count % wrap == 0) {
447             // this block of code turns the 1 million linked list of length 25 into one giant
448             //circular linked list of 25 million
449             circularLeftShift(first);
450 
451             persist(output, -1, prev, first, null);
452 
453             first = null;
454             prev = null;
455           }
456         }
457       }
458 
459       private static <T> void circularLeftShift(T[] first) {
460         T ez = first[0];
461         System.arraycopy(first, 1, first, 0, first.length - 1);
462         first[first.length - 1] = ez;
463       }
464 
465       protected void persist(Context output, long count, byte[][] prev, byte[][] current, byte[] id)
466           throws IOException {
467         for (int i = 0; i < current.length; i++) {
468           Put put = new Put(current[i]);
469           put.addColumn(FAMILY_NAME, COLUMN_PREV, prev == null ? NO_KEY : prev[i]);
470 
471           if (count >= 0) {
472             put.addColumn(FAMILY_NAME, COLUMN_COUNT, Bytes.toBytes(count + i));
473           }
474           if (id != null) {
475             put.addColumn(FAMILY_NAME, COLUMN_CLIENT, id);
476           }
477           // See if we are to write multiple columns.
478           if (this.multipleUnevenColumnFamilies) {
479             // Use any column name.
480             put.addColumn(TINY_FAMILY_NAME, TINY_FAMILY_NAME, this.tinyValue);
481             // If we've not allocated bigValue, do it now. Reuse same value each time.
482             if (this.bigValue == null) {
483               this.bigValue = new byte[current[i].length * 10];
484               ThreadLocalRandom.current().nextBytes(this.bigValue);
485             }
486             // Use any column name.
487             put.addColumn(BIG_FAMILY_NAME, BIG_FAMILY_NAME, this.bigValue);
488           }
489           mutator.mutate(put);
490 
491           if (i % 1000 == 0) {
492             // Tickle progress every so often else maprunner will think us hung
493             output.progress();
494           }
495         }
496 
497         mutator.flush();
498       }

519     protected void createSchema() throws IOException {
520       Configuration conf = getConf();
521       TableName tableName = getTableName(conf);
522       try (Connection conn = ConnectionFactory.createConnection(conf);
523           Admin admin = conn.getAdmin()) {
524         if (!admin.tableExists(tableName)) {
525           HTableDescriptor htd = new HTableDescriptor(getTableName(getConf()));
526           htd.addFamily(new HColumnDescriptor(FAMILY_NAME));
527           // Always add these families. Just skip writing to them when we do not test per CF flush.
528           htd.addFamily(new HColumnDescriptor(BIG_FAMILY_NAME));
529           htd.addFamily(new HColumnDescriptor(TINY_FAMILY_NAME));
530           int numberOfServers = admin.getClusterStatus().getServers().size();
531           if (numberOfServers == 0) {
532             throw new IllegalStateException("No live regionservers");
533           }
534           int regionsPerServer = conf.getInt(HBaseTestingUtility.REGIONS_PER_SERVER_KEY,
535                                 HBaseTestingUtility.DEFAULT_REGIONS_PER_SERVER);
536           int totalNumberOfRegions = numberOfServers * regionsPerServer;
537           LOG.info("Number of live regionservers: " + numberOfServers + ", " +
538               "pre-splitting table into " + totalNumberOfRegions + " regions " +
539               "(default regions per server: " + regionsPerServer + ")");
540 
541           byte[][] splits = new RegionSplitter.UniformSplit().split(totalNumberOfRegions);
542 
543           admin.createTable(htd, splits);
544         }
545       } catch (MasterNotRunningException e) {
546         LOG.error("Master not running", e);
547         throw new IOException(e);
548       }
549     }

735     static SortedSet<byte []> readKeysToSearch(final Configuration conf)
736     throws IOException, InterruptedException {
737       Path keysInputDir = new Path(conf.get(SEARCHER_INPUTDIR_KEY));
738       FileSystem fs = FileSystem.get(conf);
739       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
740       if (!fs.exists(keysInputDir)) {
741         throw new FileNotFoundException(keysInputDir.toString());
742       }
743       if (!fs.isDirectory(keysInputDir)) {
744         throw new UnsupportedOperationException("TODO");

757     private static SortedSet<byte[]> readFileToSearch(final Configuration conf,
758         final FileSystem fs, final LocatedFileStatus keyFileStatus) throws IOException,
759         InterruptedException {
760       SortedSet<byte []> result = new TreeSet<byte []>(Bytes.BYTES_COMPARATOR);
761       // Return entries that are flagged Counts.UNDEFINED in the value. Return the row. This is
762       // what is missing.
763       TaskAttemptContext context = new TaskAttemptContextImpl(conf, new TaskAttemptID());
764       try (SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader rr =
765           new SequenceFileAsBinaryInputFormat.SequenceFileAsBinaryRecordReader()) {
766         InputSplit is =
767           new FileSplit(keyFileStatus.getPath(), 0, keyFileStatus.getLen(), new String [] {});
768         rr.initialize(is, context);
769         while (rr.nextKeyValue()) {
770           rr.getCurrentKey();
771           BytesWritable bw = rr.getCurrentValue();
772           if (Verify.VerifyReducer.whichType(bw.getBytes()) == Verify.Counts.UNDEFINED) {
773             byte[] key = new byte[rr.getCurrentKey().getLength()];
774             System.arraycopy(rr.getCurrentKey().getBytes(), 0, key, 0, rr.getCurrentKey()
775                 .getLength());
776             result.add(key);
777           }
778         }
779       }
780       return result;
781     }

877        * @param r
878        * @return Return new byte array that has <code>ordinal</code> as prefix on front taking up
879        * Bytes.SIZEOF_SHORT bytes followed by <code>r</code>

881       public static byte[] addPrefixFlag(final int ordinal, final byte [] r) {
882         byte[] prefix = Bytes.toBytes((short)ordinal);
883         if (prefix.length != Bytes.SIZEOF_SHORT) {
884           throw new RuntimeException("Unexpected size: " + prefix.length);
885         }
886         byte[] result = new byte[prefix.length + r.length];
887         System.arraycopy(prefix, 0, result, 0, prefix.length);
888         System.arraycopy(r, 0, result, prefix.length, r.length);
889         return result;

912       @Override
913       public void reduce(BytesWritable key, Iterable<BytesWritable> values, Context context)
914           throws IOException, InterruptedException {
915         int defCount = 0;
916         boolean lostFamilies = false;
917         refs.clear();
918         for (BytesWritable type : values) {
919           if (type.getLength() == DEF.getLength()) {
920             defCount++;
921             if (type.getBytes()[0] == 1) {
922               lostFamilies = true;
923             }
924           } else {
925             byte[] bytes = new byte[type.getLength()];
926             System.arraycopy(type.getBytes(), 0, bytes, 0, type.getLength());
927             refs.add(bytes);
928           }
929         }
930 
931         // TODO check for more than one def, should not happen
932         StringBuilder refsSb = null;
933         String keyString = Bytes.toStringBinary(key.getBytes(), 0, key.getLength());
934         if (defCount == 0 || refs.size() != 1) {
935           refsSb = dumpExtraInfoOnRefs(key, context, refs);
936           LOG.error("LinkedListError: key=" + keyString + ", reference(s)=" +
937             (refsSb != null? refsSb.toString(): ""));
938         }
939         if (lostFamilies) {
940           LOG.error("LinkedListError: key=" + keyString + ", lost big or tiny families");
941           context.getCounter(Counts.LOST_FAMILIES).increment(1);
942           context.write(key, LOSTFAM);
943         }
944 
945         if (defCount == 0 && refs.size() > 0) {
946           // This is bad, found a node that is referenced but not defined. It must have been
947           // lost, emit some info about this node for debugging purposes.
948           // Write out a line per reference. If more than one, flag it.;
949           for (int i = 0; i < refs.size(); i++) {
950             byte[] bs = refs.get(i);
951             int ordinal;
952             if (i <= 0) {
953               ordinal = Counts.UNDEFINED.ordinal();
954               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
955               context.getCounter(Counts.UNDEFINED).increment(1);
956             } else {
957               ordinal = Counts.EXTRA_UNDEF_REFERENCES.ordinal();
958               context.write(key, new BytesWritable(addPrefixFlag(ordinal, bs)));
959             }
960           }
961           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
962             // Print out missing row; doing get on reference gives info on when the referencer
963             // was added which can help a little debugging. This info is only available in mapper
964             // output -- the 'Linked List error Key...' log message above. What we emit here is
965             // useless for debugging.
966             context.getCounter("undef", keyString).increment(1);
967           }
968         } else if (defCount > 0 && refs.size() == 0) {
969           // node is defined but not referenced
970           context.write(key, UNREF);
971           context.getCounter(Counts.UNREFERENCED).increment(1);
972           if (rows.addAndGet(1) < MISSING_ROWS_TO_LOG) {
973             context.getCounter("unref", keyString).increment(1);
974           }
975         } else {
976           if (refs.size() > 1) {
977             // Skip first reference.
978             for (int i = 1; i < refs.size(); i++) {
979               context.write(key,
980                 new BytesWritable(addPrefixFlag(Counts.EXTRAREFERENCES.ordinal(), refs.get(i))));
981             }
982             context.getCounter(Counts.EXTRAREFERENCES).increment(refs.size() - 1);
983           }
984           // node is defined and referenced
985           context.getCounter(Counts.REFERENCED).increment(1);
986         }
987       }

989       /**
990        * Dump out extra info around references if there are any. Helps debugging.
991        * @return StringBuilder filled with references if any.
992        * @throws IOException
993        */
994       private StringBuilder dumpExtraInfoOnRefs(final BytesWritable key, final Context context,
995           final List<byte []> refs)
996       throws IOException {
997         StringBuilder refsSb = null;
998         if (refs.isEmpty()) return refsSb;
999         refsSb = new StringBuilder();
1000         String comma = "";
1001         // If a row is a reference but has no define, print the content of the row that has
1002         // this row as a 'prev'; it will help debug.  The missing row was written just before
1003         // the row we are dumping out here.
1004         TableName tn = getTableName(context.getConfiguration());
1005         try (Table t = this.connection.getTable(tn)) {
1006           for (byte [] ref : refs) {
1007             Result r = t.get(new Get(ref));
1008             List<Cell> cells = r.listCells();
1009             String ts = (cells != null && !cells.isEmpty())?
1010                 new java.util.Date(cells.get(0).getTimestamp()).toString(): "";
1011             byte [] b = r.getValue(FAMILY_NAME, COLUMN_CLIENT);
1012             String jobStr = (b != null && b.length > 0)? Bytes.toString(b): "";
1013             b = r.getValue(FAMILY_NAME, COLUMN_COUNT);
1014             long count = (b != null && b.length > 0)? Bytes.toLong(b): -1;
1015             b = r.getValue(FAMILY_NAME, COLUMN_PREV);
1016             String refRegionLocation = "";
1017             String keyRegionLocation = "";
1018             if (b != null && b.length > 0) {
1019               try (RegionLocator rl = this.connection.getRegionLocator(tn)) {
1020                 HRegionLocation hrl = rl.getRegionLocation(b);
1021                 if (hrl != null) refRegionLocation = hrl.toString();
1022                 // Key here probably has trailing zeros on it.
1023                 hrl = rl.getRegionLocation(key.getBytes());
1024                 if (hrl != null) keyRegionLocation = hrl.toString();
1025               }
1026             }
1027             LOG.error("Extras on ref without a def, ref=" + Bytes.toStringBinary(ref) +
1028               ", refPrevEqualsKey=" +
1029                 (Bytes.compareTo(key.getBytes(), 0, key.getLength(), b, 0, b.length) == 0) +
1030                 ", key=" + Bytes.toStringBinary(key.getBytes(), 0, key.getLength()) +
1031                 ", ref row date=" + ts + ", jobStr=" + jobStr +
1032                 ", ref row count=" + count +
1033                 ", ref row regionLocation=" + refRegionLocation +
1034                 ", key row regionLocation=" + keyRegionLocation);
1035             refsSb.append(comma);
1036             comma = ",";
1037             refsSb.append(Bytes.toStringBinary(ref));
1038           }
1039         }
1040         return refsSb;
1041       }

1337   private static class Walker extends Configured implements Tool {
1338     @Override
1339     public int run(String[] args) throws IOException {
1340       Options options = new Options();
1341       options.addOption("n", "num", true, "number of queries");
1342       options.addOption("s", "start", true, "key to start at, binary string");
1343       options.addOption("l", "logevery", true, "log every N queries");
1344 
1345       GnuParser parser = new GnuParser();
1346       CommandLine cmd = null;
1347       try {
1348         cmd = parser.parse(options, args);
1349         if (cmd.getArgs().length != 0) {
1350           throw new ParseException("Command takes no arguments");
1351         }
1352       } catch (ParseException e) {
1353         System.err.println("Failed to parse command line " + e.getMessage());
1354         System.err.println();
1355         HelpFormatter formatter = new HelpFormatter();
1356         formatter.printHelp(getClass().getSimpleName(), options);
1357         System.exit(-1);
1358       }
1359 
1360       long maxQueries = Long.MAX_VALUE;
1361       if (cmd.hasOption('n')) {
1362         maxQueries = Long.parseLong(cmd.getOptionValue("n"));
1363       }
1364       Random rand = new SecureRandom();
1365       boolean isSpecificStart = cmd.hasOption('s');
1366       byte[] startKey = isSpecificStart ? Bytes.toBytesBinary(cmd.getOptionValue('s')) : null;
1367       int logEvery = cmd.hasOption('l') ? Integer.parseInt(cmd.getOptionValue('l')) : 1;
1368 
1369       Table table = new HTable(getConf(), getTableName(getConf()));
1370       long numQueries = 0;
1371       // If isSpecificStart is set, only walk one list from that particular node.
1372       // Note that in case of circular (or P-shaped) list it will walk forever, as is
1373       // the case in normal run without startKey.
1374       while (numQueries < maxQueries && (numQueries == 0 || !isSpecificStart)) {
1375         if (!isSpecificStart) {
1376           startKey = new byte[ROWKEY_LENGTH];
1377           rand.nextBytes(startKey);
1378         }
1379         CINode node = findStartNode(table, startKey);
1380         if (node == null && isSpecificStart) {
1381           System.err.printf("Start node not found: %s \n", Bytes.toStringBinary(startKey));
1382         }
1383         numQueries++;
1384         while (node != null && node.prev.length != NO_KEY.length && numQueries < maxQueries) {
1385           byte[] prev = node.prev;
1386           long t1 = System.currentTimeMillis();
1387           node = getNode(prev, table, node);
1388           long t2 = System.currentTimeMillis();
1389           if (numQueries % logEvery == 0) {
1390             System.out.printf("CQ %d: %d %s \n", numQueries, t2 - t1, Bytes.toStringBinary(prev));
1391           }
1392           numQueries++;
1393           if (node == null) {
1394             System.err.printf("UNDEFINED NODE %s \n", Bytes.toStringBinary(prev));
1395           } else if (node.prev.length == NO_KEY.length) {
1396             System.err.printf("TERMINATING NODE %s \n", Bytes.toStringBinary(node.key));
1397           }
1398         }
1399       }
1400 
1401       table.close();
1402       return 0;
1403     }

1422

1476     } else {
1477       node.count = -1;

1479     if (result.containsColumn(FAMILY_NAME, COLUMN_CLIENT)) {
1480       node.client = Bytes.toString(result.getValue(FAMILY_NAME, COLUMN_CLIENT));
1481     } else {
