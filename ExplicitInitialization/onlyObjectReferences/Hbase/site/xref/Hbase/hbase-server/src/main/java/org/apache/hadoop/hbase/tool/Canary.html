62  /**
63   * HBase Canary Tool, that that can be used to do
64   * "canary monitoring" of a running HBase cluster.
65   *
66   * Here are two modes
67   * 1. region mode - Foreach region tries to get one row per column family
68   * and outputs some information about failure or latency.
69   *
70   * 2. regionserver mode - Foreach regionserver tries to get one row from one table
71   * selected randomly and outputs some information about failure or latency.
72   */
73  public final class Canary implements Tool {
74    // Sink interface used by the canary to outputs information
75    public interface Sink {
76      public void publishReadFailure(HRegionInfo region, Exception e);
77      public void publishReadFailure(HRegionInfo region, HColumnDescriptor column, Exception e);
78      public void publishReadTiming(HRegionInfo region, HColumnDescriptor column, long msTime);
79    }
80    // new extended sink for output regionserver mode info
81    // do not change the Sink interface directly due to maintaining the API
82    public interface ExtendedSink extends Sink {
83      public void publishReadFailure(String table, String server);
84      public void publishReadTiming(String table, String server, long msTime);
85    }

99      }

102     public void publishReadTiming(HRegionInfo region, HColumnDescriptor column, long msTime) {
103       LOG.info(String.format("read from region %s column family %s in %dms",
104                region.getRegionNameAsString(), column.getNameAsString(), msTime));
105     }

160   @Override
161   public int run(String[] args) throws Exception {
162     int index = -1;
163     ChoreService choreService = null;
164 
165     // Process command line args
166     for (int i = 0; i < args.length; i++) {
167       String cmd = args[i];
168 
169       if (cmd.startsWith("-")) {
170         if (index >= 0) {
171           // command line args must be in the form: [opts] [table 1 [table 2 ...]]
172           System.err.println("Invalid command line options");
173           printUsageAndExit();
174         }
175 
176         if (cmd.equals("-help")) {
177           // user asked for help, print the help and quit.
178           printUsageAndExit();
179         } else if (cmd.equals("-daemon") && interval == 0) {
180           // user asked for daemon mode, set a default interval between checks
181           interval = DEFAULT_INTERVAL;
182         } else if (cmd.equals("-interval")) {
183           // user has specified an interval for canary breaths (-interval N)
184           i++;
185 
186           if (i == args.length) {
187             System.err.println("-interval needs a numeric value argument.");
188             printUsageAndExit();
189           }
190 
191           try {
192             interval = Long.parseLong(args[i]) * 1000;
193           } catch (NumberFormatException e) {
194             System.err.println("-interval needs a numeric value argument.");
195             printUsageAndExit();
196           }
197         } else if(cmd.equals("-regionserver")) {
198           this.regionServerMode = true;
199         } else if (cmd.equals("-e")) {
200           this.useRegExp = true;
201         } else if (cmd.equals("-t")) {
202           i++;
203 
204           if (i == args.length) {
205             System.err.println("-t needs a numeric value argument.");
206             printUsageAndExit();
207           }
208 
209           try {
210             this.timeout = Long.parseLong(args[i]);
211           } catch (NumberFormatException e) {
212             System.err.println("-t needs a numeric value argument.");
213             printUsageAndExit();
214           }
215 
216         } else if (cmd.equals("-f")) {
217           i++;
218 
219           if (i == args.length) {
220             System.err
221                 .println("-f needs a boolean value argument (true|false).");
222             printUsageAndExit();
223           }
224 
225           this.failOnError = Boolean.parseBoolean(args[i]);
226         } else {
227           // no options match
228           System.err.println(cmd + " options is invalid.");
229           printUsageAndExit();
230         }
231       } else if (index < 0) {
232         // keep track of first table name specified by the user
233         index = i;
234       }
235     }
236 
237     // Launches chore for refreshing kerberos credentials if security is enabled.
238     // Please see http://hbase.apache.org/book.html#_running_canary_in_a_kerberos_enabled_cluster
239     // for more details.
240     final ScheduledChore authChore = AuthUtil.getAuthChore(conf);
241     if (authChore != null) {
242       choreService = new ChoreService("CANARY_TOOL");
243       choreService.scheduleChore(authChore);
244     }
245 
246     // Start to prepare the stuffs
247     Monitor monitor = null;
248     Thread monitorThread = null;
249     long startTime = 0;
250     long currentTimeLength = 0;
251     // Get a connection to use in below.
252     // try-with-resources jdk7 construct. See
253     // http://docs.oracle.com/javase/tutorial/essential/exceptions/tryResourceClose.html
254     try (Connection connection = ConnectionFactory.createConnection(this.conf)) {
255       do {
256         // Do monitor !!
257         try {
258           monitor = this.newMonitor(connection, index, args);
259           monitorThread = new Thread(monitor);
260           startTime = System.currentTimeMillis();
261           monitorThread.start();
262           while (!monitor.isDone()) {
263             // wait for 1 sec
264             Thread.sleep(1000);
265             // exit if any error occurs
266             if (this.failOnError && monitor.hasError()) {
267               monitorThread.interrupt();
268               if (monitor.initialized) {
269                 System.exit(monitor.errorCode);
270               } else {
271                 System.exit(INIT_ERROR_EXIT_CODE);
272               }
273             }
274             currentTimeLength = System.currentTimeMillis() - startTime;
275             if (currentTimeLength > this.timeout) {
276               LOG.error("The monitor is running too long (" + currentTimeLength
277                   + ") after timeout limit:" + this.timeout
278                   + " will be killed itself !!");
279               if (monitor.initialized) {
280                 System.exit(TIMEOUT_ERROR_EXIT_CODE);
281               } else {
282                 System.exit(INIT_ERROR_EXIT_CODE);
283               }
284               break;
285             }
286           }
287 
288           if (this.failOnError && monitor.hasError()) {
289             monitorThread.interrupt();
290             System.exit(monitor.errorCode);
291           }
292         } finally {
293           if (monitor != null) monitor.close();
294         }
295 
296         Thread.sleep(interval);
297       } while (interval > 0);
298     } // try-with-resources close
299 
300     if (choreService != null) {
301       choreService.shutdown();
302     }
303     return(monitor.errorCode);
304   }

306   private void printUsageAndExit() {
307     System.err.printf(
308       "Usage: bin/hbase %s [opts] [table1 [table2]...] | [regionserver1 [regionserver2]..]%n",
309         getClass().getName());
310     System.err.println(" where [opts] are:");
311     System.err.println("   -help          Show this help and exit.");
312     System.err.println("   -regionserver  replace the table argument to regionserver,");
313     System.err.println("      which means to enable regionserver mode");
314     System.err.println("   -daemon        Continuous check at defined intervals.");
315     System.err.println("   -interval <N>  Interval between checks (sec)");
316     System.err.println("   -e             Use region/regionserver as regular expression");
317     System.err.println("      which means the region/regionserver is regular expression pattern");
318     System.err.println("   -f <B>         stop whole program if first error occurs," +
319         " default is true");
320     System.err.println("   -t <N>         timeout for a check, default is 600000 (milisecs)");
321     System.exit(USAGE_EXIT_CODE);
322   }

324   /**
325    * A Factory method for {@link Monitor}.
326    * Can be overridden by user.
327    * @param index a start index for monitor target
328    * @param args args passed from user
329    * @return a Monitor instance
330    */
331   public Monitor newMonitor(final Connection connection, int index, String[] args) {
332     Monitor monitor = null;
333     String[] monitorTargets = null;
334 
335     if(index >= 0) {
336       int length = args.length - index;
337       monitorTargets = new String[length];
338       System.arraycopy(args, index, monitorTargets, 0, length);
339     }
340 
341     if(this.regionServerMode) {
342       monitor = new RegionServerMonitor(
343           connection,
344           monitorTargets,
345           this.useRegExp,
346           (ExtendedSink)this.sink);
347     } else {
348       monitor = new RegionMonitor(connection, monitorTargets, this.useRegExp, this.sink);
349     }
350     return monitor;
351   }

353   // a Monitor super-class can be extended by users
354   public static abstract class Monitor implements Runnable, Closeable {
355 
356     protected Connection connection;
357     protected Admin admin;
358     protected String[] targets;
359     protected boolean useRegExp;
360     protected boolean initialized = false;
361 
362     protected boolean done = false;
363     protected int errorCode = 0;
364     protected Sink sink;
365 
366     public boolean isDone() {
367       return done;
368     }
369 
370     public boolean hasError() {
371       return errorCode != 0;
372     }
373 
374     @Override
375     public void close() throws IOException {
376       if (this.admin != null) this.admin.close();
377     }
378 
379     protected Monitor(Connection connection, String[] monitorTargets,
380         boolean useRegExp, Sink sink) {
381       if (null == connection) throw new IllegalArgumentException("connection shall not be null");
382 
383       this.connection = connection;
384       this.targets = monitorTargets;
385       this.useRegExp = useRegExp;
386       this.sink = sink;
387     }
388 
389     public abstract void run();
390 
391     protected boolean initAdmin() {
392       if (null == this.admin) {
393         try {
394           this.admin = this.connection.getAdmin();
395         } catch (Exception e) {
396           LOG.error("Initial HBaseAdmin failed...", e);
397           this.errorCode = INIT_ERROR_EXIT_CODE;
398         }
399       } else if (admin.isAborted()) {
400         LOG.error("HBaseAdmin aborted");
401         this.errorCode = INIT_ERROR_EXIT_CODE;
402       }
403       return !this.hasError();
404     }
405   }

407   // a monitor for region mode
408   private static class RegionMonitor extends Monitor {
409 
410     public RegionMonitor(Connection connection, String[] monitorTargets,
411         boolean useRegExp, Sink sink) {
412       super(connection, monitorTargets, useRegExp, sink);
413     }
414 
415     @Override
416     public void run() {
417       if(this.initAdmin()) {
418         try {
419           if (this.targets != null && this.targets.length > 0) {
420             String[] tables = generateMonitorTables(this.targets);
421             this.initialized = true;
422             for (String table : tables) {
423               Canary.sniff(admin, sink, table);
424             }
425           } else {
426             sniff();
427           }
428         } catch (Exception e) {
429           LOG.error("Run regionMonitor failed", e);
430           this.errorCode = ERROR_EXIT_CODE;
431         }
432       }
433       this.done = true;
434     }
435 
436     private String[] generateMonitorTables(String[] monitorTargets) throws IOException {
437       String[] returnTables = null;
438 
439       if(this.useRegExp) {
440         Pattern pattern = null;
441         HTableDescriptor[] tds = null;
442         Set<String> tmpTables = new TreeSet<String>();
443         try {
444           for (String monitorTarget : monitorTargets) {
445             pattern = Pattern.compile(monitorTarget);
446             tds = this.admin.listTables(pattern);
447             if (tds != null) {
448               for (HTableDescriptor td : tds) {
449                 tmpTables.add(td.getNameAsString());
450               }
451             }
452           }
453         } catch(IOException e) {
454           LOG.error("Communicate with admin failed", e);
455           throw e;
456         }
457 
458         if(tmpTables.size() > 0) {
459           returnTables = tmpTables.toArray(new String[tmpTables.size()]);
460         } else {
461           String msg = "No HTable found, tablePattern:"
462               + Arrays.toString(monitorTargets);
463           LOG.error(msg);
464           this.errorCode = INIT_ERROR_EXIT_CODE;
465           throw new TableNotFoundException(msg);
466         }
467       } else {
468         returnTables = monitorTargets;
469       }
470 
471       return returnTables;
472     }

499     if (admin.isTableAvailable(TableName.valueOf(tableName))) {

502       LOG.warn(String.format("Table %s is not available", tableName));

506   /*
507    * Loops over regions that owns this table, and output some information abouts the state.
508    */
509   private static void sniff(final Admin admin, final Sink sink, HTableDescriptor tableDesc)
510       throws Exception {
511     Table table = null;
512 
513     try {
514       table = admin.getConnection().getTable(tableDesc.getTableName());
515     } catch (TableNotFoundException e) {
516       return;
517     }
518 
519     try {
520       for (HRegionInfo region : admin.getTableRegions(tableDesc.getTableName())) {
521         try {
522           sniffRegion(admin, sink, region, table);
523         } catch (Exception e) {
524           sink.publishReadFailure(region, e);
525           LOG.debug("sniffRegion failed", e);
526         }
527       }
528     } finally {
529       table.close();
530     }
531   }

533   /*
534    * For each column family of the region tries to get one row and outputs the latency, or the
535    * failure.
536    */
537   private static void sniffRegion(
538       final Admin admin,
539       final Sink sink,
540       HRegionInfo region,
541       Table table) throws Exception {
542     HTableDescriptor tableDesc = table.getTableDescriptor();
543     byte[] startKey = null;
544     Get get = null;
545     Scan scan = null;
546     ResultScanner rs = null;
547     StopWatch stopWatch = new StopWatch();
548     for (HColumnDescriptor column : tableDesc.getColumnFamilies()) {
549       stopWatch.reset();
550       startKey = region.getStartKey();
551       // Can't do a get on empty start row so do a Scan of first element if any instead.
552       if (startKey.length > 0) {
553         get = new Get(startKey);
554         get.addFamily(column.getName());
555       } else {
556         scan = new Scan();
557         scan.setCaching(1);
558         scan.addFamily(column.getName());
559         scan.setMaxResultSize(1L);
560       }
561 
562       try {
563         if (startKey.length > 0) {
564           stopWatch.start();
565           table.get(get);
566           stopWatch.stop();
567           sink.publishReadTiming(region, column, stopWatch.getTime());
568         } else {
569           stopWatch.start();
570           rs = table.getScanner(scan);
571           stopWatch.stop();
572           sink.publishReadTiming(region, column, stopWatch.getTime());
573         }
574       } catch (Exception e) {
575         sink.publishReadFailure(region, column, e);
576       } finally {
577         if (rs != null) {
578           rs.close();
579         }
580         scan = null;
581         get = null;
582         startKey = null;
583       }
584     }

586   //a monitor for regionserver mode
587   private static class RegionServerMonitor extends Monitor {
588 
589     public RegionServerMonitor(Connection connection, String[] monitorTargets,
590         boolean useRegExp, ExtendedSink sink) {
591       super(connection, monitorTargets, useRegExp, sink);
592     }
593 
594     private ExtendedSink getSink() {
595       return (ExtendedSink) this.sink;
596     }
597 
598     @Override
599     public void run() {
600       if (this.initAdmin() && this.checkNoTableNames()) {
601         Map<String, List<HRegionInfo>> rsAndRMap = this.filterRegionServerByName();
602         this.initialized = true;
603         this.monitorRegionServers(rsAndRMap);
604       }
605       this.done = true;
606     }
607 
608     private boolean checkNoTableNames() {
609       List<String> foundTableNames = new ArrayList<String>();
610       TableName[] tableNames = null;
611 
612       try {
613         tableNames = this.admin.listTableNames();
614       } catch (IOException e) {
615         LOG.error("Get listTableNames failed", e);
616         this.errorCode = INIT_ERROR_EXIT_CODE;
617         return false;
618       }
619 
620       if (this.targets == null || this.targets.length == 0) return true;
621 
622       for (String target : this.targets) {
623         for (TableName tableName : tableNames) {
624           if (target.equals(tableName.getNameAsString())) {
625             foundTableNames.add(target);
626           }
627         }
628       }
629 
630       if (foundTableNames.size() > 0) {
631         System.err.println("Cannot pass a tablename when using the -regionserver " +
632             "option, tablenames:" + foundTableNames.toString());
633         this.errorCode = USAGE_EXIT_CODE;
634       }
635       return foundTableNames.size() == 0;
636     }
637 
638     private void monitorRegionServers(Map<String, List<HRegionInfo>> rsAndRMap) {
639       String serverName = null;
640       TableName tableName = null;
641       HRegionInfo region = null;
642       Table table = null;
643       Get get = null;
644       byte[] startKey = null;
645       Scan scan = null;
646       StopWatch stopWatch = new StopWatch();
647       // monitor one region on every region server
648       for (Map.Entry<String, List<HRegionInfo>> entry : rsAndRMap.entrySet()) {
649         stopWatch.reset();
650         serverName = entry.getKey();
651         // always get the first region
652         region = entry.getValue().get(0);
653         try {
654           tableName = region.getTable();
655           table = admin.getConnection().getTable(tableName);
656           startKey = region.getStartKey();
657           // Can't do a get on empty start row so do a Scan of first element if any instead.
658           if(startKey.length > 0) {
659             get = new Get(startKey);
660             stopWatch.start();
661             table.get(get);
662             stopWatch.stop();
663           } else {
664             scan = new Scan();
665             scan.setCaching(1);
666             scan.setMaxResultSize(1L);
667             stopWatch.start();
668             ResultScanner s = table.getScanner(scan);
669             s.close();
670             stopWatch.stop();
671           }
672           this.getSink().publishReadTiming(tableName.getNameAsString(),
673               serverName, stopWatch.getTime());
674         } catch (TableNotFoundException tnfe) {
675           // This is ignored because it doesn't imply that the regionserver is dead
676         } catch (TableNotEnabledException tnee) {
677           // This is considered a success since we got a response.
678           LOG.debug("The targeted table was disabled.  Assuming success.");
679         } catch (DoNotRetryIOException dnrioe) {
680             this.getSink().publishReadFailure(tableName.getNameAsString(), serverName);
681             LOG.error(dnrioe);
682         } catch (IOException e) {
683           this.getSink().publishReadFailure(tableName.getNameAsString(), serverName);
684           LOG.error(e);
685           this.errorCode = ERROR_EXIT_CODE;
686         } finally {
687           if (table != null) {
688             try {
689               table.close();
690             } catch (IOException e) {/* DO NOTHING */
691             }
692           }
693           scan = null;
694           get = null;
695           startKey = null;
696         }
697       }
698     }

700     private Map<String, List<HRegionInfo>> filterRegionServerByName() {
701       Map<String, List<HRegionInfo>> regionServerAndRegionsMap = this.getAllRegionServerByName();
702       regionServerAndRegionsMap = this.doFilterRegionServerByName(regionServerAndRegionsMap);
703       return regionServerAndRegionsMap;
704     }

706     private Map<String, List<HRegionInfo>> getAllRegionServerByName() {
707       Map<String, List<HRegionInfo>> rsAndRMap = new HashMap<String, List<HRegionInfo>>();
708       Table table = null;
709       RegionLocator regionLocator = null;
710       try {
711         HTableDescriptor[] tableDescs = this.admin.listTables();
712         List<HRegionInfo> regions = null;
713         for (HTableDescriptor tableDesc : tableDescs) {
714           table = this.admin.getConnection().getTable(tableDesc.getTableName());
715           regionLocator = this.admin.getConnection().getRegionLocator(tableDesc.getTableName());
716 
717           for (HRegionLocation location: regionLocator.getAllRegionLocations()) {
718             ServerName rs = location.getServerName();
719             String rsName = rs.getHostname();
720             HRegionInfo r = location.getRegionInfo();
721 
722             if (rsAndRMap.containsKey(rsName)) {
723               regions = rsAndRMap.get(rsName);
724             } else {
725               regions = new ArrayList<HRegionInfo>();
726               rsAndRMap.put(rsName, regions);
727             }
728             regions.add(r);
729           }
730           table.close();
731         }
732 
733       } catch (IOException e) {
734         String msg = "Get HTables info failed";
735         LOG.error(msg, e);
736         this.errorCode = INIT_ERROR_EXIT_CODE;
737       } finally {
738         if (table != null) {
739           try {
740             table.close();
741           } catch (IOException e) {
742             LOG.warn("Close table failed", e);
743           }
744         }
745       }
746 
747       return rsAndRMap;
748     }
749 
750     private Map<String, List<HRegionInfo>> doFilterRegionServerByName(
751         Map<String, List<HRegionInfo>> fullRsAndRMap) {
752 
753       Map<String, List<HRegionInfo>> filteredRsAndRMap = null;
754 
755       if (this.targets != null && this.targets.length > 0) {
756         filteredRsAndRMap = new HashMap<String, List<HRegionInfo>>();
757         Pattern pattern = null;
758         Matcher matcher = null;
759         boolean regExpFound = false;
760         for (String rsName : this.targets) {
761           if (this.useRegExp) {
762             regExpFound = false;
763             pattern = Pattern.compile(rsName);
764             for (Map.Entry<String,List<HRegionInfo>> entry : fullRsAndRMap.entrySet()) {
765               matcher = pattern.matcher(entry.getKey());
766               if (matcher.matches()) {
767                 filteredRsAndRMap.put(entry.getKey(), entry.getValue());
768                 regExpFound = true;
769               }
770             }
771             if (!regExpFound) {
772               LOG.info("No RegionServerInfo found, regionServerPattern:" + rsName);
773             }
774           } else {
775             if (fullRsAndRMap.containsKey(rsName)) {
776               filteredRsAndRMap.put(rsName, fullRsAndRMap.get(rsName));
777             } else {
778               LOG.info("No RegionServerInfo found, regionServerName:" + rsName);
779             }
780           }
781         }
782       } else {
783         filteredRsAndRMap = fullRsAndRMap;
784       }
785       return filteredRsAndRMap;
786     }
