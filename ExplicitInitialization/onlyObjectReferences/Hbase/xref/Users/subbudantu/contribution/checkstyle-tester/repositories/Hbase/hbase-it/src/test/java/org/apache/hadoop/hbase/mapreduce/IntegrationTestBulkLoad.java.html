88  /**
89   * Test Bulk Load and MR on a distributed cluster.
90   * It starts an MR job that creates linked chains
91   *
92   * The format of rows is like this:
93   * Row Key -> Long
94   *
95   * L:<< Chain Id >> -> Row Key of the next link in the chain
96   * S:<< Chain Id >> -> The step in the chain that his link is.
97   * D:<< Chain Id >> -> Random Data.
98   *
99   * All chains start on row 0.
100  * All rk's are > 0.
101  *
102  * After creating the linked lists they are walked over using a TableMapper based Mapreduce Job.
103  *
104  * There are a few options exposed:
105  *
106  * hbase.IntegrationTestBulkLoad.chainLength
107  * The number of rows that will be part of each and every chain.
108  *
109  * hbase.IntegrationTestBulkLoad.numMaps
110  * The number of mappers that will be run.  Each mapper creates on linked list chain.
111  *
112  * hbase.IntegrationTestBulkLoad.numImportRounds
113  * How many jobs will be run to create linked lists.
114  *
115  * hbase.IntegrationTestBulkLoad.tableName
116  * The name of the table.
117  *
118  * hbase.IntegrationTestBulkLoad.replicaCount
119  * How many region replicas to configure for the table under test.
120  */
121 @Category(IntegrationTests.class)
122 public class IntegrationTestBulkLoad extends IntegrationTestBase {

153   public static class SlowMeCoproScanOperations extends BaseRegionObserver {
154     static final AtomicLong sleepTime = new AtomicLong(2000);
155     Random r = new Random();
156     AtomicLong countOfNext = new AtomicLong(0);
157     AtomicLong countOfOpen = new AtomicLong(0);
158     public SlowMeCoproScanOperations() {}
159     @Override
160     public RegionScanner preScannerOpen(final ObserverContext<RegionCoprocessorEnvironment> e,
161         final Scan scan, final RegionScanner s) throws IOException {
162       if (countOfOpen.incrementAndGet() == 2) { //slowdown openScanner randomly
163         slowdownCode(e);
164       }
165       return s;
166     }
167 
168     @Override
169     public boolean preScannerNext(final ObserverContext<RegionCoprocessorEnvironment> e,
170         final InternalScanner s, final List<Result> results,
171         final int limit, final boolean hasMore) throws IOException {
172       //this will slow down a certain next operation if the conditions are met. The slowness
173       //will allow the call to go to a replica
174       countOfNext.incrementAndGet();
175       if (countOfNext.get() == 0 || countOfNext.get() == 4) {
176         slowdownCode(e);
177       }
178       return true;
179     }
180     protected void slowdownCode(final ObserverContext<RegionCoprocessorEnvironment> e) {
181       if (e.getEnvironment().getRegion().getRegionInfo().getReplicaId() == 0) {
182         try {
183           if (sleepTime.get() > 0) {
184             LOG.info("Sleeping for " + sleepTime.get() + " ms");
185             Thread.sleep(sleepTime.get());
186           }
187         } catch (InterruptedException e1) {
188           LOG.error(e1);
189         }
190       }
191     }
192   }

205     HBaseTestingUtility.modifyTableSync(admin, desc);
206     //sleep for sometime. Hope is that the regions are closed/opened before 
207     //the sleep returns. TODO: do this better
208     Thread.sleep(30000);

252   private void runLinkedListMRJob(int iteration) throws Exception {
253     String jobName =  IntegrationTestBulkLoad.class.getSimpleName() + " - " +
254         EnvironmentEdgeManager.currentTime();
255     Configuration conf = new Configuration(util.getConfiguration());
256     Path p = util.getDataTestDirOnTestFS(getTablename() +  "-" + iteration);
257 
258     conf.setBoolean("mapreduce.map.speculative", false);
259     conf.setBoolean("mapreduce.reduce.speculative", false);
260     conf.setInt(ROUND_NUM_KEY, iteration);
261 
262     Job job = new Job(conf);
263 
264     job.setJobName(jobName);
265 
266     // set the input format so that we can create map tasks with no data input.
267     job.setInputFormatClass(ITBulkLoadInputFormat.class);
268 
269     // Set the mapper classes.
270     job.setMapperClass(LinkedListCreationMapper.class);
271     job.setMapOutputKeyClass(ImmutableBytesWritable.class);
272     job.setMapOutputValueClass(KeyValue.class);
273 
274     // Use the identity reducer
275     // So nothing to do here.
276 
277     // Set this jar.
278     job.setJarByClass(getClass());
279 
280     // Set where to place the hfiles.
281     FileOutputFormat.setOutputPath(job, p);
282     try (Connection conn = ConnectionFactory.createConnection(conf);
283         Admin admin = conn.getAdmin();
284         Table table = conn.getTable(getTablename());
285         RegionLocator regionLocator = conn.getRegionLocator(getTablename())) {
286       
287       // Configure the partitioner and other things needed for HFileOutputFormat.
288       HFileOutputFormat2.configureIncrementalLoad(job, table.getTableDescriptor(), regionLocator);
289 
290       // Run the job making sure it works.
291       assertEquals(true, job.waitForCompletion(true));
292 
293       // Create a new loader.
294       LoadIncrementalHFiles loader = new LoadIncrementalHFiles(conf);
295 
296       // Load the HFiles in.
297       loader.doBulkLoad(p, admin, table, regionLocator);
298     }
299 
300     // Delete the files.
301     util.getTestFileSystem().delete(p, true);
302   }

316     private int index = -1;

331     @Override

348   public static class ITBulkLoadInputFormat extends InputFormat<LongWritable, LongWritable> {
349     @Override
350     public List<InputSplit> getSplits(JobContext context) throws IOException, InterruptedException {
351       int numSplits = context.getConfiguration().getInt(NUM_MAPS_KEY, NUM_MAPS);
352       ArrayList<InputSplit> ret = new ArrayList<InputSplit>(numSplits);
353       for (int i = 0; i < numSplits; ++i) {
354         ret.add(new EmptySplit());
355       }
356       return ret;
357     }
358 
359     @Override
360     public RecordReader<LongWritable, LongWritable> createRecordReader(InputSplit split,
361       TaskAttemptContext context)
362           throws IOException, InterruptedException {
363       int taskId = context.getTaskAttemptID().getTaskID().getId();
364       int numMapTasks = context.getConfiguration().getInt(NUM_MAPS_KEY, NUM_MAPS);
365       int numIterations = context.getConfiguration().getInt(NUM_IMPORT_ROUNDS_KEY, NUM_IMPORT_ROUNDS);
366       int iteration = context.getConfiguration().getInt(ROUND_NUM_KEY, 0);
367 
368       taskId = taskId + iteration * numMapTasks;
369       numMapTasks = numMapTasks * numIterations;
370 
371       long chainId = Math.abs(new Random().nextLong());
372       chainId = chainId - (chainId % numMapTasks) + taskId; // ensure that chainId is unique per task and across iterations
373       LongWritable[] keys = new LongWritable[] {new LongWritable(chainId)};
374 
375       return new FixedRecordReader<LongWritable, LongWritable>(keys, keys);
376     }
377   }

379   /**
380    * Mapper that creates a linked list of KeyValues.
381    *
382    * Each map task generates one linked list.
383    * All lists start on row key 0L.
384    * All lists should be CHAIN_LENGTH long.
385    */
386   public static class LinkedListCreationMapper
387       extends Mapper<LongWritable, LongWritable, ImmutableBytesWritable, KeyValue> {
388 
389     private Random rand = new Random();
390 
391     @Override
392     protected void map(LongWritable key, LongWritable value, Context context)
393         throws IOException, InterruptedException {
394       long chainId = value.get();
395       LOG.info("Starting mapper with chainId:" + chainId);
396 
397       byte[] chainIdArray = Bytes.toBytes(chainId);
398       long currentRow = 0;
399 
400       long chainLength = context.getConfiguration().getLong(CHAIN_LENGTH_KEY, CHAIN_LENGTH);
401       long nextRow = getNextRow(0, chainLength);
402 
403       for (long i = 0; i < chainLength; i++) {
404         byte[] rk = Bytes.toBytes(currentRow);
405 
406         // Next link in the chain.
407         KeyValue linkKv = new KeyValue(rk, CHAIN_FAM, chainIdArray, Bytes.toBytes(nextRow));
408         // What link in the chain this is.
409         KeyValue sortKv = new KeyValue(rk, SORT_FAM, chainIdArray, Bytes.toBytes(i));
410         // Added data so that large stores are created.
411         KeyValue dataKv = new KeyValue(rk, DATA_FAM, chainIdArray,
412           Bytes.toBytes(RandomStringUtils.randomAlphabetic(50))
413         );
414 
415         // Emit the key values.
416         context.write(new ImmutableBytesWritable(rk), linkKv);
417         context.write(new ImmutableBytesWritable(rk), sortKv);
418         context.write(new ImmutableBytesWritable(rk), dataKv);
419         // Move to the next row.
420         currentRow = nextRow;
421         nextRow = getNextRow(i+1, chainLength);
422       }
423     }
424 
425     /** Returns a unique row id within this chain for this index */
426     private long getNextRow(long index, long chainLength) {
427       long nextRow = Math.abs(rand.nextLong());
428       // use significant bits from the random number, but pad with index to ensure it is unique
429       // this also ensures that we do not reuse row = 0
430       // row collisions from multiple mappers are fine, since we guarantee unique chainIds
431       nextRow = nextRow - (nextRow % chainLength) + index;
432       return nextRow;
433     }
434   }

581   /**
582    * Mapper to pass over the table.
583    *
584    * For every row there could be multiple chains that landed on this row. So emit a linkKey
585    * and value for each.
586    */
587   public static class LinkedListCheckingMapper extends TableMapper<LinkKey, LinkChain> {
588     @Override
589     protected void map(ImmutableBytesWritable key, Result value, Context context)
590         throws IOException, InterruptedException {
591       long longRk = Bytes.toLong(value.getRow());
592 
593       for (Map.Entry<byte[], byte[]> entry : value.getFamilyMap(CHAIN_FAM).entrySet()) {
594         long chainId = Bytes.toLong(entry.getKey());
595         long next = Bytes.toLong(entry.getValue());
596         Cell c = value.getColumnCells(SORT_FAM, entry.getKey()).get(0);
597         long order = Bytes.toLong(CellUtil.cloneValue(c));
598         context.write(new LinkKey(chainId, order), new LinkChain(longRk, next));
599       }
600     }
601   }

603   /**
604    * Class that does the actual checking of the links.
605    *
606    * All links in the chain should be grouped and sorted when sent to this class.  Then the chain
607    * will be traversed making sure that no link is missing and that the chain is the correct length.
608    *
609    * This will throw an exception if anything is not correct.  That causes the job to fail if any
610    * data is corrupt.
611    */
612   public static class LinkedListCheckingReducer
613       extends Reducer<LinkKey, LinkChain, NullWritable, NullWritable> {
614     @Override
615     protected void reduce(LinkKey key, Iterable<LinkChain> values, Context context)
616         throws java.io.IOException, java.lang.InterruptedException {
617       long next = -1L;
618       long prev = -1L;
619       long count = 0L;
620 
621       for (LinkChain lc : values) {
622 
623         if (next == -1) {
624           if (lc.getRk() != 0L) {
625             String msg = "Chains should all start at rk 0, but read rk " + lc.getRk()
626                 + ". Chain:" + key.chainId + ", order:" + key.order;
627             logError(msg, context);
628             throw new RuntimeException(msg);
629           }
630           next = lc.getNext();
631         } else {
632           if (next != lc.getRk()) {
633             String msg = "Missing a link in the chain. Prev rk " + prev + " was, expecting "
634                 + next + " but got " + lc.getRk() + ". Chain:" + key.chainId
635                 + ", order:" + key.order;
636             logError(msg, context);
637             throw new RuntimeException(msg);
638           }
639           prev = lc.getRk();
640           next = lc.getNext();
641         }
642         count++;
643       }
644 
645       int expectedChainLen = context.getConfiguration().getInt(CHAIN_LENGTH_KEY, CHAIN_LENGTH);
646       if (count != expectedChainLen) {
647         String msg = "Chain wasn't the correct length.  Expected " + expectedChainLen + " got "
648             + count + ". Chain:" + key.chainId + ", order:" + key.order;
649         logError(msg, context);
650         throw new RuntimeException(msg);
651       }
652     }
653 
654     private static void logError(String msg, Context context) throws IOException {
655       HBaseTestingUtility util = new HBaseTestingUtility(context.getConfiguration());
656       TableName table = getTableName(context.getConfiguration());
657 
658       LOG.error("Failure in chain verification: " + msg);
659       LOG.error("cluster status:\n" + util.getHBaseClusterInterface().getClusterStatus());
660       LOG.error("table regions:\n"
661           + Joiner.on("\n").join(util.getHBaseAdmin().getTableRegions(table)));
662     }
663   }

665   /**
666    * After adding data to the table start a mr job to
667    * @throws IOException
668    * @throws ClassNotFoundException
669    * @throws InterruptedException
670    */
671   private void runCheck() throws IOException, ClassNotFoundException, InterruptedException {
672     LOG.info("Running check");
673     Configuration conf = getConf();
674     String jobName = getTablename() + "_check" + EnvironmentEdgeManager.currentTime();
675     Path p = util.getDataTestDirOnTestFS(jobName);
676 
677     Job job = new Job(conf);
678     job.setJarByClass(getClass());
679     job.setJobName(jobName);
680 
681     job.setPartitionerClass(NaturalKeyPartitioner.class);
682     job.setGroupingComparatorClass(NaturalKeyGroupingComparator.class);
683     job.setSortComparatorClass(CompositeKeyComparator.class);
684 
685     Scan scan = new Scan();
686     scan.addFamily(CHAIN_FAM);
687     scan.addFamily(SORT_FAM);
688     scan.setMaxVersions(1);
689     scan.setCacheBlocks(false);
690     scan.setBatch(1000);
691 
692     int replicaCount = conf.getInt(NUM_REPLICA_COUNT_KEY, NUM_REPLICA_COUNT_DEFAULT);
693     if (replicaCount != NUM_REPLICA_COUNT_DEFAULT) {
694       scan.setConsistency(Consistency.TIMELINE);
695     }
696 
697     TableMapReduceUtil.initTableMapperJob(
698         getTablename().getName(),
699         scan,
700         LinkedListCheckingMapper.class,
701         LinkKey.class,
702         LinkChain.class,
703         job
704     );
705 
706     job.setReducerClass(LinkedListCheckingReducer.class);
707     job.setOutputKeyClass(NullWritable.class);
708     job.setOutputValueClass(NullWritable.class);
709 
710     FileOutputFormat.setOutputPath(job, p);
711 
712     assertEquals(true, job.waitForCompletion(true));
713 
714     // Delete the files.
715     util.getTestFileSystem().delete(p, true);
716   }
