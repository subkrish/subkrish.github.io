103 /**
104  * Implementation of {@link WAL} to go against {@link FileSystem}; i.e. keep WALs in HDFS.
105  * Only one WAL is ever being written at a time.  When a WAL hits a configured maximum size,
106  * it is rolled.  This is done internal to the implementation.
107  *
108  * <p>As data is flushed from the MemStore to other on-disk structures (files sorted by
109  * key, hfiles), a WAL becomes obsolete. We can let go of all the log edits/entries for a given
110  * HRegion-sequence id.  A bunch of work in the below is done keeping account of these region
111  * sequence ids -- what is flushed out to hfiles, and what is yet in WAL and in memory only.
112  *
113  * <p>It is only practical to delete entire files. Thus, we delete an entire on-disk file
114  * <code>F</code> when all of the edits in <code>F</code> have a log-sequence-id that's older
115  * (smaller) than the most-recent flush.
116  *
117  * <p>To read an WAL, call {@link WALFactory#createReader(org.apache.hadoop.fs.FileSystem,
118  * org.apache.hadoop.fs.Path)}.
119  */

122   // IMPLEMENTATION NOTES:
123   //
124   // At the core is a ring buffer.  Our ring buffer is the LMAX Disruptor.  It tries to
125   // minimize synchronizations and volatile writes when multiple contending threads as is the case
126   // here appending and syncing on a single WAL.  The Disruptor is configured to handle multiple
127   // producers but it has one consumer only (the producers in HBase are IPC Handlers calling append
128   // and then sync).  The single consumer/writer pulls the appends and syncs off the ring buffer.
129   // When a handler calls sync, it is given back a future. The producer 'blocks' on the future so
130   // it does not return until the sync completes.  The future is passed over the ring buffer from
131   // the producer/handler to the consumer thread where it does its best to batch up the producer
132   // syncs so one WAL sync actually spans multiple producer sync invocations.  How well the
133   // batching works depends on the write rate; i.e. we tend to batch more in times of
134   // high writes/syncs.
135   //
136   // Calls to append now also wait until the append has been done on the consumer side of the
137   // disruptor.  We used to not wait but it makes the implemenation easier to grok if we have
138   // the region edit/sequence id after the append returns.
139   // 
140   // TODO: Handlers need to coordinate appending AND syncing.  Can we have the threads contend
141   // once only?  Probably hard given syncs take way longer than an append.
142   //
143   // The consumer threads pass the syncs off to multiple syncing threads in a round robin fashion
144   // to ensure we keep up back-to-back FS sync calls (FS sync calls are the long poll writing the
145   // WAL).  The consumer thread passes the futures to the sync threads for it to complete
146   // the futures when done.
147   //
148   // The 'sequence' in the below is the sequence of the append/sync on the ringbuffer.  It
149   // acts as a sort-of transaction id.  It is always incrementing.
150   //
151   // The RingBufferEventHandler class hosts the ring buffer consuming code.  The threads that
152   // do the actual FS sync are implementations of SyncRunner.  SafePointZigZagLatch is a
153   // synchronization class used to halt the consumer at a safe point --  just after all outstanding
154   // syncs and appends have completed -- so the log roller can swap the WAL out under it.

443   /**
444    * Create an edit log at the given <code>dir</code> location.
445    *
446    * You should never have to load an existing log. If there is a log at
447    * startup, it should have already been processed and deleted by the time the
448    * WAL object is started up.
449    *
450    * @param fs filesystem handle
451    * @param rootDir path to where logs and oldlogs
452    * @param logDir dir where wals are stored
453    * @param archiveDir dir where wals are archived
454    * @param conf configuration to use
455    * @param listeners Listeners on WAL events. Listeners passed here will
456    * be registered before we do anything else; e.g. the
457    * Constructor {@link #rollWriter()}.
458    * @param failIfWALExists If true IOException will be thrown if files related to this wal
459    *        already exist.
460    * @param prefix should always be hostname and port in distributed env and
461    *        it will be URL encoded before being used.
462    *        If prefix is null, "wal" will be used
463    * @param suffix will be url encoded. null is treated as empty. non-empty must start with
464    *        {@link DefaultWALProvider#WAL_FILE_NAME_DELIMITER}
465    * @throws IOException
466    */
467   public FSHLog(final FileSystem fs, final Path rootDir, final String logDir,
468       final String archiveDir, final Configuration conf,
469       final List<WALActionsListener> listeners,
470       final boolean failIfWALExists, final String prefix, final String suffix)
471       throws IOException {
472     this.fs = fs;
473     this.fullPathLogDir = new Path(rootDir, logDir);
474     this.fullPathArchiveDir = new Path(rootDir, archiveDir);
475     this.conf = conf;
476 
477     if (!fs.exists(fullPathLogDir) && !fs.mkdirs(fullPathLogDir)) {
478       throw new IOException("Unable to mkdir " + fullPathLogDir);
479     }
480 
481     if (!fs.exists(this.fullPathArchiveDir)) {
482       if (!fs.mkdirs(this.fullPathArchiveDir)) {
483         throw new IOException("Unable to mkdir " + this.fullPathArchiveDir);
484       }
485     }
486 
487     // If prefix is null||empty then just name it wal
488     this.logFilePrefix =
489       prefix == null || prefix.isEmpty() ? "wal" : URLEncoder.encode(prefix, "UTF8");
490     // we only correctly differentiate suffices when numeric ones start with '.'
491     if (suffix != null && !(suffix.isEmpty()) && !(suffix.startsWith(WAL_FILE_NAME_DELIMITER))) {
492       throw new IllegalArgumentException("wal suffix must start with '" + WAL_FILE_NAME_DELIMITER +
493           "' but instead was '" + suffix + "'");
494     }
495     // Now that it exists, set the storage policy for the entire directory of wal files related to
496     // this FSHLog instance
497     FSUtils.setStoragePolicy(fs, conf, this.fullPathLogDir, HConstants.WAL_STORAGE_POLICY,
498       HConstants.DEFAULT_WAL_STORAGE_POLICY);
499     this.logFileSuffix = (suffix == null) ? "" : URLEncoder.encode(suffix, "UTF8");
500     this.prefixPathStr = new Path(fullPathLogDir,
501         logFilePrefix + WAL_FILE_NAME_DELIMITER).toString();
502 
503     this.ourFiles = new PathFilter() {
504       @Override
505       public boolean accept(final Path fileName) {
506         // The path should start with dir/<prefix> and end with our suffix
507         final String fileNameString = fileName.toString();
508         if (!fileNameString.startsWith(prefixPathStr)) {
509           return false;
510         }
511         if (logFileSuffix.isEmpty()) {
512           // in the case of the null suffix, we need to ensure the filename ends with a timestamp.
513           return org.apache.commons.lang.StringUtils.isNumeric(
514               fileNameString.substring(prefixPathStr.length()));
515         } else if (!fileNameString.endsWith(logFileSuffix)) {
516           return false;
517         }
518         return true;
519       }
520     };
521 
522     if (failIfWALExists) {
523       final FileStatus[] walFiles = FSUtils.listStatus(fs, fullPathLogDir, ourFiles);
524       if (null != walFiles && 0 != walFiles.length) {
525         throw new IOException("Target WAL already exists within directory " + fullPathLogDir);
526       }
527     }
528 
529     // Register listeners.  TODO: Should this exist anymore?  We have CPs?
530     if (listeners != null) {
531       for (WALActionsListener i: listeners) {
532         registerWALActionsListener(i);
533       }
534     }
535     this.coprocessorHost = new WALCoprocessorHost(this, conf);
536 
537     // Get size to roll log at. Roll at 95% of HDFS block size so we avoid crossing HDFS blocks
538     // (it costs a little x'ing bocks)
539     final long blocksize = this.conf.getLong("hbase.regionserver.hlog.blocksize",
540         FSUtils.getDefaultBlockSize(this.fs, this.fullPathLogDir));
541     this.logrollsize =
542       (long)(blocksize * conf.getFloat("hbase.regionserver.logroll.multiplier", 0.95f));
543 
544     this.maxLogs = conf.getInt("hbase.regionserver.maxlogs", 32);
545     this.minTolerableReplication = conf.getInt( "hbase.regionserver.hlog.tolerable.lowreplication",
546         FSUtils.getDefaultReplication(fs, this.fullPathLogDir));
547     this.lowReplicationRollLimit =
548       conf.getInt("hbase.regionserver.hlog.lowreplication.rolllimit", 5);
549     this.closeErrorsTolerated = conf.getInt("hbase.regionserver.logroll.errors.tolerated", 0);
550     int maxHandlersCount = conf.getInt(HConstants.REGION_SERVER_HANDLER_COUNT, 200);
551 
552     LOG.info("WAL configuration: blocksize=" + StringUtils.byteDesc(blocksize) +
553       ", rollsize=" + StringUtils.byteDesc(this.logrollsize) +
554       ", prefix=" + this.logFilePrefix + ", suffix=" + logFileSuffix + ", logDir=" +
555       this.fullPathLogDir + ", archiveDir=" + this.fullPathArchiveDir);
556 
557     // rollWriter sets this.hdfs_out if it can.
558     rollWriter();
559 
560     this.slowSyncNs =
561         1000000 * conf.getInt("hbase.regionserver.hlog.slowsync.ms",
562           DEFAULT_SLOW_SYNC_TIME_MS);
563     // handle the reflection necessary to call getNumCurrentReplicas(). TODO: Replace with
564     // HdfsDataOutputStream#getCurrentBlockReplication() and go without reflection.
565     this.getNumCurrentReplicas = getGetNumCurrentReplicas(this.hdfs_out);
566     this.getPipeLine = getGetPipeline(this.hdfs_out);

568     // This is the 'writer' -- a single threaded executor.  This single thread 'consumes' what is
569     // put on the ring buffer.
570     String hostingThreadName = Thread.currentThread().getName();
571     this.appendExecutor = Executors.
572       newSingleThreadExecutor(Threads.getNamedThreadFactory(hostingThreadName + ".append"));
573     // Preallocate objects to use on the ring buffer.  The way that appends and syncs work, we will
574     // be stuck and make no progress if the buffer is filled with appends only and there is no
575     // sync. If no sync, then the handlers will be outstanding just waiting on sync completion
576     // before they return.
577     final int preallocatedEventCount =
578       this.conf.getInt("hbase.regionserver.wal.disruptor.event.count", 1024 * 16);
579     // Using BlockingWaitStrategy.  Stuff that is going on here takes so long it makes no sense
580     // spinning as other strategies do.
581     this.disruptor =
582       new Disruptor<RingBufferTruck>(RingBufferTruck.EVENT_FACTORY, preallocatedEventCount,
583         this.appendExecutor, ProducerType.MULTI, new BlockingWaitStrategy());
584     // Advance the ring buffer sequence so that it starts from 1 instead of 0,
585     // because SyncFuture.NOT_DONE = 0.
586     this.disruptor.getRingBuffer().next();
587     this.ringBufferEventHandler =
588       new RingBufferEventHandler(conf.getInt("hbase.regionserver.hlog.syncer.count", 5),
589         maxHandlersCount);
590     this.disruptor.handleExceptionsWith(new RingBufferExceptionHandler());
591     this.disruptor.handleEventsWith(new RingBufferEventHandler [] {this.ringBufferEventHandler});
592     // Presize our map of SyncFutures by handler objects.
593     this.syncFuturesByHandler = new ConcurrentHashMap<Thread, SyncFuture>(maxHandlersCount);
594     // Starting up threads in constructor is a no no; Interface should have an init call.
595     this.disruptor.start();
596   }

690   @Override
691   public byte [][] rollWriter(boolean force) throws FailedLogCloseException, IOException {
692     rollWriterLock.lock();
693     try {
694       // Return if nothing to flush.
695       if (!force && (this.writer != null && this.numEntries.get() <= 0)) return null;
696       byte [][] regionsToFlush = null;
697       if (this.closed) {
698         LOG.debug("WAL closed. Skipping rolling of writer");
699         return regionsToFlush;
700       }
701       if (!closeBarrier.beginOp()) {
702         LOG.debug("WAL closing. Skipping rolling of writer");
703         return regionsToFlush;
704       }
705       TraceScope scope = Trace.startSpan("FSHLog.rollWriter");
706       try {
707         Path oldPath = getOldPath();
708         Path newPath = getNewPath();
709         // Any exception from here on is catastrophic, non-recoverable so we currently abort.
710         Writer nextWriter = this.createWriterInstance(newPath);
711         FSDataOutputStream nextHdfsOut = null;
712         if (nextWriter instanceof ProtobufLogWriter) {
713           nextHdfsOut = ((ProtobufLogWriter)nextWriter).getStream();
714           // If a ProtobufLogWriter, go ahead and try and sync to force setup of pipeline.
715           // If this fails, we just keep going.... it is an optimization, not the end of the world.
716           preemptiveSync((ProtobufLogWriter)nextWriter);
717         }
718         tellListenersAboutPreLogRoll(oldPath, newPath);
719         // NewPath could be equal to oldPath if replaceWriter fails.
720         newPath = replaceWriter(oldPath, newPath, nextWriter, nextHdfsOut);
721         tellListenersAboutPostLogRoll(oldPath, newPath);
722         // Can we delete any of the old log files?
723         if (getNumRolledLogFiles() > 0) {
724           cleanOldLogs();
725           regionsToFlush = findRegionsToForceFlush();
726         }
727       } finally {
728         closeBarrier.endOp();
729         assert scope == NullScope.INSTANCE || !scope.isDetached();
730         scope.close();
731       }
732       return regionsToFlush;
733     } finally {
734       rollWriterLock.unlock();
735     }
736   }

770   /**
771    * Archive old logs that could be archived: a log is eligible for archiving if all its WALEdits
772    * have been flushed to hfiles.
773    * <p>
774    * For each log file, it compares its region to sequenceId map
775    * (@link {@link FSHLog#highestRegionSequenceIds} with corresponding region entries in
776    * {@link FSHLog#lowestFlushingRegionSequenceIds} and
777    * {@link FSHLog#oldestUnflushedRegionSequenceIds}. If all the regions in the map are flushed
778    * past of their value, then the wal is eligible for archiving.
779    * @throws IOException
780    */
781   private void cleanOldLogs() throws IOException {
782     Map<byte[], Long> lowestFlushingRegionSequenceIdsLocal = null;
783     Map<byte[], Long> oldestUnflushedRegionSequenceIdsLocal = null;
784     List<Path> logsToArchive = new ArrayList<Path>();
785     // make a local copy so as to avoid locking when we iterate over these maps.
786     synchronized (regionSequenceIdLock) {
787       lowestFlushingRegionSequenceIdsLocal =
788           copyMapWithLowestSeqId(this.lowestFlushingStoreSequenceIds);
789       oldestUnflushedRegionSequenceIdsLocal =
790           copyMapWithLowestSeqId(this.oldestUnflushedStoreSequenceIds);
791     }
792     for (Map.Entry<Path, Map<byte[], Long>> e : byWalRegionSequenceIds.entrySet()) {
793       // iterate over the log file.
794       Path log = e.getKey();
795       Map<byte[], Long> sequenceNums = e.getValue();
796       // iterate over the map for this log file, and tell whether it should be archive or not.
797       if (areAllRegionsFlushed(sequenceNums, lowestFlushingRegionSequenceIdsLocal,
798           oldestUnflushedRegionSequenceIdsLocal)) {
799         logsToArchive.add(log);
800         LOG.debug("WAL file ready for archiving " + log);
801       }
802     }
803     for (Path p : logsToArchive) {
804       this.totalLogSize.addAndGet(-this.fs.getFileStatus(p).getLen());
805       archiveLogFile(p);
806       this.byWalRegionSequenceIds.remove(p);
807     }
808   }

810   /**
811    * Takes a region:sequenceId map for a WAL file, and checks whether the file can be archived.
812    * It compares the region entries present in the passed sequenceNums map with the local copy of
813    * {@link #oldestUnflushedRegionSequenceIds} and {@link #lowestFlushingRegionSequenceIds}. If,
814    * for all regions, the value is lesser than the minimum of values present in the
815    * oldestFlushing/UnflushedSeqNums, then the wal file is eligible for archiving.
816    * @param sequenceNums for a WAL, at the time when it was rolled.
817    * @param oldestFlushingMap
818    * @param oldestUnflushedMap
819    * @return true if wal is eligible for archiving, false otherwise.
820    */
821    static boolean areAllRegionsFlushed(Map<byte[], Long> sequenceNums,
822       Map<byte[], Long> oldestFlushingMap, Map<byte[], Long> oldestUnflushedMap) {
823     for (Map.Entry<byte[], Long> regionSeqIdEntry : sequenceNums.entrySet()) {
824       // find region entries in the flushing/unflushed map. If there is no entry, it meansj
825       // a region doesn't have any unflushed entry.
826       long oldestFlushing = oldestFlushingMap.containsKey(regionSeqIdEntry.getKey()) ?
827           oldestFlushingMap.get(regionSeqIdEntry.getKey()) : Long.MAX_VALUE;
828       long oldestUnFlushed = oldestUnflushedMap.containsKey(regionSeqIdEntry.getKey()) ?
829           oldestUnflushedMap.get(regionSeqIdEntry.getKey()) : Long.MAX_VALUE;
830           // do a minimum to be sure to contain oldest sequence Id
831       long minSeqNum = Math.min(oldestFlushing, oldestUnFlushed);
832       if (minSeqNum <= regionSeqIdEntry.getValue()) return false;// can't archive
833     }
834     return true;
835   }

837   /**
838    * Iterates over the given map of regions, and compares their sequence numbers with corresponding
839    * entries in {@link #oldestUnflushedRegionSequenceIds}. If the sequence number is greater or
840    * equal, the region is eligible to flush, otherwise, there is no benefit to flush (from the
841    * perspective of passed regionsSequenceNums map), because the region has already flushed the
842    * entries present in the WAL file for which this method is called for (typically, the oldest
843    * wal file).
844    * @param regionsSequenceNums
845    * @return regions which should be flushed (whose sequence numbers are larger than their
846    * corresponding un-flushed entries.
847    */
848   private byte[][] findEligibleMemstoresToFlush(Map<byte[], Long> regionsSequenceNums) {
849     List<byte[]> regionsToFlush = null;
850     // Keeping the old behavior of iterating unflushedSeqNums under oldestSeqNumsLock.
851     synchronized (regionSequenceIdLock) {
852       for (Map.Entry<byte[], Long> e: regionsSequenceNums.entrySet()) {
853         long unFlushedVal = getEarliestMemstoreSeqNum(e.getKey());
854         if (unFlushedVal != HConstants.NO_SEQNUM && unFlushedVal <= e.getValue()) {
855           if (regionsToFlush == null)
856             regionsToFlush = new ArrayList<byte[]>();
857           regionsToFlush.add(e.getKey());
858         }
859       }
860     }
861     return regionsToFlush == null ? null : regionsToFlush
862         .toArray(new byte[][] { HConstants.EMPTY_BYTE_ARRAY });
863   }

893   /**
894    * Cleans up current writer closing it and then puts in place the passed in
895    * <code>nextWriter</code>.
896    *
897    * In the case of creating a new WAL, oldPath will be null.
898    *
899    * In the case of rolling over from one file to the next, none of the params will be null.
900    *
901    * In the case of closing out this FSHLog with no further use newPath, nextWriter, and
902    * nextHdfsOut will be null.
903    *
904    * @param oldPath may be null
905    * @param newPath may be null
906    * @param nextWriter may be null
907    * @param nextHdfsOut may be null
908    * @return the passed in <code>newPath</code>
909    * @throws IOException if there is a problem flushing or closing the underlying FS
910    */
911   Path replaceWriter(final Path oldPath, final Path newPath, Writer nextWriter,
912       final FSDataOutputStream nextHdfsOut)
913   throws IOException {
914     // Ask the ring buffer writer to pause at a safe point.  Once we do this, the writer
915     // thread will eventually pause. An error hereafter needs to release the writer thread
916     // regardless -- hence the finally block below.  Note, this method is called from the FSHLog
917     // constructor BEFORE the ring buffer is set running so it is null on first time through
918     // here; allow for that.
919     SyncFuture syncFuture = null;
920     SafePointZigZagLatch zigzagLatch = (this.ringBufferEventHandler == null)?
921       null: this.ringBufferEventHandler.attainSafePoint();
922     TraceScope scope = Trace.startSpan("FSHFile.replaceWriter");
923     try {
924       // Wait on the safe point to be achieved.  Send in a sync in case nothing has hit the
925       // ring buffer between the above notification of writer that we want it to go to
926       // 'safe point' and then here where we are waiting on it to attain safe point.  Use
927       // 'sendSync' instead of 'sync' because we do not want this thread to block waiting on it
928       // to come back.  Cleanup this syncFuture down below after we are ready to run again.
929       try {
930         if (zigzagLatch != null) {
931           Trace.addTimelineAnnotation("awaiting safepoint");
932           syncFuture = zigzagLatch.waitSafePoint(publishSyncOnRingBuffer());
933         }
934       } catch (FailedSyncBeforeLogCloseException e) {
935         if (isUnflushedEntries()) throw e;
936         // Else, let is pass through to the close.
937         LOG.warn("Failed last sync but no outstanding unsync edits so falling through to close; " +
938           e.getMessage());
939       }
940 
941       // It is at the safe point.  Swap out writer from under the blocked writer thread.
942       // TODO: This is close is inline with critical section.  Should happen in background?
943       try {
944         if (this.writer != null) {
945           Trace.addTimelineAnnotation("closing writer");
946           this.writer.close();
947           Trace.addTimelineAnnotation("writer closed");
948         }
949         this.closeErrorCount.set(0);
950       } catch (IOException ioe) {
951         int errors = closeErrorCount.incrementAndGet();
952         if (!isUnflushedEntries() && (errors <= this.closeErrorsTolerated)) {
953           LOG.warn("Riding over failed WAL close of " + oldPath + ", cause=\"" +
954             ioe.getMessage() + "\", errors=" + errors +
955             "; THIS FILE WAS NOT CLOSED BUT ALL EDITS SYNCED SO SHOULD BE OK");
956         } else {
957           throw ioe;
958         }
959       }
960       this.writer = nextWriter;
961       this.hdfs_out = nextHdfsOut;
962       int oldNumEntries = this.numEntries.get();
963       this.numEntries.set(0);
964       final String newPathString = (null == newPath ? null : FSUtils.getPath(newPath));
965       if (oldPath != null) {
966         this.byWalRegionSequenceIds.put(oldPath, this.highestRegionSequenceIds);
967         this.highestRegionSequenceIds = new HashMap<byte[], Long>();
968         long oldFileLen = this.fs.getFileStatus(oldPath).getLen();
969         this.totalLogSize.addAndGet(oldFileLen);
970         LOG.info("Rolled WAL " + FSUtils.getPath(oldPath) + " with entries=" + oldNumEntries +
971           ", filesize=" + StringUtils.byteDesc(oldFileLen) + "; new WAL " +
972           newPathString);
973       } else {
974         LOG.info("New WAL " + newPathString);
975       }
976     } catch (InterruptedException ie) {
977       // Perpetuate the interrupt
978       Thread.currentThread().interrupt();
979     } catch (IOException e) {
980       long count = getUnflushedEntriesCount();
981       LOG.error("Failed close of WAL writer " + oldPath + ", unflushedEntries=" + count, e);
982       throw new FailedLogCloseException(oldPath + ", unflushedEntries=" + count, e);
983     } finally {
984       try {
985         // Let the writer thread go regardless, whether error or not.
986         if (zigzagLatch != null) {
987           zigzagLatch.releaseSafePoint();
988           // It will be null if we failed our wait on safe point above.
989           if (syncFuture != null) blockOnSync(syncFuture);
990         }
991       } finally {
992         scope.close();
993       }
994     }
995     return newPath;
996   }

1114   @Override
1115   public void shutdown() throws IOException {
1116     if (shutdown.compareAndSet(false, true)) {
1117       try {
1118         // Prevent all further flushing and rolling.
1119         closeBarrier.stopAndDrainOps();
1120       } catch (InterruptedException e) {
1121         LOG.error("Exception while waiting for cache flushes and log rolls", e);
1122         Thread.currentThread().interrupt();
1123       }
1124 
1125       // Shutdown the disruptor.  Will stop after all entries have been processed.  Make sure we
1126       // have stopped incoming appends before calling this else it will not shutdown.  We are
1127       // conservative below waiting a long time and if not elapsed, then halting.
1128       if (this.disruptor != null) {
1129         long timeoutms = conf.getLong("hbase.wal.disruptor.shutdown.timeout.ms", 60000);
1130         try {
1131           this.disruptor.shutdown(timeoutms, TimeUnit.MILLISECONDS);
1132         } catch (TimeoutException e) {

1202   /**
1203    * Thread to runs the hdfs sync call. This call takes a while to complete.  This is the longest
1204    * pole adding edits to the WAL and this must complete to be sure all edits persisted.  We run
1205    * multiple threads sync'ng rather than one that just syncs in series so we have better
1206    * latencies; otherwise, an edit that arrived just after a sync started, might have to wait
1207    * almost the length of two sync invocations before it is marked done.
1208    * <p>When the sync completes, it marks all the passed in futures done.  On the other end of the
1209    * sync future is a blocked thread, usually a regionserver Handler.  There may be more than one
1210    * future passed in the case where a few threads arrive at about the same time and all invoke
1211    * 'sync'.  In this case we'll batch up the invocations and run one filesystem sync only for a
1212    * batch of Handler sync invocations.  Do not confuse these Handler SyncFutures with the futures
1213    * an ExecutorService returns when you call submit. We have no use for these in this model. These
1214    * SyncFutures are 'artificial', something to hold the Handler until the filesystem sync
1215    * completes.
1216    */
1217   private class SyncRunner extends HasThread {
1218     private volatile long sequence;
1219     private final BlockingQueue<SyncFuture> syncFutures;
1220  
1221     /**
1222      * UPDATE! 
1223      * @param syncs the batch of calls to sync that arrived as this thread was starting; when done,
1224      * we will put the result of the actual hdfs sync call as the result.
1225      * @param sequence The sequence number on the ring buffer when this thread was set running.
1226      * If this actual writer sync completes then all appends up this point have been
1227      * flushed/synced/pushed to datanodes.  If we fail, then the passed in <code>syncs</code>
1228      * futures will return the exception to their clients; some of the edits may have made it out
1229      * to data nodes but we will report all that were part of this session as failed.
1230      */
1231     SyncRunner(final String name, final int maxHandlersCount) {
1232       super(name);
1233       // LinkedBlockingQueue because of
1234       // http://www.javacodegeeks.com/2010/09/java-best-practices-queue-battle-and.html
1235       // Could use other blockingqueues here or concurrent queues.
1236       //
1237       // We could let the capacity be 'open' but bound it so we get alerted in pathological case
1238       // where we cannot sync and we have a bunch of threads all backed up waiting on their syncs
1239       // to come in.  LinkedBlockingQueue actually shrinks when you remove elements so Q should
1240       // stay neat and tidy in usual case.  Let the max size be three times the maximum handlers.
1241       // The passed in maxHandlerCount is the user-level handlers which is what we put up most of
1242       // but HBase has other handlers running too -- opening region handlers which want to write
1243       // the meta table when succesful (i.e. sync), closing handlers -- etc.  These are usually
1244       // much fewer in number than the user-space handlers so Q-size should be user handlers plus
1245       // some space for these other handlers.  Lets multiply by 3 for good-measure.
1246       this.syncFutures = new LinkedBlockingQueue<SyncFuture>(maxHandlersCount * 3);
1247     }

1249     void offer(final long sequence, final SyncFuture [] syncFutures, final int syncFutureCount) {
1250       // Set sequence first because the add to the queue will wake the thread if sleeping.
1251       this.sequence = sequence;
1252       for (int i = 0; i < syncFutureCount; ++i) {
1253         this.syncFutures.add(syncFutures[i]);
1254       }
1255     }

1257     /**
1258      * Release the passed <code>syncFuture</code>
1259      * @param syncFuture
1260      * @param currentSequence
1261      * @param t
1262      * @return Returns 1.
1263      */
1264     private int releaseSyncFuture(final SyncFuture syncFuture, final long currentSequence,
1265         final Throwable t) {
1266       if (!syncFuture.done(currentSequence, t)) throw new IllegalStateException();
1267       // This function releases one sync future only.
1268       return 1;
1269     }

1271     /**
1272      * Release all SyncFutures whose sequence is <= <code>currentSequence</code>.
1273      * @param currentSequence
1274      * @param t May be non-null if we are processing SyncFutures because an exception was thrown.
1275      * @return Count of SyncFutures we let go.
1276      */
1277     private int releaseSyncFutures(final long currentSequence, final Throwable t) {
1278       int syncCount = 0;
1279       for (SyncFuture syncFuture; (syncFuture = this.syncFutures.peek()) != null;) {
1280         if (syncFuture.getRingBufferSequence() > currentSequence) break;
1281         releaseSyncFuture(syncFuture, currentSequence, t);
1282         if (!this.syncFutures.remove(syncFuture)) {
1283           throw new IllegalStateException(syncFuture.toString());
1284         }
1285         syncCount++;
1286       }
1287       return syncCount;
1288     }

1290     /**
1291      * @param sequence The sequence we ran the filesystem sync against.
1292      * @return Current highest synced sequence.
1293      */
1294     private long updateHighestSyncedSequence(long sequence) {
1295       long currentHighestSyncedSequence;
1296       // Set the highestSyncedSequence IFF our current sequence id is the 'highest'.
1297       do {
1298         currentHighestSyncedSequence = highestSyncedSequence.get();
1299         if (currentHighestSyncedSequence >= sequence) {
1300           // Set the sync number to current highwater mark; might be able to let go more
1301           // queued sync futures
1302           sequence = currentHighestSyncedSequence;
1303           break;
1304         }
1305       } while (!highestSyncedSequence.compareAndSet(currentHighestSyncedSequence, sequence));
1306       return sequence;
1307     }

1309     public void run() {
1310       long currentSequence;
1311       while (!isInterrupted()) {
1312         int syncCount = 0;
1313         SyncFuture takeSyncFuture;
1314         try {
1315           while (true) {
1316             // We have to process what we 'take' from the queue
1317             takeSyncFuture = this.syncFutures.take();
1318             currentSequence = this.sequence;
1319             long syncFutureSequence = takeSyncFuture.getRingBufferSequence();
1320             if (syncFutureSequence > currentSequence) {
1321               throw new IllegalStateException("currentSequence=" + syncFutureSequence +
1322                 ", syncFutureSequence=" + syncFutureSequence);
1323             }
1324             // See if we can process any syncfutures BEFORE we go sync.
1325             long currentHighestSyncedSequence = highestSyncedSequence.get();
1326             if (currentSequence < currentHighestSyncedSequence) {
1327               syncCount += releaseSyncFuture(takeSyncFuture, currentHighestSyncedSequence, null);
1328               // Done with the 'take'.  Go around again and do a new 'take'.
1329               continue;
1330             }
1331             break;
1332           }
1333           // I got something.  Lets run.  Save off current sequence number in case it changes
1334           // while we run.
1335           TraceScope scope = Trace.continueSpan(takeSyncFuture.getSpan());
1336           long start = System.nanoTime();
1337           Throwable t = null;
1338           try {
1339             Trace.addTimelineAnnotation("syncing writer");
1340             writer.sync();
1341             Trace.addTimelineAnnotation("writer synced");
1342             currentSequence = updateHighestSyncedSequence(currentSequence);
1343           } catch (IOException e) {
1344             LOG.error("Error syncing, request close of wal ", e);
1345             t = e;
1346           } catch (Exception e) {
1347             LOG.warn("UNEXPECTED", e);
1348             t = e;
1349           } finally {
1350             // reattach the span to the future before releasing.
1351             takeSyncFuture.setSpan(scope.detach());
1352             // First release what we 'took' from the queue.
1353             syncCount += releaseSyncFuture(takeSyncFuture, currentSequence, t);
1354             // Can we release other syncs?
1355             syncCount += releaseSyncFutures(currentSequence, t);
1356             if (t != null) {
1357               requestLogRoll();
1358             } else checkLogRoll();
1359           }
1360           postSync(System.nanoTime() - start, syncCount);
1361         } catch (InterruptedException e) {
1362           // Presume legit interrupt.
1363           Thread.currentThread().interrupt();
1364         } catch (Throwable t) {
1365           LOG.warn("UNEXPECTED, continuing", t);
1366         }
1367       }
1368     }
1369   }

1392   /*
1393    * @return true if number of replicas for the WAL is lower than threshold
1394    */
1395   private boolean checkLowReplication() {
1396     boolean logRollNeeded = false;
1397     // if the number of replicas in HDFS has fallen below the configured
1398     // value, then roll logs.
1399     try {
1400       int numCurrentReplicas = getLogReplication();
1401       if (numCurrentReplicas != 0 && numCurrentReplicas < this.minTolerableReplication) {
1402         if (this.lowReplicationRollEnabled) {
1403           if (this.consecutiveLogRolls.get() < this.lowReplicationRollLimit) {
1404             LOG.warn("HDFS pipeline error detected. " + "Found "
1405                 + numCurrentReplicas + " replicas but expecting no less than "
1406                 + this.minTolerableReplication + " replicas. "
1407                 + " Requesting close of wal. current pipeline: "
1408                 + Arrays.toString(getPipeLine()));
1409             logRollNeeded = true;
1410             // If rollWriter is requested, increase consecutiveLogRolls. Once it
1411             // is larger than lowReplicationRollLimit, disable the
1412             // LowReplication-Roller
1413             this.consecutiveLogRolls.getAndIncrement();
1414           } else {
1415             LOG.warn("Too many consecutive RollWriter requests, it's a sign of "
1416                 + "the total number of live datanodes is lower than the tolerable replicas.");
1417             this.consecutiveLogRolls.set(0);
1418             this.lowReplicationRollEnabled = false;
1419           }
1420         }
1421       } else if (numCurrentReplicas >= this.minTolerableReplication) {
1422         if (!this.lowReplicationRollEnabled) {
1423           // The new writer's log replicas is always the default value.
1424           // So we should not enable LowReplication-Roller. If numEntries
1425           // is lower than or equals 1, we consider it as a new writer.
1426           if (this.numEntries.get() <= 1) {
1427             return logRollNeeded;
1428           }
1429           // Once the live datanode number and the replicas return to normal,
1430           // enable the LowReplication-Roller.
1431           this.lowReplicationRollEnabled = true;
1432           LOG.info("LowReplication-Roller was enabled.");
1433         }
1434       }
1435     } catch (Exception e) {
1436       LOG.warn("Unable to invoke DFSOutputStream.getNumCurrentReplicas" + e +
1437         " still proceeding ahead...");
1438     }
1439     return logRollNeeded;
1440   }

1784   /**
1785    * This class is used coordinating two threads holding one thread at a
1786    * 'safe point' while the orchestrating thread does some work that requires the first thread
1787    * paused: e.g. holding the WAL writer while its WAL is swapped out from under it by another
1788    * thread.
1789    * 
1790    * <p>Thread A signals Thread B to hold when it gets to a 'safe point'.  Thread A wait until
1791    * Thread B gets there. When the 'safe point' has been attained, Thread B signals Thread A.
1792    * Thread B then holds at the 'safe point'.  Thread A on notification that Thread B is paused,
1793    * goes ahead and does the work it needs to do while Thread B is holding.  When Thread A is done,
1794    * it flags B and then Thread A and Thread B continue along on their merry way.  Pause and
1795    * signalling 'zigzags' between the two participating threads.  We use two latches -- one the
1796    * inverse of the other -- pausing and signaling when states are achieved.
1797    * 
1798    * <p>To start up the drama, Thread A creates an instance of this class each time it would do
1799    * this zigzag dance and passes it to Thread B (these classes use Latches so it is one shot
1800    * only). Thread B notices the new instance (via reading a volatile reference or how ever) and it
1801    * starts to work toward the 'safe point'.  Thread A calls {@link #waitSafePoint()} when it
1802    * cannot proceed until the Thread B 'safe point' is attained. Thread A will be held inside in
1803    * {@link #waitSafePoint()} until Thread B reaches the 'safe point'.  Once there, Thread B
1804    * frees Thread A by calling {@link #safePointAttained()}.  Thread A now knows Thread B
1805    * is at the 'safe point' and that it is holding there (When Thread B calls
1806    * {@link #safePointAttained()} it blocks here until Thread A calls {@link #releaseSafePoint()}).
1807    * Thread A proceeds to do what it needs to do while Thread B is paused.  When finished,
1808    * it lets Thread B lose by calling {@link #releaseSafePoint()} and away go both Threads again.
1809    */
1810   static class SafePointZigZagLatch {
1811     /**
1812      * Count down this latch when safe point attained.
1813      */
1814     private volatile CountDownLatch safePointAttainedLatch = new CountDownLatch(1);
1815     /**
1816      * Latch to wait on.  Will be released when we can proceed.
1817      */
1818     private volatile CountDownLatch safePointReleasedLatch = new CountDownLatch(1);
1819  
1820     /**
1821      * For Thread A to call when it is ready to wait on the 'safe point' to be attained.
1822      * Thread A will be held in here until Thread B calls {@link #safePointAttained()}
1823      * @throws InterruptedException
1824      * @throws ExecutionException
1825      * @param syncFuture We need this as barometer on outstanding syncs.  If it comes home with
1826      * an exception, then something is up w/ our syncing.
1827      * @return The passed <code>syncFuture</code>
1828      * @throws FailedSyncBeforeLogCloseException 
1829      */
1830     SyncFuture waitSafePoint(final SyncFuture syncFuture)
1831     throws InterruptedException, FailedSyncBeforeLogCloseException {
1832       while (true) {
1833         if (this.safePointAttainedLatch.await(1, TimeUnit.NANOSECONDS)) break;
1834         if (syncFuture.isThrowable()) {
1835           throw new FailedSyncBeforeLogCloseException(syncFuture.getThrowable());
1836         }
1837       }
1838       return syncFuture;
1839     }
1840  
1841     /**
1842      * Called by Thread B when it attains the 'safe point'.  In this method, Thread B signals
1843      * Thread A it can proceed. Thread B will be held in here until {@link #releaseSafePoint()}
1844      * is called by Thread A.
1845      * @throws InterruptedException
1846      */
1847     void safePointAttained() throws InterruptedException {
1848       this.safePointAttainedLatch.countDown();
1849       this.safePointReleasedLatch.await();
1850     }
1851 
1852     /**
1853      * Called by Thread A when it is done with the work it needs to do while Thread B is
1854      * halted.  This will release the Thread B held in a call to {@link #safePointAttained()}
1855      */
1856     void releaseSafePoint() {
1857       this.safePointReleasedLatch.countDown();
1858     }
1859 
1860     /**
1861      * @return True is this is a 'cocked', fresh instance, and not one that has already fired.
1862      */
1863     boolean isCocked() {
1864       return this.safePointAttainedLatch.getCount() > 0 &&
1865         this.safePointReleasedLatch.getCount() > 0;
1866     }
1867   }

1869   /**
1870    * Handler that is run by the disruptor ringbuffer consumer. Consumer is a SINGLE
1871    * 'writer/appender' thread.  Appends edits and starts up sync runs.  Tries its best to batch up
1872    * syncs.  There is no discernible benefit batching appends so we just append as they come in
1873    * because it simplifies the below implementation.  See metrics for batching effectiveness
1874    * (In measurement, at 100 concurrent handlers writing 1k, we are batching > 10 appends and 10
1875    * handler sync invocations for every actual dfsclient sync call; at 10 concurrent handlers,
1876    * YMMV).
1877    * <p>Herein, we have an array into which we store the sync futures as they come in.  When we
1878    * have a 'batch', we'll then pass what we have collected to a SyncRunner thread to do the
1879    * filesystem sync.  When it completes, it will then call
1880    * {@link SyncFuture#done(long, Throwable)} on each of SyncFutures in the batch to release
1881    * blocked Handler threads.
1882    * <p>I've tried various effects to try and make latencies low while keeping throughput high.
1883    * I've tried keeping a single Queue of SyncFutures in this class appending to its tail as the
1884    * syncs coming and having sync runner threads poll off the head to 'finish' completed
1885    * SyncFutures.  I've tried linkedlist, and various from concurrent utils whether
1886    * LinkedBlockingQueue or ArrayBlockingQueue, etc.  The more points of synchronization, the
1887    * more 'work' (according to 'perf stats') that has to be done; small increases in stall
1888    * percentages seem to have a big impact on throughput/latencies.  The below model where we have
1889    * an array into which we stash the syncs and then hand them off to the sync thread seemed like
1890    * a decent compromise.  See HBASE-8755 for more detail.
1891    */
1892   class RingBufferEventHandler implements EventHandler<RingBufferTruck>, LifecycleAware {
1893     private final SyncRunner [] syncRunners;
1894     private final SyncFuture [] syncFutures;
1895     // Had 'interesting' issues when this was non-volatile.  On occasion, we'd not pass all
1896     // syncFutures to the next sync'ing thread.
1897     private volatile int syncFuturesCount = 0;
1898     private volatile SafePointZigZagLatch zigzagLatch;
1899     /**
1900      * Object to block on while waiting on safe point.
1901      */
1902     private final Object safePointWaiter = new Object();
1903     private volatile boolean shutdown = false;
1904 
1905     /**
1906      * Which syncrunner to use next.
1907      */
1908     private int syncRunnerIndex;

1910     RingBufferEventHandler(final int syncRunnerCount, final int maxHandlersCount) {
1911       this.syncFutures = new SyncFuture[maxHandlersCount];
1912       this.syncRunners = new SyncRunner[syncRunnerCount];
1913       for (int i = 0; i < syncRunnerCount; i++) {
1914         this.syncRunners[i] = new SyncRunner("sync." + i, maxHandlersCount);

1918     private void cleanupOutstandingSyncsOnException(final long sequence, final Exception e) {
1919       for (int i = 0; i < this.syncFuturesCount; i++) this.syncFutures[i].done(sequence, e);
1920       this.syncFuturesCount = 0;
1921     }

1923     @Override
1924     // We can set endOfBatch in the below method if at end of our this.syncFutures array
1925     public void onEvent(final RingBufferTruck truck, final long sequence, boolean endOfBatch)
1926     throws Exception {
1927       // Appends and syncs are coming in order off the ringbuffer.  We depend on this fact.  We'll
1928       // add appends to dfsclient as they come in.  Batching appends doesn't give any significant
1929       // benefit on measurement.  Handler sync calls we will batch up.
1930 
1931       try {
1932         if (truck.hasSyncFuturePayload()) {
1933           this.syncFutures[this.syncFuturesCount++] = truck.unloadSyncFuturePayload();
1934           // Force flush of syncs if we are carrying a full complement of syncFutures.
1935           if (this.syncFuturesCount == this.syncFutures.length) endOfBatch = true;
1936         } else if (truck.hasFSWALEntryPayload()) {
1937           TraceScope scope = Trace.continueSpan(truck.unloadSpanPayload());
1938           try {
1939             append(truck.unloadFSWALEntryPayload());
1940           } catch (Exception e) {
1941             // If append fails, presume any pending syncs will fail too; let all waiting handlers
1942             // know of the exception.
1943             cleanupOutstandingSyncsOnException(sequence, e);
1944             // Return to keep processing.
1945             return;
1946           } finally {
1947             assert scope == NullScope.INSTANCE || !scope.isDetached();
1948             scope.close(); // append scope is complete
1949           }
1950         } else {
1951           // They can't both be null.  Fail all up to this!!!
1952           cleanupOutstandingSyncsOnException(sequence,
1953             new IllegalStateException("Neither append nor sync"));
1954           // Return to keep processing.
1955           return;
1956         }
1957 
1958         // TODO: Check size and if big go ahead and call a sync if we have enough data.
1959 
1960         // If not a batch, return to consume more events from the ring buffer before proceeding;
1961         // we want to get up a batch of syncs and appends before we go do a filesystem sync.
1962         if (!endOfBatch || this.syncFuturesCount <= 0) return;
1963 
1964         // Now we have a batch.
1965 
1966         if (LOG.isTraceEnabled()) {
1967           LOG.trace("Sequence=" + sequence + ", syncCount=" + this.syncFuturesCount);
1968         }
1969 
1970         // Below expects that the offer 'transfers' responsibility for the outstanding syncs to the
1971         // syncRunner. We should never get an exception in here. HBASE-11145 was because queue
1972         // was sized exactly to the count of user handlers but we could have more if we factor in
1973         // meta handlers doing opens and closes.
1974         int index = Math.abs(this.syncRunnerIndex++) % this.syncRunners.length;
1975         try {
1976           this.syncRunners[index].offer(sequence, this.syncFutures, this.syncFuturesCount);
1977         } catch (Exception e) {
1978           cleanupOutstandingSyncsOnException(sequence, e);
1979           throw e;
1980         }
1981         attainSafePoint(sequence);
1982         this.syncFuturesCount = 0;
1983       } catch (Throwable t) {
1984         LOG.error("UNEXPECTED!!! syncFutures.length=" + this.syncFutures.length, t);
1985       }
1986     }

1993     /**
1994      * Check if we should attain safe point.  If so, go there and then wait till signalled before
1995      * we proceeding.
1996      */
1997     private void attainSafePoint(final long currentSequence) {
1998       if (this.zigzagLatch == null || !this.zigzagLatch.isCocked()) return;
1999       // If here, another thread is waiting on us to get to safe point.  Don't leave it hanging.
2000       try {
2001         // Wait on outstanding syncers; wait for them to finish syncing (unless we've been
2002         // shutdown or unless our latch has been thrown because we have been aborted).
2003         while (!this.shutdown && this.zigzagLatch.isCocked() &&
2004             highestSyncedSequence.get() < currentSequence) {
2005           synchronized (this.safePointWaiter) {
2006             this.safePointWaiter.wait(0, 1);
2007           }
2008         }
2009         // Tell waiting thread we've attained safe point
2010         this.zigzagLatch.safePointAttained();
2011       } catch (InterruptedException e) {
2012         LOG.warn("Interrupted ", e);
2013         Thread.currentThread().interrupt();
2014       }
2015     }
