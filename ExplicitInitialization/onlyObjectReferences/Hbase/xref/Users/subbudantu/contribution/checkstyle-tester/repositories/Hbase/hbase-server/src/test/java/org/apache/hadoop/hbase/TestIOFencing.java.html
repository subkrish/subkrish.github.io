59  /**
60   * Test for the case where a regionserver going down has enough cycles to do damage to regions
61   * that have actually been assigned elsehwere.
62   *
63   * <p>If we happen to assign a region before it fully done with in its old location -- i.e. it is on two servers at the
64   * same time -- all can work fine until the case where the region on the dying server decides to compact or otherwise
65   * change the region file set.  The region in its new location will then get a surprise when it tries to do something
66   * w/ a file removed by the region in its old location on dying server.
67   *
68   * <p>Making a test for this case is a little tough in that even if a file is deleted up on the namenode,
69   * if the file was opened before the delete, it will continue to let reads happen until something changes the
70   * state of cached blocks in the dfsclient that was already open (a block from the deleted file is cleaned
71   * from the datanode by NN).
72   *
73   * <p>What we will do below is do an explicit check for existence on the files listed in the region that
74   * has had some files removed because of a compaction.  This sort of hurry's along and makes certain what is a chance
75   * occurance.
76   */

91    public abstract static class CompactionBlockerRegion extends HRegion {
92      volatile int compactCount = 0;
93      volatile CountDownLatch compactionsBlocked = new CountDownLatch(0);
94      volatile CountDownLatch compactionsWaiting = new CountDownLatch(0);
95  
96      @SuppressWarnings("deprecation")
97      public CompactionBlockerRegion(Path tableDir, WAL log,
98          FileSystem fs, Configuration confParam, HRegionInfo info,
99          HTableDescriptor htd, RegionServerServices rsServices) {
100       super(tableDir, log, fs, confParam, info, htd, rsServices);
101     }
102 
103     public void stopCompactions() {
104       compactionsBlocked = new CountDownLatch(1);
105       compactionsWaiting = new CountDownLatch(1);
106     }
107 
108     public void allowCompactions() {
109       LOG.debug("allowing compactions");
110       compactionsBlocked.countDown();
111     }
112     public void waitForCompactionToBlock() throws IOException {
113       try {
114         LOG.debug("waiting for compaction to block");
115         compactionsWaiting.await();
116         LOG.debug("compaction block reached");
117       } catch (InterruptedException ex) {
118         throw new IOException(ex);
119       }
120     }
121 
122     @Override
123     public boolean compact(CompactionContext compaction, Store store,
124         CompactionThroughputController throughputController) throws IOException {
125       try {
126         return super.compact(compaction, store, throughputController);
127       } finally {
128         compactCount++;
129       }
130     }
131 
132     public int countStoreFiles() {
133       int count = 0;
134       for (Store store : stores.values()) {
135         count += store.getStorefilesCount();
136       }
137       return count;
138     }
139   }

145   public static class BlockCompactionsInPrepRegion extends CompactionBlockerRegion {

147     public BlockCompactionsInPrepRegion(Path tableDir, WAL log,
148         FileSystem fs, Configuration confParam, HRegionInfo info,
149         HTableDescriptor htd, RegionServerServices rsServices) {
150       super(tableDir, log, fs, confParam, info, htd, rsServices);
151     }
152     @Override
153     protected void doRegionCompactionPrep() throws IOException {
154       compactionsWaiting.countDown();
155       try {
156         compactionsBlocked.await();
157       } catch (InterruptedException ex) {
158         throw new IOException();
159       }
160       super.doRegionCompactionPrep();
161     }

181   public static class BlockCompactionsInCompletionHStore extends HStore {
182     CompactionBlockerRegion r;
183     protected BlockCompactionsInCompletionHStore(HRegion region, HColumnDescriptor family,
184         Configuration confParam) throws IOException {
185       super(region, family, confParam);
186       r = (CompactionBlockerRegion) region;
187     }
188 
189     @Override
190     protected void completeCompaction(final Collection<StoreFile> compactedFiles,
191         boolean removeFiles) throws IOException {
192       try {
193         r.compactionsWaiting.countDown();
194         r.compactionsBlocked.await();
195       } catch (InterruptedException ex) {
196         throw new IOException(ex);
197       }
198       super.completeCompaction(compactedFiles, removeFiles);
199     }
200     @Override
201     protected void completeCompaction(Collection<StoreFile> compactedFiles) throws IOException {
202       try {
203         r.compactionsWaiting.countDown();
204         r.compactionsBlocked.await();
205       } catch (InterruptedException ex) {
206         throw new IOException(ex);
207       }
208       super.completeCompaction(compactedFiles);
209     }

213   private final static TableName TABLE_NAME =

216   private static final int FIRST_BATCH_COUNT = 4000;
217   private static final int SECOND_BATCH_COUNT = FIRST_BATCH_COUNT;

243   public void doTest(Class<?> regionClass, boolean distributedLogReplay) throws Exception {
244     Configuration c = TEST_UTIL.getConfiguration();
245     c.setBoolean(HConstants.DISTRIBUTED_LOG_REPLAY_KEY, distributedLogReplay);
246     // Insert our custom region
247     c.setClass(HConstants.REGION_IMPL, regionClass, HRegion.class);
248     c.setBoolean("dfs.support.append", true);
249     // Encourage plenty of flushes
250     c.setLong("hbase.hregion.memstore.flush.size", 200000);
251     c.set(HConstants.HBASE_REGION_SPLIT_POLICY_KEY, ConstantSizeRegionSplitPolicy.class.getName());
252     // Only run compaction when we tell it to
253     c.setInt("hbase.hstore.compactionThreshold", 1000);
254     c.setLong("hbase.hstore.blockingStoreFiles", 1000);
255     // Compact quickly after we tell it to!
256     c.setInt("hbase.regionserver.thread.splitcompactcheckfrequency", 1000);
257     LOG.info("Starting mini cluster");
258     TEST_UTIL.startMiniCluster(1);
259     CompactionBlockerRegion compactingRegion = null;
260     Admin admin = null;
261     try {
262       LOG.info("Creating admin");
263       admin = TEST_UTIL.getConnection().getAdmin();
264       LOG.info("Creating table");
265       TEST_UTIL.createTable(TABLE_NAME, FAMILY);
266       Table table = TEST_UTIL.getConnection().getTable(TABLE_NAME);
267       LOG.info("Loading test table");
268       // Find the region
269       List<HRegion> testRegions = TEST_UTIL.getMiniHBaseCluster().findRegionsForTable(TABLE_NAME);
270       assertEquals(1, testRegions.size());
271       compactingRegion = (CompactionBlockerRegion)testRegions.get(0);
272       LOG.info("Blocking compactions");
273       compactingRegion.stopCompactions();
274       long lastFlushTime = compactingRegion.getEarliestFlushTimeForAllStores();
275       // Load some rows
276       TEST_UTIL.loadNumericRows(table, FAMILY, 0, FIRST_BATCH_COUNT);
277 
278       // add a compaction from an older (non-existing) region to see whether we successfully skip
279       // those entries
280       HRegionInfo oldHri = new HRegionInfo(table.getName(),
281         HConstants.EMPTY_START_ROW, HConstants.EMPTY_END_ROW);
282       CompactionDescriptor compactionDescriptor = ProtobufUtil.toCompactionDescriptor(oldHri,
283         FAMILY, Lists.newArrayList(new Path("/a")), Lists.newArrayList(new Path("/b")),
284         new Path("store_dir"));
285       WALUtil.writeCompactionMarker(compactingRegion.getWAL(), table.getTableDescriptor(),
286         oldHri, compactionDescriptor, new AtomicLong(Long.MAX_VALUE-100));
287 
288       // Wait till flush has happened, otherwise there won't be multiple store files
289       long startWaitTime = System.currentTimeMillis();
290       while (compactingRegion.getEarliestFlushTimeForAllStores() <= lastFlushTime ||
291           compactingRegion.countStoreFiles() <= 1) {
292         LOG.info("Waiting for the region to flush " +
293           compactingRegion.getRegionInfo().getRegionNameAsString());
294         Thread.sleep(1000);
295         assertTrue("Timed out waiting for the region to flush",
296           System.currentTimeMillis() - startWaitTime < 30000);
297       }
298       assertTrue(compactingRegion.countStoreFiles() > 1);
299       final byte REGION_NAME[] = compactingRegion.getRegionInfo().getRegionName();
300       LOG.info("Asking for compaction");
301       ((HBaseAdmin)admin).majorCompact(TABLE_NAME.getName());
302       LOG.info("Waiting for compaction to be about to start");
303       compactingRegion.waitForCompactionToBlock();
304       LOG.info("Starting a new server");
305       RegionServerThread newServerThread = TEST_UTIL.getMiniHBaseCluster().startRegionServer();
306       final HRegionServer newServer = newServerThread.getRegionServer();
307       LOG.info("Killing region server ZK lease");
308       TEST_UTIL.expireRegionServerSession(0);
309       CompactionBlockerRegion newRegion = null;
310       startWaitTime = System.currentTimeMillis();
311       LOG.info("Waiting for the new server to pick up the region " + Bytes.toString(REGION_NAME));
312 
313       // wait for region to be assigned and to go out of log replay if applicable
314       Waiter.waitFor(c, 60000, new Waiter.Predicate<Exception>() {
315         @Override
316         public boolean evaluate() throws Exception {
317           Region newRegion = newServer.getOnlineRegion(REGION_NAME);
318           return newRegion != null && !newRegion.isRecovering();
319         }
320       });
321 
322       newRegion = (CompactionBlockerRegion)newServer.getOnlineRegion(REGION_NAME);
323 
324       LOG.info("Allowing compaction to proceed");
325       compactingRegion.allowCompactions();
326       while (compactingRegion.compactCount == 0) {
327         Thread.sleep(1000);
328       }
329       // The server we killed stays up until the compaction that was started before it was killed completes.  In logs
330       // you should see the old regionserver now going down.
331       LOG.info("Compaction finished");
332       // After compaction of old region finishes on the server that was going down, make sure that
333       // all the files we expect are still working when region is up in new location.
334       FileSystem fs = newRegion.getFilesystem();
335       for (String f: newRegion.getStoreFileList(new byte [][] {FAMILY})) {
336         assertTrue("After compaction, does not exist: " + f, fs.exists(new Path(f)));
337       }
338       // If we survive the split keep going...
339       // Now we make sure that the region isn't totally confused.  Load up more rows.
340       TEST_UTIL.loadNumericRows(table, FAMILY, FIRST_BATCH_COUNT, FIRST_BATCH_COUNT + SECOND_BATCH_COUNT);
341       ((HBaseAdmin)admin).majorCompact(TABLE_NAME.getName());
342       startWaitTime = System.currentTimeMillis();
343       while (newRegion.compactCount == 0) {
344         Thread.sleep(1000);
345         assertTrue("New region never compacted", System.currentTimeMillis() - startWaitTime < 180000);
346       }
347       assertEquals(FIRST_BATCH_COUNT + SECOND_BATCH_COUNT, TEST_UTIL.countRows(table));
348     } finally {
349       if (compactingRegion != null) {
350         compactingRegion.allowCompactions();
351       }
352       admin.close();
353       TEST_UTIL.shutdownMiniCluster();
354     }
355   }
