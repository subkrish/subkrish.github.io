1529   @Test (timeout=180000)
1530   public void testRegionTransitionOperations() throws Exception {
1531     final TableName tableName = TableName.valueOf(name.getMethodName());
1532     MiniHBaseCluster cluster = UTIL.getHBaseCluster();
1533 
1534     HMaster master = cluster.getMaster();
1535     MasterCoprocessorHost host = master.getMasterCoprocessorHost();
1536     CPMasterObserver cp = (CPMasterObserver)host.findCoprocessor(
1537         CPMasterObserver.class.getName());
1538     cp.enableBypass(false);
1539     cp.resetStates();
1540 
1541     HTable table = UTIL.createMultiRegionTable(tableName, TEST_FAMILY);
1542 
1543     try {
1544       UTIL.waitUntilAllRegionsAssigned(tableName);
1545 
1546       NavigableMap<HRegionInfo, ServerName> regions = table.getRegionLocations();
1547       Map.Entry<HRegionInfo, ServerName> firstGoodPair = null;
1548       for (Map.Entry<HRegionInfo, ServerName> e: regions.entrySet()) {
1549         if (e.getValue() != null) {
1550           firstGoodPair = e;
1551           break;
1552         }
1553       }
1554       assertNotNull("Found a non-null entry", firstGoodPair);
1555       LOG.info("Found " + firstGoodPair.toString());
1556       // Try to force a move
1557       Collection<ServerName> servers = master.getClusterStatus().getServers();
1558       String destName = null;
1559       String serverNameForFirstRegion = firstGoodPair.getValue().toString();
1560       LOG.info("serverNameForFirstRegion=" + serverNameForFirstRegion);
1561       ServerName masterServerName = master.getServerName();
1562       boolean found = false;
1563       // Find server that is NOT carrying the first region
1564       for (ServerName info : servers) {
1565         LOG.info("ServerName=" + info);
1566         if (!serverNameForFirstRegion.equals(info.getServerName())
1567             && !masterServerName.equals(info)) {
1568           destName = info.toString();
1569           found = true;
1570           break;
1571         }
1572       }
1573       assertTrue("Found server", found);
1574       LOG.info("Found " + destName);
1575       master.getMasterRpcServices().moveRegion(null, RequestConverter.buildMoveRegionRequest(
1576           firstGoodPair.getKey().getEncodedNameAsBytes(),Bytes.toBytes(destName)));
1577       assertTrue("Coprocessor should have been called on region move",
1578         cp.wasMoveCalled());
1579 
1580       // make sure balancer is on
1581       master.balanceSwitch(true);
1582       assertTrue("Coprocessor should have been called on balance switch",
1583           cp.wasBalanceSwitchCalled());
1584 
1585       // turn balancer off
1586       master.balanceSwitch(false);
1587 
1588       // wait for assignments to finish, if any
1589       AssignmentManager mgr = master.getAssignmentManager();
1590       Collection<RegionState> transRegions =
1591         mgr.getRegionStates().getRegionsInTransition().values();
1592       for (RegionState state : transRegions) {
1593         mgr.getRegionStates().waitOnRegionToClearRegionsInTransition(state.getRegion());
1594       }
1595 
1596       // move half the open regions from RS 0 to RS 1
1597       HRegionServer rs = cluster.getRegionServer(0);
1598       byte[] destRS = Bytes.toBytes(cluster.getRegionServer(1).getServerName().toString());
1599       //Make sure no regions are in transition now
1600       waitForRITtoBeZero(master);
1601       List<HRegionInfo> openRegions = ProtobufUtil.getOnlineRegions(rs.getRSRpcServices());
1602       int moveCnt = openRegions.size()/2;
1603       for (int i=0; i<moveCnt; i++) {
1604         HRegionInfo info = openRegions.get(i);
1605         if (!info.isMetaTable()) {
1606           master.getMasterRpcServices().moveRegion(null, RequestConverter.buildMoveRegionRequest(
1607               openRegions.get(i).getEncodedNameAsBytes(), destRS));
1608         }
1609       }
1610       //Make sure no regions are in transition now
1611       waitForRITtoBeZero(master);
1612       // now trigger a balance
1613       master.balanceSwitch(true);
1614       boolean balanceRun = master.balance();
1615       assertTrue("Coprocessor should be called on region rebalancing",
1616           cp.wasBalanceCalled());
1617     } finally {
1618       Admin admin = UTIL.getHBaseAdmin();
1619       admin.disableTable(tableName);
1620       deleteTable(admin, tableName);
1621     }
1622   }
