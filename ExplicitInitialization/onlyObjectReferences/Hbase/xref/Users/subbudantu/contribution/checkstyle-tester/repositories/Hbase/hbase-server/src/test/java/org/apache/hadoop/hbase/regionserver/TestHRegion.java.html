317   /**
318    * Test we do not lose data if we fail a flush and then close.
319    * Part of HBase-10466.  Tests the following from the issue description:
320    * "Bug 1: Wrong calculation of HRegion.memstoreSize: When a flush fails, data to be flushed is
321    * kept in each MemStore's snapshot and wait for next flush attempt to continue on it. But when
322    * the next flush succeeds, the counter of total memstore size in HRegion is always deduced by
323    * the sum of current memstore sizes instead of snapshots left from previous failed flush. This
324    * calculation is problematic that almost every time there is failed flush, HRegion.memstoreSize
325    * gets reduced by a wrong value. If region flush could not proceed for a couple cycles, the size
326    * in current memstore could be much larger than the snapshot. It's likely to drift memstoreSize
327    * much smaller than expected. In extreme case, if the error accumulates to even bigger than
328    * HRegion's memstore size limit, any further flush is skipped because flush does not do anything
329    * if memstoreSize is not larger than 0."
330    * @throws Exception
331    */
332   @Test (timeout=60000)
333   public void testFlushSizeAccounting() throws Exception {
334     final Configuration conf = HBaseConfiguration.create(CONF);
335     // Only retry once.
336     conf.setInt("hbase.hstore.flush.retries.number", 1);
337     final User user =
338       User.createUserForTesting(conf, this.name.getMethodName(), new String[]{"foo"});
339     // Inject our faulty LocalFileSystem
340     conf.setClass("fs.file.impl", FaultyFileSystem.class, FileSystem.class);
341     user.runAs(new PrivilegedExceptionAction<Object>() {
342       @Override
343       public Object run() throws Exception {
344         // Make sure it worked (above is sensitive to caching details in hadoop core)
345         FileSystem fs = FileSystem.get(conf);
346         Assert.assertEquals(FaultyFileSystem.class, fs.getClass());
347         FaultyFileSystem ffs = (FaultyFileSystem)fs;
348         HRegion region = null;
349         try {
350           // Initialize region
351           region = initHRegion(tableName, name.getMethodName(), conf, COLUMN_FAMILY_BYTES);
352           long size = region.getMemstoreSize();
353           Assert.assertEquals(0, size);
354           // Put one item into memstore.  Measure the size of one item in memstore.
355           Put p1 = new Put(row);
356           p1.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual1, 1, (byte[])null));
357           region.put(p1);
358           final long sizeOfOnePut = region.getMemstoreSize();
359           // Fail a flush which means the current memstore will hang out as memstore 'snapshot'.
360           try {
361             LOG.info("Flushing");
362             region.flush(true);
363             Assert.fail("Didn't bubble up IOE!");
364           } catch (DroppedSnapshotException dse) {
365             // What we are expecting
366           }
367           // Make it so all writes succeed from here on out
368           ffs.fault.set(false);
369           // Check sizes.  Should still be the one entry.
370           Assert.assertEquals(sizeOfOnePut, region.getMemstoreSize());
371           // Now add two entries so that on this next flush that fails, we can see if we
372           // subtract the right amount, the snapshot size only.
373           Put p2 = new Put(row);
374           p2.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual2, 2, (byte[])null));
375           p2.add(new KeyValue(row, COLUMN_FAMILY_BYTES, qual3, 3, (byte[])null));
376           region.put(p2);
377           Assert.assertEquals(sizeOfOnePut * 3, region.getMemstoreSize());
378           // Do a successful flush.  It will clear the snapshot only.  Thats how flushes work.
379           // If already a snapshot, we clear it else we move the memstore to be snapshot and flush
380           // it
381           region.flush(true);
382           // Make sure our memory accounting is right.
383           Assert.assertEquals(sizeOfOnePut * 2, region.getMemstoreSize());
384         } finally {
385           HRegion.closeHRegion(region);
386         }
387         return null;
388       }
389     });
390     FileSystem.closeAllForUGI(user.getUGI());
391   }

2458   /**
2459    * @param parent
2460    *          Region to split.
2461    * @param midkey
2462    *          Key to split around.
2463    * @return The Regions we created.
2464    * @throws IOException
2465    */
2466   HRegion[] splitRegion(final HRegion parent, final byte[] midkey) throws IOException {
2467     PairOfSameType<Region> result = null;
2468     SplitTransactionImpl st = new SplitTransactionImpl(parent, midkey);
2469     // If prepare does not return true, for some reason -- logged inside in
2470     // the prepare call -- we are not ready to split just now. Just return.
2471     if (!st.prepare()) {
2472       parent.clearSplit();
2473       return null;
2474     }
2475     try {
2476       result = st.execute(null, null);
2477     } catch (IOException ioe) {
2478       try {
2479         LOG.info("Running rollback of failed split of " +
2480           parent.getRegionInfo().getRegionNameAsString() + "; " + ioe.getMessage());
2481         st.rollback(null, null);
2482         LOG.info("Successful rollback of failed split of " +
2483           parent.getRegionInfo().getRegionNameAsString());
2484         return null;
2485       } catch (RuntimeException e) {
2486         // If failed rollback, kill this server to avoid having a hole in table.
2487         LOG.info("Failed rollback of failed split of " +
2488           parent.getRegionInfo().getRegionNameAsString() + " -- aborting server", e);
2489       }
2490     }
2491     finally {
2492       parent.clearSplit();
2493     }
2494     return new HRegion[] { (HRegion)result.getFirst(), (HRegion)result.getSecond() };
2495   }
