124     public TokenServer(Configuration conf) throws IOException {
125       this.conf = conf;
126       this.startcode = EnvironmentEdgeManager.currentTime();
127       // Server to handle client requests.
128       String hostname =
129         Strings.domainNamePointerToHostName(DNS.getDefaultHost("default", "default"));
130       int port = 0;
131       // Creation of an ISA will force a resolve.
132       InetSocketAddress initialIsa = new InetSocketAddress(hostname, port);
133       if (initialIsa.getAddress() == null) {
134         throw new IllegalArgumentException("Failed resolve of " + initialIsa);
135       }
136       final List<BlockingServiceAndInterface> sai =
137         new ArrayList<BlockingServiceAndInterface>(1);
138       BlockingService service =
139         AuthenticationProtos.AuthenticationService.newReflectiveBlockingService(this);
140       sai.add(new BlockingServiceAndInterface(service,
141         AuthenticationProtos.AuthenticationService.BlockingInterface.class));
142       this.rpcServer =
143         new RpcServer(this, "tokenServer", sai, initialIsa, conf, new FifoRpcScheduler(conf, 1));
144       this.isa = this.rpcServer.getListenerAddress();
145       this.sleeper = new Sleeper(1000, this);
146     }

341   @BeforeClass
342   public static void setupBeforeClass() throws Exception {
343     TEST_UTIL = new HBaseTestingUtility();
344     TEST_UTIL.startMiniZKCluster();
345     // register token type for protocol
346     SecurityInfo.addInfo(AuthenticationProtos.AuthenticationService.getDescriptor().getName(),
347       new SecurityInfo("hbase.test.kerberos.principal",
348         AuthenticationProtos.TokenIdentifier.Kind.HBASE_AUTH_TOKEN));
349     // security settings only added after startup so that ZK does not require SASL
350     Configuration conf = TEST_UTIL.getConfiguration();
351     conf.set("hadoop.security.authentication", "kerberos");
352     conf.set("hbase.security.authentication", "kerberos");
353     conf.setBoolean(HADOOP_SECURITY_AUTHORIZATION, true);
354     server = new TokenServer(conf);
355     serverThread = new Thread(server);
356     Threads.setDaemonThreadRunning(serverThread, "TokenServer:"+server.getServerName().toString());
357     // wait for startup
358     while (!server.isStarted() && !server.isStopped()) {
359       Thread.sleep(10);
360     }
361     server.rpcServer.refreshAuthManager(new PolicyProvider() {
362       @Override
363       public Service[] getServices() {
364         return new Service [] {
365           new Service("security.client.protocol.acl",
366             AuthenticationProtos.AuthenticationService.BlockingInterface.class)};
367       }
368     });
369     ZKClusterId.setClusterId(server.getZooKeeper(), clusterId);
370     secretManager = (AuthenticationTokenSecretManager)server.getSecretManager();
371     while(secretManager.getCurrentKey() == null) {
372       Thread.sleep(1);
373     }
374   }

383   @Test
384   public void testTokenCreation() throws Exception {
385     Token<AuthenticationTokenIdentifier> token =
386         secretManager.generateToken("testuser");
387 
388     AuthenticationTokenIdentifier ident = new AuthenticationTokenIdentifier();
389     Writables.getWritable(token.getIdentifier(), ident);
390     assertEquals("Token username should match", "testuser",
391         ident.getUsername());
392     byte[] passwd = secretManager.retrievePassword(ident);
393     assertTrue("Token password and password from secret manager should match",
394         Bytes.equals(token.getPassword(), passwd));
395   }

397   @Test
398   public void testTokenAuthentication() throws Exception {
399     UserGroupInformation testuser =
400         UserGroupInformation.createUserForTesting("testuser", new String[]{"testgroup"});
401 
402     testuser.setAuthenticationMethod(
403         UserGroupInformation.AuthenticationMethod.TOKEN);
404     final Configuration conf = TEST_UTIL.getConfiguration();
405     UserGroupInformation.setConfiguration(conf);
406     Token<AuthenticationTokenIdentifier> token =
407         secretManager.generateToken("testuser");
408     LOG.debug("Got token: " + token.toString());
409     testuser.addToken(token);
410 
411     // verify the server authenticates us as this token user
412     testuser.doAs(new PrivilegedExceptionAction<Object>() {
413       public Object run() throws Exception {
414         Configuration c = server.getConfiguration();
415         RpcClient rpcClient = RpcClientFactory.createClient(c, clusterId.toString());
416         ServerName sn =
417             ServerName.valueOf(server.getAddress().getHostName(), server.getAddress().getPort(),
418                 System.currentTimeMillis());
419         try {
420           BlockingRpcChannel channel = rpcClient.createBlockingRpcChannel(sn,
421               User.getCurrent(), HConstants.DEFAULT_HBASE_RPC_TIMEOUT);
422           AuthenticationProtos.AuthenticationService.BlockingInterface stub =
423               AuthenticationProtos.AuthenticationService.newBlockingStub(channel);
424           AuthenticationProtos.WhoAmIResponse response =
425               stub.whoAmI(null, AuthenticationProtos.WhoAmIRequest.getDefaultInstance());
426           String myname = response.getUsername();
427           assertEquals("testuser", myname);
428           String authMethod = response.getAuthMethod();
429           assertEquals("TOKEN", authMethod);
430         } finally {
431           rpcClient.close();
432         }
433         return null;
434       }
435     });
436   }

438   @Test
439   public void testUseExistingToken() throws Exception {
440     User user = User.createUserForTesting(TEST_UTIL.getConfiguration(), "testuser2",
441         new String[]{"testgroup"});
442     Token<AuthenticationTokenIdentifier> token =
443         secretManager.generateToken(user.getName());
444     assertNotNull(token);
445     user.addToken(token);
446 
447     // make sure we got a token
448     Token<AuthenticationTokenIdentifier> firstToken =
449         new AuthenticationTokenSelector().selectToken(token.getService(), user.getTokens());
450     assertNotNull(firstToken);
451     assertEquals(token, firstToken);
452 
453     Connection conn = ConnectionFactory.createConnection(TEST_UTIL.getConfiguration());
454     try {
455       assertFalse(TokenUtil.addTokenIfMissing(conn, user));
456       // make sure we still have the same token
457       Token<AuthenticationTokenIdentifier> secondToken =
458           new AuthenticationTokenSelector().selectToken(token.getService(), user.getTokens());
459       assertEquals(firstToken, secondToken);
460     } finally {
461       conn.close();
462     }
463   }
