106     /** Add the default loaders for UTC-TAI offsets history files (both IERS and USNO).
107      * <p>
108      * The default loaders are {@link TAIUTCDatFilesLoader} that looks for
109      * a file named {@code tai-utc.dat} that must be in USNO format and
110      * {@link UTCTAIHistoryFilesLoader} that looks fir a file named
111      * {@code UTC-TAI.history} that must be in the IERS format. The
112      * {@link UTCTAIBulletinAFilesLoader} is <em>not</em> added by default
113      * as it is not recommended. USNO warned us that the TAI-UTC data present
114      * in bulletin A was for convenience only and was not reliable, there
115      * have been errors in several bulletins regarding these data.
116      * </p>
117      * @see <a href="http://maia.usno.navy.mil/ser7/tai-utc.dat">USNO tai-utc.dat file</a>
118      * @see <a href="http://hpiers.obspm.fr/eoppc/bul/bulc/UTC-TAI.history">IERS UTC-TAI.history file</a>
119      * @see TAIUTCDatFilesLoader
120      * @see UTCTAIHistoryFilesLoader
121      * @see #getUTC()
122      * @see #clearUTCTAIOffsetsLoaders()
123      * @since 7.1
124      */
125     public static void addDefaultUTCTAIOffsetsLoaders() {
126         addUTCTAIOffsetsLoader(new TAIUTCDatFilesLoader(TAIUTCDatFilesLoader.DEFAULT_SUPPORTED_NAMES));
127         addUTCTAIOffsetsLoader(new UTCTAIHistoryFilesLoader());
128     }

134      * @since 7.1

136     public static void clearUTCTAIOffsetsLoaders() {

155     /** Get the Universal Time Coordinate scale.
156      * <p>
157      * If no {@link UTCTAIOffsetsLoader} has been added by calling {@link
158      * #addUTCTAIOffsetsLoader(UTCTAIOffsetsLoader) addUTCTAIOffsetsLoader} or if {@link
159      * #clearUTCTAIOffsetsLoaders() clearUTCTAIOffsetsLoaders} has been called afterwards,
160      * the {@link #addDefaultUTCTAIOffsetsLoaders() addDefaultUTCTAILoaders} method
161      * will be called automatically.
162      * </p>
163      * @return Universal Time Coordinate scale
164      * @exception OrekitException if some data can't be read or some
165      * file content is corrupted
166      * @see #addDefaultUTCTAIOffsetsLoaders()
167      */
168     public static UTCScale getUTC() throws OrekitException {
169         synchronized (TimeScalesFactory.class) {
170 
171             if (utc == null) {
172                 List<OffsetModel> entries = null;
173                 if (loaders.isEmpty()) {
174                     addDefaultUTCTAIOffsetsLoaders();
175                 }
176                 for (UTCTAIOffsetsLoader loader : loaders) {
177                     entries = loader.loadOffsets();
178                     if (!entries.isEmpty()) {
179                         break;
180                     }
181                 }
182                 if (entries.isEmpty()) {
183                     throw new OrekitException(OrekitMessages.NO_IERS_UTC_TAI_HISTORY_DATA_LOADED);
184                 }
185                 utc = new UTCScale(entries);
186             }
187 
188             return utc;
189         }
190     }

192     /** Get the Universal Time 1 scale.
193      * <p>
194      * UT1 scale depends on both UTC scale and Earth Orientation Parameters,
195      * so this method loads these data sets. See the {@link #getUTC()
196      * TimeScalesFactory.getUTC()} and {@link
197      * FramesFactory#getEOPHistory(IERSConventions, boolean)} methods
198      * for an explanation of how the corresponding data loaders can be configured.
199      * </p>
200      * @param conventions IERS conventions for which EOP parameters will provide dUT1
201      * @param simpleEOP if true, tidal effects are ignored when interpolating EOP
202      * @return Universal Time 1 scale
203      * @exception OrekitException if some data can't be read or some
204      * file content is corrupted
205      * @see #getUTC()
206      * @see FramesFactory#getEOPHistory(IERSConventions, boolean)
207      */
208     public static UT1Scale getUT1(final IERSConventions conventions, final boolean simpleEOP)
209         throws OrekitException {
210         synchronized (TimeScalesFactory.class) {
211 
212             final Map<IERSConventions, UT1Scale> map =
213                     simpleEOP ? ut1MapSimpleEOP : ut1MapCompleteEOP;
214             UT1Scale ut1 = map.get(conventions);
215             if (ut1 == null) {
216                 ut1 = getUT1(FramesFactory.getEOPHistory(conventions, simpleEOP));
217                 map.put(conventions, ut1);
218             }
219             return ut1;
220         }
221     }
222 
223     /** Get the Universal Time 1 scale.
224      * <p>
225      * As this method allow associating any history with the time scale,
226      * it may involve large data sets. So this method does <em>not</em>
227      * cache the resulting {@link UT1Scale UT1Scale} instance, a new
228      * instance will be returned each time. In order to avoid wasting
229      * memory, calling {@link #getUT1(IERSConventions, boolean)}
230      * with the single enumerate corresponding to the conventions may be
231      * a better solution. This method is made available only for expert use.
232      * </p>
233      * @param history EOP parameters providing dUT1
234      * (may be null if no correction is desired)
235      * @return Universal Time 1 scale
236      * @exception OrekitException if some data can't be read or some
237      * file content is corrupted
238      * @see #getUT1(IERSConventions, boolean)
239      */
240     public static UT1Scale getUT1(final EOPHistory history) throws OrekitException {
241         return new UT1Scale(history, getUTC());
242     }
