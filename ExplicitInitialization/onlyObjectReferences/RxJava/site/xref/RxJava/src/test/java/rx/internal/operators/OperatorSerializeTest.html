149      * A thread that will pass data to onNext

151     public static class OnNextThread implements Runnable {
152 
153         private final Observer<String> observer;
154         private final int numStringsToSend;
155 
156         OnNextThread(Observer<String> observer, int numStringsToSend) {
157             this.observer = observer;
158             this.numStringsToSend = numStringsToSend;
159         }
160 
161         @Override
162         public void run() {
163             for (int i = 0; i < numStringsToSend; i++) {
164                 observer.onNext("aString");
165             }
166         }

169     /**
170      * A thread that will call onError or onNext
171      */
172     public static class CompletionThread implements Runnable {
173 
174         private final Observer<String> observer;
175         private final TestConcurrencyobserverEvent event;
176         private final Future<?>[] waitOnThese;
177 
178         CompletionThread(Observer<String> observer, TestConcurrencyobserverEvent event, Future<?>... waitOnThese) {
179             this.observer = observer;
180             this.event = event;
181             this.waitOnThese = waitOnThese;
182         }
183 
184         @Override
185         public void run() {
186             /* if we have 'waitOnThese' futures, we'll wait on them before proceeding */
187             if (waitOnThese != null) {
188                 for (Future<?> f : waitOnThese) {
189                     try {
190                         f.get();
191                     } catch (Throwable e) {
192                         System.err.println("Error while waiting on future in CompletionThread");
193                     }
194                 }
195             }
196 
197             /* send the event */
198             if (event == TestConcurrencyobserverEvent.onError) {
199                 observer.onError(new RuntimeException("mocked exception"));
200             } else if (event == TestConcurrencyobserverEvent.onCompleted) {
201                 observer.onCompleted();
202 
203             } else {
204                 throw new IllegalArgumentException("Expecting either onError or onCompleted");
205             }
206         }
207     }

209     private static enum TestConcurrencyobserverEvent {
210         onCompleted, onError, onNext
211     }

261     /**
262      * This spawns a thread for the subscription, then a separate thread for each onNext call.
263      */
264     private static class TestMultiThreadedObservable implements Observable.OnSubscribe<String> {
265         final String[] values;
266         Thread t = null;
267         AtomicInteger threadsRunning = new AtomicInteger();
268         AtomicInteger maxConcurrentThreads = new AtomicInteger();
269         ExecutorService threadPool;
270 
271         public TestMultiThreadedObservable(String... values) {
272             this.values = values;
273             this.threadPool = Executors.newCachedThreadPool();
274         }
275 
276         @Override
277         public void call(final Subscriber<? super String> observer) {
278             System.out.println("TestMultiThreadedObservable subscribed to ...");
279             t = new Thread(new Runnable() {
280 
281                 @Override
282                 public void run() {
283                     try {
284                         System.out.println("running TestMultiThreadedObservable thread");
285                         for (final String s : values) {
286                             threadPool.execute(new Runnable() {
287 
288                                 @Override
289                                 public void run() {
290                                     threadsRunning.incrementAndGet();
291                                     try {
292                                         // perform onNext call
293                                         System.out.println("TestMultiThreadedObservable onNext: " + s);
294                                         if (s == null) {
295                                             // force an error
296                                             throw new NullPointerException();
297                                         }
298                                         observer.onNext(s);
299                                         // capture 'maxThreads'
300                                         int concurrentThreads = threadsRunning.get();
301                                         int maxThreads = maxConcurrentThreads.get();
302                                         if (concurrentThreads > maxThreads) {
303                                             maxConcurrentThreads.compareAndSet(maxThreads, concurrentThreads);
304                                         }
305                                     } catch (Throwable e) {
306                                         observer.onError(e);
307                                     } finally {
308                                         threadsRunning.decrementAndGet();
309                                     }
310                                 }
311                             });
312                         }
313                         // we are done spawning threads
314                         threadPool.shutdown();
315                     } catch (Throwable e) {
316                         throw new RuntimeException(e);
317                     }
318 
319                     // wait until all threads are done, then mark it as COMPLETED
320                     try {
321                         // wait for all the threads to finish
322                         threadPool.awaitTermination(2, TimeUnit.SECONDS);
323                     } catch (InterruptedException e) {
324                         throw new RuntimeException(e);
325                     }
326                     observer.onCompleted();
327                 }
328             });
329             System.out.println("starting TestMultiThreadedObservable thread");
330             t.start();
331             System.out.println("done starting TestMultiThreadedObservable thread");
332         }
333 
334         public void waitToFinish() {
335             try {
336                 t.join();
337             } catch (InterruptedException e) {
338                 throw new RuntimeException(e);
339             }
340         }
341     }

343     private static class BusyObserver extends Subscriber<String> {
344         volatile boolean onCompleted = false;
345         volatile boolean onError = false;
346         AtomicInteger onNextCount = new AtomicInteger();
347         AtomicInteger threadsRunning = new AtomicInteger();
348         AtomicInteger maxConcurrentThreads = new AtomicInteger();
349 
350         @Override
351         public void onCompleted() {
352             threadsRunning.incrementAndGet();
353 
354             System.out.println(">>> Busyobserver received onCompleted");
355             onCompleted = true;
356 
357             int concurrentThreads = threadsRunning.get();
358             int maxThreads = maxConcurrentThreads.get();
359             if (concurrentThreads > maxThreads) {
360                 maxConcurrentThreads.compareAndSet(maxThreads, concurrentThreads);
361             }
362             threadsRunning.decrementAndGet();
363         }
364 
365         @Override
366         public void onError(Throwable e) {
367             threadsRunning.incrementAndGet();
368 
369             System.out.println(">>> Busyobserver received onError: " + e.getMessage());
370             onError = true;
371 
372             int concurrentThreads = threadsRunning.get();
373             int maxThreads = maxConcurrentThreads.get();
374             if (concurrentThreads > maxThreads) {
375                 maxConcurrentThreads.compareAndSet(maxThreads, concurrentThreads);
376             }
377             threadsRunning.decrementAndGet();
378         }
379 
380         @Override
381         public void onNext(String args) {
382             threadsRunning.incrementAndGet();
383             try {
384                 onNextCount.incrementAndGet();
385                 System.out.println(">>> Busyobserver received onNext: " + args);
386                 try {
387                     // simulate doing something computational
388                     Thread.sleep(200);
389                 } catch (InterruptedException e) {
390                     e.printStackTrace();
391                 }
392             } finally {
393                 // capture 'maxThreads'
394                 int concurrentThreads = threadsRunning.get();
395                 int maxThreads = maxConcurrentThreads.get();
396                 if (concurrentThreads > maxThreads) {
397                     maxConcurrentThreads.compareAndSet(maxThreads, concurrentThreads);
398                 }
399                 threadsRunning.decrementAndGet();
400             }
401         }
