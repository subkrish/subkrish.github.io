32  /**
33   * Returns an Observable that combines the emissions of multiple source observables. Once each
34   * source Observable has emitted at least one item, combineLatest emits an item whenever any of
35   * the source Observables emits an item, by combining the latest emissions from each source
36   * Observable with a specified function.
37   * <p>
38   * <img width="640" src="https://github.com/ReactiveX/RxJava/wiki/images/rx-operators/combineLatest.png" alt="">
39   * 
40   * @param <T>
41   *            the common basetype of the source values
42   * @param <R>
43   *            the result type of the combinator function
44   */
45  public final class OnSubscribeCombineLatest<T, R> implements OnSubscribe<R> {
46      final List<? extends Observable<? extends T>> sources;
47      final FuncN<? extends R> combinator;
48  
49      public OnSubscribeCombineLatest(List<? extends Observable<? extends T>> sources, FuncN<? extends R> combinator) {
50          this.sources = sources;
51          this.combinator = combinator;
52          if (sources.size() > 128) {
53              // For design simplicity this is limited to 128. If more are really needed we'll need to adjust 
54              // the design of how RxRingBuffer is used in the implementation below.
55              throw new IllegalArgumentException("More than 128 sources to combineLatest is not supported.");
56          }
57      }
58  
59      @Override
60      public void call(final Subscriber<? super R> child) {
61          if (sources.isEmpty()) {
62              child.onCompleted();
63              return;
64          }
65          if (sources.size() == 1) {
66              child.setProducer(new SingleSourceProducer<T, R>(child, sources.get(0), combinator));
67          } else {
68              child.setProducer(new MultiSourceProducer<T, R>(child, sources, combinator));
69          }
70  
71      }
72  
73      /*
74       * benjchristensen => This implementation uses a buffer enqueue/drain pattern. It could be optimized to have a fast-path to
75       * skip the buffer and emit directly when no conflict, but that is quite complicated and I don't have the time to attempt it right now.
76       */
77      final static class MultiSourceProducer<T, R> implements Producer {
78          private final AtomicBoolean started = new AtomicBoolean();
79          private final AtomicLong requested = new AtomicLong();
80          private final List<? extends Observable<? extends T>> sources;
81          private final Subscriber<? super R> child;
82          private final FuncN<? extends R> combinator;
83          private final MultiSourceRequestableSubscriber<T, R>[] subscribers;
84  
85          /* following are guarded by WIP */
86          private final RxRingBuffer buffer = RxRingBuffer.getSpmcInstance();
87          private final Object[] collectedValues;
88          private final BitSet haveValues;
89          private volatile int haveValuesCount; // does this need to be volatile or is WIP sufficient?
90          private final BitSet completion;
91          private volatile int completionCount; // does this need to be volatile or is WIP sufficient?
92  
93          @SuppressWarnings("unused")
94          private volatile long counter;
95          @SuppressWarnings("rawtypes")
96          private static final AtomicLongFieldUpdater<MultiSourceProducer> WIP = AtomicLongFieldUpdater.newUpdater(MultiSourceProducer.class, "counter");
97  
98          @SuppressWarnings("unchecked")
99          public MultiSourceProducer(final Subscriber<? super R> child, final List<? extends Observable<? extends T>> sources, FuncN<? extends R> combinator) {
100             this.sources = sources;
101             this.child = child;
102             this.combinator = combinator;
103 
104             int n = sources.size();
105             this.subscribers = new MultiSourceRequestableSubscriber[n];
106             this.collectedValues = new Object[n];
107             this.haveValues = new BitSet(n);
108             this.completion = new BitSet(n);
109         }
110 
111         @Override
112         public void request(long n) {
113             BackpressureUtils.getAndAddRequest(requested, n);
114             if (!started.get() && started.compareAndSet(false, true)) {
115                 /*
116                  * NOTE: this logic will ONLY work if we don't have more sources than the size of the buffer.
117                  * 
118                  * We would likely need to make an RxRingBuffer that can be sized to [numSources * n] instead
119                  * of the current global default size it has.
120                  */
121                 int sizePerSubscriber = RxRingBuffer.SIZE / sources.size();
122                 int leftOver = RxRingBuffer.SIZE % sources.size();
123                 for (int i = 0; i < sources.size(); i++) {
124                     Observable<? extends T> o = sources.get(i);
125                     int toRequest = sizePerSubscriber;
126                     if (i == sources.size() - 1) {
127                         toRequest += leftOver;
128                     }
129                     MultiSourceRequestableSubscriber<T, R> s = new MultiSourceRequestableSubscriber<T, R>(i, toRequest, child, this);
130                     subscribers[i] = s;
131                     o.unsafeSubscribe(s);
132                 }
133             }
134             tick();
135         }
136 
137         /**
138          * This will only allow one thread at a time to do the work, but ensures via `counter` increment/decrement
139          * that there is always once who acts on each `tick`. Same concept as used in OperationObserveOn.
140          */
141         void tick() {
142             if (WIP.getAndIncrement(this) == 0) {
143                 int emitted = 0;
144                 do {
145                     // we only emit if requested > 0
146                     if (requested.get() > 0) {
147                         Object o = buffer.poll();
148                         if (o != null) {
149                             if (buffer.isCompleted(o)) {
150                                 child.onCompleted();
151                             } else {
152                                 buffer.accept(o, child);
153                                 emitted++;
154                                 requested.decrementAndGet();
155                             }
156                         }
157                     }
158                 } while (WIP.decrementAndGet(this) > 0);
159                 if (emitted > 0) {
160                     for (MultiSourceRequestableSubscriber<T, R> s : subscribers) {
161                         s.requestUpTo(emitted);
162                     }
163                 }
164             }
165         }
166 
167         public void onCompleted(int index, boolean hadValue) {
168             if (!hadValue) {
169                 child.onCompleted();
170                 return;
171             }
172             boolean done = false;
173             synchronized (this) {
174                 if (!completion.get(index)) {
175                     completion.set(index);
176                     completionCount++;
177                     done = completionCount == collectedValues.length;
178                 }
179             }
180             if (done) {
181                 buffer.onCompleted();
182                 tick();
183             }
184         }
185 
186         /**
187          * @return boolean true if propagated value
188          */
189         public boolean onNext(int index, T t) {
190             synchronized (this) {
191                 if (!haveValues.get(index)) {
192                     haveValues.set(index);
193                     haveValuesCount++;
194                 }
195                 collectedValues[index] = t;
196                 if (haveValuesCount != collectedValues.length) {
197                     // haven't received value from each source yet so won't emit
198                     return false;
199                 } else {
200                     try {
201                         buffer.onNext(combinator.call(collectedValues));
202                     } catch (MissingBackpressureException e) {
203                         onError(e);
204                     } catch (Throwable e) {
205                         onError(e);
206                     }
207                 }
208             }
209             tick();
210             return true;
211         }
212 
213         public void onError(Throwable e) {
214             child.onError(e);
215         }
216     }

218     final static class MultiSourceRequestableSubscriber<T, R> extends Subscriber<T> {
219 
220         final MultiSourceProducer<T, R> producer;
221         final int index;
222         final AtomicLong emitted = new AtomicLong();
223         boolean hasValue = false;
224 
225         public MultiSourceRequestableSubscriber(int index, int initial, Subscriber<? super R> child, MultiSourceProducer<T, R> producer) {
226             super(child);
227             this.index = index;
228             this.producer = producer;
229             request(initial);
230         }
231 
232         public void requestUpTo(long n) {
233             do {
234                 long r = emitted.get();
235                 long u = Math.min(r, n);
236                 if (emitted.compareAndSet(r, r - u)) {
237                     request(u);
238                     break;
239                 }
240             } while (true);
241         }
242 
243         @Override
244         public void onCompleted() {
245             producer.onCompleted(index, hasValue);
246         }
247 
248         @Override
249         public void onError(Throwable e) {
250             producer.onError(e);
251         }
252 
253         @Override
254         public void onNext(T t) {
255             hasValue = true;
256             emitted.incrementAndGet();
257             boolean emitted = producer.onNext(index, t);
258             if (!emitted) {
259                 request(1);
260             }
261         }
262 
263     }

265     final static class SingleSourceProducer<T, R> implements Producer {
266         final AtomicBoolean started = new AtomicBoolean();
267         final Observable<? extends T> source;
268         final Subscriber<? super R> child;
269         final FuncN<? extends R> combinator;
270         final SingleSourceRequestableSubscriber<T, R> subscriber;
271 
272         public SingleSourceProducer(final Subscriber<? super R> child, Observable<? extends T> source, FuncN<? extends R> combinator) {
273             this.source = source;
274             this.child = child;
275             this.combinator = combinator;
276             this.subscriber = new SingleSourceRequestableSubscriber<T, R>(child, combinator);
277         }
278 
279         @Override
280         public void request(final long n) {
281             subscriber.requestMore(n);
282             if (started.compareAndSet(false, true)) {
283                 source.unsafeSubscribe(subscriber);
284             }
285 
286         }
287 
288     }
289 
290     final static class SingleSourceRequestableSubscriber<T, R> extends Subscriber<T> {
291 
292         private final Subscriber<? super R> child;
293         private final FuncN<? extends R> combinator;
294 
295         SingleSourceRequestableSubscriber(Subscriber<? super R> child, FuncN<? extends R> combinator) {
296             super(child);
297             this.child = child;
298             this.combinator = combinator;
299         }
300 
301         public void requestMore(long n) {
302             request(n);
303         }
304 
305         @Override
306         public void onNext(T t) {
307             child.onNext(combinator.call(t));
308         }
309 
310         @Override
311         public void onError(Throwable e) {
312             child.onError(e);
313         }
314 
315         @Override
316         public void onCompleted() {
317             child.onCompleted();
318         }
319     }
