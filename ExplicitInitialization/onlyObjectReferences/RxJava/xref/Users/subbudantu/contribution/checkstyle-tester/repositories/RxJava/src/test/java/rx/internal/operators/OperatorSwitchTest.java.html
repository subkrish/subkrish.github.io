411     @Test
412     public void testBackpressure() {
413         final Observable<String> o1 = Observable.create(new Observable.OnSubscribe<String>() {
414             @Override
415             public void call(final Subscriber<? super String> observer) {
416                 observer.setProducer(new Producer() {
417 
418                     private int emitted = 0;
419 
420                     @Override
421                     public void request(long n) {
422                         for(int i = 0; i < n && emitted < 10 && !observer.isUnsubscribed(); i++) {
423                             scheduler.advanceTimeBy(5, TimeUnit.MILLISECONDS);
424                             emitted++;
425                             observer.onNext("a" + emitted);
426                         }
427                         if(emitted == 10) {
428                             observer.onCompleted();
429                         }
430                     }
431                 });
432             }
433         });
434         final Observable<String> o2 = Observable.create(new Observable.OnSubscribe<String>() {
435             @Override
436             public void call(final Subscriber<? super String> observer) {
437                 observer.setProducer(new Producer() {
438 
439                     private int emitted = 0;
440 
441                     @Override
442                     public void request(long n) {
443                         for(int i = 0; i < n && emitted < 10 && !observer.isUnsubscribed(); i++) {
444                             scheduler.advanceTimeBy(5, TimeUnit.MILLISECONDS);
445                             emitted++;
446                             observer.onNext("b" + emitted);
447                         }
448                         if(emitted == 10) {
449                             observer.onCompleted();
450                         }
451                     }
452                 });
453             }
454         });
455         final Observable<String> o3 = Observable.create(new Observable.OnSubscribe<String>() {
456             @Override
457             public void call(final Subscriber<? super String> observer) {
458                 observer.setProducer(new Producer() {
459 
460                     private int emitted = 0;
461 
462                     @Override
463                     public void request(long n) {
464                         for(int i = 0; i < n && emitted < 10 && !observer.isUnsubscribed(); i++) {
465                             emitted++;
466                             observer.onNext("c" + emitted);
467                         }
468                         if(emitted == 10) {
469                             observer.onCompleted();
470                         }
471                     }
472                 });
473             }
474         });
475         Observable<Observable<String>> o = Observable.create(new Observable.OnSubscribe<Observable<String>>() {
476             @Override
477             public void call(Subscriber<? super Observable<String>> observer) {
478                 publishNext(observer, 10, o1);
479                 publishNext(observer, 20, o2);
480                 publishNext(observer, 30, o3);
481                 publishCompleted(observer, 30);
482             }
483         });
484         final TestSubscriber<String> testSubscriber = new TestSubscriber<String>();
485         Observable.switchOnNext(o).subscribe(new Subscriber<String>() {
486 
487             private int requested = 0;
488 
489             @Override
490             public void onStart() {
491                 requested = 3;
492                 request(3);
493             }
494 
495             @Override
496             public void onCompleted() {
497                 testSubscriber.onCompleted();
498             }
499 
500             @Override
501             public void onError(Throwable e) {
502                 testSubscriber.onError(e);
503             }
504 
505             @Override
506             public void onNext(String s) {
507                 testSubscriber.onNext(s);
508                 requested--;
509                 if(requested == 0) {
510                     requested = 3;
511                     request(3);
512                 }
513             }
514         });
515         scheduler.advanceTimeBy(10, TimeUnit.MILLISECONDS);
516         testSubscriber.assertReceivedOnNext(Arrays.asList("a1", "b1", "c1", "c2", "c3", "c4", "c5", "c6", "c7", "c8", "c9", "c10"));
517         testSubscriber.assertNoErrors();
518         testSubscriber.assertTerminalEvent();
519     }

564             }
565         };
566         src.subscribe(ts);

568         ts.awaitTerminalEvent(10, TimeUnit.SECONDS);
