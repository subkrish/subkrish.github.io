139     public DragLayer(Context context, AttributeSet attrs) {
140         super(context, attrs);
141 
142         final int srcColor = context.getResources().getColor(R.color.delete_color_filter);
143         mTrashPaint.setColorFilter(new PorterDuffColorFilter(srcColor, PorterDuff.Mode.SRC_ATOP));
144 
145         // Make estimated paint area in gray
146         int snagColor = context.getResources().getColor(R.color.snag_callout_color);
147         Paint estimatedPaint = new Paint();
148         estimatedPaint.setColor(snagColor);
149         estimatedPaint.setStrokeWidth(3);
150         estimatedPaint.setAntiAlias(true);
151 
152     }

154     public void startDrag(View v, DragSource source, Object dragInfo, int dragAction) {
155         if (PROFILE_DRAWING_DURING_DRAG) {
156             android.os.Debug.startMethodTracing("Launcher");
157         }
158 
159         // Hide soft keyboard, if visible
160         if (mInputMethodManager == null) {
161             mInputMethodManager = (InputMethodManager)
162                 getContext().getSystemService(Context.INPUT_METHOD_SERVICE);
163         }
164         mInputMethodManager.hideSoftInputFromWindow(getWindowToken(), 0);
165 
166         if (mListener != null) {
167             mListener.onDragStart(v, source, dragInfo, dragAction);
168         }
169 
170         Rect r = mDragRect;
171         r.set(v.getScrollX(), v.getScrollY(), 0, 0);
172 
173         offsetDescendantRectToMyCoords(v, r);
174         mTouchOffsetX = mLastMotionX - r.left;
175         mTouchOffsetY = mLastMotionY - r.top;
176 
177         v.clearFocus();
178         v.setPressed(false);
179 
180         boolean willNotCache = v.willNotCacheDrawing();
181         v.setWillNotCacheDrawing(false);
182 
183         // Reset the drawing cache background color to fully transparent
184         // for the duration of this operation
185         int color = v.getDrawingCacheBackgroundColor();
186         v.setDrawingCacheBackgroundColor(0);
187 
188         if (color != 0) {
189             v.destroyDrawingCache();
190         }
191         v.buildDrawingCache();
192         Bitmap viewBitmap = v.getDrawingCache();
193         int width = viewBitmap.getWidth();
194         int height = viewBitmap.getHeight();
195 
196         Matrix scale = new Matrix();
197         float scaleFactor = v.getWidth();
198         scaleFactor = (scaleFactor + DRAG_SCALE) /scaleFactor;
199         scale.setScale(scaleFactor, scaleFactor);
200 
201         mAnimationTo = 1.0f;
202         mAnimationFrom = 1.0f / scaleFactor;
203         mAnimationDuration = ANIMATION_SCALE_UP_DURATION;
204         mAnimationState = ANIMATION_STATE_STARTING;
205         mAnimationType = ANIMATION_TYPE_SCALE;
206 
207         mDragBitmap = Bitmap.createBitmap(viewBitmap, 0, 0, width, height, scale, true);
208         v.destroyDrawingCache();
209         v.setWillNotCacheDrawing(willNotCache);
210         v.setDrawingCacheBackgroundColor(color);
211 
212         final Bitmap dragBitmap = mDragBitmap;
213         mBitmapOffsetX = (dragBitmap.getWidth() - width) / 2;
214         mBitmapOffsetY = (dragBitmap.getHeight() - height) / 2;
215 
216         if (dragAction == DRAG_ACTION_MOVE) {
217             v.setVisibility(GONE);
218         }
219 
220         mDragPaint = null;
221         mDragging = true;
222         mShouldDrop = true;
223         mOriginator = v;
224         mDragSource = source;
225         mDragInfo = dragInfo;
226 
227         mVibrator.vibrate(VIBRATE_DURATION);
228 
229         mEnteredRegion = false;
230 
231         invalidate();
232     }

239     @Override
240     protected void dispatchDraw(Canvas canvas) {
241         super.dispatchDraw(canvas);
242 
243         if (mDragging && mDragBitmap != null) {
244             if (mAnimationState == ANIMATION_STATE_STARTING) {
245                 mAnimationStartTime = SystemClock.uptimeMillis();
246                 mAnimationState = ANIMATION_STATE_RUNNING;
247             }
248 
249             if (mAnimationState == ANIMATION_STATE_RUNNING) {
250                 float normalized = (float) (SystemClock.uptimeMillis() - mAnimationStartTime) /
251                         mAnimationDuration;
252                 if (normalized >= 1.0f) {
253                     mAnimationState = ANIMATION_STATE_DONE;
254                 }
255                 normalized = Math.min(normalized, 1.0f);
256                 final float value = mAnimationFrom  + (mAnimationTo - mAnimationFrom) * normalized;
257 
258                 switch (mAnimationType) {
259                     case ANIMATION_TYPE_SCALE:
260                         final Bitmap dragBitmap = mDragBitmap;
261                         canvas.save();
262                         canvas.translate(mScrollX + mLastMotionX - mTouchOffsetX - mBitmapOffsetX,
263                                 mScrollY + mLastMotionY - mTouchOffsetY - mBitmapOffsetY);
264                         canvas.translate((dragBitmap.getWidth() * (1.0f - value)) / 2,
265                                 (dragBitmap.getHeight() * (1.0f - value)) / 2);
266                         canvas.scale(value, value);
267                         canvas.drawBitmap(dragBitmap, 0.0f, 0.0f, mDragPaint);
268                         canvas.restore();
269                         break;
270                 }
271             } else {
272                 // Draw actual icon being dragged
273                 canvas.drawBitmap(mDragBitmap,
274                         mScrollX + mLastMotionX - mTouchOffsetX - mBitmapOffsetX,
275                         mScrollY + mLastMotionY - mTouchOffsetY - mBitmapOffsetY, mDragPaint);
276             }
277         }
278     }

280     private void endDrag() {
281         if (mDragging) {
282             mDragging = false;
283             if (mDragBitmap != null) {
284                 mDragBitmap.recycle();
285             }
286             if (mOriginator != null) {
287                 mOriginator.setVisibility(VISIBLE);
288             }
289             if (mListener != null) {
290                 mListener.onDragEnd();
291             }
292         }
293     }

295     @Override
296     public boolean onInterceptTouchEvent(MotionEvent ev) {
297         final int action = ev.getAction();
298 
299         final float x = ev.getX();
300         final float y = ev.getY();
301 
302         switch (action) {
303             case MotionEvent.ACTION_MOVE:
304                 break;
305 
306             case MotionEvent.ACTION_DOWN:
307                 // Remember location of down touch
308                 mLastMotionX = x;
309                 mLastMotionY = y;
310                 mLastDropTarget = null;
311                 break;
312 
313             case MotionEvent.ACTION_CANCEL:
314             case MotionEvent.ACTION_UP:
315                 if (mShouldDrop && drop(x, y)) {
316                     mShouldDrop = false;
317                 }
318                 endDrag();
319                 break;
320         }
321 
322         return mDragging;
323     }

325     @Override
326     public boolean onTouchEvent(MotionEvent ev) {
327         if (!mDragging) {
328             return false;
329         }
330 
331         final int action = ev.getAction();
332         final float x = ev.getX();
333         final float y = ev.getY();
334 
335         switch (action) {
336         case MotionEvent.ACTION_DOWN:
337 
338             // Remember where the motion event started
339             mLastMotionX = x;
340             mLastMotionY = y;
341 
342             if ((x < SCROLL_ZONE) || (x > getWidth() - SCROLL_ZONE)) {
343                 mScrollState = SCROLL_WAITING_IN_ZONE;
344                 postDelayed(mScrollRunnable, SCROLL_DELAY);
345             } else {
346                 mScrollState = SCROLL_OUTSIDE_ZONE;
347             }
348 
349             break;
350         case MotionEvent.ACTION_MOVE:
351             final int scrollX = mScrollX;
352             final int scrollY = mScrollY;
353 
354             final float touchX = mTouchOffsetX;
355             final float touchY = mTouchOffsetY;
356 
357             final int offsetX = mBitmapOffsetX;
358             final int offsetY = mBitmapOffsetY;
359 
360             int left = (int) (scrollX + mLastMotionX - touchX - offsetX);
361             int top = (int) (scrollY + mLastMotionY - touchY - offsetY);
362 
363             final Bitmap dragBitmap = mDragBitmap;
364             final int width = dragBitmap.getWidth();
365             final int height = dragBitmap.getHeight();
366 
367             final Rect rect = mRect;
368             rect.set(left - 1, top - 1, left + width + 1, top + height + 1);
369 
370             mLastMotionX = x;
371             mLastMotionY = y;
372 
373             left = (int) (scrollX + x - touchX - offsetX);
374             top = (int) (scrollY + y - touchY - offsetY);
375 
376             // Invalidate current icon position
377             rect.union(left - 1, top - 1, left + width + 1, top + height + 1);
378 
379             final int[] coordinates = mDropCoordinates;
380             DropTarget dropTarget = findDropTarget((int) x, (int) y, coordinates);
381             if (dropTarget != null) {
382                 if (mLastDropTarget == dropTarget) {
383                     dropTarget.onDragOver(mDragSource, coordinates[0], coordinates[1],
384                         (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
385                 } else {
386                     if (mLastDropTarget != null) {
387                         mLastDropTarget.onDragExit(mDragSource, coordinates[0], coordinates[1],
388                             (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
389                     }
390                     dropTarget.onDragEnter(mDragSource, coordinates[0], coordinates[1],
391                         (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
392                 }
393             } else {
394                 if (mLastDropTarget != null) {
395                     mLastDropTarget.onDragExit(mDragSource, coordinates[0], coordinates[1],
396                         (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
397                 }
398             }
399 
400             invalidate(rect);
401 
402             mLastDropTarget = dropTarget;
403 
404             boolean inDragRegion = false;
405             if (mDragRegion != null) {
406                 final RectF region = mDragRegion;
407                 final boolean inRegion = region.contains(ev.getRawX(), ev.getRawY());
408                 if (!mEnteredRegion && inRegion) {
409                     mDragPaint = mTrashPaint;
410                     mEnteredRegion = true;
411                     inDragRegion = true;
412                 } else if (mEnteredRegion && !inRegion) {
413                     mDragPaint = null;
414                     mEnteredRegion = false;
415                 }
416             }
417 
418             if (!inDragRegion && x < SCROLL_ZONE) {
419                 if (mScrollState == SCROLL_OUTSIDE_ZONE) {
420                     mScrollState = SCROLL_WAITING_IN_ZONE;
421                     mScrollRunnable.setDirection(SCROLL_LEFT);
422                     postDelayed(mScrollRunnable, SCROLL_DELAY);
423                 }
424             } else if (!inDragRegion && x > getWidth() - SCROLL_ZONE) {
425                 if (mScrollState == SCROLL_OUTSIDE_ZONE) {
426                     mScrollState = SCROLL_WAITING_IN_ZONE;
427                     mScrollRunnable.setDirection(SCROLL_RIGHT);
428                     postDelayed(mScrollRunnable, SCROLL_DELAY);
429                 }
430             } else {
431                 if (mScrollState == SCROLL_WAITING_IN_ZONE) {
432                     mScrollState = SCROLL_OUTSIDE_ZONE;
433                     mScrollRunnable.setDirection(SCROLL_RIGHT);
434                     removeCallbacks(mScrollRunnable);
435                 }
436             }
437 
438             break;
439         case MotionEvent.ACTION_UP:
440             removeCallbacks(mScrollRunnable);
441             if (mShouldDrop) {
442                 drop(x, y);
443                 mShouldDrop = false;
444             }
445             endDrag();
446 
447             break;
448         case MotionEvent.ACTION_CANCEL:
449             endDrag();
450         }
451 
452         return true;
453     }

455     private boolean drop(float x, float y) {
456         invalidate();
457 
458         final int[] coordinates = mDropCoordinates;
459         DropTarget dropTarget = findDropTarget((int) x, (int) y, coordinates);
460 
461         if (dropTarget != null) {
462             dropTarget.onDragExit(mDragSource, coordinates[0], coordinates[1],
463                     (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
464             if (dropTarget.acceptDrop(mDragSource, coordinates[0], coordinates[1],
465                     (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo)) {
466                 dropTarget.onDrop(mDragSource, coordinates[0], coordinates[1],
467                         (int) mTouchOffsetX, (int) mTouchOffsetY, mDragInfo);
468                 mDragSource.onDropCompleted((View) dropTarget, true);
469                 return true;
470             } else {
471                 mDragSource.onDropCompleted((View) dropTarget, false);
472                 return true;
473             }
474         }
475         return false;
476     }

478     DropTarget findDropTarget(int x, int y, int[] dropCoordinates) {
479         return findDropTarget(this, x, y, dropCoordinates);
480     }

482     private DropTarget findDropTarget(ViewGroup container, int x, int y, int[] dropCoordinates) {
483         final Rect r = mDragRect;
484         final int count = container.getChildCount();
485         final int scrolledX = x + container.getScrollX();
486         final int scrolledY = y + container.getScrollY();
487         final View ignoredDropTarget = mIgnoredDropTarget;
488 
489         for (int i = count - 1; i >= 0; i--) {
490             final View child = container.getChildAt(i);
491             if (child.getVisibility() == VISIBLE && child != ignoredDropTarget) {
492                 child.getHitRect(r);
493                 if (r.contains(scrolledX, scrolledY)) {
494                     DropTarget target = null;
495                     if (child instanceof ViewGroup) {
496                         x = scrolledX - child.getLeft();
497                         y = scrolledY - child.getTop();
498                         target = findDropTarget((ViewGroup) child, x, y, dropCoordinates);
499                     }
500                     if (target == null) {
501                         if (child instanceof DropTarget) {
502                             // Only consider this child if they will accept
503                             DropTarget childTarget = (DropTarget) child;
504                             if (childTarget.acceptDrop(mDragSource, x, y, 0, 0, mDragInfo)) {
505                                 dropCoordinates[0] = x;
506                                 dropCoordinates[1] = y;
507                                 return (DropTarget) child;
508                             } else {
509                                 return null;
510                             }
511                         }
512                     } else {
513                         return target;
514                     }
515                 }
516             }
517         }
518 
519         return null;
520     }

554     private class ScrollRunnable implements Runnable {
555         private int mDirection;
556 
557         ScrollRunnable() {
558         }
559 
560         public void run() {
561             if (mDragScroller != null) {
562                 if (mDirection == SCROLL_LEFT) {
563                     mDragScroller.scrollLeft();
564                 } else {
565                     mDragScroller.scrollRight();
566                 }
567                 mScrollState = SCROLL_OUTSIDE_ZONE;
568             }
569         }
570 
571         void setDirection(int direction) {
572             mDirection = direction;
573         }
