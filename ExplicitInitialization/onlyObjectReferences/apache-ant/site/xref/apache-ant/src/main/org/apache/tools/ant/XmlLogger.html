40  /**
41   * Generates a file in the current directory with
42   * an XML description of what happened during a build.
43   * The default filename is "log.xml", but this can be overridden
44   * with the property <code>XmlLogger.file</code>.
45   *
46   * This implementation assumes in its sanity checking that only one
47   * thread runs a particular target/task at a time. This is enforced
48   * by the way that parallel builds and antcalls are done - and
49   * indeed all but the simplest of tasks could run into problems
50   * if executed in parallel.
51   *
52   * @see Project#addBuildListener(BuildListener)
53   */
54  public class XmlLogger implements BuildLogger {
55  
56      private int msgOutputLevel = Project.MSG_DEBUG;
57      private PrintStream outStream;
58  
59      /** DocumentBuilder to use when creating the document to start with. */
60      private static DocumentBuilder builder = getDocumentBuilder();
61  
62      /**
63       * Returns a default DocumentBuilder instance or throws an
64       * ExceptionInInitializerError if it can't be created.
65       *
66       * @return a default DocumentBuilder instance.
67       */
68      private static DocumentBuilder getDocumentBuilder() {
69          try {
70              return DocumentBuilderFactory.newInstance().newDocumentBuilder();
71          } catch (Exception exc) {
72              throw new ExceptionInInitializerError(exc);
73          }
74      }

76      /** XML element name for a build. */
77      private static final String BUILD_TAG = "build";
78  
79      /** XML element name for a target. */
80      private static final String TARGET_TAG = "target";
81  
82      /** XML element name for a task. */
83      private static final String TASK_TAG = "task";
84  
85      /** XML element name for a message. */
86      private static final String MESSAGE_TAG = "message";
87  
88      /** XML attribute name for a name. */
89      private static final String NAME_ATTR = "name";
90  
91      /** XML attribute name for a time. */
92      private static final String TIME_ATTR = "time";
93  
94      /** XML attribute name for a message priority. */
95      private static final String PRIORITY_ATTR = "priority";
96  
97      /** XML attribute name for a file location. */
98      private static final String LOCATION_ATTR = "location";

100     /** XML attribute name for an error description. */
101     private static final String ERROR_ATTR = "error";
102 
103     /** XML element name for a stack trace. */
104     private static final String STACKTRACE_TAG = "stacktrace";
105 
106     /** The complete log document for this build. */
107     private Document doc = builder.newDocument();
108 
109     /** Mapping for when tasks started (Task to TimedElement). */
110     private Hashtable<Task, TimedElement> tasks = new Hashtable<Task, TimedElement>();
111 
112     /** Mapping for when targets started (Target to TimedElement). */
113     private Hashtable<Target, TimedElement> targets = new Hashtable<Target, XmlLogger.TimedElement>();
114 
115     /**
116      * Mapping of threads to stacks of elements
117      * (Thread to Stack of TimedElement).
118      */
119     private Hashtable<Thread, Stack<TimedElement>> threadStacks = new Hashtable<Thread, Stack<TimedElement>>();
120 
121     /**
122      * When the build started.
123      */
124     private TimedElement buildElement = null;
125 
126     /** Utility class representing the time an element started. */
127     private static class TimedElement {
128         /**
129          * Start time in milliseconds
130          * (as returned by <code>System.currentTimeMillis()</code>).
131          */
132         private long startTime;
133         /** Element created at the start time. */
134         private Element element;
135         public String toString() {
136             return element.getTagName() + ":" + element.getAttribute("name");
137         }
138     }

141      *  Constructs a new BuildListener that logs build events to an XML file.

144     }

146     /**
147      * Fired when the build starts, this builds the top-level element for the
148      * document and remembers the time of the start of the build.
149      *
150      * @param event Ignored.
151      */
152     public void buildStarted(BuildEvent event) {
153         buildElement = new TimedElement();
154         buildElement.startTime = System.currentTimeMillis();
155         buildElement.element = doc.createElement(BUILD_TAG);
156     }
157 
158     /**
159      * Fired when the build finishes, this adds the time taken and any
160      * error stacktrace to the build element and writes the document to disk.
161      *
162      * @param event An event with any relevant extra information.
163      *              Will not be <code>null</code>.
164      */
165     public void buildFinished(BuildEvent event) {
166         long totalTime = System.currentTimeMillis() - buildElement.startTime;
167         buildElement.element.setAttribute(TIME_ATTR, DefaultLogger.formatTime(totalTime));
168 
169         if (event.getException() != null) {
170             buildElement.element.setAttribute(ERROR_ATTR, event.getException().toString());
171             // print the stacktrace in the build file it is always useful...
172             // better have too much info than not enough.
173             Throwable t = event.getException();
174             Text errText = doc.createCDATASection(StringUtils.getStackTrace(t));
175             Element stacktrace = doc.createElement(STACKTRACE_TAG);
176             stacktrace.appendChild(errText);
177             synchronizedAppend(buildElement.element, stacktrace);
178         }
179         String outFilename = event.getProject().getProperty("XmlLogger.file");
180         if (outFilename == null) {
181             outFilename = "log.xml";
182         }
183         String xslUri = event.getProject().getProperty("ant.XmlLogger.stylesheet.uri");
184         if (xslUri == null) {
185             xslUri = "log.xsl";
186         }
187         Writer out = null;
188         try {
189             // specify output in UTF8 otherwise accented characters will blow
190             // up everything
191             OutputStream stream = outStream;
192             if (stream == null) {
193                 stream = new FileOutputStream(outFilename);
194             }
195             out = new OutputStreamWriter(stream, "UTF8");
196             out.write("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
197             if (xslUri.length() > 0) {
198                 out.write("<?xml-stylesheet type=\"text/xsl\" href=\"" + xslUri + "\"?>\n\n");
199             }
200             new DOMElementWriter().write(buildElement.element, out, 0, "\t");
201             out.flush();
202         } catch (IOException exc) {
203             throw new BuildException("Unable to write log file", exc);
204         } finally {
205             FileUtils.close(out);
206         }
207         buildElement = null;
208     }

210     /**
211      * Returns the stack of timed elements for the current thread.
212      * @return the stack of timed elements for the current thread
213      */
214     private Stack<TimedElement> getStack() {
215         Stack<TimedElement> threadStack = threadStacks.get(Thread.currentThread());
216         if (threadStack == null) {
217             threadStack = new Stack<TimedElement>();
218             threadStacks.put(Thread.currentThread(), threadStack);
219         }
220         /* For debugging purposes uncomment:
221         org.w3c.dom.Comment s = doc.createComment("stack=" + threadStack);
222         buildElement.element.appendChild(s);
223          */
224         return threadStack;
225     }

227     /**
228      * Fired when a target starts building, this pushes a timed element
229      * for the target onto the stack of elements for the current thread,
230      * remembering the current time and the name of the target.
231      *
232      * @param event An event with any relevant extra information.
233      *              Will not be <code>null</code>.
234      */
235     public void targetStarted(BuildEvent event) {
236         Target target = event.getTarget();
237         TimedElement targetElement = new TimedElement();
238         targetElement.startTime = System.currentTimeMillis();
239         targetElement.element = doc.createElement(TARGET_TAG);
240         targetElement.element.setAttribute(NAME_ATTR, target.getName());
241         targets.put(target, targetElement);
242         getStack().push(targetElement);
243     }
244 
245     /**
246      * Fired when a target finishes building, this adds the time taken
247      * and any error stacktrace to the appropriate target element in the log.
248      *
249      * @param event An event with any relevant extra information.
250      *              Will not be <code>null</code>.
251      */
252     public void targetFinished(BuildEvent event) {
253         Target target = event.getTarget();
254         TimedElement targetElement = (TimedElement) targets.get(target);
255         if (targetElement != null) {
256             long totalTime = System.currentTimeMillis() - targetElement.startTime;
257             targetElement.element.setAttribute(TIME_ATTR, DefaultLogger.formatTime(totalTime));
258 
259             TimedElement parentElement = null;
260             Stack<TimedElement> threadStack = getStack();
261             if (!threadStack.empty()) {
262                 TimedElement poppedStack = threadStack.pop();
263                 if (poppedStack != targetElement) {
264                     throw new RuntimeException("Mismatch - popped element = " + poppedStack
265                             + " finished target element = " + targetElement);
266                 }
267                 if (!threadStack.empty()) {
268                     parentElement = (TimedElement) threadStack.peek();
269                 }
270             }
271             if (parentElement == null) {
272                 synchronizedAppend(buildElement.element, targetElement.element);
273             } else {
274                 synchronizedAppend(parentElement.element,
275                                    targetElement.element);
276             }
277         }
278         targets.remove(target);
279     }
280 
281     /**
282      * Fired when a task starts building, this pushes a timed element
283      * for the task onto the stack of elements for the current thread,
284      * remembering the current time and the name of the task.
285      *
286      * @param event An event with any relevant extra information.
287      *              Will not be <code>null</code>.
288      */
289     public void taskStarted(BuildEvent event) {
290         TimedElement taskElement = new TimedElement();
291         taskElement.startTime = System.currentTimeMillis();
292         taskElement.element = doc.createElement(TASK_TAG);
293 
294         Task task = event.getTask();
295         String name = event.getTask().getTaskName();
296         if (name == null) {
297             name = "";
298         }
299         taskElement.element.setAttribute(NAME_ATTR, name);
300         taskElement.element.setAttribute(LOCATION_ATTR, event.getTask().getLocation().toString());
301         tasks.put(task, taskElement);
302         getStack().push(taskElement);
303     }
304 
305     /**
306      * Fired when a task finishes building, this adds the time taken
307      * and any error stacktrace to the appropriate task element in the log.
308      *
309      * @param event An event with any relevant extra information.
310      *              Will not be <code>null</code>.
311      */
312     public void taskFinished(BuildEvent event) {
313         Task task = event.getTask();
314         TimedElement taskElement = (TimedElement) tasks.get(task);
315         if (taskElement == null) {
316             throw new RuntimeException("Unknown task " + task + " not in " + tasks);
317         }
318         long totalTime = System.currentTimeMillis() - taskElement.startTime;
319         taskElement.element.setAttribute(TIME_ATTR, DefaultLogger.formatTime(totalTime));
320         Target target = task.getOwningTarget();
321         TimedElement targetElement = null;
322         if (target != null) {
323             targetElement = (TimedElement) targets.get(target);
324         }
325         if (targetElement == null) {
326             synchronizedAppend(buildElement.element, taskElement.element);
327         } else {
328             synchronizedAppend(targetElement.element, taskElement.element);
329         }
330         Stack<TimedElement> threadStack = getStack();
331         if (!threadStack.empty()) {
332             TimedElement poppedStack = threadStack.pop();
333             if (poppedStack != taskElement) {
334                 throw new RuntimeException("Mismatch - popped element = " + poppedStack
335                         + " finished task element = " + taskElement);
336             }
337         }
338         tasks.remove(task);
339     }

341     /**
342      * Get the TimedElement associated with a task.
343      *
344      * Where the task is not found directly, search for unknown elements which
345      * may be hiding the real task
346      */
347     private TimedElement getTaskElement(Task task) {
348         TimedElement element = (TimedElement) tasks.get(task);
349         if (element != null) {
350             return element;
351         }
352         for (Enumeration<Task> e = tasks.keys(); e.hasMoreElements();) {
353             Task key = e.nextElement();
354             if (key instanceof UnknownElement) {
355                 if (((UnknownElement) key).getTask() == task) {
356                     return (TimedElement) tasks.get(key);
357                 }
358             }
359         }
360         return null;
361     }

363     /**
364      * Fired when a message is logged, this adds a message element to the
365      * most appropriate parent element (task, target or build) and records
366      * the priority and text of the message.
367      *
368      * @param event An event with any relevant extra information.
369      *              Will not be <code>null</code>.
370      */
371     public void messageLogged(BuildEvent event) {
372         int priority = event.getPriority();
373         if (priority > msgOutputLevel) {
374             return;
375         }
376         Element messageElement = doc.createElement(MESSAGE_TAG);
377 
378         String name = "debug";
379         switch (priority) {
380             case Project.MSG_ERR:
381                 name = "error";
382                 break;
383             case Project.MSG_WARN:
384                 name = "warn";
385                 break;
386             case Project.MSG_INFO:
387                 name = "info";
388                 break;
389             default:
390                 name = "debug";
391                 break;
392         }
393         messageElement.setAttribute(PRIORITY_ATTR, name);
394 
395         Throwable ex = event.getException();
396         if (Project.MSG_DEBUG <= msgOutputLevel && ex != null) {
397             Text errText = doc.createCDATASection(StringUtils.getStackTrace(ex));
398             Element stacktrace = doc.createElement(STACKTRACE_TAG);
399             stacktrace.appendChild(errText);
400             synchronizedAppend(buildElement.element, stacktrace);
401         }
402         Text messageText = doc.createCDATASection(event.getMessage());
403         messageElement.appendChild(messageText);
404 
405         TimedElement parentElement = null;
406 
407         Task task = event.getTask();
408 
409         Target target = event.getTarget();
410         if (task != null) {
411             parentElement = getTaskElement(task);
412         }
413         if (parentElement == null && target != null) {
414             parentElement = (TimedElement) targets.get(target);
415         }
416         if (parentElement != null) {
417             synchronizedAppend(parentElement.element, messageElement);
418         } else {
419             synchronizedAppend(buildElement.element, messageElement);
420         }
421     }

465     private void synchronizedAppend(Node parent, Node child) {
466         synchronized(parent) {
467             parent.appendChild(child);
468         }
