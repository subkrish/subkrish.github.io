20  import java.io.IOException;
21  import java.io.Reader;
22  import java.util.ArrayList;
23  import java.util.Collections;
24  import java.util.Comparator;
25  import java.util.Iterator;
26  import java.util.List;
27  
28  import org.apache.tools.ant.BuildException;
29  import org.apache.tools.ant.types.Parameter;
30  
31  /**
32   * <p>
33   * Sort a file before and/or after the file.
34   * </p>
35   *
36   * <p>
37   * Examples:
38   * </p>
39   *
40   * <pre>
41   *   &lt;copy todir=&quot;build&quot;&gt;
42   *       &lt;fileset dir=&quot;input&quot; includes=&quot;*.txt&quot;/&gt;
43   *       &lt;filterchain&gt;
44   *           &lt;sortfilter/&gt;
45   *       &lt;/filterchain&gt;
46   *   &lt;/copy&gt;
47   * </pre>
48   *
49   * <p>
50   * Sort all files <code>*.txt</code> from <i>src</i> location and copy
51   * them into <i>build</i> location. The lines of each file are sorted
52   * in ascendant order comparing the lines via the
53   * <code>String.compareTo(Object o)</code> method.
54   * </p>
55   *
56   * <pre>
57   *   &lt;copy todir=&quot;build&quot;&gt;
58   *       &lt;fileset dir=&quot;input&quot; includes=&quot;*.txt&quot;/&gt;
59   *       &lt;filterchain&gt;
60   *           &lt;sortfilter reverse=&quot;true&quot;/&gt;
61   *       &lt;/filterchain&gt;
62   *   &lt;/copy&gt;
63   * </pre>
64   *
65   * <p>
66   * Sort all files <code>*.txt</code> from <i>src</i> location into reverse
67   * order and copy them into <i>build</i> location. If reverse parameter has
68   * value <code>true</code> (default value), then the output line of the files
69   * will be in ascendant order.
70   * </p>
71   *
72   * <pre>
73   *   &lt;copy todir=&quot;build&quot;&gt;
74   *       &lt;fileset dir=&quot;input&quot; includes=&quot;*.txt&quot;/&gt;
75   *       &lt;filterchain&gt;
76   *           &lt;filterreader classname=&quot;org.apache.tools.ant.filters.SortFilter&quot;&gt;
77   *             &lt;param name=&quot;comparator&quot; value=&quot;org.apache.tools.ant.filters.EvenFirstCmp&quot;/&gt;
78   *           &lt;/filterreader&gt;
79   *       &lt;/filterchain&gt;
80   *   &lt;/copy&gt;
81   * </pre>
82   *
83   * <p>
84   * Sort all files <code>*.txt</code> from <i>src</i> location using as
85   * sorting criterium <code>EvenFirstCmp</code> class, that sorts the file
86   * lines putting even lines first then odd lines for example. The modified files
87   * are copied into <i>build</i> location. The <code>EvenFirstCmp</code>,
88   * has to an instanciable class via <code>Class.newInstance()</code>,
89   * therefore in case of inner class has to be <em>static</em>. It also has to
90   * implement <code>java.util.Comparator</code> interface, for example:
91   * </p>
92   *
93   * <pre>
94   *         package org.apache.tools.ant.filters;
95   *         ...(omitted)
96   *           public final class EvenFirstCmp implements &lt;b&gt;Comparator&lt;/b&gt; {
97   *             public int compare(Object o1, Object o2) {
98   *             ...(omitted)
99   *             }
100  *           }
101  * </pre>
102  *
103  * <p>The example above is equivalent to:</p>
104  *
105  * <blockquote><pre>
106  *   &lt;componentdef name="evenfirst"
107  *                 classname="org.apache.tools.ant.filters.EvenFirstCmp&quot;/&gt;
108  *   &lt;copy todir=&quot;build&quot;&gt;
109  *       &lt;fileset dir=&quot;input&quot; includes=&quot;*.txt&quot;/&gt;
110  *       &lt;filterchain&gt;
111  *           &lt;sortfilter&gt;
112  *               &lt;evenfirst/&gt;
113  *           &lt;/sortfilter&gt;
114  *       &lt;/filterchain&gt;
115  *   &lt;/copy&gt;
116  * </pre></blockquote>
117  *
118  * <p> If parameter <code>comparator</code> is present, then
119  * <code>reverse</code> parameter will not be taken into account.  </p>
120  *
121  * @since Ant 1.8.0
122  */
123 public final class SortFilter extends BaseParamFilterReader
124     implements ChainableReader {
125 
126     /** Parameter name for reverse order. */
127     private static final String REVERSE_KEY = "reverse";
128 
129     /**
130      * Parameter name for specifying the comparator criteria via class that
131      * implement <code>java.util.Comparator</code> interface.
132      */
133     private static final String COMPARATOR_KEY = "comparator";
134 
135     /**
136      * Instance of comparator class to be used for sorting.
137      */
138     private Comparator<? super String> comparator = null;
139 
140     /**
141      * Controls if the sorting process will be in ascendant/descendant order. If
142      * If has value <code>true</code>, then the line of the file will be
143      * sorted on descendant order. Default value: <code>false</code>. It will
144      * be considered only if <code>comparator</code> is <code>null</code>.
145      */
146     private boolean reverse;
147 
148     /**
149      * Stores the lines to be sorted.
150      */
151     private List<String> lines;
152 
153     /**
154      * Remaining line to be read from this filter, or <code>null</code> if the
155      * next call to <code>read()</code> should read the original stream to
156      * find the next matching line.
157      */
158     private String line = null;
159 
160     private Iterator<String> iterator = null;
161 
162     /**
163      * Constructor for "dummy" instances.
164      *
165      * @see BaseFilterReader#BaseFilterReader()
166      */
167     public SortFilter() {
168         super();
169     }
170 
171     /**
172      * Creates a new filtered reader.
173      *
174      * @param in
175      *            A Reader object providing the underlying stream. Must not be
176      *            <code>null</code>.
177      */
178     public SortFilter(final Reader in) {
179         super(in);
180     }
181 
182     /**
183      * Returns the next character in the filtered stream. If the desired number
184      * of lines have already been read, the resulting stream is effectively at
185      * an end. Otherwise, the next character from the underlying stream is read
186      * and returned.
187      *
188      * @return the next character in the resulting stream, or -1 if the end of
189      *         the resulting stream has been reached
190      *
191      * @exception IOException
192      *                if the underlying stream throws an IOException during
193      *                reading
194      */
195     public int read() throws IOException {
196         if (!getInitialized()) {
197             initialize();
198             setInitialized(true);
199         }
200 
201         int ch = -1;
202         if (line != null) {
203             /*
204              * We are on the state: "reading the current line", lines are
205              * already sorted
206              */
207             ch = line.charAt(0);
208             if (line.length() == 1) {
209                 line = null;
210             } else {
211                 line = line.substring(1);
212             }
213         } else {
214             if (lines == null) {
215                 // We read all lines and sort them
216                 lines = new ArrayList<String>();
217                 for (line = readLine(); line != null; line = readLine()) {
218                     lines.add(line);
219                 }
220                 sort();
221                 iterator = lines.iterator();
222             }
223 
224             if (iterator.hasNext()) {
225                 line = (String) iterator.next();
226             } else {
227                 line = null;
228                 lines = null;
229                 iterator = null;
230             }
231             if (line != null) {
232                 return read();
233             }
234         }
235         return ch;
236     }
237 
238     /**
239      * Creates a new SortReader using the passed in Reader for instantiation.
240      *
241      * @param rdr
242      *            A Reader object providing the underlying stream. Must not be
243      *            <code>null</code>.
244      *
245      * @return a new filter based on this configuration, but filtering the
246      *         specified reader
247      */
248     public Reader chain(final Reader rdr) {
249         SortFilter newFilter = new SortFilter(rdr);
250         newFilter.setReverse(isReverse());
251         newFilter.setComparator(getComparator());
252         newFilter.setInitialized(true);
253         return newFilter;
254     }
255 
256     /**
257      * Returns <code>true</code> if the sorting process will be in reverse
258      * order, otherwise the sorting process will be in ascendant order.
259      *
260      * @return <code>true</code> if the sorting process will be in reverse
261      *         order, otherwise the sorting process will be in ascendant order.
262      */
263     public boolean isReverse() {
264         return reverse;
265     }
266 
267     /**
268      * Sets the sorting process will be in ascendant (<code>reverse=false</code>)
269      * or to descendant (<code>reverse=true</code>).
270      *
271      * @param reverse
272      *            Boolean representing reverse ordering process.
273      */
274     public void setReverse(boolean reverse) {
275         this.reverse = reverse;
276     }
277 
278     /**
279      * Returns the comparator to be used for sorting.
280      *
281      * @return the comparator
282      */
283     public Comparator<? super String> getComparator() {
284         return comparator;
285     }
286 
287     /**
288      * Set the comparator to be used as sorting criterium.
289      *
290      * @param comparator
291      *            the comparator to set
292      */
293     public void setComparator(Comparator<? super String> comparator) {
294         this.comparator = comparator;
295     }
296 
297     /**
298      * Set the comparator to be used as sorting criterion as nested element.
299      *
300      * @param comparator
301      *            the comparator to set
302      */
303     public void add(Comparator<? super String> comparator) {
304         if (this.comparator != null && comparator != null) {
305             throw new BuildException("can't have more than one comparator");
306         }
307         setComparator(comparator);
308     }
309 
310     /**
311      * Scans the parameters list
312      */
313     private void initialize() throws IOException {
314         // get parameters
315         Parameter[] params = getParameters();
316         if (params != null) {
317             for (int i = 0; i < params.length; i++) {
318                 final String paramName = params[i].getName();
319                 if (REVERSE_KEY.equals(paramName)) {
320                     setReverse(Boolean.valueOf(params[i].getValue())
321                                .booleanValue());
322                     continue;
323                 }
324                 if (COMPARATOR_KEY.equals(paramName)) {
325                     try {
326                         String className = (String) params[i].getValue();
327                         @SuppressWarnings("unchecked")
328                         final Comparator<? super String> comparatorInstance = (Comparator<? super String>) (Class
329                                 .forName(className).newInstance());
330                         setComparator(comparatorInstance);
331                         continue;
332                     } catch (InstantiationException e) {
333                         throw new BuildException(e);
334                     } catch (IllegalAccessException e) {
335                         /*
336                          * Probably a inner non-static class, this this case is
337                          * not considered
338                          */
339                         throw new BuildException(e);
340                     } catch (ClassNotFoundException e) {
341                         throw new BuildException(e);
342                     } catch (ClassCastException e) {
343                         throw new BuildException("Value of comparator attribute"
344                                                  + " should implement"
345                                                  + " java.util.Comparator"
346                                                  + " interface");
347                     } catch (Exception e) {
348                         throw new BuildException(e);
349                     }
350                 }
351             }
352         }
353     }
354 
355     /**
356      * Sorts the read lines (<code>lines</code>)acording to the sorting
357      * criteria defined by the user.
358      *
359      */
360     private void sort() {
361         if (comparator == null) {
362             if (reverse) {
363                 Collections.sort(lines, new Comparator<String>() {
364                         public int compare(String s1, String s2) {
365                             return (-s1.compareTo(s2));
366                         }
367                     });
368             } else {
369                 Collections.sort(lines);
370             }
371         } else {
372             Collections.sort(lines, comparator);
373         }
374     }
