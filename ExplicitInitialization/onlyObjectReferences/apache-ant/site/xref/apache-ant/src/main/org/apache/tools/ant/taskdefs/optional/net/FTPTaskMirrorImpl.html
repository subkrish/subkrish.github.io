253         public void scan() {
254             if (includes == null) {
255                 // No includes supplied, so set it to 'matches all'
256                 includes = new String[1];
257                 includes[0] = "**";
258             }
259             if (excludes == null) {

264             filesNotIncluded = new Vector();
265             filesExcluded = new VectorSet();
266             dirsIncluded = new VectorSet();
267             dirsNotIncluded = new Vector();
268             dirsExcluded = new VectorSet();
269 
270             try {
271                 String cwd = ftp.printWorkingDirectory();
272                 // always start from the current ftp working dir
273                 forceRemoteSensitivityCheck();
274 
275                 checkIncludePatterns();
276                 clearCaches();
277                 ftp.changeWorkingDirectory(cwd);
278             } catch (IOException e) {
279                 throw new BuildException("Unable to scan FTP server: ", e);
280             }

284         /**
285          * this routine is actually checking all the include patterns in
286          * order to avoid scanning everything under base dir
287          * @since ant1.6
288          */
289         private void checkIncludePatterns() {
290 
291             Hashtable newroots = new Hashtable();
292             // put in the newroots vector the include patterns without
293             // wildcard tokens
294             for (int icounter = 0; icounter < includes.length; icounter++) {
295                 String newpattern =
296                     SelectorUtils.rtrimWildcardTokens(includes[icounter]);
297                 newroots.put(newpattern, includes[icounter]);
298             }
299             if (task.getRemotedir() == null) {
300                 try {
301                     task.setRemotedir(ftp.printWorkingDirectory());
302                 } catch (IOException e) {
303                     throw new BuildException("could not read current ftp directory",
304                                              task.getLocation());
305                 }
306             }
307             AntFTPFile baseFTPFile = new AntFTPRootFile(ftp, task.getRemotedir());
308             rootPath = baseFTPFile.getAbsolutePath();
309             // construct it
310             if (newroots.containsKey("")) {
311                 // we are going to scan everything anyway
312                 scandir(rootPath, "", true);
313             } else {
314                 // only scan directories that can include matched files or
315                 // directories
316                 Enumeration enum2 = newroots.keys();
317 
318                 while (enum2.hasMoreElements()) {
319                     String currentelement = (String) enum2.nextElement();
320                     String originalpattern = (String) newroots.get(currentelement);
321                     AntFTPFile myfile = new AntFTPFile(baseFTPFile, currentelement);
322                     boolean isOK = true;
323                     boolean traversesSymlinks = false;
324                     String path = null;
325 
326                     if (myfile.exists()) {
327                         forceRemoteSensitivityCheck();
328                         if (remoteSensitivityChecked
329                             && remoteSystemCaseSensitive && isFollowSymlinks()) {
330                             // cool case,
331                             //we do not need to scan all the subdirs in the relative path
332                             path = myfile.getFastRelativePath();
333                         } else {
334                             // may be on a case insensitive file system.  We want
335                             // the results to show what's really on the disk, so
336                             // we need to double check.
337                             try {
338                                 path = myfile.getRelativePath();
339                                 traversesSymlinks = myfile.isTraverseSymlinks();
340                             }  catch (IOException be) {
341                                 throw new BuildException(be, task.getLocation());
342                             } catch (BuildException be) {
343                                 isOK = false;
344                             }
345                         }
346                     } else {
347                         isOK = false;
348                     }
349                     if (isOK) {
350                         currentelement = path.replace(task.getSeparator().charAt(0), File.separatorChar);
351                         if (!isFollowSymlinks()
352                             && traversesSymlinks) {
353                             continue;
354                         }
355 
356                         if (myfile.isDirectory()) {
357                             if (isIncluded(currentelement)
358                                 && currentelement.length() > 0) {
359                                 accountForIncludedDir(currentelement, myfile, true);
360                             }  else {
361                                 if (currentelement.length() > 0) {
362                                     if (currentelement.charAt(currentelement
363                                                               .length() - 1)
364                                         != File.separatorChar) {
365                                         currentelement =
366                                             currentelement + File.separatorChar;
367                                     }
368                                 }
369                                 scandir(myfile.getAbsolutePath(), currentelement, true);
370                             }
371                         } else {
372                             if (isCaseSensitive
373                                 && originalpattern.equals(currentelement)) {
374                                 accountForIncludedFile(currentelement);
375                             } else if (!isCaseSensitive
376                                        && originalpattern
377                                        .equalsIgnoreCase(currentelement)) {
378                                 accountForIncludedFile(currentelement);
379                             }
380                         }
381                     }
382                 }
383             }
384         }
385         /**
386          * scans a particular directory. populates the scannedDirs cache.
387          *
388          * @param dir directory to scan
389          * @param vpath  relative path to the base directory of the remote fileset
390          * always ended with a File.separator
391          * @param fast seems to be always true in practice
392          */
393         protected void scandir(String dir, String vpath, boolean fast) {
394             // avoid double scanning of directories, can only happen in fast mode
395             if (fast && hasBeenScanned(vpath)) {
396                 return;
397             }
398             try {
399                 if (!ftp.changeWorkingDirectory(dir)) {
400                     return;
401                 }
402                 String completePath = null;
403                 if (!vpath.equals("")) {
404                     completePath = rootPath + task.getSeparator()
405                         + vpath.replace(File.separatorChar, task.getSeparator().charAt(0));
406                 } else {
407                     completePath = rootPath;
408                 }
409                 FTPFile[] newfiles = listFiles(completePath, false);
410 
411                 if (newfiles == null) {
412                     ftp.changeToParentDirectory();
413                     return;
414                 }
415                 for (int i = 0; i < newfiles.length; i++) {
416                     FTPFile file = newfiles[i];
417                     if (file != null
418                         && !file.getName().equals(".")
419                         && !file.getName().equals("..")) {
420                         String name = vpath + file.getName();
421                         scannedDirs.put(name, new FTPFileProxy(file));
422                         if (isFunctioningAsDirectory(ftp, dir, file)) {
423                             boolean slowScanAllowed = true;
424                             if (!isFollowSymlinks() && file.isSymbolicLink()) {
425                                 dirsExcluded.addElement(name);
426                                 slowScanAllowed = false;
427                             } else if (isIncluded(name)) {
428                                 accountForIncludedDir(name,
429                                                       new AntFTPFile(ftp, file, completePath) , fast);
430                             } else {
431                                 dirsNotIncluded.addElement(name);
432                                 if (fast && couldHoldIncluded(name)) {
433                                     scandir(file.getName(),
434                                             name + File.separator, fast);
435                                 }
436                             }
437                             if (!fast && slowScanAllowed) {
438                                 scandir(file.getName(),
439                                         name + File.separator, fast);
440                             }
441                         } else {
442                             if (!isFollowSymlinks() && file.isSymbolicLink()) {
443                                 filesExcluded.addElement(name);
444                             } else if (isFunctioningAsFile(ftp, dir, file)) {
445                                 accountForIncludedFile(name);
446                             }
447                         }
448                     }
449                 }
450                 ftp.changeToParentDirectory();
451             } catch (IOException e) {
452                 throw new BuildException("Error while communicating with FTP "
453                                          + "server: ", e);
454             }
455         }
456         /**
457          * process included file
458          * @param name  path of the file relative to the directory of the fileset
459          */
460         private void accountForIncludedFile(String name) {
461             if (!filesIncluded.contains(name)
462                 && !filesExcluded.contains(name)) {
463 
464                 if (isIncluded(name)) {
465                     if (!isExcluded(name)
466                         && isSelected(name, (File) scannedDirs.get(name))) {
467                         filesIncluded.addElement(name);
468                     } else {
469                         filesExcluded.addElement(name);
470                     }
471                 } else {
472                     filesNotIncluded.addElement(name);
473                 }
474             }
475         }

478          *

480          * the fileset
481          * @param file directory as file
482          * @param fast
483          */
484         private void accountForIncludedDir(String name, AntFTPFile file, boolean fast) {
485             if (!dirsIncluded.contains(name)
486                 && !dirsExcluded.contains(name)) {
487 
488                 if (!isExcluded(name)) {
489                     if (fast) {
490                         if (file.isSymbolicLink()) {
491                             try {
492                                 file.getClient().changeWorkingDirectory(file.curpwd);
493                             } catch (IOException ioe) {
494                                 throw new BuildException("could not change directory to curpwd");
495                             }
496                             scandir(file.getLink(),
497                                     name + File.separator, fast);
498                         } else {
499                             try {
500                                 file.getClient().changeWorkingDirectory(file.curpwd);
501                             } catch (IOException ioe) {
502                                 throw new BuildException("could not change directory to curpwd");
503                             }
504                             scandir(file.getName(),
505                                     name + File.separator, fast);
506                         }
507                     }
508                     dirsIncluded.addElement(name);
509                 } else {
510                     dirsExcluded.addElement(name);
511                     if (fast && couldHoldIncluded(name)) {
512                         try {
513                             file.getClient().changeWorkingDirectory(file.curpwd);
514                         } catch (IOException ioe) {
515                             throw new BuildException("could not change directory to curpwd");
516                         }
517                         scandir(file.getName(),
518                                 name + File.separator, fast);
519                     }
520                 }
521             }

556         /**

558          * @param directory full path on the remote side
559          * @param changedir if true change to directory directory before listing
560          * @return array of FTPFile
561          */
562         public FTPFile[] listFiles(String directory, boolean changedir) {
563             //task.log("listing files in directory " + directory, Project.MSG_DEBUG);
564             String currentPath = directory;
565             if (changedir) {
566                 try {
567                     boolean result = ftp.changeWorkingDirectory(directory);
568                     if (!result) {
569                         return null;
570                     }
571                     currentPath = ftp.printWorkingDirectory();
572                 } catch (IOException ioe) {
573                     throw new BuildException(ioe, task.getLocation());
574                 }
575             }
576             if (fileListMap.containsKey(currentPath)) {
577                 task.log("filelist map used in listing files", Project.MSG_DEBUG);
578                 return ((FTPFile[]) fileListMap.get(currentPath));
579             }
580             FTPFile[] result = null;
581             try {
582                 result = ftp.listFiles();
583             } catch (IOException ioe) {
584                 throw new BuildException(ioe, task.getLocation());
585             }
586             fileListMap.put(currentPath, result);
587             if (!remoteSensitivityChecked) {
588                 checkRemoteSensitivity(result, directory);
589             }
590             return result;
591         }

611         private void checkRemoteSensitivity(FTPFile[] array, String directory) {
612             if (array == null) {
613                 return;
614             }
615             boolean candidateFound = false;
616             String target = null;
617             for (int icounter = 0; icounter < array.length; icounter++) {
618                 if (array[icounter] != null && array[icounter].isDirectory()) {
619                     if (!array[icounter].getName().equals(".")
620                         && !array[icounter].getName().equals("..")) {
621                         candidateFound = true;
622                         target = fiddleName(array[icounter].getName());
623                         task.log("will try to cd to "
624                                          + target + " where a directory called " + array[icounter].getName()
625                                          + " exists", Project.MSG_DEBUG);
626                         for (int pcounter = 0; pcounter < array.length; pcounter++) {
627                             if (array[pcounter] != null
628                                 && pcounter != icounter
629                                 && target.equals(array[pcounter].getName())) {
630                                 candidateFound = false;
631                                 break;
632                             }
633                         }
634                         if (candidateFound) {
635                             break;
636                         }
637                     }
638                 }
639             }
640             if (candidateFound) {
641                 try {
642                     task.log("testing case sensitivity, attempting to cd to "
643                                      + target, Project.MSG_DEBUG);
644                     remoteSystemCaseSensitive  = !ftp.changeWorkingDirectory(target);
645                 } catch (IOException ioe) {
646                     remoteSystemCaseSensitive = true;
647                 } finally {
648                     try {
649                         ftp.changeWorkingDirectory(directory);
650                     } catch (IOException ioe) {
651                         throw new BuildException(ioe, task.getLocation());
652                     }
653                 }
654                 task.log("remote system is case sensitive : "
655                                       + remoteSystemCaseSensitive,
656                                       Project.MSG_VERBOSE);
657                 remoteSensitivityChecked = true;
658             }
659         }
660         private String fiddleName(String origin) {
661             StringBuffer result = new StringBuffer();
662             for (int icounter = 0; icounter < origin.length(); icounter++) {
663                 if (Character.isLowerCase(origin.charAt(icounter))) {
664                     result.append(Character.toUpperCase(origin.charAt(icounter)));
665                 } else if (Character.isUpperCase(origin.charAt(icounter))) {
666                     result.append(Character.toLowerCase(origin.charAt(icounter)));
667                 } else {
668                     result.append(origin.charAt(icounter));
669                 }
670             }
671             return result.toString();
672         }
673         /**

677         protected class AntFTPFile {
678             /**
679              * ftp client
680              */
681             private FTPClient client;
682             /**
683              * parent directory of the file
684              */
685             private String curpwd;
686             /**
687              * the file itself
688              */
689             private FTPFile ftpFile;
690             /**
691              *
692              */
693             private AntFTPFile parent = null;
694             private boolean relativePathCalculated = false;
695             private boolean traversesSymlinks = false;
696             private String relativePath = "";
697             /**
698              * constructor
699              * @param client ftp client variable
700              * @param ftpFile the file
701              * @param curpwd absolute remote path where the file is found
702              */
703             public AntFTPFile(FTPClient client, FTPFile ftpFile, String curpwd) {
704                 this.client = client;
705                 this.ftpFile = ftpFile;
706                 this.curpwd = curpwd;
707             }
708             /**
709              * other constructor
710              * @param parent the parent file
711              * @param path  a relative path to the parent file
712              */
713             public AntFTPFile(AntFTPFile parent, String path) {
714                 this.parent = parent;
715                 this.client = parent.client;
716                 Vector pathElements = SelectorUtils.tokenizePath(path);
717                 try {
718                     boolean result = this.client.changeWorkingDirectory(parent.getAbsolutePath());
719                     //this should not happen, except if parent has been deleted by another process
720                     if (!result) {
721                         return;
722                     }
723                     this.curpwd = parent.getAbsolutePath();
724                 } catch (IOException ioe) {
725                     throw new BuildException("could not change working dir to "
726                                              + parent.curpwd);
727                 }
728                 final int size = pathElements.size();
729                 for (int fcount = 0; fcount < size - 1; fcount++) {
730                     String currentPathElement = (String) pathElements.elementAt(fcount);
731                     try {
732                         boolean result = this.client.changeWorkingDirectory(currentPathElement);
733                         if (!result && !isCaseSensitive()
734                             && (remoteSystemCaseSensitive || !remoteSensitivityChecked)) {
735                             currentPathElement = findPathElementCaseUnsensitive(this.curpwd,
736                                                                                 currentPathElement);
737                             if (currentPathElement == null) {
738                                 return;
739                             }
740                         } else if (!result) {
741                             return;
742                         }
743                         this.curpwd = getCurpwdPlusFileSep()
744                             + currentPathElement;
745                     } catch (IOException ioe) {
746                         throw new BuildException("could not change working dir to "
747                                                  + (String) pathElements.elementAt(fcount)
748                                                  + " from " + this.curpwd);
749                     }
750 
751                 }
752                 String lastpathelement = (String) pathElements.elementAt(size - 1);
753                 FTPFile [] theFiles = listFiles(this.curpwd);
754                 this.ftpFile = getFile(theFiles, lastpathelement);
755             }

1105      * For each file in the fileset, do the appropriate action: send, get,
1106      * delete, or list.
1107      *
1108      * @param ftp the FTPClient instance used to perform FTP actions
1109      * @param fs the fileset on which the actions are performed.
1110      *
1111      * @return the number of files to be transferred.
1112      *
1113      * @throws IOException if there is a problem reading a file
1114      * @throws BuildException if there is a problem in the configuration.
1115      */
1116     protected int transferFiles(final FTPClient ftp, FileSet fs)
1117         throws IOException, BuildException {
1118         DirectoryScanner ds;
1119         if (task.getAction() == FTPTask.SEND_FILES) {
1120             ds = fs.getDirectoryScanner(task.getProject());
1121         } else {
1122             ds = new FTPDirectoryScanner(ftp);
1123             fs.setupDirectoryScanner(ds, task.getProject());
1124             ds.setFollowSymlinks(fs.isFollowSymlinks());
1125             ds.scan();
1126         }
1127 
1128         String[] dsfiles = null;
1129         if (task.getAction() == FTPTask.RM_DIR) {
1130             dsfiles = ds.getIncludedDirectories();
1131         } else {
1132             dsfiles = ds.getIncludedFiles();
1133         }
1134         String dir = null;
1135 
1136         if ((ds.getBasedir() == null)
1137             && ((task.getAction() == FTPTask.SEND_FILES) || (task.getAction() == FTPTask.GET_FILES))) {
1138             throw new BuildException("the dir attribute must be set for send "
1139                                      + "and get actions");
1140         } else {
1141             if ((task.getAction() == FTPTask.SEND_FILES) || (task.getAction() == FTPTask.GET_FILES)) {
1142                 dir = ds.getBasedir().getAbsolutePath();
1143             }
1144         }
1145 
1146         // If we are doing a listing, we need the output stream created now.
1147         BufferedWriter bw = null;
1148 
1149         try {
1150             if (task.getAction() == FTPTask.LIST_FILES) {
1151                 File pd = task.getListing().getParentFile();
1152 
1153                 if (!pd.exists()) {
1154                     pd.mkdirs();
1155                 }
1156                 bw = new BufferedWriter(new FileWriter(task.getListing()));
1157             }
1158             RetryHandler h = new RetryHandler(task.getRetriesAllowed(), task);
1159             if (task.getAction() == FTPTask.RM_DIR) {
1160                 // to remove directories, start by the end of the list
1161                 // the trunk does not let itself be removed before the leaves
1162                 for (int i = dsfiles.length - 1; i >= 0; i--) {
1163                     final String dsfile = dsfiles[i];
1164                     executeRetryable(h, new Retryable() {
1165                             public void execute() throws IOException {
1166                                 rmDir(ftp, dsfile);
1167                             }
1168                         }, dsfile);
1169                 }
1170             } else {
1171                 final BufferedWriter fbw = bw;
1172                 final String fdir = dir;
1173                 if (task.isNewer()) {
1174                     task.setGranularityMillis(task.getTimestampGranularity()
1175                                               .getMilliseconds(task.getAction()));
1176                 }
1177                 for (int i = 0; i < dsfiles.length; i++) {
1178                     final String dsfile = dsfiles[i];
1179                     executeRetryable(h, new Retryable() {
1180                             public void execute() throws IOException {
1181                                 switch (task.getAction()) {
1182                                 case FTPTask.SEND_FILES:
1183                                     sendFile(ftp, fdir, dsfile);
1184                                     break;
1185                                 case FTPTask.GET_FILES:
1186                                     getFile(ftp, fdir, dsfile);
1187                                     break;
1188                                 case FTPTask.DEL_FILES:
1189                                     delFile(ftp, dsfile);
1190                                     break;
1191                                 case FTPTask.LIST_FILES:
1192                                     listFile(ftp, fbw, dsfile);
1193                                     break;
1194                                 case FTPTask.CHMOD:
1195                                     doSiteCommand(ftp, "chmod " + task.getChmod() + " "
1196                                                   + resolveFile(dsfile));
1197                                     transferred++;
1198                                     break;
1199                                 default:
1200                                     throw new BuildException("unknown ftp action "
1201                                                              + task.getAction());
1202                                 }
1203                             }
1204                         }, dsfile);
1205                 }
1206             }
1207         } finally {
1208             if (bw != null) {
1209                 bw.close();
1210             }
1211         }
1212 
1213         return dsfiles.length;
1214     }

1288             return;
1289         }

1293

1318                 // check if dir exists by trying to change into it.

1330                 }

1332             }

1417     protected boolean isUpToDate(FTPClient ftp, File localFile,
1418                                  String remoteFile)
1419         throws IOException, BuildException {
1420         task.log("checking date for " + remoteFile, Project.MSG_VERBOSE);
1421 
1422         FTPFile[] files = ftp.listFiles(remoteFile);
1423 
1424         // For Microsoft's Ftp-Service an Array with length 0 is
1425         // returned if configured to return listings in "MS-DOS"-Format
1426         if (files == null || files.length == 0) {
1427             // If we are sending files, then assume out of date.
1428             // If we are getting files, then throw an error
1429 
1430             if (task.getAction() == FTPTask.SEND_FILES) {
1431                 task.log("Could not date test remote file: " + remoteFile
1432                          + "assuming out of date.", Project.MSG_VERBOSE);
1433                 return false;
1434             } else {
1435                 throw new BuildException("could not date test remote file: "
1436                                          + ftp.getReplyString());
1437             }
1438         }
1439 
1440         long remoteTimestamp = files[0].getTimestamp().getTime().getTime();
1441         long localTimestamp = localFile.lastModified();
1442         long adjustedRemoteTimestamp = remoteTimestamp + task.getTimeDiffMillis()
1443             + task.getGranularityMillis();
1444 
1445         StringBuffer msg;
1446         synchronized(TIMESTAMP_LOGGING_SDF) {
1447             msg = new StringBuffer("   [")
1448                 .append(TIMESTAMP_LOGGING_SDF.format(new Date(localTimestamp)))
1449                 .append("] local");
1450         }
1451         task.log(msg.toString(), Project.MSG_VERBOSE);
1452 
1453         synchronized(TIMESTAMP_LOGGING_SDF) {
1454             msg = new StringBuffer("   [")
1455                 .append(TIMESTAMP_LOGGING_SDF.format(new Date(adjustedRemoteTimestamp)))
1456                 .append("] remote");
1457         }
1458         if (remoteTimestamp != adjustedRemoteTimestamp) {
1459             synchronized(TIMESTAMP_LOGGING_SDF) {
1460                 msg.append(" - (raw: ")
1461                     .append(TIMESTAMP_LOGGING_SDF.format(new Date(remoteTimestamp)))
1462                     .append(")");
1463             }
1464         }
1465         task.log(msg.toString(), Project.MSG_VERBOSE);
1466 
1467         if (task.getAction() == FTPTask.SEND_FILES) {
1468             return adjustedRemoteTimestamp >= localTimestamp;
1469         } else {
1470             return localTimestamp >= adjustedRemoteTimestamp;
1471         }
1472     }

1506     /**
1507      * Sends a single file to the remote host. <code>filename</code> may
1508      * contain a relative path specification. When this is the case, <code>sendFile</code>
1509      * will attempt to create any necessary parent directories before sending
1510      * the file. The file will then be sent using the entire relative path
1511      * spec - no attempt is made to change directories. It is anticipated that
1512      * this may eventually cause problems with some FTP servers, but it
1513      * simplifies the coding.
1514      * @param ftp ftp client
1515      * @param dir base directory of the file to be sent (local)
1516      * @param filename relative path of the file to be send
1517      *        locally relative to dir
1518      *        remotely relative to the remotedir attribute
1519      * @throws IOException  in unknown circumstances
1520      * @throws BuildException in unknown circumstances
1521      */
1522     protected void sendFile(FTPClient ftp, String dir, String filename)
1523         throws IOException, BuildException {
1524         InputStream instream = null;
1525 
1526         try {
1527             // TODO - why not simply new File(dir, filename)?
1528             File file = task.getProject().resolveFile(new File(dir, filename).getPath());
1529 
1530             if (task.isNewer() && isUpToDate(ftp, file, resolveFile(filename))) {
1531                 return;
1532             }
1533 
1534             if (task.isVerbose()) {
1535                 task.log("transferring " + file.getAbsolutePath());
1536             }
1537 
1538             instream = new BufferedInputStream(new FileInputStream(file));
1539 
1540             createParents(ftp, filename);
1541 
1542             ftp.storeFile(resolveFile(filename), instream);
1543 
1544             boolean success = FTPReply.isPositiveCompletion(ftp.getReplyCode());
1545 
1546             if (!success) {
1547                 String s = "could not put file: " + ftp.getReplyString();
1548 
1549                 if (task.isSkipFailedTransfers()) {
1550                     task.log(s, Project.MSG_WARN);
1551                     skipped++;
1552                 } else {
1553                     throw new BuildException(s);
1554                 }
1555 
1556             } else {
1557                 // see if we should issue a chmod command
1558                 if (task.getChmod() != null) {
1559                     doSiteCommand(ftp, "chmod " + task.getChmod() + " "
1560                                   + resolveFile(filename));
1561                 }
1562                 task.log("File " + file.getAbsolutePath() + " copied to " + task.getServer(),
1563                          Project.MSG_VERBOSE);
1564                 transferred++;
1565             }
1566         } finally {
1567             if (instream != null) {
1568                 try {
1569                     instream.close();
1570                 } catch (IOException ex) {
1571                     // ignore it
1572                 }
1573             }
1574         }
1575     }

1639     /**
1640      * Retrieve a single file from the remote host. <code>filename</code> may
1641      * contain a relative path specification. <p>
1642      *
1643      * The file will then be retreived using the entire relative path spec -
1644      * no attempt is made to change directories. It is anticipated that this
1645      * may eventually cause problems with some FTP servers, but it simplifies
1646      * the coding.</p>
1647      * @param ftp the ftp client
1648      * @param dir local base directory to which the file should go back
1649      * @param filename relative path of the file based upon the ftp remote directory
1650      *        and/or the local base directory (dir)
1651      * @throws IOException  in unknown circumstances
1652      * @throws BuildException if skipFailedTransfers is false
1653      * and the file cannot be retrieved.
1654      */
1655     protected void getFile(FTPClient ftp, String dir, String filename)
1656         throws IOException, BuildException {
1657         OutputStream outstream = null;
1658         try {
1659             File file = task.getProject().resolveFile(new File(dir, filename).getPath());
1660 
1661             if (task.isNewer() && isUpToDate(ftp, file, resolveFile(filename))) {
1662                 return;
1663             }
1664 
1665             if (task.isVerbose()) {
1666                 task.log("transferring " + filename + " to "
1667                          + file.getAbsolutePath());
1668             }
1669 
1670             File pdir = file.getParentFile();
1671 
1672             if (!pdir.exists()) {
1673                 pdir.mkdirs();
1674             }
1675             outstream = new BufferedOutputStream(new FileOutputStream(file));
1676             ftp.retrieveFile(resolveFile(filename), outstream);
1677 
1678             if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1679                 String s = "could not get file: " + ftp.getReplyString();
1680 
1681                 if (task.isSkipFailedTransfers()) {
1682                     task.log(s, Project.MSG_WARN);
1683                     skipped++;
1684                 } else {
1685                     throw new BuildException(s);
1686                 }
1687 
1688             } else {
1689                 task.log(
1690                          "File " + file.getAbsolutePath() + " copied from "
1691                          + task.getServer(), Project.MSG_VERBOSE);
1692                 transferred++;
1693                 if (task.isPreserveLastModified()) {
1694                     outstream.close();
1695                     outstream = null;
1696                     FTPFile[] remote = ftp.listFiles(resolveFile(filename));
1697                     if (remote.length > 0) {
1698                         FILE_UTILS.setFileLastModified(file,
1699                                                        remote[0].getTimestamp()
1700                                                        .getTime().getTime());
1701                     }
1702                 }
1703             }
1704         } finally {
1705             if (outstream != null) {
1706                 try {
1707                     outstream.close();
1708                 } catch (IOException ex) {
1709                     // ignore it
1710                 }
1711             }
1712         }
1713     }

1745     /**
1746      * Create the specified directory on the remote host.
1747      *
1748      * @param ftp The FTP client connection
1749      * @param dir The directory to create (format must be correct for host
1750      *      type)
1751      * @throws IOException  in unknown circumstances
1752      * @throws BuildException if ignoreNoncriticalErrors has not been set to true
1753      *         and a directory could not be created, for instance because it was
1754      *         already existing. Precisely, the codes 521, 550 and 553 will trigger
1755      *         a BuildException
1756      */
1757     protected void makeRemoteDir(FTPClient ftp, String dir)
1758         throws IOException, BuildException {
1759         String workingDirectory = ftp.printWorkingDirectory();
1760         if (task.isVerbose()) {
1761             if (dir.startsWith("/") || workingDirectory == null) {
1762                 task.log("Creating directory: " + dir + " in /");
1763             } else {
1764                 task.log("Creating directory: " + dir + " in " + workingDirectory);
1765             }
1766         }
1767         if (dir.startsWith("/")) {
1768             ftp.changeWorkingDirectory("/");
1769         }
1770         String subdir = "";
1771         StringTokenizer st = new StringTokenizer(dir, "/");
1772         while (st.hasMoreTokens()) {
1773             subdir = st.nextToken();
1774             task.log("Checking " + subdir, Project.MSG_DEBUG);
1775             if (!ftp.changeWorkingDirectory(subdir)) {
1776                 if (!ftp.makeDirectory(subdir)) {
1777                     // codes 521, 550 and 553 can be produced by FTP Servers
1778                     //  to indicate that an attempt to create a directory has
1779                     //  failed because the directory already exists.
1780                     int rc = ftp.getReplyCode();
1781                     if (!(task.isIgnoreNoncriticalErrors() && (rc == CODE_550
1782                                                                || rc == CODE_553
1783                                                                || rc == CODE_521))) {
1784                         throw new BuildException("could not create directory: "
1785                                                  + ftp.getReplyString());
1786                     }
1787                     if (task.isVerbose()) {
1788                         task.log("Directory already exists");
1789                     }
1790                 } else {
1791                     if (task.isVerbose()) {
1792                         task.log("Directory created OK");
1793                     }
1794                     ftp.changeWorkingDirectory(subdir);
1795                 }
1796             }
1797         }
1798         if (workingDirectory != null) {
1799             ftp.changeWorkingDirectory(workingDirectory);
1800         }
1801     }

1820     public void doFTP() throws BuildException {
1821         FTPClient ftp = null;
1822 
1823         try {
1824             task.log("Opening FTP connection to " + task.getServer(), Project.MSG_VERBOSE);
1825 
1826             ftp = new FTPClient();
1827             if (task.isConfigurationSet()) {
1828                 ftp = FTPConfigurator.configure(ftp, task);
1829             }
1830 
1831             ftp.setRemoteVerificationEnabled(task.getEnableRemoteVerification());
1832             ftp.connect(task.getServer(), task.getPort());
1833             if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1834                 throw new BuildException("FTP connection failed: "
1835                                          + ftp.getReplyString());
1836             }
1837 
1838             task.log("connected", Project.MSG_VERBOSE);
1839             task.log("logging in to FTP server", Project.MSG_VERBOSE);
1840 
1841             if ((task.getAccount() != null && !ftp.login(task.getUserid(), task.getPassword(), task.getAccount()))
1842                 || (task.getAccount() == null && !ftp.login(task.getUserid(), task.getPassword()))) {
1843                 throw new BuildException("Could not login to FTP server");
1844             }
1845 
1846             task.log("login succeeded", Project.MSG_VERBOSE);
1847 
1848             if (task.isBinary()) {
1849                 ftp.setFileType(org.apache.commons.net.ftp.FTP.BINARY_FILE_TYPE);
1850                 if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1851                     throw new BuildException("could not set transfer type: "
1852                                              + ftp.getReplyString());
1853                 }
1854             } else {
1855                 ftp.setFileType(org.apache.commons.net.ftp.FTP.ASCII_FILE_TYPE);
1856                 if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1857                     throw new BuildException("could not set transfer type: "
1858                                              + ftp.getReplyString());
1859                 }
1860             }
1861 
1862             if (task.isPassive()) {
1863                 task.log("entering passive mode", Project.MSG_VERBOSE);
1864                 ftp.enterLocalPassiveMode();
1865                 if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1866                     throw new BuildException("could not enter into passive "
1867                                              + "mode: " + ftp.getReplyString());
1868                 }
1869             }
1870 
1871             // If an initial command was configured then send it.
1872             // Some FTP servers offer different modes of operation,
1873             // E.G. switching between a UNIX file system mode and
1874             // a legacy file system.
1875             if (task.getInitialSiteCommand() != null) {
1876                 RetryHandler h = new RetryHandler(task.getRetriesAllowed(), task);
1877                 final FTPClient lftp = ftp;
1878                 executeRetryable(h, new Retryable() {
1879                         public void execute() throws IOException {
1880                             doSiteCommand(lftp, task.getInitialSiteCommand());
1881                         }
1882                     }, "initial site command: " + task.getInitialSiteCommand());
1883             }
1884 
1885 
1886             // For a unix ftp server you can set the default mask for all files
1887             // created.
1888 
1889             if (task.getUmask() != null) {
1890                 RetryHandler h = new RetryHandler(task.getRetriesAllowed(), task);
1891                 final FTPClient lftp = ftp;
1892                 executeRetryable(h, new Retryable() {
1893                         public void execute() throws IOException {
1894                             doSiteCommand(lftp, "umask " + task.getUmask());
1895                         }
1896                     }, "umask " + task.getUmask());
1897             }
1898 
1899             // If the action is MK_DIR, then the specified remote
1900             // directory is the directory to create.
1901 
1902             if (task.getAction() == FTPTask.MK_DIR) {
1903                 RetryHandler h = new RetryHandler(task.getRetriesAllowed(), task);
1904                 final FTPClient lftp = ftp;
1905                 executeRetryable(h, new Retryable() {
1906                         public void execute() throws IOException {
1907                             makeRemoteDir(lftp, task.getRemotedir());
1908                         }
1909                     }, task.getRemotedir());
1910             } else if (task.getAction() == FTPTask.SITE_CMD) {
1911                 RetryHandler h = new RetryHandler(task.getRetriesAllowed(), task);
1912                 final FTPClient lftp = ftp;
1913                 executeRetryable(h, new Retryable() {
1914                         public void execute() throws IOException {
1915                             doSiteCommand(lftp, task.getSiteCommand());
1916                         }
1917                     }, "Site Command: " + task.getSiteCommand());
1918             } else {
1919                 if (task.getRemotedir() != null) {
1920                     task.log("changing the remote directory", Project.MSG_VERBOSE);
1921                     ftp.changeWorkingDirectory(task.getRemotedir());
1922                     if (!FTPReply.isPositiveCompletion(ftp.getReplyCode())) {
1923                         throw new BuildException("could not change remote "
1924                                                  + "directory: " + ftp.getReplyString());
1925                     }
1926                 }
1927                 if (task.isNewer() && task.isTimeDiffAuto()) {
1928                     // in this case we want to find how much time span there is between local
1929                     // and remote
1930                     task.setTimeDiffMillis(getTimeDiff(ftp));
1931                 }
1932                 task.log(FTPTask.ACTION_STRS[task.getAction()] + " " + FTPTask.ACTION_TARGET_STRS[task.getAction()]);
1933                 transferFiles(ftp);
1934             }
1935 
1936         } catch (IOException ex) {
1937             throw new BuildException("error during FTP transfer: " + ex, ex);
1938         } finally {
1939             if (ftp != null && ftp.isConnected()) {
1940                 try {
1941                     task.log("disconnecting", Project.MSG_VERBOSE);
1942                     ftp.logout();
1943                     ftp.disconnect();
1944                 } catch (IOException ex) {
1945                     // ignore it
1946                 }
1947             }
1948         }
1949     }
