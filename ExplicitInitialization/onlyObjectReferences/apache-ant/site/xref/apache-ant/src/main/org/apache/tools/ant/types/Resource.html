20  import java.io.InputStream;
21  import java.io.IOException;
22  import java.io.OutputStream;
23  import java.math.BigInteger;
24  import java.util.Iterator;
25  import java.util.NoSuchElementException;

27  import org.apache.tools.ant.types.resources.FileProvider;

29  /**
30   * Describes a "File-like" resource (File, ZipEntry, etc.).
31   *
32   * This class is meant to be used by classes needing to record path
33   * and date/time information about a file, a zip entry or some similar
34   * resource (URL, archive in a version control repository, ...).
35   *
36   * @since Ant 1.5.2
37   * @see org.apache.tools.ant.types.resources.Touchable
38   */
39  public class Resource extends DataType implements Comparable<Resource>, ResourceCollection {
40  
41      /** Constant unknown size */
42      public static final long UNKNOWN_SIZE = -1;
43  
44      /** Constant unknown datetime for getLastModified */
45      public static final long UNKNOWN_DATETIME = 0L;
46  
47      /** Magic number */
48      protected static final int MAGIC = getMagicNumber("Resource".getBytes());
49  
50      private static final int NULL_NAME = getMagicNumber("null name".getBytes());
51  
52      /**
53       * Create a "magic number" for use in hashCode calculations.
54       * @param seed byte[] to seed with.
55       * @return a magic number as int.
56       */
57      protected static int getMagicNumber(byte[] seed) {
58          return new BigInteger(seed).intValue();
59      }
60  
61      private String name = null;
62      private Boolean exists = null;
63      private Long lastmodified = null;
64      private Boolean directory = null;
65      private Long size = null;
66  
67      /**
68       * Default constructor.
69       */
70      public Resource() {
71      }
72  
73      /**
74       * Only sets the name.
75       *
76       * <p>This is a dummy, used for not existing resources.</p>
77       *
78       * @param name relative path of the resource.  Expects
79       * &quot;/&quot; to be used as the directory separator.
80       */
81      public Resource(String name) {
82          this(name, false, 0, false);
83      }
84  
85      /**
86       * Sets the name, lastmodified flag, and exists flag.
87       *
88       * @param name relative path of the resource.  Expects
89       * &quot;/&quot; to be used as the directory separator.
90       * @param exists if true, this resource exists.
91       * @param lastmodified the last modification time of this resource.
92       */
93      public Resource(String name, boolean exists, long lastmodified) {
94          this(name, exists, lastmodified, false);
95      }
96  
97      /**
98       * Sets the name, lastmodified flag, exists flag, and directory flag.
99       *
100      * @param name relative path of the resource.  Expects
101      * &quot;/&quot; to be used as the directory separator.
102      * @param exists if true the resource exists
103      * @param lastmodified the last modification time of the resource
104      * @param directory    if true, this resource is a directory
105      */
106     public Resource(String name, boolean exists, long lastmodified, boolean directory) {
107         this(name, exists, lastmodified, directory, UNKNOWN_SIZE);
108     }
109 
110     /**
111      * Sets the name, lastmodified flag, exists flag, directory flag, and size.
112      *
113      * @param name relative path of the resource.  Expects
114      * &quot;/&quot; to be used as the directory separator.
115      * @param exists if true the resource exists
116      * @param lastmodified the last modification time of the resource
117      * @param directory    if true, this resource is a directory
118      * @param size the size of this resource.
119      */
120     public Resource(String name, boolean exists, long lastmodified, boolean directory, long size) {
121         this.name = name;
122         setName(name);
123         setExists(exists);
124         setLastModified(lastmodified);
125         setDirectory(directory);
126         setSize(size);
127     }
128 
129     /**
130      * Name attribute will contain the path of a file relative to the
131      * root directory of its fileset or the recorded path of a zip
132      * entry.
133      *
134      * <p>example for a file with fullpath /var/opt/adm/resource.txt
135      * in a file set with root dir /var/opt it will be
136      * adm/resource.txt.</p>
137      *
138      * <p>&quot;/&quot; will be used as the directory separator.</p>
139      * @return the name of this resource.
140      */
141     public String getName() {
142         return isReference() ? ((Resource) getCheckedRef()).getName() : name;
143     }
144 
145     /**
146      * Set the name of this Resource.
147      * @param name relative path of the resource.  Expects
148      * &quot;/&quot; to be used as the directory separator.
149      */
150     public void setName(String name) {
151         checkAttributesAllowed();
152         this.name = name;
153     }
154 
155     /**
156      * The exists attribute tells whether a resource exists.
157      * @return true if this resource exists.
158      */
159     public boolean isExists() {
160         if (isReference()) {
161             return ((Resource) getCheckedRef()).isExists();
162         }
163         //default true:
164         return exists == null || exists.booleanValue();
165     }
166 
167     /**
168      * Set the exists attribute.
169      * @param exists if true, this resource exists.
170      */
171     public void setExists(boolean exists) {
172         checkAttributesAllowed();
173         this.exists = exists ? Boolean.TRUE : Boolean.FALSE;
174     }
175 
176     /**
177      * Tells the modification time in milliseconds since 01.01.1970 (the "epoch").
178      *
179      * @return the modification time, if that is meaningful
180      *            (e.g. for a file resource which exists);
181      *         0 if the resource does not exist, to mirror the behavior
182      *         of {@link java.io.File#lastModified};
183      *         or 0 if the notion of modification time is meaningless for this class
184      *           of resource (e.g. an inline string)
185      */
186     public long getLastModified() {
187         if (isReference()) {
188             return ((Resource) getCheckedRef()).getLastModified();
189         }
190         if (!isExists() || lastmodified == null) {
191             return UNKNOWN_DATETIME;
192         }
193         long result = lastmodified.longValue();
194         return result < UNKNOWN_DATETIME ? UNKNOWN_DATETIME : result;
195     }
196 
197     /**
198      * Set the last modification attribute.
199      * @param lastmodified the modification time in milliseconds since 01.01.1970.
200      */
201     public void setLastModified(long lastmodified) {
202         checkAttributesAllowed();
203         this.lastmodified = new Long(lastmodified);
204     }
205 
206     /**
207      * Tells if the resource is a directory.
208      * @return boolean flag indicating if the resource is a directory.
209      */
210     public boolean isDirectory() {
211         if (isReference()) {
212             return ((Resource) getCheckedRef()).isDirectory();
213         }
214         //default false:
215         return directory != null && directory.booleanValue();
216     }
217 
218     /**
219      * Set the directory attribute.
220      * @param directory if true, this resource is a directory.
221      */
222     public void setDirectory(boolean directory) {
223         checkAttributesAllowed();
224         this.directory = directory ? Boolean.TRUE : Boolean.FALSE;
225     }
226 
227     /**
228      * Set the size of this Resource.
229      * @param size the size, as a long.
230      * @since Ant 1.6.3
231      */
232     public void setSize(long size) {
233         checkAttributesAllowed();
234         this.size = new Long(size > UNKNOWN_SIZE ? size : UNKNOWN_SIZE);
235     }
236 
237     /**
238      * Get the size of this Resource.
239      * @return the size, as a long, 0 if the Resource does not exist (for
240      *         compatibility with java.io.File), or UNKNOWN_SIZE if not known.
241      * @since Ant 1.6.3
242      */
243     public long getSize() {
244         if (isReference()) {
245             return ((Resource) getCheckedRef()).getSize();
246         }
247         return isExists()
248             ? (size != null ? size.longValue() : UNKNOWN_SIZE)
249             : 0L;
250     }
251 
252     /**
253      * Clone this Resource.
254      * @return copy of this.
255      */
256     public Object clone() {
257         try {
258             return super.clone();
259         } catch (CloneNotSupportedException e) {
260             throw new UnsupportedOperationException(
261                     "CloneNotSupportedException for a Resource caught. "
262                     + "Derived classes must support cloning.");
263         }
264     }
265 
266     /**
267      * Delegates to a comparison of names.
268      * @param other the object to compare to.
269      * @return a negative integer, zero, or a positive integer as this Resource
270      *         is less than, equal to, or greater than the specified Resource.
271      * @since Ant 1.6
272      */
273     public int compareTo(Resource other) {
274         if (isReference()) {
275             return ((Resource) getCheckedRef()).compareTo(other);
276         }
277         return toString().compareTo(other.toString());
278     }
279 
280     /**
281      * Implement basic Resource equality.
282      * @param other the object to check against.
283      * @return true if the specified Object is equal to this Resource.
284      * @since Ant 1.7
285      */
286     public boolean equals(Object other) {
287         if (isReference()) {
288             return getCheckedRef().equals(other);
289         }
290         return other != null && other.getClass().equals(getClass())
291             && compareTo((Resource) other) == 0;
292     }
293 
294     /**
295      * Get the hash code for this Resource.
296      * @return hash code as int.
297      * @since Ant 1.7
298      */
299     public int hashCode() {
300         if (isReference()) {
301             return getCheckedRef().hashCode();
302         }
303         String name = getName();
304         return MAGIC * (name == null ? NULL_NAME : name.hashCode());
305     }
306 
307     /**
308      * Get an InputStream for the Resource.
309      * @return an InputStream containing this Resource's content.
310      * @throws IOException if unable to provide the content of this
311      *         Resource as a stream.
312      * @throws UnsupportedOperationException if InputStreams are not
313      *         supported for this Resource type.
314      * @since Ant 1.7
315      */
316     public InputStream getInputStream() throws IOException {
317         if (isReference()) {
318             return ((Resource) getCheckedRef()).getInputStream();
319         }
320         throw new UnsupportedOperationException();
321     }
322 
323     /**
324      * Get an OutputStream for the Resource.
325      * @return an OutputStream to which content can be written.
326      * @throws IOException if unable to provide the content of this
327      *         Resource as a stream.
328      * @throws UnsupportedOperationException if OutputStreams are not
329      *         supported for this Resource type.
330      * @since Ant 1.7
331      */
332     public OutputStream getOutputStream() throws IOException {
333         if (isReference()) {
334             return ((Resource) getCheckedRef()).getOutputStream();
335         }
336         throw new UnsupportedOperationException();
337     }
338 
339     /**
340      * Fulfill the ResourceCollection contract.
341      * @return an Iterator of Resources.
342      * @since Ant 1.7
343      */
344     public Iterator<Resource> iterator() {
345         return isReference() ? ((Resource) getCheckedRef()).iterator()
346             : new Iterator<Resource>() {
347             private boolean done = false;
348             public boolean hasNext() {
349                 return !done;
350             }
351             public Resource next() {
352                 if (done) {
353                     throw new NoSuchElementException();
354                 }
355                 done = true;
356                 return Resource.this;
357             }
358             public void remove() {
359                 throw new UnsupportedOperationException();
360             }
361         };
362     }
363 
364     /**
365      * Fulfill the ResourceCollection contract.
366      * @return the size of this ResourceCollection.
367      * @since Ant 1.7
368      */
369     public int size() {
370         return isReference() ? ((Resource) getCheckedRef()).size() : 1;
371     }
372 
373     /**
374      * Fulfill the ResourceCollection contract.
375      * @return whether this Resource is a FileProvider.
376      * @since Ant 1.7
377      */
378     public boolean isFilesystemOnly() {
379         return (isReference() && ((Resource) getCheckedRef()).isFilesystemOnly())
380             || this.as(FileProvider.class) != null;
381     }
382 
383     /**
384      * Get the string representation of this Resource.
385      * @return this Resource formatted as a String.
386      * @since Ant 1.7
387      */
388     public String toString() {
389         if (isReference()) {
390             return getCheckedRef().toString();
391         }
392         String n = getName();
393         return n == null ? "(anonymous)" : n;
394     }
395 
396     /**
397      * Get a long String representation of this Resource.
398      * This typically should be the value of <code>toString()</code>
399      * prefixed by a type description.
400      * @return this Resource formatted as a long String.
401      * @since Ant 1.7
402      */
403     public final String toLongString() {
404         return isReference() ? ((Resource) getCheckedRef()).toLongString()
405             : getDataTypeName() + " \"" + toString() + '"';
406     }
407 
408     /**
409      * Overrides the base version.
410      * @param r the Reference to set.
411      */
412     public void setRefid(Reference r) {
413         if (name != null
414             || exists != null
415             || lastmodified != null
416             || directory != null
417             || size != null) {
418             throw tooManyAttributes();
419         }
420         super.setRefid(r);
421     }
422 
423     /**
424      * Returns a view of this resource that implements the interface
425      * given as the argument or null if there is no such view.
426      *
427      * <p>This allows extension interfaces to be added to resources
428      * without growing the number of permutations of interfaces
429      * decorators/adapters need to implement.</p>
430      *
431      * <p>This implementation of the method will return the current
432      * instance itself if it can be assigned to the given class.</p>
433      *
434      * @since Ant 1.8.0
435      */
436     public <T> T as(Class<T> clazz) {
437         return clazz.isAssignableFrom(getClass()) ? clazz.cast(this) : null;
438     }
