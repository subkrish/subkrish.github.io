62      /**
63       * Construct a new <code>Files</code> collection, shallowly cloned
64       * from the specified <code>Files</code>.
65       * @param f the <code>Files</code> to use as a template.
66       */
67      protected Files(Files f) {
68          this.defaultPatterns = f.defaultPatterns;
69          this.additionalPatterns = f.additionalPatterns;
70          this.useDefaultExcludes = f.useDefaultExcludes;
71          this.caseSensitive = f.caseSensitive;
72          this.followSymlinks = f.followSymlinks;
73          this.ds = f.ds;
74          setProject(f.getProject());
75      }

77      /**
78       * Make this instance in effect a reference to another instance.
79       *
80       * <p>You must not set another attribute or nest elements inside
81       * this element if you make it a reference.</p>
82       * @param r the <code>Reference</code> to use.
83       * @throws BuildException if there is a problem.
84       */
85      public void setRefid(Reference r) throws BuildException {
86          if (hasPatterns(defaultPatterns)) {
87              throw tooManyAttributes();
88          }
89          if (!additionalPatterns.isEmpty()) {
90              throw noChildrenAllowed();
91          }
92          if (hasSelectors()) {
93              throw noChildrenAllowed();
94          }
95          super.setRefid(r);
96      }

309     /**
310      * Fulfill the ResourceCollection contract.
311      * @return an Iterator of Resources.
312      */
313     public synchronized Iterator<Resource> iterator() {
314         if (isReference()) {
315             return getRef().iterator();
316         }
317         ensureDirectoryScannerSetup();
318         ds.scan();
319         int fct = ds.getIncludedFilesCount();
320         int dct = ds.getIncludedDirsCount();
321         if (fct + dct == 0) {
322             return EMPTY_ITERATOR;
323         }
324         FileResourceIterator result = new FileResourceIterator(getProject());
325         if (fct > 0) {
326             result.addFiles(ds.getIncludedFiles());
327         }
328         if (dct > 0) {
329             result.addFiles(ds.getIncludedDirectories());
330         }
331         return result;
332     }

334     /**
335      * Fulfill the ResourceCollection contract.
336      * @return number of elements as int.
337      */
338     public synchronized int size() {
339         if (isReference()) {
340             return getRef().size();
341         }
342         ensureDirectoryScannerSetup();
343         ds.scan();
344         return ds.getIncludedFilesCount() + ds.getIncludedDirsCount();
345     }

347     /**
348      * Find out whether this Files collection has patterns.
349      *
350      * @return whether any patterns are in this container.
351      */
352     public synchronized boolean hasPatterns() {
353         if (isReference()) {
354             return getRef().hasPatterns();
355         }
356         dieOnCircularReference();
357         if (hasPatterns(defaultPatterns)) {
358             return true;
359         }
360         for (PatternSet patternSet : additionalPatterns) {
361             if (hasPatterns(patternSet)) {
362                 return true;
363             }
364         }
365         return false;
366     }

381     /**
382      * Format this Files collection as a String.
383      * @return a descriptive <code>String</code>.
384      */
385     public String toString() {
386         if (isReference()) {
387             return getRef().toString();
388         }
389         Iterator<Resource> i = iterator();
390         if (!i.hasNext()) {
391             return "";
392         }
393         StringBuffer sb = new StringBuffer();
394         while (i.hasNext()) {
395             if (sb.length() > 0) {
396                 sb.append(File.pathSeparatorChar);
397             }
398             sb.append(i.next());
399         }
400         return sb.toString();
401     }

403     /**
404      * Create a deep clone of this instance, except for the nested selectors
405      * (the list of selectors is a shallow clone of this instance's list).
406      * @return a cloned Object.
407      */
408     public synchronized Object clone() {
409         if (isReference()) {
410             return getRef().clone();
411         }
412         Files f = (Files) super.clone();
413         f.defaultPatterns = (PatternSet) defaultPatterns.clone();
414         f.additionalPatterns = new Vector<PatternSet>(additionalPatterns.size());
415         for (PatternSet ps : additionalPatterns) {
416             f.additionalPatterns.add((PatternSet) ps.clone());
417         }
418         return f;
419     }

441     /**
442      * Get the merged patterns for this Files collection.
443      * @param p Project instance.
444      * @return the default patternset merged with the additional sets
445      * in a new PatternSet instance.
446      */
447     public synchronized PatternSet mergePatterns(Project p) {
448         if (isReference()) {
449             return getRef().mergePatterns(p);
450         }
451         dieOnCircularReference();
452         PatternSet ps = new PatternSet();
453         ps.append(defaultPatterns, p);
454         final int count = additionalPatterns.size();
455         for (int i = 0; i < count; i++) {
456             Object o = additionalPatterns.elementAt(i);
457             ps.append((PatternSet) o, p);
458         }
459         return ps;
460     }

480     private synchronized void ensureDirectoryScannerSetup() {
481         dieOnCircularReference();
482         if (ds == null) {
483             ds = new DirectoryScanner();
484             PatternSet ps = mergePatterns(getProject());
485             ds.setIncludes(ps.getIncludePatterns(getProject()));
486             ds.setExcludes(ps.getExcludePatterns(getProject()));
487             ds.setSelectors(getSelectors(getProject()));
488             if (useDefaultExcludes) {
489                 ds.addDefaultExcludes();
490             }
491             ds.setCaseSensitive(caseSensitive);
492             ds.setFollowSymlinks(followSymlinks);
493         }
494     }
495 
496     private boolean hasPatterns(PatternSet ps) {
497         String[] includePatterns = ps.getIncludePatterns(getProject());
498         String[] excludePatterns = ps.getExcludePatterns(getProject());
499         return (includePatterns != null && includePatterns.length > 0)
500             || (includePatterns != null && excludePatterns.length > 0);
501     }
