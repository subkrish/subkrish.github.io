200     /**
201      * Run the definition.
202      *
203      * @exception BuildException if an error occurs
204      */
205     public void execute() throws BuildException {
206         ClassLoader al = createLoader();
207 
208         if (!definerSet) {
209             //we arent fully defined yet. this is an error unless
210             //we are in an antlib, in which case the resource name is determined
211             //automatically.
212             //NB: URIs in the ant core package will be "" at this point.
213             if (getURI() == null) {
214                 throw new BuildException(
215                         "name, file or resource attribute of "
216                                 + getTaskName() + " is undefined",
217                         getLocation());
218             }
219 
220             if (getURI().startsWith(MagicNames.ANTLIB_PREFIX)) {
221                 //convert the URI to a resource
222                 String uri1 = getURI();
223                 setResource(makeResourceFromURI(uri1));
224             } else {
225                 throw new BuildException(
226                         "Only antlib URIs can be located from the URI alone,"
227                                 + " not the URI '" + getURI() + "'");
228             }
229         }
230 
231         if (name != null) {
232             if (classname == null) {
233                 throw new BuildException(
234                     "classname attribute of " + getTaskName() + " element "
235                     + "is undefined", getLocation());
236             }
237             addDefinition(al, name, classname);
238         } else {
239             if (classname != null) {
240                 String msg = "You must not specify classname "
241                     + "together with file or resource.";
242                 throw new BuildException(msg, getLocation());
243             }
244             final Enumeration<URL> urls;
245             if (file == null) {
246                 urls = resourceToURLs(al);
247             } else {
248                 final URL url = fileToURL();
249                 if (url == null) {
250                     return;
251                 }
252                 urls = Collections.enumeration(Collections.singleton(url));
253             }
254 
255             while (urls.hasMoreElements()) {
256                 URL url = urls.nextElement();
257 
258                 int fmt = this.format;
259                 if (url.toString().toLowerCase(Locale.ENGLISH).endsWith(".xml")) {
260                     fmt = Format.XML;
261                 }
262 
263                 if (fmt == Format.PROPERTIES) {
264                     loadProperties(al, url);
265                     break;
266                 } else {
267                     if (RESOURCE_STACK.get().get(url) != null) {
268                         log("Warning: Recursive loading of " + url
269                             + " ignored"
270                             + " at " + getLocation()
271                             + " originally loaded at "
272                             + RESOURCE_STACK.get().get(url),
273                             Project.MSG_WARN);
274                     } else {
275                         try {
276                             RESOURCE_STACK.get().put(url, getLocation());
277                             loadAntlib(al, url);
278                         } finally {
279                             RESOURCE_STACK.get().remove(url);
280                         }
281                     }
282                 }
283             }
284         }
285     }

287     /**
288      * This is where the logic to map from a URI to an antlib resource
289      * is kept.
290      * @param uri the xml namespace uri that to convert.
291      * @return the name of a resource. It may not exist
292      */
293 
294     public static String makeResourceFromURI(String uri) {
295         String path = uri.substring(MagicNames.ANTLIB_PREFIX.length());
296         String resource;
297         if (path.startsWith("//")) {
298             //handle new style full paths to an antlib, in which
299             //all but the forward slashes are allowed.
300             resource = path.substring("//".length());
301             if (!resource.endsWith(".xml")) {
302                 //if we haven't already named an XML file, it gets antlib.xml
303                 resource = resource + ANTLIB_XML;
304             }
305         } else {
306             //convert from a package to a path
307             resource = path.replace('.', '/') + ANTLIB_XML;
308         }
309         return resource;
310     }

312     /**
313      * Convert a file to a file: URL.
314      *
315      * @return the URL, or null if it isn't valid and the active error policy
316      * is not to raise a fault
317      * @throws BuildException if the file is missing/not a file and the
318      * policy requires failure at this point.
319      */
320     private URL fileToURL() {
321         String message = null;
322         if (!(file.exists())) {
323             message = "File " + file + " does not exist";
324         }
325         if (message == null && !(file.isFile())) {
326             message = "File " + file + " is not a file";
327         }
328         if (message == null) {
329             try {
330                 return FileUtils.getFileUtils().getFileURL(file);
331             } catch (Exception ex) {
332                 message =
333                     "File " + file + " cannot use as URL: "
334                     + ex.toString();
335             }
336         }
337         // Here if there is an error
338         switch (onError) {
339             case OnError.FAIL_ALL:
340                 throw new BuildException(message);
341             case OnError.FAIL:
342                 // Fall Through
343             case OnError.REPORT:
344                 log(message, Project.MSG_WARN);
345                 break;
346             case OnError.IGNORE:
347                 // log at a lower level
348                 log(message, Project.MSG_VERBOSE);
349                 break;
350             default:
351                 // Ignore the problem
352                 break;
353         }
354         return null;
355     }

357     private Enumeration<URL> resourceToURLs(ClassLoader classLoader) {
358         Enumeration<URL> ret;
359         try {
360             ret = classLoader.getResources(resource);
361         } catch (IOException e) {
362             throw new BuildException(
363                 "Could not fetch resources named " + resource,
364                 e, getLocation());
365         }
366         if (!ret.hasMoreElements()) {
367             String message = "Could not load definitions from resource "
368                 + resource + ". It could not be found.";
369             switch (onError) {
370                 case OnError.FAIL_ALL:
371                     throw new BuildException(message);
372                 case OnError.FAIL:
373                 case OnError.REPORT:
374                     log(message, Project.MSG_WARN);
375                     break;
376                 case OnError.IGNORE:
377                     log(message, Project.MSG_VERBOSE);
378                     break;
379                 default:
380                     // Ignore the problem
381                     break;
382             }
383         }
384         return ret;
385     }

387     /**
388      * Load type definitions as properties from a URL.
389      *
390      * @param al the classloader to use
391      * @param url the url to get the definitions from
392      */
393     protected void loadProperties(ClassLoader al, URL url) {
394         InputStream is = null;
395         try {
396             is = url.openStream();
397             if (is == null) {
398                 log("Could not load definitions from " + url,
399                     Project.MSG_WARN);
400                 return;
401             }
402             Properties props = new Properties();
403             props.load(is);
404             Enumeration<?> keys = props.keys();
405             while (keys.hasMoreElements()) {
406                 name = ((String) keys.nextElement());
407                 classname = props.getProperty(name);
408                 addDefinition(al, name, classname);
409             }
410         } catch (IOException ex) {
411             throw new BuildException(ex, getLocation());
412         } finally {
413             FileUtils.close(is);
414         }
415     }

417     /**
418      * Load an antlib from a URL.
419      *
420      * @param classLoader the classloader to use.
421      * @param url the url to load the definitions from.
422      */
423     private void loadAntlib(ClassLoader classLoader, URL url) {
424         try {
425             Antlib antlib = Antlib.createAntlib(getProject(), url, getURI());
426             antlib.setClassLoader(classLoader);
427             antlib.setURI(getURI());
428             antlib.execute();
429         } catch (BuildException ex) {
430             throw ProjectHelper.addLocationToBuildException(
431                 ex, getLocation());
432         }
433     }

461     /**
462      * Antlib attribute, sets resource and uri.
463      * uri is set the antlib value and, resource is set
464      * to the antlib.xml resource in the classpath.
465      * For example antlib="antlib:org.acme.bland.cola"
466      * corresponds to uri="antlib:org.acme.bland.cola"
467      * resource="org/acme/bland/cola/antlib.xml".
468      * ASF Bugzilla Bug 31999
469      * @param antlib the value to set.
470      */
471     public void setAntlib(String antlib) {
472         if (definerSet) {
473             tooManyDefinitions();
474         }
475         if (!antlib.startsWith("antlib:")) {
476             throw new BuildException(
477                 "Invalid antlib attribute - it must start with antlib:");
478         }
479         setURI(antlib);
480         this.resource = antlib.substring("antlib:".length()).replace('.', '/')
481             + "/antlib.xml";
482         definerSet = true;
483     }

563     /**
564      * Add a definition using the attributes of Definer
565      *
566      * @param al the ClassLoader to use
567      * @param name the name of the definition
568      * @param classname the classname of the definition
569      * @exception BuildException if an error occurs
570      */
571     protected void addDefinition(ClassLoader al, String name, String classname)
572         throws BuildException {
573         Class<?> cl = null;
574         try {
575             try {
576                 name = ProjectHelper.genComponentName(getURI(), name);
577 
578                 if (onError != OnError.IGNORE) {
579                     cl = Class.forName(classname, true, al);
580                 }
581 
582                 if (adapter != null) {
583                     adapterClass = Class.forName(adapter, true, al);
584                 }
585 
586                 if (adaptTo != null) {
587                     adaptToClass = Class.forName(adaptTo, true, al);
588                 }
589 
590                 AntTypeDefinition def = new AntTypeDefinition();
591                 def.setName(name);
592                 def.setClassName(classname);
593                 def.setClass(cl);
594                 def.setAdapterClass(adapterClass);
595                 def.setAdaptToClass(adaptToClass);
596                 def.setRestrict(restrict);
597                 def.setClassLoader(al);
598                 if (cl != null) {
599                     def.checkClass(getProject());
600                 }
601                 ComponentHelper.getComponentHelper(getProject())
602                         .addDataTypeDefinition(def);
603             } catch (ClassNotFoundException cnfe) {
604                 String msg = getTaskName() + " class " + classname
605                         + " cannot be found"
606                         + "\n using the classloader " + al;
607                 throw new BuildException(msg, cnfe, getLocation());
608             } catch (NoClassDefFoundError ncdfe) {
609                 String msg = getTaskName() + " A class needed by class "
610                         + classname + " cannot be found: " + ncdfe.getMessage()
611                         + "\n using the classloader " + al;
612                 throw new BuildException(msg, ncdfe, getLocation());
613             }
614         } catch (BuildException ex) {
615             switch (onError) {
616                 case OnError.FAIL_ALL:
617                 case OnError.FAIL:
618                     throw ex;
619                 case OnError.REPORT:
620                     log(ex.getLocation() + "Warning: " + ex.getMessage(),
621                         Project.MSG_WARN);
622                     break;
623                 default:
624                     log(ex.getLocation() + ex.getMessage(),
625                         Project.MSG_DEBUG);
626             }
627         }
628     }
