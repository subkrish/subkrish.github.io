65  /**
66   * Deletes a file or directory, or set of files defined by a fileset.
67   * The original delete task would delete a file, or a set of files
68   * using the include/exclude syntax.  The deltree task would delete a
69   * directory tree.  This task combines the functionality of these two
70   * originally distinct tasks.
71   * <p>Currently Delete extends MatchingTask.  This is intended <i>only</i>
72   * to provide backwards compatibility for a release.  The future position
73   * is to use nested filesets exclusively.</p>
74   *
75   * @since Ant 1.2
76   *
77   * @ant.task category="filesystem"
78   */
79  public class Delete extends MatchingTask {
80      private static final ResourceComparator REVERSE_FILESYSTEM = new Reverse(new FileSystem());
81      private static final ResourceSelector EXISTS = new Exists();
82  
83      private static class ReverseDirs implements ResourceCollection {
84          static final Comparator<Comparable<?>> REVERSE = new Comparator<Comparable<?>>() {
85              public int compare(Comparable<?> foo, Comparable<?> bar) {
86                  return ((Comparable) foo).compareTo(bar) * -1;
87              }
88          };
89          private Project project;
90          private File basedir;
91          private String[] dirs;
92          ReverseDirs(Project project, File basedir, String[] dirs) {
93              this.project = project;
94              this.basedir = basedir;
95              this.dirs = dirs;
96              Arrays.sort(this.dirs, REVERSE);
97          }
98          public Iterator<Resource> iterator() {
99              return new FileResourceIterator(project, basedir, dirs);
100         }
101         public boolean isFilesystemOnly() { return true; }
102         public int size() { return dirs.length; }
103     }

525     /**
526      * Delete the file(s).
527      * @exception BuildException if an error occurs
528      */
529     public void execute() throws BuildException {
530         if (usedMatchingTask) {
531             log("DEPRECATED - Use of the implicit FileSet is deprecated.  "
532                 + "Use a nested fileset element instead.", quiet ? Project.MSG_VERBOSE : verbosity);
533         }
534 
535         if (file == null && dir == null && filesets.size() == 0 && rcs == null) {
536             throw new BuildException("At least one of the file or dir "
537                                      + "attributes, or a nested resource collection, "
538                                      + "must be set.");
539         }
540 
541         if (quiet && failonerror) {
542             throw new BuildException("quiet and failonerror cannot both be "
543                                      + "set to true", getLocation());
544         }
545 
546         // delete the single file
547         if (file != null) {
548             if (file.exists()) {
549                 if (file.isDirectory()) {
550                     log("Directory " + file.getAbsolutePath()
551                         + " cannot be removed using the file attribute.  "
552                         + "Use dir instead.", quiet ? Project.MSG_VERBOSE : verbosity);
553                 } else {
554                     log("Deleting: " + file.getAbsolutePath());
555 
556                     if (!delete(file)) {
557                         handle("Unable to delete file " + file.getAbsolutePath());
558                     }
559                 }
560             } else if (isDanglingSymlink(file)) {
561                 log("Trying to delete file " + file.getAbsolutePath()
562                     + " which looks like a broken symlink.",
563                     quiet ? Project.MSG_VERBOSE : verbosity);
564                 if (!delete(file)) {
565                     handle("Unable to delete file " + file.getAbsolutePath());
566                 }
567             } else {
568                 log("Could not find file " + file.getAbsolutePath()
569                     + " to delete.", quiet ? Project.MSG_VERBOSE : verbosity);
570             }
571         }
572 
573         // delete the directory
574         if (dir != null && !usedMatchingTask) {
575             if (dir.exists() && dir.isDirectory()) {
576                 /*
577                   If verbosity is MSG_VERBOSE, that mean we are doing
578                   regular logging (backwards as that sounds).  In that
579                   case, we want to print one message about deleting the
580                   top of the directory tree.  Otherwise, the removeDir
581                   method will handle messages for _all_ directories.
582                 */
583                 if (verbosity == Project.MSG_VERBOSE) {
584                     log("Deleting directory " + dir.getAbsolutePath());
585                 }
586                 removeDir(dir);
587             } else if (isDanglingSymlink(dir)) {
588                 log("Trying to delete directory " + dir.getAbsolutePath()
589                     + " which looks like a broken symlink.",
590                     quiet ? Project.MSG_VERBOSE : verbosity);
591                 if (!delete(dir)) {
592                     handle("Unable to delete directory "
593                            + dir.getAbsolutePath());
594                 }
595             }
596         }
597         Resources resourcesToDelete = new Resources();
598         resourcesToDelete.setProject(getProject());
599         resourcesToDelete.setCache(true);
600         Resources filesetDirs = new Resources();
601         filesetDirs.setProject(getProject());
602         filesetDirs.setCache(true);
603         FileSet implicit = null;
604         if (usedMatchingTask && dir != null && dir.isDirectory()) {
605             //add the files from the default fileset:
606             implicit = getImplicitFileSet();
607             implicit.setProject(getProject());
608             filesets.add(implicit);
609         }
610 
611         final int size = filesets.size();
612         for (int i = 0; i < size; i++) {
613             FileSet fs = (FileSet) filesets.get(i);
614             if (fs.getProject() == null) {
615                 log("Deleting fileset with no project specified;"
616                     + " assuming executing project", Project.MSG_VERBOSE);
617                 fs = (FileSet) fs.clone();
618                 fs.setProject(getProject());
619             }
620             final File fsDir = fs.getDir();
621             if (!fs.getErrorOnMissingDir() &&
622                 (fsDir == null || !fsDir.exists())) {
623                 continue;
624             }
625             if (fsDir == null) {
626                 throw new BuildException(
627                         "File or Resource without directory or file specified");
628             } else if (!fsDir.isDirectory()) {
629                 handle("Directory does not exist: " + fsDir);
630             } else {
631                 DirectoryScanner ds = fs.getDirectoryScanner();
632                 // the previous line has already scanned the
633                 // filesystem, in order to avoid a rescan when later
634                 // iterating, capture the results now and store them
635                 final String[] files = ds.getIncludedFiles();
636                 resourcesToDelete.add(new ResourceCollection() {
637                         public boolean isFilesystemOnly() {
638                             return true;
639                         }
640                         public int size() {
641                             return files.length;
642                         }
643                         public Iterator<Resource> iterator() {
644                             return new FileResourceIterator(getProject(),
645                                                             fsDir, files);
646                         }
647                     });
648                 if (includeEmpty) {
649                     filesetDirs.add(new ReverseDirs(getProject(), fsDir,
650                                                     ds
651                                                     .getIncludedDirectories()));
652                 }
653 
654                 if (removeNotFollowedSymlinks) {
655                     String[] n = ds.getNotFollowedSymlinks();
656                     if (n.length > 0) {
657                         String[] links = new String[n.length];
658                         System.arraycopy(n, 0, links, 0, n.length);
659                         Arrays.sort(links, ReverseDirs.REVERSE);
660                         for (int l = 0; l < links.length; l++) {
661                             try {
662                                 SYMLINK_UTILS
663                                     .deleteSymbolicLink(new File(links[l]),
664                                                         this);
665                             } catch (java.io.IOException ex) {
666                                 handle(ex);
667                             }
668                         }
669                     }
670                 }
671             }
672         }
673         resourcesToDelete.add(filesetDirs);
674         if (rcs != null) {
675             // sort first to files, then dirs
676             Restrict exists = new Restrict();
677             exists.add(EXISTS);
678             exists.add(rcs);
679             Sort s = new Sort();
680             s.add(REVERSE_FILESYSTEM);
681             s.add(exists);
682             resourcesToDelete.add(s);
683         }
684         try {
685             if (resourcesToDelete.isFilesystemOnly()) {
686                 for (Resource r : resourcesToDelete) {
687                     // nonexistent resources could only occur if we already
688                     // deleted something from a fileset:
689                     File f = r.as(FileProvider.class)
690                               .getFile();
691                     if (!f.exists()) {
692                         continue;
693                     }
694                     if (!(f.isDirectory()) || f.list().length == 0) {
695                         log("Deleting " + f, verbosity);
696                         if (!delete(f) && failonerror) {
697                             handle("Unable to delete "
698                                 + (f.isDirectory() ? "directory " : "file ") + f);
699                         }
700                     }
701                 }
702             } else {
703                  handle(getTaskName() + " handles only filesystem resources");
704             }
705         } catch (Exception e) {
706             handle(e);
707         } finally {
708             if (implicit != null) {
709                 filesets.remove(implicit);
710             }
711         }
712     }

718     private void handle(String msg) {
719         handle(new BuildException(msg));
720     }

722     private void handle(Exception e) {
723         if (failonerror) {
724             throw (e instanceof BuildException)
725                 ? (BuildException) e : new BuildException(e);
726         }
727         log(e, quiet ? Project.MSG_VERBOSE : verbosity);
728     }

730     /**
731      * Accommodate Windows bug encountered in both Sun and IBM JDKs.
732      * Others possible. If the delete does not work, call System.gc(),
733      * wait a little and try again.
734      */
735     private boolean delete(File f) {
736         if (!FILE_UTILS.tryHardToDelete(f, performGc)) {
737             if (deleteOnExit) {
738                 int level = quiet ? Project.MSG_VERBOSE : Project.MSG_INFO;
739                 log("Failed to delete " + f + ", calling deleteOnExit."
740                     + " This attempts to delete the file when the Ant jvm"
741                     + " has exited and might not succeed.", level);
742                 f.deleteOnExit();
743                 return true;
744             }
745             return false;
746         }
747         return true;
748     }

750     /**
751      * Delete a directory
752      *
753      * @param d the directory to delete
754      */
755     protected void removeDir(File d) {
756         String[] list = d.list();
757         if (list == null) {
758             list = new String[0];
759         }
760         for (int i = 0; i < list.length; i++) {
761             String s = list[i];
762             File f = new File(d, s);
763             if (f.isDirectory()) {
764                 removeDir(f);
765             } else {
766                 log("Deleting " + f.getAbsolutePath(), quiet ? Project.MSG_VERBOSE : verbosity);
767                 if (!delete(f)) {
768                     handle("Unable to delete file " + f.getAbsolutePath());
769                 }
770             }
771         }
772         log("Deleting directory " + d.getAbsolutePath(), verbosity);
773         if (!delete(d)) {
774             handle("Unable to delete directory " + d.getAbsolutePath());
775         }
776     }

778     /**
779      * remove an array of files in a directory, and a list of subdirectories
780      * which will only be deleted if 'includeEmpty' is true
781      * @param d directory to work from
782      * @param files array of files to delete; can be of zero length
783      * @param dirs array of directories to delete; can of zero length
784      */
785     protected void removeFiles(File d, String[] files, String[] dirs) {
786         if (files.length > 0) {
787             log("Deleting " + files.length + " files from "
788                 + d.getAbsolutePath(), quiet ? Project.MSG_VERBOSE : verbosity);
789             for (int j = 0; j < files.length; j++) {
790                 File f = new File(d, files[j]);
791                 log("Deleting " + f.getAbsolutePath(),
792                         quiet ? Project.MSG_VERBOSE : verbosity);
793                 if (!delete(f)) {
794                     handle("Unable to delete file " + f.getAbsolutePath());
795                 }
796             }
797         }
798 
799         if (dirs.length > 0 && includeEmpty) {
800             int dirCount = 0;
801             for (int j = dirs.length - 1; j >= 0; j--) {
802                 File currDir = new File(d, dirs[j]);
803                 String[] dirFiles = currDir.list();
804                 if (dirFiles == null || dirFiles.length == 0) {
805                     log("Deleting " + currDir.getAbsolutePath(),
806                             quiet ? Project.MSG_VERBOSE : verbosity);
807                     if (!delete(currDir)) {
808                         handle("Unable to delete directory "
809                                 + currDir.getAbsolutePath());
810                     } else {
811                         dirCount++;
812                     }
813                 }
814             }
815 
816             if (dirCount > 0) {
817                 log("Deleted "
818                      + dirCount
819                      + " director" + (dirCount == 1 ? "y" : "ies")
820                      + " form " + d.getAbsolutePath(),
821                      quiet ? Project.MSG_VERBOSE : verbosity);
822             }
823         }
824     }

826     private boolean isDanglingSymlink(File f) {
827         try {
828             return SYMLINK_UTILS.isDanglingSymbolicLink(f);
829         } catch (java.io.IOException e) {
830             log("Error while trying to detect " + f.getAbsolutePath()
831                 + " as broken symbolic link. " + e.getMessage(),
832                 quiet ? Project.MSG_VERBOSE : verbosity);
833             return false;
834         }
835     }
