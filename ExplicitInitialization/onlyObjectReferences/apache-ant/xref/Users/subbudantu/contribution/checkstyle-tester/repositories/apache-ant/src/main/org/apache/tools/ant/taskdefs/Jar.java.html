303     private Manifest getManifest(File manifestFile) {
304 
305         Manifest newManifest = null;
306         FileInputStream fis = null;
307         InputStreamReader isr = null;
308         try {
309             fis = new FileInputStream(manifestFile);
310             if (manifestEncoding == null) {
311                 isr = new InputStreamReader(fis);
312             } else {
313                 isr = new InputStreamReader(fis, manifestEncoding);
314             }
315             newManifest = getManifest(isr);
316         } catch (UnsupportedEncodingException e) {
317             throw new BuildException("Unsupported encoding while reading manifest: "
318                                      + e.getMessage(), e);
319         } catch (IOException e) {
320             throw new BuildException("Unable to read manifest file: "
321                                      + manifestFile
322                                      + " (" + e.getMessage() + ")", e);
323         } finally {
324             FileUtils.close(isr);
325         }
326         return newManifest;
327     }

329     /**
330      * @return null if jarFile doesn't contain a manifest, the
331      * manifest otherwise.
332      * @since Ant 1.5.2
333      */
334     private Manifest getManifestFromJar(File jarFile) throws IOException {
335         ZipFile zf = null;
336         try {
337             zf = new ZipFile(jarFile);
338 
339             // must not use getEntry as "well behaving" applications
340             // must accept the manifest in any capitalization
341             Enumeration<? extends ZipEntry> e = zf.entries();
342             while (e.hasMoreElements()) {
343                 ZipEntry ze = e.nextElement();
344                 if (ze.getName().equalsIgnoreCase(MANIFEST_NAME)) {
345                     InputStreamReader isr =
346                         new InputStreamReader(zf.getInputStream(ze), "UTF-8");
347                     return getManifest(isr);
348                 }
349             }
350             return null;
351         } finally {
352             if (zf != null) {
353                 try {
354                     zf.close();
355                 } catch (IOException e) {
356                     // TODO - log an error?  throw an exception?
357                 }
358             }
359         }
360     }

378     private boolean jarHasIndex(File jarFile) throws IOException {
379         ZipFile zf = null;
380         try {
381             zf = new ZipFile(jarFile);
382             Enumeration<? extends ZipEntry> e = zf.entries();
383             while (e.hasMoreElements()) {
384                 ZipEntry ze = e.nextElement();
385                 if (ze.getName().equalsIgnoreCase(INDEX_NAME)) {
386                     return true;
387                 }
388             }
389             return false;
390         } finally {
391             if (zf != null) {
392                 try {
393                     zf.close();
394                 } catch (IOException e) {
395                     // TODO - log an error?  throw an exception?
396                 }
397             }
398         }
399     }

516     private Manifest createManifest()
517         throws BuildException {
518         try {
519             if (manifest == null) {
520                 if (manifestFile != null) {
521                     // if we haven't got the manifest yet, attempt to
522                     // get it now and have manifest be the final merge
523                     manifest = getManifest(manifestFile);
524                 }
525             }
526 
527             // fileset manifest must come even before the default
528             // manifest if mergewithoutmain is selected and there is
529             // no explicit manifest specified - otherwise the Main
530             // section of the fileset manifest is still merged to the
531             // final manifest.
532             boolean mergeFileSetFirst = !mergeManifestsMain
533                 && filesetManifest != null
534                 && configuredManifest == null && manifest == null;
535 
536             Manifest finalManifest;
537             if (mergeFileSetFirst) {
538                 finalManifest = new Manifest();
539                 finalManifest.merge(filesetManifest, false, mergeClassPaths);
540                 finalManifest.merge(Manifest.getDefaultManifest(),
541                                     true, mergeClassPaths);
542             } else {
543                 finalManifest = Manifest.getDefaultManifest();
544             }
545 
546             /*
547              * Precedence: manifestFile wins over inline manifest,
548              * over manifests read from the filesets over the original
549              * manifest.
550              *
551              * merge with null argument is a no-op
552              */
553 
554             if (isInUpdateMode()) {
555                 finalManifest.merge(originalManifest, false, mergeClassPaths);
556             }
557             if (!mergeFileSetFirst) {
558                 finalManifest.merge(filesetManifest, false, mergeClassPaths);
559             }
560             finalManifest.merge(configuredManifest, !mergeManifestsMain,
561                                 mergeClassPaths);
562             finalManifest.merge(manifest, !mergeManifestsMain,
563                                 mergeClassPaths);
564 
565             return finalManifest;
566 
567         } catch (ManifestException e) {
568             log("Manifest is invalid: " + e.getMessage(), Project.MSG_ERR);
569             throw new BuildException("Invalid Manifest", e, getLocation());
570         }
571     }

573     private void writeManifest(ZipOutputStream zOut, Manifest manifest)
574         throws IOException {
575         for (Enumeration<String> e = manifest.getWarnings();
576              e.hasMoreElements();) {
577             log("Manifest warning: " + e.nextElement(),
578                 Project.MSG_WARN);
579         }
580 
581         zipDir((Resource) null, zOut, "META-INF/", ZipFileSet.DEFAULT_DIR_MODE,
582                JAR_MARKER);
583         // time to write the manifest
584         ByteArrayOutputStream baos = new ByteArrayOutputStream();
585         OutputStreamWriter osw = new OutputStreamWriter(baos, Manifest.JAR_ENCODING);
586         PrintWriter writer = new PrintWriter(osw);
587         manifest.write(writer, flattenClassPaths);
588         if (writer.checkError()) {
589             throw new IOException("Encountered an error writing the manifest");
590         }
591         writer.close();
592 
593         ByteArrayInputStream bais =
594             new ByteArrayInputStream(baos.toByteArray());
595         try {
596             super.zipFile(bais, zOut, MANIFEST_NAME,
597                           System.currentTimeMillis(), null,
598                           ZipFileSet.DEFAULT_FILE_MODE);
599         } finally {
600             // not really required
601             FileUtils.close(bais);
602         }
603         super.initZipOutputStream(zOut);
604     }

621     /**
622      * Create the index list to speed up classloading.
623      * This is a JDK 1.3+ specific feature and is enabled by default. See
624      * <a href="http://java.sun.com/j2se/1.3/docs/guide/jar/jar.html#JAR%20Index">
625      * the JAR index specification</a> for more details.
626      *
627      * @param zOut the zip stream representing the jar being built.
628      * @throws IOException thrown if there is an error while creating the
629      * index and adding it to the zip stream.
630      */
631     private void createIndexList(ZipOutputStream zOut) throws IOException {
632         ByteArrayOutputStream baos = new ByteArrayOutputStream();
633         // encoding must be UTF8 as specified in the specs.
634         PrintWriter writer = new PrintWriter(new OutputStreamWriter(baos,
635                                                                     "UTF8"));
636 
637         // version-info blankline
638         writer.println("JarIndex-Version: 1.0");
639         writer.println();
640 
641         // header newline
642         writer.println(zipFile.getName());
643 
644         writeIndexLikeList(new ArrayList<String>(addedDirs.keySet()),
645                            rootEntries, writer);
646         writer.println();
647 
648         if (indexJars != null) {
649             Manifest mf = createManifest();
650             Manifest.Attribute classpath =
651                 mf.getMainSection().getAttribute(Manifest.ATTRIBUTE_CLASSPATH);
652             String[] cpEntries = null;
653             if (classpath != null && classpath.getValue() != null) {
654                 StringTokenizer tok = new StringTokenizer(classpath.getValue(),
655                                                           " ");
656                 cpEntries = new String[tok.countTokens()];
657                 int c = 0;
658                 while (tok.hasMoreTokens()) {
659                     cpEntries[c++] = tok.nextToken();
660                 }
661             }
662             String[] indexJarEntries = indexJars.list();
663             for (int i = 0; i < indexJarEntries.length; i++) {
664                 String name = findJarName(indexJarEntries[i], cpEntries);
665                 if (name != null) {
666                     ArrayList<String> dirs = new ArrayList<String>();
667                     ArrayList<String> files = new ArrayList<String>();
668                     grabFilesAndDirs(indexJarEntries[i], dirs, files);
669                     if (dirs.size() + files.size() > 0) {
670                         writer.println(name);
671                         writeIndexLikeList(dirs, files, writer);
672                         writer.println();
673                     }
674                 }
675             }
676         }
677 
678         if (writer.checkError()) {
679             throw new IOException("Encountered an error writing jar index");
680         }
681         writer.close();
682         ByteArrayInputStream bais =
683             new ByteArrayInputStream(baos.toByteArray());
684         try {
685             super.zipFile(bais, zOut, INDEX_NAME, System.currentTimeMillis(),
686                           null, ZipFileSet.DEFAULT_FILE_MODE);
687         } finally {
688             // not really required
689             FileUtils.close(bais);
690         }
691     }

693     /**
694      * Overridden from Zip class to deal with manifests and index lists.
695      * @param is the stream to read data for the entry from.  The
696      * caller of the method is responsible for closing the stream.
697      * @param zOut the zip output stream
698      * @param vPath the name this entry shall have in the archive
699      * @param lastModified last modification time for the entry.
700      * @param fromArchive the original archive we are copying this
701      *                    entry from, will be null if we are not copying from an archive.
702      * @param mode the Unix permissions to set.
703      * @throws IOException on error
704      */
705     protected void zipFile(InputStream is, ZipOutputStream zOut, String vPath,
706                            long lastModified, File fromArchive, int mode)
707         throws IOException {
708         if (MANIFEST_NAME.equalsIgnoreCase(vPath))  {
709             if (isFirstPass()) {
710                 filesetManifest(fromArchive, is);
711             }
712         } else if (INDEX_NAME.equalsIgnoreCase(vPath) && index) {
713             logWhenWriting("Warning: selected " + archiveType
714                            + " files include a " + INDEX_NAME + " which will"
715                            + " be replaced by a newly generated one.",
716                            Project.MSG_WARN);
717         } else {
718             if (index && vPath.indexOf("/") == -1) {
719                 rootEntries.addElement(vPath);
720             }
721             super.zipFile(is, zOut, vPath, lastModified, fromArchive, mode);
722         }
723     }

725     private void filesetManifest(File file, InputStream is) throws IOException {
726         if (manifestFile != null && manifestFile.equals(file)) {
727             // If this is the same name specified in 'manifest', this
728             // is the manifest to use
729             log("Found manifest " + file, Project.MSG_VERBOSE);
730             try {
731                 if (is != null) {
732                     InputStreamReader isr;
733                     if (manifestEncoding == null) {
734                         isr = new InputStreamReader(is);
735                     } else {
736                         isr = new InputStreamReader(is, manifestEncoding);
737                     }
738                     manifest = getManifest(isr);
739                 } else {
740                     manifest = getManifest(file);
741                 }
742             } catch (UnsupportedEncodingException e) {
743                 throw new BuildException("Unsupported encoding while reading "
744                     + "manifest: " + e.getMessage(), e);
745             }
746         } else if (filesetManifestConfig != null
747                     && !filesetManifestConfig.getValue().equals("skip")) {
748             // we add this to our group of fileset manifests
749             logWhenWriting("Found manifest to merge in file " + file,
750                            Project.MSG_VERBOSE);
751 
752             try {
753                 Manifest newManifest = null;
754                 if (is != null) {
755                     InputStreamReader isr;
756                     if (manifestEncoding == null) {
757                         isr = new InputStreamReader(is);
758                     } else {
759                         isr = new InputStreamReader(is, manifestEncoding);
760                     }
761                     newManifest = getManifest(isr);
762                 } else {
763                     newManifest = getManifest(file);
764                 }
765 
766                 if (filesetManifest == null) {
767                     filesetManifest = newManifest;
768                 } else {
769                     filesetManifest.merge(newManifest, false, mergeClassPaths);
770                 }
771             } catch (UnsupportedEncodingException e) {
772                 throw new BuildException("Unsupported encoding while reading "
773                     + "manifest: " + e.getMessage(), e);
774             } catch (ManifestException e) {
775                 log("Manifest in file " + file + " is invalid: "
776                     + e.getMessage(), Project.MSG_ERR);
777                 throw new BuildException("Invalid Manifest", e, getLocation());
778             }
779         } else {
780             // assuming 'skip' otherwise
781             // don't warn if skip has been requested explicitly, warn if user
782             // didn't set the attribute
783 
784             // Hide warning also as it makes no sense since
785             // the filesetmanifest attribute itself has been
786             // hidden
787 
788             //int logLevel = filesetManifestConfig == null ?
789             //    Project.MSG_WARN : Project.MSG_VERBOSE;
790             //log("File " + file
791             //    + " includes a META-INF/MANIFEST.MF which will be ignored. "
792             //    + "To include this file, set filesetManifest to a value other "
793             //    + "than 'skip'.", logLevel);
794         }
795     }

797     /**
798      * Collect the resources that are newer than the corresponding
799      * entries (or missing) in the original archive.
800      *
801      * <p>If we are going to recreate the archive instead of updating
802      * it, all resources should be considered as new, if a single one
803      * is.  Because of this, subclasses overriding this method must
804      * call <code>super.getResourcesToAdd</code> and indicate with the
805      * third arg if they already know that the archive is
806      * out-of-date.</p>
807      *
808      * @param rcs The resource collections to grab resources from
809      * @param zipFile intended archive file (may or may not exist)
810      * @param needsUpdate whether we already know that the archive is
811      * out-of-date.  Subclasses overriding this method are supposed to
812      * set this value correctly in their call to
813      * super.getResourcesToAdd.
814      * @return an array of resources to add for each fileset passed in as well
815      *         as a flag that indicates whether the archive is uptodate.
816      *
817      * @exception BuildException if it likes
818      */
819     protected ArchiveState getResourcesToAdd(ResourceCollection[] rcs,
820                                              File zipFile,
821                                              boolean needsUpdate)
822         throws BuildException {
823 
824         if (skipWriting) {
825             // this pass is only there to construct the merged
826             // manifest this means we claim an update was needed and
827             // only include the manifests, skipping any uptodate
828             // checks here deferring them for the second run
829             Resource[][] manifests = grabManifests(rcs);
830             int count = 0;
831             for (int i = 0; i < manifests.length; i++) {
832                 count += manifests[i].length;
833             }
834             log("found a total of " + count + " manifests in "
835                 + manifests.length + " resource collections",
836                 Project.MSG_VERBOSE);
837             return new ArchiveState(true, manifests);
838         }
839 
840         // need to handle manifest as a special check
841         if (zipFile.exists()) {
842             // if it doesn't exist, it will get created anyway, don't
843             // bother with any up-to-date checks.
844 
845             try {
846                 originalManifest = getManifestFromJar(zipFile);
847                 if (originalManifest == null) {
848                     log("Updating jar since the current jar has"
849                                    + " no manifest", Project.MSG_VERBOSE);
850                     needsUpdate = true;
851                 } else {
852                     Manifest mf = createManifest();
853                     if (!mf.equals(originalManifest)) {
854                         log("Updating jar since jar manifest has"
855                                        + " changed", Project.MSG_VERBOSE);
856                         needsUpdate = true;
857                     }
858                 }
859             } catch (Throwable t) {
860                 log("error while reading original manifest in file: "
861                     + zipFile.toString() + " due to " + t.getMessage(),
862                     Project.MSG_WARN);
863                 needsUpdate = true;
864             }
865 
866         } else {
867             // no existing archive
868             needsUpdate = true;
869         }
870 
871         createEmpty = needsUpdate;
872         if (!needsUpdate && index) {
873             try {
874                 needsUpdate = !jarHasIndex(zipFile);
875             } catch (IOException e) {
876                 //if we couldn't read it, we might as well recreate it?
877                 needsUpdate = true;
878             }
879         }
880         return super.getResourcesToAdd(rcs, zipFile, needsUpdate);
881     }

883     /**
884      * Create an empty jar file.
885      * @param zipFile the file to create
886      * @return true for historic reasons
887      * @throws BuildException on error
888      */
889     protected boolean createEmptyZip(File zipFile) throws BuildException {
890         if (!createEmpty) {
891             return true;
892         }
893 
894         if (emptyBehavior.equals("skip")) {
895             if (!skipWriting) {
896                 log("Warning: skipping " + archiveType + " archive "
897                     + zipFile + " because no files were included.",
898                     Project.MSG_WARN);
899             }
900             return true;
901         } else if (emptyBehavior.equals("fail")) {
902             throw new BuildException("Cannot create " + archiveType
903                                      + " archive " + zipFile
904                                      + ": no files were included.",
905                                      getLocation());
906         }
907 
908         ZipOutputStream zOut = null;
909         try {
910             if (!skipWriting) {
911                 log("Building MANIFEST-only jar: "
912                     + getDestFile().getAbsolutePath());
913             }
914             zOut = new ZipOutputStream(getDestFile());
915 
916             zOut.setEncoding(getEncoding());
917             if (isCompress()) {
918                 zOut.setMethod(ZipOutputStream.DEFLATED);
919             } else {
920                 zOut.setMethod(ZipOutputStream.STORED);
921             }
922             initZipOutputStream(zOut);
923             finalizeZipOutputStream(zOut);
924         } catch (IOException ioe) {
925             throw new BuildException("Could not create almost empty JAR archive"
926                                      + " (" + ioe.getMessage() + ")", ioe,
927                                      getLocation());
928         } finally {
929             // Close the output stream.
930             FileUtils.close(zOut);
931             createEmpty = false;
932         }
933         return true;
934     }

957     /**
958      * Check against packaging spec
959      * @see http://java.sun.com/j2se/1.3/docs/guide/versioning/spec/VersioningSpecification.html#PackageVersioning
960      */
961     // CheckStyle:LineLength ON
962     private void checkJarSpec() {
963         String br = System.getProperty("line.separator");
964         StringBuffer message = new StringBuffer();
965         Section mainSection = (configuredManifest == null)
966                             ? null
967                             : configuredManifest.getMainSection();
968 
969         if (mainSection == null) {
970             message.append("No Implementation-Title set.");
971             message.append("No Implementation-Version set.");
972             message.append("No Implementation-Vendor set.");
973         } else {
974             if (mainSection.getAttribute("Implementation-Title") == null) {
975                 message.append("No Implementation-Title set.");
976             }
977             if (mainSection.getAttribute("Implementation-Version") == null) {
978                 message.append("No Implementation-Version set.");
979             }
980             if (mainSection.getAttribute("Implementation-Vendor") == null) {
981                 message.append("No Implementation-Vendor set.");
982             }
983         }
984 
985         if (message.length() > 0) {
986             message.append(br);
987             message.append("Location: ").append(getLocation());
988             message.append(br);
989             if (strict.getValue().equalsIgnoreCase("fail")) {
990                 throw new BuildException(message.toString(), getLocation());
991             } else {
992                 logWhenWriting(message.toString(), strict.getLogLevel());
993             }
994         }

1027     /**
1028      * Writes the directory entries from the first and the filenames
1029      * from the second list to the given writer, one entry per line.
1030      *
1031      * @param dirs a list of directories
1032      * @param files a list of files
1033      * @param writer the writer to write to
1034      * @throws IOException on error
1035      * @since Ant 1.6.2
1036      */
1037     protected final void writeIndexLikeList(List<String> dirs, List<String> files,
1038                                             PrintWriter writer)
1039         throws IOException {
1040         // JarIndex is sorting the directories by ascending order.
1041         // it has no value but cosmetic since it will be read into a
1042         // hashtable by the classloader, but we'll do so anyway.
1043         Collections.sort(dirs);
1044         Collections.sort(files);
1045         for (String dir : dirs) {
1046             // try to be smart, not to be fooled by a weird directory name
1047             dir = dir.replace('\\', '/');
1048             if (dir.startsWith("./")) {
1049                 dir = dir.substring(2);
1050             }
1051             while (dir.startsWith("/")) {
1052                 dir = dir.substring(1);
1053             }
1054             int pos = dir.lastIndexOf('/');
1055             if (pos != -1) {
1056                 dir = dir.substring(0, pos);
1057             }
1058 
1059             // looks like nothing from META-INF should be added
1060             // and the check is not case insensitive.
1061             // see sun.misc.JarIndex
1062             // see also
1063             // http://bugs.sun.com/bugdatabase/view_bug.do?bug_id=4408526
1064             if (!indexMetaInf && dir.startsWith("META-INF")) {
1065                 continue;
1066             }
1067             // name newline
1068             writer.println(dir);
1069         }
1070 
1071         for (String file : files) {
1072             writer.println(file);
1073         }
1074     }

1076     /**
1077      * try to guess the name of the given file.
1078      *
1079      * <p>If this jar has a classpath attribute in its manifest, we
1080      * can assume that it will only require an index of jars listed
1081      * there.  try to find which classpath entry is most likely the
1082      * one the given file name points to.</p>
1083      *
1084      * <p>In the absence of a classpath attribute, assume the other
1085      * files will be placed inside the same directory as this jar and
1086      * use their basename.</p>
1087      *
1088      * <p>if there is a classpath and the given file doesn't match any
1089      * of its entries, return null.</p>
1090      *
1091      * @param fileName the name to look for
1092      * @param classpath the classpath to look in (may be null)
1093      * @return the matching entry, or null if the file is not found
1094      * @since Ant 1.6.2
1095      */
1096     protected static String findJarName(String fileName,
1097                                               String[] classpath) {
1098         if (classpath == null) {
1099             return (new File(fileName)).getName();
1100         }
1101         fileName = fileName.replace(File.separatorChar, '/');
1102         TreeMap<String, String> matches = new TreeMap<String, String>(new Comparator<Object>() {
1103                 // longest match comes first
1104                 public int compare(Object o1, Object o2) {
1105                     if (o1 instanceof String && o2 instanceof String) {
1106                         return ((String) o2).length()
1107                             - ((String) o1).length();
1108                     }
1109                     return 0;
1110                 }
1111             });
1112 
1113         for (int i = 0; i < classpath.length; i++) {
1114             if (fileName.endsWith(classpath[i])) {
1115                 matches.put(classpath[i], classpath[i]);
1116             } else {
1117                 int slash = classpath[i].indexOf("/");
1118                 String candidate = classpath[i];
1119                 while (slash > -1) {
1120                     candidate = candidate.substring(slash + 1);
1121                     if (fileName.endsWith(candidate)) {
1122                         matches.put(candidate, classpath[i]);
1123                         break;
1124                     }
1125                     slash = candidate.indexOf("/");
1126                 }
1127             }
1128         }
1129 
1130         return matches.size() == 0
1131             ? null : (String) matches.get(matches.firstKey());
1132     }

1134     /**
1135      * Grab lists of all root-level files and all directories
1136      * contained in the given archive.
1137      * @param file the zip file to examine
1138      * @param dirs where to place the directories found
1139      * @param files where to place the files found
1140      * @since Ant 1.7
1141      * @throws IOException on error
1142      */
1143     protected static void grabFilesAndDirs(String file, List<String> dirs,
1144                                                  List<String> files)
1145         throws IOException {
1146         org.apache.tools.zip.ZipFile zf = null;
1147         try {
1148             zf = new org.apache.tools.zip.ZipFile(file, "utf-8");
1149             Enumeration<org.apache.tools.zip.ZipEntry> entries = zf.getEntries();
1150             HashSet<String> dirSet = new HashSet<String>();
1151             while (entries.hasMoreElements()) {
1152                 org.apache.tools.zip.ZipEntry ze =
1153                     entries.nextElement();
1154                 String name = ze.getName();
1155                 if (ze.isDirectory()) {
1156                     dirSet.add(name);
1157                 } else if (name.indexOf("/") == -1) {
1158                     files.add(name);
1159                 } else {
1160                     // a file, not in the root
1161                     // since the jar may be one without directory
1162                     // entries, add the parent dir of this file as
1163                     // well.
1164                     dirSet.add(name.substring(0, name.lastIndexOf("/") + 1));
1165                 }
1166             }
1167             dirs.addAll(dirSet);
1168         } finally {
1169             if (zf != null) {
1170                 zf.close();
1171             }
1172         }
1173     }

1175     private Resource[][] grabManifests(ResourceCollection[] rcs) {
1176         Resource[][] manifests = new Resource[rcs.length][];
1177         for (int i = 0; i < rcs.length; i++) {
1178             Resource[][] resources = null;
1179             if (rcs[i] instanceof FileSet) {
1180                 resources = grabResources(new FileSet[] {(FileSet) rcs[i]});
1181             } else {
1182                 resources = grabNonFileSetResources(new ResourceCollection[] {
1183                         rcs[i]
1184                     });
1185             }
1186             for (int j = 0; j < resources[0].length; j++) {
1187                 String name = resources[0][j].getName().replace('\\', '/');
1188                 if (rcs[i] instanceof ArchiveFileSet) {
1189                     ArchiveFileSet afs = (ArchiveFileSet) rcs[i];
1190                     if (!"".equals(afs.getFullpath(getProject()))) {
1191                         name = afs.getFullpath(getProject());
1192                     } else if (!"".equals(afs.getPrefix(getProject()))) {
1193                         String prefix = afs.getPrefix(getProject());
1194                         if (!prefix.endsWith("/") && !prefix.endsWith("\\")) {
1195                             prefix += "/";
1196                         }
1197                         name = prefix + name;
1198                     }
1199                 }
1200                 if (name.equalsIgnoreCase(MANIFEST_NAME)) {
1201                     manifests[i] = new Resource[] {resources[0][j]};
1202                     break;
1203                 }
1204             }
1205             if (manifests[i] == null) {
1206                 manifests[i] = new Resource[0];
1207             }
1208         }
1209         return manifests;
1210     }
