104     /**
105      * Helper class, holds the nested &lt;map&gt; values. Elements will look like
106      * this: &lt;map from=&quot;d:&quot; to=&quot;/foo&quot;/&gt;
107      *
108      * When running on windows, the prefix comparison will be case
109      * insensitive.
110      */
111     public class MapEntry {
112 
113         // Members
114         private String from = null;
115         private String to = null;
116 
117         /**
118          * Set the &quot;from&quot; attribute of the map entry.
119          * @param from the prefix string to search for; required.
120          * Note that this value is case-insensitive when the build is
121          * running on a Windows platform and case-sensitive when running on
122          * a Unix platform.
123          */
124         public void setFrom(String from) {
125             this.from = from;
126         }
127 
128         /**
129          * Set the replacement text to use when from is matched; required.
130          * @param to new prefix.
131          */
132         public void setTo(String to) {
133             this.to = to;
134         }
135 
136         /**
137          * Apply this map entry to a given path element.
138          *
139          * @param elem Path element to process.
140          * @return String Updated path element after mapping.
141          */
142         public String apply(String elem) {
143             if (from == null || to == null) {
144                 throw new BuildException("Both 'from' and 'to' must be set "
145                      + "in a map entry");
146             }
147             // If we're on windows, then do the comparison ignoring case
148             // and treat the two directory characters the same
149             String cmpElem =
150                 onWindows ? elem.toLowerCase().replace('\\', '/') : elem;
151             String cmpFrom =
152                 onWindows ? from.toLowerCase().replace('\\', '/') : from;
153 
154             // If the element starts with the configured prefix, then
155             // convert the prefix to the configured 'to' value.
156 
157             return cmpElem.startsWith(cmpFrom)
158                 ? to + elem.substring(from.length()) : elem;
159         }
160     }

233     /**
234      * Set targetos to a platform to one of
235      * "windows", "unix", "netware", or "os/2";
236      * current platform settings are used by default.
237      * @param target the target os
238      *
239      * @since Ant 1.5
240      */
241     public void setTargetos(TargetOs target) {
242         targetOS = target.getValue();
243 
244         // Currently, we deal with only two path formats: Unix and Windows
245         // And Unix is everything that is not Windows
246 
247         // for NetWare and OS/2, piggy-back on Windows, since in the
248         // validateSetup code, the same assumptions can be made as
249         // with windows - that ; is the path separator
250 
251         targetWindows = !targetOS.equals("unix") && !targetOS.equals("tandem");
252     }

329     /**
330      * Do the execution.
331      * @throws BuildException if something is invalid.
332      */
333     public void execute() throws BuildException {
334         Resources savedPath = path;
335         String savedPathSep = pathSep; // may be altered in validateSetup
336         String savedDirSep = dirSep; // may be altered in validateSetup
337 
338         try {
339             // If we are a reference, create a Path from the reference
340             if (isReference()) {
341                 Object o = refid.getReferencedObject(getProject());
342                 if (!(o instanceof ResourceCollection)) {
343                     throw new BuildException("refid '" + refid.getRefId()
344                         + "' does not refer to a resource collection.");
345                 }
346                 getPath().add((ResourceCollection) o);
347             }
348             validateSetup(); // validate our setup
349 
350             // Currently, we deal with only two path formats: Unix and Windows
351             // And Unix is everything that is not Windows
352             // (with the exception for NetWare and OS/2 below)
353 
354             // for NetWare and OS/2, piggy-back on Windows, since here and
355             // in the apply code, the same assumptions can be made as with
356             // windows - that \\ is an OK separator, and do comparisons
357             // case-insensitive.
358             String fromDirSep = onWindows ? "\\" : "/";
359 
360             StringBuffer rslt = new StringBuffer();
361 
362             ResourceCollection resources = isPreserveDuplicates() ? (ResourceCollection) path : new Union(path);
363             List ret = new ArrayList();
364             FileNameMapper mapperImpl = mapper == null ? new IdentityMapper() : mapper.getImplementation();
365             for (Resource r : resources) {
366                 String[] mapped = mapperImpl.mapFileName(String.valueOf(r));
367                 for (int m = 0; mapped != null && m < mapped.length; ++m) {
368                     ret.add(mapped[m]);
369                 }
370             }
371             boolean first = true;
372             for (Iterator mappedIter = ret.iterator(); mappedIter.hasNext(); ) {
373                 String elem = mapElement((String) mappedIter.next()); // Apply the path prefix map
374 
375                 // Now convert the path and file separator characters from the
376                 // current os to the target os.
377 
378                 if (!first) {
379                     rslt.append(pathSep);
380                 }
381                 first = false;
382 
383                 StringTokenizer stDirectory = new StringTokenizer(elem, fromDirSep, true);
384 
385                 while (stDirectory.hasMoreTokens()) {
386                     String token = stDirectory.nextToken();
387                     rslt.append(fromDirSep.equals(token) ? dirSep : token);
388                 }
389             }
390             // Place the result into the specified property,
391             // unless setonempty == false
392             if (setonempty || rslt.length() > 0) {
393                 String value = rslt.toString();
394                 if (property == null) {
395                     log(value);
396                 } else {
397                     log("Set property " + property + " = " + value, Project.MSG_VERBOSE);
398                     getProject().setNewProperty(property, value);
399                 }
400             }
401         } finally {
402             path = savedPath;
403             dirSep = savedDirSep;
404             pathSep = savedPathSep;
405         }
406     }

408     /**
409      * Apply the configured map to a path element. The map is used to convert
410      * between Windows drive letters and Unix paths. If no map is configured,
411      * then the input string is returned unchanged.
412      *
413      * @param elem The path element to apply the map to.
414      * @return String Updated element.
415      */
416     private String mapElement(String elem) {
417 
418         int size = prefixMap.size();
419 
420         if (size != 0) {
421 
422             // Iterate over the map entries and apply each one.
423             // Stop when one of the entries actually changes the element.
424 
425             for (int i = 0; i < size; i++) {
426                 MapEntry entry = (MapEntry) prefixMap.elementAt(i);
427                 String newElem = entry.apply(elem);
428 
429                 // Note I'm using "!=" to see if we got a new object back from
430                 // the apply method.
431 
432                 if (newElem != elem) {
433                     elem = newElem;
434                     break; // We applied one, so we're done
435                 }
436             }
437         }
438         return elem;
439     }

465     /**
466      * Validate that all our parameters have been properly initialized.
467      *
468      * @throws BuildException if something is not set up properly.
469      */
470     private void validateSetup() throws BuildException {
471 
472         if (path == null) {
473             throw new BuildException("You must specify a path to convert");
474         }
475         // Determine the separator strings.  The dirsep and pathsep attributes
476         // override the targetOS settings.
477         String dsep = File.separator;
478         String psep = File.pathSeparator;
479 
480         if (targetOS != null) {
481             psep = targetWindows ? ";" : ":";
482             dsep = targetWindows ? "\\" : "/";
483         }
484         if (pathSep != null) {
485             // override with pathsep=
486             psep = pathSep;
487         }
488         if (dirSep != null) {
489             // override with dirsep=
490             dsep = dirSep;
491         }
492         pathSep = psep;
493         dirSep = dsep;
494     }
