37  /**
38   * Signs JAR or ZIP files with the javasign command line tool. The tool detailed
39   * dependency checking: files are only signed if they are not signed. The
40   * <tt>signjar</tt> attribute can point to the file to generate; if this file
41   * exists then its modification date is used as a cue as to whether to resign
42   * any JAR file.
43   *
44   * Timestamp driven signing is based on the unstable and inadequately documented
45   * information in the Java1.5 docs
46   * @see <a href="http://java.sun.com/j2se/1.5.0/docs/guide/security/time-of-signing-beta1.html">
47   * beta documentation</a>
48   * @ant.task category="java"
49   * @since Ant 1.1
50   */

320     /**
321      * sign the jar(s)
322      *
323      * @throws BuildException on errors
324      */
325     public void execute() throws BuildException {
326         //validation logic
327         final boolean hasJar = jar != null;
328         final boolean hasSignedJar = signedjar != null;
329         final boolean hasDestDir = destDir != null;
330         final boolean hasMapper = mapper != null;
331 
332         if (!hasJar && !hasResources()) {
333             throw new BuildException(ERROR_NO_SOURCE);
334         }
335         if (null == alias) {
336             throw new BuildException(ERROR_NO_ALIAS);
337         }
338 
339         if (null == storepass) {
340             throw new BuildException(ERROR_NO_STOREPASS);
341         }
342 
343         if (hasDestDir && hasSignedJar) {
344             throw new BuildException(ERROR_TODIR_AND_SIGNEDJAR);
345         }
346 
347 
348         if (hasResources() && hasSignedJar) {
349             throw new BuildException(ERROR_SIGNEDJAR_AND_PATHS);
350         }
351 
352         //this isnt strictly needed, but by being fussy now,
353         //we can change implementation details later
354         if (!hasDestDir && hasMapper) {
355             throw new BuildException(ERROR_MAPPER_WITHOUT_DEST);
356         }
357 
358         beginExecution();
359 
360 
361         try {
362             //special case single jar handling with signedjar attribute set
363             if (hasJar && hasSignedJar) {
364                 // single jar processing
365                 signOneJar(jar, signedjar);
366                 //return here.
367                 return;
368             }
369 
370             //the rest of the method treats single jar like
371             //a nested path with one file
372 
373             Path sources = createUnifiedSourcePath();
374             //set up our mapping policy
375             FileNameMapper destMapper;
376             if (hasMapper) {
377                 destMapper = mapper;
378             } else {
379                 //no mapper? use the identity policy
380                 destMapper = new IdentityMapper();
381             }
382 
383 
384             //at this point the paths are set up with lists of files,
385             //and the mapper is ready to map from source dirs to dest files
386             //now we iterate through every JAR giving source and dest names
387             // deal with the paths
388             for (Resource r : sources) {
389                 FileResource fr = ResourceUtils
390                     .asFileResource(r.as(FileProvider.class));
391 
392                 //calculate our destination directory; it is either the destDir
393                 //attribute, or the base dir of the fileset (for in situ updates)
394                 File toDir = hasDestDir ? destDir : fr.getBaseDir();
395 
396                 //determine the destination filename via the mapper
397                 String[] destFilenames = destMapper.mapFileName(fr.getName());
398                 if (destFilenames == null || destFilenames.length != 1) {
399                     //we only like simple mappers.
400                     throw new BuildException(ERROR_BAD_MAP + fr.getFile());
401                 }
402                 File destFile = new File(toDir, destFilenames[0]);
403                 signOneJar(fr.getFile(), destFile);
404             }
405         } finally {
406             endExecution();
407         }
408     }

410     /**
411      * Sign one jar.
412      * <p/>
413      * The signing only takes place if {@link #isUpToDate(File, File)} indicates
414      * that it is needed.
415      *
416      * @param jarSource source to sign
417      * @param jarTarget target; may be null
418      * @throws BuildException
419      */
420     private void signOneJar(File jarSource, File jarTarget)
421         throws BuildException {
422 
423 
424         File targetFile = jarTarget;
425         if (targetFile == null) {
426             targetFile = jarSource;
427         }
428         if (isUpToDate(jarSource, targetFile)) {
429             return;
430         }
431 
432         long lastModified = jarSource.lastModified();
433         final ExecTask cmd = createJarSigner();
434 
435         setCommonOptions(cmd);
436 
437         bindToKeystore(cmd);
438         if (null != sigfile) {
439             addValue(cmd, "-sigfile");
440             String value = this.sigfile;
441             addValue(cmd, value);
442         }
443 
444         try {
445             //DO NOT SET THE -signedjar OPTION if source==dest
446             //unless you like fielding hotspot crash reports
447             if (!FILE_UTILS.areSame(jarSource, targetFile)) {
448                 addValue(cmd, "-signedjar");
449                 addValue(cmd, targetFile.getPath());
450             }
451         } catch (IOException ioex) {
452             throw new BuildException(ioex);
453         }
454 
455         if (internalsf) {
456             addValue(cmd, "-internalsf");
457         }
458 
459         if (sectionsonly) {
460             addValue(cmd, "-sectionsonly");
461         }
462 
463         if (sigAlg != null) {
464             addValue(cmd, "-sigalg");
465             addValue(cmd, sigAlg);
466         }
467 
468         if (digestAlg != null) {
469             addValue(cmd, "-digestalg");
470             addValue(cmd, digestAlg);
471         }
472 
473         //add -tsa operations if declared
474         addTimestampAuthorityCommands(cmd);
475 
476         //JAR source is required
477         addValue(cmd, jarSource.getPath());
478 
479         //alias is required for signing
480         addValue(cmd, alias);
481 
482         log("Signing JAR: "
483             + jarSource.getAbsolutePath()
484             + " to "
485             + targetFile.getAbsolutePath()
486             + " as " + alias);
487 
488         cmd.execute();
489 
490         // restore the lastModified attribute
491         if (preserveLastModified) {
492             FILE_UTILS.setFileLastModified(targetFile, lastModified);
493         }
494     }

496     /**
497      * If the tsa parameters are set, this passes them to the command.
498      * There is no validation of java version, as third party JDKs
499      * may implement this on earlier/later jarsigner implementations.
500      * @param cmd the exec task.
501      */
502     private void addTimestampAuthorityCommands(final ExecTask cmd) {
503         if (tsaurl != null) {
504             addValue(cmd, "-tsa");
505             addValue(cmd, tsaurl);
506         }
507         if (tsacert != null) {
508             addValue(cmd, "-tsacert");
509             addValue(cmd, tsacert);
510         }
511     }

513     /**
514      * <p>Compare a jar file with its corresponding signed jar. The logic for this
515      * is complex, and best explained in the source itself. Essentially if
516      * either file doesnt exist, or the destfile has an out of date timestamp,
517      * then the return value is false.</p>
518      * 
519      * <p>If we are signing ourself, the check {@link #isSigned(File)} is used to
520      * trigger the process.</p>
521      *
522      * @param jarFile       the unsigned jar file
523      * @param signedjarFile the result signed jar file
524      * @return true if the signedjarFile is considered up to date
525      */
526     protected boolean isUpToDate(File jarFile, File signedjarFile) {
527         if (isForce() || null == jarFile || !jarFile.exists()) {
528             //these are pathological cases, but retained in case somebody
529             //subclassed us.
530             return false;
531         }
532 
533         //we normally compare destination with source
534         File destFile = signedjarFile;
535         if (destFile == null) {
536             //but if no dest is specified, compare source to source
537             destFile = jarFile;
538         }
539 
540         //if, by any means, the destfile and source match,
541         if (jarFile.equals(destFile)) {
542             if (lazy) {
543                 //we check the presence of signatures on lazy signing
544                 return isSigned(jarFile);
545             }
546             //unsigned or non-lazy self signings are always false
547             return false;
548         }
549 
550         //if they are different, the timestamps are used
551         return FILE_UTILS.isUpToDate(jarFile, destFile);
552     }

554     /**
555      * test for a file being signed, by looking for a signature in the META-INF
556      * directory with our alias/sigfile.
557      *
558      * @param file the file to be checked
559      * @return true if the file is signed
560      * @see IsSigned#isSigned(File, String)
561      */
562     protected boolean isSigned(File file) {
563         try {
564             return IsSigned.isSigned(file, sigfile == null ? alias : sigfile);
565         } catch (IOException e) {
566             //just log this
567             log(e.toString(), Project.MSG_VERBOSE);
568             return false;
569         }
570     }
