35  /**
36   * Runs a JSP compiler.
37   *
38   * <p> This task takes the given jsp files and compiles them into java
39   * files. It is then up to the user to compile the java files into classes.
40   *
41   * <p> The task requires the srcdir and destdir attributes to be
42   * set. This Task is a MatchingTask, so the files to be compiled can be
43   * specified using includes/excludes attributes or nested include/exclude
44   * elements. Optional attributes are verbose (set the verbosity level passed
45   * to jasper), package (name of the destination package for generated java
46   * classes and classpath (the classpath to use when running the jsp
47   * compiler).
48   * <p> This task supports the nested elements classpath (A Path) and
49   * classpathref (A Reference) which can be used in preference to the
50   * attribute classpath, if the jsp compiler is not already in the ant
51   * classpath.
52   *
53   * <p><h4>Usage</h4>
54   * <pre>
55   * &lt;jspc srcdir="${basedir}/src/war"
56   *       destdir="${basedir}/gensrc"
57   *       package="com.i3sp.jsp"
58   *       verbose="9"&gt;
59   *   &lt;include name="**\/*.jsp" /&gt;
60   * &lt;/jspc&gt;
61   * </pre>
62   *
63   * <p> Large Amount of cutting and pasting from the Javac task...
64   * @since 1.5
65   */

419     /**
420      * execute by building up a list of files that
421      * have changed and hand them off to a jsp compiler
422      * @throws BuildException on error.
423      */
424     public void execute()
425         throws BuildException {
426 
427         // make sure that we've got a destdir
428         if (destDir == null) {
429             throw new BuildException("destdir attribute must be set!",
430                                      getLocation());
431         }
432 
433         if (!destDir.isDirectory()) {
434             throw new BuildException("destination directory \"" + destDir
435                                      + "\" does not exist or is not a directory",
436                                      getLocation());
437         }
438 
439         File dest = getActualDestDir();
440 
441         AntClassLoader al = null;
442         try {
443             //bind to a compiler
444             JspCompilerAdapter compiler =
445                 JspCompilerAdapterFactory
446                 .getCompiler(compilerName, this,
447                              al = getProject().createClassLoader(compilerClasspath));
448 
449             //if we are a webapp, hand off to the compiler, which had
450             //better handle it
451             if (webApp != null) {
452                 doCompilation(compiler);
453                 return;
454             }
455 
456             // make sure that we've got a srcdir
457             if (src == null) {
458                 throw new BuildException("srcdir attribute must be set!",
459                                          getLocation());
460             }
461             String [] list = src.list();
462             if (list.length == 0) {
463                 throw new BuildException("srcdir attribute must be set!",
464                                          getLocation());
465             }
466 
467 
468             // if the compiler does its own dependency stuff, we just
469             // call it right now
470             if (compiler.implementsOwnDependencyChecking()) {
471                 doCompilation(compiler);
472                 return;
473             }
474 
475             //the remainder of this method is only for compilers that
476             //need their dependency work done
477             JspMangler mangler = compiler.createMangler();
478 
479             // scan source directories and dest directory to build up both copy
480             // lists and compile lists
481             resetFileLists();
482             int filecount = 0;
483             for (int i = 0; i < list.length; i++) {
484                 File srcDir = getProject().resolveFile(list[i]);
485                 if (!srcDir.exists()) {
486                     throw new BuildException("srcdir \"" + srcDir.getPath()
487                                              + "\" does not exist!",
488                                              getLocation());
489                 }
490                 DirectoryScanner ds = this.getDirectoryScanner(srcDir);
491                 String[] files = ds.getIncludedFiles();
492                 filecount = files.length;
493                 scanDir(srcDir, dest, mangler, files);
494             }
495 
496             // compile the source files
497 
498             log("compiling " + compileList.size() + " files",
499                 Project.MSG_VERBOSE);
500 
501             if (compileList.size() > 0) {
502 
503                 log("Compiling " + compileList.size() + " source file"
504                     + (compileList.size() == 1 ? "" : "s")
505                     + " to "
506                     + dest);
507                 doCompilation(compiler);
508 
509             } else {
510                 if (filecount == 0) {
511                     log("there were no files to compile", Project.MSG_INFO);
512                 } else {
513                     log("all files are up to date", Project.MSG_VERBOSE);
514                 }
515             }
516         } finally {
517             if (al != null) {
518                 al.cleanup();
519             }
520         }
521     }

523     /**
524      * calculate where the files will end up:
525      * this is destDir or it id destDir + the package name
526      */
527     private File getActualDestDir() {
528         File dest = null;
529         if (packageName == null) {
530             dest = destDir;
531         } else {
532             String path = destDir.getPath() + File.separatorChar
533                 + packageName.replace('.', File.separatorChar);
534             dest = new File(path);
535         }
536         return dest;
537     }

539     /**
540      * do the compile
541      */
542     private void doCompilation(JspCompilerAdapter compiler)
543         throws BuildException {
544         // now we need to populate the compiler adapter
545         compiler.setJspc(this);
546 
547         // finally, lets execute the compiler!!
548         if (!compiler.execute()) {
549             if (failOnError) {
550                 throw new BuildException(FAIL_MSG, getLocation());
551             } else {
552                 log(FAIL_MSG, Project.MSG_ERR);
553             }
554         }
555     }

564     /**
565      * Scans the directory looking for source files to be compiled.
566      * The results are returned in the class variable compileList
567      * @param srcDir the source directory.
568      * @param dest   the destination directory.
569      * @param mangler the jsp filename mangler.
570      * @param files   the file names to mangle.
571      */
572     protected void scanDir(File srcDir, File dest, JspMangler mangler,
573                            String[] files) {
574 
575         long now = (new Date()).getTime();
576 
577         for (int i = 0; i < files.length; i++) {
578             String filename = files[i];
579             File srcFile = new File(srcDir, filename);
580             File javaFile = mapToJavaFile(mangler, srcFile, srcDir, dest);
581             if (javaFile == null) {
582                 continue;
583             }
584 
585             if (srcFile.lastModified() > now) {
586                 log("Warning: file modified in the future: " + filename,
587                     Project.MSG_WARN);
588             }
589             boolean shouldCompile = false;
590             shouldCompile = isCompileNeeded(srcFile, javaFile);
591             if (shouldCompile) {
592                 compileList.addElement(srcFile.getAbsolutePath());
593                 javaFiles.addElement(javaFile);
594             }
595         }
596     }

598     /**
599      * Test whether or not compilation is needed. A return value of
600      * <code>true<code> means yes, <code>false</code> means
601      * our tests do not indicate this, but as the TLDs are
602      * not used for dependency checking this is not guaranteed.
603      * The current tests are
604      * <ol>
605      * <li>no dest file
606      * <li>dest file out of date w.r.t source
607      * <li>dest file zero bytes long
608      * </ol>
609      * @param srcFile JSP source file
610      * @param javaFile JSP dest file
611      * @return true if a compile is definitely needed.
612      *
613      */
614     private boolean isCompileNeeded(File srcFile, File javaFile) {
615         boolean shouldCompile = false;
616         if (!javaFile.exists()) {
617             shouldCompile = true;
618             log("Compiling " + srcFile.getPath()
619                 + " because java file " + javaFile.getPath()
620                 + " does not exist", Project.MSG_VERBOSE);
621         } else {
622             if (srcFile.lastModified() > javaFile.lastModified()) {
623                 shouldCompile = true;
624                 log("Compiling " + srcFile.getPath()
625                     + " because it is out of date with respect to "
626                     + javaFile.getPath(),
627                     Project.MSG_VERBOSE);
628             } else {
629                 if (javaFile.length() == 0) {
630                     shouldCompile = true;
631                     log("Compiling " + srcFile.getPath()
632                         + " because java file " + javaFile.getPath()
633                         + " is empty", Project.MSG_VERBOSE);
634                 }
635             }
636         }
637         return shouldCompile;
638     }

641     /**
642      * get a filename from our jsp file.
643      * @param mangler the jsp filename manager.
644      * @param srcFile the source file.
645      * @param srcDir  the source directory.
646      * @param dest    the destination directory.
647      * @return the filename.
648      * @todo support packages and subdirs
649      */
650     protected File mapToJavaFile(JspMangler mangler, File srcFile, File srcDir,
651                                  File dest) {
652         if (!srcFile.getName().endsWith(".jsp")) {
653             return null;
654         }
655         String javaFileName = mangler.mapJspToJavaName(srcFile);
656         //        String srcFileDir=srcFile.getParent();
657         return new File(dest, javaFileName);
658     }

660     /**
661      * delete any java output files that are empty
662      * this is to get around a little defect in jasper: when it
663      * fails, it leaves incomplete files around.
664      */
665     public void deleteEmptyJavaFiles() {
666         if (javaFiles != null) {
667             Enumeration e = javaFiles.elements();
668             while (e.hasMoreElements()) {
669                 File file = (File) e.nextElement();
670                 if (file.exists() && file.length() == 0) {
671                     log("deleting empty output file " + file);
672                     file.delete();
673                 }
674             }
675         }
676     }
