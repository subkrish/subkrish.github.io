59  /**
60   * Creates, Deletes, Records and Restores Symlinks.
61   *
62   * <p> This task performs several related operations. In the most trivial
63   * and default usage, it creates a link specified in the link attribute to
64   * a resource specified in the resource attribute. The second usage of this
65   * task is to traverse a directory structure specified by a fileset,
66   * and write a properties file in each included directory describing the
67   * links found in that directory. The third usage is to traverse a
68   * directory structure specified by a fileset, looking for properties files
69   * (also specified as included in the fileset) and recreate the links
70   * that have been previously recorded for each directory. Finally, it can be
71   * used to remove a symlink without deleting the associated resource.
72   *
73   * <p> Usage examples:
74   *
75   * <p> Make a link named &quot;foo&quot; to a resource named
76   * &quot;bar.foo&quot; in subdir:
77   * <pre>
78   * &lt;symlink link=&quot;${dir.top}/foo&quot; resource=&quot;${dir.top}/subdir/bar.foo&quot;/&gt;
79   * </pre>
80   *
81   * <p> Record all links in subdir and its descendants in files named
82   * &quot;dir.links&quot;:
83   * <pre>
84   * &lt;symlink action=&quot;record&quot; linkfilename=&quot;dir.links&quot;&gt;
85   *    &lt;fileset dir=&quot;${dir.top}&quot; includes=&quot;subdir&#47;**&quot; /&gt;
86   * &lt;/symlink&gt;
87   * </pre>
88   *
89   * <p> Recreate the links recorded in the previous example:
90   * <pre>
91   * &lt;symlink action=&quot;recreate&quot;&gt;
92   *    &lt;fileset dir=&quot;${dir.top}&quot; includes=&quot;subdir&#47;**&#47;dir.links&quot; /&gt;
93   * &lt;/symlink&gt;
94   * </pre>
95   *
96   * <p> Delete a link named &quot;foo&quot; to a resource named
97   * &quot;bar.foo&quot; in subdir:
98   * <pre>
99   * &lt;symlink action=&quot;delete&quot; link=&quot;${dir.top}/foo&quot;/&gt;
100  * </pre>
101  *
102  * <p><strong>LIMITATIONS:</strong> Because Java has no direct support for
103  * handling symlinks this task divines them by comparing canonical and
104  * absolute paths. On non-unix systems this may cause false positives.
105  * Furthermore, any operating system on which the command
106  * <code>ln -s link resource</code> is not a valid command on the command line
107  * will not be able to use action=&quot;delete&quot;, action=&quot;single&quot;
108  * or action=&quot;recreate&quot;, but action=&quot;record&quot; should still
109  * work. Finally, the lack of support for symlinks in Java means that all links
110  * are recorded as links to the <strong>canonical</strong> resource name.
111  * Therefore the link: <code>link --> subdir/dir/../foo.bar</code> will be
112  * recorded as <code>link=subdir/foo.bar</code> and restored as
113  * <code>link --> subdir/foo.bar</code>.
114  *
115  */

155     /**
156      * Create a symlink.
157      * @throws BuildException on error.
158      * @since Ant 1.7
159      */
160     public void single() throws BuildException {
161         try {
162             if (resource == null) {
163                 handleError("Must define the resource to symlink to!");
164                 return;
165             }
166             if (link == null) {
167                 handleError("Must define the link name for symlink!");
168                 return;
169             }
170             doLink(resource, link);
171         } finally {
172             setDefaults();
173         }
174     }

176     /**
177      * Delete a symlink.
178      * @throws BuildException on error.
179      * @since Ant 1.7
180      */
181     public void delete() throws BuildException {
182         try {
183             if (link == null) {
184                 handleError("Must define the link name for symlink!");
185                 return;
186             }
187             log("Removing symlink: " + link);
188             SYMLINK_UTILS.deleteSymbolicLink(FILE_UTILS
189                                              .resolveFile(new File("."), link),
190                                              this);
191         } catch (FileNotFoundException fnfe) {
192             handleError(fnfe.toString());
193         } catch (IOException ioe) {
194             handleError(ioe.toString());
195         } finally {
196             setDefaults();
197         }
198     }

200     /**
201      * Restore symlinks.
202      * @throws BuildException on error.
203      * @since Ant 1.7
204      */
205     public void recreate() throws BuildException {
206         try {
207             if (fileSets.isEmpty()) {
208                 handleError("File set identifying link file(s) "
209                             + "required for action recreate");
210                 return;
211             }
212             Properties links = loadLinks(fileSets);
213 
214             for (Iterator kitr = links.keySet().iterator(); kitr.hasNext();) {
215                 String lnk = (String) kitr.next();
216                 String res = links.getProperty(lnk);
217                 // handle the case where lnk points to a directory (bug 25181)
218                 try {
219                     File test = new File(lnk);
220                     if (!SYMLINK_UTILS.isSymbolicLink(lnk)) {
221                         doLink(res, lnk);
222                     } else if (!test.getCanonicalPath().equals(
223                         new File(res).getCanonicalPath())) {
224                         SYMLINK_UTILS.deleteSymbolicLink(test, this);
225                         doLink(res, lnk);
226                     } // else lnk exists, do nothing
227                 } catch (IOException ioe) {
228                     handleError("IO exception while creating link");
229                 }
230             }
231         } finally {
232             setDefaults();
233         }
234     }

236     /**
237      * Record symlinks.
238      * @throws BuildException on error.
239      * @since Ant 1.7
240      */
241     public void record() throws BuildException {
242         try {
243             if (fileSets.isEmpty()) {
244                 handleError("Fileset identifying links to record required");
245                 return;
246             }
247             if (linkFileName == null) {
248                 handleError("Name of file to record links in required");
249                 return;
250             }
251             // create a hashtable to group them by parent directory:
252             Hashtable byDir = new Hashtable();
253 
254             // get an Iterator of file objects representing links (canonical):
255             for (Iterator litr = findLinks(fileSets).iterator();
256                 litr.hasNext();) {
257                 File thisLink = (File) litr.next();
258                 File parent = thisLink.getParentFile();
259                 Vector v = (Vector) byDir.get(parent);
260                 if (v == null) {
261                     v = new Vector();
262                     byDir.put(parent, v);
263                 }
264                 v.addElement(thisLink);
265             }
266             // write a Properties file in each directory:
267             for (Iterator dirs = byDir.keySet().iterator(); dirs.hasNext();) {
268                 File dir = (File) dirs.next();
269                 Vector linksInDir = (Vector) byDir.get(dir);
270                 Properties linksToStore = new Properties();
271 
272                 // fill up a Properties object with link and resource names:
273                 for (Iterator dlnk = linksInDir.iterator(); dlnk.hasNext();) {
274                     File lnk = (File) dlnk.next();
275                     try {
276                         linksToStore.put(lnk.getName(), lnk.getCanonicalPath());
277                     } catch (IOException ioe) {
278                         handleError("Couldn't get canonical name of parent link");
279                     }
280                 }
281                 writePropertyFile(linksToStore, dir);
282             }
283         } finally {
284             setDefaults();
285         }
286     }

392     /**
393      * Delete a symlink (without deleting the associated resource).
394      *
395      * <p>This is a utility method that removes a unix symlink without removing
396      * the resource that the symlink points to. If it is accidentally invoked
397      * on a real file, the real file will not be harmed.</p>
398      *
399      * <p>This method works by
400      * getting the canonical path of the link, using the canonical path to
401      * rename the resource (breaking the link) and then deleting the link.
402      * The resource is then returned to its original name inside a finally
403      * block to ensure that the resource is unharmed even in the event of
404      * an exception.</p>
405      *
406      * <p>Since Ant 1.8.0 this method will try to delete the File object if
407      * it reports it wouldn't exist (as symlinks pointing nowhere usually do). 
408      * Prior version would throw a FileNotFoundException in that case.</p>
409      *
410      * @param linkfil    A <code>File</code> object of the symlink to delete.
411      *
412      * @throws IOException             If calls to <code>File.rename</code>,
413      *                                   <code>File.delete</code> or
414      *                                   <code>File.getCanonicalPath</code>
415      *                                   fail.
416      * @deprecated use
417      * org.apache.tools.ant.util.SymbolicLinkUtils#deleteSymbolicLink
418      * instead
419      */
420     public static void deleteSymlink(File linkfil)
421         throws IOException {
422         SYMLINK_UTILS.deleteSymbolicLink(linkfil, null);
423     }

425     /**
426      * Write a properties file. This method uses <code>Properties.store</code>
427      * and thus may throw exceptions that occur while writing the file.
428      *
429      * @param properties     The properties object to be written.
430      * @param dir            The directory for which we are writing the links.
431      * @throws BuildException if the property file could not be written
432      */
433     private void writePropertyFile(Properties properties, File dir)
434         throws BuildException {
435         BufferedOutputStream bos = null;
436         try {
437             bos = new BufferedOutputStream(
438                 new FileOutputStream(new File(dir, linkFileName)));
439             properties.store(bos, "Symlinks from " + dir);
440         } catch (IOException ioe) {
441             throw new BuildException(ioe, getLocation());
442         } finally {
443             FileUtils.close(bos);
444         }
445     }

461     /**
462      * Conduct the actual construction of a link.
463      *
464      * <p> The link is constructed by calling <code>Execute.runCommand</code>.
465      *
466      * @param res   The path of the resource we are linking to.
467      * @param lnk       The name of the link we wish to make.
468      * @throws BuildException when things go wrong
469      */
470     private void doLink(String res, String lnk) throws BuildException {
471         File linkfil = new File(lnk);
472         String options = "-s";
473         if (overwrite) {
474             options += "f";
475             if (linkfil.exists()) {
476                 try {
477                     SYMLINK_UTILS.deleteSymbolicLink(linkfil, this);
478                 } catch (FileNotFoundException fnfe) {
479                     log("Symlink disappeared before it was deleted: " + lnk);
480                 } catch (IOException ioe) {
481                     log("Unable to overwrite preexisting link or file: " + lnk,
482                         ioe, Project.MSG_INFO);
483                 }
484             }
485         }
486         String[] cmd = new String[] {"ln", options, res, lnk};
487         try {
488             Execute.runCommand(this, cmd);
489         } catch (BuildException failedToExecute) {
490             if (failonerror) {
491                 throw failedToExecute;
492             } else {
493                 //log at the info level, and keep going.
494                 log(failedToExecute.getMessage(), failedToExecute, Project.MSG_INFO);
495             }
496         }
497     }

499     /**
500      * Find all the links in all supplied filesets.
501      *
502      * <p> This method is invoked when the action attribute is
503      * &quot;record&quot;. This means that filesets are interpreted
504      * as the directories in which links may be found.
505      *
506      * @param v   The filesets specified by the user.
507      * @return A HashSet of <code>File</code> objects containing the
508      *         links (with canonical parent directories).
509      */
510     private HashSet findLinks(Vector v) {
511         HashSet result = new HashSet();
512         final int size = v.size();
513         for (int i = 0; i < size; i++) {
514             FileSet fs = (FileSet) v.get(i);
515             DirectoryScanner ds = fs.getDirectoryScanner(getProject());
516             String[][] fnd = new String[][]
517                 {ds.getIncludedFiles(), ds.getIncludedDirectories()};
518             File dir = fs.getDir(getProject());
519             for (int j = 0; j < fnd.length; j++) {
520                 for (int k = 0; k < fnd[j].length; k++) {
521                     try {
522                         File f = new File(dir, fnd[j][k]);
523                         File pf = f.getParentFile();
524                         String name = f.getName();
525                         if (SYMLINK_UTILS.isSymbolicLink(pf, name)) {
526                             result.add(new File(pf.getCanonicalFile(), name));
527                         }
528                     } catch (IOException e) {
529                         handleError("IOException: " + fnd[j][k] + " omitted");
530                     }
531                 }
532             }
533         }
534         return result;
535     }

537     /**
538      * Load links from properties files included in one or more FileSets.
539      *
540      * <p> This method is only invoked when the action attribute is set to
541      * &quot;recreate&quot;. The filesets passed in are assumed to specify the
542      * names of the property files with the link information and the
543      * subdirectories in which to look for them.
544      *
545      * @param v    The <code>FileSet</code>s for this task.
546      * @return            The links to be made.
547      */
548     private Properties loadLinks(Vector v) {
549         Properties finalList = new Properties();
550         // loop through the supplied file sets:
551         final int size = v.size();
552         for (int i = 0; i < size; i++) {
553             FileSet fs = (FileSet) v.elementAt(i);
554             DirectoryScanner ds = new DirectoryScanner();
555             fs.setupDirectoryScanner(ds, getProject());
556             ds.setFollowSymlinks(false);
557             ds.scan();
558             String[] incs = ds.getIncludedFiles();
559             File dir = fs.getDir(getProject());
560 
561             // load included files as properties files:
562             for (int j = 0; j < incs.length; j++) {
563                 File inc = new File(dir, incs[j]);
564                 File pf = inc.getParentFile();
565                 Properties lnks = new Properties();
566                 InputStream is = null;
567                 try {
568                     is = new BufferedInputStream(new FileInputStream(inc));
569                     lnks.load(is);
570                     pf = pf.getCanonicalFile();
571                 } catch (FileNotFoundException fnfe) {
572                     handleError("Unable to find " + incs[j] + "; skipping it.");
573                     continue;
574                 } catch (IOException ioe) {
575                     handleError("Unable to open " + incs[j]
576                                 + " or its parent dir; skipping it.");
577                     continue;
578                 } finally {
579                     FileUtils.close(is);
580                 }
581                 lnks.list(new PrintStream(
582                     new LogOutputStream(this, Project.MSG_INFO)));
583                 // Write the contents to our master list of links
584                 // This method assumes that all links are defined in
585                 // terms of absolute paths, or paths relative to the
586                 // working directory:
587                 for (Iterator kitr = lnks.keySet().iterator(); kitr.hasNext();) {
588                     String key = (String) kitr.next();
589                     finalList.put(new File(pf, key).getAbsolutePath(),
590                         lnks.getProperty(key));
591                 }
592             }
593         }
594         return finalList;
595     }
