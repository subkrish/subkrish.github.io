38  /**
39   * This object represents a path as used by CLASSPATH or PATH
40   * environment variable. A path might also be described as a collection
41   * of unique filesystem resources.
42   * <p>
43   * <code>
44   * &lt;sometask&gt;<br>
45   * &nbsp;&nbsp;&lt;somepath&gt;<br>
46   * &nbsp;&nbsp;&nbsp;&nbsp;&lt;pathelement location="/path/to/file.jar" /&gt;<br>
47   * &nbsp;&nbsp;&nbsp;&nbsp;&lt;pathelement
48   *  path="/path/to/file2.jar:/path/to/class2;/path/to/class3" /&gt;
49   * <br>
50   * &nbsp;&nbsp;&nbsp;&nbsp;&lt;pathelement location="/path/to/file3.jar" /&gt;<br>
51   * &nbsp;&nbsp;&nbsp;&nbsp;&lt;pathelement location="/path/to/file4.jar" /&gt;<br>
52   * &nbsp;&nbsp;&lt;/somepath&gt;<br>
53   * &lt;/sometask&gt;<br>
54   * </code>
55   * <p>
56   * The object implementation <code>sometask</code> must provide a method called
57   * <code>createSomepath</code> which returns an instance of <code>Path</code>.
58   * Nested path definitions are handled by the Path object and must be labeled
59   * <code>pathelement</code>.<p>
60   *
61   * The path element takes a parameter <code>path</code> which will be parsed
62   * and split into single elements. It will usually be used
63   * to define a path from an environment variable.
64   */
65

320     /**
321      * Same as addExisting, but support classpath behavior if tryUserDir
322      * is true. Classpaths are relative to user dir, not the project base.
323      * That used to break jspc test
324      *
325      * @param source the source path
326      * @param tryUserDir  if true try the user directory if the file is not present
327      */
328     public void addExisting(Path source, boolean tryUserDir) {
329         String[] list = source.list();
330         File userDir = (tryUserDir) ? new File(System.getProperty("user.dir"))
331                 : null;
332 
333         for (int i = 0; i < list.length; i++) {
334             File f = resolveFile(getProject(), list[i]);
335 
336             // probably not the best choice, but it solves the problem of
337             // relative paths in CLASSPATH
338             if (tryUserDir && !f.exists()) {
339                 f = new File(userDir, list[i]);
340             }
341             if (f.exists()) {
342                 setLocation(f);
343             } else if (f.getParentFile() != null && f.getParentFile().exists()
344                        && containsWildcards(f.getName())) {
345                 setLocation(f);
346                 log("adding " + f + " which contains wildcards and may not"
347                     + " do what you intend it to do depending on your OS or"
348                     + " version of Java", Project.MSG_VERBOSE);
349             } else {
350                 log("dropping " + f + " from path as it doesn't exist",
351                     Project.MSG_VERBOSE);
352             }
353         }
354     }

390     /**
391      * Splits a PATH (with : or ; as separators) into its parts.
392      * @param project the project to use
393      * @param source a <code>String</code> value
394      * @return an array of strings, one for each path element
395      */
396     public static String[] translatePath(Project project, String source) {
397         final Vector<String> result = new Vector<String>();
398         if (source == null) {
399             return new String[0];
400         }
401         PathTokenizer tok = new PathTokenizer(source);
402         StringBuffer element = new StringBuffer();
403         while (tok.hasMoreTokens()) {
404             String pathElement = tok.nextToken();
405             try {
406                 element.append(resolveFile(project, pathElement).getPath());
407             } catch (BuildException e) {
408                 project.log("Dropping path element " + pathElement
409                     + " as it is not valid relative to the project",
410                     Project.MSG_VERBOSE);
411             }
412             for (int i = 0; i < element.length(); i++) {
413                 translateFileSep(element, i);
414             }
415             result.addElement(element.toString());
416             element = new StringBuffer();
417         }
418         return result.toArray(new String[result.size()]);
419     }

421     /**
422      * Returns its argument with all file separator characters
423      * replaced so that they match the local OS conventions.
424      * @param source the path to convert
425      * @return the converted path
426      */
427     public static String translateFile(String source) {
428         if (source == null) {
429           return "";
430         }
431         final StringBuffer result = new StringBuffer(source);
432         for (int i = 0; i < result.length(); i++) {
433             translateFileSep(result, i);
434         }
435         return result.toString();
436     }

438     /**
439      * Translates occurrences at a position of / or \ to correct separator of the
440      * current platform and returns whether it had to do a
441      * replacement.
442      * @param buffer a buffer containing a string
443      * @param pos the position in the string buffer to convert
444      * @return true if the character was a / or \
445      */
446     protected static boolean translateFileSep(StringBuffer buffer, int pos) {
447         if (buffer.charAt(pos) == '/' || buffer.charAt(pos) == '\\') {
448             buffer.setCharAt(pos, File.separatorChar);
449             return true;
450         }
451         return false;
452     }

542     /**
543      * Concatenates a class path in the order specified by the
544      * ${build.sysclasspath} property - using the supplied value if
545      * ${build.sysclasspath} has not been set.
546      */
547     private Path concatSpecialPath(String defValue, Path p) {
548         Path result = new Path(getProject());
549 
550         String order = defValue;
551         String o = getProject() != null
552             ? getProject().getProperty(MagicNames.BUILD_SYSCLASSPATH)
553             : System.getProperty(MagicNames.BUILD_SYSCLASSPATH);
554         if (o != null) {
555             order = o;
556         }
557         if (order.equals("only")) {
558             // only: the developer knows what (s)he is doing
559             result.addExisting(p, true);
560 
561         } else if (order.equals("first")) {
562             // first: developer could use a little help
563             result.addExisting(p, true);
564             result.addExisting(this);
565 
566         } else if (order.equals("ignore")) {
567             // ignore: don't trust anyone
568             result.addExisting(this);
569 
570         } else {
571             // last: don't trust the developer
572             if (!order.equals("last")) {
573                 log("invalid value for " + MagicNames.BUILD_SYSCLASSPATH
574                     + ": " + order,
575                     Project.MSG_WARN);
576             }
577             result.addExisting(this);
578             result.addExisting(p, true);
579         }
580         return result;
581     }

583     /**
584      * Add the Java Runtime classes to this Path instance.
585      */
586     public void addJavaRuntime() {
587         if (JavaEnvUtils.isKaffe()) {
588             // newer versions of Kaffe (1.1.1+) won't have this,
589             // but this will be sorted by FileSet anyway.
590             File kaffeShare = new File(System.getProperty("java.home")
591                                        + File.separator + "share"
592                                        + File.separator + "kaffe");
593             if (kaffeShare.isDirectory()) {
594                 FileSet kaffeJarFiles = new FileSet();
595                 kaffeJarFiles.setDir(kaffeShare);
596                 kaffeJarFiles.setIncludes("*.jar");
597                 addFileset(kaffeJarFiles);
598             }
599         } else if ("GNU libgcj".equals(System.getProperty("java.vm.name"))) {
600             addExisting(systemBootClasspath);
601         }
602 
603         if (System.getProperty("java.vendor").toLowerCase(Locale.ENGLISH).indexOf("microsoft") >= 0) {
604             // TODO is this code still necessary? is there any 1.2+ port?
605             // Pull in *.zip from packages directory
606             FileSet msZipFiles = new FileSet();
607             msZipFiles.setDir(new File(System.getProperty("java.home")
608                 + File.separator + "Packages"));
609             msZipFiles.setIncludes("*.ZIP");
610             addFileset(msZipFiles);
611         } else {
612             // JDK 1.2+ seems to set java.home to the JRE directory.
613             addExisting(new Path(null,
614                                  System.getProperty("java.home")
615                                  + File.separator + "lib"
616                                  + File.separator + "rt.jar"));
617             // Just keep the old version as well and let addExisting
618             // sort it out.
619             addExisting(new Path(null,
620                                  System.getProperty("java.home")
621                                  + File.separator + "jre"
622                                  + File.separator + "lib"
623                                  + File.separator + "rt.jar"));
624 
625             // Sun's and Apple's 1.4 have JCE and JSSE in separate jars.
626             String[] secJars = {"jce", "jsse"};
627             for (int i = 0; i < secJars.length; i++) {
628                 addExisting(new Path(null,
629                                      System.getProperty("java.home")
630                                      + File.separator + "lib"
631                                      + File.separator + secJars[i] + ".jar"));
632                 addExisting(new Path(null,
633                                      System.getProperty("java.home")
634                                      + File.separator + ".."
635                                      + File.separator + "Classes"
636                                      + File.separator + secJars[i] + ".jar"));
637             }
638 
639             // IBM's 1.4 has rt.jar split into 4 smaller jars and a combined
640             // JCE/JSSE in security.jar.
641             String[] ibmJars
642                 = {"core", "graphics", "security", "server", "xml"};
643             for (int i = 0; i < ibmJars.length; i++) {
644                 addExisting(new Path(null,
645                                      System.getProperty("java.home")
646                                      + File.separator + "lib"
647                                      + File.separator + ibmJars[i] + ".jar"));
648             }
649 
650             // Added for MacOS X
651             addExisting(new Path(null,
652                                  System.getProperty("java.home")
653                                  + File.separator + ".."
654                                  + File.separator + "Classes"
655                                  + File.separator + "classes.jar"));
656             addExisting(new Path(null,
657                                  System.getProperty("java.home")
658                                  + File.separator + ".."
659                                  + File.separator + "Classes"
660                                  + File.separator + "ui.jar"));
661         }
662     }

664     /**
665      * Emulation of extdirs feature in java >= 1.2.
666      * This method adds all files in the given
667      * directories (but not in sub-directories!) to the classpath,
668      * so that you don't have to specify them all one by one.
669      * @param extdirs - Path to append files to
670      */
671     public void addExtdirs(Path extdirs) {
672         if (extdirs == null) {
673             String extProp = System.getProperty("java.ext.dirs");
674             if (extProp != null) {
675                 extdirs = new Path(getProject(), extProp);
676             } else {
677                 return;
678             }
679         }
680 
681         String[] dirs = extdirs.list();
682         for (int i = 0; i < dirs.length; i++) {
683             File dir = resolveFile(getProject(), dirs[i]);
684             if (dir.exists() && dir.isDirectory()) {
685                 FileSet fs = new FileSet();
686                 fs.setDir(dir);
687                 fs.setIncludes("*");
688                 addFileset(fs);
689             }
690         }
691     }

726      * @param rc the ResourceCollection to check.
727      * @throws BuildException if <code>rc</code> is not filesystem-only.
728      * @return the passed in ResourceCollection.
729      */
730     protected ResourceCollection assertFilesystemOnly(ResourceCollection rc) {
731         if (rc != null && !(rc.isFilesystemOnly())) {
732             throw new BuildException(getDataTypeName()
733                 + " allows only filesystem resources.");
734         }
735         return rc;
736     }

738     /**
739      * Helps determine whether to preserve BC by calling <code>list()</code> on subclasses.
740      * The default behavior of this method is to return <code>true</code> for any subclass
741      * that implements <code>list()</code>; this can, of course, be avoided by overriding
742      * this method to return <code>false</code>. It is not expected that the result of this
743      * method should change over time, thus it is called only once.
744      * @return <code>true</code> if <code>iterator()</code> should delegate to <code>list()</code>.
745      */
746     protected boolean delegateIteratorToList() {
747         if (getClass().equals(Path.class)) {
748             return false;
749         }
750         try {
751             Method listMethod = getClass().getMethod("list", (Class[]) null);
752             return !listMethod.getDeclaringClass().equals(Path.class);
753         } catch (Exception e) {
754             //shouldn't happen, but
755             return false;
756         }
757     }
758 
759     private synchronized boolean getPreserveBC() {
760         if (preserveBC == null) {
761             preserveBC = delegateIteratorToList() ? Boolean.TRUE : Boolean.FALSE;
762         }
763         return preserveBC.booleanValue();
764     }

766     /**
767      * Does the given file name contain wildcards?
768      * @since Ant 1.8.2
769      */
770     private static boolean containsWildcards(String path) {
771         return path != null
772             && (path.indexOf("*") > -1 || path.indexOf("?") > -1);
