23  import java.util.HashSet;

32  /**
33   * This class implements a security manager meant for usage by tasks that run inside the
34   * Ant VM. An examples are the Java Task and JUnitTask.
35   *
36   * The basic functionality is that nothing (except for a base set of permissions) is allowed, unless
37   * the permission is granted either explicitly or implicitly.
38   * If a permission is granted this can be overruled by explicitly revoking the permission.
39   *
40   * It is not permissible to add permissions (either granted or revoked) while the Security Manager
41   * is active (after calling setSecurityManager() but before calling restoreSecurityManager()).
42   *
43   * @since Ant 1.6
44   */
45  public class Permissions {
46  
47      private List<Permission> grantedPermissions = new LinkedList<Permission>();
48      private List<Permission> revokedPermissions = new LinkedList<Permission>();
49      private java.security.Permissions granted = null;
50      private SecurityManager origSm = null;
51      private boolean active = false;
52      private boolean delegateToOldSM;
53  
54      // Mandatory constructor for permission object.
55      private static final Class<?>[] PARAMS = {String.class, String.class};
56  
57      /**
58       * Create a set of Permissions.  Equivalent to calling
59       * <code>new Permissions(false)</code>.
60       */
61      public Permissions() {
62          this(false);
63      }
64  
65      /**
66       * Create a set of permissions.
67       * @param delegateToOldSM  if <code>true</code> the old security manager
68       * will be used if the permission has not been explicitly granted or revoked
69       * in this instance.
70       */
71      public Permissions(boolean delegateToOldSM) {
72          this.delegateToOldSM = delegateToOldSM;
73      }
74  
75      /**
76       * Adds a permission to be granted.
77       * @param perm The Permissions.Permission to be granted.
78       */
79      public void addConfiguredGrant(Permissions.Permission perm) {
80          grantedPermissions.add(perm);
81      }
82  
83      /**
84       * Adds a permission to be revoked.
85       * @param perm The Permissions.Permission to be revoked
86       */
87      public void addConfiguredRevoke(Permissions.Permission perm) {
88          revokedPermissions.add(perm);
89      }
90  
91      /**
92       * To be used by tasks wishing to use this security model before executing the part to be
93       * subject to these Permissions. Note that setting the SecurityManager too early may
94       * prevent your part from starting, as for instance changing classloaders may be prohibited.
95       * The classloader for the new situation is supposed to be present.
96       * @throws BuildException on error
97       */
98      public synchronized void setSecurityManager() throws BuildException {
99          origSm = System.getSecurityManager();
100         init();
101         System.setSecurityManager(new MySM());
102         active = true;
103     }
104 
105     /**
106      * Initializes the list of granted permissions, checks the list of revoked permissions.
107      */
108     private void init() throws BuildException {
109         granted = new java.security.Permissions();
110         for (Permissions.Permission p : revokedPermissions) {
111             if (p.getClassName() == null) {
112                 throw new BuildException("Revoked permission " + p + " does not contain a class.");
113             }
114         }
115         for (Permissions.Permission p : grantedPermissions) {
116             if (p.getClassName() == null) {
117                 throw new BuildException("Granted permission " + p
118                         + " does not contain a class.");
119             } else {
120                 java.security.Permission perm = createPermission(p);
121                 granted.add(perm);
122             }
123         }
124         // Add base set of permissions
125         granted.add(new java.net.SocketPermission("localhost:1024-", "listen"));
126         granted.add(new java.util.PropertyPermission("java.version", "read"));
127         granted.add(new java.util.PropertyPermission("java.vendor", "read"));
128         granted.add(new java.util.PropertyPermission("java.vendor.url", "read"));
129         granted.add(new java.util.PropertyPermission("java.class.version", "read"));
130         granted.add(new java.util.PropertyPermission("os.name", "read"));
131         granted.add(new java.util.PropertyPermission("os.version", "read"));
132         granted.add(new java.util.PropertyPermission("os.arch", "read"));
133         granted.add(new java.util.PropertyPermission("file.encoding", "read"));
134         granted.add(new java.util.PropertyPermission("file.separator", "read"));
135         granted.add(new java.util.PropertyPermission("path.separator", "read"));
136         granted.add(new java.util.PropertyPermission("line.separator", "read"));
137         granted.add(new java.util.PropertyPermission("java.specification.version", "read"));
138         granted.add(new java.util.PropertyPermission("java.specification.vendor", "read"));
139         granted.add(new java.util.PropertyPermission("java.specification.name", "read"));
140         granted.add(new java.util.PropertyPermission("java.vm.specification.version", "read"));
141         granted.add(new java.util.PropertyPermission("java.vm.specification.vendor", "read"));
142         granted.add(new java.util.PropertyPermission("java.vm.specification.name", "read"));
143         granted.add(new java.util.PropertyPermission("java.vm.version", "read"));
144         granted.add(new java.util.PropertyPermission("java.vm.vendor", "read"));
145         granted.add(new java.util.PropertyPermission("java.vm.name", "read"));
146     }
147 
148     private java.security.Permission createPermission(
149             Permissions.Permission permission) {
150         try {
151             // First add explicitly already resolved permissions will not be
152             // resolved when added as unresolved permission.
153             Class<? extends java.security.Permission> clazz = Class.forName(
154                     permission.getClassName()).asSubclass(java.security.Permission.class);
155             String name = permission.getName();
156             String actions = permission.getActions();
157             Constructor<? extends java.security.Permission> ctr = clazz.getConstructor(PARAMS);
158             return ctr.newInstance(new Object[] { name, actions });
159         } catch (Exception e) {
160             // Let the UnresolvedPermission handle it.
161             return new UnresolvedPermission(permission.getClassName(),
162                     permission.getName(), permission.getActions(), null);
163         }
164     }
165 
166     /**
167      * To be used by tasks that just finished executing the parts subject to these permissions.
168      */
169     public synchronized void restoreSecurityManager() {
170         active = false;
171         System.setSecurityManager(origSm);
172     }
173 
174     /**
175      * This inner class implements the actual SecurityManager that can be used by tasks
176      * supporting Permissions.
177      */
178     private class MySM extends SecurityManager {
179 
180         /**
181          * Exit is treated in a special way in order to be able to return the exit code
182          * towards tasks.
183          * An ExitException is thrown instead of a simple SecurityException to indicate the exit
184          * code.
185          * Overridden from java.lang.SecurityManager
186          * @param status The exit status requested.
187          */
188         public void checkExit(int status) {
189             java.security.Permission perm = new java.lang.RuntimePermission("exitVM", null);
190             try {
191                 checkPermission(perm);
192             } catch (SecurityException e) {
193                 throw new ExitException(e.getMessage(), status);
194             }
195         }
196 
197         /**
198          * The central point in checking permissions.
199          * Overridden from java.lang.SecurityManager
200          *
201          * @param perm The permission requested.
202          */
203         public void checkPermission(java.security.Permission perm) {
204             if (active) {
205                 if (delegateToOldSM && !perm.getName().equals("exitVM")) {
206                     boolean permOK = false;
207                     if (granted.implies(perm)) {
208                         permOK = true;
209                     }
210                     checkRevoked(perm);
211                     /*
212                      if the permission was not explicitly granted or revoked
213                      the original security manager will do its work
214                     */
215                     if (!permOK && origSm != null) {
216                         origSm.checkPermission(perm);
217                     }
218                 }  else {
219                     if (!granted.implies(perm)) {
220                         throw new SecurityException("Permission " + perm + " was not granted.");
221                     }
222                     checkRevoked(perm);
223                 }
224             }
225         }
226 
227         /**
228          * throws an exception if this permission is revoked
229          * @param perm the permission being checked
230          */
231         private void checkRevoked(java.security.Permission perm) {
232             for (Permissions.Permission revoked : revokedPermissions) {
233                 if (revoked.matches(perm)) {
234                     throw new SecurityException("Permission " + perm + " was revoked.");
235                 }
236             }
237         }
238     }

240     /** Represents a permission. */
241     public static class Permission {
242         private String className;
243         private String name;
244         private String actionString;
245         private Set<String> actions;
246 
247         /**
248          * Set the class, mandatory.
249          * @param aClass The class name of the permission.
250          */
251         public void setClass(String aClass) {
252             className = aClass.trim();
253         }
254 
255         /**
256          * Get the class of the permission.
257          * @return The class name of the permission.
258          */
259         public String getClassName() {
260             return className;
261         }
262 
263         /**
264          * Set the name of the permission.
265          * @param aName The name of the permission.
266          */
267         public void setName(String aName) {
268             name = aName.trim();
269         }
270 
271         /**
272          * Get the name of the permission.
273          * @return The name of the permission.
274          */
275         public String getName() {
276             return name;
277         }
278 
279         /**
280          * Set the actions.
281          * @param actions The actions of the permission.
282          */
283         public void setActions(String actions) {
284             actionString = actions;
285             if (actions.length() > 0) {
286                 this.actions = parseActions(actions);
287             }
288         }
289 
290         /**
291          * Get the actions.
292          * @return The actions of the permission.
293          */
294         public String getActions() {
295             return actionString;
296         }
297 
298         /**
299          * Learn whether the permission matches in case of a revoked permission.
300          * @param perm The permission to check against.
301          */
302         boolean matches(java.security.Permission perm) {
303             if (!className.equals(perm.getClass().getName())) {
304                 return false;
305             }
306             if (name != null) {
307                 if (name.endsWith("*")) {
308                     if (!perm.getName().startsWith(name.substring(0, name.length() - 1))) {
309                         return false;
310                     }
311                 } else {
312                     if (!name.equals(perm.getName())) {
313                         return false;
314                     }
315                 }
316             }
317             if (actions != null) {
318                 Set<String> as = parseActions(perm.getActions());
319                 int size = as.size();
320                 as.removeAll(actions);
321                 if (as.size() == size) {
322                     // None of the actions revoked, so all allowed.
323                     return false;
324                 }
325             }
326             return true;
327         }
328 
329         /**
330          * Parses the actions into a set of separate strings.
331          * @param actions The actions to be parsed.
332          */
333         private Set<String> parseActions(String actions) {
334             Set<String> result = new HashSet<String>();
335             StringTokenizer tk = new StringTokenizer(actions, ",");
336             while (tk.hasMoreTokens()) {
337                 String item = tk.nextToken().trim();
338                 if (!item.equals("")) {
339                     result.add(item);
340                 }
341             }
342             return result;
343         }
344 
345         /**
346          * Get a string description of the permissions.
347          * @return string description of the permissions.
348          */
349         public String toString() {
350             return ("Permission: " + className + " (\"" + name + "\", \"" + actions + "\")");
351         }
352     }
