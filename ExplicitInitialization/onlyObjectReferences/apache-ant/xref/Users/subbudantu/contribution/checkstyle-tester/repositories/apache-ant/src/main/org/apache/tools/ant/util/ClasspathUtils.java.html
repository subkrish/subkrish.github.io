30  /**
31   * Offers some helper methods on the Path structure in ant.
32   *
33   * <p>The basic idea behind this utility class is to use it from inside the
34   * different Ant objects (and user defined objects) that need classLoading
35   * for their operation.
36   * Normally those would have a setClasspathRef() {for the @classpathref}
37   * and/or a createClasspath() {for the nested &lt;classpath&gt;}
38   * Typically one would have in your Ant Task or DataType</p>
39   *
40   * <pre><code>
41   * ClasspathUtils.Delegate cpDelegate;
42   *
43   * public void init() {
44   *     this.cpDelegate = ClasspathUtils.getDelegate(this);
45   *     super.init();
46   * }
47   *
48   * public void setClasspathRef(Reference r) {
49   *     this.cpDelegate.setClasspathRef(r);
50   * }
51   *
52   * public Path createClasspath() {
53   *     return this.cpDelegate.createClasspath();
54   * }
55   *
56   * public void setClassname(String fqcn) {
57   *     this.cpDelegate.setClassname(fqcn);
58   * }
59   * </code></pre>
60   *
61   * <p>At execution time, when you actually need the classloading
62   * you can just:</p>
63   *
64   * <pre><code>
65   *     Object o = this.cpDelegate.newInstance();
66   * </code></pre>
67   *
68   * @since Ant 1.6
69   */
70  public class ClasspathUtils {
71  
72      /**
73       * Name of the magic property that controls classloader reuse in Ant 1.4.
74       */
75      public static final String REUSE_LOADER_REF = MagicNames.REFID_CLASSPATH_REUSE_LOADER;

77      /**
78       * Convenience overloaded version of {@link
79       * #getClassLoaderForPath(Project, Reference, boolean)}.
80       *
81       * <p>Assumes the logical 'false' for the reverseLoader.</p>
82       *
83       * @param p the project
84       * @param ref the reference
85       * @return The class loader
86       */
87      public static ClassLoader getClassLoaderForPath(Project p, Reference ref) {
88          return getClassLoaderForPath(p, ref, false);
89      }
90  
91      /**
92       * Convenience overloaded version of {@link #getClassLoaderForPath(Project, Path,
93       * String, boolean)}.
94       *
95       * <p>Delegates to the other one after extracting the referenced
96       * Path from the Project. This checks also that the passed
97       * Reference is pointing to a Path all right.</p>
98       * @param p current Ant project
99       * @param ref Reference to Path structure
100      * @param reverseLoader if set to true this new loader will take
101      * precedence over its parent (which is contra the regular
102      * classloader behaviour)
103      * @return The class loader
104      */
105     public static ClassLoader getClassLoaderForPath(
106         Project p, Reference ref, boolean reverseLoader) {
107         String pathId = ref.getRefId();
108         Object path = p.getReference(pathId);
109         if (!(path instanceof Path)) {
110             throw new BuildException("The specified classpathref " + pathId
111                     + " does not reference a Path.");
112         }
113         String loaderId = MagicNames.REFID_CLASSPATH_LOADER_PREFIX + pathId;
114         return getClassLoaderForPath(p, (Path) path, loaderId, reverseLoader);
115     }

120      *
121      * <p>Assumes the logical 'false' for the reverseLoader.</p>
122      *
123      * @param p current Ant project
124      * @param path the path
125      * @param loaderId the loader id string
126      * @return The class loader
127      */
128     public static ClassLoader getClassLoaderForPath(Project p, Path path, String loaderId) {
129         return getClassLoaderForPath(p, path, loaderId, false);

132     /**
133      * Convenience overloaded version of {@link
134      * #getClassLoaderForPath(Project, Path, String, boolean, boolean)}.
135      *
136      * <p>Sets value for 'reuseLoader' to true if the magic property
137      * has been set.</p>
138      *
139      * @param p the project
140      * @param path the path
141      * @param loaderId the loader id string
142      * @param reverseLoader if set to true this new loader will take
143      * precedence over its parent (which is contra the regular
144      * classloader behaviour)
145      * @return The class loader
146      */
147     public static ClassLoader getClassLoaderForPath(
148             Project p, Path path, String loaderId, boolean reverseLoader) {
149         return getClassLoaderForPath(p, path, loaderId, reverseLoader, isMagicPropertySet(p));
150     }

152     /**
153      * Gets a classloader that loads classes from the classpath
154      * defined in the path argument.
155      *
156      * <p>Based on the setting of the magic property
157      * 'ant.reuse.loader' this will try to reuse the previously
158      * created loader with that id, and of course store it there upon
159      * creation.</p>
160      * @param p             Ant Project where the handled components are living in.
161      * @param path          Path object to be used as classpath for this classloader
162      * @param loaderId      identification for this Loader,
163      * @param reverseLoader if set to true this new loader will take
164      *                      precedence over its parent (which is contra the regular
165      *                      classloader behaviour)
166      * @param reuseLoader   if true reuse the loader if it is found
167      * @return              ClassLoader that uses the Path as its classpath.
168      */
169     public static ClassLoader getClassLoaderForPath(
170             Project p, Path path, String loaderId, boolean reverseLoader, boolean reuseLoader) {
171         ClassLoader cl = null;
172 
173         // magic property
174         if (loaderId != null && reuseLoader) {
175             Object reusedLoader = p.getReference(loaderId);
176             if (reusedLoader != null && !(reusedLoader instanceof ClassLoader)) {
177                 throw new BuildException("The specified loader id " + loaderId
178                         + " does not reference a class loader");
179             }
180             cl = (ClassLoader) reusedLoader;
181         }
182         if (cl == null) {
183             cl = getUniqueClassLoaderForPath(p, path, reverseLoader);
184             if (loaderId != null && reuseLoader) {
185                 p.addReference(loaderId, cl);
186             }
187         }
188         return cl;
189     }
190 
191     /**
192      * Gets a fresh, different, previously unused classloader that uses the
193      * passed path as its classpath.
194      *
195      * <p>This method completely ignores the ant.reuse.loader magic
196      * property and should be used with caution.</p>
197      * @param p             Ant Project where the handled components are living in.
198      * @param path          the classpath for this loader
199      * @param reverseLoader if set to true this new loader will take
200      *                      precedence over its parent (which is contra the regular
201      *                      classloader behaviour)
202      * @return The fresh, different, previously unused class loader.
203      */
204     public static ClassLoader getUniqueClassLoaderForPath(Project p, Path path,
205             boolean reverseLoader) {
206         AntClassLoader acl = p.createClassLoader(path);
207         if (reverseLoader) {
208             acl.setParentFirst(false);
209             acl.addJavaLibraries();
210         }
211         return acl;
212     }

214     /**
215      * Creates a fresh object instance of the specified classname.
216      *
217      * <p> This uses the userDefinedLoader to load the specified class,
218      * and then makes an instance using the default no-argument constructor.
219      * </p>
220      *
221      * @param className the full qualified class name to load.
222      * @param userDefinedLoader the classloader to use.
223      * @return The fresh object instance
224      * @throws BuildException when loading or instantiation failed.
225      */
226     public static Object newInstance(String className, ClassLoader userDefinedLoader) {
227         return newInstance(className, userDefinedLoader, Object.class);
228     }
229 
230     /**
231      * Creates a fresh object instance of the specified classname.
232      *
233      * <p> This uses the userDefinedLoader to load the specified class,
234      * and then makes an instance using the default no-argument constructor.
235      * </p>
236      *
237      * @param className the full qualified class name to load.
238      * @param userDefinedLoader the classloader to use.
239      * @param expectedType the Class that the result should be assignment
240      * compatible with. (No ClassCastException will be thrown in case
241      * the result of this method is casted to the expectedType)
242      * @return The fresh object instance
243      * @throws BuildException when loading or instantiation failed.
244      * @since Ant 1.7
245      */
246     public static Object newInstance(String className, ClassLoader userDefinedLoader,
247             Class expectedType) {
248         try {
249             Class clazz = Class.forName(className, true, userDefinedLoader);
250             Object o = clazz.newInstance();
251             if (!expectedType.isInstance(o)) {
252                 throw new BuildException("Class of unexpected Type: " + className + " expected :"
253                         + expectedType);
254             }
255             return o;
256         } catch (ClassNotFoundException e) {
257             throw new BuildException("Class not found: " + className, e);
258         } catch (InstantiationException e) {
259             throw new BuildException("Could not instantiate " + className
260                     + ". Specified class should have a no " + "argument constructor.", e);
261         } catch (IllegalAccessException e) {
262             throw new BuildException("Could not instantiate " + className
263                     + ". Specified class should have a " + "public constructor.", e);
264         } catch (LinkageError e) {
265             throw new BuildException("Class " + className
266                     + " could not be loaded because of an invalid dependency.", e);
267         }
268     }

283      * for <taskdef> and <typedef> in Ant 1.5 and earlier.

289     /**
290      * Delegate that helps out any specific ProjectComponent that needs
291      * dynamic classloading.
292      *
293      * <p>Ant ProjectComponents that need a to be able to dynamically load
294      * Classes and instantiate them often expose the following ant syntax
295      * sugar: </p>
296      *
297      * <ul><li> nested &lt;classpath&gt; </li>
298      * <li> attribute @classpathref </li>
299      * <li> attribute @classname </li></ul>
300      *
301      * <p> This class functions as a delegate handling the configuration
302      * issues for this recurring pattern.  Its usage pattern, as the name
303      * suggests, is delegation rather than inheritance. </p>
304      *
305      * @since Ant 1.6
306      */
307     public static class Delegate {
308         private final ProjectComponent component;
309         private Path classpath;
310         private String classpathId;
311         private String className;
312         private String loaderId;
313         private boolean reverseLoader = false;
314 
315         /**
316          * Construct a Delegate
317          * @param component the ProjectComponent this delegate is for.
318          */
319         Delegate(ProjectComponent component) {
320             this.component = component;
321         }
322 
323         /**
324          * This method is a Delegate method handling the @classpath attribute.
325          *
326          * <p>This attribute can set a path to add to the classpath.</p>
327          *
328          * @param classpath the path to use for the classpath.
329          */
330         public void setClasspath(Path classpath) {
331             if (this.classpath == null) {
332                 this.classpath = classpath;
333             } else {
334                 this.classpath.append(classpath);
335             }
336         }
337 
338         /**
339          * Delegate method handling the &lt;classpath&gt; tag.
340          *
341          * <p>This nested path-like structure can set a path to add to the
342          * classpath.</p>
343          *
344          * @return the created path.
345          */
346         public Path createClasspath() {
347             if (this.classpath == null) {
348                 this.classpath = new Path(component.getProject());
349             }
350             return this.classpath.createPath();
351         }
352 
353         /**
354          * Delegate method handling the @classname attribute.
355          *
356          * <p>This attribute sets the full qualified class name of the class
357          * to load and instantiate.</p>
358          *
359          * @param fcqn the name of the class to load.
360          */
361         public void setClassname(String fcqn) {
362             this.className = fcqn;
363         }
364 
365         /**
366          * Delegate method handling the @classpathref attribute.
367          *
368          * <p>This attribute can add a referenced path-like structure to the
369          * classpath.</p>
370          *
371          * @param r the reference to the classpath.
372          */
373         public void setClasspathref(Reference r) {
374             this.classpathId = r.getRefId();
375             createClasspath().setRefid(r);
376         }
377 
378         /**
379          * Delegate method handling the @reverseLoader attribute.
380          *
381          * <p>This attribute can set a boolean indicating that the used
382          * classloader should NOT follow the classical parent-first scheme.
383          * </p>
384          *
385          * <p>By default this is supposed to be false.</p>
386          *
387          * <p>Caution: this behaviour is contradictory to the normal way
388          * classloaders work.  Do not let your ProjectComponent use it if
389          * you are not really sure.</p>
390          *
391          * @param reverseLoader if true reverse the order of looking up a class.
392          */
393         public void setReverseLoader(boolean reverseLoader) {
394             this.reverseLoader = reverseLoader;
395         }
396 
397         /**
398          * Sets the loaderRef.
399          * @param r the reference to the loader.
400          */
401         public void setLoaderRef(Reference r) {
402             this.loaderId = r.getRefId();
403         }
404 
405 
406         /**
407          * Finds or creates the classloader for this object.
408          * @return The class loader.
409          */
410         public ClassLoader getClassLoader() {
411             return getClassLoaderForPath(getContextProject(), classpath, getClassLoadId(),
412                     reverseLoader, loaderId != null || isMagicPropertySet(getContextProject()));
413         }
414 
415         /**
416          * The project of the ProjectComponent we are working for.
417          */
418         private Project getContextProject() {
419             return component.getProject();
420         }
421 
422         /**
423          * Computes the loaderId based on the configuration of the component.
424          * @return a loader identifier.
425          */
426         public String getClassLoadId() {
427             if (loaderId == null && classpathId != null) {
428                 return MagicNames.REFID_CLASSPATH_LOADER_PREFIX + classpathId;
429             } else {
430                 return loaderId;
431             }
432         }
433 
434         /**
435          * Helper method obtaining a fresh instance of the class specified
436          * in the @classname and using the specified classpath.
437          *
438          * @return the fresh instantiated object.
439          */
440         public Object newInstance() {
441             return ClasspathUtils.newInstance(this.className, getClassLoader());
442         }
443 
444         /**
445          * The classpath.
446          * @return the classpath.
447          */
448         public Path getClasspath() {
449             return classpath;
450         }
451 
452         /**
453          * Get the reverseLoader setting.
454          * @return true if looking up in reverse order.
455          */
456         public boolean isReverseLoader() {
457             return reverseLoader;
458         }

462     }
