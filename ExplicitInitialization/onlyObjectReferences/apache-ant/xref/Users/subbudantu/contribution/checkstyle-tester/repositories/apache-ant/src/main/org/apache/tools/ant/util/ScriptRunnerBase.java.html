39  /**
40   * This is a common abstract base case for script runners.
41   * These classes need to implement executeScript, evaluateScript
42   * and supportsLanguage.
43   * @since Ant 1.7.0
44   */
45  public abstract class ScriptRunnerBase {
46      /** Whether to keep the engine between calls to execute/eval */
47      private boolean keepEngine = false;
48  
49      /** Script language */
50      private String language;
51  
52      /** Script content */
53      private String script = "";
54  
55      /** Project this runner is used in */
56      private Project project;
57  
58      /** Classloader to be used when running the script. */
59      private ClassLoader scriptLoader;
60  
61      /** Beans to be provided to the script */
62      private Map beans = new HashMap();
63  
64      /**
65       * Add a list of named objects to the list to be exported to the script
66       *
67       * @param dictionary a map of objects to be placed into the script context
68       *        indexed by String names.
69       */
70      public void addBeans(Map dictionary) {
71          for (Iterator i = dictionary.keySet().iterator(); i.hasNext();) {
72              String key = (String) i.next();
73              try {
74                  Object val = dictionary.get(key);
75                  addBean(key, val);
76              } catch (BuildException ex) {
77                  // The key is in the dictionary but cannot be retrieved
78                  // This is usually due references that refer to tasks
79                  // that have not been taskdefed in the current run.
80                  // Ignore
81              }
82          }
83      }
84  
85      /**
86       * Add a single object into the script context.
87       *
88       * @param key the name in the context this object is to stored under.
89       * @param bean the object to be stored in the script context.
90       */
91      public void addBean(String key, Object bean) {
92          boolean isValid = key.length() > 0
93              && Character.isJavaIdentifierStart(key.charAt(0));
94  
95          for (int i = 1; isValid && i < key.length(); i++) {
96              isValid = Character.isJavaIdentifierPart(key.charAt(i));
97          }
98  
99          if (isValid) {
100             beans.put(key, bean);
101         }
102     }
103 
104     /**
105      * Get the beans used for the script.
106      * @return the map of beans.
107      */
108     protected Map getBeans() {
109         return beans;
110     }
111 
112     /**
113      * Do the work.
114      * @param execName the name that will be passed to BSF for this script
115      *        execution.
116      */
117     public abstract void executeScript(String execName);
118 
119     /**
120      * Evaluate the script.
121      * @param execName the name that will be passed to the
122      *                 scripting engine for this script execution.
123      * @return the result of evaluating the script.
124      */
125     public abstract Object evaluateScript(String execName);
126 
127     /**
128      * Check if a script engine can be created for
129      * this language.
130      * @return true if a script engine can be created, false
131      *              otherwise.
132      */
133     public abstract boolean supportsLanguage();
134 
135     /**
136      * Get the name of the manager prefix used for this
137      * scriptrunner.
138      * @return the prefix string.
139      */
140     public abstract String getManagerName();
141 
142     /**
143      * Defines the language (required).
144      * @param language the scripting language name for the script.
145      */
146     public void setLanguage(String language) {
147         this.language = language;
148     }
149 
150     /**
151      * Get the script language
152      * @return the script language
153      */
154     public String getLanguage() {
155         return language;
156     }
157 
158     /**
159      * Set the script classloader.
160      * @param classLoader the classloader to use.
161      */
162     public void setScriptClassLoader(ClassLoader classLoader) {
163         this.scriptLoader = classLoader;
164     }
165 
166     /**
167      * Get the classloader used to load the script engine.
168      * @return the classloader.
169      */
170     protected ClassLoader getScriptClassLoader() {
171         return scriptLoader;
172     }
173 
174     /**
175      * Whether to keep the script engine between calls.
176      * @param keepEngine if true, keep the engine.
177      */
178     public void setKeepEngine(boolean keepEngine) {
179         this.keepEngine = keepEngine;
180     }
181 
182     /**
183      * Get the keep engine attribute.
184      * @return the attribute.
185      */
186     public boolean getKeepEngine() {
187         return keepEngine;
188     }
189 
190     /**
191      * Load the script from an external file; optional.
192      * @param file the file containing the script source.
193      */
194     public void setSrc(File file) {
195         String filename = file.getPath();
196         if (!file.exists()) {
197             throw new BuildException("file " + filename + " not found.");
198         }
199         try {
200             readSource(new FileReader(file), filename);
201         } catch (FileNotFoundException e) {
202             //this can only happen if the file got deleted a short moment ago
203             throw new BuildException("file " + filename + " not found.");
204         }
205     }
206 
207     /**
208      * Read some source in from the given reader
209      * @param reader the reader; this is closed afterwards.
210      * @param name the name to use in error messages
211      */
212     private void readSource(Reader reader, String name) {
213         BufferedReader in = null;
214         try {
215             in = new BufferedReader(reader);
216             script += FileUtils.safeReadFully(in);
217         } catch (IOException ex) {
218             throw new BuildException("Failed to read " + name, ex);
219         } finally {
220             FileUtils.close(in);
221         }
222     }
223 
224 
225     /**
226      * Add a resource to the source list.
227      * @since Ant 1.7.1
228      * @param sourceResource the resource to load
229      * @throws BuildException if the resource cannot be read
230      */
231     public void loadResource(Resource sourceResource) {
232         String name = sourceResource.toLongString();
233         InputStream in = null;
234         try {
235             in = sourceResource.getInputStream();
236         } catch (IOException e) {
237             throw new BuildException("Failed to open " + name, e);
238         } catch (UnsupportedOperationException e) {
239             throw new BuildException(
240                 "Failed to open " + name + " -it is not readable", e);
241         }
242         readSource(new InputStreamReader(in), name);
243     }
244 
245     /**
246      * Add all resources in a resource collection to the source list.
247      * @since Ant 1.7.1
248      * @param collection the resource to load
249      * @throws BuildException if a resource cannot be read
250      */
251     public void loadResources(ResourceCollection collection) {
252         for (Resource resource : collection) {
253             loadResource(resource);
254         }
255     }

261      */
262     public void addText(String text) {
263         script += text;
264     }

270     public String getScript() {
271         return script;
272     }

275      * Clear the current script text content.
276      */

278         this.script = "";
279     }

297     /**
298      * Bind the runner to a project component.
299      * Properties, targets and references are all added as beans;
300      * project is bound to project, and self to the component.
301      * @param component to become <code>self</code>
302      */
303     public void bindToComponent(ProjectComponent component) {
304         project = component.getProject();
305         addBeans(project.getProperties());
306         addBeans(project.getUserProperties());
307         addBeans(project.getCopyOfTargets());
308         addBeans(project.getCopyOfReferences());
309         addBean("project", project);
310         addBean("self", component);
311     }
312 
313     /**
314      * Bind the runner to a project component.
315      * The project and self are the only beans set.
316      * @param component to become <code>self</code>
317      */
318     public void bindToComponentMinimum(ProjectComponent component) {
319         project = component.getProject();
320         addBean("project", project);
321         addBean("self", component);
322     }

324     /**
325      * Check if the language attribute is set.
326      * @throws BuildException if it is not.
327      */
328     protected void checkLanguage() {
329         if (language == null) {
330             throw new BuildException(
331                 "script language must be specified");
332         }
333     }
334 
335     /**
336      * Replace the current context classloader with the
337      * script context classloader.
338      * @return the current context classloader.
339      */
340     protected ClassLoader replaceContextLoader() {
341         ClassLoader origContextClassLoader =
342             Thread.currentThread().getContextClassLoader();
343         if (getScriptClassLoader() == null) {
344             setScriptClassLoader(getClass().getClassLoader());
345         }
346         Thread.currentThread().setContextClassLoader(getScriptClassLoader());
347         return origContextClassLoader;
348     }
349 
350     /**
351      * Restore the context loader with the original context classloader.
352      *
353      * script context loader.
354      * @param origLoader the original context classloader.
355      */
356     protected void restoreContextLoader(ClassLoader origLoader) {
357         Thread.currentThread().setContextClassLoader(
358                  origLoader);
359     }
