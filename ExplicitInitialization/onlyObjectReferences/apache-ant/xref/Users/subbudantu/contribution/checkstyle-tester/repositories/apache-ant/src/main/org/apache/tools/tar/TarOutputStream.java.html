207     /**
208      * Ends the TAR archive without closing the underlying OutputStream.
209      * 
210      * An archive consists of a series of file entries terminated by an
211      * end-of-archive entry, which consists of two 512 blocks of zero bytes. 
212      * POSIX.1 requires two EOF records, like some other implementations.
213      * 
214      * @throws IOException on error
215      */
216     public void finish() throws IOException {
217         if (finished) {
218             throw new IOException("This archive has already been finished");
219         }
220 
221         if (haveUnclosedEntry) {
222             throw new IOException("This archives contains unclosed entries.");
223         }
224         writeEOFRecord();
225         writeEOFRecord();
226         buffer.flushBlock();
227         finished = true;
228     }
229 
230     /**
231      * Ends the TAR archive and closes the underlying OutputStream.
232      * This means that finish() is called followed by calling the
233      * TarBuffer's close().
234      * @throws IOException on error
235      */
236     @Override
237     public void close() throws IOException {
238         if(!finished) {
239             finish();
240         }
241 
242         if (!closed) {
243             buffer.close();
244             out.close();
245             closed = true;
246         }
247     }

258     /**
259      * Put an entry on the output stream. This writes the entry's
260      * header record and positions the output stream for writing
261      * the contents of the entry. Once this method is called, the
262      * stream is ready for calls to write() to write the entry's
263      * contents. Once the contents are written, closeEntry()
264      * <B>MUST</B> be called to ensure that all buffered data
265      * is completely written to the output stream.
266      *
267      * @param entry The TarEntry to be written to the archive.
268      * @throws IOException on error
269      */
270     public void putNextEntry(TarEntry entry) throws IOException {
271         if(finished) {
272             throw new IOException("Stream has already been finished");
273         }
274         Map<String, String> paxHeaders = new HashMap<String, String>();
275         final String entryName = entry.getName();
276         boolean paxHeaderContainsPath = handleLongName(entryName, paxHeaders, "path",
277                                                        TarConstants.LF_GNUTYPE_LONGNAME, "file name");
278 
279         final String linkName = entry.getLinkName();
280         boolean paxHeaderContainsLinkPath = linkName != null
281             && handleLongName(linkName, paxHeaders, "linkpath",
282                               TarConstants.LF_GNUTYPE_LONGLINK, "link name");
283 
284         if (bigNumberMode == BIGNUMBER_POSIX) {
285             addPaxHeadersForBigNumbers(paxHeaders, entry);
286         } else if (bigNumberMode != BIGNUMBER_STAR) {
287             failForBigNumbers(entry);
288         }
289 
290         if (addPaxHeadersForNonAsciiNames && !paxHeaderContainsPath
291             && !ASCII.canEncode(entryName)) {
292             paxHeaders.put("path", entryName);
293         }
294 
295         if (addPaxHeadersForNonAsciiNames && !paxHeaderContainsLinkPath
296             && (entry.isLink() || entry.isSymbolicLink())
297             && !ASCII.canEncode(linkName)) {
298             paxHeaders.put("linkpath", linkName);
299         }
300 
301         if (paxHeaders.size() > 0) {
302             writePaxHeaders(entryName, paxHeaders);
303         }
304 
305         entry.writeEntryHeader(recordBuf, encoding,
306                                bigNumberMode == BIGNUMBER_STAR);
307         buffer.writeRecord(recordBuf);
308 
309         currBytes = 0;
310 
311         if (entry.isDirectory()) {
312             currSize = 0;
313         } else {
314             currSize = entry.getSize();
315         }
316         currName = entryName;
317         haveUnclosedEntry = true;
318     }
319 
320     /**
321      * Close an entry. This method MUST be called for all file
322      * entries that contain data. The reason is that we must
323      * buffer data written to the stream in order to satisfy
324      * the buffer's record based writes. Thus, there may be
325      * data fragments still being assembled that must be written
326      * to the output stream before this entry is closed and the
327      * next entry written.
328      * @throws IOException on error
329      */
330     public void closeEntry() throws IOException {
331         if (finished) {
332             throw new IOException("Stream has already been finished");
333         }
334         if (!haveUnclosedEntry){
335             throw new IOException("No current entry to close");
336         }
337         if (assemLen > 0) {
338             for (int i = assemLen; i < assemBuf.length; ++i) {
339                 assemBuf[i] = 0;
340             }
341 
342             buffer.writeRecord(assemBuf);
343 
344             currBytes += assemLen;
345             assemLen = 0;
346         }
347 
348         if (currBytes < currSize) {
349             throw new IOException("entry '" + currName + "' closed at '"
350                                   + currBytes
351                                   + "' before the '" + currSize
352                                   + "' bytes specified in the header were written");
353         }
354         haveUnclosedEntry = false;
355     }

384     /**
385      * Writes bytes to the current tar archive entry. This method
386      * is aware of the current entry and will throw an exception if
387      * you attempt to write bytes past the length specified for the
388      * current entry. The method is also (painfully) aware of the
389      * record buffering required by TarBuffer, and manages buffers
390      * that are not a multiple of recordsize in length, including
391      * assembling records from small buffers.
392      *
393      * @param wBuf The buffer to write to the archive.
394      * @param wOffset The offset in the buffer from which to get bytes.
395      * @param numToWrite The number of bytes to write.
396      * @throws IOException on error
397      */
398     @Override
399     public void write(byte[] wBuf, int wOffset, int numToWrite) throws IOException {
400         if ((currBytes + numToWrite) > currSize) {
401             throw new IOException("request to write '" + numToWrite
402                                   + "' bytes exceeds size in header of '"
403                                   + currSize + "' bytes for entry '"
404                                   + currName + "'");
405 
406             //
407             // We have to deal with assembly!!!
408             // The programmer can be writing little 32 byte chunks for all
409             // we know, and we must assemble complete records for writing.
410             // REVIEW Maybe this should be in TarBuffer? Could that help to
411             // eliminate some of the buffer copying.
412             //
413         }
414 
415         if (assemLen > 0) {
416             if ((assemLen + numToWrite) >= recordBuf.length) {
417                 int aLen = recordBuf.length - assemLen;
418 
419                 System.arraycopy(assemBuf, 0, recordBuf, 0,
420                                  assemLen);
421                 System.arraycopy(wBuf, wOffset, recordBuf,
422                                  assemLen, aLen);
423                 buffer.writeRecord(recordBuf);
424 
425                 currBytes += recordBuf.length;
426                 wOffset += aLen;
427                 numToWrite -= aLen;
428                 assemLen = 0;
429             } else {
430                 System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
431                                  numToWrite);
432 
433                 wOffset += numToWrite;
434                 assemLen += numToWrite;
435                 numToWrite = 0;
436             }
437         }
438 
439         //
440         // When we get here we have EITHER:
441         // o An empty "assemble" buffer.
442         // o No bytes to write (numToWrite == 0)
443         //
444         while (numToWrite > 0) {
445             if (numToWrite < recordBuf.length) {
446                 System.arraycopy(wBuf, wOffset, assemBuf, assemLen,
447                                  numToWrite);
448 
449                 assemLen += numToWrite;
450 
451                 break;
452             }
453 
454             buffer.writeRecord(wBuf, wOffset);
455 
456             int num = recordBuf.length;
457 
458             currBytes += num;
459             numToWrite -= num;
460             wOffset += num;
461         }
462     }

464     /**
465      * Writes a PAX extended header with the given map as contents.
466      */
467     void writePaxHeaders(String entryName,
468                          Map<String, String> headers) throws IOException {
469         String name = "./PaxHeaders.X/" + stripTo7Bits(entryName);
470         if (name.length() >= TarConstants.NAMELEN) {
471             name = name.substring(0, TarConstants.NAMELEN - 1);
472         }
473         while (name.endsWith("/")) {
474             // TarEntry's constructor would think this is a directory
475             // and not allow any data to be written
476             name = name.substring(0, name.length() - 1);
477         }
478         TarEntry pex = new TarEntry(name,
479                                     TarConstants.LF_PAX_EXTENDED_HEADER_LC);
480 
481         StringWriter w = new StringWriter();
482         for (Map.Entry<String, String> h : headers.entrySet()) {
483             String key = h.getKey();
484             String value = h.getValue();
485             int len = key.length() + value.length()
486                 + 3 /* blank, equals and newline */
487                 + 2 /* guess 9 < actual length < 100 */;
488             String line = len + " " + key + "=" + value + "\n";
489             int actualLength = line.getBytes("UTF-8").length;
490             while (len != actualLength) {
491                 // Adjust for cases where length < 10 or > 100
492                 // or where UTF-8 encoding isn't a single octet
493                 // per character.
494                 // Must be in loop as size may go from 99 to 100 in
495                 // first pass so we'd need a second.
496                 len = actualLength;
497                 line = len + " " + key + "=" + value + "\n";
498                 actualLength = line.getBytes("UTF-8").length;
499             }
500             w.write(line);
501         }
502         byte[] data = w.toString().getBytes("UTF-8");
503         pex.setSize(data.length);
504         putNextEntry(pex);
505         write(data);
506         closeEntry();
507     }

509     private String stripTo7Bits(String name) {
510         final int length = name.length();
511         StringBuffer result = new StringBuffer(length);
512         for (int i = 0; i < length; i++) {
513             char stripped = (char) (name.charAt(i) & 0x7F);
514             if (stripped != 0) { // would be read as Trailing null
515                 result.append(stripped);
516             }
517         }
518         return result.toString();
519     }

521     /**
522      * Write an EOF (end of archive) record to the tar archive.
523      * An EOF record consists of a record of all zeros.
524      */
525     private void writeEOFRecord() throws IOException {
526         for (int i = 0; i < recordBuf.length; ++i) {
527             recordBuf[i] = 0;
528         }
529 
530         buffer.writeRecord(recordBuf);
531     }

533     private void addPaxHeadersForBigNumbers(Map<String, String> paxHeaders,
534                                             TarEntry entry) {
535         addPaxHeaderForBigNumber(paxHeaders, "size", entry.getSize(),
536                                  TarConstants.MAXSIZE);
537         addPaxHeaderForBigNumber(paxHeaders, "gid", entry.getGroupId(),
538                                  TarConstants.MAXID);
539         addPaxHeaderForBigNumber(paxHeaders, "mtime",                                 entry.getModTime().getTime() / 1000,
540                                  TarConstants.MAXSIZE);
541         addPaxHeaderForBigNumber(paxHeaders, "uid", entry.getUserId(),
542                                  TarConstants.MAXID);
543         // star extensions by J\u00f6rg Schilling
544         addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devmajor",
545                                  entry.getDevMajor(), TarConstants.MAXID);
546         addPaxHeaderForBigNumber(paxHeaders, "SCHILY.devminor",
547                                  entry.getDevMinor(), TarConstants.MAXID);
548         // there is no PAX header for file mode
549         failForBigNumber("mode", entry.getMode(), TarConstants.MAXID);
550     }

552     private void addPaxHeaderForBigNumber(Map<String, String> paxHeaders,
553                                           String header, long value,
554                                           long maxValue) {
555         if (value < 0 || value > maxValue) {
556             paxHeaders.put(header, String.valueOf(value));
557         }
558     }
559 
560     private void failForBigNumbers(TarEntry entry) {
561         failForBigNumber("entry size", entry.getSize(), TarConstants.MAXSIZE);
562         failForBigNumber("group id", entry.getGroupId(), TarConstants.MAXID);
563         failForBigNumber("last modification time",
564                          entry.getModTime().getTime() / 1000,
565                          TarConstants.MAXSIZE);
566         failForBigNumber("user id", entry.getUserId(), TarConstants.MAXID);
567         failForBigNumber("mode", entry.getMode(), TarConstants.MAXID);
568         failForBigNumber("major device number", entry.getDevMajor(),
569                          TarConstants.MAXID);
570         failForBigNumber("minor device number", entry.getDevMinor(),
571                          TarConstants.MAXID);
572     }

574     private void failForBigNumber(String field, long value, long maxValue) {
575         if (value < 0 || value > maxValue) {
576             throw new RuntimeException(field + " '" + value
577                                        + "' is too big ( > "
578                                        + maxValue + " )");
579         }
580     }

582     /**
583      * Handles long file or link names according to the longFileMode setting.
584      *
585      * <p>I.e. if the given name is too long to be written to a plain
586      * tar header then
587      * <ul>
588      *   <li>it creates a pax header who's name is given by the
589      *   paxHeaderName parameter if longFileMode is POSIX</li>
590      *   <li>it creates a GNU longlink entry who's type is given by
591      *   the linkType parameter if longFileMode is GNU</li>
592      *   <li>it throws an exception if longFileMode is ERROR</li>
593      *   <li>it truncates the name if longFileMode is TRUNCATE</li>
594      * </ul></p>
595      *
596      * @param name the name to write
597      * @param paxHeaders current map of pax headers
598      * @param paxHeaderName name of the pax header to write
599      * @param linkType type of the GNU entry to write
600      * @param fieldName the name of the field
601      * @return whether a pax header has been written.
602      */
603     private boolean handleLongName(String name,
604                                    Map<String, String> paxHeaders,
605                                    String paxHeaderName, byte linkType, String fieldName)
606         throws IOException {
607         final ByteBuffer encodedName = encoding.encode(name);
608         final int len = encodedName.limit() - encodedName.position();
609         if (len >= TarConstants.NAMELEN) {
610 
611             if (longFileMode == LONGFILE_POSIX) {
612                 paxHeaders.put(paxHeaderName, name);
613                 return true;
614             } else if (longFileMode == LONGFILE_GNU) {
615                 // create a TarEntry for the LongLink, the contents
616                 // of which are the link's name
617                 TarEntry longLinkEntry =
618                     new TarEntry(TarConstants.GNU_LONGLINK, linkType);
619 
620                 longLinkEntry.setSize(len + 1); // +1 for NUL
621                 putNextEntry(longLinkEntry);
622                 write(encodedName.array(), encodedName.arrayOffset(), len);
623                 write(0); // NUL terminator
624                 closeEntry();
625             } else if (longFileMode != LONGFILE_TRUNCATE) {
626                 throw new RuntimeException(fieldName + " '" + name
627                                            + "' is too long ( > "
628                                            + TarConstants.NAMELEN + " bytes)");
629             }
630         }
631         return false;
632     }
