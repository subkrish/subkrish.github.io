19  package org.apache.tools.zip;
20  
21  import java.util.zip.CRC32;
22  import java.util.zip.ZipException;
23  
24  /**
25   * Adds Unix file permission and UID/GID fields as well as symbolic
26   * link handling.
27   *
28   * <p>This class uses the ASi extra field in the format:
29   * <pre>
30   *         Value         Size            Description
31   *         -----         ----            -----------
32   * (Unix3) 0x756e        Short           tag for this extra block type
33   *         TSize         Short           total data size for this block
34   *         CRC           Long            CRC-32 of the remaining data
35   *         Mode          Short           file permissions
36   *         SizDev        Long            symlink'd size OR major/minor dev num
37   *         UID           Short           user ID
38   *         GID           Short           group ID
39   *         (var.)        variable        symbolic link filename
40   * </pre>
41   * taken from appnote.iz (Info-ZIP note, 981119) found at <a
42   * href="ftp://ftp.uu.net/pub/archiving/zip/doc/">ftp://ftp.uu.net/pub/archiving/zip/doc/</a></p>
43  
44   *
45   * <p>Short is two bytes and Long is four bytes in big endian byte and
46   * word order, device numbers are currently not supported.</p>
47   *
48   * <p>Since the documentation this class is based upon doesn't mention
49   * the character encoding of the file name at all, it is assumed that
50   * it uses the current platform's default encoding.</p>
51   */
52  public class AsiExtraField implements ZipExtraField, UnixStat, Cloneable {
53  
54      private static final ZipShort HEADER_ID = new ZipShort(0x756E);
55      private static final int      WORD = 4;
56      /**
57       * Standard Unix stat(2) file mode.
58       *
59       * @since 1.1
60       */
61      private int mode = 0;
62      /**
63       * User ID.
64       *
65       * @since 1.1
66       */
67      private int uid = 0;
68      /**
69       * Group ID.
70       *
71       * @since 1.1
72       */
73      private int gid = 0;
74      /**
75       * File this entry points to, if it is a symbolic link.
76       *
77       * <p>empty string - if entry is not a symbolic link.</p>
78       *
79       * @since 1.1
80       */
81      private String link = "";
82      /**
83       * Is this an entry for a directory?
84       *
85       * @since 1.1
86       */
87      private boolean dirFlag = false;
88  
89      /**
90       * Instance used to calculate checksums.
91       *
92       * @since 1.1
93       */
94      private CRC32 crc = new CRC32();
95  
96      /** Constructor for AsiExtraField. */
97      public AsiExtraField() {
98      }
99  
100     /**
101      * The Header-ID.
102      * @return the value for the header id for this extrafield
103      * @since 1.1
104      */
105     public ZipShort getHeaderId() {
106         return HEADER_ID;
107     }
108 
109     /**
110      * Length of the extra field in the local file data - without
111      * Header-ID or length specifier.
112      * @return a <code>ZipShort</code> for the length of the data of this extra field
113      * @since 1.1
114      */
115     public ZipShort getLocalFileDataLength() {
116         return new ZipShort(WORD         // CRC
117                           + 2         // Mode
118                           + WORD         // SizDev
119                           + 2         // UID
120                           + 2         // GID
121                           + getLinkedFile().getBytes().length);
122                           // Uses default charset - see class Javadoc
123     }
124 
125     /**
126      * Delegate to local file data.
127      * @return the centralDirectory length
128      * @since 1.1
129      */
130     public ZipShort getCentralDirectoryLength() {
131         return getLocalFileDataLength();
132     }
133 
134     /**
135      * The actual data to put into local file data - without Header-ID
136      * or length specifier.
137      * @return get the data
138      * @since 1.1
139      */
140     public byte[] getLocalFileDataData() {
141         // CRC will be added later
142         byte[] data = new byte[getLocalFileDataLength().getValue() - WORD];
143         System.arraycopy(ZipShort.getBytes(getMode()), 0, data, 0, 2);
144 
145         byte[] linkArray = getLinkedFile().getBytes(); // Uses default charset - see class Javadoc
146         // CheckStyle:MagicNumber OFF
147         System.arraycopy(ZipLong.getBytes(linkArray.length),
148                          0, data, 2, WORD);
149 
150         System.arraycopy(ZipShort.getBytes(getUserId()),
151                          0, data, 6, 2);
152         System.arraycopy(ZipShort.getBytes(getGroupId()),
153                          0, data, 8, 2);
154 
155         System.arraycopy(linkArray, 0, data, 10, linkArray.length);
156         // CheckStyle:MagicNumber ON
157 
158         crc.reset();
159         crc.update(data);
160         long checksum = crc.getValue();
161 
162         byte[] result = new byte[data.length + WORD];
163         System.arraycopy(ZipLong.getBytes(checksum), 0, result, 0, WORD);
164         System.arraycopy(data, 0, result, WORD, data.length);
165         return result;
166     }
167 
168     /**
169      * Delegate to local file data.
170      * @return the local file data
171      * @since 1.1
172      */
173     public byte[] getCentralDirectoryData() {
174         return getLocalFileDataData();
175     }
176 
177     /**
178      * Set the user id.
179      * @param uid the user id
180      * @since 1.1
181      */
182     public void setUserId(int uid) {
183         this.uid = uid;
184     }
185 
186     /**
187      * Get the user id.
188      * @return the user id
189      * @since 1.1
190      */
191     public int getUserId() {
192         return uid;
193     }
194 
195     /**
196      * Set the group id.
197      * @param gid the group id
198      * @since 1.1
199      */
200     public void setGroupId(int gid) {
201         this.gid = gid;
202     }
203 
204     /**
205      * Get the group id.
206      * @return the group id
207      * @since 1.1
208      */
209     public int getGroupId() {
210         return gid;
211     }
212 
213     /**
214      * Indicate that this entry is a symbolic link to the given filename.
215      *
216      * @param name Name of the file this entry links to, empty String
217      *             if it is not a symbolic link.
218      *
219      * @since 1.1
220      */
221     public void setLinkedFile(String name) {
222         link = name;
223         mode = getMode(mode);
224     }
225 
226     /**
227      * Name of linked file
228      *
229      * @return name of the file this entry links to if it is a
230      *         symbolic link, the empty string otherwise.
231      *
232      * @since 1.1
233      */
234     public String getLinkedFile() {
235         return link;
236     }
237 
238     /**
239      * Is this entry a symbolic link?
240      * @return true if this is a symbolic link
241      * @since 1.1
242      */
243     public boolean isLink() {
244         return getLinkedFile().length() != 0;
245     }
246 
247     /**
248      * File mode of this file.
249      * @param mode the file mode
250      * @since 1.1
251      */
252     public void setMode(int mode) {
253         this.mode = getMode(mode);
254     }
255 
256     /**
257      * File mode of this file.
258      * @return the file mode
259      * @since 1.1
260      */
261     public int getMode() {
262         return mode;
263     }
264 
265     /**
266      * Indicate whether this entry is a directory.
267      * @param dirFlag if true, this entry is a directory
268      * @since 1.1
269      */
270     public void setDirectory(boolean dirFlag) {
271         this.dirFlag = dirFlag;
272         mode = getMode(mode);
273     }
274 
275     /**
276      * Is this entry a directory?
277      * @return true if this entry is a directory
278      * @since 1.1
279      */
280     public boolean isDirectory() {
281         return dirFlag && !isLink();
282     }
283 
284     /**
285      * Populate data from this array as if it was in local file data.
286      * @param data an array of bytes
287      * @param offset the start offset
288      * @param length the number of bytes in the array from offset
289      * @since 1.1
290      * @throws ZipException on error
291      */
292     public void parseFromLocalFileData(byte[] data, int offset, int length)
293         throws ZipException {
294 
295         long givenChecksum = ZipLong.getValue(data, offset);
296         byte[] tmp = new byte[length - WORD];
297         System.arraycopy(data, offset + WORD, tmp, 0, length - WORD);
298         crc.reset();
299         crc.update(tmp);
300         long realChecksum = crc.getValue();
301         if (givenChecksum != realChecksum) {
302             throw new ZipException("bad CRC checksum "
303                                    + Long.toHexString(givenChecksum)
304                                    + " instead of "
305                                    + Long.toHexString(realChecksum));
306         }
307 
308         int newMode = ZipShort.getValue(tmp, 0);
309         // CheckStyle:MagicNumber OFF
310         byte[] linkArray = new byte[(int) ZipLong.getValue(tmp, 2)];
311         uid = ZipShort.getValue(tmp, 6);
312         gid = ZipShort.getValue(tmp, 8);
313 
314         if (linkArray.length == 0) {
315             link = "";
316         } else {
317             System.arraycopy(tmp, 10, linkArray, 0, linkArray.length);
318             link = new String(linkArray); // Uses default charset - see class Javadoc
319         }
320         // CheckStyle:MagicNumber ON
321         setDirectory((newMode & DIR_FLAG) != 0);
322         setMode(newMode);
323     }
324 
325     /**
326      * Get the file mode for given permissions with the correct file type.
327      * @param mode the mode
328      * @return the type with the mode
329      * @since 1.1
330      */
331     protected int getMode(int mode) {
332         int type = FILE_FLAG;
333         if (isLink()) {
334             type = LINK_FLAG;
335         } else if (isDirectory()) {
336             type = DIR_FLAG;
337         }
338         return type | (mode & PERM_MASK);
339     }
340 
341     @Override
342     public Object clone() {
343         try {
344             AsiExtraField cloned = (AsiExtraField) super.clone();
345             cloned.crc = new CRC32();
346             return cloned;
347         } catch (CloneNotSupportedException cnfe) {
348             // impossible
349             throw new RuntimeException(cnfe);
350         }
351     }
