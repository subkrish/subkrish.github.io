19  package org.jsecurity.authc.credential;
20  
21  import org.jsecurity.authc.AuthenticationInfo;
22  import org.jsecurity.authc.AuthenticationToken;
23  import org.jsecurity.codec.Base64;
24  import org.jsecurity.codec.Hex;
25  import org.jsecurity.crypto.hash.AbstractHash;
26  import org.jsecurity.crypto.hash.Hash;
27  
28  /**
29   * A <tt>HashedCredentialMatcher</tt> provides support for hashing of supplied <tt>AuthenticationToken</tt> credentials
30   * before being compared to those in the <tt>AuthenticationInfo</tt> from the data store.
31   *
32   * <p>Credential hashing is one of the most common security techniques when safeguarding a user's private credentials
33   * (passwords, keys, etc).  Most developers never want to store their users' credentials in plain form, viewable by
34   * anyone, so they often hash the users' credentials before they are saved in the data store.</p>
35   *
36   * <p>This class (and its subclasses) function as follows:</p>
37   *
38   * <p>It first hashes the <tt>AuthenticationToken</tt> credentials supplied by the user during their login.  It then
39   * compares this hashed value directly with the <tt>AuthenticationInfo</tt> credentials stored in the system.  The stored account
40   * credentials are expected to already be in hashed form.  If these two values are equal, the submitted credentials
41   * match.</p>
42   *
43   * <h3>Salting and Multiple Hash Iterations</h3>
44   *
45   * <p>Because simple hashing is sometimes not good enough for many applications, this class also supports 'salting'
46   * and multiple hash iterations.  Please read this excellent
47   * <a href="http://www.owasp.org/index.php/Hashing_Java" _target="blank">Hashing Java article</a> to learn about
48   * salting and multiple iterations and why you might want to use them. (Note of sections 5
49   * &quot;Why add salt?&quot; and 6 "Hardening against the attacker's attack").
50   *
51   * <p>We should also note here that all of JSecurity's Hash implementations (for example,
52   * {@link org.jsecurity.crypto.hash.Md5Hash Md5Hash}, {@link org.jsecurity.crypto.hash.Sha1Hash Sha1Hash}, etc)
53   * support salting and multiple hash iterations via overloaded constructors.</p>
54   *
55   * <h4>Salting</h4>
56   *
57   * <p>Salting of the authentication token's credentials hash is disabled by default, but you may enable it by setting
58   * {@link #setHashSalted hashSalted} to
59   * <tt>true</tt>.  If you do enable it, the value used to salt the hash will be
60   * obtained from {@link #getSalt(AuthenticationToken) getSalt(authenticationToken)}.
61   *
62   * <p>The default <tt>getSalt</tt> implementation merely returns
63   * <code>token.getPrincipal()</code>, effectively using the user's identity as the salt, a most common
64   * technique.  If you wish to provide the authentication token's salt another way, you may override this
65   * <tt>getSalt</tt> method.
66   *
67   * <h4>Multiple Hash Iterations</h4>
68   *
69   * <p>If you hash your users' credentials multiple times before persisting to the data store, you will also need to
70   * set this class's {@link #setHashIterations(int) hashIterations} property.</p>
71   *
72   * <p><b>Note:</b> <a href="http://en.wikipedia.org/wiki/MD5">MD5</a> and
73   * <a href="http://en.wikipedia.org/wiki/SHA_hash_functions">SHA-1</a> algorithms are now known to be vulnerable to
74   * compromise and/or collisions (read the linked pages for more).  While most applications are ok with either of these
75   * two, if your application mandates high security, use the SHA-256 (or higher) hashing algorithms and their
76   * supporting <code>CredentialsMatcher</code> implementations.</p>
77   *
78   * @author Les Hazlewood
79   * @see org.jsecurity.crypto.hash.Md5Hash
80   * @see org.jsecurity.crypto.hash.Sha1Hash
81   * @see org.jsecurity.crypto.hash.Sha256Hash
82   * @since 0.9
83   */
84  public abstract class HashedCredentialsMatcher extends SimpleCredentialsMatcher {
85  
86      private boolean storedCredentialsHexEncoded = true; //false means base64 encoded
87      private boolean hashSalted = false;
88      private int hashIterations = 1;
89  
90      /**
91       * Returns <tt>true</tt> if the system's stored credential hash is Hex encoded, <tt>false</tt> if it
92       * is Base64 encoded.
93       *
94       * <p>Default value is <tt>true</tt> for convenience - all of JSecurity's {@link Hash Hash#toString()}
95       * implementations return Hex encoded values by default, making this class's use with those implementations
96       * easier.</p>
97       *
98       * @return <tt>true</tt> if the system's stored credential hash is Hex encoded, <tt>false</tt> if it
99       *         is Base64 encoded.  Default is <tt>true</tt>
100      */
101     public boolean isStoredCredentialsHexEncoded() {
102         return storedCredentialsHexEncoded;
103     }
104 
105     /**
106      * Sets the indicator if this system's stored credential hash is Hex encoded or not.
107      *
108      * <p>A value of <tt>true</tt> will cause this class to decode the system credential from Hex, a
109      * value of <tt>false</tt> will cause this class to decode the system credential from Base64.</p>
110      *
111      * <p>Unless overridden via this method, the default value is <tt>true</tt> for convenience - all of JSecurity's
112      * {@link Hash Hash#toString()} implementations return Hex encoded values by default, making this class's use with
113      * those implementations easier.</p>.
114      *
115      * @param storedCredentialsHexEncoded the indicator if this system's stored credential hash is Hex
116      *                                    encoded or not ('not' automatically implying it is Base64 encoded).
117      */
118     public void setStoredCredentialsHexEncoded(boolean storedCredentialsHexEncoded) {
119         this.storedCredentialsHexEncoded = storedCredentialsHexEncoded;
120     }
121 
122     /**
123      * Returns <tt>true</tt> if a submitted <tt>AuthenticationToken</tt>'s credentials should be salted when hashing,
124      * <tt>false</tt> if it should not be salted.
125      *
126      * <p>If enabled, the salt used will be obtained via the {@link #getSalt(AuthenticationToken) getSalt} method.
127      *
128      * <p>The default value is <tt>false</tt>.
129      *
130      * @return <tt>true</tt> if a submitted <tt>AuthenticationToken</tt>'s credentials should be salted when hashing,
131      *         <tt>false</tt> if it should not be salted.
132      */
133     public boolean isHashSalted() {
134         return hashSalted;
135     }
136 
137     /**
138      * Sets whether or not to salt a submitted <tt>AuthenticationToken</tt>'s credentials when hashing.
139      *
140      * <p>If enabled, the salt used will be obtained via the {@link #getSalt(AuthenticationToken) getSalt} method.
141      *
142      * <p>The default value is <tt>false</tt>.
143      *
144      * @param hashSalted whether or not to salt a submitted <tt>AuthenticationToken</tt>'s credentials when hashing.
145      */
146     public void setHashSalted(boolean hashSalted) {
147         this.hashSalted = hashSalted;
148     }
149 
150     /**
151      * Returns the number of times a submitted <tt>AuthenticationToken</tt>'s credentials will be hashed before
152      * comparing to the credentials stored in the system.
153      *
154      * <p>Unless overridden, the default value is <tt>1</tt>, meaning a normal hash execution will occur.
155      *
156      * @return the number of times a submitted <tt>AuthenticationToken</tt>'s credentials will be hashed before
157      *         comparing to the credentials stored in the system.
158      */
159     public int getHashIterations() {
160         return hashIterations;
161     }
162 
163     /**
164      * Sets the number of times a submitted <tt>AuthenticationToken</tt>'s credentials will be hashed before comparing
165      * to the credentials stored in the system.
166      *
167      * <p>Unless overridden, the default value is <tt>1</tt>, meaning a normal single hash execution will occur.
168      *
169      * <p>If this argument is less than 1 (i.e. 0 or negative), the default value of 1 is applied.  There must always be
170      * at least 1 hash iteration (otherwise there would be no hash).
171      *
172      * @param hashIterations the number of times to hash a submitted <tt>AuthenticationToken</tt>'s credentials.
173      */
174     public void setHashIterations(int hashIterations) {
175         if (hashIterations < 1) {
176             this.hashIterations = 1;
177         } else {
178             this.hashIterations = hashIterations;
179         }
180     }
181 
182     /**
183      * Returns a salt value used to hash the token's credentials.
184      *
185      * <p>This default implementation merely returns <code>token.getPrincipal()</code>, effectively using the user's
186      * identity (username, user id, etc) as the salt, a most common technique.  If you wish to provide the
187      * authentication token's salt another way, you may override this method.
188      *
189      * @param token the AuthenticationToken submitted during the authentication attempt.
190      * @return a salt value to use to hash the authentication token's credentials.
191      */
192     protected Object getSalt(AuthenticationToken token) {
193         return token.getPrincipal();
194     }
195 
196     /**
197      * As this is a HashedCredentialMatcher, this method overrides the parent method by returning a hashed value
198      * of the submitted token's credentials.
199      *
200      * <p>Based on this class's configuration, the return value may be salted and/or
201      * hashed multiple times (see the class-level JavaDoc for more information on salting and
202      * multiple hash iterations).
203      *
204      * @param token the authentication token submitted during the authentication attempt.
205      * @return the hashed value of the authentication token's credentials.
206      */
207     protected Object getCredentials(AuthenticationToken token) {
208         Object credentials = token.getCredentials();
209         Object salt = isHashSalted() ? getSalt(token) : null;
210         return hashProvidedCredentials(credentials, salt, getHashIterations());
211     }
212 
213     /**
214      * Returns a {@link Hash Hash} instance representing the already-hashed AuthenticationInfo credentials stored in the system.
215      *
216      * <p>This method reconstructs a {@link Hash Hash} instance based on a <code>info.getCredentials</code> call,
217      * but it does <em>not</em> hash that value - it is expected that method call will return an already-hashed value.
218      *
219      * <p>This implementation's reconstruction effort functions as follows:
220      *
221      * <ol>
222      * <li>Convert <code>account.getCredentials()</code> to a byte array via the {@link #toBytes toBytes} method.
223      * <li>If <code>account.getCredentials()</code> was originally a String or char[] before <tt>toBytes</tt> was
224      * called, check for encoding:
225      * <li>If {@link #storedCredentialsHexEncoded storedCredentialsHexEncoded}, Hex decode that byte array, otherwise
226      * Base64 decode the byte array</li>
227      * <li>Set the byte[] array directly on the <tt>Hash</tt> implementation and return it.</li>
228      * </ol>
229      *
230      * @param info the AuthenticationInfo from which to retrive the credentials which assumed to be in already-hashed form.
231      * @return a {@link Hash Hash} instance representing the given AuthenticationInfo's stored credentials.
232      */
233     protected Object getCredentials(AuthenticationInfo info) {
234         Object credentials = info.getCredentials();
235 
236         byte[] storedBytes = toBytes(credentials);
237 
238         if (credentials instanceof String || credentials instanceof char[]) {
239             //account.credentials were a char[] or String, so
240             //we need to do text decoding first:
241             if (isStoredCredentialsHexEncoded()) {
242                 storedBytes = Hex.decode(storedBytes);
243             } else {
244                 storedBytes = Base64.decode(storedBytes);
245             }
246         }
247         AbstractHash hash = newHashInstance();
248         hash.setBytes(storedBytes);
249         return hash;
250     }
251 
252     /**
253      * Hashes the provided credentials a total of <tt>hashIterations</tt> times, using the given salt.  The hash
254      * implementation/algorithm used is left to subclasses.
255      *
256      * @param credentials    the submitted authentication token's credentials to hash
257      * @param salt           the value to salt the hash, or <tt>null</tt> if a salt will not be used.
258      * @param hashIterations the number of times to hash the credentials.  At least one hash will always occur though,
259      *                       even if this argument is 0 or negative.
260      * @return the hashed value of the provided credentials, according to the specified salt and hash iterations.
261      */
262     protected abstract Hash hashProvidedCredentials(Object credentials, Object salt, int hashIterations);
263 
264     /**
265      * Returns a new, <em>uninitialized</em> instance, without its byte array set.  Used as a utility method in the
266      * {@link SimpleCredentialsMatcher#getCredentials(org.jsecurity.authc.AuthenticationInfo) getCredentials(AuthenticationInfo)} implementation.
267      *
268      * @return a new, <em>uninitialized</em> instance, without its byte array set.
269      */
270     protected abstract AbstractHash newHashInstance();
271 
272 }
