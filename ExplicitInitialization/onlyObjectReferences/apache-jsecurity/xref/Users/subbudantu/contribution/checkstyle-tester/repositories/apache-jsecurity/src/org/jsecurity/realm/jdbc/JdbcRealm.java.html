40  /**
41   * <p>
42   * Realm that allows authentication and authorization via JDBC calls.  The default queries suggest a potential schema
43   * for retrieving the user's password for authentication, and querying for a user's roles and permissions.  The
44   * default queries can be overridden by setting the query properties of the realm.
45   * </p>
46   *
47   * <p>
48   * If the default implementation
49   * of authentication and authorization cannot handle your schema, this class can be subclassed and the
50   * appropriate methods overridden. (usually {@link #doGetAuthenticationInfo(org.jsecurity.authc.AuthenticationToken)},
51   * {@link #getRoleNamesForUser(java.sql.Connection,String)}, and/or {@link #getPermissions(java.sql.Connection,String,java.util.Collection)}
52   * </p>
53   *
54   * <p>
55   * This realm supports caching by extending from {@link org.jsecurity.realm.AuthorizingRealm}.
56   * </p>
57   *
58   * @author Jeremy Haile
59   * @since 0.2
60   */
61  public class JdbcRealm extends AuthorizingRealm {

67       * The default query used to retrieve account data for the user.
68       */
69      protected static final String DEFAULT_AUTHENTICATION_QUERY = "select password from users where username = ?";

72       * The default query used to retrieve the roles that apply to a user.

76      /**
77       * The default query used to retrieve permissions that apply to a particular role.
78       */
79      protected static final String DEFAULT_PERMISSIONS_QUERY = "select permission from roles_permissions where role_name = ?";

104     /**
105      * Sets the datasource that should be used to retrieve connections used by this realm.

107      * @param dataSource the SQL data source.
108      */
109     public void setDataSource(DataSource dataSource) {
110         this.dataSource = dataSource;

113     /**
114      * Overrides the default query used to retrieve a user's password during authentication.  When using the default
115      * implementation, this query must take the user's username as a single parameter and return a single result
116      * with the user's password as the first column.  If you require a solution that does not match this query
117      * structure, you can override {@link #doGetAuthenticationInfo(org.jsecurity.authc.AuthenticationToken)} or
118      * just {@link #getPasswordForUser(java.sql.Connection,String)}
119      *
120      * @param authenticationQuery the query to use for authentication.
121      * @see #DEFAULT_AUTHENTICATION_QUERY
122      */
123     public void setAuthenticationQuery(String authenticationQuery) {
124         this.authenticationQuery = authenticationQuery;
125     }
126 
127     /**
128      * Overrides the default query used to retrieve a user's roles during authorization.  When using the default
129      * implementation, this query must take the user's username as a single parameter and return a row
130      * per role with a single column containing the role name.  If you require a solution that does not match this query
131      * structure, you can override {@link #doGetAuthorizationInfo(PrincipalCollection)} or just
132      * {@link #getRoleNamesForUser(java.sql.Connection,String)}
133      *
134      * @param userRolesQuery the query to use for retrieving a user's roles.
135      * @see #DEFAULT_USER_ROLES_QUERY
136      */
137     public void setUserRolesQuery(String userRolesQuery) {
138         this.userRolesQuery = userRolesQuery;
139     }
140 
141     /**
142      * <p>
143      * Overrides the default query used to retrieve a user's permissions during authorization.  When using the default
144      * implementation, this query must take a role name as the single parameter and return a row
145      * per permission with three columns containing the fully qualified name of the permission class, the permission
146      * name, and the permission actions (in that order).  If you require a solution that does not match this query
147      * structure, you can override {@link #doGetAuthorizationInfo(PrincipalCollection)} or just
148      * {@link #getPermissions(java.sql.Connection,String,java.util.Collection)}</p>
149      *
150      * <p><b>Permissions are only retrieved if you set {@link #permissionsLookupEnabled} to true.  Otherwise,
151      * this query is ignored.</b></p>
152      *
153      * @param permissionsQuery the query to use for retrieving permissions for a role.
154      * @see #DEFAULT_PERMISSIONS_QUERY
155      * @see #setPermissionsLookupEnabled(boolean)
156      */
157     public void setPermissionsQuery(String permissionsQuery) {
158         this.permissionsQuery = permissionsQuery;
159     }
160 
161     /**
162      * Enables lookup of permissions during authorization.  The default is "false" - meaning that only roles
163      * are associated with a user.  Set this to true in order to lookup roles <b>and</b> permissions.
164      *
165      * @param permissionsLookupEnabled true if permissions should be looked up during authorization, or false if only
166      *                                 roles should be looked up.
167      */
168     public void setPermissionsLookupEnabled(boolean permissionsLookupEnabled) {
169         this.permissionsLookupEnabled = permissionsLookupEnabled;
170     }

176     protected AuthenticationInfo doGetAuthenticationInfo(AuthenticationToken token) throws AuthenticationException {
177 
178         UsernamePasswordToken upToken = (UsernamePasswordToken) token;
179         String username = upToken.getUsername();
180 
181         // Null username is invalid
182         if (username == null) {
183             throw new AccountException("Null usernames are not allowed by this realm.");
184         }
185 
186         Connection conn = null;
187         AuthenticationInfo info = null;
188         try {
189             conn = dataSource.getConnection();
190 
191             String password = getPasswordForUser(conn, username);
192 
193             if (password == null) {
194                 throw new UnknownAccountException("No account found for user [" + username + "]");
195             }
196 
197             info = buildAuthenticationInfo(username, password.toCharArray());
198 
199         } catch (SQLException e) {
200             final String message = "There was a SQL error while authenticating user [" + username + "]";
201             if (log.isErrorEnabled()) {
202                 log.error(message, e);
203             }
204 
205             // Rethrow any SQL errors as an authentication exception
206             throw new AuthenticationException(message, e);
207         } finally {
208             JdbcUtils.closeConnection(conn);
209         }
210 
211         return info;
212     }
213 
214     protected AuthenticationInfo buildAuthenticationInfo(String username, char[] password) {
215         return new SimpleAuthenticationInfo(username, password, getName());
216     }
217 
218     private String getPasswordForUser(Connection conn, String username) throws SQLException {
219 
220         PreparedStatement ps = null;
221         ResultSet rs = null;
222         String password = null;
223         try {
224             ps = conn.prepareStatement(authenticationQuery);
225             ps.setString(1, username);
226 
227             // Execute query
228             rs = ps.executeQuery();
229 
230             // Loop over results - although we are only expecting one result, since usernames should be unique
231             boolean foundResult = false;
232             while (rs.next()) {
233 
234                 // Check to ensure only one row is processed
235                 if (foundResult) {
236                     throw new AuthenticationException("More than one user row found for user [" + username + "]. Usernames must be unique.");
237                 }
238 
239                 password = rs.getString(1);
240 
241                 foundResult = true;
242             }
243         } finally {
244             JdbcUtils.closeResultSet(rs);
245             JdbcUtils.closeStatement(ps);
246         }
247 
248         return password;
249     }

251     /**
252      * This implementation of the interface expects the principals collection to return a String username keyed off of
253      * this realm's {@link #getName() name}
254      *
255      * @see AuthorizingRealm#getAuthorizationInfo(org.jsecurity.subject.PrincipalCollection)
256      */
257     @Override
258     protected AuthorizationInfo doGetAuthorizationInfo(PrincipalCollection principals) {
259 
260         //null usernames are invalid
261         if (principals == null) {
262             throw new AuthorizationException("PrincipalCollection method argument cannot be null.");
263         }
264 
265         String username = (String) principals.fromRealm(getName()).iterator().next();
266 
267         Connection conn = null;
268         Set<String> roleNames = null;
269         Set<String> permissions = null;
270         try {
271             conn = dataSource.getConnection();
272 
273             // Retrieve roles and permissions from database
274             roleNames = getRoleNamesForUser(conn, username);
275             permissions = getPermissions(conn, username, roleNames);
276 
277         } catch (SQLException e) {
278             final String message = "There was a SQL error while authorizing user [" + username + "]";
279             if (log.isErrorEnabled()) {
280                 log.error(message, e);
281             }
282 
283             // Rethrow any SQL errors as an authorization exception
284             throw new AuthorizationException(message, e);
285         } finally {
286             JdbcUtils.closeConnection(conn);
287         }
288 
289         SimpleAuthorizationInfo info = new SimpleAuthorizationInfo(roleNames);
290         info.setStringPermissions( permissions );
291         return info;
292 
293     }
294 
295     protected Set<String> getRoleNamesForUser(Connection conn, String username) throws SQLException {
296         PreparedStatement ps = null;
297         ResultSet rs = null;
298         Set<String> roleNames = new LinkedHashSet<String>();
299         try {
300             ps = conn.prepareStatement(userRolesQuery);
301             ps.setString(1, username);
302 
303             // Execute query
304             rs = ps.executeQuery();
305 
306             // Loop over results and add each returned role to a set
307             while (rs.next()) {
308 
309                 String roleName = rs.getString(1);
310 
311                 // Add the role to the list of names if it isn't null
312                 if (roleName != null) {
313                     roleNames.add(roleName);
314                 } else {
315                     if (log.isWarnEnabled()) {
316                         log.warn("Null role name found while retrieving role names for user [" + username + "]");
317                     }
318                 }
319             }
320         } finally {
321             JdbcUtils.closeResultSet(rs);
322             JdbcUtils.closeStatement(ps);
323         }
324         return roleNames;
325     }
326 
327     protected Set<String> getPermissions(Connection conn, String username, Collection<String> roleNames) throws SQLException {
328         PreparedStatement ps = null;
329         ResultSet rs = null;
330         Set<String> permissions = new LinkedHashSet<String>();
331         try {
332             for (String roleName : roleNames) {
333 
334                 ps = conn.prepareStatement(permissionsQuery);
335                 ps.setString(1, roleName);
336 
337                 // Execute query
338                 rs = ps.executeQuery();
339 
340                 // Loop over results and add each returned role to a set
341                 while (rs.next()) {
342 
343                     String permissionString = rs.getString(1);
344 
345                     // Add the permission to the set of permissions
346                     permissions.add(permissionString);
347                 }
348 
349             }
350         } finally {
351             JdbcUtils.closeResultSet(rs);
352             JdbcUtils.closeStatement(ps);
353         }
354 
355         return permissions;
356     }
