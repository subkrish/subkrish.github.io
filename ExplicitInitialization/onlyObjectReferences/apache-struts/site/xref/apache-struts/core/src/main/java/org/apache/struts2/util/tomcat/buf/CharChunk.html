23   * Utilities to manipulate char chunks. While String is
24   * the easiest way to manipulate chars ( search, substrings, etc),
25   * it is known to not be the most efficient solution - Strings are

235     /** Add data to the buffer
236      */
237     public void append( char src[], int off, int len )
238         throws IOException
239     {
240         // will grow, up to limit
241         makeSpace( len );
242 
243         // if we don't have limit: makeSpace can grow as it wants
244         if( limit < 0 ) {
245             // assert: makeSpace made enough space
246             System.arraycopy( src, off, buff, end, len );
247             end+=len;
248             return;
249         }
250 
251         // Optimize on a common case.
252         // If the source is going to fill up all the space in buffer, may
253         // as well write it directly to the output, and avoid an extra copy
254         if ( optimizedWrite && len == limit && end == start && out != null ) {
255             out.realWriteChars( src, off, len );
256             return;
257         }
258 
259         // if we have limit and we're below
260         if( len <= limit - end ) {
261             // makeSpace will grow the buffer to the limit,
262             // so we have space
263             System.arraycopy( src, off, buff, end, len );
264 
265             end+=len;
266             return;
267         }
268 
269         // need more space than we can afford, need to flush
270         // buffer
271 
272         // the buffer is already at ( or bigger than ) limit
273 
274         // Optimization:
275         // If len-avail < length ( i.e. after we fill the buffer with
276         // what we can, the remaining will fit in the buffer ) we'll just
277         // copy the first part, flush, then copy the second part - 1 write
278         // and still have some space for more. We'll still have 2 writes, but
279         // we write more on the first.
280 
281         if( len + end < 2 * limit ) {
282             /* If the request length exceeds the size of the output buffer,
283                flush the output buffer and then write the data directly.
284                We can't avoid 2 writes, but we can write more on the second
285             */
286             int avail=limit-end;
287             System.arraycopy(src, off, buff, end, avail);
288             end += avail;
289 
290             flushBuffer();
291 
292             System.arraycopy(src, off+avail, buff, end, len - avail);
293             end+= len - avail;
294 
295         } else {        // len > buf.length + avail
296             // long write - flush the buffer and write the rest
297             // directly from source
298             flushBuffer();
299 
300             out.realWriteChars( src, off, len );
301         }
302     }

311     /** Append a string to the buffer
312      */
313     public void append(String s, int off, int len) throws IOException {
314         if (s==null) {
315             return;
316         }
317 
318         // will grow, up to limit
319         makeSpace( len );
320 
321         // if we don't have limit: makeSpace can grow as it wants
322         if( limit < 0 ) {
323             // assert: makeSpace made enough space
324             s.getChars(off, off+len, buff, end );
325             end+=len;
326             return;
327         }
328 
329         int sOff = off;
330         int sEnd = off + len;
331         while (sOff < sEnd) {
332             int d = min(limit - end, sEnd - sOff);
333             s.getChars( sOff, sOff+d, buff, end);
334             sOff += d;
335             end += d;
336             if (end >= limit) {
337                 flushBuffer();
338             }
339         }
340     }

344     public int substract()
345         throws IOException {
346 
347         if ((end - start) == 0) {
348             if (in == null) {
349                 return -1;
350             }
351             int n = in.realReadChars(buff, end, buff.length - end);
352             if (n < 0) {
353                 return -1;
354             }
355         }
356 
357         return (buff[start++]);
358 
359     }

361     public int substract( char src[], int off, int len )
362         throws IOException {
363 
364         if ((end - start) == 0) {
365             if (in == null) {
366                 return -1;
367             }
368             int n = in.realReadChars( buff, end, buff.length - end);
369             if (n < 0) {
370                 return -1;
371             }
372         }
373 
374         int n = len;
375         if (len > getLength()) {
376             n = getLength();
377         }
378         System.arraycopy(buff, start, src, off, n);
379         start += n;
380         return n;
381 
382     }

385     public void flushBuffer()
386         throws IOException
387     {
388         //assert out!=null
389         if( out==null ) {
390             throw new IOException( "Buffer overflow, no sink " + limit + " " +
391                                    buff.length  );
392         }
393         out.realWriteChars( buff, start, end - start );
394         end=start;
395     }

397     /** Make space for len chars. If len is small, allocate
398      *        a reserve space too. Never grow bigger than limit.
399      */
400     public void makeSpace(int count)
401     {
402         char[] tmp = null;
403 
404         int newSize;
405         int desiredSize=end + count;
406 
407         // Can't grow above the limit
408         if( limit > 0 &&
409             desiredSize > limit) {
410             desiredSize=limit;
411         }
412 
413         if( buff==null ) {
414             if( desiredSize < 256 )
415              {
416                 desiredSize=256; // take a minimum
417             }
418             buff=new char[desiredSize];
419         }
420 
421         // limit < buf.length ( the buffer is already big )
422         // or we already have space XXX
423         if( desiredSize <= buff.length) {
424             return;
425         }
426         // grow in larger chunks
427         if( desiredSize < 2 * buff.length ) {
428             newSize= buff.length * 2;
429             if( limit >0 &&
430                 newSize > limit ) {
431                 newSize=limit;
432             }
433             tmp=new char[newSize];
434         } else {
435             newSize= buff.length * 2 + count ;
436             if( limit > 0 &&
437                 newSize > limit ) {
438                 newSize=limit;
439             }
440             tmp=new char[newSize];
441         }
442 
443         System.arraycopy(buff, 0, tmp, 0, end);
444         buff = tmp;
445         tmp = null;
446     }

474     /**
475      * Compares the message bytes to the specified String object.
476      * @param s the String to compare
477      * @return true if the comparison succeeded, false otherwise
478      */
479     public boolean equals(String s) {
480         char[] c = buff;
481         int len = end-start;
482         if (c == null || len != s.length()) {
483             return false;
484         }
485         int off = start;
486         for (int i = 0; i < len; i++) {
487             if (c[off++] != s.charAt(i)) {
488                 return false;
489             }
490         }
491         return true;
492     }

518     public boolean equals(char b2[], int off2, int len2) {
519         char b1[]=buff;
520         if( b1==null && b2==null ) {
521             return true;
522         }
523 
524         if (b1== null || b2==null || end-start != len2) {
525             return false;
526         }
527         int off1 = start;
528         int len=end-start;
529         while ( len-- > 0) {
530             if (b1[off1++] != b2[off2++]) {
531                 return false;
532             }
533         }
534         return true;

597     @Override
598     public int hashCode() {
599         if (hasHashCode) {
600             return hashCode;
601         }
602         int code = 0;
603 
604         code = hash();
605         hashCode = code;
606         hasHashCode = true;
607         return code;
608     }

610     // normal hash.
611     public int hash() {
612         int code=0;
613         for (int i = start; i < start + end-start; i++) {
614             code = code * 37 + buff[i];
615         }
616         return code;
617     }

632     public static int indexOf( char chars[], int off, int cend, char qq )
633     {
634         while( off < cend ) {
635             char b=chars[off];
636             if( b==qq ) {
637                 return off;
638             }
639             off++;
640         }
641         return -1;
642     }

645     public int indexOf(String src, int srcOff, int srcLen, int myOff ) {
646         char first=src.charAt( srcOff );
647 
648         // Look for first char
649         int srcEnd = srcOff + srcLen;
650 
651         for( int i=myOff+start; i <= (end - srcLen); i++ ) {
652             if( buff[i] != first ) {
653                 continue;
654             }
655             // found first char, now look for a match
656             int myPos=i+1;
657             for( int srcPos=srcOff + 1; srcPos< srcEnd;) {
658                 if( buff[myPos++] != src.charAt( srcPos++ )) {
659                     break;
660                 }
661                 if( srcPos==srcEnd )
662                  {
663                     return i-start; // found it
664                 }
665             }
666         }
667         return -1;
668     }

684     public CharSequence subSequence(int start, int end) {
685         try {
686             CharChunk result = (CharChunk) this.clone();
687             result.setOffset(this.start + start);
688             result.setEnd(this.start + end);
689             return result;
690         } catch (CloneNotSupportedException e) {
691             // Cannot happen
692             return null;
693         }
