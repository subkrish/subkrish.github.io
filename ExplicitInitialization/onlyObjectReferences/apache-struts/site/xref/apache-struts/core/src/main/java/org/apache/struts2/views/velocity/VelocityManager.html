119     /**
120      * <p>
121      * This method is responsible for creating the standard VelocityContext used by all WW2 velocity views.  The
122      * following context parameters are defined:
123      * </p>
124      *
125      * <ul>
126      * <li><strong>request</strong> - the current HttpServletRequest</li>
127      * <li><strong>response</strong> - the current HttpServletResponse</li>
128      * <li><strong>stack</strong> - the current {@link ValueStack}</li>
129      * <li><strong>ognl</strong> - an {@link OgnlTool}</li>
130      * <li><strong>struts</strong> - an instance of {@link org.apache.struts2.util.StrutsUtil}</li>
131      * <li><strong>action</strong> - the current Struts action</li>
132      * </ul>
133      *
134      * @param stack the current {@link ValueStack}
135      * @param req the current HttpServletRequest
136      * @param res the current HttpServletResponse
137      * @return a new StrutsVelocityContext
138      */
139     public Context createContext(ValueStack stack, HttpServletRequest req, HttpServletResponse res) {
140         Context result = null;
141         VelocityContext[] chainedContexts = prepareChainedContexts(req, res, stack.getContext());
142         StrutsVelocityContext context = new StrutsVelocityContext(chainedContexts, stack);
143         Map standardMap = ContextUtil.getStandardContext(stack, req, res);
144         for (Iterator iterator = standardMap.entrySet().iterator(); iterator.hasNext();) {
145             Map.Entry entry = (Map.Entry) iterator.next();
146             context.put((String) entry.getKey(), entry.getValue());
147         }
148         context.put(STRUTS, new VelocityStrutsUtil(velocityEngine, context, stack, req, res));
149 
150 
151         ServletContext ctx = null;
152         try {
153             ctx = ServletActionContext.getServletContext();
154         } catch (NullPointerException npe) {
155             // in case this was used outside the lifecycle of struts servlet
156             LOG.debug("internal toolbox context ignored");
157         }
158 
159         if (toolboxManager != null && ctx != null) {
160             ChainedContext chained = new ChainedContext(context, velocityEngine, req, res, ctx);
161             chained.setToolbox(toolboxManager.getToolbox(chained));
162             result = chained;
163         } else {
164             result = context;
165         }
166 
167         req.setAttribute(KEY_VELOCITY_STRUTS_CONTEXT, result);
168         return result;
169     }

171     /**
172      * constructs contexts for chaining on this request.  This method does not
173      * perform any initialization of the contexts.  All that must be done in the
174      * context itself.
175      *
176      * @param servletRequest the servlet request object
177      * @param servletResponse the servlet response object
178      * @param extraContext map with extra context
179      * @return an VelocityContext[] of contexts to chain
180      */
181     protected VelocityContext[] prepareChainedContexts(HttpServletRequest servletRequest, HttpServletResponse servletResponse, Map extraContext) {
182         if (this.chainedContextNames == null) {
183             return null;
184         }
185         List contextList = new ArrayList();
186         for (int i = 0; i < chainedContextNames.length; i++) {
187             String className = chainedContextNames[i];
188             try {
189                 VelocityContext velocityContext = (VelocityContext) objectFactory.buildBean(className, null);
190                 contextList.add(velocityContext);
191             } catch (Exception e) {
192                 LOG.warn("Warning. {} caught while attempting to instantiate a chained VelocityContext, {} -- skipping", e.getClass().getName(), className);
193             }
194         }
195         if (contextList.size() > 0) {
196             VelocityContext[] extraContexts = new VelocityContext[contextList.size()];
197             contextList.toArray(extraContexts);
198             return extraContexts;
199         } else {
200             return null;
201         }
202     }

204     /**
205      * initializes the VelocityManager.  this should be called during the initialization process, say by
206      * ServletDispatcher.  this may be called multiple times safely although calls beyond the first won't do anything
207      *
208      * @param context the current servlet context
209      */
210     public synchronized void init(ServletContext context) {
211         if (velocityEngine == null) {
212             velocityEngine = newVelocityEngine(context);
213         }
214         this.initToolbox(context);
215     }

217     /**
218      * load optional velocity properties using the following loading strategy
219      * <ul>
220      * <li>relative to the servlet context path</li>
221      * <li>relative to the WEB-INF directory</li>
222      * <li>on the classpath</li>
223      * </ul>
224      *
225      * @param context the current ServletContext.  may <b>not</b> be null
226      * @return the optional properties if struts.velocity.configfile was specified, an empty Properties file otherwise
227      */
228     public Properties loadConfiguration(ServletContext context) {
229         if (context == null) {
230             String gripe = "Error attempting to create a loadConfiguration from a null ServletContext!";
231             LOG.error(gripe);
232             throw new IllegalArgumentException(gripe);
233         }
234 
235         Properties properties = new Properties();
236 
237         // now apply our systemic defaults, then allow user to override
238         applyDefaultConfiguration(context, properties);
239 
240 
241         String defaultUserDirective = properties.getProperty("userdirective");
242 
243         /**
244          * if the user has specified an external velocity configuration file, we'll want to search for it in the
245          * following order
246          *
247          * 1. relative to the context path
248          * 2. relative to /WEB-INF
249          * 3. in the class path
250          */
251         String configfile;
252 
253         if (customConfigFile != null) {
254             configfile = customConfigFile;
255         } else {
256             configfile = "velocity.properties";
257         }
258 
259         configfile = configfile.trim();
260 
261         InputStream in = null;
262         String resourceLocation = null;
263 
264         try {
265             if (context.getRealPath(configfile) != null) {
266                 // 1. relative to context path, i.e. /velocity.properties
267                 String filename = context.getRealPath(configfile);
268 
269                 if (filename != null) {
270                     File file = new File(filename);
271 
272                     if (file.isFile()) {
273                         resourceLocation = file.getCanonicalPath() + " from file system";
274                         in = new FileInputStream(file);
275                     }
276 
277                     // 2. if nothing was found relative to the context path, search relative to the WEB-INF directory
278                     if (in == null) {
279                         file = new File(context.getRealPath("/WEB-INF/" + configfile));
280 
281                         if (file.isFile()) {
282                             resourceLocation = file.getCanonicalPath() + " from file system";
283                             in = new FileInputStream(file);
284                         }
285                     }
286                 }
287             }
288 
289             // 3. finally, if there's no physical file, how about something in our classpath
290             if (in == null) {
291                 in = VelocityManager.class.getClassLoader().getResourceAsStream(configfile);
292                 if (in != null) {
293                     resourceLocation = configfile + " from classloader";
294                 }
295             }
296 
297             // if we've got something, load 'er up
298             if (in != null) {
299                 LOG.info("Initializing velocity using {}", resourceLocation);
300                 properties.load(in);
301             }
302         } catch (IOException e) {
303             LOG.warn("Unable to load velocity configuration {}", resourceLocation, e);
304         } finally {
305             if (in != null) {
306                 try {
307                     in.close();
308                 } catch (IOException e) {
309                 }
310             }
311         }
312 
313         // overide with programmatically set properties
314         if (this.velocityProperties != null) {
315             Iterator keys = this.velocityProperties.keySet().iterator();
316             while (keys.hasNext()) {
317                 String key = (String) keys.next();
318                 properties.setProperty(key, this.velocityProperties.getProperty(key));
319             }
320         }
321 
322         String userdirective = properties.getProperty("userdirective");
323 
324         if ((userdirective == null) || userdirective.trim().equals("")) {
325             userdirective = defaultUserDirective;
326         } else {
327             userdirective = userdirective.trim() + "," + defaultUserDirective;
328         }
329 
330         properties.setProperty("userdirective", userdirective);
331 
332 
333         // for debugging purposes, allows users to dump out the properties that have been configured
334         if (LOG.isDebugEnabled()) {
335             LOG.debug("Initializing Velocity with the following properties ...");
336 
337             for (Iterator iter = properties.keySet().iterator(); iter.hasNext(); ) {
338                 String key = (String) iter.next();
339                 String value = properties.getProperty(key);
340                 LOG.debug("    '{}' = '{}'", key, value);
341             }
342         }
343 
344         return properties;
345     }

361     /**
362      * allow users to specify via the struts.properties file a set of additional VelocityContexts to chain to the
363      * the StrutsVelocityContext.  The intent is to allow these contexts to store helper objects that the ui
364      * developer may want access to.  Examples of reasonable VelocityContexts would be an IoCVelocityContext, a
365      * SpringReferenceVelocityContext, and a ToolboxVelocityContext
366      *
367      * @param contexts comma separated velocity context's
368      */
369     @Inject(StrutsConstants.STRUTS_VELOCITY_CONTEXTS)
370     public void setChainedContexts(String contexts) {
371         // we expect contexts to be a comma separated list of classnames
372         StringTokenizer st = new StringTokenizer(contexts, ",");
373         List<String> contextList = new ArrayList<>();
374 
375         while (st.hasMoreTokens()) {
376             String classname = st.nextToken();
377             contextList.add(classname);
378         }
379         if (contextList.size() > 0) {
380             String[] chainedContexts = new String[contextList.size()];
381             contextList.toArray(chainedContexts);
382             this.chainedContextNames = chainedContexts;
383         }
384     }

404     /**
405      * <p>
406      * Instantiates a new VelocityEngine.
407      * </p>
408      * <p>
409      * The following is the default Velocity configuration
410      * </p>
411      *
412      * <pre>
413      *  resource.loader = file, class
414      *  file.resource.loader.path = real path of webapp
415      *  class.resource.loader.description = Velocity Classpath Resource Loader
416      *  class.resource.loader.class = org.apache.struts2.views.velocity.StrutsResourceLoader
417      * </pre>
418      * <p>
419      * this default configuration can be overridden by specifying a struts.velocity.configfile property in the
420      * struts.properties file.  the specified config file will be searched for in the following order:
421      * </p>
422      *
423      * <ul>
424      * <li>relative to the servlet context path</li>
425      * <li>relative to the WEB-INF directory</li>
426      * <li>on the classpath</li>
427      * </ul>
428      *
429      * @param context the current ServletContext. may <b>not</b> be null
430      *
431      * @return the new velocity engine
432      */
433     protected VelocityEngine newVelocityEngine(ServletContext context) {
434         if (context == null) {
435             String gripe = "Error attempting to create a new VelocityEngine from a null ServletContext!";
436             LOG.error(gripe);
437             throw new IllegalArgumentException(gripe);
438         }
439 
440         Properties p = loadConfiguration(context);
441 
442         VelocityEngine velocityEngine = new VelocityEngine();
443 
444         //  Set the velocity attribute for the servlet context
445         //  if this is not set the webapp loader WILL NOT WORK
446         velocityEngine.setApplicationAttribute(ServletContext.class.getName(),
447                 context);
448 
449         try {
450             velocityEngine.init(p);
451         } catch (Exception e) {
452             throw new StrutsException("Unable to instantiate VelocityEngine!", e);
453         }
454 
455         return velocityEngine;
456     }

458     /**
459      * once we've loaded up the user defined configurations, we will want to apply Struts specification configurations.
460      * <ul>
461      * <li>if Velocity.RESOURCE_LOADER has not been defined, then we will use the defaults which is a joined file,
462      * class loader for unpackaed wars and a straight class loader otherwise</li>
463      * <li>we need to define the various Struts custom user directives such as #param, #tag, and #bodytag</li>
464      * </ul>
465      *
466      * @param context the servlet context
467      * @param properties velocity properties
468      */
469     private void applyDefaultConfiguration(ServletContext context, Properties properties) {
470         // ensure that caching isn't overly aggressive
471 
472         /**
473          * Load a default resource loader definition if there isn't one present.
474          * Ben Hall (22/08/2003)
475          */
476         if (properties.getProperty(Velocity.RESOURCE_LOADER) == null) {
477             properties.setProperty(Velocity.RESOURCE_LOADER, "strutsfile, strutsclass");
478         }
479 
480         /**
481          * If there's a "real" path add it for the strutsfile resource loader.
482          * If there's no real path and they haven't configured a loader then we change
483          * resource loader property to just use the strutsclass loader
484          * Ben Hall (22/08/2003)
485          */
486         if (context.getRealPath("") != null) {
487             properties.setProperty("strutsfile.resource.loader.description", "Velocity File Resource Loader");
488             properties.setProperty("strutsfile.resource.loader.class", "org.apache.velocity.runtime.resource.loader.FileResourceLoader");
489             properties.setProperty("strutsfile.resource.loader.path", context.getRealPath(""));
490             properties.setProperty("strutsfile.resource.loader.modificationCheckInterval", "2");
491             properties.setProperty("strutsfile.resource.loader.cache", "true");
492         } else {
493             // remove strutsfile from resource loader property
494             String prop = properties.getProperty(Velocity.RESOURCE_LOADER);
495             if (prop.indexOf("strutsfile,") != -1) {
496                 prop = replace(prop, "strutsfile,", "");
497             } else if (prop.indexOf(", strutsfile") != -1) {
498                 prop = replace(prop, ", strutsfile", "");
499             } else if (prop.indexOf("strutsfile") != -1) {
500                 prop = replace(prop, "strutsfile", "");
501             }
502 
503             properties.setProperty(Velocity.RESOURCE_LOADER, prop);
504         }
505 
506         /**
507          * Refactored the Velocity templates for the Struts taglib into the classpath from the web path.  This will
508          * enable Struts projects to have access to the templates by simply including the Struts jar file.
509          * Unfortunately, there does not appear to be a macro for the class loader keywords
510          * Matt Ho - Mon Mar 17 00:21:46 PST 2003
511          */
512         properties.setProperty("strutsclass.resource.loader.description", "Velocity Classpath Resource Loader");
513         properties.setProperty("strutsclass.resource.loader.class", "org.apache.struts2.views.velocity.StrutsResourceLoader");
514         properties.setProperty("strutsclass.resource.loader.modificationCheckInterval", "2");
515         properties.setProperty("strutsclass.resource.loader.cache", "true");
516 
517         // components
518         StringBuilder sb = new StringBuilder();
519 
520         for (TagLibraryDirectiveProvider tagLibrary : tagLibraries) {
521             List<Class> directives = tagLibrary.getDirectiveClasses();
522             for (Class directive : directives) {
523                 addDirective(sb, directive);
524             }
525         }
526 
527         String directives = sb.toString();
528 
529         String userdirective = properties.getProperty("userdirective");
530         if ((userdirective == null) || userdirective.trim().equals("")) {
531             userdirective = directives;
532         } else {
533             userdirective = userdirective.trim() + "," + directives;
534         }
535 
536         properties.setProperty("userdirective", userdirective);
537     }
538 
539     private void addDirective(StringBuilder sb, Class clazz) {
540         sb.append(clazz.getName()).append(",");
541     }

543     private static final String replace(String string, String oldString, String newString) {
544         if (string == null) {
545             return null;
546         }
547         // If the newString is null, just return the string since there's nothing to replace.
548         if (newString == null) {
549             return string;
550         }
551         int i = 0;
552         // Make sure that oldString appears at least once before doing any processing.
553         if ((i = string.indexOf(oldString, i)) >= 0) {
554             // Use char []'s, as they are more efficient to deal with.
555             char[] string2 = string.toCharArray();
556             char[] newString2 = newString.toCharArray();
557             int oLength = oldString.length();
558             StringBuilder buf = new StringBuilder(string2.length);
559             buf.append(string2, 0, i).append(newString2);
560             i += oLength;
561             int j = i;
562             // Replace all remaining instances of oldString with newString.
563             while ((i = string.indexOf(oldString, i)) > 0) {
564                 buf.append(string2, j, i - j).append(newString2);
565                 i += oLength;
566                 j = i;
567             }
568             buf.append(string2, j, string2.length - j);
569             return buf.toString();
570         }
571         return string;
572     }
