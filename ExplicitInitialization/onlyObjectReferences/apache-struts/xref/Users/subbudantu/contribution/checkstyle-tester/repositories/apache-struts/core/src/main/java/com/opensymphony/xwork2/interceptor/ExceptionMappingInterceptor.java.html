18  import com.opensymphony.xwork2.ActionInvocation;
19  import com.opensymphony.xwork2.config.entities.ExceptionMappingConfig;
20  import org.apache.logging.log4j.Logger;
21  import org.apache.logging.log4j.LogManager;

24  import java.util.List;

27  /**
28   * <!-- START SNIPPET: description -->
29   * <p>
30   * This interceptor forms the core functionality of the exception handling feature. Exception handling allows you to map
31   * an exception to a result code, just as if the action returned a result code instead of throwing an unexpected
32   * exception. When an exception is encountered, it is wrapped with an {@link ExceptionHolder} and pushed on the stack,
33   * providing easy access to the exception from within your result.
34   * </p>
35   *
36   * <p>
37   * <b>Note:</b> While you can configure exception mapping in your configuration file at any point, the configuration
38   * will not have any effect if this interceptor is not in the interceptor stack for your actions. It is recommended that
39   * you make this interceptor the first interceptor on the stack, ensuring that it has full access to catch any
40   * exception, even those caused by other interceptors.
41   * </p>
42   *
43   * <!-- END SNIPPET: description -->
44   *
45   * <p><u>Interceptor parameters:</u></p>
46   *
47   * <!-- START SNIPPET: parameters -->
48   *
49   * <ul>
50   *
51   * <li>logEnabled (optional) - Should exceptions also be logged? (boolean true|false)</li>
52   * 
53   * <li>logLevel (optional) - what log level should we use (<code>trace, debug, info, warn, error, fatal</code>)? - defaut is <code>debug</code></li>
54   * 
55   * <li>logCategory (optional) - If provided we would use this category (eg. <code>com.mycompany.app</code>).
56   * Default is to use <code>com.opensymphony.xwork2.interceptor.ExceptionMappingInterceptor</code>.</li>
57   *
58   * </ul>
59   *
60   * <p>
61   * The parameters above enables us to log all thrown exceptions with stacktace in our own logfile,
62   * and present a friendly webpage (with no stacktrace) to the end user.
63   * </p>
64   *
65   * <!-- END SNIPPET: parameters -->
66   *
67   * <p><u>Extending the interceptor:</u></p>
68   *
69   * <!-- START SNIPPET: extending -->
70   * <p>
71   * If you want to add custom handling for publishing the Exception, you may override
72   * {@link #publishException(com.opensymphony.xwork2.ActionInvocation, ExceptionHolder)}. The default implementation
73   * pushes the given ExceptionHolder on value stack. A custom implementation could add additional logging etc.
74   * </p>
75   * <!-- END SNIPPET: extending -->
76   *
77   * <p><u>Example code:</u></p>
78   *
79   * <pre>
80   * <!-- START SNIPPET: example -->
81   * &lt;xwork&gt;
82   *     &lt;package name="default" extends="xwork-default"&gt;
83   *         &lt;global-results&gt;
84   *             &lt;result name="error" type="freemarker"&gt;error.ftl&lt;/result&gt;
85   *         &lt;/global-results&gt;
86   *
87   *         &lt;global-exception-mappings&gt;
88   *             &lt;exception-mapping exception="java.lang.Exception" result="error"/&gt;
89   *         &lt;/global-exception-mappings&gt;
90   *
91   *         &lt;action name="test"&gt;
92   *             &lt;interceptor-ref name="exception"/&gt;
93   *             &lt;interceptor-ref name="basicStack"/&gt;
94   *             &lt;exception-mapping exception="com.acme.CustomException" result="custom_error"/&gt;
95   *             &lt;result name="custom_error"&gt;custom_error.ftl&lt;/result&gt;
96   *             &lt;result name="success" type="freemarker"&gt;test.ftl&lt;/result&gt;
97   *         &lt;/action&gt;
98   *     &lt;/package&gt;
99   * &lt;/xwork&gt;
100  * <!-- END SNIPPET: example -->
101  * </pre>
102  * 
103  * <p>
104  * This second example will also log the exceptions using our own category
105  * <code>com.mycompany.app.unhandled</code> at WARN level.
106  * </p>
107  *
108  * <pre>
109  * <!-- START SNIPPET: example2 -->
110  * &lt;xwork&gt;
111  *   &lt;package name="something" extends="xwork-default"&gt;
112  *      &lt;interceptors&gt;
113  *          &lt;interceptor-stack name="exceptionmappingStack"&gt;
114  *              &lt;interceptor-ref name="exception"&gt;
115  *                  &lt;param name="logEnabled"&gt;true&lt;/param&gt;
116  *                  &lt;param name="logCategory"&gt;com.mycompany.app.unhandled&lt;/param&gt;
117  *                  &lt;param name="logLevel"&gt;WARN&lt;/param&gt;	        		
118  *              &lt;/interceptor-ref&gt;	
119  *              &lt;interceptor-ref name="i18n"/&gt;
120  *              &lt;interceptor-ref name="staticParams"/&gt;
121  *              &lt;interceptor-ref name="params"/&gt;
122  *              &lt;interceptor-ref name="validation"&gt;
123  *                  &lt;param name="excludeMethods"&gt;input,back,cancel,browse&lt;/param&gt;
124  *              &lt;/interceptor-ref&gt;
125  *          &lt;/interceptor-stack&gt;
126  *      &lt;/interceptors&gt;
127  *
128  *      &lt;default-interceptor-ref name="exceptionmappingStack"/&gt;
129  *    
130  *      &lt;global-results&gt;
131  *           &lt;result name="unhandledException"&gt;/unhandled-exception.jsp&lt;/result&gt;
132  *      &lt;/global-results&gt;
133  *
134  *      &lt;global-exception-mappings&gt;
135  *           &lt;exception-mapping exception="java.lang.Exception" result="unhandledException"/&gt;
136  *      &lt;/global-exception-mappings&gt;
137  *        
138  *      &lt;action name="exceptionDemo" class="org.apache.struts2.showcase.exceptionmapping.ExceptionMappingAction"&gt;
139  *          &lt;exception-mapping exception="org.apache.struts2.showcase.exceptionmapping.ExceptionMappingException"
140  *                             result="damm"/&gt;
141  *          &lt;result name="input"&gt;index.jsp&lt;/result&gt;
142  *          &lt;result name="success"&gt;success.jsp&lt;/result&gt;            
143  *          &lt;result name="damm"&gt;damm.jsp&lt;/result&gt;
144  *      &lt;/action&gt;
145  *
146  *   &lt;/package&gt;
147  * &lt;/xwork&gt;
148  * <!-- END SNIPPET: example2 -->
149  * </pre>
150  *
151  * @author Matthew E. Porter (matthew dot porter at metissian dot com) 
152  * @author Claus Ibsen
153  */
154 public class ExceptionMappingInterceptor extends AbstractInterceptor {
155     
156     private static final Logger LOG = LogManager.getLogger(ExceptionMappingInterceptor.class);
157 
158     protected Logger categoryLogger;
159     protected boolean logEnabled = false;
160     protected String logCategory;
161     protected String logLevel;
162     
163 
164     public boolean isLogEnabled() {
165         return logEnabled;
166     }
167 
168     public void setLogEnabled(boolean logEnabled) {
169         this.logEnabled = logEnabled;
170     }
171 
172     public String getLogCategory() {
173 		return logCategory;
174 	}
175 
176 	public void setLogCategory(String logCatgory) {
177 		this.logCategory = logCatgory;
178 	}
179 
180 	public String getLogLevel() {
181 		return logLevel;
182 	}
183 
184 	public void setLogLevel(String logLevel) {
185 		this.logLevel = logLevel;
186 	}
187 
188     @Override
189     public String intercept(ActionInvocation invocation) throws Exception {
190         String result;
191 
192         try {
193             result = invocation.invoke();
194         } catch (Exception e) {
195             if (isLogEnabled()) {
196                 handleLogging(e);
197             }
198             List<ExceptionMappingConfig> exceptionMappings = invocation.getProxy().getConfig().getExceptionMappings();
199             ExceptionMappingConfig mappingConfig = this.findMappingFromExceptions(exceptionMappings, e);
200             if (mappingConfig != null && mappingConfig.getResult()!=null) {
201                 Map<String, String> mappingParams = mappingConfig.getParams();
202                 // create a mutable HashMap since some interceptors will remove parameters, and parameterMap is immutable
203                 HttpParameters parameters = HttpParameters.create(mappingParams).build();
204                 invocation.getInvocationContext().setParameters(parameters);
205                 result = mappingConfig.getResult();
206                 publishException(invocation, new ExceptionHolder(e));
207             } else {
208                 throw e;
209             }
210         }
211 
212         return result;
213     }
214 
215     /**
216      * Handles the logging of the exception.
217      * 
218      * @param e the exception to log.
219      */
220     protected void handleLogging(Exception e) {
221     	if (logCategory != null) {
222         	if (categoryLogger == null) {
223         		// init category logger
224         		categoryLogger = LogManager.getLogger(logCategory);
225         	}
226         	doLog(categoryLogger, e);
227     	} else {
228     		doLog(LOG, e);
229     	}
230     }
231     
232     /**
233      * Performs the actual logging.
234      * 
235      * @param logger  the provided logger to use.
236      * @param e  the exception to log.
237      */
238     protected void doLog(Logger logger, Exception e) {
239     	if (logLevel == null) {
240     		logger.debug(e.getMessage(), e);
241     		return;
242     	}
243     	
244     	if ("trace".equalsIgnoreCase(logLevel)) {
245     		logger.trace(e.getMessage(), e);
246     	} else if ("debug".equalsIgnoreCase(logLevel)) {
247     		logger.debug(e.getMessage(), e);
248     	} else if ("info".equalsIgnoreCase(logLevel)) {
249     		logger.info(e.getMessage(), e);
250     	} else if ("warn".equalsIgnoreCase(logLevel)) {
251     		logger.warn(e.getMessage(), e);
252     	} else if ("error".equalsIgnoreCase(logLevel)) {
253     		logger.error(e.getMessage(), e);
254     	} else if ("fatal".equalsIgnoreCase(logLevel)) {
255     		logger.fatal(e.getMessage(), e);
256     	} else {
257     		throw new IllegalArgumentException("LogLevel [" + logLevel + "] is not supported");
258     	}
259     }
260 
261     /**
262      * Try to find appropriate {@link ExceptionMappingConfig} based on provided Throwable
263      *
264      * @param exceptionMappings list of defined exception mappings
265      * @param t caught exception
266      * @return appropriate mapping or null
267      */
268     protected ExceptionMappingConfig findMappingFromExceptions(List<ExceptionMappingConfig> exceptionMappings, Throwable t) {
269     	ExceptionMappingConfig config = null;
270         // Check for specific exception mappings.
271         if (exceptionMappings != null) {
272             int deepest = Integer.MAX_VALUE;
273             for (Object exceptionMapping : exceptionMappings) {
274                 ExceptionMappingConfig exceptionMappingConfig = (ExceptionMappingConfig) exceptionMapping;
275                 int depth = getDepth(exceptionMappingConfig.getExceptionClassName(), t);
276                 if (depth >= 0 && depth < deepest) {
277                     deepest = depth;
278                     config = exceptionMappingConfig;
279                 }
280             }
281         }
282         return config;
283     }
284 
285     /**
286      * Return the depth to the superclass matching. 0 means ex matches exactly. Returns -1 if there's no match.
287      * Otherwise, returns depth. Lowest depth wins.
288      *
289      * @param exceptionMapping  the mapping classname
290      * @param t  the cause
291      * @return the depth, if not found -1 is returned.
292      */
293     public int getDepth(String exceptionMapping, Throwable t) {
294         return getDepth(exceptionMapping, t.getClass(), 0);
295     }
296 
297     private int getDepth(String exceptionMapping, Class exceptionClass, int depth) {
298         if (exceptionClass.getName().contains(exceptionMapping)) {
299             // Found it!
300             return depth;
301         }
302         // If we've gone as far as we can go and haven't found it...
303         if (exceptionClass.equals(Throwable.class)) {
304             return -1;
305         }
306         return getDepth(exceptionMapping, exceptionClass.getSuperclass(), depth + 1);
307     }
308 
309     /**
310      * Default implementation to handle ExceptionHolder publishing. Pushes given ExceptionHolder on the stack.
311      * Subclasses may override this to customize publishing.
312      *
313      * @param invocation The invocation to publish Exception for.
314      * @param exceptionHolder The exceptionHolder wrapping the Exception to publish.
315      */
316     protected void publishException(ActionInvocation invocation, ExceptionHolder exceptionHolder) {
317         invocation.getStack().push(exceptionHolder);
318     }
