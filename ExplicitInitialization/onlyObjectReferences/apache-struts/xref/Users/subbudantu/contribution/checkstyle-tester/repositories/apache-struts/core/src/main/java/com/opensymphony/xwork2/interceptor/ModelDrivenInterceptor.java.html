16  package com.opensymphony.xwork2.interceptor;
17  
18  import com.opensymphony.xwork2.ActionInvocation;
19  import com.opensymphony.xwork2.ModelDriven;
20  import com.opensymphony.xwork2.util.CompoundRoot;
21  import com.opensymphony.xwork2.util.ValueStack;
22  
23  
24  /**
25   * <!-- START SNIPPET: description -->
26   *
27   * Watches for {@link ModelDriven} actions and adds the action's model on to the value stack.
28   *
29   * <p> <b>Note:</b>  The ModelDrivenInterceptor must come before the both {@link StaticParametersInterceptor} and
30   * {@link ParametersInterceptor} if you want the parameters to be applied to the model.
31   * </p>
32   * <p> <b>Note:</b>  The ModelDrivenInterceptor will only push the model into the stack when the
33   * model is not null, else it will be ignored.
34   * </p>
35   *
36   * <!-- END SNIPPET: description -->
37   *
38   * <p><u>Interceptor parameters:</u></p>
39   *
40   * <!-- START SNIPPET: parameters -->
41   *
42   * <ul>
43   *
44   * <li>refreshModelBeforeResult - set to true if you want the model to be refreshed on the value stack after action
45   * execution and before result execution.  The setting is useful if you want to change the model instance during the
46   * action execution phase, like when loading it from the data layer.  This will result in getModel() being called at
47   * least twice.</li>
48   *
49   * </ul>
50   *
51   * <!-- END SNIPPET: parameters -->
52   *
53   * <p><u>Extending the interceptor:</u></p>
54   *
55   * <!-- START SNIPPET: extending -->
56   *
57   * There are no known extension points to this interceptor.
58   *
59   * <!-- END SNIPPET: extending -->
60   *
61   * <p><u>Example code:</u></p>
62   *
63   * <pre>
64   * <!-- START SNIPPET: example -->
65   * &lt;action name=&quot;someAction&quot; class=&quot;com.examples.SomeAction&quot;&gt;
66   *     &lt;interceptor-ref name=&quot;modelDriven&quot;/&gt;
67   *     &lt;interceptor-ref name=&quot;basicStack&quot;/&gt;
68   *     &lt;result name=&quot;success&quot;&gt;good_result.ftl&lt;/result&gt;
69   * &lt;/action&gt;
70   * <!-- END SNIPPET: example -->
71   * </pre>
72   * 
73   * @author tm_jee
74   * @version $Date$ $Id$
75   */
76  public class ModelDrivenInterceptor extends AbstractInterceptor {
77  
78      protected boolean refreshModelBeforeResult = false;
79  
80      public void setRefreshModelBeforeResult(boolean val) {
81          this.refreshModelBeforeResult = val;
82      }
83  
84      @Override
85      public String intercept(ActionInvocation invocation) throws Exception {
86          Object action = invocation.getAction();
87  
88          if (action instanceof ModelDriven) {
89              ModelDriven modelDriven = (ModelDriven) action;
90              ValueStack stack = invocation.getStack();
91              Object model = modelDriven.getModel();
92              if (model !=  null) {
93              	stack.push(model);
94              }
95              if (refreshModelBeforeResult) {
96                  invocation.addPreResultListener(new RefreshModelBeforeResult(modelDriven, model));
97              }
98          }
99          return invocation.invoke();
100     }
101 
102     /**
103      * Refreshes the model instance on the value stack, if it has changed
104      */
105     protected static class RefreshModelBeforeResult implements PreResultListener {
106         private Object originalModel = null;
107         protected ModelDriven action;
108 
109 
110         public RefreshModelBeforeResult(ModelDriven action, Object model) {
111             this.originalModel = model;
112             this.action = action;
113         }
114 
115         public void beforeResult(ActionInvocation invocation, String resultCode) {
116             ValueStack stack = invocation.getStack();
117             CompoundRoot root = stack.getRoot();
118 
119             boolean needsRefresh = true;
120             Object newModel = action.getModel();
121 
122             // Check to see if the new model instance is already on the stack
123             for (Object item : root) {
124                 if (item.equals(newModel)) {
125                     needsRefresh = false;
126                     break;
127                 }
128             }
129 
130             // Add the new model on the stack
131             if (needsRefresh) {
132 
133                 // Clear off the old model instance
134                 if (originalModel != null) {
135                     root.remove(originalModel);
136                 }
137                 if (newModel != null) {
138                     stack.push(newModel);
139                 }
140             }
141         }
142     }
143 }
