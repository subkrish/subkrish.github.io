18  
19  import com.opensymphony.xwork2.ActionContext;
20  import com.opensymphony.xwork2.ActionInvocation;

22  import com.opensymphony.xwork2.spring.SpringObjectFactory;

26  import org.springframework.context.ApplicationContext;
27  import org.springframework.context.ApplicationContextAware;
28  import org.springframework.web.context.WebApplicationContext;

37   * <!-- END SNIPPET: parameters -->
38   *
39   * <!-- START SNIPPET: extending -->
40   * TODO: Discuss some possible extension of the Interceptor.
41   * <!-- END SNIPPET: extending -->
42   *
43   * <pre>
44   * <!-- START SNIPPET: example -->
45   * &lt;!-- TODO: Describe how the Interceptor reference will effect execution --&gt;
46   * &lt;action name="someAction" class="com.examples.SomeAction"&gt;
47   *      TODO: fill in the interceptor reference.
48   *     &lt;interceptor-ref name=""/&gt;
49   *     &lt;result name="success"&gt;good_result.ftl&lt;/result&gt;
50   * &lt;/action&gt;
51   * <!-- END SNIPPET: example -->
52   * </pre>
53   *
54   * <p>
55   * Autowires action classes to Spring beans.  The strategy for autowiring the beans can be configured
56   * by setting the parameter on the interceptor.  Actions that need access to the <code>ActionContext</code>
57   * can implements the <code>ApplicationContextAware</code> interface.  The context will also be placed on
58   * the action context under the APPLICATION_CONTEXT attribute.
59   * </p>
60   *
61   * @author Simon Stewart
62   * @author Eric Hauser
63   */
64  public class ActionAutowiringInterceptor extends AbstractInterceptor implements ApplicationContextAware {
65      private static final Logger LOG = LogManager.getLogger(ActionAutowiringInterceptor.class);
66  
67      public static final String APPLICATION_CONTEXT = "com.opensymphony.xwork2.spring.interceptor.ActionAutowiringInterceptor.applicationContext";
68  
69      private boolean initialized = false;
70      private ApplicationContext context;
71      private SpringObjectFactory factory;
72      private Integer autowireStrategy;
73  
74      /**
75       * @param autowireStrategy the autowire strategy
76       */
77      public void setAutowireStrategy(Integer autowireStrategy) {
78          this.autowireStrategy = autowireStrategy;
79      }
80  
81      /**
82       * <p>
83       * Looks for the <code>ApplicationContext</code> under the attribute that the Spring listener sets in
84       * the servlet context.  The configuration is done the first time here instead of in init() since the
85       * <code>ActionContext</code> is not available during <code>Interceptor</code> initialization.
86       * </p>
87       *
88       * <p>
89       * Autowires the action to Spring beans and places the <code>ApplicationContext</code>
90       * on the <code>ActionContext</code>
91       * </p>
92       *
93       * <p>
94       * TODO: Should this check to see if the <code>SpringObjectFactory</code> has already been configured instead of instantiating a new one?  Or is there a good reason for the interceptor to have it's own factory?
95       * </p>
96       *
97       * @param invocation the action invocation
98       * @throws Exception in case of any errors
99       */
100     @Override public String intercept(ActionInvocation invocation) throws Exception {
101         if (!initialized) {
102             ApplicationContext applicationContext = (ApplicationContext) ActionContext.getContext().getApplication().get(
103                     WebApplicationContext.ROOT_WEB_APPLICATION_CONTEXT_ATTRIBUTE);
104 
105             if (applicationContext == null) {
106                 LOG.warn("ApplicationContext could not be found.  Action classes will not be autowired.");
107             } else {
108                 setApplicationContext(applicationContext);
109                 factory = new SpringObjectFactory();
110                 factory.setApplicationContext(getApplicationContext());
111                 if (autowireStrategy != null) {
112                     factory.setAutowireStrategy(autowireStrategy);
113                 }
114             }
115             initialized = true;
116         }
117 
118         if (factory != null) {
119             Object bean = invocation.getAction();
120             factory.autoWireBean(bean);
121     
122             ActionContext.getContext().put(APPLICATION_CONTEXT, context);
123         }
124         return invocation.invoke();
125     }
126 
127     /**
128      * @param applicationContext the application context
129      * @throws BeansException in case of errors
130      */
131     public void setApplicationContext(ApplicationContext applicationContext) throws BeansException {
132         context = applicationContext;
133     }
134 
135     /**
136      * @return the application context
137      */
138     protected ApplicationContext getApplicationContext() {
139         return context;
140     }

142 }
