40  /**
41   * <!-- START SNIPPET: javadoc -->
42   * <p>
43   * A tag that creates a HTML &lt;a &gt;.This tag supports the same attributes as the "url" tag,
44   * including nested parameters using the "param" tag.
45   * </p>
46   * <!-- END SNIPPET: javadoc -->
47   *
48   * <p><b>Examples</b></p>
49   *
50   * <pre>
51   * <!-- START SNIPPET: example1 -->
52   * &lt;s:a id="link1" href="/do-it.action"&gt;
53   *     &lt;img border="none" src="&lt;%=request.getContextPath()%&gt;/images/delete.gif"/&gt;
54   *     &lt;s:param name="id" value="1"/&gt;
55   * &lt;/s:a&gt;
56   * <!-- END SNIPPET: example1 -->
57   * </pre>
58   */
59  @StrutsTag(
60          name = "a",
61          tldTagClass = "org.apache.struts2.views.jsp.ui.AnchorTag",
62          description = "Render a HTML href element",
63          allowDynamicAttributes = true)
64  public class Anchor extends ClosingUIBean {
65      private static final Logger LOG = LogManager.getLogger(Anchor.class);
66  
67      public static final String OPEN_TEMPLATE = "a";
68      public static final String TEMPLATE = "a-close";
69      public static final String COMPONENT_NAME = Anchor.class.getName();
70  
71      protected String href;
72      protected UrlProvider urlProvider;
73      protected UrlRenderer urlRenderer;
74      protected boolean processingTagBody = false;
75      
76      //these params are passed by the Param tag
77      protected Map urlParameters = new LinkedHashMap();
78  
79      public Anchor(ValueStack stack, HttpServletRequest request, HttpServletResponse response) {
80          super(stack, request, response);
81          urlProvider = new ComponentUrlProvider(this, this.urlParameters);
82          urlProvider.setHttpServletRequest(request);
83          urlProvider.setHttpServletResponse(response);
84      }
85  
86      public String getDefaultOpenTemplate() {
87          return OPEN_TEMPLATE;
88      }
89  
90      protected String getDefaultTemplate() {
91          return TEMPLATE;
92      }
93  
94      public boolean usesBody() {
95          return true;
96      }
97  
98      @Override
99      protected void evaluateExtraParams() {
100         super.evaluateExtraParams();
101 
102         if (href != null)
103             addParameter("href", ensureAttributeSafelyNotEscaped(findString(href)));
104         else {
105             //no href, build it from URL attributes
106             StringWriter sw = new StringWriter();
107             urlRenderer.beforeRenderUrl(urlProvider);
108             urlRenderer.renderUrl(sw, urlProvider);
109             String builtHref = sw.toString();
110             if (StringUtils.isNotEmpty(builtHref))
111                 addParameter("href", ensureAttributeSafelyNotEscaped(builtHref));
112         }
113     }
114 
115     @Inject(StrutsConstants.STRUTS_URL_INCLUDEPARAMS)
116     public void setUrlIncludeParams(String urlIncludeParams) {
117        urlProvider.setUrlIncludeParams(urlIncludeParams);
118     }
119 
120     @Inject
121 	public void setUrlRenderer(UrlRenderer urlRenderer) {
122 		urlProvider.setUrlRenderer(urlRenderer);
123         this.urlRenderer = urlRenderer;
124 	}
125 
126     @Inject(required=false)
127     public void setExtraParameterProvider(ExtraParameterProvider provider) {
128         urlProvider.setExtraParameterProvider(provider);
129     }
130 
131     @Override
132     public boolean start(Writer writer) {
133         boolean result = super.start(writer);
134         this.processingTagBody = true;
135         return result;
136     }
137 
138     /**
139      * Overrides to be able to render body in a template rather than always before the template
140      */
141     public boolean end(Writer writer, String body) {
142         this.processingTagBody = false;
143         evaluateParams();
144         try {
145             addParameter("body", body);
146             mergeTemplate(writer, buildTemplateName(template, getDefaultTemplate()));
147         } catch (Exception e) {
148             LOG.error("error when rendering", e);
149         }
150         finally {
151             popComponentStack();
152         }
153 
154         return false;
155     }
156 
157 
158     public void addParameter(String key, Object value) {
159         /*
160           the parameters added by this method are used in the template. this method is also
161           called by Param to add params into ancestestor. This tag needs to keep both set of parameters
162           separated (You gotta keep 'em separated!)
163          */
164         if (processingTagBody) {
165             this.urlParameters.put(key, value);
166         } else
167             super.addParameter(key, value);
168     }
169 
170     @Override
171     public void addAllParameters(Map params) {
172         /*
173           the parameters added by this method are used in the template. this method is also
174           called by Param to add params into ancestestor. This tag needs to keep both set of parameters
175           separated (You gotta keep 'em separated!)
176          */
177         if (processingTagBody) {
178             this.urlParameters.putAll(params);
179         } else
180             super.addAllParameters(params);
181     }
182

184         return urlProvider;
185     }
