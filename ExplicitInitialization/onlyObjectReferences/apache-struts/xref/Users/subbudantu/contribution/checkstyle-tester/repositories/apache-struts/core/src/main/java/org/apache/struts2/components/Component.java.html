88       * Gets the name of this component.
89       * @return the name of this component.
90       */
91      private String getComponentName() {
92          Class c = getClass();
93          String name = c.getName();
94          int dot = name.lastIndexOf('.');
95  
96          return name.substring(dot + 1).toLowerCase();
97      }

126     /**
127      * Gets the component stack of this component.
128      * @return the component stack of this component, never <tt>null</tt>.
129      */
130     public Stack<Component> getComponentStack() {
131         Stack<Component> componentStack = (Stack<Component>) stack.getContext().get(COMPONENT_STACK);
132         if (componentStack == null) {
133             componentStack = new Stack<>();
134             stack.getContext().put(COMPONENT_STACK, componentStack);
135         }
136         return componentStack;
137     }

163     /**
164      * Callback for the start tag of this component.
165      * Should the body be evaluated again?
166      * <br>
167      * <b>NOTE:</b> has a parameter to determine to pop the component stack.
168      * @param writer  the output writer.
169      * @param body    the rendered body.
170      * @param popComponentStack  should the component stack be popped?
171      * @return true if the body should be evaluated again
172      */
173     protected boolean end(Writer writer, String body, boolean popComponentStack) {
174         assert(body != null);
175 
176         try {
177             writer.write(body);
178         } catch (IOException e) {
179             throw new StrutsException("IOError while writing the body: " + e.getMessage(), e);
180         }
181         if (popComponentStack) {
182             popComponentStack();
183         }
184         return false;
185     }

194     /**
195      * Finds the nearest ancestor of this component stack.
196      * @param clazz the class to look for, or if assignable from.
197      * @return  the component if found, <tt>null</tt> if not.
198      */
199     protected Component findAncestor(Class clazz) {
200         Stack componentStack = getComponentStack();
201         int currPosition = componentStack.search(this);
202         if (currPosition >= 0) {
203             int start = componentStack.size() - currPosition - 1;
204 
205             //for (int i = componentStack.size() - 2; i >= 0; i--) {
206             for (int i = start; i >=0; i--) {
207                 Component component = (Component) componentStack.get(i);
208                 if (clazz.isAssignableFrom(component.getClass()) && component != this) {
209                     return component;
210                 }
211             }
212         }
213 
214         return null;
215     }

226     /**
227      * Evaluates the OGNL stack to find a String value.
228      * <br>
229      * If the given expression is <tt>null</tt> a error is logged and a <code>RuntimeException</code> is thrown
230      * constructed with a messaged based on the given field and errorMsg parameter.
231      *
232      * @param expr  OGNL expression.
233      * @param field   field name used when throwing <code>RuntimeException</code>.
234      * @param errorMsg  error message used when throwing <code>RuntimeException</code>.
235      * @return  the String value found.
236      * @throws StrutsException is thrown in case of expression is null.
237      */
238     protected String findString(String expr, String field, String errorMsg) {
239         if (expr == null) {
240             throw fieldError(field, errorMsg, null);
241         } else {
242             return findString(expr);
243         }
244     }

246     /**
247      * Constructs a <code>RuntimeException</code> based on the given information.
248      * <br>
249      * A message is constructed and logged at ERROR level before being returned
250      * as a <code>RuntimeException</code>.
251      * @param field   field name used when throwing <code>RuntimeException</code>.
252      * @param errorMsg  error message used when throwing <code>RuntimeException</code>.
253      * @param e  the caused exception, can be <tt>null</tt>.
254      * @return  the constructed <code>StrutsException</code>.
255      */
256     protected StrutsException fieldError(String field, String errorMsg, Exception e) {
257         String msg = "tag '" + getComponentName() + "', field '" + field +
258                 ( parameters != null && parameters.containsKey("name")?"', name '" + parameters.get("name"):"") +
259                 "': " + errorMsg;
260         throw new StrutsException(msg, e);
261     }

263     /**
264      * Finds a value from the OGNL stack based on the given expression.
265      * Will always evaluate <code>expr</code> against stack except when <code>expr</code>
266      * is null. If altsyntax (%{...}) is applied, simply strip it off.
267      *
268      * @param expr  the expression. Returns <tt>null</tt> if expr is null.
269      * @return the value, <tt>null</tt> if not found.
270      */
271     protected Object findValue(String expr) {
272         if (expr == null) {
273             return null;
274         }
275 
276         expr = stripExpressionIfAltSyntax(expr);
277 
278         return getStack().findValue(expr, throwExceptionOnELFailure);
279     }

285      * the parameter expression is returned as is.
286      */
287 	protected String stripExpressionIfAltSyntax(String expr) {

306 		if (altSyntax()) {
307 			return "%{" + expr + "}";
308 		}
309 		return expr;

314      * @param expr the expression.
315      * @return the found string if altSyntax is enabled. The parameter
316      * expression otherwise.

319 		if (altSyntax()) {
320 		    return findString(expr);
321 		}
322 		return expr;

325     /**
326      * <p>
327      * Evaluates the OGNL stack to find an Object value.
328      * </p>
329      *
330      * <p>
331      * Function just like <code>findValue(String)</code> except that if the
332      * given expression is <tt>null</tt> a error is logged and
333      * a <code>RuntimeException</code> is thrown constructed with a
334      * messaged based on the given field and errorMsg parameter.
335      * </p>
336      *
337      * @param expr  OGNL expression.
338      * @param field   field name used when throwing <code>RuntimeException</code>.
339      * @param errorMsg  error message used when throwing <code>RuntimeException</code>.
340      * @return  the Object found, is never <tt>null</tt>.
341      * @throws StrutsException is thrown in case of not found in the OGNL stack, or expression is <tt>null</tt>.
342      */
343     protected Object findValue(String expr, String field, String errorMsg) {
344         if (expr == null) {
345             throw fieldError(field, errorMsg, null);
346         } else {
347             Object value = null;
348             Exception problem = null;
349             try {
350                 value = findValue(expr);
351             } catch (Exception e) {
352                 problem = e;
353             }
354 
355             if (value == null) {
356                 throw fieldError(field, errorMsg, problem);
357             }
358 
359             return value;
360         }
361     }

363     /**
364      * Evaluates the OGNL stack to find an Object of the given type. Will evaluate
365      * <code>expr</code> the portion wrapped with altSyntax (%{...})
366      * against stack when altSyntax is on, else the whole <code>expr</code>
367      * is evaluated against the stack.
368      * <br>
369      * This method only supports the altSyntax. So this should be set to true.
370      * @param expr  OGNL expression.
371      * @param toType  the type expected to find.
372      * @return  the Object found, or <tt>null</tt> if not found.
373      */
374     protected Object findValue(String expr, Class toType) {
375         if (altSyntax() && toType == String.class) {
376             if (ComponentUtils.containsExpression(expr)) {
377                 return TextParseUtil.translateVariables('%', expr, stack);
378             } else {
379                 return expr;
380             }
381         } else {
382             expr = stripExpressionIfAltSyntax(expr);
383 
384             return getStack().findValue(expr, toType, throwExceptionOnELFailure);
385         }
386     }

388     /**
389      * Renders an action URL by consulting the {@link org.apache.struts2.dispatcher.mapper.ActionMapper}.
390      * @param action      the action
391      * @param namespace   the namespace
392      * @param method      the method
393      * @param req         HTTP request
394      * @param res         HTTP response
395      * @param parameters  parameters
396      * @param scheme      http or https
397      * @param includeContext  should the context path be included or not
398      * @param encodeResult    should the url be encoded
399      * @param forceAddSchemeHostAndPort    should the scheme host and port be forced
400      * @param escapeAmp    should ampersand (&amp;) be escaped to &amp;amp;
401      * @return the action url.
402      */
403     protected String determineActionURL(String action, String namespace, String method,
404                                         HttpServletRequest req, HttpServletResponse res, Map parameters, String scheme,
405                                         boolean includeContext, boolean encodeResult, boolean forceAddSchemeHostAndPort,
406                                         boolean escapeAmp) {
407         String finalAction = findString(action);
408         String finalMethod = method != null ? findString(method) : null;
409         String finalNamespace = determineNamespace(namespace, getStack(), req);
410         ActionMapping mapping = new ActionMapping(finalAction, finalNamespace, finalMethod, parameters);
411         String uri = actionMapper.getUriFromActionMapping(mapping);
412         return urlHelper.buildUrl(uri, req, res, parameters, scheme, includeContext, encodeResult, forceAddSchemeHostAndPort, escapeAmp);
413     }

415     /**
416      * Determines the namespace of the current page being renderdd. Useful for Form, URL, and href generations.
417      * @param namespace  the namespace
418      * @param stack      OGNL value stack
419      * @param req        HTTP request
420      * @return  the namepsace of the current page being rendered, is never <tt>null</tt>.
421      */
422     protected String determineNamespace(String namespace, ValueStack stack, HttpServletRequest req) {
423         String result;
424 
425         if (namespace == null) {
426             result = TagUtils.buildNamespace(actionMapper, stack, req);
427         } else {
428             result = findString(namespace);
429         }
430 
431         if (result == null) {
432             result = "";
433         }
434 
435         return result;
436     }

438     /**
439      * Pushes this component's parameter Map as well as the component itself on to the stack
440      * and then copies the supplied parameters over. Because the component's parameter Map is
441      * pushed before the component itself, any key-value pair that can't be assigned to component
442      * will be set in the parameters Map.
443      *
444      * @param params  the parameters to copy.
445      */
446     public void copyParams(Map params) {
447         stack.push(parameters);
448         stack.push(this);
449         try {
450             for (Object o : params.entrySet()) {
451                 Map.Entry entry = (Map.Entry) o;
452                 String key = (String) entry.getKey();
453 
454                 if (key.indexOf('-') >= 0) {
455                     // UI component attributes may contain hypens (e.g. data-ajax), but ognl
456                     // can't handle that, and there can't be a component property with a hypen
457                     // so into the parameters map it goes. See WW-4493
458                     parameters.put(key, entry.getValue());
459                 } else {
460                     stack.setValue(key, entry.getValue());
461                 }
462             }
463         } finally {
464             stack.pop();
465             stack.pop();
466         }
467     }

498     /**
499      * Adds the given key and value to this component's own parameter.
500      * <br>
501      * If the provided key is <tt>null</tt> nothing happens.
502      * If the provided value is <tt>null</tt> any existing parameter with
503      * the given key name is removed.
504      * @param key  the key of the new parameter to add.
505      * @param value the value associated with the key.
506      */
507     public void addParameter(String key, Object value) {
508         if (key != null) {
509             Map params = getParameters();
510 
511             if (value == null) {
512                 params.remove(key);
513             } else {
514                 params.put(key, value);
515             }
516         }
517     }

537     /**
538      * If needed caches all methods annotated by given annotation to avoid further scans
539      *
540      * @return list of attributes
541      */
542     protected Collection<String> getStandardAttributes() {
543         Class clz = getClass();
544         Collection<String> standardAttributes = standardAttributesMap.get(clz);
545         if (standardAttributes == null) {
546             Collection<Method> methods = AnnotationUtils.getAnnotatedMethods(clz, StrutsTagAttribute.class);
547             standardAttributes = new HashSet<>(methods.size());
548             for(Method m : methods) {
549                 standardAttributes.add(StringUtils.uncapitalize(m.getName().substring(3)));
550             }
551             standardAttributesMap.putIfAbsent(clz, standardAttributes);
552         }
553         return standardAttributes;
554     }
