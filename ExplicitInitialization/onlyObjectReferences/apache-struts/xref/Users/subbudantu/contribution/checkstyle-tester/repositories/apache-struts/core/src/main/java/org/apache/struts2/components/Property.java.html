22  package org.apache.struts2.components;
23  
24  import com.opensymphony.xwork2.util.ValueStack;
25  import org.apache.commons.lang3.StringEscapeUtils;
26  import org.apache.logging.log4j.LogManager;
27  import org.apache.logging.log4j.Logger;
28  import org.apache.struts2.views.annotations.StrutsTag;
29  import org.apache.struts2.views.annotations.StrutsTagAttribute;
30  
31  import java.io.IOException;
32  import java.io.Writer;
33  
34  /**
35   * <!-- START SNIPPET: javadoc -->
36   * <p>
37   * Used to get the property of a <i>value</i>, which will default to the top of
38   * the stack if none is specified.
39   * </p>
40   * <!-- END SNIPPET: javadoc -->
41   *
42   * <!-- START SNIPPET: params -->
43   *
44   * <ul>
45   *      <li>default (String) - The default value to be used if <u>value</u> attribute is null</li>
46   *      <li>escapeCsv (Boolean) - Escape CSV. Defaults to false</li>
47   *      <li>escapeHtml (Boolean) - Escape HTML. Defaults to true</li>
48   *      <li>escapeJavaScript (Boolean) - Escape JavaScript. Defaults to false</li>
49   *      <li>escapeXml (Boolean) - Escape XML. Defaults to false</li>
50   *      <li>value (Object) - value to be displayed</li>
51   * </ul>
52   *
53   * <!-- END SNIPPET: params -->
54   *
55   *
56   * <pre>
57   * <!-- START SNIPPET: example -->
58   *
59   * &lt;s:push value="myBean"&gt;
60   *     <!-- Example 1: -->
61   *     &lt;s:property value="myBeanProperty" /&gt;
62   *
63   *     <!-- Example 2: -->TextUtils
64   *     &lt;s:property value="myBeanProperty" default="a default value" /&gt;
65   * &lt;/s:push&gt;
66   *
67   * <!-- END SNIPPET: example -->
68   * </pre>
69   *
70   * <pre>
71   * <!-- START SNIPPET: exampledescription -->
72   * Example 1 prints the result of myBean's getMyBeanProperty() method.
73   * Example 2 prints the result of myBean's getMyBeanProperty() method and if it is null, print 'a default value' instead.
74   * <!-- END SNIPPET: exampledescription -->
75   * </pre>
76   *
77   *
78   * <pre>
79   * <!-- START SNIPPET: i18nExample -->
80   *
81   * &lt;s:property value="getText('some.key')" /&gt;
82   *
83   * <!-- END SNIPPET: i18nExample -->
84   * </pre>
85   *
86   */
87  @StrutsTag(name="property", tldBodyContent="empty", tldTagClass="org.apache.struts2.views.jsp.PropertyTag",
88      description="Print out expression which evaluates against the stack")
89  public class Property extends Component {
90      private static final Logger LOG = LogManager.getLogger(Property.class);
91  
92      public Property(ValueStack stack) {
93          super(stack);
94      }
95  
96      private String defaultValue;
97      private String value;
98      private boolean escapeHtml = true;
99      private boolean escapeJavaScript = false;
100     private boolean escapeXml = false;
101     private boolean escapeCsv = false;
102 
103     @StrutsTagAttribute(description="The default value to be used if <u>value</u> attribute is null")
104     public void setDefault(String defaultValue) {
105         this.defaultValue = defaultValue;
106     }
107 
108     @StrutsTagAttribute(description="Whether to escape HTML", type="Boolean", defaultValue="true")
109     public void setEscapeHtml(boolean escape) {
110         this.escapeHtml = escape;
111     }
112 
113     @StrutsTagAttribute(description="Whether to escape Javascript", type="Boolean", defaultValue="false")
114     public void setEscapeJavaScript(boolean escapeJavaScript) {
115         this.escapeJavaScript = escapeJavaScript;
116     }
117 
118     @StrutsTagAttribute(description="Value to be displayed", type="Object", defaultValue="&lt;top of stack&gt;")
119     public void setValue(String value) {
120         this.value = value;
121     }
122 
123     public void setDefaultValue(String defaultValue) {
124         this.defaultValue = defaultValue;
125     }
126 
127     @StrutsTagAttribute(description="Whether to escape CSV (useful to escape a value for a column)", type="Boolean", defaultValue="false")
128     public void setEscapeCsv(boolean escapeCsv) {
129         this.escapeCsv = escapeCsv;
130     }
131 
132     @StrutsTagAttribute(description="Whether to escape XML", type="Boolean", defaultValue="false")
133     public void setEscapeXml(boolean escapeXml) {
134         this.escapeXml = escapeXml;
135     }
136 
137     public boolean start(Writer writer) {
138         boolean result = super.start(writer);
139 
140         String actualValue = null;
141 
142         if (value == null) {
143             value = "top";
144         }
145         else {
146         	value = stripExpressionIfAltSyntax(value);
147         }
148 
149         // exception: don't call findString(), since we don't want the
150         //            expression parsed in this one case. it really
151         //            doesn't make sense, in fact.
152         actualValue = (String) getStack().findValue(value, String.class, throwExceptionOnELFailure);
153 
154         try {
155             if (actualValue != null) {
156                 writer.write(prepare(actualValue));
157             } else if (defaultValue != null) {
158                 writer.write(prepare(defaultValue));
159             }
160         } catch (IOException e) {
161             LOG.info("Could not print out value '{}'", value, e);
162         }
163 
164         return result;
165     }
166 
167     private String prepare(String value) {
168     	String result = value;
169         if (escapeHtml) {
170         	result = StringEscapeUtils.escapeHtml4(result);
171         }
172         if (escapeJavaScript) {
173         	result = StringEscapeUtils.escapeEcmaScript(result);
174         }
175         if (escapeXml) {
176         	result = StringEscapeUtils.escapeXml(result);
177         }
178         if (escapeCsv) {
179             result = StringEscapeUtils.escapeCsv(result);
180         }
181 
182         return result;
183     }
184 }
