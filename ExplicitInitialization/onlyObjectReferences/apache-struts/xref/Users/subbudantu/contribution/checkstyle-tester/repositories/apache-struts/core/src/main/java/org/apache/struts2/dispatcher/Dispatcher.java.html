286     /**
287      * Releases all instances bound to this dispatcher instance.
288      */
289     public void cleanup() {
290 
291     	// clean up ObjectFactory
292         ObjectFactory objectFactory = getContainer().getInstance(ObjectFactory.class);
293         if (objectFactory == null) {
294         	LOG.warn("Object Factory is null, something is seriously wrong, no clean up will be performed");
295         }
296         if (objectFactory instanceof ObjectFactoryDestroyable) {
297             try {
298                 ((ObjectFactoryDestroyable)objectFactory).destroy();
299             }
300             catch(Exception e) {
301                 // catch any exception that may occurred during destroy() and log it
302                 LOG.error("Exception occurred while destroying ObjectFactory [{}]", objectFactory.toString(), e);
303             }
304         }
305 
306         // clean up Dispatcher itself for this thread
307         instance.set(null);
308 
309         // clean up DispatcherListeners
310         if (!dispatcherListeners.isEmpty()) {
311             for (DispatcherListener l : dispatcherListeners) {
312                 l.dispatcherDestroyed(this);
313             }
314         }
315 
316         // clean up all interceptors by calling their destroy() method
317         Set<Interceptor> interceptors = new HashSet<>();
318         Collection<PackageConfig> packageConfigs = configurationManager.getConfiguration().getPackageConfigs().values();
319         for (PackageConfig packageConfig : packageConfigs) {
320             for (Object config : packageConfig.getAllInterceptorConfigs().values()) {
321                 if (config instanceof InterceptorStackConfig) {
322                     for (InterceptorMapping interceptorMapping : ((InterceptorStackConfig) config).getInterceptors()) {
323                 	    interceptors.add(interceptorMapping.getInterceptor());
324                     }
325                 }
326             }
327         }
328         for (Interceptor interceptor : interceptors) {
329         	interceptor.destroy();
330         }
331 
332         // Clear container holder when application is unloaded / server shutdown
333         ContainerHolder.clear();
334 
335         //cleanup action context
336         ActionContext.setContext(null);
337 
338         // clean up configuration
339     	configurationManager.destroyConfiguration();
340     	configurationManager = null;
341     }

343     private void init_FileManager() throws ClassNotFoundException {
344         if (initParams.containsKey(StrutsConstants.STRUTS_FILE_MANAGER)) {
345             final String fileManagerClassName = initParams.get(StrutsConstants.STRUTS_FILE_MANAGER);
346             final Class<FileManager> fileManagerClass = (Class<FileManager>) Class.forName(fileManagerClassName);
347             LOG.info("Custom FileManager specified: {}", fileManagerClassName);
348             configurationManager.addContainerProvider(new FileManagerProvider(fileManagerClass, fileManagerClass.getSimpleName()));
349         } else {
350             // add any other Struts 2 provided implementations of FileManager
351             configurationManager.addContainerProvider(new FileManagerProvider(JBossFileManager.class, "jboss"));
352         }
353         if (initParams.containsKey(StrutsConstants.STRUTS_FILE_MANAGER_FACTORY)) {
354             final String fileManagerFactoryClassName = initParams.get(StrutsConstants.STRUTS_FILE_MANAGER_FACTORY);
355             final Class<FileManagerFactory> fileManagerFactoryClass = (Class<FileManagerFactory>) Class.forName(fileManagerFactoryClassName);
356             LOG.info("Custom FileManagerFactory specified: {}", fileManagerFactoryClassName);
357             configurationManager.addContainerProvider(new FileManagerFactoryProvider(fileManagerFactoryClass));
358         }
359     }

369     private void init_TraditionalXmlConfigurations() {
370         String configPaths = initParams.get("config");
371         if (configPaths == null) {
372             configPaths = DEFAULT_CONFIGURATION_PATHS;
373         }
374         String[] files = configPaths.split("\\s*[,]\\s*");
375         for (String file : files) {
376             if (file.endsWith(".xml")) {
377                 configurationManager.addContainerProvider(createStrutsXmlConfigurationProvider(file, false, servletContext));
378             } else {
379                 throw new IllegalArgumentException("Invalid configuration file name");
380             }
381         }
382     }

388     private void init_CustomConfigurationProviders() {
389         String configProvs = initParams.get("configProviders");
390         if (configProvs != null) {
391             String[] classes = configProvs.split("\\s*[,]\\s*");
392             for (String cname : classes) {
393                 try {
394                     Class cls = ClassLoaderUtil.loadClass(cname, this.getClass());
395                     ConfigurationProvider prov = (ConfigurationProvider)cls.newInstance();
396                     if (prov instanceof ServletContextAwareConfigurationProvider) {
397                         ((ServletContextAwareConfigurationProvider)prov).initWithContext(servletContext);
398                     }
399                     configurationManager.addContainerProvider(prov);
400                 } catch (InstantiationException e) {
401                     throw new ConfigurationException("Unable to instantiate provider: "+cname, e);
402                 } catch (IllegalAccessException e) {
403                     throw new ConfigurationException("Unable to access provider: "+cname, e);
404                 } catch (ClassNotFoundException e) {
405                     throw new ConfigurationException("Unable to locate provider class: "+cname, e);
406                 }
407             }
408         }
409     }

451     /**
452      * Load configurations, including both XML and zero-configuration strategies,
453      * and update optional settings, including whether to reload configurations and resource files.
454      */
455     public void init() {
456 
457     	if (configurationManager == null) {
458     		configurationManager = createConfigurationManager(Container.DEFAULT_NAME);
459     	}
460 
461         try {
462             init_FileManager();
463             init_DefaultProperties(); // [1]
464             init_TraditionalXmlConfigurations(); // [2]
465             init_LegacyStrutsProperties(); // [3]
466             init_CustomConfigurationProviders(); // [5]
467             init_FilterInitParameters() ; // [6]
468             init_AliasStandardObjects() ; // [7]
469 
470             Container container = init_PreloadConfiguration();
471             container.inject(this);
472             init_CheckWebLogicWorkaround(container);
473 
474             if (!dispatcherListeners.isEmpty()) {
475                 for (DispatcherListener l : dispatcherListeners) {
476                     l.dispatcherInitialized(this);
477                 }
478             }
479             errorHandler.init(servletContext);
480 
481         } catch (Exception ex) {
482             LOG.error("Dispatcher initialization failed", ex);
483             throw new StrutsException(ex);
484         }
485     }

491     /**
492      * <p>
493      * Load Action class for mapping and invoke the appropriate Action method, or go directly to the Result.
494      * </p>
495      *
496      * <p>
497      * This method first creates the action context from the given parameters,
498      * and then loads an <tt>ActionProxy</tt> from the given action name and namespace.
499      * After that, the Action method is executed and output channels through the response object.
500      * Actions not found are sent back to the user via the {@link Dispatcher#sendError} method,
501      * using the 404 return code.
502      * All other errors are reported by throwing a ServletException.
503      * </p>
504      *
505      * @param request  the HttpServletRequest object
506      * @param response the HttpServletResponse object
507      * @param mapping  the action mapping object
508      * @throws ServletException when an unknown error occurs (not a 404, but typically something that
509      *                          would end up as a 5xx by the servlet container)
510      *
511      * @since 2.3.17
512      */
513     public void serviceAction(HttpServletRequest request, HttpServletResponse response, ActionMapping mapping)
514             throws ServletException {
515 
516         Map<String, Object> extraContext = createContextMap(request, response, mapping);
517 
518         // If there was a previous value stack, then create a new copy and pass it in to be used by the new Action
519         ValueStack stack = (ValueStack) request.getAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY);
520         boolean nullStack = stack == null;
521         if (nullStack) {
522             ActionContext ctx = ActionContext.getContext();
523             if (ctx != null) {
524                 stack = ctx.getValueStack();
525             }
526         }
527         if (stack != null) {
528             extraContext.put(ActionContext.VALUE_STACK, valueStackFactory.createValueStack(stack));
529         }
530 
531         String timerKey = "Handling request from Dispatcher";
532         try {
533             UtilTimerStack.push(timerKey);
534             String namespace = mapping.getNamespace();
535             String name = mapping.getName();
536             String method = mapping.getMethod();
537 
538             ActionProxy proxy = getContainer().getInstance(ActionProxyFactory.class).createActionProxy(
539                     namespace, name, method, extraContext, true, false);
540 
541             request.setAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY, proxy.getInvocation().getStack());
542 
543             // if the ActionMapping says to go straight to a result, do it!
544             if (mapping.getResult() != null) {
545                 Result result = mapping.getResult();
546                 result.execute(proxy.getInvocation());
547             } else {
548                 proxy.execute();
549             }
550 
551             // If there was a previous value stack then set it back onto the request
552             if (!nullStack) {
553                 request.setAttribute(ServletActionContext.STRUTS_VALUESTACK_KEY, stack);
554             }
555         } catch (ConfigurationException e) {
556             logConfigurationException(request, e);
557             sendError(request, response, HttpServletResponse.SC_NOT_FOUND, e);
558         } catch (Exception e) {
559             e.printStackTrace();
560             if (handleException || devMode) {
561                 sendError(request, response, HttpServletResponse.SC_INTERNAL_SERVER_ERROR, e);
562             } else {
563                 throw new ServletException(e);
564             }
565         } finally {
566             UtilTimerStack.pop(timerKey);
567         }
568     }

570     /**
571      * Performs logging of missing action/result configuration exception
572      *
573      * @param request current {@link HttpServletRequest}
574      * @param e {@link ConfigurationException} that occurred
575      */
576     protected void logConfigurationException(HttpServletRequest request, ConfigurationException e) {
577         // WW-2874 Only log error if in devMode
578         String uri = request.getRequestURI();
579         if (request.getQueryString() != null) {
580             uri = uri + "?" + request.getQueryString();
581         }
582         if (devMode) {
583             LOG.error("Could not find action or result: {}", uri, e);
584         } else if (LOG.isWarnEnabled()) {
585             LOG.warn("Could not find action or result: {}", uri, e);
586         }
587     }

589     /**
590      * Create a context map containing all the wrapped request objects
591      *
592      * @param request The servlet request
593      * @param response The servlet response
594      * @param mapping The action mapping
595      * @return A map of context objects
596      *
597      * @since 2.3.17
598      */
599     public Map<String,Object> createContextMap(HttpServletRequest request, HttpServletResponse response,
600             ActionMapping mapping) {
601 
602         // request map wrapping the http request objects
603         Map requestMap = new RequestMap(request);
604 
605         // parameters map wrapping the http parameters.  ActionMapping parameters are now handled and applied separately
606         HttpParameters params = HttpParameters.create(request.getParameterMap()).build();
607 
608         // session map wrapping the http session
609         Map session = new SessionMap(request);
610 
611         // application map wrapping the ServletContext
612         Map application = new ApplicationMap(servletContext);
613 
614         Map<String,Object> extraContext = createContextMap(requestMap, params, session, application, request, response);
615 
616         if (mapping != null) {
617             extraContext.put(ServletActionContext.ACTION_MAPPING, mapping);
618         }
619         return extraContext;
620     }

622     /**
623      * Merge all application and servlet attributes into a single <tt>HashMap</tt> to represent the entire
624      * <tt>Action</tt> context.
625      *
626      * @param requestMap     a Map of all request attributes.
627      * @param parameters     an Object of all request parameters.
628      * @param sessionMap     a Map of all session attributes.
629      * @param applicationMap a Map of all servlet context attributes.
630      * @param request        the HttpServletRequest object.
631      * @param response       the HttpServletResponse object.
632      * @return a HashMap representing the <tt>Action</tt> context.
633      *
634      * @since 2.3.17
635      */
636     public HashMap<String,Object> createContextMap(Map requestMap,
637                                     HttpParameters parameters,
638                                     Map sessionMap,
639                                     Map applicationMap,
640                                     HttpServletRequest request,
641                                     HttpServletResponse response) {
642         HashMap<String, Object> extraContext = new HashMap<>();
643         extraContext.put(ActionContext.PARAMETERS, parameters);
644         extraContext.put(ActionContext.SESSION, sessionMap);
645         extraContext.put(ActionContext.APPLICATION, applicationMap);
646 
647         Locale locale;
648         if (defaultLocale != null) {
649             locale = DefaultLocalizedTextProvider.localeFromString(defaultLocale, request.getLocale());
650         } else {
651             locale = request.getLocale();
652         }
653 
654         extraContext.put(ActionContext.LOCALE, locale);
655 
656         extraContext.put(StrutsStatics.HTTP_REQUEST, request);
657         extraContext.put(StrutsStatics.HTTP_RESPONSE, response);
658         extraContext.put(StrutsStatics.SERVLET_CONTEXT, servletContext);
659 
660         // helpers to get access to request/session/application scope
661         extraContext.put("request", requestMap);
662         extraContext.put("session", sessionMap);
663         extraContext.put("application", applicationMap);
664         extraContext.put("parameters", parameters);
665 
666         AttributeMap attrMap = new AttributeMap(extraContext);
667         extraContext.put("attr", attrMap);
668 
669         return extraContext;
670     }

672     /**
673      * Return the path to save uploaded files to (this is configurable).
674      *
675      * @return the path to save uploaded files to
676      */
677     protected String getSaveDir() {
678         String saveDir = multipartSaveDir.trim();
679 
680         if (saveDir.equals("")) {
681             File tempdir = (File) servletContext.getAttribute("javax.servlet.context.tempdir");
682         	LOG.info("Unable to find 'struts.multipart.saveDir' property setting. Defaulting to javax.servlet.context.tempdir");
683 
684             if (tempdir != null) {
685                 saveDir = tempdir.toString();
686                 setMultipartSaveDir(saveDir);
687             }
688         } else {
689             File multipartSaveDir = new File(saveDir);
690 
691             if (!multipartSaveDir.exists()) {
692                 if (!multipartSaveDir.mkdirs()) {
693                     String logMessage;
694                     try {
695                         logMessage = "Could not find create multipart save directory '" + multipartSaveDir.getCanonicalPath() + "'.";
696                     } catch (IOException e) {
697                         logMessage = "Could not find create multipart save directory '" + multipartSaveDir.toString() + "'.";
698                     }
699                     if (devMode) {
700                         LOG.error(logMessage);
701                     } else {
702                         LOG.warn(logMessage);
703                     }
704                 }
705             }
706         }
707 
708         LOG.debug("saveDir={}", saveDir);
709 
710         return saveDir;
711     }

713     /**
714      * Prepare a request, including setting the encoding and locale.
715      *
716      * @param request The request
717      * @param response The response
718      */
719     public void prepare(HttpServletRequest request, HttpServletResponse response) {
720         String encoding = null;
721         if (defaultEncoding != null) {
722             encoding = defaultEncoding;
723         }
724         // check for Ajax request to use UTF-8 encoding strictly http://www.w3.org/TR/XMLHttpRequest/#the-send-method
725         if ("XMLHttpRequest".equals(request.getHeader("X-Requested-With"))) {
726             encoding = "UTF-8";
727         }
728 
729         Locale locale = null;
730         if (defaultLocale != null) {
731             locale = DefaultLocalizedTextProvider.localeFromString(defaultLocale, request.getLocale());
732         }
733 
734         if (encoding != null) {
735             applyEncoding(request, encoding);
736         }
737 
738         if (locale != null) {
739             response.setLocale(locale);
740         }
741 
742         if (paramsWorkaroundEnabled) {
743             request.getParameter("foo"); // simply read any parameter (existing or not) to "prime" the request
744         }
745     }

747     private void applyEncoding(HttpServletRequest request, String encoding) {
748         try {
749             if (!encoding.equals(request.getCharacterEncoding())) {
750                 // if the encoding is already correctly set and the parameters have been already read
751                 // do not try to set encoding because it is useless and will cause an error
752                 request.setCharacterEncoding(encoding);
753             }
754         } catch (Exception e) {
755             LOG.error("Error setting character encoding to '{}' - ignoring.", encoding, e);
756         }
757     }

759     /**
760      * <p>
761      * Wrap and return the given request or return the original request object.
762      * </p>
763      *
764      * <p>
765      * This method transparently handles multipart data as a wrapped class around the given request.
766      * Override this method to handle multipart requests in a special way or to handle other types of requests.
767      * Note, {@link org.apache.struts2.dispatcher.multipart.MultiPartRequestWrapper} is
768      * flexible - look first to that object before overriding this method to handle multipart data.
769      * </p>
770      *
771      * @param request the HttpServletRequest object.
772      * @return a wrapped request or original request.
773      * @see org.apache.struts2.dispatcher.multipart.MultiPartRequestWrapper
774      * @throws java.io.IOException on any error.
775      *
776      * @since 2.3.17
777      */
778     public HttpServletRequest wrapRequest(HttpServletRequest request) throws IOException {
779         // don't wrap more than once
780         if (request instanceof StrutsRequestWrapper) {
781             return request;
782         }
783 
784         String content_type = request.getContentType();
785         if (content_type != null && content_type.contains("multipart/form-data")) {
786             MultiPartRequest multiPartRequest = getMultiPartRequest();
787             LocaleProviderFactory localeProviderFactory = getContainer().getInstance(LocaleProviderFactory.class);
788 
789             request = new MultiPartRequestWrapper(
790                     multiPartRequest,
791                     request,
792                     getSaveDir(),
793                     localeProviderFactory.createLocaleProvider(),
794                     disableRequestAttributeValueStackLookup
795             );
796         } else {
797             request = new StrutsRequestWrapper(request, disableRequestAttributeValueStackLookup);
798         }
799 
800         return request;
801     }

803     /**
804      * On each request it must return a new instance as implementation could be not thread safe
805      * and thus ensure of resource clean up
806      *
807      * @return a multi part request object
808      */
809     protected MultiPartRequest getMultiPartRequest() {
810         MultiPartRequest mpr = null;
811         //check for alternate implementations of MultiPartRequest
812         Set<String> multiNames = getContainer().getInstanceNames(MultiPartRequest.class);
813         for (String multiName : multiNames) {
814             if (multiName.equals(multipartHandlerName)) {
815                 mpr = getContainer().getInstance(MultiPartRequest.class, multiName);
816             }
817         }
818         if (mpr == null ) {
819             mpr = getContainer().getInstance(MultiPartRequest.class);
820         }
821         return mpr;
822     }

885     /**
886      * Expose the dependency injection container.
887      * @return Our dependency injection container
888      */
889     public Container getContainer() {
890         if (ContainerHolder.get() != null) {
891             return ContainerHolder.get();
892         }
893         ConfigurationManager mgr = getConfigurationManager();
894         if (mgr == null) {
895             throw new IllegalStateException("The configuration manager shouldn't be null");
896         } else {
897             Configuration config = mgr.getConfiguration();
898             if (config == null) {
899                 throw new IllegalStateException("Unable to load configuration");
900             } else {
901                 Container container = config.getContainer();
902                 ContainerHolder.store(container);
903                 return container;
904             }
905         }
