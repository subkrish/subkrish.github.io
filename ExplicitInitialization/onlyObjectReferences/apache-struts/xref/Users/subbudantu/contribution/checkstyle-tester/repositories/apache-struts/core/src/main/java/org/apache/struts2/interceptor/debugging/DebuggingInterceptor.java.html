51  /**
52   * <!-- START SNIPPET: description -->
53   * <p>
54   * Provides several different debugging screens to provide insight into the
55   * data behind the page.
56   * </p>
57   * <!-- END SNIPPET: description -->
58   * <p>
59   * The value of the 'debug' request parameter determines
60   * the screen:
61   * </p>
62   * <!-- START SNIPPET: parameters -->
63   * <ul>
64   * <li> <code>xml</code> - Dumps the parameters, context, session, and value
65   * stack as an XML document.</li>
66   * <li> <code>console</code> - Shows a popup 'OGNL Console' that allows the
67   * user to test OGNL expressions against the value stack. The XML data from
68   * the 'xml' mode is inserted at the top of the page.</li>
69   * <li> <code>command</code> - Tests an OGNL expression and returns the
70   * string result. Only used by the OGNL console.</li>
71   * <li><code>browser</code> Shows field values of an object specified in the 
72   * <code>object</code> parameter (#context by default). When the <code>object</code>
73   * parameters is set, the '#' character needs to be escaped to '%23'. Like
74   * debug=browser&amp;object=%23parameters</li>
75   * </ul>
76   * <!-- END SNIPPET: parameters -->
77   * <p>
78   * Example:
79   * </p>
80   * <!-- START SNIPPET: example -->
81   *  http://localhost:8080/Welcome.action?debug=xml
82   * <!-- END SNIPPET: example -->
83   * <p>
84   * <!-- START SNIPPET: remarks -->
85   * This interceptor only is activated when devMode is enabled in
86   * struts.properties. The 'debug' parameter is removed from the parameter list
87   * before the action is executed. All operations occur before the natural
88   * Result has a chance to execute.
89   * <!-- END SNIPPET: remarks -->
90   * </p>
91   */
92  public class DebuggingInterceptor extends AbstractInterceptor {

139     /*
140      * (non-Javadoc)
141      *
142      * @see com.opensymphony.xwork2.interceptor.Interceptor#invoke(com.opensymphony.xwork2.ActionInvocation)
143      */
144     public String intercept(ActionInvocation inv) throws Exception {
145         boolean actionOnly = false;
146         boolean cont = true;
147         Boolean devModeOverride = PrepareOperations.getDevModeOverride();
148         boolean devMode = devModeOverride != null ? devModeOverride : this.devMode;
149         if (devMode) {
150             final ActionContext ctx = ActionContext.getContext();
151             String type = getParameter(DEBUG_PARAM);
152             ctx.getParameters().remove(DEBUG_PARAM);
153             if (XML_MODE.equals(type)) {
154                 inv.addPreResultListener(
155                         new PreResultListener() {
156                             public void beforeResult(ActionInvocation inv, String result) {
157                                 printContext();
158                             }
159                         });
160             } else if (CONSOLE_MODE.equals(type)) {
161                 consoleEnabled = true;
162                 inv.addPreResultListener(
163                         new PreResultListener() {
164                             public void beforeResult(ActionInvocation inv, String actionResult) {
165                                 String xml = "";
166                                 if (enableXmlWithConsole) {
167                                     StringWriter writer = new StringWriter();
168                                     printContext(new PrettyPrintWriter(writer));
169                                     xml = writer.toString();
170                                     xml = xml.replaceAll("&", "&amp;");
171                                     xml = xml.replaceAll(">", "&gt;");
172                                     xml = xml.replaceAll("<", "&lt;");
173                                 }
174                                 ActionContext.getContext().put("debugXML", xml);
175 
176                                 FreemarkerResult result = new FreemarkerResult();
177                                 result.setFreemarkerManager(freemarkerManager);
178                                 result.setContentType("text/html");
179                                 result.setLocation("/org/apache/struts2/interceptor/debugging/console.ftl");
180                                 result.setParse(false);
181                                 try {
182                                     result.execute(inv);
183                                 } catch (Exception ex) {
184                                     LOG.error("Unable to create debugging console", ex);
185                                 }
186 
187                             }
188                         });
189             } else if (COMMAND_MODE.equals(type)) {
190                 ValueStack stack = (ValueStack) ctx.getSession().get(SESSION_KEY);
191                 if (stack == null) {
192                     //allows it to be embedded on another page
193                     stack = (ValueStack) ctx.get(ActionContext.VALUE_STACK);
194                     ctx.getSession().put(SESSION_KEY, stack);
195                 }
196                 String cmd = getParameter(EXPRESSION_PARAM);
197 
198                 ServletActionContext.getRequest().setAttribute("decorator", "none");
199                 HttpServletResponse res = ServletActionContext.getResponse();
200                 res.setContentType("text/plain");
201 
202                 try (PrintWriter writer =
203                             ServletActionContext.getResponse().getWriter()) {
204                     writer.print(stack.findValue(cmd));
205                 } catch (IOException ex) {
206                     ex.printStackTrace();
207                 }
208                 cont = false;
209             } else if (BROWSER_MODE.equals(type)) {
210                 actionOnly = true;
211                 inv.addPreResultListener(
212                     new PreResultListener() {
213                         public void beforeResult(ActionInvocation inv, String actionResult) {
214                             String rootObjectExpression = getParameter(OBJECT_PARAM);
215                             if (rootObjectExpression == null)
216                                 rootObjectExpression = "#context";
217                             String decorate = getParameter(DECORATE_PARAM);
218                             ValueStack stack = (ValueStack) ctx.get(ActionContext.VALUE_STACK);
219                             Object rootObject = stack.findValue(rootObjectExpression);
220                             
221                             try (StringWriter writer = new StringWriter()) {
222                                 ObjectToHTMLWriter htmlWriter = new ObjectToHTMLWriter(writer);
223                                 htmlWriter.write(reflectionProvider, rootObject, rootObjectExpression);
224                                 String html = writer.toString();
225                                 writer.close();
226                                 
227                                 stack.set("debugHtml", html);
228                                 
229                                 //on the first request, response can be decorated
230                                 //but we need plain text on the other ones
231                                 if ("false".equals(decorate))
232                                     ServletActionContext.getRequest().setAttribute("decorator", "none");
233                                 
234                                 FreemarkerResult result = new FreemarkerResult();
235                                 result.setFreemarkerManager(freemarkerManager);
236                                 result.setContentType("text/html");
237                                 result.setLocation("/org/apache/struts2/interceptor/debugging/browser.ftl");
238                                 result.execute(inv);
239                             } catch (Exception ex) {
240                                 LOG.error("Unable to create debugging console", ex);
241                             }
242 
243                         }
244                     });
245             }
246         } 
247         if (cont) {
248             try {
249                 if (actionOnly) {
250                     inv.invokeActionOnly();
251                     return null;
252                 } else {
253                     return inv.invoke();
254                 }
255             } finally {
256                 if (devMode && consoleEnabled) {
257                     final ActionContext ctx = ActionContext.getContext();
258                     ctx.getSession().put(SESSION_KEY, ctx.get(ActionContext.VALUE_STACK));
259                 }
260             }
261         } else {
262             return null;
263         }
264     }

277     /**
278      * Prints the current context to the response in XML format.
279      */
280     protected void printContext() {
281         HttpServletResponse res = ServletActionContext.getResponse();
282         res.setContentType("text/xml");
283 
284         try {
285             PrettyPrintWriter writer = new PrettyPrintWriter(
286                     ServletActionContext.getResponse().getWriter());
287             printContext(writer);
288             writer.close();
289         } catch (IOException ex) {
290             ex.printStackTrace();
291         }
292     }
293 
294     /**
295      * Prints the current request to the existing writer.
296      *
297      * @param writer The XML writer
298      */
299     protected void printContext(PrettyPrintWriter writer) {
300         ActionContext ctx = ActionContext.getContext();
301         writer.startNode(DEBUG_PARAM);
302         serializeIt(ctx.getParameters(), "parameters", writer, new ArrayList<>());
303         writer.startNode("context");
304         String key;
305         Map ctxMap = ctx.getContextMap();
306         for (Object o : ctxMap.keySet()) {
307             key = o.toString();
308             boolean print = !ignoreKeys.contains(key);
309 
310             for (String ignorePrefixe : ignorePrefixes) {
311                 if (key.startsWith(ignorePrefixe)) {
312                     print = false;
313                     break;
314                 }
315             }
316             if (print) {
317                 serializeIt(ctxMap.get(key), key, writer, new ArrayList<>());
318             }
319         }
320         writer.endNode();
321         Map requestMap = (Map) ctx.get("request");
322         serializeIt(requestMap, "request", writer, filterValueStack(requestMap));
323         serializeIt(ctx.getSession(), "session", writer, new ArrayList<>());
324 
325         ValueStack stack = (ValueStack) ctx.get(ActionContext.VALUE_STACK);
326         serializeIt(stack.getRoot(), "valueStack", writer, new ArrayList<>());
327         writer.endNode();
328     }

330     /**
331      * Recursive function to serialize objects to XML. Currently it will
332      * serialize Collections, maps, Arrays, and JavaBeans. It maintains a stack
333      * of objects serialized already in the current function call. This is used
334      * to avoid looping (stack overflow) of circular linked objects. Struts and
335      * XWork objects are ignored.
336      *
337      * @param bean   The object you want serialized.
338      * @param name   The name of the object, used for element &lt;name/&gt;
339      * @param writer The XML writer
340      * @param stack  List of objects we're serializing since the first calling
341      *               of this function (to prevent looping on circular references).
342      */
343     protected void serializeIt(Object bean, String name,
344                                PrettyPrintWriter writer, List<Object> stack) {
345         writer.flush();
346         // Check stack for this object
347         if ((bean != null) && (stack.contains(bean))) {
348             LOG.info("Circular reference detected, not serializing object: {}", name);
349             return;
350         } else if (bean != null) {
351             // Push object onto stack.
352             // Don't push null objects ( handled below)
353             stack.add(bean);
354         }
355         if (bean == null) {
356             return;
357         }
358         String clsName = bean.getClass().getName();
359 
360         writer.startNode(name);
361 
362         // It depends on the object and it's value what todo next:
363         if (bean instanceof Collection) {
364             Collection col = (Collection) bean;
365 
366             // Iterate through components, and call ourselves to process
367             // elements
368             for (Object aCol : col) {
369                 serializeIt(aCol, "value", writer, stack);
370             }
371         } else if (bean instanceof Map) {
372 
373             Map<Object, Object> map = (Map) bean;
374 
375             // Loop through keys and call ourselves
376             for (Map.Entry<Object, Object> entry : map.entrySet()) {
377                 Object objValue = entry.getValue();
378                 serializeIt(objValue, entry.getKey().toString(), writer, stack);
379             }
380         } else if (bean.getClass().isArray()) {
381             // It's an array, loop through it and keep calling ourselves
382             for (int i = 0; i < Array.getLength(bean); i++) {
383                 serializeIt(Array.get(bean, i), "arrayitem", writer, stack);
384             }
385         } else {
386             if (clsName.startsWith("java.lang")) {
387                 writer.setValue(bean.toString());
388             } else {
389                 // Not java.lang, so we can call ourselves with this object's
390                 // values
391                 try {
392                     BeanInfo info = Introspector.getBeanInfo(bean.getClass());
393                     PropertyDescriptor[] props = info.getPropertyDescriptors();
394 
395                     for (PropertyDescriptor prop : props) {
396                         String n = prop.getName();
397                         Method m = prop.getReadMethod();
398 
399                         // Call ourselves with the result of the method
400                         // invocation
401                         if (m != null) {
402                             serializeIt(m.invoke(bean), n, writer, stack);
403                         }
404                     }
405                 } catch (Exception e) {
406                     LOG.error(e.toString(), e);
407                 }
408             }
409         }
410 
411         writer.endNode();
412 
413         // Remove object from stack
414         stack.remove(bean);
415     }

424     private List<Object> filterValueStack(Map requestMap) {
425         List<Object> filter = new ArrayList<>();
426         Object valueStack = requestMap.get("struts.valueStack");
427     	if(valueStack != null) {
428     		filter.add(valueStack);
429     	}
430     	return filter;
431     }
