48  /**
49   * <!-- START SNIPPET: description -->
50   * <p>
51   * Calls the {@link HttpServletResponse#sendRedirect(String) sendRedirect}
52   * method to the location specified. The response is told to redirect the
53   * browser to the specified location (a new request from the client). The
54   * consequence of doing this means that the action (action instance, action
55   * errors, field errors, etc) that was just executed is lost and no longer
56   * available. This is because actions are built on a single-thread model. The
57   * only way to pass data is through the session or with web parameters
58   * (url?name=value) which can be OGNL expressions.
59   * </p>
60   * <!-- END SNIPPET: description -->
61   * <p>
62   * <b>This result type takes the following parameters:</b>
63   * </p>
64   * <!-- START SNIPPET: params -->
65   * 
66   * <ul>
67   * 
68   * <li><b>location (default)</b> - the location to go to after execution.</li>
69   * 
70   * <li><b>parse</b> - true by default. If set to false, the location param will
71   * not be parsed for Ognl expressions.</li>
72   * 
73   * <li><b>anchor</b> - Optional.  Also known as "fragment" or colloquially as 
74   * "hash".  You can specify an anchor for a result.</li>
75   * </ul>
76   * 
77   * <p>
78   * This result follows the same rules from {@link StrutsResultSupport}.
79   * </p>
80   * 
81   * <!-- END SNIPPET: params -->
82   * <p>
83   * <b>Example:</b>
84   * </p>
85   *
86   * <pre>
87   * <!-- START SNIPPET: example -->
88   * &lt;!--
89   *   The redirect URL generated will be:
90   *   /foo.jsp#FRAGMENT
91   * --&gt;
92   * &lt;result name="success" type="redirect"&gt;
93   *   &lt;param name="location"&gt;foo.jsp&lt;/param&gt;
94   *   &lt;param name="parse"&gt;false&lt;/param&gt;
95   *   &lt;param name="anchor"&gt;FRAGMENT&lt;/param&gt;
96   * &lt;/result&gt;
97   * <!-- END SNIPPET: example -->
98   * </pre>
99   * 
100  */
101 public class ServletRedirectResult extends StrutsResultSupport implements ReflectionExceptionHandler, Redirectable {
102 
103     private static final long serialVersionUID = 6316947346435301270L;
104 
105     private static final Logger LOG = LogManager.getLogger(ServletRedirectResult.class);

107     protected boolean prependServletContext = true;
108     protected ActionMapper actionMapper;
109     protected int statusCode = SC_FOUND;
110     protected boolean suppressEmptyParameters = false;

112     protected String anchor;

152     /**
153      * Sets whether or not to prepend the servlet context path to the redirected
154      * URL.
155      * 
156      * @param prependServletContext <tt>true</tt> to prepend the location with the servlet context path, <tt>false</tt> otherwise.
157      */
158     public void setPrependServletContext(boolean prependServletContext) {
159         this.prependServletContext = prependServletContext;
160     }

162     public void execute(ActionInvocation invocation) throws Exception {
163         if (anchor != null) {
164             anchor = conditionalParse(anchor, invocation);
165         }
166         super.execute(invocation);
167     }
168 
169     /**
170      * Redirects to the location specified by calling
171      * {@link HttpServletResponse#sendRedirect(String)}.
172      * 
173      * @param finalLocation the location to redirect to.
174      * @param invocation an encapsulation of the action execution state.
175      * @throws Exception if an error occurs when redirecting.
176      */
177     protected void doExecute(String finalLocation, ActionInvocation invocation) throws Exception {
178         ActionContext ctx = invocation.getInvocationContext();
179         HttpServletRequest request = (HttpServletRequest) ctx.get(ServletActionContext.HTTP_REQUEST);
180         HttpServletResponse response = (HttpServletResponse) ctx.get(ServletActionContext.HTTP_RESPONSE);
181 
182         if (isPathUrl(finalLocation)) {
183             if (!finalLocation.startsWith("/")) {
184                 ActionMapping mapping = actionMapper.getMapping(request, Dispatcher.getInstance().getConfigurationManager());
185                 String namespace = null;
186                 if (mapping != null) {
187                     namespace = mapping.getNamespace();
188                 }
189 
190                 if ((namespace != null) && (namespace.length() > 0) && (!"/".equals(namespace))) {
191                     finalLocation = namespace + "/" + finalLocation;
192                 } else {
193                     finalLocation = "/" + finalLocation;
194                 }
195             }
196 
197             // if the URL's are relative to the servlet context, append the servlet context path
198             if (prependServletContext && (request.getContextPath() != null) && (request.getContextPath().length() > 0)) {
199                 finalLocation = request.getContextPath() + finalLocation;
200             }
201         }
202         ResultConfig resultConfig = invocation.getProxy().getConfig().getResults().get(invocation.getResultCode());
203         if (resultConfig != null) {
204             Map<String, String> resultConfigParams = resultConfig.getParams();
205 
206             List<String> prohibitedResultParams = getProhibitedResultParams();
207             for (Map.Entry<String, String> e : resultConfigParams.entrySet()) {
208                 if (!prohibitedResultParams.contains(e.getKey())) {
209                     Collection<String> values = conditionalParseCollection(e.getValue(), invocation, suppressEmptyParameters);
210                     if (!suppressEmptyParameters || !values.isEmpty()) {
211                         requestParameters.put(e.getKey(), values);
212                     }
213                 }
214             }
215         }
216 
217         StringBuilder tmpLocation = new StringBuilder(finalLocation);
218         urlHelper.buildParametersString(requestParameters, tmpLocation, "&");
219 
220         // add the anchor
221         if (anchor != null) {
222             tmpLocation.append('#').append(anchor);
223         }
224 
225         finalLocation = response.encodeRedirectURL(tmpLocation.toString());
226 
227         LOG.debug("Redirecting to finalLocation: {}", finalLocation);
228 
229         sendRedirect(response, finalLocation);
230     }
231 
232     protected List<String> getProhibitedResultParams() {
233         return Arrays.asList(
234                 DEFAULT_PARAM,
235                 "namespace",
236                 "method",
237                 "encode",
238                 "parse",
239                 "location",
240                 "prependServletContext",
241                 "suppressEmptyParameters",
242                 "anchor",
243                 "statusCode"
244         );
245     }
246 
247     /**
248      * Sends the redirection. Can be overridden to customize how the redirect is
249      * handled (i.e. to use a different status code)
250      * 
251      * @param response The response
252      * @param finalLocation The location URI
253      * @throws IOException in case of IO errors
254      */
255     protected void sendRedirect(HttpServletResponse response, String finalLocation) throws IOException {
256         if (SC_FOUND == statusCode) {
257             response.sendRedirect(finalLocation);
258         } else {
259             response.setStatus(statusCode);
260             response.setHeader("Location", finalLocation);
261             response.getWriter().write(finalLocation);
262             response.getWriter().close();
263         }
264 
265     }
266 
267     /**
268      * Checks if url is simple path or either full url
269      *
270      * @param url string
271      * @return true if it's just a path not a full url
272      */
273     protected boolean isPathUrl(String url) {
274         try {
275             String rawUrl = url;
276             if (url.contains("?")) {
277                 rawUrl = url.substring(0, url.indexOf("?"));
278             }
279             URI uri = URI.create(rawUrl.replaceAll(" ", "%20"));
280             if (uri.isAbsolute()) {
281                 URL validUrl = uri.toURL();
282                 LOG.debug("[{}] is full url, not a path", url);
283                 return validUrl.getProtocol() == null;
284             } else {
285                 LOG.debug("[{}] isn't absolute URI, assuming it's a path", url);
286                 return true;
287             }
288         } catch (IllegalArgumentException e) {
289             LOG.debug("[{}] isn't a valid URL, assuming it's a path", url, e);
290             return true;
291         } catch (MalformedURLException e) {
292             LOG.debug("[{}] isn't a valid URL, assuming it's a path", url, e);
293             return true;
294         }
295     }

304     }

306     /**
307      * Adds a request parameter to be added to the redirect url
308      * 
309      * @param key The parameter name
310      * @param value The parameter value
311      *
312      * @return the servlet redirect result
313      */
314     public ServletRedirectResult addParameter(String key, Object value) {
315         requestParameters.put(key, String.valueOf(value));
316         return this;
317     }

319     public void handle(ReflectionException ex) {
320         // Only log as debug as they are probably parameters to be appended to the url

322             LOG.debug(ex.getMessage(), ex);
323         }
