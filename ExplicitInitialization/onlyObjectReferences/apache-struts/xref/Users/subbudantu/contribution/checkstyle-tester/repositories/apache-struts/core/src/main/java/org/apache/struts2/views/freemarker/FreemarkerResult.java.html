52  /**
53   * <!-- START SNIPPET: description -->
54   *
55   * Renders a view using the Freemarker template engine.
56   * <p>
57   * The FreemarkarManager class configures the template loaders so that the
58   * template location can be either
59   * </p>
60   *
61   * <ul>
62   *
63   * <li>relative to the web root folder. eg <code>/WEB-INF/views/home.ftl</code>
64   * </li>
65   *
66   * <li>a classpath resuorce. eg <code>/com/company/web/views/home.ftl</code></li>
67   *
68   * </ul>
69   *
70   * <!-- END SNIPPET: description -->
71   *
72   * <b>This result type takes the following parameters:</b>
73   *
74   * <!-- START SNIPPET: params -->
75   *
76   * <ul>
77   *
78   * <li><b>location (default)</b> - the location of the template to process.</li>
79   *
80   * <li><b>parse</b> - true by default. If set to false, the location param will
81   * not be parsed for Ognl expressions.</li>
82   *
83   * <li><b>contentType</b> - defaults to "text/html" unless specified.</li>
84   * 
85   * <li><b>writeIfCompleted</b> - false by default, write to stream only if there isn't any error 
86   * processing the template. Setting template_exception_handler=rethrow in freemarker.properties
87   * will have the same effect.</li>
88   *
89   * </ul>
90   *
91   * <!-- END SNIPPET: params -->
92   *
93   * <b>Example:</b>
94   *
95   * <pre>
96   * <!-- START SNIPPET: example -->
97   *
98   * &lt;result name="success" type="freemarker"&gt;foo.ftl&lt;/result&gt;
99   *
100  * <!-- END SNIPPET: example -->
101  * </pre>
102  */

151     /**
152      * Execute this result, using the specified template locationArg.
153      * <p>
154      * The template locationArg has already been interpolated for any variable substitutions
155      * </p>
156      * <p>
157      * this method obtains the freemarker configuration and the object wrapper from the provided hooks.
158      * It them implements the template processing workflow by calling the hooks for
159      * preTemplateProcess and postTemplateProcess
160      * </p>
161      *
162      * @param locationArg location argument
163      * @param invocation the action invocation
164      *
165      * @throws IOException in case of IO errors
166      * @throws TemplateException in case of freemarker template errors
167      */
168     public void doExecute(String locationArg, ActionInvocation invocation) throws IOException, TemplateException {
169         this.location = locationArg;
170         this.invocation = invocation;
171         this.configuration = getConfiguration();
172         this.wrapper = getObjectWrapper();
173 
174         ActionContext ctx = invocation.getInvocationContext();
175         HttpServletRequest req = (HttpServletRequest) ctx.get(ServletActionContext.HTTP_REQUEST);
176 
177         String absoluteLocation;
178         if (location.startsWith("/")) {
179             absoluteLocation = location; 
180         } else { 
181             String namespace = invocation.getProxy().getNamespace();
182             if (namespace == null || namespace.length() == 0 || namespace.equals("/")) {
183                 absoluteLocation = "/" + location;
184             } else if (namespace.startsWith("/")) {
185                 absoluteLocation = namespace + "/" + location;
186             } else {
187                 absoluteLocation = "/" + namespace + "/" + location;
188             }
189         }
190 
191         Template template = configuration.getTemplate(absoluteLocation, deduceLocale());
192         TemplateModel model = createModel();
193 
194         // Give subclasses a chance to hook into preprocessing
195         if (preTemplateProcess(template, model)) {
196             try {
197                 // Process the template
198                 Writer writer = getWriter();
199                 if (isWriteIfCompleted() || configuration.getTemplateExceptionHandler() == TemplateExceptionHandler.RETHROW_HANDLER) {
200                     CharArrayWriter parentCharArrayWriter = (CharArrayWriter) req.getAttribute(PARENT_TEMPLATE_WRITER);
201                     boolean isTopTemplate = false;
202                     if (isTopTemplate = (parentCharArrayWriter == null)) {
203                         //this is the top template
204                         parentCharArrayWriter = new CharArrayWriter();
205                         //set it in the request because when the "action" tag is used a new VS and ActionContext is created
206                         req.setAttribute(PARENT_TEMPLATE_WRITER, parentCharArrayWriter);
207                     }
208 
209                     try {
210                         template.process(model, parentCharArrayWriter);
211 
212                         if (isTopTemplate) {
213                             parentCharArrayWriter.flush();
214                             parentCharArrayWriter.writeTo(writer);
215                         }
216                     } catch (TemplateException e) {
217                         if (LOG.isErrorEnabled()) {
218                             LOG.error("Error processing Freemarker result!", e);
219                         }
220                         throw e;
221                     } catch (IOException e) {
222                         if (LOG.isErrorEnabled()){
223                             LOG.error("Error processing Freemarker result!", e);
224                         }
225                         throw e;
226                     } finally {
227                         if (isTopTemplate && parentCharArrayWriter != null) {
228                             req.removeAttribute(PARENT_TEMPLATE_WRITER);
229                             parentCharArrayWriter.close();
230                         }
231                     }
232                 } else {
233                     template.process(model, writer);
234                 }
235             } finally {
236                 // Give subclasses a chance to hook into postprocessing
237                 postTemplateProcess(template, model);
238             }
239         }
240     }

300     /**
301      * Build the instance of the ScopesHashModel, including JspTagLib support
302      * <p>
303      * Objects added to the model are
304      * </p>
305      *
306      * <ul>
307      * <li>Application - servlet context attributes hash model
308      * <li>JspTaglibs - jsp tag lib factory model
309      * <li>Request - request attributes hash model
310      * <li>Session - session attributes hash model
311      * <li>request - the HttpServletRequst object for direct access
312      * <li>response - the HttpServletResponse object for direct access
313      * <li>stack - the OgnLValueStack instance for direct access
314      * <li>ognl - the instance of the OgnlTool
315      * <li>action - the action itself
316      * <li>exception - optional : the JSP or Servlet exception as per the servlet spec (for JSP Exception pages)
317      * <li>struts - instance of the StrutsUtil class
318      * </ul>
319      *
320      * @return TemplateModel returns the created template model
321      * @throws TemplateModelException in case of errors during creating the model
322      */
323     protected TemplateModel createModel() throws TemplateModelException {
324         ServletContext servletContext = ServletActionContext.getServletContext();
325         HttpServletRequest request = ServletActionContext.getRequest();
326         HttpServletResponse response = ServletActionContext.getResponse();
327         ValueStack stack = ServletActionContext.getContext().getValueStack();
328 
329         Object action = null;
330         if(invocation!= null ) action = invocation.getAction(); //Added for NullPointException
331         return freemarkerManager.buildTemplateModel(stack, action, servletContext, request, response, wrapper);
332     }

334     /**
335      * Returns the locale used for the {@link Configuration#getTemplate(String, Locale)} call. The base implementation
336      * simply returns the locale setting of the action (assuming the action implements {@link LocaleProvider}) or, if
337      * the action does not the configuration's locale is returned. Override this method to provide different behaviour,
338      *
339      * @return the locale from action if action implements the {@link LocaleProvider}) or local from configuration
340      */
341     protected Locale deduceLocale() {
342         if (invocation.getAction() instanceof LocaleProvider) {
343             return ((LocaleProvider) invocation.getAction()).getLocale();
344         } else {
345             return configuration.getLocale();
346         }
347     }

360     /**
361      * Called before the execution is passed to template.process().
362      * This is a generic hook you might use in subclasses to perform a specific
363      * action before the template is processed. By default does nothing.
364      * A typical action to perform here is to inject application-specific
365      * objects into the model root
366      *
367      * @param template the freemarker template
368      * @param model the template model
369      * @return true to process the template, false to suppress template processing.
370      * @throws IOException in case of IO errors
371      */
372     protected boolean preTemplateProcess(Template template, TemplateModel model) throws IOException {
373         Object attrContentType = template.getCustomAttribute("content_type");
374 
375         HttpServletResponse response = ServletActionContext.getResponse();
376         if (response.getContentType() == null) {
377             if (attrContentType != null) {
378                 response.setContentType(attrContentType.toString());
379             } else {
380                 String contentType = getContentType();
381 
382                 if (contentType == null) {
383                     contentType = "text/html";
384                 }
385 
386                 String encoding = template.getEncoding();
387 
388                 if (encoding != null) {
389                     contentType = contentType + "; charset=" + encoding;
390                 }
391 
392                 response.setContentType(contentType);
393             }
394         } else if(isInsideActionTag()){
395              //trigger com.opensymphony.module.sitemesh.filter.PageResponseWrapper.deactivateSiteMesh()
396             response.setContentType(response.getContentType());
397         }
398 
399         return true;
400     }

402     private boolean isInsideActionTag() {
403         Object attribute = ServletActionContext.getRequest().getAttribute(StrutsStatics.STRUTS_ACTION_TAG_INVOCATION);
404         return (Boolean) ObjectUtils.defaultIfNull(attribute, Boolean.FALSE);
405     }
