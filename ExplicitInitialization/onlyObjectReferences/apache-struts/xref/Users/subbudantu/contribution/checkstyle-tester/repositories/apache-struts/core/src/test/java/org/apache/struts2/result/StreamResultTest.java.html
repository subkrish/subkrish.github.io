76      public void testStreamResultDefault() throws Exception {
77          result.setInputName("streamForImage");
78  
79          result.doExecute("helloworld", mai);
80  
81          assertEquals(contentLength, response.getContentLength());
82          assertEquals("text/plain", response.getContentType());
83          assertEquals("streamForImage", result.getInputName());
84          assertEquals(1024, result.getBufferSize()); // 1024 is default
85          assertEquals("inline", result.getContentDisposition());
86  
87          assertEquals("text/plain", response.getContentType());
88          assertEquals(contentLength, response.getContentLength());
89          assertEquals("inline", response.getHeader("Content-disposition"));
90      }

92      public void testStreamResultWithCharSet() throws Exception {
93          result.setInputName("streamForImage");
94          result.setContentCharSet("ISO-8859-1");
95          result.doExecute("helloworld", mai);

97          assertEquals(contentLength, response.getContentLength());

167     public void testStreamResultParse1() throws Exception {
168         ///////////////////
169         result.setParse(true);
170         // ${...} conditionalParse of Result, returns String,
171         // which gets evaluated to the stack, that's how it works.
172         // We use ${streamForImageAsString} that returns "streamForImage"
173         // which is a property that returns an InputStream object.
174         result.setInputName("${streamForImageAsString}");
175         result.setBufferSize(128);
176         result.setContentLength(String.valueOf(contentLength));
177         result.setContentDisposition("filename=\"logo.png\"");
178         result.setContentType("image/jpeg");
179 
180         result.doExecute("helloworld", mai);
181 
182         assertEquals(String.valueOf(contentLength), result.getContentLength());
183         assertEquals("image/jpeg", result.getContentType());
184         assertEquals("${streamForImageAsString}", result.getInputName());
185         assertEquals(128, result.getBufferSize());
186         assertEquals("filename=\"logo.png\"", result.getContentDisposition());
187 
188         assertEquals("image/jpeg", response.getContentType());
189         assertEquals(contentLength, response.getContentLength());
190         assertEquals("filename=\"logo.png\"", response.getHeader("Content-disposition"));
191     }

193     public void testStreamResultParse2() throws Exception {

195         result.setParse(true);
196         // This time we dun use ${...}, so streamForImage will
197         // be evaluated to the stack, which should reaturn an
198         // InputStream object, cause there's such a property in
199         // the action object itself.
200         result.setInputName("streamForImage");
201         result.setBufferSize(128);
202         result.setContentLength(String.valueOf(contentLength));
203         result.setContentDisposition("filename=\"logo.png\"");
204         result.setContentType("image/jpeg");
205 
206         result.doExecute("helloworld", mai);
207 
208         assertEquals(String.valueOf(contentLength), result.getContentLength());
209         assertEquals("image/jpeg", result.getContentType());
210         assertEquals("streamForImage", result.getInputName());
211         assertEquals(128, result.getBufferSize());
212         assertEquals("filename=\"logo.png\"", result.getContentDisposition());
213 
214         assertEquals("image/jpeg", response.getContentType());
215         assertEquals(contentLength, response.getContentLength());
216         assertEquals("filename=\"logo.png\"", response.getHeader("Content-disposition"));

219     protected void setUp() throws Exception {
220         super.setUp();
221         response = new MockHttpServletResponse();
222 
223         result = new StreamResult();
224         result.setContentLength("${contentLength}");
225         stack = ActionContext.getContext().getValueStack();
226 
227         MyImageAction action = new MyImageAction();
228         contentLength = (int) action.getContentLength();
229 
230         mai = new com.opensymphony.xwork2.mock.MockActionInvocation();
231         mai.setAction(action);
232         mai.setStack(stack);
233         mai.setInvocationContext(ActionContext.getContext());
234         stack.push(action);
235 
236         ActionContext.getContext().put(ServletActionContext.HTTP_RESPONSE, response);
237     }
