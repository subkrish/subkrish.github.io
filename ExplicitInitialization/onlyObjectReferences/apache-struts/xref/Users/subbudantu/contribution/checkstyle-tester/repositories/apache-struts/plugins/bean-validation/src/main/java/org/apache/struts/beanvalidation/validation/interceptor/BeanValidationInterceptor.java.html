45  /**
46   * <p>
47   * Bean Validation interceptor. This Interceptor does not itself provide any Bean validation functionality but
48   * works as a bridge between Bean validation implementations like Apache Bval or Hibernate Validator and Struts2 validation mechanism.
49   * </p>
50   * <p>
51   * Interceptor will create a Validation Factory based on the provider class and will validate requested method or Action
52   * class. Hibernate bean validator will be used as a default validator in case of no provider class will be supplied to
53   * the interceptor.
54   * </p>
55   */
56  public class BeanValidationInterceptor extends MethodFilterInterceptor {
57  
58      private static final Logger LOG = LogManager.getLogger(BeanValidationInterceptor.class);
59  
60      protected BeanValidationManager beanValidationManager;
61      protected TextProviderFactory textProviderFactory;
62      protected boolean convertToUtf8 = false;
63      protected String convertFromEncoding = "ISO-8859-1";
64  
65      @Inject()
66      public void setBeanValidationManager(BeanValidationManager beanValidationManager) {
67          this.beanValidationManager = beanValidationManager;
68      }
69  
70      @Inject
71      public void setTextProviderFactory(TextProviderFactory textProviderFactory) {
72          this.textProviderFactory = textProviderFactory;
73      }
74  
75      @Inject(value = ValidatorConstants.CONVERT_MESSAGE_TO_UTF8, required = false)
76      public void setConvertToUtf8(String convertToUtf8) {
77          this.convertToUtf8 = BooleanUtils.toBoolean(convertToUtf8);
78      }
79  
80      @Inject(value = ValidatorConstants.CONVERT_MESSAGE_FROM, required = false)
81      public void setConvertFromEncoding(String convertFromEncoding) {
82          this.convertFromEncoding = convertFromEncoding;
83      }
84  
85      @Override
86      protected String doIntercept(ActionInvocation invocation) throws Exception {
87          Validator validator = this.beanValidationManager.getValidator();
88          if (validator == null) {
89              LOG.debug("There is no Bean Validator configured in class path. Skipping Bean validation..");
90              return invocation.invoke();
91          }
92          LOG.debug("Starting bean validation using validator: {}", validator.getClass());
93  
94          Object action = invocation.getAction();
95          ActionProxy actionProxy = invocation.getProxy();
96          String methodName = actionProxy.getMethod();
97  
98          if (LOG.isDebugEnabled()) {
99              LOG.debug("Validating [{}/{}] with method [{}]", invocation.getProxy().getNamespace(), invocation.getProxy().getActionName(), methodName);
100         }
101 
102         Collection<Method> annotatedMethods = AnnotationUtils.getAnnotatedMethods(action.getClass(), SkipValidation.class);
103 
104         if (!annotatedMethods.contains(getActionMethod(action.getClass(), methodName))) {
105             // performing bean validation on action
106             performBeanValidation(action, validator);
107         }
108 
109         return invocation.invoke();
110     }
111 
112     protected void performBeanValidation(Object action, Validator validator) {
113 
114         LOG.trace("Initiating bean validation..");
115 
116         Set<ConstraintViolation<Object>> constraintViolations;
117 
118         if (action instanceof ModelDriven) {
119             LOG.trace("Performing validation on model..");
120             Object model = (Object)((ModelDriven<?>) action).getModel();
121             constraintViolations = validator.validate(model);
122         } else {
123             LOG.trace("Performing validation on action..");
124             constraintViolations = validator.validate(action);
125         }
126 
127         addBeanValidationErrors(constraintViolations, action);
128     }
129 
130     @SuppressWarnings("nls")
131     private void addBeanValidationErrors(Set<ConstraintViolation<Object>> constraintViolations, Object action) {
132         if (constraintViolations != null) {
133             ValidatorContext validatorContext = new DelegatingValidatorContext(action, textProviderFactory);
134             for (ConstraintViolation<Object> constraintViolation : constraintViolations) {
135                 String key = constraintViolation.getMessage();
136                 String message = key;
137                 try {
138                     message = validatorContext.getText(key);
139                     if (convertToUtf8 && StringUtils.isNotBlank(message)) {
140                         message = new String(message.getBytes(convertFromEncoding), "UTF-8");
141                     }
142                 } catch (Exception e) {
143                     LOG.error("Error while trying to fetch message: {}", key, e);
144                 }
145 
146                 if (isActionError(constraintViolation)) {
147                     LOG.debug("Adding action error [{}]", message);
148                     validatorContext.addActionError(message);
149                 } else {
150                     ValidationError validationError = buildBeanValidationError(constraintViolation, message);
151                     String fieldName = validationError.getFieldName();
152                     if (action instanceof ModelDriven && fieldName.startsWith(ValidatorConstants.MODELDRIVEN_PREFIX)) {
153                         fieldName = fieldName.replace("model.", ValidatorConstants.EMPTY_SPACE);
154                     }
155                     if (LOG.isDebugEnabled()) {
156                         LOG.debug("Adding field error [{}] with message [{}]", fieldName, validationError.getMessage());
157                     }
158                     validatorContext.addFieldError(fieldName, validationError.getMessage());
159                 }
160             }
161         }
162     }
163 
164     protected ValidationError buildBeanValidationError(ConstraintViolation<Object> violation, String message) {
165 
166         if (violation.getPropertyPath().iterator().next().getName() != null) {
167             String fieldName = violation.getPropertyPath().toString();
168             String finalMessage = StringUtils.removeStart(message, fieldName + ValidatorConstants.FIELD_SEPERATOR);
169             return new ValidationError(fieldName, finalMessage);
170         }
171 
172         return null;
173     }
174 
175     /**
176      * Decide if a violation should be added to the fieldErrors or actionErrors
177      *
178      * @param violation the violation
179      *
180      * @return true if violation should be added to the fieldErrors or actionErrors
181      */
182     protected boolean isActionError(ConstraintViolation<Object> violation) {
183         return violation.getLeafBean() == violation.getInvalidValue();
184     }
185 
186     /**
187      * This is copied from DefaultActionInvocation
188      *
189      * @param actionClass the action class
190      * @param methodName the method name
191      *
192      * @return Method
193      *
194      * @throws NoSuchMethodException if no method with this name was found
195      */
196     protected Method getActionMethod(Class<?> actionClass, String methodName) throws NoSuchMethodException {
197         Method method;
198 
199         method = actionClass.getMethod(methodName);
200 
201         return method;
202     }
203 
204     /**
205      * Inner class for validation error
206      * Nice concept taken from  Oval plugin.
207      */
208 
209     class ValidationError {
210         private final String fieldName;
211         private final String message;
212 
213         ValidationError(String fieldName, String message) {
214             this.fieldName = fieldName;
215             this.message = message;
216         }
217 
218         public String getFieldName() {
219             return this.fieldName;
220         }
221 
222         public String getMessage() {
223             return this.message;
224         }
225     }
