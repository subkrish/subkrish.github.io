96      /**
97       * Constructs actions based on a list of packages.
98       *
99       * @param configuration         The XWork configuration that the new package configs and action configs
100      *                              are added to.
101      * @param container             Xwork Container
102      * @param objectFactory         The ObjectFactory used to create the actions and such.
103      * @param redirectToSlash       A boolean parameter that controls whether or not this will create an
104      *                              action for indexes. If this is set to true, index actions are not created because
105      *                              the unknown handler will redirect from /foo to /foo/. The only action that is created
106      *                              is to the empty action in the namespace (e.g. the namespace /foo and the action "").
107      * @param defaultParentPackage  The default parent package for all the configuration.
108      */
109     @Inject
110     public PackageBasedActionConfigBuilder(Configuration configuration, Container container, ObjectFactory objectFactory,
111                                            @Inject("struts.convention.redirect.to.slash") String redirectToSlash,
112                                            @Inject("struts.convention.default.parent.package") String defaultParentPackage) {
113 
114         // Validate that the parameters are okay
115         this.configuration = configuration;
116         this.actionNameBuilder = container.getInstance(ActionNameBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_ACTION_NAME_BUILDER));
117         this.resultMapBuilder = container.getInstance(ResultMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_RESULT_MAP_BUILDER));
118         this.interceptorMapBuilder = container.getInstance(InterceptorMapBuilder.class, container.getInstance(String.class, ConventionConstants.CONVENTION_INTERCEPTOR_MAP_BUILDER));
119         this.objectFactory = objectFactory;
120         this.redirectToSlash = Boolean.parseBoolean(redirectToSlash);
121 
122         if (LOG.isTraceEnabled()) {
123             LOG.trace("Setting action default parent package to [{}]", defaultParentPackage);
124         }
125 
126         this.defaultParentPackage = defaultParentPackage;
127     }

301     /**
302      * Builds the action configurations by loading all classes in the packages specified by the
303      * property <b>struts.convention.action.packages</b> and then figuring out which classes implement Action
304      * or have Action in their name. Next, if this class is in a Java package that hasn't been
305      * inspected a new PackageConfig (XWork) is created for that Java package using the Java package
306      * name. This will contain all the ActionConfigs for all the Action classes that are discovered
307      * within that Java package. Next, each class is inspected for the {@link ParentPackage}
308      * annotation which is used to control the parent package for a specific action. Lastly, the
309      * {@link ResultMapBuilder} is used to create ResultConfig instances of the action.
310      */
311     public void buildActionConfigs() {
312         //setup reload class loader based on dev settings
313         initReloadClassLoader();
314 
315         if (!disableActionScanning) {
316             if (actionPackages == null && packageLocators == null) {
317                 throw new ConfigurationException("At least a list of action packages or action package locators " +
318                         "must be given using one of the properties [struts.convention.action.packages] or " +
319                         "[struts.convention.package.locators]");
320             }
321 
322             if (LOG.isTraceEnabled()) {
323                 LOG.trace("Loading action configurations");
324                 if (actionPackages != null) {
325                     LOG.trace("Actions being loaded from action packages: {}", actionPackages);
326                 }
327                 if (packageLocators != null) {
328                     LOG.trace("Actions being loaded using package locator's: {}", packageLocators);
329                 }
330                 if (excludePackages != null) {
331                     LOG.trace("Excluding actions from packages: {}", excludePackages);
332                 }
333             }
334 
335             Set<Class> classes = findActions();
336             buildConfiguration(classes);
337         }
338     }

340     protected ClassLoaderInterface getClassLoaderInterface() {
341         if (isReloadEnabled()) {
342             return new ClassLoaderInterfaceDelegate(this.reloadingClassLoader);
343         }
344         else {
345             /*
346             if there is a ClassLoaderInterface in the context, use it, otherwise
347             default to the default ClassLoaderInterface (a wrapper around the current
348             thread classloader)
349             using this, other plugins (like OSGi) can plugin their own classloader for a while
350             and it will be used by Convention (it cannot be a bean, as Convention is likely to be
351             called multiple times, and it needs to use the default ClassLoaderInterface during normal startup)
352             */
353             ClassLoaderInterface classLoaderInterface = null;
354             ActionContext ctx = ActionContext.getContext();
355             if (ctx != null) {
356                 classLoaderInterface = (ClassLoaderInterface) ctx.get(ClassLoaderInterface.CLASS_LOADER_INTERFACE);
357             }
358             return ObjectUtils.defaultIfNull(classLoaderInterface, new ClassLoaderInterfaceDelegate(getClassLoader()));
359         }
360     }

366     @SuppressWarnings("unchecked")
367     protected Set<Class> findActions() {
368         Set<Class> classes = new HashSet<>();
369         try {
370             if (actionPackages != null || (packageLocators != null && !disablePackageLocatorsScanning)) {
371 
372                 // By default, ClassFinder scans EVERY class in the specified
373                 // url set, which can produce spurious warnings for non-action
374                 // classes that can't be loaded. We pass a package filter that
375                 // only considers classes that match the action packages
376                 // specified by the user
377                 Test<String> classPackageTest = getClassPackageTest();
378                 List<URL> urls = readUrls();
379                 ClassFinder finder = buildClassFinder(classPackageTest, urls);
380 
381                 Test<ClassFinder.ClassInfo> test = getActionClassTest();
382                 classes.addAll(finder.findClasses(test));
383             }
384         } catch (Exception ex) {
385             LOG.error("Unable to scan named packages", ex);
386         }
387 
388         return classes;
389     }

391     protected ClassFinder buildClassFinder(Test<String> classPackageTest, List<URL> urls) {
392         if (classFinderFactory != null) {
393             LOG.trace("Using ClassFinderFactory to create instance of ClassFinder!");
394             return classFinderFactory.buildClassFinder(getClassLoaderInterface(), urls, EXTRACT_BASE_INTERFACES, fileProtocols, classPackageTest);
395         } else {
396             LOG.trace("ClassFinderFactory not defined, fallback to default ClassFinder implementation");
397             return new DefaultClassFinder(getClassLoaderInterface(), urls, EXTRACT_BASE_INTERFACES, fileProtocols, classPackageTest);
398         }
399     }

401     private List<URL> readUrls() throws IOException {
402         List<URL> resourceUrls = new ArrayList<>();
403         // Usually the "classes" dir.
404         ArrayList<URL> classesList = Collections.list(getClassLoaderInterface().getResources(""));
405         for (URL url : classesList) {
406             resourceUrls.addAll(fileManager.getAllPhysicalUrls(url));
407         }
408         return buildUrlSet(resourceUrls).getUrls();
409     }
410 
411     private UrlSet buildUrlSet(List<URL> resourceUrls) throws IOException {
412         ClassLoaderInterface classLoaderInterface = getClassLoaderInterface();
413         UrlSet urlSet = new UrlSet(resourceUrls);
414         urlSet = urlSet.include(new UrlSet(classLoaderInterface, this.fileProtocols));
415 
416         //excluding the urls found by the parent class loader is desired, but fails in JBoss (all urls are removed)
417         if (excludeParentClassLoader) {
418             //exclude parent of classloaders
419             ClassLoaderInterface parent = classLoaderInterface.getParent();
420             //if reload is enabled, we need to step up one level, otherwise the UrlSet will be empty
421             //this happens because the parent of the reloading class loader is the web app classloader
422             if (parent != null && isReloadEnabled())
423                 parent = parent.getParent();
424 
425             if (parent != null)
426                 urlSet = urlSet.exclude(parent);
427 
428             try {
429                 // This may fail in some sandboxes, ie GAE
430                 ClassLoader systemClassLoader = ClassLoader.getSystemClassLoader();
431                 urlSet = urlSet.exclude(new ClassLoaderInterfaceDelegate(systemClassLoader.getParent()));
432 
433             } catch (SecurityException e) {
434                 LOG.warn("Could not get the system classloader due to security constraints, there may be improper urls left to scan");
435             }
436         }
437 
438         //try to find classes dirs inside war files
439         urlSet = urlSet.includeClassesUrl(classLoaderInterface, new UrlSet.FileProtocolNormalizer() {
440             public URL normalizeToFileProtocol(URL url) {
441                 return fileManager.normalizeToFileProtocol(url);
442             }
443         });
444 
445 
446         urlSet = urlSet.excludeJavaExtDirs()
447                     .excludeJavaEndorsedDirs()
448                     .excludeUserExtensionsDir();
449 
450         try {
451         	urlSet = urlSet.excludeJavaHome();
452         } catch (NullPointerException e) {
453         	// This happens in GAE since the sandbox contains no java.home directory
454       	    LOG.warn("Could not exclude JAVA_HOME, is this a sandbox jvm?");
455         }
456         urlSet = urlSet.excludePaths(System.getProperty("sun.boot.class.path", ""));
457         urlSet = urlSet.exclude(".*/JavaVM.framework/.*");
458 
459         if (includeJars == null) {
460             urlSet = urlSet.exclude(".*?\\.jar(!/|/)?");
461         } else {
462             if(LOG.isDebugEnabled()) {
463                 LOG.debug("jar urls regexes were specified: {}", Arrays.asList(includeJars));
464             }
465             List<URL> rawIncludedUrls = urlSet.getUrls();
466             Set<URL> includeUrls = new HashSet<>();
467             boolean[] patternUsed = new boolean[includeJars.length];
468 
469             for (URL url : rawIncludedUrls) {
470                 if (fileProtocols.contains(url.getProtocol())) {
471                     //it is a jar file, make sure it matches at least a url regex
472                     for (int i = 0; i < includeJars.length; i++) {
473                         String includeJar = includeJars[i];
474                         if (Pattern.matches(includeJar, url.toExternalForm())) {
475                             includeUrls.add(url);
476                             patternUsed[i] = true;
477                             break;
478                         }
479                     }
480                 } else {
481                     LOG.debug("It is not a jar [{}]", url);
482                     includeUrls.add(url);
483                 }
484             }
485 
486             if (LOG.isWarnEnabled()) {
487                 for (int i = 0; i < patternUsed.length; i++) {
488                     if (!patternUsed[i]) {
489                         LOG.warn("The includeJars pattern [{}] did not match any jars in the classpath", includeJars[i]);
490                     }
491                 }
492             }
493             return new UrlSet(includeUrls);
494         }
495 
496         return urlSet;
497     }

499     /**
500      * Note that we can't include the test for {@link #actionSuffix} here
501      * because a class is included if its name ends in {@link #actionSuffix} OR
502      * it implements {@link com.opensymphony.xwork2.Action}. Since the whole
503      * goal is to avoid loading the class if we don't have to, the (actionSuffix
504      * || implements Action) test will have to remain until later. See
505      * {@link #getActionClassTest()} for the test performed on the loaded
506      * {@link ClassFinder.ClassInfo} structure.
507      *
508      * @param className the name of the class to test
509      * @return true if the specified class should be included in the
510      *         package-based action scan
511      */
512     protected boolean includeClassNameInActionScan(String className) {
513         String classPackageName = StringUtils.substringBeforeLast(className, ".");
514         return (checkActionPackages(classPackageName) || checkPackageLocators(classPackageName)) && checkExcludePackages(classPackageName);
515     }

517     /**
518      * Checks if provided class package is on the exclude list
519      *
520      * @param classPackageName name of class package
521      * @return false if class package is on the {@link #excludePackages} list
522      */
523     protected boolean checkExcludePackages(String classPackageName) {
524         if(excludePackages != null && excludePackages.length > 0) {
525             WildcardHelper wildcardHelper = new WildcardHelper();
526 
527             //we really don't care about the results, just the boolean
528             Map<String, String> matchMap = new HashMap<>();
529 
530             for(String packageExclude : excludePackages) {
531                 int[] packagePattern = wildcardHelper.compilePattern(packageExclude);
532                 if(wildcardHelper.match(matchMap, classPackageName, packagePattern)) {
533                     return false;
534                 }
535             }
536         }
537         return true;
538     }

540     /**
541      * Checks if class package match provided list of action packages
542      *
543      * @param classPackageName name of class package
544      * @return true if class package is on the {@link #actionPackages} list
545      */
546     protected boolean checkActionPackages(String classPackageName) {
547         if (actionPackages != null) {
548             for (String packageName : actionPackages) {
549                 String strictPackageName = packageName + ".";
550                 if (classPackageName.equals(packageName) || classPackageName.startsWith(strictPackageName)) {
551                     return true;
552                 }
553             }
554         }
555         return false;
556     }

558     /**
559      * Checks if class package match provided list of package locators
560      *
561      * @param classPackageName name of class package
562      * @return true if class package is on the {@link #packageLocators} list
563      */
564     protected boolean checkPackageLocators(String classPackageName) {
565         if (packageLocators != null && !disablePackageLocatorsScanning && classPackageName.length() > 0
566                 && (packageLocatorsBasePackage == null || classPackageName
567                         .startsWith(packageLocatorsBasePackage))) {
568             for (String packageLocator : packageLocators) {
569                 String[] splitted = classPackageName.split("\\.");
570 
571                 if (StringTools.contains(splitted, packageLocator, false)) {
572                     return true;
573                 }
574             }
575         }
576         return false;
577     }

581      * should be included in the package scan based on the clazz's package name.
582      * Note that the goal is to avoid loading the class, so the test should only
583      * rely on information in the class name itself. The default implementation
584      * is to return the result of {@link #includeClassNameInActionScan(String)}.
585      *

587      *         name should be included in the package scan

590         return new Test<String>() {
591             public boolean test(String className) {
592                 return includeClassNameInActionScan(className);
593             }
594         };

597     /**
598      * Construct a {@link Test} Object that determines if a specified class
599      * should be included in the package scan based on the full {@link ClassFinder.ClassInfo}
600      * of the class. At this point, the class has been loaded, so it's ok to
601      * perform tests such as checking annotations or looking at interfaces or
602      * super-classes of the specified class.
603      *
604      * @return a {@link Test} object that returns true if the specified class
605      *         should be included in the package scan
606      */
607     protected Test<ClassFinder.ClassInfo> getActionClassTest() {
608         return new Test<ClassFinder.ClassInfo>() {
609             public boolean test(ClassFinder.ClassInfo classInfo) {
610 
611                 // Why do we call includeClassNameInActionScan here, when it's
612                 // already been called to in the initial call to ClassFinder?
613                 // When some action class passes our package filter in that step,
614                 // ClassFinder automatically includes parent classes of that action,
615                 // such as com.opensymphony.xwork2.ActionSupport.  We repeat the
616                 // package filter here to filter out such results.
617                 boolean inPackage = includeClassNameInActionScan(classInfo.getName());
618                 boolean nameMatches = classInfo.getName().endsWith(actionSuffix);
619 
620                 try {
621                     return inPackage && (nameMatches || (checkImplementsAction && com.opensymphony.xwork2.Action.class.isAssignableFrom(classInfo.get())));
622                 } catch (ClassNotFoundException ex) {
623                     LOG.error("Unable to load class [{}]", classInfo.getName(), ex);
624                     return false;
625                 }
626             }
627         };
628     }

630     @SuppressWarnings("unchecked")
631     protected void buildConfiguration(Set<Class> classes) {
632         Map<String, PackageConfig.Builder> packageConfigs = new HashMap<>();
633 
634         for (Class<?> actionClass : classes) {
635             Actions actionsAnnotation = actionClass.getAnnotation(Actions.class);
636             Action actionAnnotation = actionClass.getAnnotation(Action.class);
637 
638             // Skip classes that can't be instantiated
639             if (cannotInstantiate(actionClass)) {
640                 LOG.trace("Class [{}] did not pass the instantiation test and will be ignored", actionClass.getName());
641                 continue;
642             }
643 
644             if (eagerLoading) {
645                 // Tell the ObjectFactory about this class
646                 try {
647                     objectFactory.getClassInstance(actionClass.getName());
648                 } catch (ClassNotFoundException e) {
649                     LOG.error("Object Factory was unable to load class [{}]", actionClass.getName(), e);
650                     throw new StrutsException("Object Factory was unable to load class " + actionClass.getName(), e);
651                 }
652             }
653 
654             // Determine the action package
655             String actionPackage = actionClass.getPackage().getName();
656             LOG.debug("Processing class [{}] in package [{}]", actionClass.getName(), actionPackage);
657 
658             Set<String> allowedMethods = getAllowedMethods(actionClass);
659 
660             // Determine the default namespace and action name
661             List<String> namespaces = determineActionNamespace(actionClass);
662             for (String namespace : namespaces) {
663                 String defaultActionName = determineActionName(actionClass);
664                 PackageConfig.Builder defaultPackageConfig = getPackageConfig(packageConfigs, namespace,
665                         actionPackage, actionClass, null);
666 
667                 // Verify that the annotations have no errors and also determine if the default action
668                 // configuration should still be built or not.
669                 Map<String, List<Action>> map = getActionAnnotations(actionClass);
670                 Set<String> actionNames = new HashSet<>();
671                 boolean hasDefaultMethod = ReflectionTools.containsMethod(actionClass, DEFAULT_METHOD);
672                 if (!map.containsKey(DEFAULT_METHOD)
673                         && hasDefaultMethod
674                         && actionAnnotation == null && actionsAnnotation == null
675                         && (alwaysMapExecute || map.isEmpty())) {
676                     boolean found = false;
677                     for (String method : map.keySet()) {
678                         List<Action> actions = map.get(method);
679                         for (Action action : actions) {
680 
681                             // Check if there are duplicate action names in the annotations.
682                             String actionName = action.value().equals(Action.DEFAULT_VALUE) ? defaultActionName : action.value();
683                             if (actionNames.contains(actionName)) {
684                                 throw new ConfigurationException("The action class [" + actionClass +
685                                         "] contains two methods with an action name annotation whose value " +
686                                         "is the same (they both might be empty as well).");
687                             } else {
688                                 actionNames.add(actionName);
689                             }
690 
691                             // Check this annotation is the default action
692                             if (action.value().equals(Action.DEFAULT_VALUE)) {
693                                 found = true;
694                             }
695                         }
696                     }
697 
698                     // Build the default
699                     if (!found) {
700                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, DEFAULT_METHOD, null, allowedMethods);
701                     }
702                 }
703 
704                 // Build the actions for the annotations
705                 for (String method : map.keySet()) {
706                     List<Action> actions = map.get(method);
707                     for (Action action : actions) {
708                         PackageConfig.Builder pkgCfg = defaultPackageConfig;
709                         if (action.value().contains("/") && !slashesInActionNames) {
710                             pkgCfg = getPackageConfig(packageConfigs, namespace, actionPackage,
711                                     actionClass, action);
712                         }
713 
714                         createActionConfig(pkgCfg, actionClass, defaultActionName, method, action, allowedMethods);
715                     }
716                 }
717 
718                 // some actions will not have any @Action or a default method, like the rest actions
719                 // where the action mapper is the one that finds the right method at runtime
720                 if (map.isEmpty() && mapAllMatches && actionAnnotation == null && actionsAnnotation == null) {
721                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, null, actionAnnotation, allowedMethods);
722                 }
723 
724                 //if there are @Actions or @Action at the class level, create the mappings for them
725                 String methodName = hasDefaultMethod ? DEFAULT_METHOD : null;
726                 if (actionsAnnotation != null) {
727                     List<Action> actionAnnotations = checkActionsAnnotation(actionsAnnotation);
728                     for (Action actionAnnotation2 : actionAnnotations)
729                         createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation2, allowedMethods);
730                 } else if (actionAnnotation != null)
731                     createActionConfig(defaultPackageConfig, actionClass, defaultActionName, methodName, actionAnnotation, allowedMethods);
732             }
733         }
734 
735         buildIndexActions(packageConfigs);
736 
737         // Add the new actions to the configuration
738         Set<String> packageNames = packageConfigs.keySet();
739         for (String packageName : packageNames) {
740             configuration.addPackageConfig(packageName, packageConfigs.get(packageName).build());
741         }
742     }

744     private Set<String> getAllowedMethods(Class<?> actionClass) {
745         AllowedMethods annotation = AnnotationUtils.findAnnotation(actionClass, AllowedMethods.class);
746         if (annotation == null) {
747             return Collections.emptySet();
748         } else {
749             Set<String> methods = new HashSet<>();
750             for (String method : annotation.value()) {
751                 methods.add(method);
752             }
753             return methods;
754         }

757     /**
758      * Interfaces, enums, annotations, and abstract classes cannot be instantiated.
759      * @param actionClass class to check
760      * @return returns true if the class cannot be instantiated or should be ignored
761      */
762     protected boolean cannotInstantiate(Class<?> actionClass) {
763         return actionClass.isAnnotation() || actionClass.isInterface() || actionClass.isEnum() ||
764                 (actionClass.getModifiers() & Modifier.ABSTRACT) != 0 || actionClass.isAnonymousClass();
765     }

767     /**
768      * Determines the namespace(s) for the action based on the action class. If there is a {@link Namespace}
769      * annotation on the class (including parent classes) or on the package that the class is in, than
770      * it is used. Otherwise, the Java package name that the class is in is used in conjunction with
771      * either the <b>struts.convention.action.packages</b> or <b>struts.convention.package.locators</b>
772      * configuration values. These are used to determine which part of the Java package name should
773      * be converted into the namespace for the XWork PackageConfig.
774      *
775      * @param actionClass The action class.
776      * @return The namespace or an empty string.
777      */
778     protected List<String> determineActionNamespace(Class<?> actionClass) {
779         List<String> namespaces = new ArrayList<>();
780 
781         // Check if there is a class or package level annotation for the namespace
782         //single namespace
783         Namespace namespaceAnnotation = AnnotationUtils.findAnnotation(actionClass, Namespace.class);
784         if (namespaceAnnotation != null) {
785             LOG.trace("Using non-default action namespace from Namespace annotation of [{}]", namespaceAnnotation.value());
786 
787             namespaces.add(namespaceAnnotation.value());
788         }
789 
790         //multiple annotations
791         Namespaces namespacesAnnotation = AnnotationUtils.findAnnotation(actionClass, Namespaces.class);
792         if (namespacesAnnotation != null) {
793             if (LOG.isTraceEnabled()) {
794                 StringBuilder sb = new StringBuilder();
795                 for (Namespace namespace : namespacesAnnotation.value())
796                     sb.append(namespace.value()).append(",");
797                 sb.deleteCharAt(sb.length() - 1);
798                 LOG.trace("Using non-default action namespaces from Namespaces annotation of [{}]", sb.toString());
799             }
800 
801             for (Namespace namespace : namespacesAnnotation.value()) {
802                 namespaces.add(namespace.value());
803             }
804         }
805 
806         //don't use default if there are annotations
807         if (!namespaces.isEmpty())
808             return namespaces;
809 
810         String pkg = actionClass.getPackage().getName();
811         String pkgPart = null;
812         if (actionPackages != null) {
813             for (String actionPackage : actionPackages) {
814                 if (pkg.startsWith(actionPackage)) {
815                     pkgPart = actionClass.getName().substring(actionPackage.length() + 1);
816                 }
817             }
818         }
819 
820         if (pkgPart == null && packageLocators != null) {
821             for (String packageLocator : packageLocators) {
822                 // check subpackage and not a part of package name, eg. actions -> my.actions.transactions - WW-3803
823                 int index = pkg.lastIndexOf("." + packageLocator + ".");
824 
825                 // This ensures that the match is at the end, beginning or has a dot on each side of it
826                 if (index >= 0 && (index + packageLocator.length() == pkg.length() || index == 0 ||
827                         (pkg.charAt(index) == '.' && pkg.charAt(index + 1 + packageLocator.length()) == '.'))) {
828                     pkgPart = actionClass.getName().substring(index + packageLocator.length() + 2);
829                 }
830             }
831         }
832 
833         if (pkgPart != null) {
834             final int indexOfDot = pkgPart.lastIndexOf('.');
835             if (indexOfDot >= 0) {
836                 String convertedNamespace = actionNameBuilder.build(pkgPart.substring(0, indexOfDot));
837                 namespaces.add("/" + convertedNamespace.replace('.', '/'));
838                 return namespaces;
839             }
840         }
841 
842         namespaces.add("");
843         return namespaces;
844     }

859     /**
860      * Locates all of the {@link Actions} and {@link Action} annotations on methods within the Action
861      * class and its parent classes.
862      *
863      * @param actionClass The action class.
864      * @return The list of annotations or an empty list if there are none.
865      */
866     protected Map<String, List<Action>> getActionAnnotations(Class<?> actionClass) {
867         Method[] methods = actionClass.getMethods();
868         Map<String, List<Action>> map = new HashMap<>();
869         for (Method method : methods) {
870             Actions actionsAnnotation = method.getAnnotation(Actions.class);
871             if (actionsAnnotation != null) {
872                 List<Action> actions = checkActionsAnnotation(actionsAnnotation);
873                 map.put(method.getName(), actions);
874             } else {
875                 Action ann = method.getAnnotation(Action.class);
876                 if (ann != null) {
877                     map.put(method.getName(), Arrays.asList(ann));
878                 }
879             }
880         }
881 
882         return map;
883     }

885     /**
886      *  Builds a list of actions from an @Actions annotation, and check that they are not all empty
887      * @param actionsAnnotation Actions annotation
888      * @return a list   of Actions
889      */
890     protected List<Action> checkActionsAnnotation(Actions actionsAnnotation) {
891         Action[] actionArray = actionsAnnotation.value();
892         boolean valuelessSeen = false;
893         List<Action> actions = new ArrayList<>();
894         for (Action ann : actionArray) {
895             if (ann.value().equals(Action.DEFAULT_VALUE) && !valuelessSeen) {
896                 valuelessSeen = true;
897             } else if (ann.value().equals(Action.DEFAULT_VALUE)) {
898                 throw new ConfigurationException("You may only add a single Action annotation that has no value parameter.");
899             }
900 
901             actions.add(ann);
902         }
903         return actions;
904     }

906     /**
907      * Creates a single ActionConfig object.
908      *
909      * @param pkgCfg       The package the action configuration instance will belong to.
910      * @param actionClass  The action class.
911      * @param actionName   The name of the action.
912      * @param actionMethod The method that the annotation was on (if the annotation is not null) or
913      *                     the default method (execute).
914      * @param annotation   The ActionName annotation that might override the action name and possibly
915      */
916     protected void createActionConfig(PackageConfig.Builder pkgCfg, Class<?> actionClass, String actionName,
917                                       String actionMethod, Action annotation, Set<String> allowedMethods) {
918     	String className = actionClass.getName();
919         if (annotation != null) {
920             actionName = annotation.value() != null && annotation.value().equals(Action.DEFAULT_VALUE) ? actionName : annotation.value();
921             actionName = StringUtils.contains(actionName, "/") && !slashesInActionNames ? StringUtils.substringAfterLast(actionName, "/") : actionName;
922             if(!Action.DEFAULT_VALUE.equals(annotation.className())){
923             	className = annotation.className();
924             }
925         }
926 
927         ActionConfig.Builder actionConfig = new ActionConfig.Builder(pkgCfg.getName(), actionName, className);
928         actionConfig.methodName(actionMethod);
929 
930         if (pkgCfg.isStrictMethodInvocation()) {
931             actionConfig.addAllowedMethod(actionMethod);
932             actionConfig.addAllowedMethod(allowedMethods);
933             actionConfig.addAllowedMethod(pkgCfg.getGlobalAllowedMethods());
934         } else {
935             actionConfig.addAllowedMethod(ActionConfig.WILDCARD);
936         }
937 
938         if (LOG.isDebugEnabled()) {
939             LOG.debug("Creating action config for class [{}], name [{}] and package name [{}] in namespace [{}]",
940                     actionClass.toString(), actionName, pkgCfg.getName(), pkgCfg.getNamespace());
941         }
942 
943         //build interceptors
944         List<InterceptorMapping> interceptors = interceptorMapBuilder.build(actionClass, pkgCfg, actionName, annotation);
945         actionConfig.addInterceptors(interceptors);
946 
947         //build results
948         Map<String, ResultConfig> results = resultMapBuilder.build(actionClass, annotation, actionName, pkgCfg.build());
949         actionConfig.addResultConfigs(results);
950 
951         //add params
952         if (annotation != null)
953             actionConfig.addParams(StringTools.createParameterMap(annotation.params()));
954 
955         //add exception mappings from annotation
956         if (annotation != null && annotation.exceptionMappings() != null)
957             actionConfig.addExceptionMappings(buildExceptionMappings(annotation.exceptionMappings(), actionName));
958 
959         //add exception mapping from class
960         ExceptionMappings exceptionMappings = actionClass.getAnnotation(ExceptionMappings.class);
961         if (exceptionMappings != null)
962             actionConfig.addExceptionMappings(buildExceptionMappings(exceptionMappings.value(), actionName));
963 
964         //add
965         pkgCfg.addActionConfig(actionName, actionConfig.build());
966 
967         //check if an action with the same name exists on that package (from XML config probably)
968         PackageConfig existingPkg = configuration.getPackageConfig(pkgCfg.getName());
969         if (existingPkg != null) {
970             // there is a package already with that name, check action
971             ActionConfig existingActionConfig = existingPkg.getActionConfigs().get(actionName);
972             if (existingActionConfig != null && LOG.isWarnEnabled()) {
973                 LOG.warn("Duplicated action definition in package [{}] with name [{}].", pkgCfg.getName(), actionName);
974             }
975         }
976 
977         //watch class file
978         if (isReloadEnabled()) {
979             URL classFile = actionClass.getResource(actionClass.getSimpleName() + ".class");
980             fileManager.monitorFile(classFile);
981             loadedFileUrls.add(classFile.toString());
982         }
983     }

985     protected List<ExceptionMappingConfig> buildExceptionMappings(ExceptionMapping[] exceptions, String actionName) {
986         List<ExceptionMappingConfig> exceptionMappings = new ArrayList<>();
987 
988         for (ExceptionMapping exceptionMapping : exceptions) {
989             LOG.trace("Mapping exception [{}] to result [{}] for action [{}]", exceptionMapping.exception(),
990                         exceptionMapping.result(), actionName);
991             ExceptionMappingConfig.Builder builder = new ExceptionMappingConfig.Builder(null, exceptionMapping
992                     .exception(), exceptionMapping.result());
993             if (exceptionMapping.params() != null)
994                 builder.addParams(StringTools.createParameterMap(exceptionMapping.params()));
995             exceptionMappings.add(builder.build());
996         }
997 
998         return exceptionMappings;
999     }

1001     protected PackageConfig.Builder getPackageConfig(final Map<String, PackageConfig.Builder> packageConfigs,
1002                                                    String actionNamespace, final String actionPackage, final Class<?> actionClass,
1003                                                    Action action) {
1004         if (action != null && !action.value().equals(Action.DEFAULT_VALUE)) {
1005             LOG.trace("Using non-default action namespace from the Action annotation of [{}]", action.value());
1006             String actionName = action.value();
1007             actionNamespace = StringUtils.contains(actionName, "/") ? StringUtils.substringBeforeLast(actionName, "/") : StringUtils.EMPTY;
1008         }
1009 
1010         // Next grab the parent annotation from the class
1011         ParentPackage parent = AnnotationUtils.findAnnotation(actionClass, ParentPackage.class);
1012         String parentName = null;
1013         if (parent != null) {
1014             LOG.trace("Using non-default parent package from annotation of [{}]", parent.value());
1015             parentName = parent.value();
1016         }
1017 
1018         // Finally use the default
1019         if (parentName == null) {
1020             parentName = defaultParentPackage;
1021         }
1022 
1023         if (parentName == null) {
1024             throw new ConfigurationException("Unable to determine the parent XWork package for the action class [" +
1025                     actionClass.getName() + "]");
1026         }
1027 
1028         PackageConfig parentPkg = configuration.getPackageConfig(parentName);
1029         if (parentPkg == null) {
1030             throw new ConfigurationException("Unable to locate parent package [" + parentName + "] for [" + actionClass + "]");
1031         }
1032 
1033         // Grab based on package-namespace and if it exists, we need to ensure the existing one has
1034         // the correct parent package. If not, we need to create a new package config
1035         String name = actionPackage + "#" + parentPkg.getName() + "#" + actionNamespace;
1036         PackageConfig.Builder pkgConfig = packageConfigs.get(name);
1037         if (pkgConfig == null) {
1038             pkgConfig = new PackageConfig.Builder(name).namespace(actionNamespace).addParent(parentPkg);
1039             packageConfigs.put(name, pkgConfig);
1040 
1041             //check for @DefaultInterceptorRef in the package
1042             DefaultInterceptorRef defaultInterceptorRef = AnnotationUtils.findAnnotation(actionClass, DefaultInterceptorRef.class);
1043             if (defaultInterceptorRef != null) {
1044                 pkgConfig.defaultInterceptorRef(defaultInterceptorRef.value());
1045 
1046                 LOG.trace("Setting [{}] as the default interceptor ref for [{}]", defaultInterceptorRef.value(), pkgConfig.getName());
1047             }
1048         }
1049 
1050         LOG.trace("Created package config named [{}] with a namespace [{}]", name, actionNamespace);
1051 
1052         return pkgConfig;
1053     }

1055     /**
1056      * Determine all the index handling actions and results based on this logic:
1057      *
1058      * 1. Loop over all the namespaces such as /foo and see if it has an action named index
1059      * 2. If an action doesn't exists in the parent namespace of the same name, create an action
1060      * in the parent namespace of the same name as the namespace that points to the index
1061      * action in the namespace. e.g. /foo -&gt; /foo/index
1062      * 3. Create the action in the namespace for empty string if it doesn't exist. e.g. /foo/
1063      * the action is "" and the namespace is /foo
1064      *
1065      * @param packageConfigs Used to store the actions.
1066      */
1067     protected void buildIndexActions(Map<String, PackageConfig.Builder> packageConfigs) {
1068         Map<String, PackageConfig.Builder> byNamespace = new HashMap<>();
1069         Collection<PackageConfig.Builder> values = packageConfigs.values();
1070         for (PackageConfig.Builder packageConfig : values) {
1071             byNamespace.put(packageConfig.getNamespace(), packageConfig);
1072         }
1073 
1074         // Step #1
1075         Set<String> namespaces = byNamespace.keySet();
1076         for (String namespace : namespaces) {
1077             // First see if the namespace has an index action
1078             PackageConfig.Builder pkgConfig = byNamespace.get(namespace);
1079             ActionConfig indexActionConfig = pkgConfig.build().getAllActionConfigs().get("index");
1080             if (indexActionConfig == null) {
1081                 continue;
1082             }
1083 
1084             // Step #2
1085             if (!redirectToSlash) {
1086                 int lastSlash = namespace.lastIndexOf('/');
1087                 if (lastSlash >= 0) {
1088                     String parentAction = namespace.substring(lastSlash + 1);
1089                     String parentNamespace = namespace.substring(0, lastSlash);
1090                     PackageConfig.Builder parent = byNamespace.get(parentNamespace);
1091                     if (parent == null || parent.build().getAllActionConfigs().get(parentAction) == null) {
1092                         if (parent == null) {
1093                             parent = new PackageConfig.Builder(parentNamespace).namespace(parentNamespace).
1094                                     addParents(pkgConfig.build().getParents());
1095                             packageConfigs.put(parentNamespace, parent);
1096                         }
1097 
1098                         if (parent.build().getAllActionConfigs().get(parentAction) == null) {
1099                             parent.addActionConfig(parentAction, indexActionConfig);
1100                         }
1101                     } else {
1102                         LOG.trace("The parent namespace [{}] already contains an action [{}]", parentNamespace, parentAction);
1103                     }
1104                 }
1105             }
1106 
1107             // Step #3
1108             if (pkgConfig.build().getAllActionConfigs().get("") == null) {
1109                 LOG.trace("Creating index ActionConfig with an action name of [] for the action class [{}]", indexActionConfig.getClassName());
1110                 pkgConfig.addActionConfig("", indexActionConfig);
1111             }
1112         }
1113     }
