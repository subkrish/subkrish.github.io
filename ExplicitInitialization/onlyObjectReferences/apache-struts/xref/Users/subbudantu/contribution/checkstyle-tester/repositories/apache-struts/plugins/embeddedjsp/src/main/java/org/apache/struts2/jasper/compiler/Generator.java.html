107     /**
108      * @param s
109      *            the input string
110      * @return quoted and escaped string, per Java rule
111      */
112     static String quote(String s) {
113 
114         if (s == null)
115             return "null";
116 
117         return '"' + escape(s) + '"';
118     }

120     /**
121      * @param s
122      *            the input string
123      * @return escaped string, per Java rule
124      */
125     static String escape(String s) {
126 
127         if (s == null)
128             return "";
129 
130         StringBuffer b = new StringBuffer();
131         for (int i = 0; i < s.length(); i++) {
132             char c = s.charAt(i);
133             if (c == '"')
134                 b.append('\\').append('"');
135             else if (c == '\\')
136                 b.append('\\').append('\\');
137             else if (c == '\n')
138                 b.append('\\').append('n');
139             else if (c == '\r')
140                 b.append('\\').append('r');
141             else
142                 b.append(c);
143         }
144         return b.toString();
145     }
146 
147     /**
148      * Single quote and escape a character
149      */
150     static String quote(char c) {
151 
152         StringBuffer b = new StringBuffer();
153         b.append('\'');
154         if (c == '\'')
155             b.append('\\').append('\'');
156         else if (c == '\\')
157             b.append('\\').append('\\');
158         else if (c == '\n')
159             b.append('\\').append('n');
160         else if (c == '\r')
161             b.append('\\').append('r');
162         else
163             b.append(c);
164         b.append('\'');
165         return b.toString();
166     }

168     private String createJspId() throws JasperException {
169         if (this.jspIdPrefix == null) {
170             StringBuffer sb = new StringBuffer(32);
171             String name = ctxt.getServletJavaFileName();
172             sb.append("jsp_").append(Math.abs(name.hashCode())).append('_');

174         }
175         return this.jspIdPrefix + (this.jspId++);

178     /**
179      * Generates declarations. This includes "info" of the page directive, and
180      * scriptlet declarations.
181      */
182     private void generateDeclarations(Node.Nodes page) throws JasperException {
183 
184         class DeclarationVisitor extends Node.Visitor {
185 
186             private boolean getServletInfoGenerated = false;
187 
188             /*
189              * Generates getServletInfo() method that returns the value of the
190              * page directive's 'info' attribute, if present.
191              * 
192              * The Validator has already ensured that if the translation unit
193              * contains more than one page directive with an 'info' attribute,
194              * their values match.
195              */
196             public void visit(Node.PageDirective n) throws JasperException {
197 
198                 if (getServletInfoGenerated) {
199                     return;
200                 }
201 
202                 String info = n.getAttributeValue("info");
203                 if (info == null)
204                     return;
205 
206                 getServletInfoGenerated = true;
207                 out.printil("public String getServletInfo() {");
208                 out.pushIndent();
209                 out.printin("return ");
210                 out.print(quote(info));
211                 out.println(";");
212                 out.popIndent();
213                 out.printil("}");
214                 out.println();
215             }
216 
217             public void visit(Node.Declaration n) throws JasperException {
218                 n.setBeginJavaLine(out.getJavaLine());
219                 out.printMultiLn(new String(n.getText()));
220                 out.println();
221                 n.setEndJavaLine(out.getJavaLine());
222             }
223 
224             // Custom Tags may contain declarations from tag plugins.
225             public void visit(Node.CustomTag n) throws JasperException {
226                 if (n.useTagPlugin()) {
227                     if (n.getAtSTag() != null) {
228                         n.getAtSTag().visit(this);
229                     }
230                     visitBody(n);
231                     if (n.getAtETag() != null) {
232                         n.getAtETag().visit(this);
233                     }
234                 } else {
235                     visitBody(n);
236                 }
237             }
238         }
239 
240         out.println();
241         page.visit(new DeclarationVisitor());
242     }

244     /**
245      * Compiles list of tag handler pool names.
246      */
247     private void compileTagHandlerPoolList(Node.Nodes page)
248             throws JasperException {
249 
250         class TagHandlerPoolVisitor extends Node.Visitor {
251 
252             private Vector names;
253 
254             /*
255              * Constructor
256              * 
257              * @param v Vector of tag handler pool names to populate
258              */
259             TagHandlerPoolVisitor(Vector v) {
260                 names = v;
261             }
262 
263             /*
264              * Gets the name of the tag handler pool for the given custom tag
265              * and adds it to the list of tag handler pool names unless it is
266              * already contained in it.
267              */
268             public void visit(Node.CustomTag n) throws JasperException {
269 
270                 if (!n.implementsSimpleTag()) {
271                     String name = createTagHandlerPoolName(n.getPrefix(), n
272                             .getLocalName(), n.getAttributes(), 
273                             n.getNamedAttributeNodes(), n.hasEmptyBody());
274                     n.setTagHandlerPoolName(name);
275                     if (!names.contains(name)) {
276                         names.add(name);
277                     }
278                 }
279                 visitBody(n);
280             }
281 
282             /*
283              * Creates the name of the tag handler pool whose tag handlers may
284              * be (re)used to service this action.
285              * 
286              * @return The name of the tag handler pool
287              */
288             private String createTagHandlerPoolName(String prefix,
289                     String shortName, Attributes attrs, Node.Nodes namedAttrs,
290                     boolean hasEmptyBody) {
291                 String poolName = null;
292 
293                 poolName = "_jspx_tagPool_" + prefix + "_" + shortName;
294                 if (attrs != null) {
295                     String[] attrNames =
296                         new String[attrs.getLength() + namedAttrs.size()];
297                     for (int i = 0; i < attrNames.length; i++) {
298                         attrNames[i] = attrs.getQName(i);
299                     }
300                     for (int i = 0; i < namedAttrs.size(); i++) {
301                         attrNames[attrs.getLength() + i] =
302                             ((NamedAttribute) namedAttrs.getNode(i)).getQName();
303                     }
304                     Arrays.sort(attrNames, Collections.reverseOrder());
305                     if (attrNames.length > 0) {
306                         poolName = poolName + "&";
307                     }
308                     for (int i = 0; i < attrNames.length; i++) {
309                         poolName = poolName + "_" + attrNames[i];
310                     }
311                 }
312                 if (hasEmptyBody) {
313                     poolName = poolName + "_nobody";
314                 }
315                 return JspUtil.makeJavaIdentifier(poolName);
316             }
317         }
318 
319         page.visit(new TagHandlerPoolVisitor(tagHandlerPoolNames));
320     }

322     private void declareTemporaryScriptingVars(Node.Nodes page)
323             throws JasperException {
324 
325         class ScriptingVarVisitor extends Node.Visitor {
326 
327             private Vector vars;
328 
329             ScriptingVarVisitor() {
330                 vars = new Vector();
331             }
332 
333             public void visit(Node.CustomTag n) throws JasperException {
334 
335                 if (n.getCustomNestingLevel() > 0) {
336                     TagVariableInfo[] tagVarInfos = n.getTagVariableInfos();
337                     VariableInfo[] varInfos = n.getVariableInfos();
338 
339                     if (varInfos.length > 0) {
340                         for (int i = 0; i < varInfos.length; i++) {
341                             String varName = varInfos[i].getVarName();
342                             String tmpVarName = "_jspx_" + varName + "_"
343                                     + n.getCustomNestingLevel();
344                             if (!vars.contains(tmpVarName)) {
345                                 vars.add(tmpVarName);
346                                 out.printin(varInfos[i].getClassName());
347                                 out.print(" ");
348                                 out.print(tmpVarName);
349                                 out.print(" = ");
350                                 out.print(null);
351                                 out.println(";");
352                             }
353                         }
354                     } else {
355                         for (int i = 0; i < tagVarInfos.length; i++) {
356                             String varName = tagVarInfos[i].getNameGiven();
357                             if (varName == null) {
358                                 varName = n.getTagData().getAttributeString(
359                                         tagVarInfos[i].getNameFromAttribute());
360                             } else if (tagVarInfos[i].getNameFromAttribute() != null) {
361                                 // alias
362                                 continue;
363                             }
364                             String tmpVarName = "_jspx_" + varName + "_"
365                                     + n.getCustomNestingLevel();
366                             if (!vars.contains(tmpVarName)) {
367                                 vars.add(tmpVarName);
368                                 out.printin(tagVarInfos[i].getClassName());
369                                 out.print(" ");
370                                 out.print(tmpVarName);
371                                 out.print(" = ");
372                                 out.print(null);
373                                 out.println(";");
374                             }
375                         }
376                     }
377                 }
378 
379                 visitBody(n);
380             }
381         }
382 
383         page.visit(new ScriptingVarVisitor());
384     }

668     /**
669      * Generates an XML Prolog, which includes an XML declaration and an XML
670      * doctype declaration.
671      */
672     private void generateXmlProlog(Node.Nodes page) {
673 
674         /*
675          * An XML declaration is generated under the following conditions: -
676          * 'omit-xml-declaration' attribute of <jsp:output> action is set to
677          * "no" or "false" - JSP document without a <jsp:root>
678          */
679         String omitXmlDecl = pageInfo.getOmitXmlDecl();
680         if ((omitXmlDecl != null && !JspUtil.booleanValue(omitXmlDecl))
681                 || (omitXmlDecl == null && page.getRoot().isXmlSyntax()
682                         && !pageInfo.hasJspRoot() && !ctxt.isTagFile())) {
683             String cType = pageInfo.getContentType();
684             String charSet = cType.substring(cType.indexOf("charset=") + 8);
685             out.printil("out.write(\"<?xml version=\\\"1.0\\\" encoding=\\\""
686                     + charSet + "\\\"?>\\n\");");
687         }
688 
689         /*
690          * Output a DOCTYPE declaration if the doctype-root-element appears. If
691          * doctype-public appears: <!DOCTYPE name PUBLIC "doctypePublic"
692          * "doctypeSystem"> else <!DOCTYPE name SYSTEM "doctypeSystem" >
693          */
694 
695         String doctypeName = pageInfo.getDoctypeName();
696         if (doctypeName != null) {
697             String doctypePublic = pageInfo.getDoctypePublic();
698             String doctypeSystem = pageInfo.getDoctypeSystem();
699             out.printin("out.write(\"<!DOCTYPE ");
700             out.print(doctypeName);
701             if (doctypePublic == null) {
702                 out.print(" SYSTEM \\\"");
703             } else {
704                 out.print(" PUBLIC \\\"");
705                 out.print(doctypePublic);
706                 out.print("\\\" \\\"");
707             }
708             out.print(doctypeSystem);
709             out.println("\\\">\\n\");");
710         }
711     }

788         /**
789          * Returns an attribute value, optionally URL encoded. If the value is a
790          * runtime expression, the result is the expression itself, as a string.
791          * If the result is an EL expression, we insert a call to the
792          * interpreter. If the result is a Named Attribute we insert the
793          * generated variable name. Otherwise the result is a string literal,
794          * quoted and escaped.
795          * 
796          * @param attr
797          *            An JspAttribute object
798          * @param encode
799          *            true if to be URL encoded
800          * @param expectedType
801          *            the expected type for an EL evaluation (ignored for
802          *            attributes that aren't EL expressions)
803          */
804         private String attributeValue(Node.JspAttribute attr, boolean encode,
805                 Class expectedType) {
806             String v = attr.getValue();
807             if (!attr.isNamedAttribute() && (v == null))
808                 return "";
809 
810             if (attr.isExpression()) {
811                 if (encode) {
812                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode(String.valueOf("
813                             + v + "), request.getCharacterEncoding())";
814                 }
815                 return v;
816             } else if (attr.isELInterpreterInput()) {
817                 v = attributeValueWithEL(this.isTagFile, v, expectedType,
818                         attr.getEL().getMapName());
819                 if (encode) {
820                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode("
821                             + v + ", request.getCharacterEncoding())";
822                 }
823                 return v;
824             } else if (attr.isNamedAttribute()) {
825                 return attr.getNamedAttributeNode().getTemporaryVariableName();
826             } else {
827                 if (encode) {
828                     return "org.apache.struts2.jasper.runtime.JspRuntimeLibrary.URLEncode("
829                             + quote(v) + ", request.getCharacterEncoding())";
830                 }
831                 return quote(v);
832             }
833         }

836         /*
837          * When interpreting the EL attribute value, literals outside the EL
838          * must not be unescaped but the EL processor will unescape them.
839          * Therefore, make sure only the EL expressions are processed by the EL
840          * processor.
841          */
842         private String attributeValueWithEL(boolean isTag, String tx,
843                 Class<?> expectedType, String mapName) {
844             if (tx==null) return null;
845             Class<?> type = expectedType;
846             int size = tx.length();
847             StringBuffer output = new StringBuffer(size);
848             boolean el = false;
849             int i = 0;
850             int mark = 0;
851             char ch;
852             
853             while(i < size){
854                 ch = tx.charAt(i);
855                 
856                 // Start of an EL expression
857                 if (!el && i+1 < size && ch == '$' && tx.charAt(i+1)=='{') {
858                     if (mark < i) {
859                         if (output.length() > 0) {
860                             output.append(" + ");
861                             // Composite expression - must coerce to String
862                             type = String.class;
863                         }
864                         output.append(quote(tx.substring(mark, i)));
865                     }
866                     mark = i;
867                     el = true;
868                     i += 2;
869                 } else if (ch=='\\' && i+1 < size &&
870                         (tx.charAt(i+1)=='$' || tx.charAt(i+1)=='}')) { 
871                     // Skip an escaped $ or }
872                     i += 2;
873                 } else if (el && ch=='}') {
874                     // End of an EL expression
875                     if (output.length() > 0) {
876                         output.append(" + ");
877                         // Composite expression - must coerce to String
878                         type = String.class;
879                     }
880                     output.append(
881                             JspUtil.interpreterCall(isTag,
882                                     tx.substring(mark, i+1), type,
883                                     mapName, false));
884                     mark = i + 1;
885                     el = false;
886                     ++i;
887                 } else {
888                     // Nothing to see here - move to next character
889                     ++i;
890                 }
891             }
892             if (!el && mark < i) {
893                 if (output.length() > 0) {
894                     output.append(" + ");
895                 }
896                 output.append(quote(tx.substring(mark, i)));
897             }
898             return output.toString();
899         }
