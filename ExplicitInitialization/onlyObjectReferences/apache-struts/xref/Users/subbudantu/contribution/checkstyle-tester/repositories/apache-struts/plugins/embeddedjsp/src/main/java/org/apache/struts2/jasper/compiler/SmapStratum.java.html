33      // Class for storing LineInfo data

35      /**
36       * Represents a single LineSection in an SMAP, associated with
37       * a particular stratum.
38       */
39      public static class LineInfo {
40          private int inputStartLine = -1;
41          private int outputStartLine = -1;
42          private int lineFileID = 0;
43          private int inputLineCount = 1;
44          private int outputLineIncrement = 1;
45          private boolean lineFileIDSet = false;
46  
47          /**
48           * @param inputStartLine  Sets InputStartLine.
49           */
50          public void setInputStartLine(int inputStartLine) {
51              if (inputStartLine < 0)
52                  throw new IllegalArgumentException("" + inputStartLine);
53              this.inputStartLine = inputStartLine;
54          }
55  
56          /** @param outputStartLine Sets OutputStartLine. */
57          public void setOutputStartLine(int outputStartLine) {
58              if (outputStartLine < 0)
59                  throw new IllegalArgumentException("" + outputStartLine);
60              this.outputStartLine = outputStartLine;
61          }
62  
63          /**
64               * @param lineFileID  Sets lineFileID.  Should be called only when different from
65               * that of prior LineInfo object (in any given context) or 0
66               * if the current LineInfo has no (logical) predecessor.
67               * <tt>LineInfo</tt> will print this file number no matter what.
68               */
69          public void setLineFileID(int lineFileID) {
70              if (lineFileID < 0)
71                  throw new IllegalArgumentException("" + lineFileID);
72              this.lineFileID = lineFileID;
73              this.lineFileIDSet = true;
74          }
75  
76          /** @param inputLineCount  Sets InputLineCount. */
77          public void setInputLineCount(int inputLineCount) {
78              if (inputLineCount < 0)
79                  throw new IllegalArgumentException("" + inputLineCount);
80              this.inputLineCount = inputLineCount;
81          }
82  
83          /** @param outputLineIncrement Sets OutputLineIncrement. */
84          public void setOutputLineIncrement(int outputLineIncrement) {
85              if (outputLineIncrement < 0)
86                  throw new IllegalArgumentException("" + outputLineIncrement);
87              this.outputLineIncrement = outputLineIncrement;
88          }
89  
90          /**
91           * @return Retrieves the current LineInfo as a String, print all values
92           * only when appropriate (but LineInfoID if and only if it's been
93           * specified, as its necessity is sensitive to context).
94           */
95          public String getString() {
96              if (inputStartLine == -1 || outputStartLine == -1)
97                  throw new IllegalStateException();
98              StringBuffer out = new StringBuffer();
99              out.append(inputStartLine);
100             if (lineFileIDSet)
101                 out.append("#" + lineFileID);
102             if (inputLineCount != 1)
103                 out.append("," + inputLineCount);
104             out.append(":" + outputStartLine);
105             if (outputLineIncrement != 1)
106                 out.append("," + outputLineIncrement);
107             out.append('\n');
108             return out.toString();
109         }
110 
111         public String toString() {
112             return getString();
113         }
114     }

154     /**
155      * Adds record of a new file, by filename and path.  The path
156      * may be relative to a source compilation path.
157      *
158      * @param filename the filename to add, unqualified by path
159      * @param filePath the path for the filename, potentially relative
160      *                 to a source compilation path
161      */
162     public void addFile(String filename, String filePath) {
163         int pathIndex = filePathList.indexOf(filePath);
164         if (pathIndex == -1) {
165             fileNameList.add(filename);
166             filePathList.add(filePath);
167         }
168     }

170     /**
171      * Combines consecutive LineInfos wherever possible
172      */
173     public void optimizeLineSection() {
174 
175         //Incorporate each LineInfo into the previous LineInfo's
176         //outputLineIncrement, if possible
177         int i = 0;
178         while (i < lineData.size() - 1) {
179             LineInfo li = (LineInfo)lineData.get(i);
180             LineInfo liNext = (LineInfo)lineData.get(i + 1);
181             if (!liNext.lineFileIDSet
182                 && liNext.inputStartLine == li.inputStartLine
183                 && liNext.inputLineCount == 1
184                 && li.inputLineCount == 1
185                 && liNext.outputStartLine
186                     == li.outputStartLine
187                         + li.inputLineCount * li.outputLineIncrement) {
188                 li.setOutputLineIncrement(
189                     liNext.outputStartLine
190                         - li.outputStartLine
191                         + liNext.outputLineIncrement);
192                 lineData.remove(i + 1);
193             } else {
194                 i++;
195             }
196         }
197 
198         //Incorporate each LineInfo into the previous LineInfo's
199         //inputLineCount, if possible
200         i = 0;
201         while (i < lineData.size() - 1) {
202             LineInfo li = (LineInfo)lineData.get(i);
203             LineInfo liNext = (LineInfo)lineData.get(i + 1);
204             if (!liNext.lineFileIDSet
205                 && liNext.inputStartLine == li.inputStartLine + li.inputLineCount
206                 && liNext.outputLineIncrement == li.outputLineIncrement
207                 && liNext.outputStartLine
208                     == li.outputStartLine
209                         + li.inputLineCount * li.outputLineIncrement) {
210                 li.setInputLineCount(li.inputLineCount + liNext.inputLineCount);
211                 lineData.remove(i + 1);
212             } else {
213                 i++;
214             }
215         }
216     }

218     /**
219      * Adds complete information about a simple line mapping.  Specify
220      * all the fields in this method; the back-end machinery takes care
221      * of printing only those that are necessary in the final SMAP.
222      * (My view is that fields are optional primarily for spatial efficiency,
223      * not for programmer convenience.  Could always add utility methods
224      * later.)
225      *
226      * @param inputStartLine starting line in the source file
227      *        (SMAP <tt>InputStartLine</tt>)
228      * @param inputFileName the filepath (or name) from which the input comes
229      *        (yields SMAP <tt>LineFileID</tt>)  Use unqualified names
230      *        carefully, and only when they uniquely identify a file.
231      * @param inputLineCount the number of lines in the input to map
232      *        (SMAP <tt>LineFileCount</tt>)
233      * @param outputStartLine starting line in the output file 
234      *        (SMAP <tt>OutputStartLine</tt>)
235      * @param outputLineIncrement number of output lines to map to each
236      *        input line (SMAP <tt>OutputLineIncrement</tt>).  <i>Given the
237      *        fact that the name starts with "output", I continuously have
238      *        the subconscious urge to call this field
239      *        <tt>OutputLineExcrement</tt>.</i>
240      */
241     public void addLineData(
242         int inputStartLine,
243         String inputFileName,
244         int inputLineCount,
245         int outputStartLine,
246         int outputLineIncrement) {
247         // check the input - what are you doing here??
248         int fileIndex = filePathList.indexOf(inputFileName);
249         if (fileIndex == -1) // still
250             throw new IllegalArgumentException(
251                 "inputFileName: " + inputFileName);
252 
253         //Jasper incorrectly SMAPs certain Nodes, giving them an 
254         //outputStartLine of 0.  This can cause a fatal error in
255         //optimizeLineSection, making it impossible for Jasper to
256         //compile the JSP.  Until we can fix the underlying
257         //SMAPping problem, we simply ignore the flawed SMAP entries.
258         if (outputStartLine == 0)
259             return;
260 
261         // build the LineInfo
262         LineInfo li = new LineInfo();
263         li.setInputStartLine(inputStartLine);
264         li.setInputLineCount(inputLineCount);
265         li.setOutputStartLine(outputStartLine);
266         li.setOutputLineIncrement(outputLineIncrement);
267         if (fileIndex != lastFileID)
268             li.setLineFileID(fileIndex);
269         lastFileID = fileIndex;
270 
271         // save it
272         lineData.add(li);
273     }

285     /**
286      * @return  the given stratum as a String:  a StratumSection,
287      * followed by at least one FileSection and at least one LineSection.
288      */
289     public String getString() {
290         // check state and initialize buffer
291         if (fileNameList.size() == 0 || lineData.size() == 0)
292             return null;
293 
294         StringBuffer out = new StringBuffer();
295 
296         // print StratumSection
297         out.append("*S " + stratumName + "\n");
298 
299         // print FileSection
300         out.append("*F\n");
301         int bound = fileNameList.size();
302         for (int i = 0; i < bound; i++) {
303             if (filePathList.get(i) != null) {
304                 out.append("+ " + i + " " + fileNameList.get(i) + "\n");
305                 // Source paths must be relative, not absolute, so we
306                 // remove the leading "/", if one exists.
307                 String filePath = (String)filePathList.get(i);
308                 if (filePath.startsWith("/")) {
309                     filePath = filePath.substring(1);
310                 }
311                 out.append(filePath + "\n");
312             } else {
313                 out.append(i + " " + fileNameList.get(i) + "\n");
314             }
315         }
316 
317         // print LineSection
318         out.append("*L\n");
319         bound = lineData.size();
320         for (int i = 0; i < bound; i++) {
321             LineInfo li = (LineInfo)lineData.get(i);
322             out.append(li.getString());
323         }
324 
325         return out.toString();
326     }
327 
328     public String toString() {
329         return getString();
330     }
