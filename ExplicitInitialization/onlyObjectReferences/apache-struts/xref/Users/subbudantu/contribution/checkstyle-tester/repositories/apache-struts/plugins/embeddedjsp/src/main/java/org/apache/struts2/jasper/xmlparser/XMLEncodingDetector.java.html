86      /**
87       * <p>
88       * Autodetects the encoding of the XML document supplied by the given
89       * input stream.
90       * </p>
91       *
92       * <p>
93       * Encoding autodetection is done according to the XML 1.0 specification,
94       * Appendix F.1: Detection Without External Encoding Information.
95       * </p>
96       *
97       * @param fname   file name
98       * @param jarFile jar file
99       * @param ctxt    compilation context
100      * @param err     error dispatcher
101      * @return Two-element array, where the first element (of type
102      * java.lang.String) contains the name of the (auto)detected encoding, and
103      * the second element (of type java.lang.Boolean) specifies whether the
104      * encoding was specified using the 'encoding' attribute of an XML prolog
105      * (TRUE) or autodetected (FALSE).
106      * @throws IOException     in case of IO errors
107      * @throws JasperException in case of Jasper errors
108      */
109     public static Object[] getEncoding(String fname, JarFile jarFile,
110                                        JspCompilationContext ctxt,
111                                        ErrorDispatcher err)
112             throws IOException, JasperException {
113         InputStream inStream = JspUtil.getInputStream(fname, jarFile, ctxt,
114                 err);
115         XMLEncodingDetector detector = new XMLEncodingDetector();
116         Object[] ret = detector.getEncoding(inStream, err);
117         inStream.close();
118 
119         return ret;
120     }

122     private Object[] getEncoding(InputStream in, ErrorDispatcher err)
123             throws IOException, JasperException {
124         this.stream = in;
125         this.err = err;
126         createInitialReader();
127         scanXMLDecl();
128 
129         return new Object[]{this.encoding,
130                 Boolean.valueOf(this.isEncodingSetInProlog),
131                 Boolean.valueOf(this.isBomPresent),
132                 Integer.valueOf(this.skip)};
133     }

141     private void createInitialReader() throws IOException, JasperException {
142 
143         // wrap this stream in RewindableInputStream
144         stream = new RewindableInputStream(stream);
145 
146         // perform auto-detect of encoding if necessary
147         if (encoding == null) {
148             // read first four bytes and determine encoding
149             final byte[] b4 = new byte[4];
150             int count = 0;
151             for (; count < 4; count++) {
152                 b4[count] = (byte) stream.read();
153             }
154             if (count == 4) {
155                 Object[] encodingDesc = getEncodingName(b4, count);
156                 encoding = (String) (encodingDesc[0]);
157                 isBigEndian = (Boolean) (encodingDesc[1]);
158 
159                 if (encodingDesc.length > 3) {
160                     isBomPresent = (Boolean) (encodingDesc[2]);
161                     skip = (Integer) (encodingDesc[3]);
162                 } else {
163                     isBomPresent = true;
164                     skip = (Integer) (encodingDesc[2]);
165                 }
166 
167                 stream.reset();
168                 // Special case UTF-8 files with BOM created by Microsoft
169                 // tools. It's more efficient to consume the BOM than make
170                 // the reader perform extra checks. -Ac
171                 if (count > 2 && encoding.equals("UTF-8")) {
172                     int b0 = b4[0] & 0xFF;
173                     int b1 = b4[1] & 0xFF;
174                     int b2 = b4[2] & 0xFF;
175                     if (b0 == 0xEF && b1 == 0xBB && b2 == 0xBF) {
176                         // ignore first three bytes...
177                         stream.skip(3);
178                     }
179                 }
180                 reader = createReader(stream, encoding, isBigEndian);
181             } else {
182                 reader = createReader(stream, encoding, isBigEndian);
183             }
184         }
185     }

190     /**
191      * Creates a reader capable of reading the given input stream in
192      * the specified encoding.
193      *
194      * @param inputStream The input stream.
195      * @param encoding    The encoding name that the input stream is
196      *                    encoded using. If the user has specified that
197      *                    Java encoding names are allowed, then the
198      *                    encoding name may be a Java encoding name;
199      *                    otherwise, it is an ianaEncoding name.
200      * @param isBigEndian For encodings (like uCS-4), whose names cannot
201      *                    specify a byte order, this tells whether the order
202      *                    is bigEndian. null means unknown or not relevant.
203      * @return Returns a reader.
204      * @throws IOException     in case of IO errors
205      * @throws JasperException in case of Jasper errors
206      */
207     private Reader createReader(InputStream inputStream, String encoding,
208                                 Boolean isBigEndian)
209             throws IOException, JasperException {
210 
211         // normalize encoding name
212         if (encoding == null) {
213             encoding = "UTF-8";
214         }
215 
216         // try to use an optimized reader
217         String ENCODING = encoding.toUpperCase(Locale.ENGLISH);
218         if (ENCODING.equals("UTF-8")) {
219             return new UTF8Reader(inputStream, fBufferSize);
220         }
221         if (ENCODING.equals("US-ASCII")) {
222             return new ASCIIReader(inputStream, fBufferSize);
223         }
224         if (ENCODING.equals("ISO-10646-UCS-4")) {
225             if (isBigEndian != null) {
226                 boolean isBE = isBigEndian.booleanValue();
227                 if (isBE) {
228                     return new UCSReader(inputStream, UCSReader.UCS4BE);
229                 } else {
230                     return new UCSReader(inputStream, UCSReader.UCS4LE);
231                 }
232             } else {
233                 err.jspError("jsp.error.xml.encodingByteOrderUnsupported",
234                         encoding);
235             }
236         }
237         if (ENCODING.equals("ISO-10646-UCS-2")) {
238             if (isBigEndian != null) { // sould never happen with this encoding...
239                 boolean isBE = isBigEndian.booleanValue();
240                 if (isBE) {
241                     return new UCSReader(inputStream, UCSReader.UCS2BE);
242                 } else {
243                     return new UCSReader(inputStream, UCSReader.UCS2LE);
244                 }
245             } else {
246                 err.jspError("jsp.error.xml.encodingByteOrderUnsupported",
247                         encoding);
248             }
249         }
250 
251         // check for valid name
252         boolean validIANA = XMLChar.isValidIANAEncoding(encoding);
253         boolean validJava = XMLChar.isValidJavaEncoding(encoding);
254         if (!validIANA || (fAllowJavaEncodings && !validJava)) {
255             err.jspError("jsp.error.xml.encodingDeclInvalid", encoding);
256             // NOTE: AndyH suggested that, on failure, we use ISO Latin 1
257             //       because every byte is a valid ISO Latin 1 character.
258             //       It may not translate correctly but if we failed on
259             //       the encoding anyway, then we're expecting the content
260             //       of the document to be bad. This will just prevent an
261             //       invalid UTF-8 sequence to be detected. This is only
262             //       important when continue-after-fatal-error is turned
263             //       on. -Ac
264             encoding = "ISO-8859-1";
265         }
266 
267         // try to use a Java reader
268         String javaEncoding = EncodingMap.getIANA2JavaMapping(ENCODING);
269         if (javaEncoding == null) {
270             if (fAllowJavaEncodings) {
271                 javaEncoding = encoding;
272             } else {
273                 err.jspError("jsp.error.xml.encodingDeclInvalid", encoding);
274                 // see comment above.
275                 javaEncoding = "ISO8859_1";
276             }
277         }
278         return new InputStreamReader(inputStream, javaEncoding);
279 
280     } // createReader(InputStream,String, Boolean): Reader

285     /**
286      * Returns the IANA encoding name that is auto-detected from
287      * the bytes specified, with the endian-ness of that encoding where
288      * appropriate.
289      *
290      * @param b4    The first four bytes of the input.
291      * @param count The number of bytes actually read.
292      * @return a 2-element array:  the first element, an IANA-encoding string,
293      * the second element a Boolean which is true iff the document is big
294      * endian, false if it's little-endian, and null if the distinction isn't
295      * relevant.
296      */
297     private Object[] getEncodingName(byte[] b4, int count) {
298 
299         if (count < 2) {
300             return new Object[]{"UTF-8", null, Boolean.FALSE, Integer.valueOf(0)};
301         }
302 
303         // UTF-16, with BOM
304         int b0 = b4[0] & 0xFF;
305         int b1 = b4[1] & 0xFF;
306         if (b0 == 0xFE && b1 == 0xFF) {
307             // UTF-16, big-endian
308             return new Object[]{"UTF-16BE", Boolean.TRUE, Integer.valueOf(2)};
309         }
310         if (b0 == 0xFF && b1 == 0xFE) {
311             // UTF-16, little-endian
312             return new Object[]{"UTF-16LE", Boolean.FALSE, Integer.valueOf(2)};
313         }
314 
315         // default to UTF-8 if we don't have enough bytes to make a
316         // good determination of the encoding
317         if (count < 3) {
318             return new Object[]{"UTF-8", null, Boolean.FALSE, Integer.valueOf(0)};
319         }
320 
321         // UTF-8 with a BOM
322         int b2 = b4[2] & 0xFF;
323         if (b0 == 0xEF && b1 == 0xBB && b2 == 0xBF) {
324             return new Object[]{"UTF-8", null, Integer.valueOf(3)};
325         }
326 
327         // default to UTF-8 if we don't have enough bytes to make a
328         // good determination of the encoding
329         if (count < 4) {
330             return new Object[]{"UTF-8", null, Integer.valueOf(0)};
331         }
332 
333         // other encodings
334         int b3 = b4[3] & 0xFF;
335         if (b0 == 0x00 && b1 == 0x00 && b2 == 0x00 && b3 == 0x3C) {
336             // UCS-4, big endian (1234)
337             return new Object[]{"ISO-10646-UCS-4", new Boolean(true), Integer.valueOf(4)};
338         }
339         if (b0 == 0x3C && b1 == 0x00 && b2 == 0x00 && b3 == 0x00) {
340             // UCS-4, little endian (4321)
341             return new Object[]{"ISO-10646-UCS-4", new Boolean(false), Integer.valueOf(4)};
342         }
343         if (b0 == 0x00 && b1 == 0x00 && b2 == 0x3C && b3 == 0x00) {
344             // UCS-4, unusual octet order (2143)
345             // REVISIT: What should this be?
346             return new Object[]{"ISO-10646-UCS-4", null, Integer.valueOf(4)};
347         }
348         if (b0 == 0x00 && b1 == 0x3C && b2 == 0x00 && b3 == 0x00) {
349             // UCS-4, unusual octect order (3412)
350             // REVISIT: What should this be?
351             return new Object[]{"ISO-10646-UCS-4", null, Integer.valueOf(4)};
352         }
353         if (b0 == 0x00 && b1 == 0x3C && b2 == 0x00 && b3 == 0x3F) {
354             // UTF-16, big-endian, no BOM
355             // (or could turn out to be UCS-2...
356             // REVISIT: What should this be?
357             return new Object[]{"UTF-16BE", new Boolean(true), Integer.valueOf(4)};
358         }
359         if (b0 == 0x3C && b1 == 0x00 && b2 == 0x3F && b3 == 0x00) {
360             // UTF-16, little-endian, no BOM
361             // (or could turn out to be UCS-2...
362             return new Object[]{"UTF-16LE", new Boolean(false), Integer.valueOf(4)};
363         }
364         if (b0 == 0x4C && b1 == 0x6F && b2 == 0xA7 && b3 == 0x94) {
365             // EBCDIC
366             // a la xerces1, return CP037 instead of EBCDIC here
367             return new Object[]{"CP037", null, Integer.valueOf(4)};
368         }
369 
370         // default encoding
371         return new Object[]{"UTF-8", null, Boolean.FALSE, Integer.valueOf(0)};
372 
373     }

396 
397         // load more characters, if needed
398         if (fCurrentEntity.position == fCurrentEntity.count) {

400         }
401 
402         // peek at character

404 
405         // return peeked character
406         if (fCurrentEntity.isExternal()) {
407             return c != '\r' ? c : '\n';
408         } else {
409             return c;

417     /**
418      * @return the next character on the input.
419      * <p>
420      * <strong>Note:</strong> The character is consumed.
421      * </p>
422      * @throws IOException Thrown if i/o error occurs.
423      */
424     public int scanChar() throws IOException {
425 
426         // load more characters, if needed
427         if (fCurrentEntity.position == fCurrentEntity.count) {
428             load(0, true);
429         }
430 
431         // scan character
432         int c = fCurrentEntity.ch[fCurrentEntity.position++];
433         boolean external = false;
434         if (c == '\n' ||
435                 (c == '\r' && (external = fCurrentEntity.isExternal()))) {
436             fCurrentEntity.lineNumber++;
437             fCurrentEntity.columnNumber = 1;
438             if (fCurrentEntity.position == fCurrentEntity.count) {
439                 fCurrentEntity.ch[0] = (char) c;
440                 load(1, false);
441             }
442             if (c == '\r' && external) {
443                 if (fCurrentEntity.ch[fCurrentEntity.position++] != '\n') {
444                     fCurrentEntity.position--;
445                 }
446                 c = '\n';
447             }
448         }
449 
450         // return character that was scanned
451         fCurrentEntity.columnNumber++;
452         return c;
453 
454     }

459     /**
460      * @return a string matching the Name production appearing immediately
461      * on the input as a symbol, or null if no Name string is present.
462      * <p>
463      * <strong>Note:</strong> The Name characters are consumed.
464      * </p>
465      * <p>
466      * <strong>Note:</strong> The string returned must be a symbol. The
467      * SymbolTable can be used for this purpose.
468      * </p>
469      * @throws IOException Thrown if i/o error occurs.
470      * @see SymbolTable
471      * @see XMLChar#isName
472      * @see XMLChar#isNameStart
473      */
474     public String scanName() throws IOException {
475 
476         // load more characters, if needed
477         if (fCurrentEntity.position == fCurrentEntity.count) {
478             load(0, true);
479         }
480 
481         // scan name
482         int offset = fCurrentEntity.position;
483         if (XMLChar.isNameStart(fCurrentEntity.ch[offset])) {
484             if (++fCurrentEntity.position == fCurrentEntity.count) {
485                 fCurrentEntity.ch[0] = fCurrentEntity.ch[offset];
486                 offset = 0;
487                 if (load(1, false)) {
488                     fCurrentEntity.columnNumber++;
489                     String symbol = fSymbolTable.addSymbol(fCurrentEntity.ch,
490                             0, 1);
491                     return symbol;
492                 }
493             }
494             while (XMLChar.isName(fCurrentEntity.ch[fCurrentEntity.position])) {
495                 if (++fCurrentEntity.position == fCurrentEntity.count) {
496                     int length = fCurrentEntity.position - offset;
497                     if (length == fBufferSize) {
498                         // bad luck we have to resize our buffer
499                         char[] tmp = new char[fBufferSize * 2];
500                         System.arraycopy(fCurrentEntity.ch, offset,
501                                 tmp, 0, length);
502                         fCurrentEntity.ch = tmp;
503                         fBufferSize *= 2;
504                     } else {
505                         System.arraycopy(fCurrentEntity.ch, offset,
506                                 fCurrentEntity.ch, 0, length);
507                     }
508                     offset = 0;
509                     if (load(length, false)) {
510                         break;
511                     }
512                 }
513             }
514         }
515         int length = fCurrentEntity.position - offset;
516         fCurrentEntity.columnNumber += length;
517 
518         // return name
519         String symbol = null;
520         if (length > 0) {
521             symbol = fSymbolTable.addSymbol(fCurrentEntity.ch, offset, length);
522         }
523         return symbol;
524 
525     }

530     /**
531      * <p>
532      * Scans a range of attribute value data, setting the fields of the
533      * XMLString structure, appropriately.
534      * </p>
535      *
536      * <p>
537      * <strong>Note:</strong> The characters are consumed.
538      * </p>
539      *
540      * <p>
541      * <strong>Note:</strong> This method does not guarantee to return
542      * the longest run of attribute value data. This method may return
543      * before the quote character due to reaching the end of the input
544      * buffer or any other reason.
545      * </p>
546      *
547      * <p>
548      * <strong>Note:</strong> The fields contained in the XMLString
549      * structure are not guaranteed to remain valid upon subsequent calls
550      * to the entity scanner. Therefore, the caller is responsible for
551      * immediately using the returned character data or making a copy of
552      * the character data.
553      * </p>
554      *
555      * @param quote   The quote character that signifies the end of the
556      *                attribute value data.
557      * @param content The content structure to fill.
558      * @return Returns the next character on the input, if known. This
559      * value may be -1 but this does <em>note</em> designate
560      * end of file.
561      * @throws IOException Thrown if i/o error occurs.
562      */
563     public int scanLiteral(int quote, XMLString content)
564             throws IOException {
565 
566         // load more characters, if needed
567         if (fCurrentEntity.position == fCurrentEntity.count) {
568             load(0, true);
569         } else if (fCurrentEntity.position == fCurrentEntity.count - 1) {
570             fCurrentEntity.ch[0] = fCurrentEntity.ch[fCurrentEntity.count - 1];
571             load(1, false);
572             fCurrentEntity.position = 0;
573         }
574 
575         // normalize newlines
576         int offset = fCurrentEntity.position;
577         int c = fCurrentEntity.ch[offset];
578         int newlines = 0;
579         boolean external = fCurrentEntity.isExternal();
580         if (c == '\n' || (c == '\r' && external)) {
581             do {
582                 c = fCurrentEntity.ch[fCurrentEntity.position++];
583                 if (c == '\r' && external) {
584                     newlines++;
585                     fCurrentEntity.lineNumber++;
586                     fCurrentEntity.columnNumber = 1;
587                     if (fCurrentEntity.position == fCurrentEntity.count) {
588                         offset = 0;
589                         fCurrentEntity.position = newlines;
590                         if (load(newlines, false)) {
591                             break;
592                         }
593                     }
594                     if (fCurrentEntity.ch[fCurrentEntity.position] == '\n') {
595                         fCurrentEntity.position++;
596                         offset++;
597                     }
598                     /*** NEWLINE NORMALIZATION ***/
599                     else {
600                         newlines++;
601                     }
602                     /***/
603                 } else if (c == '\n') {
604                     newlines++;
605                     fCurrentEntity.lineNumber++;
606                     fCurrentEntity.columnNumber = 1;
607                     if (fCurrentEntity.position == fCurrentEntity.count) {
608                         offset = 0;
609                         fCurrentEntity.position = newlines;
610                         if (load(newlines, false)) {
611                             break;
612                         }
613                     }
614                     /*** NEWLINE NORMALIZATION ***
615                      if (fCurrentEntity.ch[fCurrentEntity.position] == '\r'
616                      && external) {
617                      fCurrentEntity.position++;
618                      offset++;
619                      }
620                      /***/
621                 } else {
622                     fCurrentEntity.position--;
623                     break;
624                 }
625             } while (fCurrentEntity.position < fCurrentEntity.count - 1);
626             for (int i = offset; i < fCurrentEntity.position; i++) {
627                 fCurrentEntity.ch[i] = '\n';
628             }
629             int length = fCurrentEntity.position - offset;
630             if (fCurrentEntity.position == fCurrentEntity.count - 1) {
631                 content.setValues(fCurrentEntity.ch, offset, length);
632                 return -1;
633             }
634         }
635 
636         // scan literal value
637         while (fCurrentEntity.position < fCurrentEntity.count) {
638             c = fCurrentEntity.ch[fCurrentEntity.position++];
639             if ((c == quote &&
640                     (!fCurrentEntity.literal || external))
641                     || c == '%' || !XMLChar.isContent(c)) {
642                 fCurrentEntity.position--;
643                 break;
644             }
645         }
646         int length = fCurrentEntity.position - offset;
647         fCurrentEntity.columnNumber += length - newlines;
648         content.setValues(fCurrentEntity.ch, offset, length);
649 
650         // return next character
651         if (fCurrentEntity.position != fCurrentEntity.count) {
652             c = fCurrentEntity.ch[fCurrentEntity.position];
653             // NOTE: We don't want to accidentally signal the
654             //       end of the literal if we're expanding an
655             //       entity appearing in the literal. -Ac
656             if (c == quote && fCurrentEntity.literal) {
657                 c = -1;
658             }
659         } else {
660             c = -1;
661         }
662         return c;
663 
664     }

666     /**
667      * Scans a range of character data up to the specified delimiter,
668      * setting the fields of the XMLString structure, appropriately.
669      * <p>
670      * <strong>Note:</strong> The characters are consumed.
671      * </p>
672      * <p>
673      * <strong>Note:</strong> This assumes that the internal buffer is
674      * at least the same size, or bigger, than the length of the delimiter
675      * and that the delimiter contains at least one character.
676      * </p>
677      * <p>
678      * <strong>Note:</strong> This method does not guarantee to return
679      * the longest run of character data. This method may return before
680      * the delimiter due to reaching the end of the input buffer or any
681      * other reason.
682      * </p>
683      * <p>
684      * <strong>Note:</strong> The fields contained in the XMLString
685      * structure are not guaranteed to remain valid upon subsequent calls
686      * to the entity scanner. Therefore, the caller is responsible for
687      * immediately using the returned character data or making a copy of
688      * the character data.
689      * </p>
690      *
691      * @param delimiter The string that signifies the end of the character
692      *                  data to be scanned.
693      * @param buffer    The data structure to fill.
694      * @return Returns true if there is more data to scan, false otherwise.
695      * @throws IOException Thrown if i/o error occurs.
696      */
697     public boolean scanData(String delimiter, XMLStringBuffer buffer)
698             throws IOException {
699 
700         boolean done = false;
701         int delimLen = delimiter.length();
702         char charAt0 = delimiter.charAt(0);
703         boolean external = fCurrentEntity.isExternal();
704         do {
705 
706             // load more characters, if needed
707 
708             if (fCurrentEntity.position == fCurrentEntity.count) {
709                 load(0, true);
710             } else if (fCurrentEntity.position >= fCurrentEntity.count - delimLen) {
711                 System.arraycopy(fCurrentEntity.ch, fCurrentEntity.position,
712                         fCurrentEntity.ch, 0, fCurrentEntity.count - fCurrentEntity.position);
713                 load(fCurrentEntity.count - fCurrentEntity.position, false);
714                 fCurrentEntity.position = 0;
715             }
716             if (fCurrentEntity.position >= fCurrentEntity.count - delimLen) {
717                 // something must be wrong with the input: e.g., file ends an
718                 // unterminated comment
719                 int length = fCurrentEntity.count - fCurrentEntity.position;
720                 buffer.append(fCurrentEntity.ch, fCurrentEntity.position,
721                         length);
722                 fCurrentEntity.columnNumber += fCurrentEntity.count;
723                 fCurrentEntity.position = fCurrentEntity.count;
724                 load(0, true);
725                 return false;
726             }
727 
728             // normalize newlines
729             int offset = fCurrentEntity.position;
730             int c = fCurrentEntity.ch[offset];
731             int newlines = 0;
732             if (c == '\n' || (c == '\r' && external)) {
733                 do {
734                     c = fCurrentEntity.ch[fCurrentEntity.position++];
735                     if (c == '\r' && external) {
736                         newlines++;
737                         fCurrentEntity.lineNumber++;
738                         fCurrentEntity.columnNumber = 1;
739                         if (fCurrentEntity.position == fCurrentEntity.count) {
740                             offset = 0;
741                             fCurrentEntity.position = newlines;
742                             if (load(newlines, false)) {
743                                 break;
744                             }
745                         }
746                         if (fCurrentEntity.ch[fCurrentEntity.position] == '\n') {
747                             fCurrentEntity.position++;
748                             offset++;
749                         }
750                         /*** NEWLINE NORMALIZATION ***/
751                         else {
752                             newlines++;
753                         }
754                     } else if (c == '\n') {
755                         newlines++;
756                         fCurrentEntity.lineNumber++;
757                         fCurrentEntity.columnNumber = 1;
758                         if (fCurrentEntity.position == fCurrentEntity.count) {
759                             offset = 0;
760                             fCurrentEntity.position = newlines;
761                             fCurrentEntity.count = newlines;
762                             if (load(newlines, false)) {
763                                 break;
764                             }
765                         }
766                     } else {
767                         fCurrentEntity.position--;
768                         break;
769                     }
770                 } while (fCurrentEntity.position < fCurrentEntity.count - 1);
771                 for (int i = offset; i < fCurrentEntity.position; i++) {
772                     fCurrentEntity.ch[i] = '\n';
773                 }
774                 int length = fCurrentEntity.position - offset;
775                 if (fCurrentEntity.position == fCurrentEntity.count - 1) {
776                     buffer.append(fCurrentEntity.ch, offset, length);
777                     return true;
778                 }
779             }
780 
781             // iterate over buffer looking for delimiter
782             OUTER:
783             while (fCurrentEntity.position < fCurrentEntity.count) {
784                 c = fCurrentEntity.ch[fCurrentEntity.position++];
785                 if (c == charAt0) {
786                     // looks like we just hit the delimiter
787                     int delimOffset = fCurrentEntity.position - 1;
788                     for (int i = 1; i < delimLen; i++) {
789                         if (fCurrentEntity.position == fCurrentEntity.count) {
790                             fCurrentEntity.position -= i;
791                             break OUTER;
792                         }
793                         c = fCurrentEntity.ch[fCurrentEntity.position++];
794                         if (delimiter.charAt(i) != c) {
795                             fCurrentEntity.position--;
796                             break;
797                         }
798                     }
799                     if (fCurrentEntity.position == delimOffset + delimLen) {
800                         done = true;
801                         break;
802                     }
803                 } else if (c == '\n' || (external && c == '\r')) {
804                     fCurrentEntity.position--;
805                     break;
806                 } else if (XMLChar.isInvalid(c)) {
807                     fCurrentEntity.position--;
808                     int length = fCurrentEntity.position - offset;
809                     fCurrentEntity.columnNumber += length - newlines;
810                     buffer.append(fCurrentEntity.ch, offset, length);
811                     return true;
812                 }
813             }
814             int length = fCurrentEntity.position - offset;
815             fCurrentEntity.columnNumber += length - newlines;
816             if (done) {
817                 length -= delimLen;
818             }
819             buffer.append(fCurrentEntity.ch, offset, length);
820 
821             // return true if string was skipped
822         } while (!done);
823         return !done;

1052     /**
1053      * <p>
1054      * This class wraps the byte inputstreams we're presented with.
1055      * We need it because java.io.InputStreams don't provide
1056      * functionality to reread processed bytes, and they have a habit
1057      * of reading more than one character when you call their read()
1058      * methods.  This means that, once we discover the true (declared)
1059      * encoding of a document, we can neither backtrack to read the
1060      * whole doc again nor start reading where we are with a new
1061      * reader.
1062      * </p>
1063      * <p/>
1064      * <p>
1065      * This class allows rewinding an inputStream by allowing a mark
1066      * to be set, and the stream reset to that position.  <strong>The
1067      * class assumes that it needs to read one character per
1068      * invocation when it's read() method is inovked, but uses the
1069      * underlying InputStream's read(char[], offset length) method--it
1070      * won't buffer data read this way!</strong>
1071      * </p>
1072      *
1073      * @author Neil Graham, IBM
1074      * @author Glenn Marcy, IBM
1075      */
1076     private final class RewindableInputStream extends InputStream {
1077 
1078         private InputStream fInputStream;
1079         private byte[] fData;
1080         private int fStartOffset;
1081         private int fEndOffset;
1082         private int fOffset;
1083         private int fLength;
1084         private int fMark;
1085 
1086         public RewindableInputStream(InputStream is) {
1087             fData = new byte[DEFAULT_XMLDECL_BUFFER_SIZE];
1088             fInputStream = is;
1089             fStartOffset = 0;
1090             fEndOffset = -1;
1091             fOffset = 0;
1092             fLength = 0;
1093             fMark = 0;
1094         }
1095 
1096         public void setStartOffset(int offset) {
1097             fStartOffset = offset;
1098         }
1099 
1100         public void rewind() {
1101             fOffset = fStartOffset;
1102         }
1103 
1104         public int read() throws IOException {
1105             int b = 0;
1106             if (fOffset < fLength) {
1107                 return fData[fOffset++] & 0xff;
1108             }
1109             if (fOffset == fEndOffset) {
1110                 return -1;
1111             }
1112             if (fOffset == fData.length) {
1113                 byte[] newData = new byte[fOffset << 1];
1114                 System.arraycopy(fData, 0, newData, 0, fOffset);
1115                 fData = newData;
1116             }
1117             b = fInputStream.read();
1118             if (b == -1) {
1119                 fEndOffset = fOffset;
1120                 return -1;
1121             }
1122             fData[fLength++] = (byte) b;
1123             fOffset++;
1124             return b & 0xff;
1125         }
1126 
1127         public int read(byte[] b, int off, int len) throws IOException {
1128             int bytesLeft = fLength - fOffset;
1129             if (bytesLeft == 0) {
1130                 if (fOffset == fEndOffset) {
1131                     return -1;
1132                 }
1133                 // better get some more for the voracious reader...
1134                 if (fCurrentEntity.mayReadChunks) {
1135                     return fInputStream.read(b, off, len);
1136                 }
1137                 int returnedVal = read();
1138                 if (returnedVal == -1) {
1139                     fEndOffset = fOffset;
1140                     return -1;
1141                 }
1142                 b[off] = (byte) returnedVal;
1143                 return 1;
1144             }
1145             if (len < bytesLeft) {
1146                 if (len <= 0) {
1147                     return 0;
1148                 }
1149             } else {
1150                 len = bytesLeft;
1151             }
1152             if (b != null) {
1153                 System.arraycopy(fData, fOffset, b, off, len);
1154             }
1155             fOffset += len;
1156             return len;
1157         }
1158 
1159         public long skip(long n)
1160                 throws IOException {
1161             int bytesLeft;
1162             if (n <= 0) {
1163                 return 0;
1164             }
1165             bytesLeft = fLength - fOffset;
1166             if (bytesLeft == 0) {
1167                 if (fOffset == fEndOffset) {
1168                     return 0;
1169                 }
1170                 return fInputStream.skip(n);
1171             }
1172             if (n <= bytesLeft) {
1173                 fOffset += n;
1174                 return n;
1175             }
1176             fOffset += bytesLeft;
1177             if (fOffset == fEndOffset) {
1178                 return bytesLeft;
1179             }
1180             n -= bytesLeft;
1181         /*
1182          * In a manner of speaking, when this class isn't permitting more
1183 	     * than one byte at a time to be read, it is "blocking".  The
1184 	     * available() method should indicate how much can be read without
1185 	     * blocking, so while we're in this mode, it should only indicate
1186 	     * that bytes in its buffer are available; otherwise, the result of
1187 	     * available() on the underlying InputStream is appropriate.
1188 	     */
1189             return fInputStream.skip(n) + bytesLeft;

1551      * @param target The PI target

1553      */
1554     private void scanPIData(String target, XMLString data)
1555             throws IOException, JasperException {
1556 
1557         // check target
1558         if (target.length() == 3) {
1559             char c0 = Character.toLowerCase(target.charAt(0));
1560             char c1 = Character.toLowerCase(target.charAt(1));
1561             char c2 = Character.toLowerCase(target.charAt(2));

1564             }

1568         if (!skipSpaces()) {
1569             if (skipString("?>")) {

1571                 data.clear();

1573             } else {

1602     /*
1603      * Scans surrogates and append them to the specified buffer.
1604      * <p>
1605      * <strong>Note:</strong> This assumes the current char has already been
1606      * identified as a high surrogate.
1607      * </p>
1608      *
1609      * @param buf The StringBuffer to append the read surrogates to.
1610      * @returns True if it succeeded.
1611      */
1612     private boolean scanSurrogates(XMLStringBuffer buf)
1613             throws IOException, JasperException {
1614 
1615         int high = scanChar();
1616         int low = peekChar();
1617         if (!XMLChar.isLowSurrogate(low)) {
1618             err.jspError("jsp.error.xml.invalidCharInContent",
1619                     Integer.toString(high, 16));
1620             return false;
1621         }
1622         scanChar();
1623 
1624         // convert surrogates to supplemental character
1625         int c = XMLChar.supplemental((char) high, (char) low);
1626 
1627         // supplemental character must be a valid XML character
1628         if (!XMLChar.isValid(c)) {
1629             err.jspError("jsp.error.xml.invalidCharInContent",
1630                     Integer.toString(c, 16));
1631             return false;
1632         }
1633 
1634         // fill in the buffer
1635         buf.append((char) high);
1636         buf.append((char) low);
1637 
1638         return true;
1639 
1640     }
