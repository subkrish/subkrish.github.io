51  /**
52   * Struts package provider that starts the OSGi container and deelgates package loading
53   */
54  public class OsgiConfigurationProvider implements PackageProvider, BundleListener {
55  
56      private static final Logger LOG = LogManager.getLogger(OsgiConfigurationProvider.class);
57  
58      private Configuration configuration;
59      private ObjectFactory objectFactory;
60      private FileManagerFactory fileManagerFactory;
61  
62      private OsgiHost osgiHost;
63      private BundleContext bundleContext;
64      private BundleAccessor bundleAccessor;
65      private boolean bundlesChanged = false;
66      private ServletContext servletContext;
67  
68      public void init(Configuration configuration) throws ConfigurationException {
69          osgiHost = (OsgiHost) servletContext.getAttribute(StrutsOsgiListener.OSGI_HOST);
70          bundleContext = osgiHost.getBundleContext();
71          bundleAccessor.setBundleContext(bundleContext);
72          bundleAccessor.setOsgiHost(osgiHost);
73          this.configuration = configuration;
74  
75          //this class loader interface can be used by other plugins to lookup resources
76          //from the bundles. A temporary class loader interface is set during other configuration
77          //loading as well
78          servletContext.setAttribute(ClassLoaderInterface.CLASS_LOADER_INTERFACE, new BundleClassLoaderInterface());
79      }
80  
81      public synchronized void loadPackages() throws ConfigurationException {
82          if (LOG.isTraceEnabled())
83              LOG.trace("Loading packages from XML and Convention on startup");                
84  
85          //init action context
86          ActionContext ctx = ActionContext.getContext();
87          if (ctx == null) {
88              ctx = createActionContext();
89              ActionContext.setContext(ctx);
90          }
91  
92          Set<String> bundleNames = new HashSet<>();
93  
94          //iterate over the bundles and load packages from them
95          for (Bundle bundle : osgiHost.getBundles().values()) {
96              String bundleName = bundle.getSymbolicName();
97              if (shouldProcessBundle(bundle) && !bundleNames.contains(bundleName)) {
98                  bundleNames.add(bundleName);
99                  //load XML and Convention config
100                 loadConfigFromBundle(bundle);
101             }
102         }
103 
104         bundlesChanged = false;
105         bundleContext.addBundleListener(this);
106     }
107 
108     protected ActionContext createActionContext() {
109         return new ActionContext(new HashMap<String, Object>());
110     }
111 
112     /**
113      * Loads XML config as well as Convention config from a bundle
114      * Limitation: Constants and Beans are ignored on XML config
115      *
116      * @param bundle the bundle
117      */
118     protected void loadConfigFromBundle(Bundle bundle) {
119         String bundleName = bundle.getSymbolicName();
120         LOG.debug("Loading packages from bundle [{}]", bundleName);
121 
122         //init action context
123         ActionContext ctx = ActionContext.getContext();
124         if (ctx == null) {
125             ctx = createActionContext();
126             ActionContext.setContext(ctx);
127         }
128 
129         try {
130             //the Convention plugin will use BundleClassLoaderInterface from the ActionContext to find resources
131             //and load classes
132             ctx.put(ClassLoaderInterface.CLASS_LOADER_INTERFACE, new BundleClassLoaderInterface());
133             ctx.put(BundleAccessor.CURRENT_BUNDLE_NAME, bundleName);
134 
135             LOG.trace("Loading XML config from bundle [{}]", bundleName);
136 
137             //XML config
138             PackageLoader loader = new BundlePackageLoader();
139             for (PackageConfig pkg : loader.loadPackages(bundle, bundleContext, objectFactory, fileManagerFactory, configuration.getPackageConfigs())) {
140                 configuration.addPackageConfig(pkg.getName(), pkg);
141                 bundleAccessor.addPackageFromBundle(bundle, pkg.getName());
142             }
143 
144             //Convention
145             //get the existing packages before reloading the provider (se we can figure out what are the new packages)
146             Set<String> packagesBeforeLoading = new HashSet<>(configuration.getPackageConfigNames());
147 
148             PackageProvider conventionPackageProvider = configuration.getContainer().getInstance(PackageProvider.class, "convention.packageProvider");
149             if (conventionPackageProvider != null) {
150                 LOG.trace("Loading Convention config from bundle [{}]", bundleName);
151                 conventionPackageProvider.loadPackages();
152             }
153 
154             Set<String> packagesAfterLoading = new HashSet<String>(configuration.getPackageConfigNames());
155             packagesAfterLoading.removeAll(packagesBeforeLoading);
156             if (!packagesAfterLoading.isEmpty()) {
157                 //add the new packages to the map of bundle -> package
158                 for (String packageName : packagesAfterLoading)
159                     bundleAccessor.addPackageFromBundle(bundle, packageName);
160             }
161 
162             if (this.configuration.getRuntimeConfiguration() != null) {
163                 //if there is a runtime config, it meas that this method was called froma bundle start event
164                 //instead of the initial load, in that case, reload the config
165                 this.configuration.rebuildRuntimeConfiguration();
166             }
167         } finally {
168             ctx.put(BundleAccessor.CURRENT_BUNDLE_NAME, null);
169             ctx.put(ClassLoaderInterface.CLASS_LOADER_INTERFACE, null);
170         }
171     }
172 
173     /**
174      * Checks for "Struts2-Enabled" header in the bundle
175      *
176      * @param bundle the bundle
177      *
178      * @return true is struts2 enabled
179      */
180     protected boolean shouldProcessBundle(Bundle bundle) {
181         // Cast to String is required on JDK7
182         String strutsEnabled = (String) bundle.getHeaders().get(OsgiHost.OSGI_HEADER_STRUTS_ENABLED);
183 
184         return "true".equalsIgnoreCase(strutsEnabled);
185     }
186 
187     public synchronized boolean needsReload() {
188         return bundlesChanged;
189     }

192     public void setObjectFactory(ObjectFactory factory) {
193         this.objectFactory = factory;
194     }

197     public void setBundleAccessor(BundleAccessor acc) {
198         this.bundleAccessor = acc;
199     }

201     @Inject
202     public void setVelocityManager(VelocityManager vm) {
203         Properties props = new Properties();
204         props.setProperty("osgi.resource.loader.description", "OSGI bundle loader");
205         props.setProperty("osgi.resource.loader.class", VelocityBundleResourceLoader.class.getName());
206         props.setProperty(Velocity.RESOURCE_LOADER, "strutsfile,strutsclass,osgi");
207         vm.setVelocityProperties(props);
208     }

211     public void setServletContext(ServletContext servletContext) {
212         this.servletContext = servletContext;
213     }

216     public void setFileManagerFactory(FileManagerFactory fmFactory) {
217         this.fileManagerFactory = fmFactory;
218     }
219 
220     public void destroy() {
221         try {
222             osgiHost.destroy();
223         } catch (Exception e) {
224             if (LOG.isErrorEnabled()) {
225                 LOG.error("Failed to stop OSGi container", e);
226             }
227         }
228     }
229 
230     /**
231      * Listens to bundle event to load/unload config
232      *
233      * @param bundleEvent the bundle event
234      */
235     public void bundleChanged(BundleEvent bundleEvent) {
236         Bundle bundle = bundleEvent.getBundle();
237         String bundleName = bundle.getSymbolicName();
238         if (bundleName != null && shouldProcessBundle(bundle)) {
239             switch (bundleEvent.getType()) {
240                 case BundleEvent.STARTED:
241                     LOG.trace("The bundle [{}] has been activated and will be scanned for struts configuration", bundleName);
242                     loadConfigFromBundle(bundle);
243                     break;
244                 case BundleEvent.STOPPED:
245                     onBundleStopped(bundle);
246                     break;
247             }
248         }
249     }
250 
251     /**
252      * This method is called when a bundle is stopped, so the config that is related to it is removed
253      *
254      * @param bundle the bundle that stopped
255      */
256     protected void onBundleStopped(Bundle bundle) {
257         Set<String> packages = bundleAccessor.getPackagesByBundle(bundle);
258         if (!packages.isEmpty()) {
259             if (LOG.isTraceEnabled()) {
260                 LOG.trace("The bundle [{}] has been stopped. The packages [{}] will be disabled", bundle.getSymbolicName(), StringUtils.join(packages, ","));
261             }
262             for (String packageName : packages) {
263                 configuration.removePackageConfig(packageName);
264             }
265         }
266     }
