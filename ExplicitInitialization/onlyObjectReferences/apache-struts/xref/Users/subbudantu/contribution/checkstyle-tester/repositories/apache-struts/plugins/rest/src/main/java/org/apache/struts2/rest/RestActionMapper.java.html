39  /**
40   * <!-- START SNIPPET: description -->
41   *
42   * This Restful action mapper enforces Ruby-On-Rails Rest-style mappings.  If the method 
43   * is not specified (via '!' or 'method:' prefix), the method is "guessed" at using 
44   * ReST-style conventions that examine the URL and the HTTP method.  Special care has
45   * been given to ensure this mapper works correctly with the codebehind plugin so that
46   * XML configuration is unnecessary.
47   *  
48   * <p>
49   *   This mapper supports the following parameters:
50   * </p>
51   * <ul>
52   *   <li><code>struts.mapper.idParameterName</code> - If set, this value will be the name
53   *       of the parameter under which the id is stored.  The id will then be removed
54   *       from the action name.  Whether or not the method is specified, the mapper will 
55   *       try to truncate the identifier from the url and store it as a parameter.
56   *   </li>
57   *   <li><code>struts.mapper.indexMethodName</code> - The method name to call for a GET
58   *       request with no id parameter. Defaults to 'index'.
59   *   </li>
60   *   <li><code>struts.mapper.getMethodName</code> - The method name to call for a GET
61   *       request with an id parameter. Defaults to 'show'.
62   *   </li>
63   *   <li><code>struts.mapper.postMethodName</code> - The method name to call for a POST
64   *       request with no id parameter. Defaults to 'create'.
65   *   </li>
66   *   <li><code>struts.mapper.putMethodName</code> - The method name to call for a PUT
67   *       request with an id parameter. Defaults to 'update'.
68   *   </li>
69   *   <li><code>struts.mapper.deleteMethodName</code> - The method name to call for a DELETE
70   *       request with an id parameter. Defaults to 'destroy'.
71   *   </li>
72   *   <li><code>struts.mapper.editMethodName</code> - The method name to call for a GET
73   *       request with an id parameter and the 'edit' view specified. Defaults to 'edit'.
74   *   </li>
75   *   <li><code>struts.mapper.newMethodName</code> - The method name to call for a GET
76   *       request with no id parameter and the 'new' view specified. Defaults to 'editNew'.
77   *   </li>
78   * </ul>
79   * <p>
80   * The following URL's will invoke its methods:
81   * </p>
82   * <ul> 
83   *  <li><code>GET:    /movies                =&gt; method="index"</code></li>
84   *  <li><code>GET:    /movies/Thrillers      =&gt; method="show", id="Thrillers"</code></li>
85   *  <li><code>GET:    /movies/Thrillers;edit =&gt; method="edit", id="Thrillers"</code></li>
86   *  <li><code>GET:    /movies/Thrillers/edit =&gt; method="edit", id="Thrillers"</code></li>
87   *  <li><code>GET:    /movies/new            =&gt; method="editNew"</code></li>
88   *  <li><code>POST:   /movies                =&gt; method="create"</code></li>
89   *  <li><code>PUT:    /movies/Thrillers      =&gt; method="update", id="Thrillers"</code></li>
90   *  <li><code>DELETE: /movies/Thrillers      =&gt; method="destroy", id="Thrillers"</code></li>
91   * </ul>
92   * <p>
93   * To simulate the HTTP methods PUT and DELETE, since they aren't supported by HTML,
94   * the HTTP parameter "_method" will be used.
95   * </p>
96   * <!-- END SNIPPET: description -->
97   */

191     public ActionMapping getMapping(HttpServletRequest request,
192             ConfigurationManager configManager) {
193         ActionMapping mapping = new ActionMapping();
194         String uri = RequestUtils.getUri(request);
195 
196         uri = dropExtension(uri, mapping);
197         if (uri == null) {
198             return null;
199         }
200 
201         parseNameAndNamespace(uri, mapping, configManager);
202 
203         handleSpecialParameters(request, mapping);
204 
205         if (mapping.getName() == null) {
206             return null;
207         }
208 
209         // handle "name!method" convention.
210         handleDynamicMethodInvocation(mapping, mapping.getName());
211 
212         String fullName = mapping.getName();
213         // Only try something if the action name is specified
214         if (StringUtils.isNotEmpty(fullName)) {
215 
216             // cut off any ;jsessionid= type appendix but allow the rails-like ;edit
217             int scPos = fullName.indexOf(';');
218             if (scPos > -1 && !"edit".equals(fullName.substring(scPos + 1))) {
219                 fullName = fullName.substring(0, scPos);
220             }
221 
222             int lastSlashPos = fullName.lastIndexOf('/');
223             String id = null;
224             if (lastSlashPos > -1) {
225 
226                 // fun trickery to parse 'actionName/id/methodName' in the case of 'animals/dog/edit'
227                 int prevSlashPos = fullName.lastIndexOf('/', lastSlashPos - 1);
228                 //WW-4589 do not overwrite explicit method name
229                 if (prevSlashPos > -1 && mapping.getMethod() == null) {
230                     mapping.setMethod(fullName.substring(lastSlashPos + 1));
231                     fullName = fullName.substring(0, lastSlashPos);
232                     lastSlashPos = prevSlashPos;
233                 }
234                 id = fullName.substring(lastSlashPos + 1);
235             }
236 
237 
238 
239             // If a method hasn't been explicitly named, try to guess using ReST-style patterns
240             if (mapping.getMethod() == null) {
241 
242             	if (isOptions(request)) {
243                 	mapping.setMethod(optionsMethodName);
244                 
245             	// Handle uris with no id, possibly ending in '/'
246             	} else if (lastSlashPos == -1 || lastSlashPos == fullName.length() -1) {
247 
248                     // Index e.g. foo
249                     if (isGet(request)) {
250                         mapping.setMethod(indexMethodName);
251                         
252                     // Creating a new entry on POST e.g. foo
253                     } else if (isPost(request)) {
254                     	if (isExpectContinue(request)) {
255                             mapping.setMethod(postContinueMethodName);
256                     	} else {
257                             mapping.setMethod(postMethodName);
258                     	}
259                     }
260 
261                 // Handle uris with an id at the end
262                 } else if (id != null) {
263                     
264                     // Viewing the form to edit an item e.g. foo/1;edit
265                     if (isGet(request) && id.endsWith(";edit")) {
266                         id = id.substring(0, id.length() - ";edit".length());
267                         mapping.setMethod(editMethodName);
268                         
269                     // Viewing the form to create a new item e.g. foo/new
270                     } else if (isGet(request) && "new".equals(id)) {
271                         mapping.setMethod(newMethodName);
272 
273                     // Removing an item e.g. foo/1
274                     } else if (isDelete(request)) {
275                         mapping.setMethod(deleteMethodName);
276                         
277                     // Viewing an item e.g. foo/1
278                     } else if (isGet(request)) {
279                         mapping.setMethod(getMethodName);
280                     
281                     // Updating an item e.g. foo/1    
282                     }  else if (isPut(request)) {
283                     	if (isExpectContinue(request)) {
284                             mapping.setMethod(putContinueMethodName);
285                     	} else {
286                             mapping.setMethod(putMethodName);
287                     	}
288                     }
289                 }
290             }
291 
292             // cut off the id parameter, even if a method is specified
293             if (id != null) {
294                 if (!"new".equals(id)) {
295                     if (mapping.getParams() == null) {
296                         mapping.setParams(new HashMap<String, Object>());
297                     }
298                     mapping.getParams().put(idParameterName, new String[]{id});
299                 }
300                 fullName = fullName.substring(0, lastSlashPos);
301             }
302 
303             mapping.setName(cleanupActionName(fullName));
304         }
305         return mapping;
306     }

308     private void handleDynamicMethodInvocation(ActionMapping mapping, String name) {
309         int exclamation = name.lastIndexOf("!");
310         if (exclamation != -1) {
311             String actionName = name.substring(0, exclamation);
312             String actionMethod = name.substring(exclamation + 1);
313 
314             // WW-4585
315             // add any ; appendix to name, it will be handled later in getMapping method
316             int scPos = actionMethod.indexOf(';');
317             if (scPos != -1) {
318                 actionName = actionName + actionMethod.substring(scPos);
319                 actionMethod = actionMethod.substring(0, scPos);
320             }
321 
322             mapping.setName(actionName);
323             if (allowDynamicMethodCalls) {
324                 mapping.setMethod(cleanupMethodName(actionMethod));
325             } else {
326                 mapping.setMethod(null);
327             }
328         }
329     }

331     /**
332      * Parses the name and namespace from the uri.  Uses the configured package 
333      * namespaces to determine the name and id parameter, to be parsed later.
334      *
335      * @param uri
336      *            The uri
337      * @param mapping
338      *            The action mapping to populate
339      */
340     protected void parseNameAndNamespace(String uri, ActionMapping mapping, ConfigurationManager configManager) {
341         String namespace, name;
342         int lastSlash = uri.lastIndexOf("/");
343         if (lastSlash == -1) {
344             namespace = "";
345             name = uri;
346         } else if (lastSlash == 0) {
347             // ww-1046, assume it is the root namespace, it will fallback to
348             // default
349             // namespace anyway if not found in root namespace.
350             namespace = "/";
351             name = uri.substring(lastSlash + 1);
352         } else {
353             // Try to find the namespace in those defined, defaulting to ""
354             Configuration config = configManager.getConfiguration();
355             String prefix = uri.substring(0, lastSlash);
356             namespace = "";
357             // Find the longest matching namespace, defaulting to the default
358             for (Object o : config.getPackageConfigs().values()) {
359                 String ns = ((PackageConfig) o).getNamespace();
360                 if (ns != null && prefix.startsWith(ns) && (prefix.length() == ns.length() || prefix.charAt(ns.length()) == '/')) {
361                     if (ns.length() > namespace.length()) {
362                         namespace = ns;
363                     }
364                 }
365             }
366 
367             name = uri.substring(namespace.length() + 1);
368         }
369 
370         mapping.setNamespace(namespace);
371         mapping.setName(name);
372     }

374     protected boolean isGet(HttpServletRequest request) {
375         return GET.equalsIgnoreCase(request.getMethod());
376     }
377 
378     protected boolean isPost(HttpServletRequest request) {
379         return POST.equalsIgnoreCase(request.getMethod());
380     }
381 
382     protected boolean isPut(HttpServletRequest request) {
383         return PUT.equalsIgnoreCase(request.getMethod()) || isPost(request) && PUT.equalsIgnoreCase(request.getParameter(HTTP_METHOD_PARAM));
384     }
385 
386     protected boolean isDelete(HttpServletRequest request) {
387         return DELETE.equalsIgnoreCase(request.getMethod()) || DELETE.equalsIgnoreCase(request.getParameter(HTTP_METHOD_PARAM));
388     }
389 
390     protected boolean isOptions(HttpServletRequest request) {
391         return OPTIONS.equalsIgnoreCase(request.getMethod());
392     }
393     
394     protected boolean isExpectContinue(HttpServletRequest request) {
395     	String expect = request.getHeader("Expect");
396     	return (expect != null && expect.toLowerCase().contains("100-continue")); 
397     }
