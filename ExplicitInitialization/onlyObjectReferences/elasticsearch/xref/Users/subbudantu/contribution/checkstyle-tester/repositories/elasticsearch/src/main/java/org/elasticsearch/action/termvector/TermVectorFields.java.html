36  /**
37   * This class represents the result of a {@link TermVectorRequest}. It works
38   * exactly like the {@link Fields} class except for one thing: It can return
39   * offsets and payloads even if positions are not present. You must call
40   * nextPosition() anyway to move the counter although this method only returns
41   * <tt>-1,</tt>, if no positions were returned by the {@link TermVectorRequest}.
42   * <p/>
43   * The data is stored in two byte arrays ({@code headerRef} and
44   * {@code termVectors}, both {@link ByteRef}) that have the following format:
45   * <p/>
46   * {@code headerRef}: Stores offsets per field in the {@code termVectors} array
47   * and some header information as {@link BytesRef}. Format is
48   * <ul>
49   * <li>String : "TV"</li>
50   * <li>vint: version (=-1)</li>
51   * <li>boolean: hasTermStatistics (are the term statistics stored?)</li>
52   * <li>boolean: hasFieldStatitsics (are the field statistics stored?)</li>
53   * <li>vint: number of fields</li>
54   * <ul>
55   * <li>String: field name 1</li>
56   * <li>vint: offset in {@code termVectors} for field 1</li>
57   * <li>...</li>
58   * <li>String: field name last field</li>
59   * <li>vint: offset in {@code termVectors} for last field</li>
60   * </ul>
61   * </ul>
62   * <p/>
63   * termVectors: Stores the actual term vectors as a {@link BytesRef}.
64   * <p/>
65   * Term vectors for each fields are stored in blocks, one for each field. The
66   * offsets in {@code headerRef} are used to find where the block for a field
67   * starts. Each block begins with a
68   * <ul>
69   * <li>vint: number of terms</li>
70   * <li>boolean: positions (has it positions stored?)</li>
71   * <li>boolean: offsets (has it offsets stored?)</li>
72   * <li>boolean: payloads (has it payloads stored?)</li>
73   * </ul>
74   * If the field statistics were requested ({@code hasFieldStatistics} is true,
75   * see {@code headerRef}), the following numbers are stored:
76   * <ul>
77   * <li>vlong: sum of total term freqencies of the field (sumTotalTermFreq)</li>
78   * <li>vlong: sum of document frequencies for each term (sumDocFreq)</li>
79   * <li>vint: number of documents in the shard that has an entry for this field
80   * (docCount)</li>
81   * </ul>
82   * <p/>
83   * After that, for each term it stores
84   * <ul>
85   * <ul>
86   * <li>vint: term lengths</li>
87   * <li>BytesRef: term name</li>
88   * </ul>
89   * <p/>
90   * If term statistics are requested ({@code hasTermStatistics} is true, see
91   * {@code headerRef}):
92   * <ul>
93   * <li>vint: document frequency, how often does this term appear in documents?</li>
94   * <li>vlong: total term frequency. Sum of terms in this field.</li>
95   * </ul>
96   * After that
97   * <ul>
98   * <li>vint: frequency (always returned)</li>
99   * <ul>
100  * <li>vint: position_1 (if positions == true)</li>
101  * <li>vint: startOffset_1 (if offset == true)</li>
102  * <li>vint: endOffset_1 (if offset == true)</li>
103  * <li>BytesRef: payload_1 (if payloads == true)</li>
104  * <li>...</li>
105  * <li>vint: endOffset_freqency (if offset == true)</li>
106  * <li>BytesRef: payload_freqency (if payloads == true)</li>
107  * <ul>
108  * </ul> </ul>
109  */
110 
111 public final class TermVectorFields extends Fields {
112 
113     private final ObjectLongOpenHashMap<String> fieldMap;
114     private final BytesReference termVectors;
115     final boolean hasTermStatistic;
116     final boolean hasFieldStatistic;
117 
118     /**
119      * @param headerRef   Stores offsets per field in the {@code termVectors} and some
120      *                    header information as {@link BytesRef}.
121      * @param termVectors Stores the actual term vectors as a {@link BytesRef}.
122      */
123     public TermVectorFields(BytesReference headerRef, BytesReference termVectors) throws IOException {
124         BytesStreamInput header = new BytesStreamInput(headerRef);
125         fieldMap = new ObjectLongOpenHashMap<>();
126 
127         // here we read the header to fill the field offset map
128         String headerString = header.readString();
129         assert headerString.equals("TV");
130         int version = header.readInt();
131         assert version == -1;
132         hasTermStatistic = header.readBoolean();
133         hasFieldStatistic = header.readBoolean();
134         final int numFields = header.readVInt();
135         for (int i = 0; i < numFields; i++) {
136             fieldMap.put((header.readString()), header.readVLong());
137         }
138         header.close();
139         // reference to the term vector data
140         this.termVectors = termVectors;
141     }
142 
143     @Override
144     public Iterator<String> iterator() {
145         final Iterator<ObjectLongCursor<String>> iterator = fieldMap.iterator();
146         return new Iterator<String>() {
147             @Override
148             public boolean hasNext() {
149                 return iterator.hasNext();
150             }
151 
152             @Override
153             public String next() {
154                 return iterator.next().key;
155             }
156 
157             @Override
158             public void remove() {
159                 throw new UnsupportedOperationException();
160             }
161         };
162     }
163 
164     @Override
165     public Terms terms(String field) throws IOException {
166         // first, find where in the termVectors bytes the actual term vector for
167         // this field is stored
168         if (!fieldMap.containsKey(field)) {
169             return null; // we don't have it.
170         }
171         long readOffset = fieldMap.lget();
172         return new TermVector(termVectors, readOffset);
173     }
174 
175     @Override
176     public int size() {
177         return fieldMap.size();
178     }

180     private final class TermVector extends Terms {
181 
182         private final BytesStreamInput perFieldTermVectorInput;
183         private final long readOffset;
184 
185         private long numTerms;
186         private boolean hasPositions;
187         private boolean hasOffsets;
188         private boolean hasPayloads;
189         private long sumTotalTermFreq;
190         private long sumDocFreq;
191         private int docCount;
192 
193         public TermVector(BytesReference termVectors, long readOffset) throws IOException {
194             this.perFieldTermVectorInput = new BytesStreamInput(termVectors);
195             this.readOffset = readOffset;
196             reset();
197         }
198 
199         private void reset() throws IOException {
200             this.perFieldTermVectorInput.reset();
201             this.perFieldTermVectorInput.skip(readOffset);
202 
203             // read how many terms....
204             this.numTerms = perFieldTermVectorInput.readVLong();
205             // ...if positions etc. were stored....
206             this.hasPositions = perFieldTermVectorInput.readBoolean();
207             this.hasOffsets = perFieldTermVectorInput.readBoolean();
208             this.hasPayloads = perFieldTermVectorInput.readBoolean();
209             // read the field statistics
210             this.sumTotalTermFreq = hasFieldStatistic ? readPotentiallyNegativeVLong(perFieldTermVectorInput) : -1;
211             this.sumDocFreq = hasFieldStatistic ? readPotentiallyNegativeVLong(perFieldTermVectorInput) : -1;
212             this.docCount = hasFieldStatistic ? readPotentiallyNegativeVInt(perFieldTermVectorInput) : -1;
213         }
214 
215         @Override
216         public TermsEnum iterator(TermsEnum reuse) throws IOException {
217             // reset before asking for an iterator
218             reset();
219             // convert bytes ref for the terms to actual data
220             return new TermsEnum() {
221                 int currentTerm = 0;
222                 int freq = 0;
223                 int docFreq = -1;
224                 long totalTermFrequency = -1;
225                 int[] positions = new int[1];
226                 int[] startOffsets = new int[1];
227                 int[] endOffsets = new int[1];
228                 BytesRefBuilder[] payloads = new BytesRefBuilder[1];
229                 final BytesRefBuilder spare = new BytesRefBuilder();
230 
231                 @Override
232                 public BytesRef next() throws IOException {
233                     if (currentTerm++ < numTerms) {
234                         // term string. first the size...
235                         int termVectorSize = perFieldTermVectorInput.readVInt();
236                         spare.grow(termVectorSize);
237                         // ...then the value.
238                         perFieldTermVectorInput.readBytes(spare.bytes(), 0, termVectorSize);
239                         spare.setLength(termVectorSize);
240                         if (hasTermStatistic) {
241                             docFreq = readPotentiallyNegativeVInt(perFieldTermVectorInput);
242                             totalTermFrequency = readPotentiallyNegativeVLong(perFieldTermVectorInput);
243                         }
244 
245                         freq = readPotentiallyNegativeVInt(perFieldTermVectorInput);
246                         // grow the arrays to read the values. this is just
247                         // for performance reasons. Re-use memory instead of
248                         // realloc.
249                         growBuffers();
250                         // finally, read the values into the arrays
251                         // curentPosition etc. so that we can just iterate
252                         // later
253                         writeInfos(perFieldTermVectorInput);
254                         return spare.get();
255 
256                     } else {
257                         return null;
258                     }
259                 }
260 
261                 private void writeInfos(final BytesStreamInput input) throws IOException {
262                     for (int i = 0; i < freq; i++) {
263                         if (hasPositions) {
264                             positions[i] = input.readVInt();
265                         }
266                         if (hasOffsets) {
267                             startOffsets[i] = input.readVInt();
268                             endOffsets[i] = input.readVInt();
269                         }
270                         if (hasPayloads) {
271                             int payloadLength = input.readVInt();
272                             if (payloads[i] == null) {
273                                 payloads[i] = new BytesRefBuilder();
274                             }
275                             payloads[i].grow(payloadLength);
276                             input.readBytes(payloads[i].bytes(), 0, payloadLength);
277                             payloads[i].setLength(payloadLength);
278                         }
279                     }
280                 }
281 
282                 private void growBuffers() {
283                     if (hasPositions) {
284                         positions = grow(positions, freq);
285                     }
286                     if (hasOffsets) {
287                         startOffsets = grow(startOffsets, freq);
288                         endOffsets = grow(endOffsets, freq);
289                     }
290                     if (hasPayloads) {
291                         if (payloads.length < freq) {
292                             payloads = Arrays.copyOf(payloads, ArrayUtil.oversize(freq, RamUsageEstimator.NUM_BYTES_OBJECT_REF));
293                         }
294                     }
295                 }
296 
297                 @Override
298                 public Comparator<BytesRef> getComparator() {
299                     return BytesRef.getUTF8SortedAsUnicodeComparator();
300                 }
301 
302                 @Override
303                 public SeekStatus seekCeil(BytesRef text) throws IOException {
304                     throw new UnsupportedOperationException();
305                 }
306 
307                 @Override
308                 public void seekExact(long ord) throws IOException {
309                     throw new UnsupportedOperationException("Seek is not supported");
310                 }
311 
312                 @Override
313                 public BytesRef term() throws IOException {
314                     return spare.get();
315                 }
316 
317                 @Override
318                 public long ord() throws IOException {
319                     throw new UnsupportedOperationException("ordinals are not supported");
320                 }
321 
322                 @Override
323                 public int docFreq() throws IOException {
324                     return docFreq;
325                 }
326 
327                 @Override
328                 public long totalTermFreq() throws IOException {
329                     return totalTermFrequency;
330                 }
331 
332                 @Override
333                 public DocsEnum docs(Bits liveDocs, DocsEnum reuse, int flags) throws IOException {
334                     return docsAndPositions(liveDocs, reuse instanceof DocsAndPositionsEnum ? (DocsAndPositionsEnum) reuse : null, 0);
335                 }
336 
337                 @Override
338                 public DocsAndPositionsEnum docsAndPositions(Bits liveDocs, DocsAndPositionsEnum reuse, int flags) throws IOException {
339                     final TermVectorDocsAndPosEnum retVal = (reuse instanceof TermVectorDocsAndPosEnum ? (TermVectorDocsAndPosEnum) reuse
340                             : new TermVectorDocsAndPosEnum());
341                     return retVal.reset(hasPositions ? positions : null, hasOffsets ? startOffsets : null, hasOffsets ? endOffsets
342                             : null, hasPayloads ? payloads : null, freq);
343                 }
344 
345             };
346         }
347 
348         @Override
349         public Comparator<BytesRef> getComparator() {
350             return BytesRef.getUTF8SortedAsUnicodeComparator();
351         }
352 
353         @Override
354         public long size() throws IOException {
355             return numTerms;
356         }
357 
358         @Override
359         public long getSumTotalTermFreq() throws IOException {
360             return sumTotalTermFreq;
361         }
362 
363         @Override
364         public long getSumDocFreq() throws IOException {
365             return sumDocFreq;
366         }
367 
368         @Override
369         public int getDocCount() throws IOException {
370             return docCount;
371         }
372 
373         @Override
374         public boolean hasFreqs() {
375             return true;
376         }
377 
378         @Override
379         public boolean hasOffsets() {
380             return hasOffsets;
381         }
382 
383         @Override
384         public boolean hasPositions() {
385             return hasPositions;
386         }
387 
388         @Override
389         public boolean hasPayloads() {
390             return hasPayloads;
391         }
392     }

394     private final class TermVectorDocsAndPosEnum extends DocsAndPositionsEnum {
395         private boolean hasPositions;
396         private boolean hasOffsets;
397         private boolean hasPayloads;
398         int curPos = -1;
399         int doc = -1;
400         private int freq;
401         private int[] startOffsets;
402         private int[] positions;
403         private BytesRefBuilder[] payloads;
404         private int[] endOffsets;
405 
406         private DocsAndPositionsEnum reset(int[] positions, int[] startOffsets, int[] endOffsets, BytesRefBuilder[] payloads, int freq) {
407             curPos = -1;
408             doc = -1;
409             this.hasPositions = positions != null;
410             this.hasOffsets = startOffsets != null;
411             this.hasPayloads = payloads != null;
412             this.freq = freq;
413             this.startOffsets = startOffsets;
414             this.endOffsets = endOffsets;
415             this.payloads = payloads;
416             this.positions = positions;
417             return this;
418         }
419 
420         @Override
421         public int nextDoc() throws IOException {
422             return doc = (doc == -1 ? 0 : NO_MORE_DOCS);
423         }
424 
425         @Override
426         public int docID() {
427             return doc;
428         }
429 
430         @Override
431         public int advance(int target) throws IOException {
432             while (nextDoc() < target && doc != NO_MORE_DOCS) {
433             }
434             return doc;
435         }
436 
437         @Override
438         public int freq() throws IOException {
439             return freq;
440         }
441 
442         // call nextPosition once before calling this one
443         // because else counter is not advanced
444         @Override
445         public int startOffset() throws IOException {
446             assert curPos < freq && curPos >= 0;
447             return hasOffsets ? startOffsets[curPos] : -1;
448 
449         }
450 
451         @Override
452         // can return -1 if posistions were not requested or
453         // stored but offsets were stored and requested
454         public int nextPosition() throws IOException {
455             assert curPos + 1 < freq;
456             ++curPos;
457             // this is kind of cheating but if you don't need positions
458             // we safe lots fo space on the wire
459             return hasPositions ? positions[curPos] : -1;
460         }
461 
462         @Override
463         public BytesRef getPayload() throws IOException {
464             assert curPos < freq && curPos >= 0;
465             if (hasPayloads) {
466                 final BytesRefBuilder payload = payloads[curPos];
467                 if (payload != null) {
468                     return payload.get();
469                 }
470             }
471             return null;
472         }
473 
474         @Override
475         public int endOffset() throws IOException {
476             assert curPos < freq && curPos >= 0;
477             return hasOffsets ? endOffsets[curPos] : -1;
478         }
479 
480         @Override
481         public long cost() {
482             return 1;
483         }
484     }

486     // read a vInt. this is used if the integer might be negative. In this case,
487     // the writer writes a 0 for -1 or value +1 and accordingly we have to
488     // substract 1 again
489     // adds one to mock not existing term freq
490     int readPotentiallyNegativeVInt(BytesStreamInput stream) throws IOException {
491         return stream.readVInt() - 1;
492     }
493 
494     // read a vLong. this is used if the integer might be negative. In this
495     // case, the writer writes a 0 for -1 or value +1 and accordingly we have to
496     // substract 1 again
497     // adds one to mock not existing term freq
498     long readPotentiallyNegativeVLong(BytesStreamInput stream) throws IOException {
499         return stream.readVLong() - 1;
500     }
