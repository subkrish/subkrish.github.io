87              T readFrom(StreamInput in) throws IOException;

97      /**
98       * Register a custom index meta data factory. Make sure to call it from a static block.
99       */
100     public static void registerFactory(String type, Custom.Factory factory) {
101         customFactories.put(type, factory);
102     }
103 
104     @Nullable
105     public static <T extends Custom> Custom.Factory<T> lookupFactory(String type) {
106         return customFactories.get(type);
107     }
108 
109     public static <T extends Custom> Custom.Factory<T> lookupFactorySafe(String type) throws ElasticsearchIllegalArgumentException {
110         Custom.Factory<T> factory = customFactories.get(type);
111         if (factory == null) {
112             throw new ElasticsearchIllegalArgumentException("No custom state factory registered for type [" + type + "]");
113         }
114         return factory;
115     }

265     public enum Metric {
266         VERSION("version"),
267         MASTER_NODE("master_node"),
268         BLOCKS("blocks"),
269         NODES("nodes"),
270         METADATA("metadata"),
271         ROUTING_TABLE("routing_table"),
272         CUSTOMS("customs");
273 
274         private static Map<String, Metric> valueToEnum;
275 
276         static {
277             valueToEnum = new HashMap<>();
278             for (Metric metric : Metric.values()) {
279                 valueToEnum.put(metric.value, metric);
280             }
281         }
282 
283         private final String value;
284 
285         private Metric(String value) {
286             this.value = value;
287         }
288 
289         public static EnumSet<Metric> parseString(String param, boolean ignoreUnknown) {
290             String[] metrics = Strings.splitStringByCommaToArray(param);
291             EnumSet<Metric> result = EnumSet.noneOf(Metric.class);
292             for (String metric : metrics) {
293                 if ("_all".equals(metric)) {
294                     result = EnumSet.allOf(Metric.class);
295                     break;
296                 }
297                 Metric m = valueToEnum.get(metric);
298                 if (m == null) {
299                     if (!ignoreUnknown) {
300                         throw new ElasticsearchIllegalArgumentException("Unknown metric [" + metric + "]");
301                     }
302                 } else {
303                     result.add(m);
304                 }
305             }
306             return result;
307         }
308 
309         @Override
310         public String toString() {
311             return value;
312         }
313     }

317     @Override
318     public XContentBuilder toXContent(XContentBuilder builder, Params params) throws IOException {
319         EnumSet<Metric> metrics = Metric.parseString(params.param("metric", "_all"), true);
320 
321         if (metrics.contains(Metric.VERSION)) {
322             builder.field("version", version);
323         }
324 
325         if (metrics.contains(Metric.MASTER_NODE)) {
326             builder.field("master_node", nodes().masterNodeId());
327         }
328 
329         if (metrics.contains(Metric.BLOCKS)) {
330             builder.startObject("blocks");
331 
332             if (!blocks().global().isEmpty()) {
333                 builder.startObject("global");
334                 for (ClusterBlock block : blocks().global()) {
335                     block.toXContent(builder, params);
336                 }
337                 builder.endObject();
338             }
339 
340             if (!blocks().indices().isEmpty()) {
341                 builder.startObject("indices");
342                 for (Map.Entry<String, ImmutableSet<ClusterBlock>> entry : blocks().indices().entrySet()) {
343                     builder.startObject(entry.getKey());
344                     for (ClusterBlock block : entry.getValue()) {
345                         block.toXContent(builder, params);
346                     }
347                     builder.endObject();
348                 }
349                 builder.endObject();
350             }
351 
352             builder.endObject();
353         }
354 
355         // nodes
356         if (metrics.contains(Metric.NODES)) {
357             builder.startObject("nodes");
358             for (DiscoveryNode node : nodes()) {
359                 builder.startObject(node.id(), XContentBuilder.FieldCaseConversion.NONE);
360                 builder.field("name", node.name());
361                 builder.field("transport_address", node.address().toString());
362 
363                 builder.startObject("attributes");
364                 for (Map.Entry<String, String> attr : node.attributes().entrySet()) {
365                     builder.field(attr.getKey(), attr.getValue());
366                 }
367                 builder.endObject();
368 
369                 builder.endObject();
370             }
371             builder.endObject();
372         }
373 
374         // meta data
375         if (metrics.contains(Metric.METADATA)) {
376             builder.startObject("metadata");
377 
378             builder.startObject("templates");
379             for (ObjectCursor<IndexTemplateMetaData> cursor : metaData().templates().values()) {
380                 IndexTemplateMetaData templateMetaData = cursor.value;
381                 builder.startObject(templateMetaData.name(), XContentBuilder.FieldCaseConversion.NONE);
382 
383                 builder.field("template", templateMetaData.template());
384                 builder.field("order", templateMetaData.order());
385 
386                 builder.startObject("settings");
387                 Settings settings = templateMetaData.settings();
388                 if (settingsFilter != null) {
389                     settings = settingsFilter.filterSettings(settings);
390                 }
391                 settings.toXContent(builder, params);
392                 builder.endObject();
393 
394                 builder.startObject("mappings");
395                 for (ObjectObjectCursor<String, CompressedString> cursor1 : templateMetaData.mappings()) {
396                     byte[] mappingSource = cursor1.value.uncompressed();
397                     XContentParser parser = XContentFactory.xContent(mappingSource).createParser(mappingSource);
398                     Map<String, Object> mapping = parser.map();
399                     if (mapping.size() == 1 && mapping.containsKey(cursor1.key)) {
400                         // the type name is the root value, reduce it
401                         mapping = (Map<String, Object>) mapping.get(cursor1.key);
402                     }
403                     builder.field(cursor1.key);
404                     builder.map(mapping);
405                 }
406                 builder.endObject();
407 
408 
409                 builder.endObject();
410             }
411             builder.endObject();
412 
413             builder.startObject("indices");
414             for (IndexMetaData indexMetaData : metaData()) {
415                 builder.startObject(indexMetaData.index(), XContentBuilder.FieldCaseConversion.NONE);
416 
417                 builder.field("state", indexMetaData.state().toString().toLowerCase(Locale.ENGLISH));
418 
419                 builder.startObject("settings");
420                 Settings settings = indexMetaData.settings();
421                 if (settingsFilter != null) {
422                     settings = settingsFilter.filterSettings(settings);
423                 }
424                 settings.toXContent(builder, params);
425                 builder.endObject();
426 
427                 builder.startObject("mappings");
428                 for (ObjectObjectCursor<String, MappingMetaData> cursor : indexMetaData.mappings()) {
429                     byte[] mappingSource = cursor.value.source().uncompressed();
430                     XContentParser parser = XContentFactory.xContent(mappingSource).createParser(mappingSource);
431                     Map<String, Object> mapping = parser.map();
432                     if (mapping.size() == 1 && mapping.containsKey(cursor.key)) {
433                         // the type name is the root value, reduce it
434                         mapping = (Map<String, Object>) mapping.get(cursor.key);
435                     }
436                     builder.field(cursor.key);
437                     builder.map(mapping);
438                 }
439                 builder.endObject();
440 
441                 builder.startArray("aliases");
442                 for (ObjectCursor<String> cursor : indexMetaData.aliases().keys()) {
443                     builder.value(cursor.value);
444                 }
445                 builder.endArray();
446 
447                 builder.endObject();
448             }
449             builder.endObject();
450 
451             for (ObjectObjectCursor<String, MetaData.Custom> cursor : metaData.customs()) {
452                 builder.startObject(cursor.key);
453                 MetaData.lookupFactorySafe(cursor.key).toXContent(cursor.value, builder, params);
454                 builder.endObject();
455             }
456 
457             builder.endObject();
458         }
459 
460         // routing table
461         if (metrics.contains(Metric.ROUTING_TABLE)) {
462             builder.startObject("routing_table");
463             builder.startObject("indices");
464             for (IndexRoutingTable indexRoutingTable : routingTable()) {
465                 builder.startObject(indexRoutingTable.index(), XContentBuilder.FieldCaseConversion.NONE);
466                 builder.startObject("shards");
467                 for (IndexShardRoutingTable indexShardRoutingTable : indexRoutingTable) {
468                     builder.startArray(Integer.toString(indexShardRoutingTable.shardId().id()));
469                     for (ShardRouting shardRouting : indexShardRoutingTable) {
470                         shardRouting.toXContent(builder, params);
471                     }
472                     builder.endArray();
473                 }
474                 builder.endObject();
475                 builder.endObject();
476             }
477             builder.endObject();
478             builder.endObject();
479         }
480 
481         // routing nodes
482         if (metrics.contains(Metric.ROUTING_TABLE)) {
483             builder.startObject("routing_nodes");
484             builder.startArray("unassigned");
485             for (ShardRouting shardRouting : readOnlyRoutingNodes().unassigned()) {
486                 shardRouting.toXContent(builder, params);
487             }
488             builder.endArray();
489 
490             builder.startObject("nodes");
491             for (RoutingNode routingNode : readOnlyRoutingNodes()) {
492                 builder.startArray(routingNode.nodeId(), XContentBuilder.FieldCaseConversion.NONE);
493                 for (ShardRouting shardRouting : routingNode) {
494                     shardRouting.toXContent(builder, params);
495                 }
496                 builder.endArray();
497             }
498             builder.endObject();
499 
500             builder.endObject();
501         }
502         if (metrics.contains(Metric.ROUTING_TABLE)) {
503             // An empty array is written to maintain REST backwards compatibility for 1.0 -> 1.x
504             builder.startArray("allocations");
505             builder.endArray();
506         }
507 
508         if (metrics.contains(Metric.CUSTOMS)) {
509             for (ObjectObjectCursor<String, Custom> cursor : customs) {
510                 builder.startObject(cursor.key);
511                 lookupFactorySafe(cursor.key).toXContent(cursor.value, builder, params);
512                 builder.endObject();
513             }
514         }
515 
516         return builder;
517     }

623         /**
624          * @param data               input bytes
625          * @param localNode          used to set the local node in the cluster state.
626          * @param defaultClusterName this cluster name will be used of if the deserialized cluster state does not have a name set
627          *                           (which is only introduced in version 1.1.1)
628          */
629         public static ClusterState fromBytes(byte[] data, DiscoveryNode localNode, ClusterName defaultClusterName) throws IOException {
630             return readFrom(new BytesStreamInput(data, false), localNode, defaultClusterName);
631         }
632 
633         public static void writeTo(ClusterState state, StreamOutput out) throws IOException {
634             if (out.getVersion().onOrAfter(Version.V_1_1_1)) {
635                 out.writeBoolean(state.clusterName != null);
636                 if (state.clusterName != null) {
637                     state.clusterName.writeTo(out);
638                 }
639             }
640             out.writeLong(state.version());
641             MetaData.Builder.writeTo(state.metaData(), out);
642             RoutingTable.Builder.writeTo(state.routingTable(), out);
643             DiscoveryNodes.Builder.writeTo(state.nodes(), out);
644             ClusterBlocks.Builder.writeClusterBlocks(state.blocks(), out);
645             if (out.getVersion().before(Version.V_1_1_0)) {
646                 // Versions before 1.1.0 are expecting AllocationExplanation
647                 AllocationExplanation.EMPTY.writeTo(out);
648             }
649             out.writeVInt(state.customs().size());
650             for (ObjectObjectCursor<String, Custom> cursor : state.customs()) {
651                 out.writeString(cursor.key);
652                 lookupFactorySafe(cursor.key).writeTo(cursor.value, out);
653             }
654         }
655 
656         /**
657          * @param in                 input stream
658          * @param localNode          used to set the local node in the cluster state. can be null.
659          * @param defaultClusterName this cluster name will be used of receiving a cluster state from a node on version older than 1.1.1
660          *                           or if the sending node did not set a cluster name
661          */
662         public static ClusterState readFrom(StreamInput in, @Nullable DiscoveryNode localNode, @Nullable ClusterName defaultClusterName) throws IOException {
663             ClusterName clusterName = defaultClusterName;
664             if (in.getVersion().onOrAfter(Version.V_1_1_1)) {
665                 // it might be null even if it comes from a >= 1.1.1 node since it's origin might be an older node
666                 if (in.readBoolean()) {
667                     clusterName = ClusterName.readClusterName(in);
668                 }
669             }
670             Builder builder = new Builder(clusterName);
671             builder.version = in.readLong();
672             builder.metaData = MetaData.Builder.readFrom(in);
673             builder.routingTable = RoutingTable.Builder.readFrom(in);
674             builder.nodes = DiscoveryNodes.Builder.readFrom(in, localNode);
675             builder.blocks = ClusterBlocks.Builder.readClusterBlocks(in);
676             if (in.getVersion().before(Version.V_1_1_0)) {
677                 // Ignore the explanation read, since after 1.1.0 it's not part of the cluster state
678                 AllocationExplanation.readAllocationExplanation(in);
679             }
680             int customSize = in.readVInt();
681             for (int i = 0; i < customSize; i++) {
682                 String type = in.readString();
683                 Custom customIndexMetaData = lookupFactorySafe(type).readFrom(in);
684                 builder.putCustom(type, customIndexMetaData);
685             }
686             return builder.build();
687         }
