38      private boolean useTimestamp = false;
39      private boolean skipExisting = false;
40  
41      public boolean download(URL source, File dest, @Nullable DownloadProgress progress, TimeValue timeout) throws Exception {
42          if (dest.exists() && skipExisting) {
43              return true;
44          }
45  
46          //don't do any progress, unless asked
47          if (progress == null) {
48              progress = new NullProgress();
49          }
50  
51          //set the timestamp to the file date.
52          long timestamp = 0;
53  
54          boolean hasTimestamp = false;
55          if (useTimestamp && dest.exists()) {
56              timestamp = dest.lastModified();
57              hasTimestamp = true;
58          }
59  
60          GetThread getThread = new GetThread(source, dest, hasTimestamp, timestamp, progress);
61  
62          try {
63              getThread.setDaemon(true);
64              getThread.start();
65              getThread.join(timeout.millis());
66  
67              if (getThread.isAlive()) {
68                  throw new ElasticsearchTimeoutException("The GET operation took longer than " + timeout + ", stopping it.");
69              }
70          }
71          catch (InterruptedException ie) {
72              return false;
73          } finally {
74              getThread.closeStreams();
75          }
76  
77          return getThread.wasSuccessful();
78      }

128     /**
129      * verbose progress system prints to some output stream
130      */
131     public static class VerboseProgress implements DownloadProgress {
132         private int dots = 0;
133         // CheckStyle:VisibilityModifier OFF - bc
134         PrintWriter writer;
135         // CheckStyle:VisibilityModifier ON
136 
137         /**
138          * Construct a verbose progress reporter.
139          *
140          * @param out the output stream.
141          */
142         public VerboseProgress(PrintStream out) {
143             this.writer = new PrintWriter(out);
144         }
145 
146         /**
147          * Construct a verbose progress reporter.
148          *
149          * @param writer the output stream.
150          */
151         public VerboseProgress(PrintWriter writer) {
152             this.writer = writer;
153         }
154 
155         /**
156          * begin a download
157          */
158         public void beginDownload() {
159             writer.print("Downloading ");
160             dots = 0;
161         }
162 
163         /**
164          * tick handler
165          */
166         public void onTick() {
167             writer.print(".");
168             if (dots++ > 50) {
169                 writer.flush();
170                 dots = 0;
171             }
172         }
173 
174         /**
175          * end a download
176          */
177         public void endDownload() {
178             writer.println("DONE");
179             writer.flush();
180         }
181     }

183     private class GetThread extends Thread {
184 
185         private final URL source;
186         private final File dest;
187         private final boolean hasTimestamp;
188         private final long timestamp;
189         private final DownloadProgress progress;
190 
191         private boolean success = false;
192         private IOException ioexception = null;
193         private InputStream is = null;
194         private OutputStream os = null;
195         private URLConnection connection;
196         private int redirections = 0;
197 
198         GetThread(URL source, File dest, boolean h, long t, DownloadProgress p) {
199             this.source = source;
200             this.dest = dest;
201             hasTimestamp = h;
202             timestamp = t;
203             progress = p;
204         }
205 
206         public void run() {
207             try {
208                 success = get();
209             } catch (IOException ioex) {
210                 ioexception = ioex;
211             }
212         }
213 
214         private boolean get() throws IOException {
215 
216             connection = openConnection(source);
217 
218             if (connection == null) {
219                 return false;
220             }
221 
222             boolean downloadSucceeded = downloadFile();
223 
224             //if (and only if) the use file time option is set, then
225             //the saved file now has its timestamp set to that of the
226             //downloaded file
227             if (downloadSucceeded && useTimestamp) {
228                 updateTimeStamp();
229             }
230 
231             return downloadSucceeded;
232         }
233

236             // Argh, github does this...

241 //                        + aSource.getProtocol() + " to " + aDest.getProtocol()
242 //                        + ". Protocol switch unsafe, not allowed.";
243 //                throw new IOException(message);

246             redirections++;
247             if (redirections > 5) {

249                 throw new IOException(message);
250             }

256         private URLConnection openConnection(URL aSource) throws IOException {
257 
258             // set up the URL connection
259             URLConnection connection = aSource.openConnection();
260             // modify the headers
261             // NB: things like user authentication could go in here too.
262             if (hasTimestamp) {
263                 connection.setIfModifiedSince(timestamp);
264             }
265 
266             if (connection instanceof HttpURLConnection) {
267                 ((HttpURLConnection) connection).setInstanceFollowRedirects(false);
268                 ((HttpURLConnection) connection).setUseCaches(true);
269                 ((HttpURLConnection) connection).setConnectTimeout(5000);
270             }
271             connection.setRequestProperty("ES-Version", Version.CURRENT.toString());
272             connection.setRequestProperty("User-Agent", "elasticsearch-plugin-manager");
273 
274             // connect to the remote site (may take some time)
275             connection.connect();
276 
277             // First check on a 301 / 302 (moved) response (HTTP only)
278             if (connection instanceof HttpURLConnection) {
279                 HttpURLConnection httpConnection = (HttpURLConnection) connection;
280                 int responseCode = httpConnection.getResponseCode();
281                 if (responseCode == HttpURLConnection.HTTP_MOVED_PERM ||
282                         responseCode == HttpURLConnection.HTTP_MOVED_TEMP ||
283                         responseCode == HttpURLConnection.HTTP_SEE_OTHER) {
284                     String newLocation = httpConnection.getHeaderField("Location");
285                     String message = aSource
286                             + (responseCode == HttpURLConnection.HTTP_MOVED_PERM ? " permanently"
287                             : "") + " moved to " + newLocation;
288                     URL newURL = new URL(newLocation);
289                     if (!redirectionAllowed(aSource, newURL)) {
290                         return null;
291                     }
292                     return openConnection(newURL);
293                 }
294                 // next test for a 304 result (HTTP only)
295                 long lastModified = httpConnection.getLastModified();
296                 if (responseCode == HttpURLConnection.HTTP_NOT_MODIFIED
297                         || (lastModified != 0 && hasTimestamp && timestamp >= lastModified)) {
298                     // not modified so no file download. just return
299                     // instead and trace out something so the user
300                     // doesn't think that the download happened when it
301                     // didn't
302                     return null;
303                 }
304                 // test for 401 result (HTTP only)
305                 if (responseCode == HttpURLConnection.HTTP_UNAUTHORIZED) {
306                     String message = "HTTP Authorization failure";
307                     throw new IOException(message);
308                 }
309             }
310 
311             //REVISIT: at this point even non HTTP connections may
312             //support the if-modified-since behaviour -we just check
313             //the date of the content and skip the write if it is not
314             //newer. Some protocols (FTP) don't include dates, of
315             //course.
316             return connection;
317         }
318 
319         private boolean downloadFile() throws FileNotFoundException, IOException {
320             IOException lastEx = null;
321             for (int i = 0; i < 3; i++) {
322                 // this three attempt trick is to get round quirks in different
323                 // Java implementations. Some of them take a few goes to bind
324                 // property; we ignore the first couple of such failures.
325                 try {
326                     is = connection.getInputStream();
327                     break;
328                 } catch (IOException ex) {
329                     lastEx = ex;
330                 }
331             }
332             if (is == null) {
333                 throw new IOException("Can't get " + source + " to " + dest, lastEx);
334             }
335 
336             os = new FileOutputStream(dest);
337             progress.beginDownload();
338             boolean finished = false;
339             try {
340                 byte[] buffer = new byte[1024 * 100];
341                 int length;
342                 while (!isInterrupted() && (length = is.read(buffer)) >= 0) {
343                     os.write(buffer, 0, length);
344                     progress.onTick();
345                 }
346                 finished = !isInterrupted();
347             } finally {
348                 if (!finished) {
349                     // we have started to (over)write dest, but failed.
350                     // Try to delete the garbage we'd otherwise leave
351                     // behind.
352                     IOUtils.closeWhileHandlingException(os, is);
353                     dest.delete();
354                 } else {
355                     IOUtils.close(os, is);
356                 }
357             }
358             progress.endDownload();
359             return true;
360         }
361 
362         private void updateTimeStamp() {
363             long remoteTimestamp = connection.getLastModified();
364             if (remoteTimestamp != 0) {
365                 dest.setLastModified(remoteTimestamp);
366             }
367         }
368 
369         /**
370          * Has the download completed successfully?
371          * <p/>
372          * <p>Re-throws any exception caught during executaion.</p>
373          */
374         boolean wasSuccessful() throws IOException {
375             if (ioexception != null) {
376                 throw ioexception;
377             }
378             return success;
379         }
380 
381         /**
382          * Closes streams, interrupts the download, may delete the
383          * output file.
384          */
385         void closeStreams() throws IOException {
386             interrupt();
387             if (success) {
388                 IOUtils.close(is, os);
389             } else {
390                 IOUtils.closeWhileHandlingException(is, os);
391                 if (dest != null && dest.exists()) {
392                     dest.delete();
393                 }
394             }
395         }
396     }
