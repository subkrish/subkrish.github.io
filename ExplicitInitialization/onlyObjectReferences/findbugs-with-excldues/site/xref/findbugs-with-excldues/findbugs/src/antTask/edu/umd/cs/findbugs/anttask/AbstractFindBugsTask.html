234     @Override
235     public void execute() throws BuildException {
236         checkParameters();
237         try {
238             execFindbugs();
239         } catch (BuildException e) {
240             // log("Oops: " + e.getMessage());
241             if (errorProperty != null) {
242                 getProject().setProperty(errorProperty, "true");
243             }
244             if (failOnError) {
245                 throw e;
246             }
247         }
248     }

250     /**
251      * Check that all required attributes have been set.
252      */
253     protected void checkParameters() {
254         if (homeDir == null && classpath == null) {
255             throw new BuildException("either home attribute or " + "classpath attributes "
256                     + " must be defined for task <" + getTaskName() + "/>", getLocation());
257         }
258 
259         if (pluginList != null) {
260             // Make sure that all plugins are actually Jar files.
261             String[] pluginFileList = pluginList.list();
262             for (String pluginFile : pluginFileList) {
263                 if (!pluginFile.endsWith(".jar")) {
264                     throw new BuildException("plugin file " + pluginFile + " is not a Jar file " + "in task <" + getTaskName()
265                             + "/>", getLocation());
266                 }
267             }
268         }
269 
270         for (SystemProperty systemProperty : systemPropertyList) {
271             if (systemProperty.getName() == null || systemProperty.getValue() == null) {
272                 throw new BuildException("systemProperty elements must have name and value attributes");
273             }
274         }
275     }

277     /**
278      * Create the FindBugs engine (the Java process that will run whatever
279      * FindBugs-related program this task is going to execute).
280      */
281     protected void createFindbugsEngine() {
282         findbugsEngine = new Java();
283         findbugsEngine.setProject(getProject());
284         findbugsEngine.setTaskName(getTaskName());
285         findbugsEngine.setFork(true);
286         if (jvm.length() > 0) {
287             findbugsEngine.setJvm(jvm);
288         }
289         findbugsEngine.setTimeout(timeout);
290 
291         if (debug) {
292             jvmargs = jvmargs + " -Dfindbugs.debug=true";
293         }
294         jvmargs = jvmargs + " -Dfindbugs.hostApp=FBAntTask";
295         findbugsEngine.createJvmarg().setLine(jvmargs);
296 
297         // Add JVM arguments for system properties
298         for (SystemProperty systemProperty : systemPropertyList) {
299             String jvmArg = "-D" + systemProperty.getName() + "=" + systemProperty.getValue();
300             findbugsEngine.createJvmarg().setValue(jvmArg);
301         }
302 
303         if (homeDir != null) {
304             // Use findbugs.home to locate findbugs.jar and the standard
305             // plugins. This is the usual means of initialization.
306             File findbugsLib = new File(homeDir, "lib");
307             if (!findbugsLib.exists() && "lib".equals(homeDir.getName())) {
308                 findbugsLib = homeDir;
309                 homeDir = homeDir.getParentFile();
310             }
311             File findbugsLibFindBugs = new File(findbugsLib, "findbugs.jar");
312             // log("executing using home dir [" + homeDir + "]");
313             if (findbugsLibFindBugs.exists()) {
314                 findbugsEngine.setClasspath(new Path(getProject(), findbugsLibFindBugs.getPath()));
315             } else {
316                 throw new IllegalArgumentException("Can't find findbugs.jar in " + findbugsLib);
317             }
318             findbugsEngine.createJvmarg().setValue("-Dfindbugs.home=" + homeDir.getPath());
319         } else {
320             // Use an explicitly specified classpath and list of plugin Jars
321             // to initialize. This is useful for other tools which may have
322             // FindBugs installed using a non-standard directory layout.
323 
324             findbugsEngine.setClasspath(classpath);
325         }
326         if (pluginList != null) {
327             addArg("-pluginList");
328             addArg(pluginList.toString());
329         }
330         // Set the main class to be whatever the subclass's constructor
331         // specified.
332         findbugsEngine.setClassname(mainClass);
333     }

360     /**
361      * Create a new JVM to do the work.
362      *
363      * @since Ant 1.5
364      */
365     private void execFindbugs() throws BuildException {
366 
367         System.out.println("Executing findbugs " + this.getClass().getSimpleName() + " from ant task");
368         createFindbugsEngine();
369         configureFindbugsEngine();
370 
371         beforeExecuteJavaProcess();
372 
373         if (getDebug()) {
374             log(getFindbugsEngine().getCommandLine().describeCommand());
375         }
376 
377         /*
378          * set property containing return code of child process using a task
379          * identifier and a UUID to ensure exit code corresponds to this
380          * execution (the base Ant Task won't overwrite return code once it's
381          * been set, so unique identifiers must be used for each execution if we
382          * want to get the exit code)
383          */
384         String execReturnCodeIdentifier = execResultProperty + "." + UUID.randomUUID().toString();
385         getFindbugsEngine().setResultProperty(execReturnCodeIdentifier);
386 
387         /*
388          * if the execution fails, we'll report it ourself -- prevent the
389          * underlying Ant Java object from throwing an exception
390          */
391         getFindbugsEngine().setFailonerror(false);
392         try {
393             getFindbugsEngine().execute();
394         } catch (BuildException be) {
395             // setFailonerror(false) should ensure that this doesn't happen,
396             // but...
397             log(be.toString());
398         }
399         String returnProperty = getFindbugsEngine().getProject().getProperty(execReturnCodeIdentifier);
400         int rc = returnProperty == null ? 0 : Integer.parseInt(returnProperty);
401 
402         afterExecuteJavaProcess(rc);
403     }
