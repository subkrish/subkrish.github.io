39  /**
40   * A simple BugReporter which simply prints the formatted message to the output
41   * stream.
42   */
43  public class PrintingBugReporter extends TextUIBugReporter {
44      private final HashSet<BugInstance> seenAlready = new HashSet<BugInstance>();
45  
46      @Override
47      public void observeClass(ClassDescriptor classDescriptor) {
48          // Don't need to do anything special, since we won't be
49          // reporting statistics.
50      }
51  
52      @Override
53      protected void doReportBug(BugInstance bugInstance) {
54          if (seenAlready.add(bugInstance)) {
55              printBug(bugInstance);
56              notifyObservers(bugInstance);
57          }
58      }
59  
60      @Override
61      public void finish() {
62          outputStream.close();
63      }
64  
65      class PrintingCommandLine extends CommandLine {
66          private String stylesheet = null;
67  
68          private boolean annotationUploadFormat = false;
69  
70          private int maxRank = BugRanker.VISIBLE_RANK_MAX;
71  
72          private int summarizeMaxRank = maxRank;
73  
74          private final Project project;
75  
76          private boolean setExitCode;
77  
78          public PrintingCommandLine() {
79              project = new Project();
80              addSwitch("-longBugCodes", "use long bug codes when generating text");
81              addSwitch("-rank", "list rank when generating text");
82              addOption("-maxRank", "max rank", "only list bugs of this rank or less");
83              addOption("-summarizeMaxRank", "max rank", "summary bugs with of this rank or less");
84              addSwitch("-designations", "report user designations for each bug");
85              addSwitch("-history", "report first and last versions for each bug");
86              addSwitch("-applySuppression", "exclude any bugs that match suppression filters");
87              addSwitch("-annotationUpload", "generate annotations in upload format");
88              addSwitchWithOptionalExtraPart("-html", "stylesheet", "Generate HTML output (default stylesheet is default.xsl)");
89              addOption("-pluginList", "jar1[" + File.pathSeparator + "jar2...]", "specify list of plugin Jar files to load");
90              addSwitch("-exitcode", "set exit code of process");
91          }
92  
93          public @Nonnull
94          Project getProject() {
95              return project;
96          }
97  
98          @Override
99          protected void handleOption(String option, String optionExtraPart) throws IOException {
100             if ("-longBugCodes".equals(option)) {
101                 setUseLongBugCodes(true);
102             } else if ("-rank".equals(option)) {
103                 setShowRank(true);
104             } else if ("-designations".equals(option)) {
105                 setReportUserDesignations(true);
106             } else if ("-applySuppression".equals(option)) {
107                 setApplySuppressions(true);
108             } else if ("-history".equals(option)) {
109                 setReportHistory(true);
110             } else if ("-annotationUpload".equals(option)) {
111                 annotationUploadFormat = true;
112             } else if ("-html".equals(option)) {
113                 if (!"".equals(optionExtraPart)) {
114                     stylesheet = optionExtraPart;
115                 } else {
116                     stylesheet = "default.xsl";
117                 }
118             } else if ("-exitcode".equals(option)) {
119                 setExitCode = true;
120             } else {
121                 throw new IllegalArgumentException("Unknown option '" + option + "'");
122             }
123         }
124 
125         @Override
126         protected void handleOptionWithArgument(String option, String argument) throws IOException {
127             if ("-pluginList".equals(option)) {
128                 String pluginListStr = argument;
129                 Map<String, Boolean> customPlugins = getProject().getConfiguration().getCustomPlugins();
130                 StringTokenizer tok = new StringTokenizer(pluginListStr, File.pathSeparator);
131                 while (tok.hasMoreTokens()) {
132                     File file = new File(tok.nextToken());
133                     Boolean enabled = Boolean.valueOf(file.isFile());
134                     customPlugins.put(file.getAbsolutePath(), enabled);
135                     if(enabled.booleanValue()) {
136                         try {
137                             Plugin.loadCustomPlugin(file, getProject());
138                         } catch (PluginException e) {
139                             throw new IllegalStateException("Failed to load plugin " +
140                                     "specified by the '-pluginList', file: " + file, e);
141                         }
142                     }
143                 }
144             } else if ("-maxRank".equals(option)) {
145                 maxRank = Integer.parseInt(argument);
146             } else if ("-summarizeMaxRank".equals(option)) {
147                 summarizeMaxRank = Integer.parseInt(argument);
148             } else {
149                 throw new IllegalStateException();
150             }
151         }
152     }

154     public static void main(String[] args) throws Exception {
155 
156         FindBugs.setNoAnalysis();
157         PrintingBugReporter reporter = new PrintingBugReporter();
158         PrintingCommandLine commandLine = reporter.new PrintingCommandLine();
159 
160         int argCount = commandLine.parse(args, 0, 2, "Usage: " + PrintingCommandLine.class.getName()
161                 + " [options] [<xml results> [<test results]] ");
162 
163 
164         if (commandLine.stylesheet != null) {
165             // actually do xsl via HTMLBugReporter instead of
166             // PrintingBugReporter
167             xslt(commandLine.stylesheet, reporter.isApplySuppressions(), args, argCount);
168             return;
169         }
170 
171         SortedBugCollection bugCollection = new SortedBugCollection(commandLine.getProject());
172         if (argCount < args.length) {
173             bugCollection.readXML(args[argCount++]);
174         } else {
175             bugCollection.readXML(System.in);
176         }
177 
178         if (argCount < args.length) {
179             reporter.setOutputStream(UTF8.printStream(new FileOutputStream(args[argCount++]), true));
180         }
181         boolean bugsReported = false;
182         RuntimeException storedException = null;
183         if (commandLine.annotationUploadFormat) {
184             bugCollection.computeBugHashes();
185             for (Iterator<BugInstance> i = bugCollection.iterator(); i.hasNext();) {
186                 BugInstance warning = i.next();
187                 try {
188                     String fHash = "fb-" + warning.getInstanceHash() + "-" + warning.getInstanceOccurrenceNum() + "-"
189                             + warning.getInstanceOccurrenceMax();
190 
191                     System.out.print("#" + fHash);
192                     String key = warning.getUserDesignationKey();
193                     if (key.equals(BugDesignation.UNCLASSIFIED) || "NEEDS_FURTHER_STUDY".equals(key)) {
194                         System.out.print("#-1#" + key);
195                     } else if ("MUST_FIX".equals(key) || "SHOULD_FIX".equals(key) || "I_WILL_FIX".equals(key)) {
196                         System.out.print("#7#" + key);
197                     } else {
198                         System.out.print("#0#" + key);
199                     }
200                     SourceLineAnnotation sourceLine = warning.getPrimarySourceLineAnnotation();
201                     System.out.println("#" + sourceLine.getSourceFile() + "#" + sourceLine.getStartLine());
202                     System.out.println(warning.getAnnotationText());
203                 } catch (RuntimeException e) {
204                     if (storedException == null) {
205                         storedException = e;
206                     }
207                 }
208             }
209         } else {
210 
211             Bag<String> lowRank = new Bag<String>(new TreeMap<String, Integer>());
212             for (BugInstance warning : bugCollection.getCollection()) {
213                 if (!reporter.isApplySuppressions() || !bugCollection.getProject().getSuppressionFilter().match(warning)) {
214                     int rank = warning.getBugRank();
215                     BugPattern pattern = warning.getBugPattern();
216                     if (rank <= commandLine.maxRank) {
217                         bugsReported = true;
218                         try {
219                             reporter.printBug(warning);
220                         } catch (RuntimeException e) {
221                             if (storedException == null) {
222                                 storedException = e;
223                             }
224                         }
225                     } else if (rank <= commandLine.summarizeMaxRank) {
226                         bugsReported = true;
227                         lowRank.add(pattern.getCategory());
228                     }
229 
230                 }
231             }
232 
233             reporter.finish();
234             for (Map.Entry<String, Integer> e : lowRank.entrySet()) {
235                 System.out.printf("%4d low ranked %s issues%n", e.getValue(),
236                         I18N.instance().getBugCategoryDescription(e.getKey()));
237             }
238 
239         }
240         if(commandLine.setExitCode){
241             int exitCode = 0;
242             System.err.println("Calculating exit code...");
243             if (storedException != null) {
244                 exitCode |= ExitCodes.ERROR_FLAG;
245                 System.err.println("Setting 'errors encountered' flag (" + ExitCodes.ERROR_FLAG + ")");
246                 storedException.printStackTrace(System.err);
247             }
248             if (bugsReported) {
249                 exitCode |= ExitCodes.BUGS_FOUND_FLAG;
250                 System.err.println("Setting 'bugs found' flag (" + ExitCodes.BUGS_FOUND_FLAG + ")");
251             }
252             System.err.println("Exit code set to: " + exitCode);
253             System.exit(exitCode);
254         } else
255             if (storedException != null) {
256                 throw storedException;
257             }
258 
259     }

261     public static void xslt(String stylesheet, boolean applySuppression, String[] args, int argCount) throws Exception {
262         Project proj = new Project();
263         HTMLBugReporter reporter = new HTMLBugReporter(proj, stylesheet);
264         BugCollection bugCollection = reporter.getBugCollection();
265 
266         bugCollection.setApplySuppressions(applySuppression);
267         if (argCount < args.length) {
268             bugCollection.readXML(args[argCount++]);
269         } else {
270             bugCollection.readXML(System.in);
271         }
272 
273         if (argCount < args.length) {
274             reporter.setOutputStream(UTF8.printStream(new FileOutputStream(args[argCount++]), true));
275         }
276 
277         reporter.finish();
278         Exception e = reporter.getFatalException();
279         if (e != null) {
280             throw e;
281         }
282     }

287         return null;
288     }
