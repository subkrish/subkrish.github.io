241     /**
242      * Sets the title of the source tabs for either docking or non-docking
243      * versions.
244      */
245     void setSourceTab(String title, @CheckForNull BugInstance bug) {
246         JComponent label = mainFrame.getGuiLayout().getSourceViewComponent();
247         if (label != null) {
248             URL u = null;
249             if (bug != null) {
250                 Cloud plugin = mainFrame.getBugCollection().getCloud();
251                 if (plugin.supportsSourceLinks()) {
252                     u = plugin.getSourceLink(bug);
253                 }
254             }
255             if (u != null) {
256                 addLink(label, u);
257             } else {
258                 removeLink(label);
259             }
260 
261         }
262         mainFrame.getGuiLayout().setSourceTitle(title);
263     }

265     private void addLink(JComponent component, URL source) {
266         this.sourceLink = source;
267         component.setEnabled(true);
268         if (!listenerAdded) {
269             listenerAdded = true;
270             component.addMouseListener(new MouseAdapter() {
271                 @Override
272                 public void mouseClicked(MouseEvent e) {
273                     URL u = sourceLink;
274                     if (u != null) {
275                         LaunchBrowser.showDocument(u);
276                     }
277 
278                 }
279             });
280         }
281         component.setCursor(new Cursor(Cursor.HAND_CURSOR));
282         Cloud plugin = mainFrame.getBugCollection().getCloud();
283         if (plugin != null) {
284             component.setToolTipText(plugin.getSourceLinkToolTip(null));
285         }
286 
287     }

300     Component bugSummaryComponent(BugAnnotation value, BugInstance bug) {
301         JLabel label = new JLabel();
302         label.setFont(label.getFont().deriveFont(Driver.getFontSize()));
303         label.setFont(label.getFont().deriveFont(Font.PLAIN));
304         label.setForeground(Color.BLACK);
305         ClassAnnotation primaryClass = bug.getPrimaryClass();
306 
307         String sourceCodeLabel = L10N.getLocalString("summary.source_code", "source code.");
308         String summaryLine = L10N.getLocalString("summary.line", "Line");
309         String summaryLines = L10N.getLocalString("summary.lines", "Lines");
310         String clickToGoToText = L10N.getLocalString("tooltip.click_to_go_to", "Click to go to");
311         if (value instanceof SourceLineAnnotation) {
312             final SourceLineAnnotation link = (SourceLineAnnotation) value;
313             if (sourceCodeExists(link)) {
314                 String srcStr = "";
315                 int start = link.getStartLine();
316                 int end = link.getEndLine();
317                 if (start < 0 && end < 0) {
318                     srcStr = sourceCodeLabel;
319                 } else if (start == end) {
320                     srcStr = " [" + summaryLine + " " + start + "]";
321                 } else if (start < end) {
322                     srcStr = " [" + summaryLines + " " + start + " - " + end + "]";
323                 }
324 
325                 label.setToolTipText(clickToGoToText + " " + srcStr);
326 
327                 label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
328             }
329 
330             label.setText(link.toString());
331         } else if (value instanceof BugAnnotationWithSourceLines) {
332             BugAnnotationWithSourceLines note = (BugAnnotationWithSourceLines) value;
333             final SourceLineAnnotation link = note.getSourceLines();
334             String srcStr = "";
335             if (link != null && sourceCodeExists(link)) {
336                 int start = link.getStartLine();
337                 int end = link.getEndLine();
338                 if (start < 0 && end < 0) {
339                     srcStr = sourceCodeLabel;
340                 } else if (start == end) {
341                     srcStr = " [" + summaryLine + " " + start + "]";
342                 } else if (start < end) {
343                     srcStr = " [" + summaryLines + " " + start + " - " + end + "]";
344                 }
345 
346                 if (!"".equals(srcStr)) {
347                     label.setToolTipText(clickToGoToText + " " + srcStr);
348                     label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
349                 }
350             }
351             String noteText;
352             if (note == bug.getPrimaryMethod() || note == bug.getPrimaryField()) {
353                 noteText = note.toString();
354             } else {
355                 noteText = note.toString(primaryClass);
356             }
357             if (!srcStr.equals(sourceCodeLabel)) {
358                 label.setText(noteText + srcStr);
359             } else {
360                 label.setText(noteText);
361             }
362         } else {
363             label.setText(value.toString(primaryClass));
364         }
365 
366         return label;
367     }

369     /**
370      * Creates bug summary component. If obj is a string will create a JLabel
371      * with that string as it's text and return it. If obj is an annotation will
372      * return a JLabel with the annotation's toString(). If that annotation is a
373      * SourceLineAnnotation or has a SourceLineAnnotation connected to it and
374      * the source file is available will attach a listener to the label.
375      */
376     public Component bugSummaryComponent(String str, BugInstance bug) {
377         JLabel label = new JLabel();
378         label.setFont(label.getFont().deriveFont(Driver.getFontSize()));
379         label.setFont(label.getFont().deriveFont(Font.PLAIN));
380         label.setForeground(Color.BLACK);
381 
382         label.setText(str);
383 
384         SourceLineAnnotation link = bug.getPrimarySourceLineAnnotation();
385         if (link != null) {
386             label.addMouseListener(new BugSummaryMouseListener(bug, label, link));
387         }
388 
389         return label;
390     }

392     private boolean sourceCodeExists(@Nonnull SourceLineAnnotation note) {
393         try {
394             mainFrame.getProject().getSourceFinder().findSourceFile(note);
395         } catch (FileNotFoundException e) {
396             return false;
397         } catch (IOException e) {
398             return false;
399         }
400         return true;
401     }

442             }

519     /**
520      * Listens for when cursor is over the label and when it is clicked. When
521      * the cursor is over the label will make the label text blue and the cursor
522      * the hand cursor. When clicked will take the user to the source code tab
523      * and to the lines of code connected to the SourceLineAnnotation.
524      *
525      * @author Kristin Stephens
526      *
527      */
528     private class BugSummaryMouseListener extends MouseAdapter {
529         private final BugInstance bugInstance;
530 
531         private final JLabel label;
532 
533         private final SourceLineAnnotation note;
534 
535         BugSummaryMouseListener(@Nonnull BugInstance bugInstance, @Nonnull JLabel label, @Nonnull SourceLineAnnotation link) {
536             this.bugInstance = bugInstance;
537             this.label = label;
538             this.note = link;
539         }
540 
541         @Override
542         public void mouseClicked(MouseEvent e) {
543             mainFrame.getSourceCodeDisplayer().displaySource(bugInstance, note);
544         }
545 
546         @Override
547         public void mouseEntered(MouseEvent e) {
548             label.setForeground(Color.blue);
549             mainFrame.setCursor(new Cursor(Cursor.HAND_CURSOR));
550         }
551 
552         @Override
553         public void mouseExited(MouseEvent e) {
554             label.setForeground(Color.black);
555             mainFrame.setCursor(new Cursor(Cursor.DEFAULT_CURSOR));
556         }
557     }
