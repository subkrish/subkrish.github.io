87      public void addDefaultAnnotation(Target target, String c, AnnotationEnum n) {
88          if (!defaultAnnotation.containsKey(target)) {
89              return;
90          }
91          if (DEBUG) {
92              System.out.println("Default annotation " + target + " " + c + " " + n);
93          }
94          defaultAnnotation.get(target).put(c, n);
95          seen.add(n);
96      }

107     @CheckForNull
108     public AnnotationEnum getResolvedAnnotation(Object o, boolean getMinimal) {
109         if (o instanceof XMethod) {
110             XMethod m = (XMethod) o;
111             if (m.getName().startsWith("access$")) {
112                 InnerClassAccessMap icam = AnalysisContext.currentAnalysisContext().getInnerClassAccessMap();
113                 try {
114                     InnerClassAccess ica = icam.getInnerClassAccess(m.getClassName(), m.getName());
115                     if (ica != null && ica.isLoad()) {
116                         o = ica.getField();
117                     }
118                 } catch (ClassNotFoundException e) {
119                     AnalysisContext.reportMissingClass(e);
120                     return null;
121                 }
122 
123             }
124         }
125         Map<Object, AnnotationEnum> cache;
126         if (getMinimal) {
127             cache = cachedMinimal;
128         } else {
129             cache = cachedMaximal;
130         }
131 
132         if (cache.containsKey(o)) {
133             return cache.get(o);
134         }
135         AnnotationEnum n = getUncachedResolvedAnnotation(o, getMinimal);
136         if (DEBUG) {
137             System.out.println("TTT: " + o + " " + n);
138         }
139         cache.put(o, n);
140         return n;
141     }

147     @CheckForNull
148     public AnnotationEnum getUncachedResolvedAnnotation(final Object o, boolean getMinimal) {
149 
150         AnnotationEnum n = getDirectAnnotation(o);
151         if (n != null) {
152             return n;
153         }
154 
155         try {
156 
157             String className;
158             Target kind;
159             boolean isParameterToInitMethodofAnonymousInnerClass = false;
160             boolean isSyntheticMethod = false;
161             if (o instanceof XMethod || o instanceof XMethodParameter) {
162 
163                 XMethod m;
164                 if (o instanceof XMethod) {
165                     m = (XMethod) o;
166                     isSyntheticMethod = m.isSynthetic();
167                     kind = Target.METHOD;
168                     className = m.getClassName();
169                 } else if (o instanceof XMethodParameter) {
170                     m = ((XMethodParameter) o).getMethod();
171                     // Don't
172                     isSyntheticMethod = m.isSynthetic();
173                     className = m.getClassName();
174                     kind = Target.PARAMETER;
175                     if ("<init>".equals(m.getName())) {
176                         int i = className.lastIndexOf('$');
177                         if (i + 1 < className.length() && Character.isDigit(className.charAt(i + 1))) {
178                             isParameterToInitMethodofAnonymousInnerClass = true;
179                         }
180                     }
181                 } else {
182                     throw new IllegalStateException("impossible");
183                 }
184 
185                 if (!m.isStatic() && !"<init>".equals(m.getName())) {
186                     JavaClass c = Repository.lookupClass(className);
187                     // get inherited annotation
188                     TreeSet<AnnotationEnum> inheritedAnnotations = new TreeSet<AnnotationEnum>();
189                     if (c.getSuperclassNameIndex() > 0) {
190 
191                         n = lookInOverriddenMethod(o, c.getSuperclassName(), m, getMinimal);
192                         if (n != null) {
193                             inheritedAnnotations.add(n);
194                         }
195                     }
196                     for (String implementedInterface : c.getInterfaceNames()) {
197                         n = lookInOverriddenMethod(o, implementedInterface, m, getMinimal);
198                         if (n != null) {
199                             inheritedAnnotations.add(n);
200                         }
201                     }
202                     if (DEBUG) {
203                         System.out.println("# of inherited annotations : " + inheritedAnnotations.size());
204                     }
205                     if (!inheritedAnnotations.isEmpty()) {
206                         if (inheritedAnnotations.size() == 1) {
207                             return inheritedAnnotations.first();
208                         }
209                         if (!getMinimal) {
210                             return inheritedAnnotations.last();
211                         }
212 
213                         AnnotationEnum min = inheritedAnnotations.first();
214                         if (min.getIndex() == 0) {
215                             inheritedAnnotations.remove(min);
216                             min = inheritedAnnotations.first();
217                         }
218                         return min;
219                     }
220                     // check to see if method is defined in this class;
221                     // if not, on't consider default annotations
222                     if (!classDefinesMethod(c, m)) {
223                         return null;
224                     }
225                     if (DEBUG) {
226                         System.out.println("looking for default annotations: " + c.getClassName() + " defines " + m);
227                     }
228                 } // if not static
229             } // associated with method
230             else if (o instanceof XField) {
231 
232                 className = ((XField) o).getClassName();
233                 kind = Target.FIELD;
234             } else if (o instanceof String) {
235                 assert false;
236                 className = (String) o;
237                 kind = Target.CLASS;
238             } else {
239                 throw new IllegalArgumentException("Can't look up annotation for " + o.getClass().getName());
240             }
241 
242             // <init> method parameters for inner classes don't inherit default
243             // annotations
244             // since some of them are synthetic
245             if (isParameterToInitMethodofAnonymousInnerClass) {
246                 return null;
247             }
248 
249             // synthetic elements should not inherit default annotations
250             if (isSyntheticMethod) {
251                 return null;
252             }
253             try {
254                 XClass c = Global.getAnalysisCache().getClassAnalysis(XClass.class,
255                         DescriptorFactory.createClassDescriptorFromDottedClassName(className));
256 
257                 if (c != null && c.isSynthetic()) {
258                     return null;
259                 }
260             } catch (CheckedAnalysisException e) {
261                 assert true;
262             }
263 
264             // look for default annotation
265             n = defaultAnnotation.get(kind).get(className);
266             if (DEBUG) {
267                 System.out.println("Default annotation for " + kind + " is " + n);
268             }
269             if (n != null) {
270                 return n;
271             }
272 
273             n = defaultAnnotation.get(Target.ANY).get(className);
274             if (DEBUG) {
275                 System.out.println("Default annotation for any is " + n);
276             }
277             if (n != null) {
278                 return n;
279             }
280 
281             int p = className.lastIndexOf('.');
282             className = className.substring(0, p + 1) + "package-info";
283             n = defaultAnnotation.get(kind).get(className);
284             if (DEBUG) {
285                 System.out.println("Default annotation for " + kind + " is " + n);
286             }
287             if (n != null) {
288                 return n;
289             }
290 
291             n = defaultAnnotation.get(Target.ANY).get(className);
292             if (DEBUG) {
293                 System.out.println("Default annotation for any is " + n);
294             }
295             if (n != null) {
296                 return n;
297             }
298 
299             return n;
300         } catch (ClassNotFoundException e) {
301             AnalysisContext.reportMissingClass(e);
302             return null;
303         }
304 
305     }

311     private boolean classDefinesMethod(JavaClass c, XMethod m) {
312         for (Method definedMethod : c.getMethods()) {
313             if (definedMethod.getName().equals(m.getName()) && definedMethod.getSignature().equals(m.getSignature())
314                     && definedMethod.isStatic() == m.isStatic()) {
315                 return true;
316             }
317         }
318         return false;
319     }

321     private AnnotationEnum lookInOverriddenMethod(final Object originalQuery, String classToLookIn, XMethod originalMethod,
322             boolean getMinimal) {
323         try {
324             AnnotationEnum n;
325             // Look in supermethod
326             XMethod superMethod = XFactory.createXMethod(classToLookIn, originalMethod.getName(), originalMethod.getSignature(),
327                     originalMethod.isStatic());
328             if (!superMethod.isResolved()) {
329                 return null;
330             }
331             if (DEBUG) {
332                 System.out.println("Looking for overridden method " + superMethod);
333             }
334 
335             Object probe;
336             if (originalQuery instanceof XMethod) {
337                 probe = superMethod;
338             } else if (originalQuery instanceof XMethodParameter) {
339                 probe = new XMethodParameter(superMethod, ((XMethodParameter) originalQuery).getParameterNumber());
340             } else {
341                 throw new IllegalStateException("impossible");
342             }
343 
344             n = getResolvedAnnotation(probe, getMinimal);
345             return n;
346         } catch (RuntimeException e) {
347             AnalysisContext.logError("Exception while looking for annotation of " + originalMethod + "in " + classToLookIn, e);
348             return null;
349         }
350     }

413     private boolean onlyAppliesToReferenceParameters(AnnotationEnum annotation) {
414         // return annotation instanceof NullnessAnnotation; work around JDK bug

418     protected void addMethodParameterAnnotation(String cName, String mName, String mSig, boolean isStatic, int param,
419             AnnotationEnum annotation) {
420         // if (!Subtypes.DO_NOT_USE) {
421         // subtypes.addNamedClass(cName);
422         // }
423         if (addClassOnly) {
424             return;
425         }
426         SignatureParser parser = new SignatureParser(mSig);
427         if (param < 0 || param >= parser.getNumParameters()) {
428             throw new IllegalArgumentException("can't annotation parameter #" + param + " of " + cName + "." + mName + mSig);
429         }
430         String signature = parser.getParameter(param);
431         char firstChar = signature.charAt(0);
432         boolean isReference = firstChar == 'L' || firstChar == '[';
433         if (onlyAppliesToReferenceParameters(annotation) && !isReference) {
434             AnalysisContext.logError("Can't apply " + annotation + " to parameter " + param + " with signature " + signature
435                     + " of " + cName + "." + mName + " : " + mSig);
436             return;
437         }
438         XMethod m = XFactory.createXMethod(cName, mName, mSig, isStatic);
439         addDirectAnnotation(new XMethodParameter(m, param), annotation);
440     }
