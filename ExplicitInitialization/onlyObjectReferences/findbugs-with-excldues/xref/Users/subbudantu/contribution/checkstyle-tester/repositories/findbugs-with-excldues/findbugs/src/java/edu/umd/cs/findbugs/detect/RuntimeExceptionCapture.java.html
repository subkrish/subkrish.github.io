63  /**
64   * RuntimeExceptionCapture
65   *
66   * @author Brian Goetz
67   * @author Bill Pugh
68   * @author David Hovemeyer
69   */
70  public class RuntimeExceptionCapture extends OpcodeStackDetector implements StatelessDetector {
71      private static final boolean DEBUG = SystemProperties.getBoolean("rec.debug");
72  
73      private final BugReporter bugReporter;
74  
75      private final List<ExceptionCaught> catchList = new ArrayList<ExceptionCaught>();
76  
77      private final List<ExceptionThrown> throwList = new ArrayList<ExceptionThrown>();
78  
79      private final BugAccumulator accumulator;
80  
81      private static class ExceptionCaught {
82          public String exceptionClass;
83  
84          public int startOffset, endOffset, sourcePC;
85  
86          public boolean seen = false;
87  
88          public boolean dead = false;
89  
90          public ExceptionCaught(String exceptionClass, int startOffset, int endOffset, int sourcePC) {
91              this.exceptionClass = exceptionClass;
92              this.startOffset = startOffset;
93              this.endOffset = endOffset;
94              this.sourcePC = sourcePC;
95          }
96      }
97  
98      private static class ExceptionThrown {
99          public @DottedClassName
100         String exceptionClass;
101 
102         public int offset;
103 
104         public ExceptionThrown(@DottedClassName String exceptionClass, int offset) {
105             this.exceptionClass = exceptionClass;
106             this.offset = offset;
107         }
108     }
109 
110     public RuntimeExceptionCapture(BugReporter bugReporter) {
111         this.bugReporter = bugReporter;
112         accumulator = new BugAccumulator(bugReporter);
113     }
114 
115     @Override
116     public void visitJavaClass(JavaClass c) {
117         super.visitJavaClass(c);
118         accumulator.reportAccumulatedBugs();
119     }

121     @Override
122     public void visitAfter(Code obj) {
123         for (ExceptionCaught caughtException : catchList) {
124             Set<String> thrownSet = new HashSet<String>();
125             for (ExceptionThrown thrownException : throwList) {
126                 if (thrownException.offset >= caughtException.startOffset && thrownException.offset < caughtException.endOffset) {
127                     thrownSet.add(thrownException.exceptionClass);
128                     if (thrownException.exceptionClass.equals(caughtException.exceptionClass)) {
129                         caughtException.seen = true;
130                     }
131                 }
132             }
133             int catchClauses = 0;
134             if ("java.lang.Exception".equals(caughtException.exceptionClass) && !caughtException.seen) {
135                 // Now we have a case where Exception is caught, but not thrown
136                 boolean rteCaught = false;
137                 for (ExceptionCaught otherException : catchList) {
138                     if (otherException.startOffset == caughtException.startOffset
139                             && otherException.endOffset == caughtException.endOffset) {
140                         catchClauses++;
141                         if ("java.lang.RuntimeException".equals(otherException.exceptionClass)) {
142                             rteCaught = true;
143                         }
144                     }
145                 }
146                 int range = caughtException.endOffset - caughtException.startOffset;
147                 if (!rteCaught) {
148                     int priority = LOW_PRIORITY + 1;
149                     if (range > 300) {
150                         priority--;
151                     } else if (range < 30) {
152                         priority++;
153                     }
154                     if (catchClauses > 1) {
155                         priority++;
156                     }
157                     if (thrownSet.size() > 1) {
158                         priority--;
159                     }
160                     if (caughtException.dead) {
161                         priority--;
162                     }
163                     accumulator.accumulateBug(new BugInstance(this, "REC_CATCH_EXCEPTION", priority).addClassAndMethod(this),
164                             SourceLineAnnotation.fromVisitedInstruction(getClassContext(), this, caughtException.sourcePC));
165                 }
166             }
167         }
168         catchList.clear();
169         throwList.clear();
170     }
171 
172     @Override
173     public void visit(CodeException obj) {
174         try {
175             super.visit(obj);
176             int type = obj.getCatchType();
177             if (type == 0) {
178                 return;
179             }
180             String name = getConstantPool().constantToString(getConstantPool().getConstant(type));
181 
182             ExceptionCaught caughtException = new ExceptionCaught(name, obj.getStartPC(), obj.getEndPC(), obj.getHandlerPC());
183             catchList.add(caughtException);
184 
185             // See if the store that saves the exception object
186             // is alive or dead. We rely on the fact that javac
187             // always (?) emits an ASTORE instruction to save
188             // the caught exception.
189             LiveLocalStoreDataflow dataflow = getClassContext().getLiveLocalStoreDataflow(getMethod());
190             CFG cfg = getClassContext().getCFG(getMethod());
191             Collection<BasicBlock> blockList = cfg.getBlocksContainingInstructionWithOffset(obj.getHandlerPC());
192             for (BasicBlock block : blockList) {
193                 InstructionHandle first = block.getFirstInstruction();
194                 if (first != null && first.getPosition() == obj.getHandlerPC() && first.getInstruction() instanceof ASTORE) {
195                     ASTORE astore = (ASTORE) first.getInstruction();
196                     BitSet liveStoreSet = dataflow.getFactAtLocation(new Location(first, block));
197                     if (!liveStoreSet.get(astore.getIndex())) {
198                         // The ASTORE storing the exception object is dead
199                         if (DEBUG) {
200                             System.out.println("Dead exception store at " + first);
201                         }
202                         caughtException.dead = true;
203                         break;
204                     }
205                 }
206             }
207         } catch (MethodUnprofitableException e) {
208             Method m = getMethod();
209             bugReporter.reportSkippedAnalysis(DescriptorFactory.instance().getMethodDescriptor(getClassName(), getMethodName(),
210                     getMethodSig(), m.isStatic()));
211         } catch (DataflowAnalysisException e) {
212             bugReporter.logError("Error checking for dead exception store", e);
213         } catch (CFGBuilderException e) {
214             bugReporter.logError("Error checking for dead exception store", e);
215         }
216     }

218     @Override
219     public void sawOpcode(int seen) {
220         switch (seen) {
221         case ATHROW:
222             if (stack.getStackDepth() > 0) {
223                 OpcodeStack.Item item = stack.getStackItem(0);
224                 String signature = item.getSignature();
225                 if (signature != null && signature.length() > 0) {
226                     if (signature.startsWith("L")) {
227                         signature = SignatureConverter.convert(signature);
228                     } else {
229                         signature = signature.replace('/', '.');
230                     }
231                     throwList.add(new ExceptionThrown(signature, getPC()));
232                 }
233             }
234             break;
235 
236         case INVOKEVIRTUAL:
237         case INVOKESPECIAL:
238         case INVOKESTATIC:
239             String className = getClassConstantOperand();
240             if (!className.startsWith("[")) {
241                 try {
242                     XClass c = Global.getAnalysisCache().getClassAnalysis(XClass.class,
243                             DescriptorFactory.createClassDescriptor(className));
244                     XMethod m = Hierarchy2.findInvocationLeastUpperBound(c, getNameConstantOperand(), getSigConstantOperand(),
245                             seen == INVOKESTATIC, seen == INVOKEINTERFACE);
246                     if (m == null) {
247                         break;
248                     }
249                     String[] exceptions = m.getThrownExceptions();
250                     if (exceptions != null) {
251                         for (String name : exceptions) {
252                             throwList.add(new ExceptionThrown(ClassName.toDottedClassName(name), getPC()));
253                         }
254                     }
255                 } catch (MissingClassException e) {
256                     bugReporter.reportMissingClass(e.getClassDescriptor());
257                 } catch (CheckedAnalysisException e) {
258                     bugReporter.logError("Error looking up " + className, e);
259                 }
260             }
261             break;
262         default:
263             break;
264         }
265 
266     }
