204     @Override
205     public boolean initialize() throws IOException {
206         // noinspection ConstantConditions
207         if (false && initialized) {
208             LOGGER.warning("Already initialized " + getClass().getSimpleName());
209             return true;
210         }
211 
212         LOGGER.fine("Initializing " + getClass().getSimpleName());
213         setSigninState(SigninState.UNAUTHENTICATED);
214         
215         addStatusListener(new CloudStatusListener() {
216             
217             public void handleStateChange(SigninState oldState, SigninState state) {
218                 if (oldState.canUpload() && !state.canUpload())
219                     evaluationsFromXmlUploader.reset();
220             }
221             
222             public void handleIssueDataDownloadedEvent() {
223                 // TODO Auto-generated method stub
224                 
225             }
226         });
227 
228         try {
229             if (!super.initialize()) {
230                 setSigninState(SigninState.SIGNIN_FAILED);
231                 return false;
232             }
233 
234             boolean networkInitialization = networkClient.initialize();
235             initialized = true;
236             if (networkInitialization) {
237                 setSigninState(SigninState.SIGNED_IN);
238             } else {
239                 // soft init didn't work
240                 setSigninState(SigninState.UNAUTHENTICATED);
241             }
242             
243             return true;
244 
245         } catch (UnknownHostException e) {
246             initialized = true;
247             setSigninState(SigninState.DISCONNECTED);
248             return true;
249         } catch (IOException e) {
250             setSigninState(SigninState.SIGNIN_FAILED);
251             throw e;
252         }
253 
254     }

261     public void signInIfNecessary(@CheckForNull String reason) throws SignInCancelledException {
262         if (couldSignIn()) {
263 
264             if (reason != null) {
265                 IGuiCallback callback = getGuiCallback();
266                 int result = callback.showConfirmDialog(reason, getCloudName(), "Sign in", "Cancel");
267                 if (result != 0) {
268                     setSigninState(SigninState.SIGNIN_DECLINED);
269                     throw new SignInCancelledException();
270                 }
271             }
272             try {
273                 signIn();
274             } catch (Exception e) {
275                 getGuiCallback().showMessageDialog("Could not sign into " + getCloudName() + ": "
276                         + Util.getNetworkErrorMessage(e));
277                 setSigninState(SigninState.SIGNIN_FAILED);
278                 throw new SignInCancelledException(e);
279             }
280 
281         } else if (getSigninState() == SigninState.SIGNING_IN) {
282             // thread that set state to signing in should already hold lock
283             synchronized (signInLock) {
284             }
285         } else if (getSigninState() == SigninState.SIGNED_IN) {
286             // great!
287         }
288     }

333     public void initiateCommunication() {
334         checkInitialized();
335         bugsPopulated();
336 
337         if (communicationInitiated)
338             return;
339 
340         if (!networkClient.ready())
341             return;
342         synchronized (initiationLock) {
343             if (communicationInitiated)
344                 return;
345             communicationInitiated = true;
346             backgroundExecutorService.execute(new Runnable() {
347                 public void run() {
348                     try {
349                         mockFailure();
350                         actuallyCheckBugsAgainstCloud();
351                     } catch (RejectedExecutionException e) {
352                         // I think this only happens on purpose -Keith
353                     } catch (Throwable e) {
354                         if (e instanceof ExecutionException)
355                             e = e.getCause();
356                         String errorMsg = Util.getNetworkErrorMessage(e);
357                         getGuiCallback().showMessageDialog("Could not connect to " + getCloudName()
358                                 + "\n\n" +  errorMsg 
359                                 + "\nsignin status: " + getSigninState());
360                         LOGGER.log(Level.SEVERE, "Error while checking bugs against cloud in background" , e);
361                     }
362                 }
363             });
364 
365         }
366     }

398     private final Object signInLock = new Object();
399 
400     public void signIn() throws IOException {
401         checkInitialized();
402         SigninState oldState = getSigninState();
403         synchronized (signInLock) {
404             if (getSigninState() == SigninState.SIGNED_IN)
405                 return;
406             setSigninState(SigninState.SIGNING_IN);
407             MutableCloudTask task = createTask("Signing into " + getCloudName());
408             try {
409                 try {
410                     if (oldState == SigninState.DISCONNECTED) {
411                         if (networkClient.initialize()) {
412                             networkClient.signIn(true);
413                             networkClient.logIntoCloudForce();
414                             setSigninState(SigninState.SIGNED_IN);
415                         } else {
416                             // soft init didn't work
417                             setSigninState(SigninState.UNAUTHENTICATED);
418                         }
419                     } else {
420                         networkClient.signIn(true);
421                         networkClient.logIntoCloudForce();
422                         setSigninState(SigninState.SIGNED_IN);
423                     }
424                 } catch (UnknownHostException e) {
425                     getGuiCallback().showMessageDialog("Could not connect to " + getCloudName());
426                     setSigninState(SigninState.DISCONNECTED);
427                 } catch (IOException e) {
428                     setSigninState(SigninState.SIGNIN_FAILED);
429                     throw e;
430                 } catch (RuntimeException e) {
431                     setSigninState(SigninState.SIGNIN_FAILED);
432                     throw e;
433                 }
434 
435             } finally {
436                 task.finished();
437             }
438         }
439         if (getSigninState().canDownload()) {
440         initiateCommunication();
441         SigninState newState = getSigninState();
442         if (!oldState.canUpload()  && newState.canUpload()) {
443             evaluationsFromXmlUploader.tryUploadingLocalAnnotations(true);
444         }
445         }
446     }

685     private List<BugDesignation> sortDesignations(Collection<BugDesignation> bugDesignations) {
686         List<BugDesignation> designations = new ArrayList<BugDesignation>(bugDesignations);
687         Collections.sort(designations, new Comparator<BugDesignation>() {
688             public int compare(BugDesignation o1, BugDesignation o2) {
689                 return Util.compare(o1.getTimestamp(), o2.getTimestamp());
690             }
691         });
692         return designations;
693     }

695     private List<Evaluation> sortEvals(List<Evaluation> evaluationsList) {
696         List<Evaluation> evals = new ArrayList<Evaluation>(evaluationsList);
697         Collections.sort(evals, new Comparator<Evaluation>() {
698             public int compare(Evaluation o1, Evaluation o2) {
699                 return Util.compare(o1.getWhen(), o2.getWhen());
700             }
701         });
702         return evals;
703     }

705     volatile int updateInternal = EVALUATION_CHECK_SECS_MIN_SECS;
706     
707     private final class GetUpdatedEvaluations extends TimerTask {
708         @Override
709         public void run() {
710             if (CloudFactory.DEBUG)
711                 System.out.printf("checking for updates from server%n");
712            
713             try {
714                 int count = updateEvaluationsFromServer();
715                 if (count == 0)
716                     updateInternal = Math.min(updateInternal*2, EVALUATION_CHECK_SECS_MAX_SECS);
717                 else
718                     updateInternal = EVALUATION_CHECK_SECS_MIN_SECS;
719             } catch (ServerReturnedErrorCodeException e) {
720                 updateInternal = Math.min(updateInternal*2, EVALUATION_CHECK_SECS_MAX_SECS);
721                 LOGGER.log(Level.WARNING, "Error during periodic review update check");
722                 LOGGER.log(Level.FINEST, "", e);
723             } catch (IOException e) {
724                 updateInternal = Math.min(updateInternal*2, EVALUATION_CHECK_SECS_MAX_SECS);
725                 LOGGER.log(Level.WARNING, "Error during periodic review update check", e);
726                 LOGGER.log(Level.FINEST, "", e);
727             } finally {
728                 scheduleUpdateEvaluationsFromServer();
729             }
730         }
731     }

757     private void actuallyCheckBugsAgainstCloud() throws ExecutionException, InterruptedException {
758         checkInitialized();
759         ConcurrentHashMap<String, BugInstance> bugsByHash = new ConcurrentHashMap<String, BugInstance>();
760 
761         for (BugInstance b : bugCollection.getCollection()) {
762             bugsByHash.put(b.getInstanceHash(), b);
763         }
764 
765         int numBugs = bugsByHash.size();
766         MutableCloudTask task = createTask("Checking " + getCloudName());
767         try {
768             LOGGER.info("Checking " + numBugs + " bugs against the " + getCloudName() + "...");
769 
770             boolean exceptionThrown = true;
771             try {
772                 List<Callable<Object>> tasks = new ArrayList<Callable<Object>>();
773                 ArrayList<String> hashes = new ArrayList<String>(bugsByHash.keySet());
774                 networkClient.generateHashCheckRunnables(task, hashes, tasks, bugsByHash);
775                 executeAndWaitForAll(tasks);
776                 exceptionThrown = false;
777 
778             } catch (InterruptedException e) {
779                 LOGGER.log(Level.WARNING, "Find issues interrupted!", e);
780 
781             } finally {
782                 issueDataDownloaded.countDown();
783                 fireIssueDataDownloadedEvent();
784                 if (exceptionThrown)
785                     fireNewIssuesUploadedEvent();
786             }
787 
788             if (getSigninState() == SigninState.SIGNIN_FAILED) {
789                // TODO: Why would we do this?
790                if (false) 
791                    fireNewIssuesUploadedEvent();
792                 return;
793             }
794         } finally {
795             task.finished();
796         }
797 
798         Collection<BugInstance> newBugs = bugsByHash.values();
799         boolean hasTimestampsToUpdate = !networkClient.getTimestampsToUpdate().isEmpty();
800         boolean hasBugsToUpload = !newBugs.isEmpty();
801         if (!getGuiCallback().isHeadless()) {
802             startEvaluationCheckThread();
803         };
804         String cloudTokenProperty = getCloudTokenProperty();
805         boolean canUpload = cloudTokenProperty != null
806         || !getGuiCallback().isHeadless()
807         || getSigninState() == SigninState.SIGNED_IN;
808         if ((hasBugsToUpload || hasTimestampsToUpdate)) {
809             if (canUpload) {
810                 uploadAndUpdateBugsInBackground(new ArrayList<BugInstance>(newBugs));
811             } else {
812                 setStatusMsg("not able to automatically upload bugs to the " + getCloudName());
813 
814                 fireNewIssuesUploadedEvent();
815             }
816         } else {
817             fireNewIssuesUploadedEvent();
818             setStatusMsg("All " + numBugs + " bugs are already stored in the " + getCloudName());
819         }
820 
821     }

835     private <T> void executeAndWaitForAll(List<Callable<T>> tasks) throws ExecutionException, InterruptedException {
836         if (backgroundExecutorService.isShutdown()) {
837             LOGGER.log(Level.SEVERE, "backgroundExecutor service is shutdown in executeAndWaitForAll");
838             return;
839         }
840 
841         try {
842             List<Future<T>> results = backgroundExecutorService.invokeAll(tasks);
843             for (Future<T> result : results) {
844                 result.get();
845             }
846 
847         } catch (RejectedExecutionException e) {
848             if (backgroundExecutorService.isShutdown())
849                 LOGGER.log(Level.SEVERE, "backgroundExecutor service is shutdown ", e);
850             else if (backgroundExecutorService.isTerminated())
851                 LOGGER.log(Level.SEVERE, "backgroundExecutor service is terminated ", e);
852             else
853                 LOGGER.log(Level.SEVERE, "Rejected execution", e);
854 
855         }
856     }

858     /** package-private for testing */
859     int updateEvaluationsFromServer() throws IOException {
860         checkInitialized();
861         if (!networkClient.ready())
862             return 0;
863         if (issueDataDownloaded.getCount() > 0)
864             return 0;
865         MutableCloudTask task = createTask("Checking " + getCloudName() + " for updates");
866 
867         int count = 0;
868         RecentEvaluations evals;
869         try {
870             int i = 0;
871             do {
872                 evals = networkClient.getRecentEvaluationsFromServer();
873                 count += mergeUpdatedEvaluations(evals);
874                 task.update("found " + count + " so far...", 0);
875             } while (evals.hasAskAgain() && evals.getAskAgain() && ++i < MAX_RECENT_EVALUATION_PAGES);
876 
877         } catch (ServerReturnedErrorCodeException e) {
878             task.failed(e.getMessage());
879             throw e;
880 
881         } catch (IOException e) {
882             if (getSigninState() == SigninState.SIGNED_IN) {
883                 signOut(true);
884                 getGuiCallback().showMessageDialog(
885                         "A network error occurred while checking the " + getCloudName() + " for updates.\n"
886                                 + "\n"
887                                 + "You have been automatically signed out of the Cloud. Any reviews\n"
888                                 + "you add or change will only be stored on your computer if you save the\n"
889                                 + "analysis via the File->Save menu.\n"
890                                 + "\n"
891                                 + "To attempt to sign back in, click the Sign In button on the left.");
892             } else {
893                 task.failed(Util.getNetworkErrorMessage(e));
894             }
895             throw e;
896 
897         } catch (RuntimeException e) {
898             task.failed(e.getClass().getSimpleName() + ": " + e.getMessage());
899             throw e;
900 
901         } finally {
902             task.finished();
903         }
904         if (count > 0)
905             setStatusMsg(getCloudName() + ": found " + count + " updated bug reviews");
906         return count;
907     }

917     private int mergeUpdatedEvaluations(RecentEvaluations evals) {
918         int found = 0;
919         for (Issue updatedIssue : evals.getIssuesList()) {
920             String protoHash = WebCloudProtoUtil.decodeHash(updatedIssue.getHash());
921             BugInstance bugInstance = getBugByHash(protoHash);
922             if (bugInstance == null) {
923                 if (CloudFactory.DEBUG) 
924                     System.out.printf("No match for %s in %s, updated %tc%n", updatedIssue.getBugPattern(), updatedIssue.getPrimaryClass(), 
925                             lastEvaluation(updatedIssue));
926                 continue;
927             }
928             Issue existingIssue = networkClient.getIssueByHash(protoHash);
929             Issue issueToStore;
930             if (existingIssue != null) {
931                 Issue oldIssue = Issue.newBuilder(existingIssue).build();
932                 issueToStore = mergeIssues(existingIssue, updatedIssue);
933                 String newHash = WebCloudProtoUtil.decodeHash(issueToStore.getHash());
934                 assert newHash.equals(protoHash) : newHash + " vs " + protoHash;
935                 if (oldIssue.equals(issueToStore)) {
936                     if (CloudFactory.DEBUG) 
937                         System.out.println("no new information");
938                     continue;
939                 } else {
940                     if (CloudFactory.DEBUG) 
941                         System.out.printf("Got new information for %s in %s @ %tc : %s%n", issueToStore.getBugPattern(), issueToStore.getPrimaryClass() ,
942                                 lastEvaluation(updatedIssue), issueToStore.getHash());
943                 }
944 
945             } else {
946                 issueToStore = updatedIssue;
947                 System.out.printf("Got new issue %s in %s @ %tc : %s%n", issueToStore.getBugPattern(), 
948                         issueToStore.getPrimaryClass() , 
949                         lastEvaluation(updatedIssue), issueToStore.getHash().toStringUtf8());
950                 
951                 
952             }
953             networkClient.storeIssueDetails(protoHash, issueToStore);
954             updateBugInstanceAndNotify(bugInstance);
955             found++;
956         }
957         return found;
958     }

960     private void uploadAndUpdateBugsInBackground(final List<BugInstance> newBugs) {
961         checkInitialized();
962         backgroundExecutorService.execute(new Runnable() {
963             public void run() {
964                 List<Callable<Void>> callables = new ArrayList<Callable<Void>>();
965                 MutableCloudTask taskA = null;
966                 MutableCloudTask taskB = null;
967                 try {
968                     taskA = networkClient.generateUploadRunnables(newBugs, callables);
969                     taskB = networkClient.generateUpdateTimestampRunnables(callables);
970                     executeAndWaitForAll(callables);
971 
972                 } catch (SignInCancelledException e) {
973                     // OK!
974                 } catch (Exception e) {
975                     LOGGER.log(Level.SEVERE, "", e);
976                 } finally {
977                     if (taskA != null)
978                         taskA.finished();
979                     if (taskB != null)
980                         taskB.finished();
981                 }
982                 fireNewIssuesUploadedEvent();
983             }
984         });
985     }

991     private Issue mergeIssues(Issue existingIssue, Issue updatedIssue) {
992         List<Evaluation> allEvaluations = new ArrayList<Evaluation>();
993         allEvaluations.addAll(existingIssue.getEvaluationsList());
994         allEvaluations.addAll(updatedIssue.getEvaluationsList());
995         removeAllButLatestEvaluationPerUser(allEvaluations);
996 
997         return Issue.newBuilder(updatedIssue).clearEvaluations().addAllEvaluations(allEvaluations).build();
998     }

1000     private void removeAllButLatestEvaluationPerUser(List<Evaluation> allEvaluations) {
1001         Set<String> seenUsernames = new HashSet<String>();
1002         for (ListIterator<Evaluation> it = reverseIterator(allEvaluations); it.hasPrevious();) {
1003             Evaluation evaluation = it.previous();
1004             boolean isNewUsername = seenUsernames.add(evaluation.getWho());
1005             if (!isNewUsername)
1006                 it.remove();
1007         }
1008     }
