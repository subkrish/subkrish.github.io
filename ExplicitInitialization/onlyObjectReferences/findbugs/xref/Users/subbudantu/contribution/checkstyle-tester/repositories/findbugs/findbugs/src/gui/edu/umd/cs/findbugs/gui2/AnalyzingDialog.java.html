69      public static void show(@Nonnull final Project project) {
70          AnalysisCallback callback = new AnalysisCallback() {
71              @Override
72              public void analysisFinished(BugCollection results) {
73                  MainFrame instance = MainFrame.getInstance();
74                  assert results.getProject() == project;
75                  instance.setBugCollection(results);
76                  try {
77                      instance.releaseDisplayWait();
78                  } catch (Exception e) {
79                      Logger.getLogger(AnalyzingDialog.class.getName()).log(Level.FINE, "", e);
80                  }
81                  results.reinitializeCloud();
82              }
83  
84              @Override
85              public void analysisInterrupted() {
86                  MainFrame instance = MainFrame.getInstance();
87                  instance.updateProjectAndBugCollection(null);
88                  instance.releaseDisplayWait();
89              }
90          };
91          show(project, callback, false);
92  
93      }

95      /**
96       *
97       * @param project
98       *            The Project to analyze
99       * @param callback
100      *            contains what to do if the analysis is interrupted and what to
101      *            do if it finishes normally
102      * @param joinThread
103      *            Whether or not this constructor should return before the
104      *            analysis is complete. If true, the constructor does not return
105      *            until the analysis is either finished or interrupted.
106      */
107 
108     public static void show(@Nonnull
109             Project project, AnalysisCallback callback, boolean joinThread) {
110         AnalyzingDialog dialog = new AnalyzingDialog(project, callback, joinThread);
111         MainFrame.getInstance().acquireDisplayWait();
112         try {
113             dialog.analysisThread.start();
114             if (joinThread) {
115                 try {
116                     dialog.analysisThread.join();
117                 } catch (InterruptedException e) {
118                 }
119             }
120         } finally {
121             if (joinThread) {
122                 MainFrame.getInstance().releaseDisplayWait();
123             }
124         }
125     }

141     private AnalyzingDialog(@Nonnull Project project, AnalysisCallback callback, boolean joinThread) {
142         if (project == null) {
143             throw new NullPointerException("null project");
144         }
145         this.project = project;
146         this.callback = callback;
147         statusLabel = new JLabel(" ");
148         progressBar = new JProgressBar();
149         progressBar.setStringPainted(true);
150         cancelButton = new JButton(edu.umd.cs.findbugs.L10N.getLocalString("dlg.cancel_btn", "Cancel"));
151         cancelButton.addActionListener(new ActionListener() {
152             @Override
153             public void actionPerformed(ActionEvent evt) {
154                 cancel();
155             }
156         });
157         setDefaultCloseOperation(DO_NOTHING_ON_CLOSE);
158         addWindowListener(new WindowAdapter() {
159             @Override
160             public void windowClosed(WindowEvent evt) {
161                 cancel();
162             }
163         });
164 
165         SwingUtilities.invokeLater(new Runnable() {
166             @Override
167             public void run() {
168                 setLayout(new BoxLayout(getContentPane(), BoxLayout.Y_AXIS));
169                 add(statusLabel);
170                 add(progressBar);
171                 add(cancelButton);
172                 statusLabel.setAlignmentX(CENTER_ALIGNMENT);
173                 progressBar.setAlignmentX(CENTER_ALIGNMENT);
174                 cancelButton.setAlignmentX(CENTER_ALIGNMENT);
175                 pack();
176                 setSize(300, getHeight());
177                 setLocationRelativeTo(MainFrame.getInstance());
178                 setResizable(false);
179                 setModal(true);// Why was this set to false before?
180                 try {
181                     setVisible(true);
182                 } catch (Throwable e) {
183                     AnalyzingDialog.this.project.getGuiCallback().showMessageDialog("ERROR DURING ANALYSIS:\n\n"
184                             + e.getClass().getSimpleName() + ": " + e.getMessage());
185                 }
186             }
187         });
188 
189     }

191     private void cancel() {
192         if (!analysisFinished) {
193             analysisThread.interrupt();
194             setVisible(false);
195             // TODO there should be a call to dispose() here, but it seems to
196             // cause repainting issues
197         }
198     }

205         count++;
206         SwingUtilities.invokeLater(new Runnable() {

215     private void updateCount(final int count, final int goal) {
216         this.count = count;
217         this.goal = goal;
218         SwingUtilities.invokeLater(new Runnable() {
219             @Override
220             public void run() {
221                 progressBar.setString(count + "/" + goal);
222                 progressBar.setValue(count);

224             }
225         });
226     }

249     int pass = 0;
250 
251     @Override
252     public void startAnalysis(int numClasses) {
253         pass++;
254         String localString = edu.umd.cs.findbugs.L10N.getLocalString("progress.analyzing_classes", "Analyzing classes...");
255         updateStage(localString + ", pass " + pass + "/" + classesPerPass.length);
256         updateCount(0, numClasses);

259     private class AnalysisThread extends Thread {
260         {
261             // Give the analysis thread its (possibly user-defined) priority.
262             // The default is a slightly lower priority than the UI.
263             setPriority(Driver.getPriority());
264             setName("Analysis Thread");
265         }
266 
267         @Override
268         public void run() {
269             if (project == null) {
270                 throw new NullPointerException("null project");
271             }
272 
273             BugCollection data;
274             try {
275                 data = BugLoader.doAnalysis(project, AnalyzingDialog.this);
276             } catch (InterruptedException e) {
277                 callback.analysisInterrupted();
278                 // We don't have to clean up the dialog because the
279                 // cancel button handler does this already.
280                 return;
281             } catch (IOException e) {
282                 Logger.getLogger(AnalyzingDialog.class.getName()).log(Level.WARNING, "IO Error while performing analysis", e);
283                 callback.analysisInterrupted();
284                 scheduleDialogCleanup();
285                 scheduleErrorDialog("Analysis failed", e.getClass().getSimpleName()  + ": " + e.getMessage());
286                 return;
287             } catch (Throwable e) {
288                 callback.analysisInterrupted();
289                 scheduleDialogCleanup();
290                 scheduleErrorDialog("Analysis failed", e.getClass().getSimpleName()  + ": " + e.getMessage());
291                 return;
292             }
293 
294             // Analysis succeeded
295             analysisFinished = true;
296             scheduleDialogCleanup();
297             callback.analysisFinished(data);
298             MainFrame.getInstance().newProject();
299         }
300 
301         private void scheduleDialogCleanup() {
302             SwingUtilities.invokeLater(new Runnable() {
303                 /*
304                  * (non-Javadoc)
305                  *
306                  * @see java.lang.Runnable#run()
307                  */
308                 @Override
309                 public void run() {
310                     AnalyzingDialog.this.setVisible(false);
311                 }
312             });
313         }
314 
315         private void scheduleErrorDialog(final String title, final String message) {
316             SwingUtilities.invokeLater(new Runnable() {
317                 /*
318                  * (non-Javadoc)
319                  *
320                  * @see java.lang.Runnable#run()
321                  */
322                 @Override
323                 public void run() {
324                     JOptionPane.showMessageDialog(MainFrame.getInstance(), message, title, JOptionPane.ERROR_MESSAGE);
325                 }
326             });
327         }
