123         Sortables[] sortables;

125         for (Sortables s : Sortables.values()) {
126             if (s.isAvailable(mainFrame)) {
127                 a.add(s);

129         }
130         sortables = new Sortables[a.size()];

152     public void updateBugTree() {
153         mainFrame.acquireDisplayWait();
154         try {
155             BugTreeModel model = (BugTreeModel) getTree().getModel();
156             BugSet bs;
157             if (mainFrame.getBugCollection() != null) {
158                 bs = new BugSet(mainFrame.getBugCollection());
159             } else {
160                 bs = new BugSet(Collections.<BugLeafNode>emptySet());
161             }
162             model.getOffListenerList();
163             model.changeSet(bs);
164             if (bs.size() == 0 && bs.sizeUnfiltered() > 0) {
165                 warnUserOfFilters();
166             }
167 
168             mainFrame.updateStatusBar();
169             mainFrame.updateTitle();
170         } finally {
171             mainFrame.releaseDisplayWait();
172         }
173     }

230     JPopupMenu createBranchPopUpMenu() {
231         JPopupMenu popupMenu = new JPopupMenu();
232 
233         JMenuItem filterMenuItem = MainFrameHelper.newJMenuItem("menu.filterTheseBugs", "Filter these bugs");
234 
235         filterMenuItem.addActionListener(new ActionListener() {
236             @Override
237             public void actionPerformed(ActionEvent evt) {
238                 // TODO This code does a smarter version of filtering that is
239                 // only possible for branches, and does so correctly
240                 // However, it is still somewhat of a hack, because if we ever
241                 // add more tree listeners than simply the bugtreemodel,
242                 // They will not be called by this code. Using FilterActivity to
243                 // notify all listeners will however destroy any
244                 // benefit of using the smarter deletion method.
245 
246                 try {
247                     if (!mainFrame.canNavigateAway()) {
248                         return;
249                     }
250                     int startCount;
251                     TreePath path = MainFrame.getInstance().getTree().getSelectionPath();
252                     TreePath deletePath = path;
253                     startCount = ((BugAspects) (path.getLastPathComponent())).getCount();
254                     int count = ((BugAspects) (path.getParentPath().getLastPathComponent())).getCount();
255                     while (count == startCount) {
256                         deletePath = deletePath.getParentPath();
257                         if (deletePath.getParentPath() == null)// We are at the
258                             // top of the
259                             // tree, don't
260                             // let this be
261                             // removed,
262                             // rebuild tree
263                             // from root.
264                         {
265                             Matcher m = mainFrame.getCurrentSelectedBugAspects().getMatcher();
266                             Filter suppressionFilter = MainFrame.getInstance().getProject().getSuppressionFilter();
267                             suppressionFilter.addChild(m);
268                             PreferencesFrame.getInstance().updateFilterPanel();
269                             FilterActivity.notifyListeners(FilterListener.Action.FILTERING, null);
270                             return;
271                         }
272                         count = ((BugAspects) (deletePath.getParentPath().getLastPathComponent())).getCount();
273                     }
274                     /*
275                      * deletePath should now be a path to the highest ancestor
276                      * branch with the same number of elements as the branch to
277                      * be deleted in other words, the branch that we actually
278                      * have to remove in order to correctly remove the selected
279                      * branch.
280                      */
281                     BugTreeModel model = MainFrame.getInstance().getBugTreeModel();
282                     TreeModelEvent event = new TreeModelEvent(mainFrame, deletePath.getParentPath(),
283                             new int[] { model.getIndexOfChild(deletePath.getParentPath().getLastPathComponent(),
284                                     deletePath.getLastPathComponent()) }, new Object[] { deletePath.getLastPathComponent() });
285                     Matcher m = mainFrame.getCurrentSelectedBugAspects().getMatcher();
286                     Filter suppressionFilter = MainFrame.getInstance().getProject().getSuppressionFilter();
287                     suppressionFilter.addChild(m);
288                     PreferencesFrame.getInstance().updateFilterPanel();
289                     model.sendEvent(event, BugTreeModel.TreeModification.REMOVE);
290                     // FilterActivity.notifyListeners(FilterListener.Action.FILTERING,
291                     // null);
292 
293                     mainFrame.setProjectChanged(true);
294 
295                     MainFrame.getInstance().getTree().setSelectionRow(0);// Selects
296                     // the
297                     // top
298                     // of
299                     // the
300                     // Jtree
301                     // so
302                     // the
303                     // CommentsArea
304                     // syncs
305                     // up.
306                 } catch (RuntimeException e) {
307                     MainFrame.getInstance().showMessageDialog("Unable to create filter: " + e.getMessage());
308                 }
309             }
310         });
311 
312         popupMenu.add(filterMenuItem);
313 
314         JMenu changeDesignationMenu = MainFrameHelper.newJMenu("menu.changeDesignation", "Change bug designation");
315 
316         int i = 0;
317         int keyEvents[] = { KeyEvent.VK_1, KeyEvent.VK_2, KeyEvent.VK_3, KeyEvent.VK_4, KeyEvent.VK_5, KeyEvent.VK_6,
318                 KeyEvent.VK_7, KeyEvent.VK_8, KeyEvent.VK_9 };
319         for (String key : I18N.instance().getUserDesignationKeys(true)) {
320             String name = I18N.instance().getUserDesignation(key);
321             mainFrame.addDesignationItem(changeDesignationMenu, key, name, keyEvents[i++]);
322         }
323 
324         popupMenu.add(changeDesignationMenu);
325 
326         return popupMenu;
327     }

342         Debug.printf("expandTree(%d)\n", max);
343         JTree jTree = getTree();
344         int i = 0;
345         while (true) {
346             int rows = jTree.getRowCount();
347             if (i >= rows || rows >= max) {
348                 break;
349             }
350             jTree.expandRow(i++);
351         }
352     }

354     @SwingThread
355     boolean leavesShown() {
356         JTree jTree = getTree();
357 
358         int rows = jTree.getRowCount();
359         for (int i = 0; i < rows; i++) {
360             TreePath treePath = jTree.getPathForRow(i);
361             Object lastPathComponent = treePath.getLastPathComponent();
362             if (lastPathComponent instanceof BugLeafNode) {
363                 return true;
364             }
365         }
366         return false;
367     }

371         Debug.println("expand to first leaf");
372         if (leavesShown()) {
373             return;
374         }
375         JTree jTree = getTree();
376         int i = 0;
377         while (true) {
378             int rows = jTree.getRowCount();

380                 break;

383             Object lastPathComponent = treePath.getLastPathComponent();
384             if (lastPathComponent instanceof BugLeafNode) {
385                 return;
386             }
387             jTree.expandRow(i++);
388         }

612     private class TreeMouseListener implements MouseListener {
613         @Override
614         public void mouseClicked(MouseEvent e) {
615             TreePath path = tree.getPathForLocation(e.getX(), e.getY());
616 
617             if (path == null) {
618                 return;
619             }
620 
621             if(currentSelectedBugLeaf == path.getLastPathComponent()) {
622                 // sync mainFrame if user just clicks on the same bug
623                 mainFrame.syncBugInformation();
624             }
625 
626             if ((e.getButton() == MouseEvent.BUTTON3) || (e.getButton() == MouseEvent.BUTTON1 && e.isControlDown())) {
627 
628                 if (tree.getModel().isLeaf(path.getLastPathComponent())) {
629                     tree.setSelectionPath(path);
630                     bugPopupMenu.show(tree, e.getX(), e.getY());
631                 } else {
632                     tree.setSelectionPath(path);
633                     if (!(path.getParentPath() == null)) {
634                         // path is null, the
635                         // root was selected,
636                         // dont allow them to
637                         // filter out the root.
638                         branchPopupMenu.show(tree, e.getX(), e.getY());
639                     }
640                 }
641             }
642         }

661     private class MyTreeSelectionListener implements TreeSelectionListener {
662         private volatile boolean ignoreSelection = false;
663         @Override
664         public void valueChanged(TreeSelectionEvent selectionEvent) {
665             if (ignoreSelection) {
666                 return;
667             }
668 
669             TreePath path = selectionEvent.getNewLeadSelectionPath();
670             if (path != null) {
671                 if (!mainFrame.canNavigateAway()) {
672                     try {
673                         ignoreSelection = true;
674                         tree.clearSelection();
675                         tree.setSelectionPath(selectionEvent.getOldLeadSelectionPath());
676                     } finally {
677                         ignoreSelection = false;
678                     }
679                     return;
680                 }
681 
682                 Object lastPathComponent = path.getLastPathComponent();
683                 if (lastPathComponent instanceof BugLeafNode) {
684                     boolean beforeProjectChanged = mainFrame.isProjectChanged();
685                     currentSelectedBugLeaf = (BugLeafNode) lastPathComponent;
686                     mainFrame.setCurrentSelectedBugAspects(null);
687                     mainFrame.syncBugInformation();

693                     mainFrame.setCurrentSelectedBugAspects((BugAspects) lastPathComponent);
694                     mainFrame.syncBugInformation();
695                     mainFrame.setProjectChanged(beforeProjectChanged);
696                 }
697             }
