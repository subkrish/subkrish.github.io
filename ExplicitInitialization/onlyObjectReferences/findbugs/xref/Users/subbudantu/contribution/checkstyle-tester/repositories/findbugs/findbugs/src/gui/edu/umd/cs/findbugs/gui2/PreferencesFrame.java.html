194     private void handleWindowClose() {
195         TreeModel bt = (MainFrame.getInstance().getTree().getModel());
196         if (bt instanceof BugTreeModel) {
197             ((BugTreeModel) bt).checkSorter();
198         }
199         Project project = getCurrentProject();
200 
201         boolean changed = pluginsAdded;
202         pluginsAdded = false;
203         List<String> enabledPlugins = new ArrayList<String>();
204         List<String> disabledPlugins = new ArrayList<String>();
205         for (Map.Entry<Plugin, EnabledSettings> entry : pluginEnabledStatus.entrySet()) {
206             Plugin plugin = entry.getKey();
207             EnabledSettings enabled = entry.getValue();
208             if (project != null) {
209                 Boolean newSetting = enabled.project;
210                 Boolean existingSetting = project.getPluginStatus(plugin);
211                 boolean sameSettings = Objects.equals(existingSetting, newSetting);
212                 if (!sameSettings) {
213                     project.setPluginStatusTrinary(plugin.getPluginId(), newSetting);
214                     changed = true;
215                 }
216             }
217             if (enabled.global) {
218                 enabledPlugins.add(plugin.getPluginId());
219             } else {
220                 disabledPlugins.add(plugin.getPluginId());
221             }
222             if (plugin.isGloballyEnabled() != enabled.global) {
223                 plugin.setGloballyEnabled(enabled.global);
224                 changed = true;
225             }
226         }
227 
228         if (changed && project != null) {
229             MainFrame.getInstance().updateBugTree();
230             MainFrame.getInstance().setProjectChanged(true);
231         }
232         if (project == null) {
233             GUISaveState.getInstance().setPluginsEnabled(enabledPlugins, disabledPlugins);
234             GUISaveState.getInstance().save();
235         }
236     }

244         final JPanel pluginPanel = new JPanel();

261                 chooser.addChoosableFileFilter(new FileFilter() {
262 
263                     @Override
264                     public String getDescription() {
265                         return "FindBugs Plugin (*.jar)";
266                     }
267 
268                     @Override
269                     public boolean accept(File f) {
270                         if (f.isDirectory()) {
271                             return true;
272                         }
273                         if (!f.canRead()) {
274                             return false;
275                         }
276                         if (f.getName().endsWith(".jar")) {
277                             return true;
278                         }
279                         return false;
280                     }
281                 });
282                 chooser.setDialogTitle("Select a FindBugs plugin jar");
283                 int retvalue = chooser.showDialog(PreferencesFrame.this, "Install");
284 
285                 if (retvalue == JFileChooser.APPROVE_OPTION) {
286                     File f = chooser.getSelectedFile();
287                     try {
288                         // load and enable for project (if loaded)
289                         Plugin plugin = Plugin.loadCustomPlugin(f, PreferencesFrame.this.getCurrentProject());
290 
291                         GUISaveState guiSaveState = GUISaveState.getInstance();
292                         URL url = f.toURI().toURL();
293                         // add to FBGUI custom plugins list
294                         guiSaveState.addCustomPlugin(url);
295                         // add to list of enabled plugins
296                         guiSaveState.setPluginEnabled(plugin.getPluginId());
297                         plugin.setGloballyEnabled(true);
298                         guiSaveState.save();
299                         pluginsAdded = true;
300                         rebuildPluginCheckboxes();
301 
302                     } catch (PluginException | MalformedURLException e1) {
303                         LOGGER.log(Level.WARNING, "Could not load " + f.getPath(), e1);
304                         JOptionPane.showMessageDialog(PreferencesFrame.this, "Could not load " + f.getPath()
305                                 + "\n\n"
306                                 + e1.getClass().getSimpleName() + ": " + e1.getMessage(),
307                                 "Error Loading Plugin", JOptionPane.ERROR_MESSAGE);
308                     }
309                 }
310             }
311         });
312 
313         pluginPanel.add(south, BorderLayout.SOUTH);
314 
315         return pluginPanel;
316     }

322     private void rebuildPluginCheckboxes() {
323         Project currentProject = getCurrentProject();
324         pluginPanelCenter.removeAll();
325         if (currentProject != null) {
326             GridBagConstraints g = new GridBagConstraints();
327             g.fill = GridBagConstraints.NONE;
328             g.insets = new Insets(5,5,5,5);
329             g.gridy = 0;
330             //        g.anchor = GridBagConstraints.WEST;
331             //        g.gridx = 1;
332             //        pluginPanelCenter.add(new JLabel("Global Setting"), g);
333             g.anchor = GridBagConstraints.CENTER;
334             g.gridx = 2;
335             pluginPanelCenter.add(new JLabel("Project Setting"), g);
336         }
337         Collection<Plugin> plugins = Plugin.getAllPlugins();
338         int added = 0;
339         for (final Plugin plugin : plugins) {
340             if (plugin.isCorePlugin()) {
341                 continue;
342             }
343             String text = plugin.getShortDescription();
344             String id = plugin.getPluginId();
345             if (text == null) {
346                 text = id;
347             }
348             final URL url = plugin.getPluginLoader().getURL();
349             String pluginUrlStr = url.toExternalForm();
350             if ("file".equals(url.getProtocol())) {
351                 try {
352                     pluginUrlStr = new File(URLDecoder.decode(url.getPath(), "UTF-8")).getAbsolutePath();
353                 } catch (UnsupportedEncodingException e) {
354 
355                 }
356             }
357             text = String.format("<html>%s<br><font style='font-weight:normal;font-style:italic'>%s",
358                     text, pluginUrlStr);
359 
360             EnabledSettings enabled = isEnabled(currentProject, plugin);
361             final JCheckBox checkGlobal = new JCheckBox(text, enabled.global);
362             final boolean cannotDisable = plugin.isEnabledByDefault() && plugin.cannotDisable();
363             if (cannotDisable) {
364                 if (!enabled.global) {
365                     throw new IllegalStateException(
366                             plugin.getPluginId() + " is enabled by default and cannot be disabled, but is disabled");
367                 }
368                 checkGlobal.setEnabled(false);
369             } else {
370                 checkGlobal.addMouseListener(new MouseAdapter() {
371 
372                     @Override
373                     public void mousePressed(MouseEvent e) {
374                         if (SwingUtilities.isRightMouseButton(e)) {
375                             JPopupMenu menu = new JPopupMenu();
376                             JMenuItem item = new JMenuItem("Uninstall " + plugin.getShortDescription() + "...");
377                             item.addActionListener(new UninstallClickListener(plugin, url));
378                             menu.add(item);
379                             menu.show(checkGlobal, e.getX(), e.getY());
380                         }
381                     }
382                 });
383                 checkGlobal.addActionListener(new ActionListener() {
384                     @Override
385                     public void actionPerformed(ActionEvent e) {
386                         pluginEnabledStatus.get(plugin).global = checkGlobal.isSelected();
387                     }
388                 });
389             }
390             checkGlobal.setVerticalTextPosition(SwingConstants.TOP);
391             String longText = plugin.getDetailedDescription();
392             if (longText != null) {
393                 checkGlobal.setToolTipText("<html>" + longText + "</html>");
394             }
395             pluginEnabledStatus.put(plugin, enabled);
396 
397             GridBagConstraints gbc = new GridBagConstraints();
398             gbc.fill = GridBagConstraints.BOTH;
399             gbc.weightx = 1;
400             gbc.gridx = 1;
401             gbc.insets = new Insets(0,5,0,5);
402             gbc.gridy = added+1;
403             gbc.anchor = GridBagConstraints.NORTHWEST;
404             pluginPanelCenter.add(checkGlobal, gbc);
405 
406             if (currentProject != null && !cannotDisable) {
407                 final JComboBox<String> combo = new WideComboBox<>(new String[]{"DEFAULT", "DISABLED", "ENABLED"});
408                 if (enabled.project == null) {
409                     combo.setSelectedIndex(0);
410                 } else {
411                     combo.setSelectedIndex(enabled.project ? 2 : 1);
412                 }
413                 combo.setRenderer(new DefaultListCellRenderer() {
414                     @Override
415                     public Component getListCellRendererComponent(JList list, Object value, int index, boolean isSelected, boolean cellHasFocus) {
416                         if (index == -1) {
417                             if (value.equals("DEFAULT")) {
418                                 value = "Default";
419                             }
420                             if (value.equals("DISABLED")) {
421                                 value = "Disabled";
422                             }
423                             if (value.equals("ENABLED")) {
424                                 value = "Enabled";
425                             }
426                         } else {
427                             if (value.equals("DEFAULT")) {
428                                 value = "Default (use global setting)";
429                             }
430                             if (value.equals("DISABLED")) {
431                                 value = "Disabled for this project";
432                             }
433                             if (value.equals("ENABLED")) {
434                                 value = "Enabled for this project";
435                             }
436                         }
437                         return super.getListCellRendererComponent(list, value, index, isSelected, cellHasFocus);
438                     }
439                 });
440                 combo.addActionListener(new ActionListener() {
441                     @Override
442                     public void actionPerformed(ActionEvent e) {
443                         Boolean[] array = { null, false, true };
444                         int i = combo.getSelectedIndex();
445                         if (i < 0 || i > 2)
446                         {
447                             return; // ??
448                         }
449                         pluginEnabledStatus.get(plugin).project = array[i];
450                     }
451                 });
452                 gbc.gridx = 2;
453                 gbc.fill = GridBagConstraints.NONE;
454                 gbc.weightx = 0;
455                 pluginPanelCenter.add(combo, gbc);
456             }
457             added++;
458         }
459         if (added == 0) {
460             JLabel label = new JLabel("<html>No plugins are loaded.<br> " +
461                     "Try installing <u><font color=blue>fb-contrib</font></u> - or write your own<br>" +
462                     "plugin for your project's needs!");
463             label.setCursor(Cursor.getPredefinedCursor(Cursor.HAND_CURSOR));
464             label.addMouseListener(new MouseAdapter() {
465                 @Override
466                 public void mouseClicked(MouseEvent e) {
467                     try {
468                         LaunchBrowser.showDocument(new URL("https://sourceforge.net/projects/fb-contrib/"));
469                     } catch (MalformedURLException e1) {
470                         throw new IllegalStateException(e1);
471                     }
472                 }
473             });
474             label.setBorder(new EmptyBorder(10,10,10,10));
475             pluginPanelCenter.add(label);
476         }
477         PreferencesFrame.this.pack();
478     }

551     private void changeTabSize() {
552         int tabSize;
553 
554         try {
555             tabSize = Integer.decode(tabTextField.getText());
556         } catch (NumberFormatException exc) {
557             JOptionPane.showMessageDialog(instance, "Error in tab size field.", "Tab Size Error",
558                     JOptionPane.INFORMATION_MESSAGE);
559             return;
560         }
561 
562         if (tabSize < TAB_MIN || tabSize > TAB_MAX) {
563             JOptionPane.showMessageDialog(instance, "Tab size exceedes range (" + TAB_MIN + " - " + TAB_MAX + ").",
564                     "Tab Size Excedes Range", JOptionPane.INFORMATION_MESSAGE);
565             return;
566         }
567 
568         if (tabSize != GUISaveState.getInstance().getTabSize()) {
569             GUISaveState.getInstance().setTabSize(tabSize);
570             MainFrame.getInstance().getSourceCodeDisplayer().clearCache();
571             // This causes the GUI to redisplay the current code
572             MainFrame.getInstance().syncBugInformation();
573         }
574     }

576     private void changeFontSize() {
577         float fontSize;
578 
579         try {
580             fontSize = Float.parseFloat(fontTextField.getText());
581         } catch (NumberFormatException exc) {
582             JOptionPane.showMessageDialog(instance, "Error in font size field.", "Font Size Error",
583                     JOptionPane.INFORMATION_MESSAGE);
584             return;
585         }
586 
587         if (fontSize < FONT_MIN || fontSize > FONT_MAX) {
588             JOptionPane.showMessageDialog(instance, "Font size exceedes range (" + FONT_MIN + " - " + FONT_MAX + ").",
589                     "Font Size Exceedes Range", JOptionPane.INFORMATION_MESSAGE);
590             return;
591         }
592 
593         if (fontSize != GUISaveState.getInstance().getFontSize()) {
594             GUISaveState.getInstance().setFontSize(fontSize);
595             JOptionPane.showMessageDialog(instance, "To implement the new font size please restart FindBugs.",
596                     "Changing Font",
597                     JOptionPane.INFORMATION_MESSAGE);
598         }
599     }

601     private void changePackagePrefixLength() {
602         int value;
603 
604         try {
605             value = Integer.parseInt(packagePrefixLengthTextField.getText());
606         } catch (NumberFormatException exc) {
607             JOptionPane.showMessageDialog(instance, "Error in package prefix length field.", "Package Prefix Length Error",
608                     JOptionPane.INFORMATION_MESSAGE);
609             return;
610         }
611 
612         if (value < 1 || value > 12) {
613             JOptionPane.showMessageDialog(instance, "package prefix length exceedes range (" + 1 + " - " + 12 + ").",
614                     "package prefix lengthe exceedes range", JOptionPane.INFORMATION_MESSAGE);
615             return;
616         }
617 
618         if (value != GUISaveState.getInstance().getPackagePrefixSegments()) {
619             GUISaveState.getInstance().setPackagePrefixSegments(value);
620             BugTreeModel bt = (BugTreeModel) (MainFrame.getInstance().getTree().getModel());
621             bt.needToRebuild();
622             bt.checkSorter();
623         }

674             }
675         });
676 
677         gbc.gridy = 2;
678         gbc.insets = new Insets(5, 0, 0, 0);
679         filterPanel.add(removeButton, gbc);
680         removeButton.addActionListener(new ActionListener() {
681             @Override
682             public void actionPerformed(ActionEvent evt) {
683                 Object[] selected = filterCheckBoxList.getSelectedValues();
684                 if (selected.length == 0) {
685                     return;
686                 }
687                 for (Object o : selected) {
688                     MatchBox box = (MatchBox) o;
689                     MainFrame.getInstance().getProject().getSuppressionFilter().removeChild(box.getMatcher());
690                 }
691                 FilterActivity.notifyListeners(FilterListener.Action.UNFILTERING, null);
692                 MainFrame.getInstance().setProjectChanged(true);
693                 updateFilterPanel();
694             }
695         });
696         gbc.gridy = 3;
697         gbc.weighty = 0;
698         gbc.insets = new Insets(5, 0, 0, 0);
699         filterPanel.add(removeAllButton, gbc);
700         removeAllButton.addActionListener(new ActionListener() {
701             @Override
702             public void actionPerformed(ActionEvent evt) {
703                 boolean needsRebuild = false;
704                 Filter suppressionFilter = MainFrame.getInstance().getProject().getSuppressionFilter();
705                 if (!suppressionFilter.isEmpty()) {
706                     needsRebuild = true;
707                 }
708                 suppressionFilter.clear();
709 
710                 if (needsRebuild) {
711                     // filters being cleared were disabled
712                     FilterActivity.notifyListeners(FilterListener.Action.UNFILTERING, null);
713                 }
714                 MainFrame.getInstance().setProjectChanged(true);
715                 updateFilterPanel();
716             }
717         });
718         gbc.gridy = 4;
719         gbc.weighty = 1;
720         gbc.insets = new Insets(0, 0, 0, 0);
721         filterPanel.add(Box.createGlue(), gbc);
722 
723         return filterPanel;
724     }

726     private static class MatchBox extends JCheckBox {
727         Matcher m;
728 
729         MatchBox(String text, Matcher m) {
730             super(text);
731             this.m = m;
732         }
733 
734         Matcher getMatcher() {
735             return m;
736         }
737     }
738 
739     void updateFilterPanel() {
740         ArrayList<MatchBox> boxes = new ArrayList<MatchBox>();
741         final Filter f = MainFrame.getInstance().getProject().getSuppressionFilter();
742 
743         for (final Matcher m : f.getChildren()) {
744             MatchBox box = new MatchBox(m.toString(), m);
745             box.addItemListener(new ItemListener() {
746                 @Override
747                 public void itemStateChanged(ItemEvent evt) {
748                     boolean isSelected = ((JCheckBox) evt.getSource()).isSelected();
749                     boolean wasSelected = f.isEnabled(m);
750                     if (isSelected == wasSelected) {
751                         return;
752                     }
753                     f.setEnabled(m, isSelected);
754                     updateFilters(isSelected);
755 
756                 }
757             });
758             box.setSelected(f.isEnabled(m));
759             boxes.add(box);
760         }
761 
762         filterCheckBoxList.setListData(boxes.toArray(new MatchBox[boxes.size()]));
763     }

767         FilterActivity.notifyListeners(addedFilter ? FilterListener.Action.FILTERING
768                 : FilterListener.Action.UNFILTERING, null);
769         MainFrame.getInstance().setProjectChanged(true);
770     }

772     private class UninstallClickListener implements ActionListener {
773         private final Plugin plugin;
774         private final URL url;
775 
776         public UninstallClickListener(Plugin plugin, URL url) {
777             this.plugin = plugin;
778             this.url = url;
779         }
780 
781         @Override
782         public void actionPerformed(ActionEvent e) {
783             int result = JOptionPane.showOptionDialog(PreferencesFrame.this,
784                     "Are you sure you want to uninstall " + plugin.getShortDescription() + "?" +
785                             "\n\nNo files will be deleted from your computer.", "",
786                             JOptionPane.YES_NO_OPTION, JOptionPane.QUESTION_MESSAGE, null,
787                             new Object[]{"Uninstall", "Cancel"}, "Cancel");
788             if (result == 0) {
789                 if (!GUISaveState.getInstance().removeCustomPlugin(url)) {
790                     if ("file".equals(url.getProtocol())) {
791                         String path = url.toExternalForm();
792                         try {
793                             path = new File(URLDecoder.decode(url.getPath(), "UTF-8")).getAbsolutePath();
794                         } catch (UnsupportedEncodingException ex) {
795                         }
796                         try {
797                             StringSelection contents = new StringSelection(path);
798                             Toolkit.getDefaultToolkit().getSystemClipboard().setContents(contents, contents);
799                         } catch (Exception e1) {
800                         }
801                         JOptionPane.showMessageDialog(PreferencesFrame.this,
802                                 "The plugin could not be uninstalled automatically.\n\n" +
803                                         "You can try to delete this plugin manually: \n"
804                                         + path + "\n\n(This path has been copied to your clipboard)",
805                                         "Error", JOptionPane.ERROR_MESSAGE);
806                     } else {
807                         JOptionPane.showMessageDialog(PreferencesFrame.this,
808                                 "This plugin is not actually in the list of plugins...\n" +
809                                         "Not sure what to do...\n "
810                                         + url.toExternalForm()
811                                         + "\n\nPlugin URL's:\n" +
812                                         GUISaveState.getInstance().getCustomPlugins(),
813                                         "Error", JOptionPane.ERROR_MESSAGE);
814                     }
815                 } else {
816                     JOptionPane.showMessageDialog(PreferencesFrame.this,
817                             "Changes will take effect after you restart FindBugs.");
818                 }
819             }
820         }
821     }
