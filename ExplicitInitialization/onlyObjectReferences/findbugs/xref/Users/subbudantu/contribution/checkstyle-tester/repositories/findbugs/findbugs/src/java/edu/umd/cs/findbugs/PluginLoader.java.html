189     /**
190      * Constructor.
191      *
192      * @param url
193      *            the URL of the plugin Jar file
194      * @param uri
195      * @param parent
196      *            the parent classloader
197      * @param isInitial
198      *          is this plugin loaded from one of the standard locations for plugins
199      * @param optional
200      *          is this an optional plugin
201      */
202     private PluginLoader(@Nonnull URL url, URI uri, ClassLoader parent, boolean isInitial, boolean optional) throws PluginException {
203         URL[] loaderURLs = createClassloaderUrls(url);
204         classLoaderForResources = new URLClassLoader(loaderURLs);
205         loadedFrom = url;
206         loadedFromUri = uri;
207         jarName = getJarName(url);
208         corePlugin = false;
209         initialPlugin = isInitial;
210         optionalPlugin = optional;
211         plugin = init();
212         if (!hasParent()) {
213             classLoader = new URLClassLoader(loaderURLs, parent);
214         } else {
215             if (parent != PluginLoader.class.getClassLoader()) {
216                 throw new IllegalArgumentException("Can't specify parentid " + parentId + " and provide a seperate class loader");
217             }
218             Plugin parentPlugin = Plugin.getByPluginId(parentId);
219             if (parentPlugin != null) {
220                 parent = parentPlugin.getClassLoader();
221                 classLoader = new URLClassLoader(loaderURLs, parent);
222             }
223         }
224         if (classLoader == null) {
225             if (!lazyInitialization) {
226                 throw new IllegalStateException("Can't find parent plugin " + parentId);
227             }
228             partiallyInitialized.add(this);
229         } else {
230             loadPluginComponents();
231             Plugin.putPlugin(loadedFromUri, plugin);
232         }
233     }

235     private static void finishLazyInitialization()  {
236         if (!lazyInitialization) {
237             throw new IllegalStateException("Not in lazy initialization mode");
238         }
239         while (!partiallyInitialized.isEmpty()) {
240             boolean changed = false;
241             LinkedList<String>  unresolved = new LinkedList<String>();
242             Set<String>  needed = new TreeSet<String>();
243 
244             for (Iterator<PluginLoader> i = partiallyInitialized.iterator(); i.hasNext();) {
245                 PluginLoader pluginLoader = i.next();
246                 String pluginId = pluginLoader.getPlugin().getPluginId();
247                 assert pluginLoader.hasParent();
248                 String parentid = pluginLoader.parentId;
249                 Plugin parent = Plugin.getByPluginId(parentid);
250                 if (parent != null) {
251                     i.remove();
252                     try {
253                         URL[] loaderURLs = PluginLoader.createClassloaderUrls(pluginLoader.loadedFrom);
254                         pluginLoader.classLoader = new URLClassLoader(loaderURLs, parent.getClassLoader());
255                         pluginLoader.loadPluginComponents();
256                         Plugin.putPlugin(pluginLoader.loadedFromUri, pluginLoader.plugin);
257                     } catch (PluginException e) {
258                         throw new RuntimeException("Unable to load plugin " + pluginId, e);
259                     }
260                     changed = true;
261                 } else {
262                     unresolved.add(pluginId);
263                     needed.add(parentid);
264                 }
265             }
266             if (!changed) {
267                 String msg = "Unable to load parent plugins " + needed + " in order to load " + unresolved;
268                 System.err.println(msg);
269                 AnalysisContext.logError(msg);
270                 msg = "Available plugins are " + Plugin.getAllPluginIds();
271                 System.err.println(msg);
272                 AnalysisContext.logError(msg);
273 
274                 for (Iterator<PluginLoader> i = partiallyInitialized.iterator(); i.hasNext();) {
275                     Plugin.removePlugin(i.next().loadedFromUri);
276                 }
277                 partiallyInitialized.clear();
278             }
279         }
280         lazyInitialization = false;
281     }

283     /**
284      * Patch for issue 3429143: allow plugins load classes/resources from 3rd
285      * party jars
286      *
287      * @param url
288      *            plugin jar location as url
289      * @return non empty list with url to be used for loading classes from given
290      *         plugin. If plugin jar contains standard Java manifest file, all
291      *         entries of its "Class-Path" attribute will be translated to the
292      *         jar-relative url's and added to the returned list. If plugin jar
293      *         does not contains a manifest, or manifest does not have
294      *         "Class-Path" attribute, the given argument will be the only entry
295      *         in the array.
296      * @throws PluginException
297      */
298     private static @Nonnull URL[] createClassloaderUrls(@Nonnull URL url) throws PluginException {
299         List<URL> urls = new ArrayList<URL>();
300         urls.add(url);
301 
302         Manifest mf = null;
303         File f = new File(url.getPath());
304         // default: try with jar/zip/war etc files
305         if(!f.isDirectory()) {
306             JarInputStream jis = null;
307             try {
308                 jis = new JarInputStream(url.openStream());
309                 mf = jis.getManifest();
310             } catch (IOException ioe) {
311                 throw new PluginException("Failed loading manifest for plugin jar: " + url, ioe);
312             } finally {
313                 IO.close(jis);
314             }
315         } else {
316             // If this is not a jar/zip/war etc file, can we can load from directory?
317             // Allow plugins be loaded from "exploded jar" directories (e.g. while debugging
318             // 3rd party FB plugin projects in Eclipse without packaging them to jars at all)
319             File manifest = guessManifest(f);
320             if(manifest != null){
321                 FileInputStream is = null;
322                 try {
323                     is = new FileInputStream(manifest);
324                     mf = new Manifest(is);
325                 } catch (IOException e) {
326                     throw new PluginException("Failed loading manifest for plugin jar: " + url, e);
327                 } finally {
328                     IO.close(is);
329                 }
330             }
331         }
332         if(mf != null){
333             try {
334                 addClassPathFromManifest(url, urls, mf);
335             } catch (MalformedURLException e) {
336                 throw new PluginException("Failed loading manifest for plugin jar: " + url, e);
337             }
338         }
339         return urls.toArray(new URL[urls.size()]);
340     }

342     private static void addClassPathFromManifest(@Nonnull URL url, @Nonnull List<URL> urls,
343             @Nonnull Manifest mf) throws MalformedURLException {
344         Attributes atts = mf.getMainAttributes();
345         if (atts == null)  {
346             return;
347         }
348         String classPath = atts.getValue(Attributes.Name.CLASS_PATH);
349         if (classPath != null) {
350             String jarRoot = url.toString();
351             jarRoot = jarRoot.substring(0, jarRoot.lastIndexOf('/') + 1);
352             String[] jars = classPath.split(",");
353             for (String jar : jars) {
354                 jar = jarRoot + jar.trim();
355                 urls.add(new URL(jar));
356             }
357         }
358     }

362      * location or "standard" Eclipse location (sibling to the current classpath)
363      */
364     @CheckForNull
365     private static File guessManifest(@Nonnull File parent) {
366         File file = new File(parent, "MANIFEST.MF");
367         if(!file.isFile()){
368             file = new File(parent, "META-INF/MANIFEST.MF");
369         }
370         if(!file.isFile()){
371             file = new File(parent, "../META-INF/MANIFEST.MF");
372         }
373         if(file.isFile()){
374             return file;
375         }
376         return null;
377     }

426     private static URL computeCoreUrl() {
427         URL from;
428         String findBugsClassFile = ClassName.toSlashedClassName(FindBugs.class) + ".class";
429         URL me = FindBugs.class.getClassLoader().getResource(findBugsClassFile);
430         if (DEBUG) {
431             System.out.println("FindBugs.class loaded from " + me);
432         }
433         if(me == null) {
434             throw new IllegalStateException("Failed to load " + findBugsClassFile);
435         }
436         try {
437             String u = me.toString();
438             if (u.startsWith("jar:") && u.endsWith("!/" + findBugsClassFile)) {
439                 u = u.substring(4, u.indexOf("!/"));
440 
441                 from = new URL(u);
442 
443             } else if (u.endsWith(findBugsClassFile)) {
444                 u = u.substring(0, u.indexOf(findBugsClassFile));
445                 from = new URL(u);
446             } else {
447                 throw new IllegalArgumentException("Unknown url shema: " + u);
448             }
449 
450         } catch (MalformedURLException e) {
451             throw new IllegalArgumentException("Failed to parse url: " + me);
452         }
453         if (DEBUG) {
454             System.out.println("Core class files loaded from " + from);
455         }
456         return from;
457     }

507     private static URL resourceFromPlugin(URL u, String args) throws MalformedURLException {
508         String path = u.getPath();
509         if (path.endsWith(".zip") || path.endsWith(".jar")) {
510             return new URL("jar:" + u.toString() + "!/" + args);
511         } else if (path.endsWith("/")) {
512             return new URL(u.toString() + args);
513         } else {
514             return new URL(u.toString() + "/" + args);
515 
516         }
517     }

519     /**
520      * Get a resource using the URLClassLoader classLoader. We try findResource
521      * first because (based on experiment) we can trust it to prefer resources
522      * in the jarfile to resources on the filesystem. Simply calling
523      * classLoader.getResource() allows the filesystem to override the jarfile,
524      * which can mess things up if, for example, there is a findbugs.xml or
525      * messages.xml in the current directory.
526      *
527      * @param name
528      *            resource to get
529      * @return URL for the resource, or null if it could not be found
530      */
531     public URL getResource(String name) {
532         if (isCorePlugin()) {
533             URL url = getCoreResource(name);
534             if (url != null &&  IO.verifyURL(url)) {
535                 return url;
536             }
537         }
538         if (loadedFrom != null) {
539             try {
540                 URL url = resourceFromPlugin(loadedFrom, name);
541                 if (DEBUG) {
542                     System.out.println("Trying to load " + name + " from " + url);
543                 }
544                 if (IO.verifyURL(url)) {
545                     return url;
546                 }
547             } catch (MalformedURLException e) {
548                 assert true;
549             }
550 
551         }
552 
553         if (classLoaderForResources instanceof URLClassLoader) {
554 
555             URLClassLoader urlClassLoader = (URLClassLoader) classLoaderForResources;
556             if (DEBUG) {
557                 System.out.println("Trying to load " + name + " using URLClassLoader.findResource");
558                 System.out.println("  from urls: " + Arrays.asList(urlClassLoader.getURLs()));
559             }
560             URL url = urlClassLoader.findResource(name);
561             if (url == null) {
562                 url = urlClassLoader.findResource("/" + name);
563             }
564             if (IO.verifyURL(url)) {
565                 return url;
566             }
567         }
568 
569         if (DEBUG) {
570             System.out.println("Trying to load " + name + " using ClassLoader.getResource");
571         }
572         URL url = classLoaderForResources.getResource(name);
573         if (url == null) {
574             url = classLoaderForResources.getResource("/" + name);
575         }
576         if (IO.verifyURL(url)) {
577             return url;
578         }
579 
580         return null;
581     }

583     static @CheckForNull
584     URL getCoreResource(String name) {
585         URL u = loadFromFindBugsPluginDir(name);
586         if (u != null) {
587             return u;
588         }
589         u = loadFromFindBugsEtcDir(name);
590         if (u != null) {
591             return u;
592         }
593         u = PluginLoader.class.getResource(name);
594         if (u != null) {
595             return u;
596         }
597         u = PluginLoader.class.getResource("/"+name);
598         return u;
599     }

637     private static <T> Class<? extends T> getClass(ClassLoader loader, @DottedClassName String className, Class<T> type) throws PluginException {
638         try {
639             return loader.loadClass(className).asSubclass(type);
640         } catch (ClassNotFoundException e) {
641             throw new PluginException("Unable to load " + className, e);
642         } catch (ClassCastException e) {
643             throw new PluginException("Cannot cast " + className + " to " + type.getName(), e);
644         }
645     }

647     private Plugin init() throws PluginException {
648         if (DEBUG) {
649             System.out.println("Loading plugin from " + loadedFrom);
650         }
651         // Plugin descriptor (a.k.a, "findbugs.xml"). Defines
652         // the bug detectors and bug patterns that the plugin provides.
653         Document pluginDescriptor = getPluginDescriptor();
654         List<Document> messageCollectionList = getMessageDocuments();
655 
656         Plugin constructedPlugin = constructMinimalPlugin(pluginDescriptor, messageCollectionList);
657 
658         // Success!
659         if (DEBUG) {
660             System.out.println("Loaded " + constructedPlugin.getPluginId() + " from " + loadedFrom);
661         }
662         return constructedPlugin;
663     }

665     private void loadPluginComponents()
666             throws PluginException {
667         Document pluginDescriptor = getPluginDescriptor();
668         List<Document> messageCollectionList = getMessageDocuments();
669         List<Node> cloudNodeList = XMLUtil.selectNodes(pluginDescriptor, "/FindbugsPlugin/Cloud");
670         for (Node cloudNode : cloudNodeList) {
671 
672             String cloudClassname = cloudNode.valueOf("@cloudClass");
673             String cloudId = cloudNode.valueOf("@id");
674             String usernameClassname = cloudNode.valueOf("@usernameClass");
675             boolean onlineStorage = Boolean.valueOf(cloudNode.valueOf("@onlineStorage"));
676             String propertiesLocation = cloudNode.valueOf("@properties");
677             boolean disabled = Boolean.valueOf(cloudNode.valueOf("@disabled")) && !cloudId.equals(CloudFactory.DEFAULT_CLOUD);
678             if (disabled) {
679                 continue;
680             }
681             boolean hidden = Boolean.valueOf(cloudNode.valueOf("@hidden")) && !cloudId.equals(CloudFactory.DEFAULT_CLOUD);
682 
683             Class<? extends Cloud> cloudClass = getClass(classLoader, cloudClassname, Cloud.class);
684 
685             Class<? extends NameLookup> usernameClass = getClass(classLoader, usernameClassname, NameLookup.class);
686             Node cloudMessageNode = findMessageNode(messageCollectionList, "/MessageCollection/Cloud[@id='" + cloudId + "']",
687                     "Missing Cloud description for cloud " + cloudId);
688             String description = getChildText(cloudMessageNode, "Description").trim();
689             String details = getChildText(cloudMessageNode, "Details").trim();
690             PropertyBundle properties = new PropertyBundle();
691             if (propertiesLocation != null && propertiesLocation.length() > 0) {
692                 URL properiesURL = classLoader.getResource(propertiesLocation);
693                 if (properiesURL == null) {
694                     continue;
695                 }
696                 properties.loadPropertiesFromURL(properiesURL);
697             }
698             List<Node> propertyNodes = XMLUtil.selectNodes(cloudNode, "Property");
699             for (Node node : propertyNodes) {
700                 String key = node.valueOf("@key");

1154         Document pluginDescriptor;

1158         URL findbugsXML_URL = getResource(name);
1159         if (findbugsXML_URL == null) {

1161         }

1166         if (jarName != null && !findbugsXML_URL.toString().contains(jarName)
1167                 && !(corePlugin && findbugsXML_URL.toString().endsWith("etc/findbugs.xml"))) {

1170                 classloaderName += Arrays.asList(((URLClassLoader) classLoader).getURLs());
1171             }
1172             throw new PluginDoesntContainMetadataException((corePlugin ? "Core plugin" : "Plugin ") + jarName
1173                     + " doesn't contain findbugs.xml; got " + findbugsXML_URL + " from " + classloaderName);
1174         }
1175         SAXReader reader = new SAXReader();
1176 
1177         Reader r = null;

1180             pluginDescriptor = reader.read(r);

1256     private static DetectorFactorySelector getConstraintSelector(Element constraintElement, Plugin plugin,
1257             String singleDetectorElementName/*
1258              * , String
1259              * detectorCategoryElementName
1260              */) throws PluginException {
1261         Node node = constraintElement.selectSingleNode("./" + singleDetectorElementName);
1262         if (node != null) {
1263             String detectorClass = node.valueOf("@class");
1264             return new SingleDetectorFactorySelector(plugin, detectorClass);
1265         }
1266 
1267         node = constraintElement.selectSingleNode("./" + singleDetectorElementName + "Category");
1268         if (node != null) {
1269             boolean spanPlugins = Boolean.valueOf(node.valueOf("@spanplugins")).booleanValue();
1270 
1271             String categoryName = node.valueOf("@name");
1272             if (!"".equals(categoryName)) {
1273                 if ("reporting".equals(categoryName)) {
1274                     return new ReportingDetectorFactorySelector(spanPlugins ? null : plugin);
1275                 } else if ("training".equals(categoryName)) {
1276                     return new ByInterfaceDetectorFactorySelector(spanPlugins ? null : plugin, TrainingDetector.class);
1277                 } else if ("interprocedural".equals(categoryName)) {
1278                     return new ByInterfaceDetectorFactorySelector(spanPlugins ? null : plugin,
1279                             InterproceduralFirstPassDetector.class);
1280                 } else {
1281                     throw new PluginException("Invalid category name " + categoryName + " in constraint selector node");
1282                 }
1283             }
1284         }
1285 
1286         node = constraintElement.selectSingleNode("./" + singleDetectorElementName + "Subtypes");
1287         if (node != null) {
1288             boolean spanPlugins = Boolean.valueOf(node.valueOf("@spanplugins")).booleanValue();
1289 
1290             String superName = node.valueOf("@super");
1291             if (!"".equals(superName)) {
1292                 try {

1294                     return new ByInterfaceDetectorFactorySelector(spanPlugins ? null : plugin, superClass);
1295                 } catch (ClassNotFoundException e) {
1296                     throw new PluginException("Unknown class " + superName + " in constraint selector node");
1297                 }
1298             }
1299         }
1300         throw new PluginException("Invalid constraint selector node");
1301     }

1446                 }

1450         if (Plugin.getAllPlugins().size() > 1 && JavaWebStart.isRunningViaJavaWebstart()) {

1492     static void installWebStartPlugins() {
1493         URL pluginListProperties = getCoreResource("pluginlist.properties");
1494         BufferedReader in = null;
1495         if (pluginListProperties != null) {
1496             try {
1497                 DetectorFactoryCollection.jawsDebugMessage(pluginListProperties.toString());
1498                 URL base = getUrlBase(pluginListProperties);
1499 
1500                 in = UTF8.bufferedReader(pluginListProperties.openStream());
1501                 while (true) {
1502                     String plugin = in.readLine();
1503 
1504                     if (plugin == null) {
1505                         break;
1506                     }
1507                     URL url = new URL(base, plugin);
1508                     try {
1509                         URLConnection connection = url.openConnection();
1510                         String contentType = connection.getContentType();
1511                         DetectorFactoryCollection.jawsDebugMessage("contentType : " + contentType);
1512                         if (connection instanceof HttpURLConnection) {
1513                             ((HttpURLConnection) connection).disconnect();
1514                         }
1515                         loadInitialPlugin(url, true, false);
1516                     } catch (IOException e) {
1517                         DetectorFactoryCollection.jawsDebugMessage("error loading " + url + " : " + e.getMessage());
1518                     }
1519                 }
1520             } catch (IOException e) {
1521                 DetectorFactoryCollection.jawsDebugMessage("error : " + e.getMessage());
1522             } finally {
1523                 Util.closeSilently(in);
1524             }
1525         }
1526     }

1562     public static Summary validate(File file) throws IllegalArgumentException {
1563         String path = file.getPath();
1564         if (!file.getName().endsWith(".jar")) {
1565             String message = "File " + path + " is not a .jar file";
1566             throw new IllegalArgumentException(message);
1567         }
1568         if (!file.isFile() || !file.canRead()) {
1569             String message = "File " + path
1570                     + " is not a file or is not readable";
1571             throw new IllegalArgumentException(message);
1572         }
1573         if (file.length() == 0) {
1574             String message = "File " + path + " is empty";
1575             throw new IllegalArgumentException(message);
1576         }
1577 
1578         ZipFile zip = null;
1579         try {
1580             zip = new ZipFile(file);
1581             ZipEntry findbugsXML = zip.getEntry("findbugs.xml");
1582             if (findbugsXML == null) {
1583                 throw new IllegalArgumentException(
1584                         "plugin doesn't contain a findbugs.xml file");
1585             }
1586             ZipEntry messagesXML = zip.getEntry("messages.xml");
1587             if (messagesXML == null) {
1588                 throw new IllegalArgumentException(
1589                         "plugin doesn't contain a messages.xml file");
1590             }
1591             Document pluginDocument = parseDocument(zip.getInputStream(findbugsXML));
1592             String pluginId = pluginDocument.valueOf(XPATH_PLUGIN_PLUGINID).trim();
1593             String provider = pluginDocument.valueOf(XPATH_PLUGIN_PROVIDER).trim();
1594             String website = pluginDocument.valueOf(XPATH_PLUGIN_WEBSITE).trim();
1595             List<Document> msgDocuments = new ArrayList<Document>(3);
1596             for (String msgFile : getPotentialMessageFiles()) {
1597                 ZipEntry msgEntry = zip.getEntry(msgFile);
1598                 if (msgEntry == null) {
1599                     continue;
1600                 }
1601                 Document msgDocument = parseDocument(zip.getInputStream(msgEntry));
1602                 msgDocuments.add(msgDocument);
1603             }
1604             String shortDesc = findMessageText(msgDocuments,
1605                     XPATH_PLUGIN_SHORT_DESCRIPTION, "");
1606             return new Summary(pluginId, shortDesc, provider, website);
1607         } catch (DocumentException e) {
1608             throw new IllegalArgumentException(e);
1609         } catch (IOException e) {
1610             throw new IllegalArgumentException(e);
1611         } finally {
1612             Util.closeSilently(zip);
1613         }
1614     }
