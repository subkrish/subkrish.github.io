184     @Override
185     public @Nonnull Cloud getCloud() {
186         if (shouldNotUsePlugin) {
187             return CloudFactory.getPlainCloud(this);
188         }
189         Cloud result = cloud;
190         if (result == null) {
191             IGuiCallback callback = getProject().getGuiCallback();
192             result = cloud = CloudFactory.createCloudWithoutInitializing(this);
193             try {
194                 CloudFactory.initializeCloud(this, result);
195             } catch (Exception e) {
196                 LOGGER.log(Level.SEVERE, "Could not load cloud plugin "+ result.getCloudName(), e);
197                 callback.showMessageDialog("Unable to connect to " + result.getCloudName() + ": " + Util.getNetworkErrorMessage(e));
198                 if (CloudFactory.FAIL_ON_CLOUD_ERROR) {
199                     throw new IllegalStateException("Could not load FindBugs Cloud plugin - to avoid this message, " +
200                             "set -D" + CloudFactory.FAIL_ON_CLOUD_ERROR_PROP + "=false", e);
201                 }
202                 result = cloud = CloudFactory.getPlainCloud(this);
203             }
204             callback.registerCloud(getProject(), this, result);
205 
206 
207         }
208         if (!result.isInitialized()) {
209             LOGGER.log(Level.SEVERE, "Cloud " + result.getCloudName() + " is not initialized ");
210         }
211         if (bugsPopulated) {
212             result.bugsPopulated();
213         }
214         return result;
215     }

380     private void doReadXML(@WillClose InputStream in, @CheckForNull File base) throws IOException, DocumentException {
381         try {
382             checkInputStream(in);
383             Reader reader = Util.getReader(in);
384             doReadXML(reader, base);
385         } catch (RuntimeException e) {
386             in.close();
387             throw e;
388         } catch (IOException e) {
389             in.close();
390             throw e;
391         }
392     }

394     private void doReadXML(@WillClose Reader reader, @CheckForNull File base) throws IOException, DocumentException {
395         timeStartedLoading = System.currentTimeMillis();
396 
397         SAXBugCollectionHandler handler = new SAXBugCollectionHandler(this, base);
398         Profiler profiler = getProjectStats().getProfiler();
399         profiler.start(handler.getClass());
400         try {
401 
402             XMLReader xr;
403             try {
404                 xr = XMLReaderFactory.createXMLReader();
405             } catch (SAXException e) {
406                 AnalysisContext.logError("Couldn't create XMLReaderFactory", e);
407                 throw new DocumentException("Sax error ", e);
408             }
409             xr.setContentHandler(handler);
410             xr.setErrorHandler(handler);
411 
412             xr.parse(new InputSource(reader));
413         } catch (SAXParseException e) {
414             if (base != null) {
415                 throw new DocumentException("Parse error at line " + e.getLineNumber() + " : " + e.getColumnNumber() + " of "
416                         + base, e);
417             }
418             throw new DocumentException("Parse error at line " + e.getLineNumber() + " : " + e.getColumnNumber(), e);
419         } catch (SAXException e) {
420             // FIXME: throw SAXException from method?
421             if (base != null) {
422                 throw new DocumentException("Sax error while parsing " + base, e);
423             }
424             throw new DocumentException("Sax error ", e);
425         } finally {
426             Util.closeSilently(reader);
427             profiler.end(handler.getClass());
428         }
429         timeFinishedLoading = System.currentTimeMillis();
430         bugsPopulated();
431         // Presumably, project is now up-to-date
432         project.setModified(false);
433     }

537     public void computeBugHashes() {
538         if (preciseHashOccurrenceNumbersAvailable) {
539             return;
540         }
541         invalidateHashes();
542         HashMap<String, Integer> seen = new HashMap<String, Integer>();
543 
544         for (BugInstance bugInstance : getCollection()) {
545             String hash = bugInstance.getInstanceHash();
546 
547             Integer count = seen.get(hash);
548             if (count == null) {
549                 bugInstance.setInstanceOccurrenceNum(0);
550                 seen.put(hash, 0);
551             } else {
552                 bugInstance.setInstanceOccurrenceNum(count + 1);
553                 seen.put(hash, count + 1);
554             }
555         }
556         for (BugInstance bugInstance : getCollection()) {
557             bugInstance.setInstanceOccurrenceMax(seen.get(bugInstance.getInstanceHash()));
558         }
559         preciseHashOccurrenceNumbersAvailable = true;
560     }

562     /**
563      * Write the BugCollection to an XMLOutput object. The finish() method of
564      * the XMLOutput object is guaranteed to be called.
565      *
566      * <p>
567      * To write the SummaryHTML element, set property
568      * findbugs.report.SummaryHTML to "true".
569      * </p>
570      *
571      * @param xmlOutput
572      *            the XMLOutput object
573      */
574     @Override
575     public void writeXML(@WillClose XMLOutput xmlOutput) throws IOException {
576         assert project != null;
577         try {
578             writePrologue(xmlOutput);
579             if (withMessages) {
580                 computeBugHashes();
581                 getProjectStats().computeFileStats(this);
582                 String commonBase = null;
583                 for (String s : project.getSourceDirList()) {
584                     if (commonBase == null) {
585                         commonBase = s;
586                     } else {
587                         commonBase = commonBase.substring(0, commonPrefix(commonBase, s));
588                     }
589 
590                 }
591                 if (commonBase != null && commonBase.length() > 0) {
592                     if (commonBase.indexOf("/./") > 0) {
593                         commonBase = commonBase.substring(0, commonBase.indexOf("/."));
594                     }
595                     File base = new File(commonBase);
596                     if (base.exists() && base.isDirectory() && base.canRead()) {
597                         SourceLineAnnotation.generateRelativeSource(base, project);
598                     }
599                 }
600             }
601             if (earlyStats && !minimalXML) {
602                 getProjectStats().writeXML(xmlOutput, withMessages);
603             }
604 
605             // Write BugInstances
606             for (BugInstance bugInstance : getCollection()) {
607                 if (!applySuppressions || !project.getSuppressionFilter().match(bugInstance)) {
608                     bugInstance.writeXML(xmlOutput, this, withMessages);
609                 }
610             }
611 
612             writeEpilogue(xmlOutput);
613 
614         } finally {
615             xmlOutput.finish();
616             SourceLineAnnotation.clearGenerateRelativeSource();
617         }
618     }

620     int commonPrefix(String s1, String s2) {
621         int pos = 0;
622         while (pos < s1.length() && pos < s2.length() && s1.charAt(pos) == s2.charAt(pos)) {
623             pos++;
624         }
625         return pos;
626     }

628     @Override
629     public void writeEpilogue(XMLOutput xmlOutput) throws IOException {
630         if (withMessages) {
631             writeBugCategories(xmlOutput);
632             writeBugPatterns(xmlOutput);
633             writeBugCodes(xmlOutput);
634         }
635         // Errors, missing classes
636         if (!minimalXML) {
637             emitErrors(xmlOutput);
638         }
639 
640         if (!earlyStats && !minimalXML) {
641             // Statistics
642             getProjectStats().writeXML(xmlOutput, withMessages);
643         }
644 
645         // // Class and method hashes
646         // xmlOutput.openTag(CLASS_HASHES_ELEMENT_NAME);
647         // for (Iterator<ClassHash> i = classHashIterator(); i.hasNext();) {
648         // ClassHash classHash = i.next();
649         // classHash.writeXML(xmlOutput);
650         // }
651         // xmlOutput.closeTag(CLASS_HASHES_ELEMENT_NAME);
652 
653         // Class features
654         xmlOutput.openTag("ClassFeatures");
655         for (Iterator<ClassFeatureSet> i = classFeatureSetIterator(); i.hasNext();) {
656             ClassFeatureSet classFeatureSet = i.next();
657             classFeatureSet.writeXML(xmlOutput);
658         }
659         xmlOutput.closeTag("ClassFeatures");
660 
661         // AppVersions
662         xmlOutput.openTag(HISTORY_ELEMENT_NAME);
663         for (Iterator<AppVersion> i = appVersionIterator(); i.hasNext();) {
664             AppVersion appVersion = i.next();
665             appVersion.writeXML(xmlOutput);
666         }
667         xmlOutput.closeTag(HISTORY_ELEMENT_NAME);
668 
669         // Summary HTML
670         if (REPORT_SUMMARY_HTML) {
671             String html = getSummaryHTML();
672             if (html != null && !"".equals(html)) {
673                 xmlOutput.openTag(SUMMARY_HTML_ELEMENT_NAME);
674                 xmlOutput.writeCDATA(html);
675                 xmlOutput.closeTag(SUMMARY_HTML_ELEMENT_NAME);
676             }
677         }
678         xmlOutput.closeTag(ROOT_ELEMENT_NAME);
679     }

681     private void writeBugPatterns(XMLOutput xmlOutput) throws IOException {
682         // Find bug types reported
683         Set<String> bugTypeSet = new HashSet<String>();
684         for (Iterator<BugInstance> i = iterator(); i.hasNext();) {
685             BugInstance bugInstance = i.next();
686             BugPattern bugPattern = bugInstance.getBugPattern();
687             bugTypeSet.add(bugPattern.getType());
688         }
689         // Emit element describing each reported bug pattern
690         for (String bugType : bugTypeSet) {
691             BugPattern bugPattern = DetectorFactoryCollection.instance().lookupBugPattern(bugType);
692             if (bugPattern == null) {
693                 continue;
694             }
695 
696             XMLAttributeList attributeList = new XMLAttributeList();
697             attributeList.addAttribute("type", bugType);
698             attributeList.addAttribute("abbrev", bugPattern.getAbbrev());
699             attributeList.addAttribute("category", bugPattern.getCategory());
700             if (bugPattern.getCWEid() != 0) {
701                 attributeList.addAttribute("cweid", Integer.toString(bugPattern.getCWEid()));
702             }
703             xmlOutput.openTag("BugPattern", attributeList);
704 
705             xmlOutput.openTag("ShortDescription");
706             xmlOutput.writeText(bugPattern.getShortDescription());
707             xmlOutput.closeTag("ShortDescription");
708 
709             xmlOutput.openTag("Details");
710             xmlOutput.writeCDATA(bugPattern.getDetailText());
711             xmlOutput.closeTag("Details");
712 
713             xmlOutput.closeTag("BugPattern");
714         }
715     }

717     private void writeBugCodes(XMLOutput xmlOutput) throws IOException {
718         // Find bug codes reported
719         Set<String> bugCodeSet = new HashSet<String>();
720         for (Iterator<BugInstance> i = iterator(); i.hasNext();) {
721             BugInstance bugInstance = i.next();
722             String bugCode = bugInstance.getAbbrev();
723             if (bugCode != null) {
724                 bugCodeSet.add(bugCode);
725             }
726         }
727         // Emit element describing each reported bug code
728         for (String bugCodeAbbrev : bugCodeSet) {
729             BugCode bugCode = DetectorFactoryCollection.instance().getBugCode(bugCodeAbbrev);
730             String bugCodeDescription = bugCode.getDescription();
731             if (bugCodeDescription == null) {
732                 continue;
733             }
734 
735             XMLAttributeList attributeList = new XMLAttributeList();
736             attributeList.addAttribute("abbrev", bugCodeAbbrev);
737             if (bugCode.getCWEid() != 0) {
738                 attributeList.addAttribute("cweid", Integer.toString(bugCode.getCWEid()));
739             }
740             xmlOutput.openTag("BugCode", attributeList);
741 
742             xmlOutput.openTag("Description");
743             xmlOutput.writeText(bugCodeDescription);
744             xmlOutput.closeTag("Description");
745 
746             xmlOutput.closeTag("BugCode");
747         }
748     }

750     private void writeBugCategories(XMLOutput xmlOutput) throws IOException {
751         // Find bug categories reported
752         Set<String> bugCatSet = new HashSet<String>();
753         for (Iterator<BugInstance> i = iterator(); i.hasNext();) {
754             BugInstance bugInstance = i.next();
755             BugPattern bugPattern = bugInstance.getBugPattern();
756             bugCatSet.add(bugPattern.getCategory());
757         }
758         // Emit element describing each reported bug code
759         for (String bugCat : bugCatSet) {
760             String bugCatDescription = I18N.instance().getBugCategoryDescription(bugCat);
761             if (bugCatDescription == null) {
762                 continue;
763             }
764 
765             XMLAttributeList attributeList = new XMLAttributeList();
766             attributeList.addAttribute("category", bugCat);
767 
768             xmlOutput.openTag("BugCategory", attributeList);
769 
770             xmlOutput.openTag("Description");
771             xmlOutput.writeText(bugCatDescription);
772             xmlOutput.closeTag("Description");
773 
774             xmlOutput.closeTag("BugCategory");
775         }
776     }

831     private static void checkInputStream(@WillNotClose InputStream in) throws IOException {
832         if (!in.markSupported()) {
833             return;
834         }
835 
836         byte[] buf = new byte[200];
837         in.mark(buf.length);
838 
839         int numRead = 0;
840 
841         boolean isEOF = false;
842         while (numRead < buf.length && !isEOF) {
843             int n = in.read(buf, numRead, buf.length - numRead);
844             if (n < 0) {
845                 isEOF = true;
846             } else {
847                 numRead += n;
848             }
849         }
850 
851         in.reset();
852 
853         BufferedReader reader = new BufferedReader(Util.getReader(new ByteArrayInputStream(buf)));
854         try {
855             String line;
856             while ((line = reader.readLine()) != null) {
857                 if (line.startsWith("<BugCollection")) {
858                     return;
859                 }
860             }
861         } finally {
862             reader.close();
863         }
864 
865         throw new IOException("XML does not contain saved bug data");
866     }

893     public static class BugInstanceComparator implements Comparator<BugInstance> {
894 
895         private BugInstanceComparator() {
896         }
897 
898         @Override
899         public int compare(BugInstance lhs, BugInstance rhs) {
900             ClassAnnotation lca = lhs.getPrimaryClass();
901             ClassAnnotation rca = rhs.getPrimaryClass();
902             if (lca == null || rca == null) {
903                 throw new IllegalStateException("null class annotation: " + lca + "," + rca);
904             }
905             int cmp = lca.getClassName().compareTo(rca.getClassName());
906             if (cmp != 0) {
907                 return cmp;
908             }
909             return lhs.compareTo(rhs);
910         }
911 
912         public static final BugInstanceComparator instance = new BugInstanceComparator();
913     }

915     public static class MultiversionBugInstanceComparator extends BugInstanceComparator {
916 
917         private MultiversionBugInstanceComparator(){
918         }
919 
920         @Override
921         public int compare(BugInstance lhs, BugInstance rhs) {
922             int result = super.compare(lhs, rhs);
923             if (result != 0) {
924                 return result;
925             }
926             long diff = lhs.getFirstVersion() - rhs.getFirstVersion();
927             if (diff == 0) {
928                 diff = lhs.getLastVersion() - rhs.getLastVersion();
929             }
930             if (diff < 0) {
931                 return -1;
932             }
933             if (diff > 0) {
934                 return 1;
935             }
936             return 0;
937         }
938 
939         public static final MultiversionBugInstanceComparator instance = new MultiversionBugInstanceComparator();
940     }
