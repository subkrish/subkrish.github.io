20  package edu.umd.cs.findbugs.ba.jsr305;
21  
22  import edu.umd.cs.findbugs.ba.XMethod;
23  
24  /**
25   * Accumulate type qualifier annotations for a method parameter.
26   *
27   * @author David Hovemeyer
28   */
29  public class ParameterAnnotationAccumulator extends AbstractMethodAnnotationAccumulator {
30      private final int parameter;
31  
32      private final ParameterAnnotationLookupResult result;
33  
34      private boolean overrides = false;
35  
36      /**
37       * Constructor.
38       *
39       * @param typeQualifierValue
40       *            TypeQualifierValue specifying kind of application to lookup
41       * @param xmethod
42       *            method we want to find parameter annotation for
43       * @param parameter
44       *            the parameter (0 == first parameter)
45       */
46      protected ParameterAnnotationAccumulator(TypeQualifierValue<?> typeQualifierValue, XMethod xmethod, int parameter) {
47          super(typeQualifierValue, xmethod);
48          this.parameter = parameter;
49          this.result = new ParameterAnnotationLookupResult();
50      }
51  
52      @Override
53      public TypeQualifierAnnotationLookupResult getResult() {
54          return result;
55      }
56  
57      /**
58       * Returns true if the method overrides/implements a method in a superclass
59       * or interface
60       */
61      @Override
62      public boolean overrides() {
63          return overrides;
64      }
65  
66      @Override
67      protected TypeQualifierAnnotation lookupAnnotation(XMethod xm) {
68          overrides = true;
69          TypeQualifierAnnotation result1 = TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(xm, parameter,
70                  getTypeQualifierValue());
71          if (TypeQualifierApplications.DEBUG && result1 != null) {
72              System.out.println("Inherit " + result1.when + " from " + xm);
73          }
74          return result1;
75  
76      }
77  
78  }
