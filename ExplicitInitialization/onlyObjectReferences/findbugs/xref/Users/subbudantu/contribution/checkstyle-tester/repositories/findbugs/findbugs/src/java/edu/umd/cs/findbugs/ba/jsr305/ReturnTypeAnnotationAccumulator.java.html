20  package edu.umd.cs.findbugs.ba.jsr305;
21  
22  import edu.umd.cs.findbugs.ba.XMethod;
23  
24  /**
25   * Accumulate relevant return type annotations for a given method by traversing
26   * its supertypes.
27   *
28   * @author David Hovemeyer
29   */
30  class ReturnTypeAnnotationAccumulator extends AbstractMethodAnnotationAccumulator {
31      private final TypeQualifierAnnotationLookupResult result;
32  
33      private boolean overrides = false;
34  
35      /**
36       * Constructor.
37       *
38       * @param typeQualifierValue
39       *            TypeQualifierValue specifying the kind of annotation we want
40       *            to look up
41       * @param xmethod
42       *            method whose effective return type annotation we want
43       */
44      public ReturnTypeAnnotationAccumulator(TypeQualifierValue<?> typeQualifierValue, XMethod xmethod) {
45          super(typeQualifierValue, xmethod);
46          this.result = new ReturnTypeAnnotationLookupResult();
47      }
48  
49      @Override
50      public TypeQualifierAnnotationLookupResult getResult() {
51          return result;
52      }
53  
54      /**
55       * Returns true if the method overrides/implements a method in a superclass
56       * or interface
57       */
58      @Override
59      public boolean overrides() {
60          return overrides;
61      }
62  
63      @Override
64      protected TypeQualifierAnnotation lookupAnnotation(XMethod xm) {
65          overrides = true;
66          return TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(xm, getTypeQualifierValue());
67      }
68  
69  }
