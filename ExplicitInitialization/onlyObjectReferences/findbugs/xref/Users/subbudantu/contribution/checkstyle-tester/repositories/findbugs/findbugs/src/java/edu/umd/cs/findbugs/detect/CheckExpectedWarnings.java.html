67  /**
68   * Check uses of the ExpectWarning and NoWarning annotations. This is for
69   * internal testing of FindBugs (against findbugsTestCases).
70   *
71   * @author David Hovemeyer
72   */
73  public class CheckExpectedWarnings implements Detector2, NonReportingDetector {
74      private static final boolean DEBUG = SystemProperties.getBoolean("cew.debug");
75  
76      private BugReporter reporter;
77  
78      private final BugCollection bugCollection;

80      private boolean initialized = false;
81      private Map<ClassDescriptor, Collection<BugInstance>> warningsByClass;
82      private Map<MethodDescriptor, Collection<BugInstance>> warningsByMethod;
83      private Map<FieldDescriptor, Collection<BugInstance>> warningsByField;
84  
85      private ClassDescriptor expectWarning;
86  
87      private ClassDescriptor noWarning;
88  
89      private ClassDescriptor desireWarning;
90  
91      private ClassDescriptor desireNoWarning;
92  
93      private boolean warned;

95      public CheckExpectedWarnings(BugReporter bugReporter) {
96          bugCollection = bugReporter.getBugCollection();
97          if (bugCollection != null) {
98              reporter = bugReporter;
99              expectWarning = DescriptorFactory.createClassDescriptor(ExpectWarning.class);
100             noWarning = DescriptorFactory.createClassDescriptor(NoWarning.class);
101             desireWarning = DescriptorFactory.createClassDescriptor(DesireWarning.class);
102             desireNoWarning = DescriptorFactory.createClassDescriptor(DesireNoWarning.class);
103         }
104     }

106     @Override
107     public void visitClass(ClassDescriptor classDescriptor) throws CheckedAnalysisException {
108         if (reporter == null) {
109             if (!warned) {
110                 System.err
111                 .println("*** NOTE ***: CheckExpectedWarnings disabled because bug reporter doesn't use a BugCollection");
112                 warned = true;
113             }
114             return;
115         }
116 
117         if (!initialized) {
118 
119             initialized = true;
120             //
121             // Build index of all warnings reported so far, by method.
122             // Because this detector runs in a later pass than any
123             // reporting detector, all warnings should have been
124             // produced by this point.
125             //
126 
127             warningsByClass = new HashMap<ClassDescriptor, Collection<BugInstance>>();
128             warningsByMethod = new HashMap<MethodDescriptor, Collection<BugInstance>>();
129             warningsByField = new HashMap<FieldDescriptor, Collection<BugInstance>>();
130 
131             for (Iterator<BugInstance> i = bugCollection.iterator(); i.hasNext();) {
132                 BugInstance warning = i.next();
133                 MethodAnnotation method = warning.getPrimaryMethod();
134                 if (method != null) {
135                     MethodDescriptor methodDesc = method.toMethodDescriptor();
136                     Collection<BugInstance> warnings = warningsByMethod.get(methodDesc);
137                     if (warnings == null) {
138                         warnings = new LinkedList<BugInstance>();
139                         warningsByMethod.put(methodDesc, warnings);
140                     }
141                     warnings.add(warning);
142                 }
143                 FieldAnnotation field = warning.getPrimaryField();
144                 if (field != null) {
145                     if (DEBUG) {
146                         System.out.println("primary field of " + field + " for " + warning);
147                     }
148                     FieldDescriptor fieldDescriptor = field.toFieldDescriptor();
149                     Collection<BugInstance> warnings = warningsByField.get(fieldDescriptor);
150 
151                     if (warnings == null) {
152                         warnings = new LinkedList<BugInstance>();
153                         warningsByField.put(fieldDescriptor, warnings);
154                     }
155                     warnings.add(warning);
156                 }
157 
158                 ClassAnnotation clazz = warning.getPrimaryClass();
159                 if (clazz != null) {
160                     ClassDescriptor classDesc = clazz.getClassDescriptor();
161                     if(field != null && classDesc.equals(field.getClassDescriptor())) {
162                         continue;
163                     }
164                     if (method != null && classDesc.equals(method.getClassDescriptor())) {
165                         continue;
166                     }
167                     Collection<BugInstance> warnings = warningsByClass.get(classDesc);
168                     if (warnings == null) {
169                         warnings = new LinkedList<BugInstance>();
170                         warningsByClass.put(classDesc, warnings);
171                     }
172                     warnings.add(warning);
173                 }
174 
175             }
176 
177         }
178 
179         XClass xclass = Global.getAnalysisCache().getClassAnalysis(XClass.class, classDescriptor);
180         List<? extends XMethod> methods = xclass.getXMethods();
181         if (DEBUG) {
182             System.out.println("CEW: checking " + xclass.toString());
183         }
184         if (xclass.isSynthetic()) {
185             if (DEBUG) {
186                 System.out.println("Skipping synthetic classxclass " + xclass.toString());
187             }
188             return;
189         }
190         check(xclass, expectWarning, true, HIGH_PRIORITY);
191         check(xclass, desireWarning, true, NORMAL_PRIORITY);
192         check(xclass, noWarning, false, HIGH_PRIORITY);
193         check(xclass, desireNoWarning, false, NORMAL_PRIORITY);
194 
195         for (XMethod xmethod : methods) {
196             if (DEBUG) {
197                 System.out.println("CEW: checking " + xmethod.toString());
198             }
199             if (xmethod.isSynthetic()) {
200                 if (DEBUG) {
201                     System.out.println("Skipping synthetic method " + xmethod.toString());
202                 }
203                 continue;
204             }
205             check(xmethod, expectWarning, true, HIGH_PRIORITY);
206             check(xmethod, desireWarning, true, NORMAL_PRIORITY);
207             check(xmethod, noWarning, false, HIGH_PRIORITY);
208             check(xmethod, desireNoWarning, false, NORMAL_PRIORITY);
209         }
210         for (XField xfield : xclass.getXFields()) {
211             if (DEBUG) {
212                 System.out.println("CEW: checking " + xfield.toString());
213             }
214             if (xfield.isSynthetic()) {
215                 if (DEBUG) {
216                     System.out.println("Skipping synthetic field " + xfield.toString());
217                 }
218                 continue;
219             }
220             check(xfield, expectWarning, true, HIGH_PRIORITY);
221             check(xfield, desireWarning, true, NORMAL_PRIORITY);
222             check(xfield, noWarning, false, HIGH_PRIORITY);
223             check(xfield, desireNoWarning, false, NORMAL_PRIORITY);
224         }
225 
226     }

228     private void check(XClass xclass, ClassDescriptor annotation, boolean expectWarnings, int priority) {
229         AnnotationValue expect = xclass.getAnnotation(annotation);
230         if (expect == null) {
231             return;
232         }
233         if (DEBUG) {
234             System.out.println("*** Found " + annotation + " annotation on " + xclass);
235         }
236         ClassDescriptor descriptor = xclass.getClassDescriptor();
237         Collection<BugInstance> warnings = warningsByClass.get(descriptor);
238         check(expect, descriptor, warnings, expectWarnings, priority, descriptor);
239     }

241     private void check(XMethod xmethod, ClassDescriptor annotation, boolean expectWarnings, int priority) {
242         AnnotationValue expect = xmethod.getAnnotation(annotation);
243         if (expect == null) {
244             return;
245         }
246         if (DEBUG) {
247             System.out.println("*** Found " + annotation + " annotation on " + xmethod);
248         }
249         FieldOrMethodDescriptor descriptor = xmethod.getMethodDescriptor();
250         Collection<BugInstance> warnings = warningsByMethod.get(descriptor);
251         check(expect, descriptor, warnings, expectWarnings, priority, descriptor.getClassDescriptor());
252     }

254     private void check(XField xfield, ClassDescriptor annotation, boolean expectWarnings, int priority) {
255         AnnotationValue expect = xfield.getAnnotation(annotation);
256         if (expect == null) {
257             return;
258         }
259 
260         if (DEBUG) {
261             System.out.println("*** Found " + annotation + " annotation on " + xfield);
262         }
263         FieldOrMethodDescriptor descriptor = xfield.getFieldDescriptor();
264         Collection<BugInstance> warnings = warningsByField.get(descriptor);
265         check(expect, descriptor, warnings, expectWarnings, priority, descriptor.getClassDescriptor());
266     }

268     private void check(AnnotationValue expect, Object descriptor,
269             Collection<BugInstance> warnings, boolean expectWarnings, int priority, ClassDescriptor cd) {
270 
271         if (expect != null) {
272 
273             String expectedBugCodes = (String) expect.getValue("value");
274             EnumValue wantedConfidence = (EnumValue) expect.getValue("confidence");
275             EnumValue wantedPriority = (EnumValue) expect.getValue("priority");
276             Integer num = (Integer) expect.getValue("num");
277             if (num == null) {
278                 num = (expectWarnings ? 1 : 0);
279             }
280             Integer rank = (Integer) expect.getValue("rank");
281             if (rank == null) {
282                 rank = BugRanker.VISIBLE_RANK_MAX;
283             }
284 
285             int minPriority = Confidence.LOW.getConfidenceValue();
286             if (wantedConfidence != null) {
287                 minPriority = Confidence.valueOf(wantedConfidence.value).getConfidenceValue();
288             } else if (wantedPriority != null) {
289                 minPriority = Priority.valueOf(wantedPriority.value).getPriorityValue();
290             }
291 
292             if (DEBUG)  {
293                 if (warnings == null) {
294                     System.out.println("Checking " + expectedBugCodes + " against no bugs");
295                 } else {
296                     System.out.println("Checking " + expectedBugCodes + " against " + warnings.size() + " bugs");
297                     for (BugInstance b : warnings) {
298                         System.out.println("  " + b.getType());
299                     }
300                 }
301             }
302             if (expectedBugCodes == null || expectedBugCodes.trim().length() == 0) {
303                 checkAnnotation(null, warnings, expectWarnings, priority, rank, num, descriptor, minPriority, cd);
304             } else {
305                 StringTokenizer tok = new StringTokenizer(expectedBugCodes, ",");
306                 while (tok.hasMoreTokens()) {
307                     String bugCode = tok.nextToken().trim();
308                     checkAnnotation(bugCode, warnings, expectWarnings, priority, rank, num, descriptor, minPriority, cd);
309                 }
310             }
311         }
312     }

314     public void checkAnnotation(@CheckForNull String bugCode, Collection<BugInstance> warnings, boolean expectWarnings, int priority,
315             Integer rank, Integer num, Object methodDescriptor, int minPriority, ClassDescriptor cd) {
316 
317         String bugCodeMessage = bugCode != null ? bugCode : "any bug";
318         Collection<SourceLineAnnotation> bugs = countWarnings(warnings, bugCode, minPriority,
319                 rank);
320         if (expectWarnings && bugs.size() < num) {
321             if (DetectorFactoryCollection.instance().isDisabledByDefault(bugCode)) {
322                 return;
323             }
324             BugInstance bug = makeWarning("FB_MISSING_EXPECTED_WARNING", methodDescriptor, priority, cd).addString(bugCodeMessage);
325             if (!bugs.isEmpty()) {
326                 bug.addString(String.format("Expected %d bugs, saw %d", num, bugs.size()));
327             }
328             reporter.reportBug(bug);
329         } else if (bugs.size() > num) {
330             // More bugs than expected
331             BugInstance bug = makeWarning("FB_UNEXPECTED_WARNING", methodDescriptor, priority, cd).addString(bugCodeMessage);
332             if (!expectWarnings) {
333                 // Wanted no more than this many warnings
334                 for (SourceLineAnnotation s : bugs) {
335                     reporter.reportBug(bug.add(s));
336                 }
337             } else if(num > 1){
338                 // For example, we told it that we expected 3 warnings, and saw 4 warnings
339                 // num == 1 is default value. So if we set a non default value, and see more warnings
340                 // as expected, it's a problem
341                 bug.addString(String.format("Expected %d bugs, saw %d", num, bugs.size()));
342                 reporter.reportBug(bug);
343             }
344         }
345     }

347     public BugInstance makeWarning(String bugPattern, Object descriptor, int priority, ClassDescriptor cd) {
348         BugInstance bug = new BugInstance(this, bugPattern, priority).addClass(cd);
349         if (descriptor instanceof FieldDescriptor) {
350             bug.addField((FieldDescriptor)descriptor);
351         } else if (descriptor instanceof MethodDescriptor) {
352             bug.addMethod((MethodDescriptor)descriptor);
353         } else if (descriptor instanceof ClassDescriptor) {
354             bug.addClass((ClassDescriptor)descriptor);
355         }
356         if (DEBUG) {
357             System.out.println("Reporting " + bug);
358         }
359         return bug;
360 
361     }

363     private static Collection<SourceLineAnnotation> countWarnings( Collection<BugInstance> warnings,
364             @CheckForNull String bugCode,
365             int desiredPriority, int rank) {
366 
367         Collection<SourceLineAnnotation> matching = new HashSet<SourceLineAnnotation>();
368         DetectorFactoryCollection i18n = DetectorFactoryCollection.instance();
369         boolean matchPattern = false;
370         try {
371             i18n.getBugCode(bugCode);
372         } catch (IllegalArgumentException e) {
373             matchPattern = true;
374         }
375 
376         if (warnings != null) {
377             for (BugInstance warning : warnings) {
378                 if (warning.getPriority() > desiredPriority) {
379                     continue;
380                 }
381                 if (warning.getBugRank() > rank) {
382                     continue;
383                 }
384                 if (bugCode == null) {
385                     matching.add(warning.getPrimarySourceLineAnnotation());
386                     matching.addAll(warning.getAnotherInstanceSourceLineAnnotations());
387                     continue;
388                 }
389                 BugPattern pattern = warning.getBugPattern();
390                 String match;
391                 if (matchPattern) {
392                     match = pattern.getType();
393                 } else {
394                     match = pattern.getAbbrev();
395                 }
396                 if (match.equals(bugCode)) {
397                     matching.add(warning.getPrimarySourceLineAnnotation());
398                     matching.addAll(warning.getAnotherInstanceSourceLineAnnotations());
399 
400                 }
401             }
402         }
403         return matching;
404     }

406     @Override
407     public void finishPass() {
408         HashSet<BugPattern> claimedReported = new HashSet<BugPattern>();
409         for (DetectorFactory d : DetectorFactoryCollection.instance().getFactories()) {
410             claimedReported.addAll(d.getReportedBugPatterns());
411         }
412         for (BugPattern b : DetectorFactoryCollection.instance().getBugPatterns()) {
413             String category = b.getCategory();
414             if (!b.isDeprecated() && !"EXPERIMENTAL".equals(category) && !claimedReported.contains(b)) {
415                 AnalysisContext.logError("No detector claims " + b.getType());
416             }
417         }
418 
419     }
