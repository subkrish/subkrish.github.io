115     /**
116      * Minimum percent of unbiased field accesses that must be synchronized in
117      * order to report a field as inconsistently synchronized. This is meant to
118      * distinguish incidental synchronization from intentional synchronization.
119      */
120     private static final int MIN_SYNC_PERCENT = SystemProperties.getInt("findbugs.fis.minSyncPercent", 50);
121 
122     /**
123      * Bias that writes are given with respect to reads. The idea is that this
124      * should be above 1.0, because unsynchronized writes are more dangerous
125      * than unsynchronized reads.
126      */
127     private static final double WRITE_BIAS = Double.parseDouble(SystemProperties.getProperty("findbugs.fis.writeBias", "2.0"));
128 
129     /**
130      * Factor which the biased number of unsynchronized accesses is multiplied
131      * by. I.e., for factor <i>f</i>, if <i>nUnsync</i> is the biased number of
132      * unsynchronized accesses, and <i>nSync</i> is the biased number of
133      * synchronized accesses, and
134      *
135      * <pre>
136      *      <i>f</i>(<i>nUnsync</i>) &gt; <i>nSync</i>
137      * </pre>
138      *
139      * then we report a bug. Default value is 2.0, which means that we report a
140      * bug if more than 1/3 of accesses are unsynchronized.
141      * <p/>
142      * <p>
143      * Note that <code>MIN_SYNC_PERCENT</code> also influences whether we report
144      * a bug: it specifies the minimum unbiased percentage of synchronized
145      * accesses.
146      */
147     private static final double UNSYNC_FACTOR = Double.parseDouble(SystemProperties.getProperty("findbugs.fis.unsyncFactor",
148             "1.6"));

207     public static boolean isServletField(XField field) {
208         ClassDescriptor classDescriptor = field.getClassDescriptor();
209 
210         try {
211             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
212             if (subtypes2.isSubtype(classDescriptor, servlet) && !subtypes2.isSubtype(classDescriptor, singleThreadedServlet)) {
213                 return true;
214             }
215         } catch (ClassNotFoundException e) {
216             assert true;
217         }
218         if (classDescriptor.getClassName().endsWith("Servlet")) {
219             return true;
220         }
221         return false;
222     }

224     /**
225      * The access statistics for a field. Stores the number of locked and
226      * unlocked reads and writes, as well as the number of accesses made with a
227      * lock held.
228      */
229     private static class FieldStats {
230 
231         private final int[] countList = new int[6];
232 
233         private int numLocalLocks = 0;
234 
235         private int numGetterMethodAccesses = 0;
236 
237         private List<FieldAccess> unsyncAccessList = Collections.emptyList();
238 
239         private List<FieldAccess> syncAccessList = Collections.emptyList();
240 
241         boolean interesting = true;
242 
243         final boolean servletField;
244 
245         FieldStats(XField field) {
246             servletField = FindInconsistentSync2.isServletField(field);
247         }
248 
249         public void addAccess(int kind) {
250             countList[kind]++;
251         }
252 
253         public int getNumAccesses(int kind) {
254             return countList[kind];
255         }
256 
257         public void addLocalLock() {
258             numLocalLocks++;
259         }
260 
261         public int getNumLocalLocks() {
262             return numLocalLocks;
263         }
264 
265         public void addGetterMethodAccess() {
266             numGetterMethodAccesses++;
267         }
268 
269         public int getNumGetterMethodAccesses() {
270             return numGetterMethodAccesses;
271         }
272 
273         public boolean isInteresting() {
274             return interesting;
275         }
276 
277         public boolean isServletField() {
278             return servletField;
279         }
280 
281         public boolean hasAnySynchronizedAccesses() {
282             return interesting && !syncAccessList.isEmpty();
283         }
284 
285         public void addAccess(MethodDescriptor method, InstructionHandle handle, boolean isLocked) {
286             if (!interesting) {
287                 return;
288             }
289             if (!SYNC_ACCESS && isLocked) {
290                 return;
291             }
292 
293             if (!servletField && !isLocked && syncAccessList.size() == 0 && unsyncAccessList.size() > 6) {
294                 interesting = false;
295                 syncAccessList = null;
296                 unsyncAccessList = null;
297                 return;
298             }
299             FieldAccess fa = new FieldAccess(method, handle.getPosition());
300             if (isLocked) {
301                 syncAccessList = Util.addTo(syncAccessList, fa);
302             } else {
303                 unsyncAccessList = Util.addTo(unsyncAccessList, fa);
304             }
305         }
306 
307         public Iterator<SourceLineAnnotation> unsyncAccessIterator() {
308             if (!interesting) {
309                 throw new IllegalStateException("Not interesting");
310             }
311             return FieldAccess.asSourceLineAnnotation(unsyncAccessList).iterator();
312         }
313 
314         public Iterator<SourceLineAnnotation> syncAccessIterator() {
315             if (!interesting) {
316                 throw new IllegalStateException("Not interesting");
317             }
318             return FieldAccess.asSourceLineAnnotation(syncAccessList).iterator();
319         }

342     @Override
343     public void visitClassContext(ClassContext classContext) {
344         JavaClass javaClass = classContext.getJavaClass();
345         if (DEBUG) {
346             System.out.println("******** Analyzing class " + javaClass.getClassName());
347         }
348 
349         // Build self-call graph
350         SelfCalls selfCalls = new SelfCalls(classContext) {
351             @Override
352             public boolean wantCallsFor(Method method) {
353                 return !method.isPublic();
354             }
355         };
356 
357         Set<Method> lockedMethodSet;
358         // Set<Method> publicReachableMethods;
359         Set<Method> allMethods = new HashSet<Method>(Arrays.asList(javaClass.getMethods()));
360 
361         try {
362             selfCalls.execute();
363             CallGraph callGraph = selfCalls.getCallGraph();
364             if (DEBUG) {
365                 System.out.println("Call graph (not unlocked methods): " + callGraph.getNumVertices() + " nodes, "
366                         + callGraph.getNumEdges() + " edges");
367             }
368             // Find call edges that are obviously locked
369             Set<CallSite> obviouslyLockedSites = findObviouslyLockedCallSites(classContext, selfCalls);
370             lockedMethodSet = findNotUnlockedMethods(classContext, selfCalls, obviouslyLockedSites);
371             lockedMethodSet.retainAll(findLockedMethods(classContext, selfCalls, obviouslyLockedSites));
372             // publicReachableMethods = findPublicReachableMethods(classContext,
373             // selfCalls);
374         } catch (CFGBuilderException e) {
375             bugReporter.logError("Error finding locked call sites", e);
376             return;
377         } catch (DataflowAnalysisException e) {
378             bugReporter.logError("Error finding locked call sites", e);
379             return;
380         }
381 
382 
383         for (Method method : allMethods) {
384             if (DEBUG) {
385                 System.out.println("******** considering method " + method.getName());
386             }
387 
388             if (classContext.getMethodGen(method) == null) {
389                 continue;
390             }
391 
392             if (method.getName().startsWith("access$")) {
393                 // Ignore inner class access methods;
394                 // we will treat calls to them as field accesses
395                 continue;
396             }
397 
398             String name = method.getName();
399 
400             boolean inConstructor = "<init>".equals(name) || "<clinit>".equals(name)
401                     || "readObject".equals(name) || "clone".equals(name) || "close".equals(name)
402                     || "finalize".equals(name);
403 
404             if (inConstructor) {
405                 continue;
406             }
407 
408             if (DEBUG) {
409                 System.out.println("******** Analyzing method " + method.getName());
410             }
411 
412             try {
413                 analyzeMethod(classContext, method, lockedMethodSet);
414             } catch (CFGBuilderException e) {
415                 bugReporter.logError("Error analyzing method", e);
416             } catch (DataflowAnalysisException e) {
417                 bugReporter.logError("Error analyzing method", e);
418             }
419         }
420         for (Field f : javaClass.getFields()) {
421             if (f.isPrivate()) {
422                 XField xf = XFactory.getExactXField(classContext.getClassDescriptor().getClassName(), f);
423                 FieldStats stats = statMap.get(xf);
424                 if (stats == null) {
425                     continue;
426                 }
427                 if (!stats.isServletField() && !stats.hasAnySynchronizedAccesses()) {
428                     statMap.remove(xf);
429                 }
430             }
431         }
432     }

434     @Override
435     public void report() {
436         if(statMap.isEmpty()){
437             return;
438         }
439         JCIPAnnotationDatabase jcipAnotationDatabase = AnalysisContext.currentAnalysisContext().getJCIPAnnotationDatabase();
440         for (Entry<XField, FieldStats> entry : statMap.entrySet()) {
441             XField xfield = entry.getKey();
442             FieldStats stats = entry.getValue();
443             if (!stats.isInteresting()) {
444                 continue;
445             }
446             boolean notThreadSafe = jcipAnotationDatabase.hasClassAnnotation(xfield.getClassName(), "NotThreadSafe");
447             if (notThreadSafe) {
448                 continue;
449             }
450             ElementValue guardedByValue = jcipAnotationDatabase.getFieldAnnotation(xfield, "GuardedBy");
451             boolean guardedByThis;
452             if(guardedByValue != null){
453                 guardedByThis = "this".equals(guardedByValue.stringifyValue());
454             } else {
455                 guardedByThis = false;
456             }
457             boolean threadSafe = jcipAnotationDatabase.hasClassAnnotation(xfield.getClassName(), "ThreadSafe");
458 
459             WarningPropertySet<InconsistentSyncWarningProperty> propertySet = new WarningPropertySet<InconsistentSyncWarningProperty>();
460 
461             int numReadUnlocked = stats.getNumAccesses(READ_UNLOCKED);
462             int numWriteUnlocked = stats.getNumAccesses(WRITE_UNLOCKED);
463             int numNullCheckUnlocked = stats.getNumAccesses(NULLCHECK_UNLOCKED);
464 
465             int numReadLocked = stats.getNumAccesses(READ_LOCKED);
466             int numWriteLocked = stats.getNumAccesses(WRITE_LOCKED);
467             int numNullCheckLocked = stats.getNumAccesses(NULLCHECK_LOCKED);
468 
469             int extra = 0;
470             if (numWriteUnlocked > 0) {
471                 extra = numNullCheckLocked;
472             }
473             int locked = numReadLocked + numWriteLocked + numNullCheckLocked;
474             int biasedLocked = numReadLocked + (int) (WRITE_BIAS * (numWriteLocked + numNullCheckLocked + extra));
475             int unlocked = numReadUnlocked + numWriteUnlocked + numNullCheckUnlocked;
476             int biasedUnlocked = numReadUnlocked + (int) (WRITE_BIAS * (numWriteUnlocked));
477             // int writes = numWriteLocked + numWriteUnlocked;
478 
479             if (unlocked == 0) {
480                 continue;
481                 // propertySet.addProperty(InconsistentSyncWarningProperty.NEVER_UNLOCKED);
482             }
483 
484             if (guardedByThis) {
485                 propertySet.addProperty(InconsistentSyncWarningProperty.ANNOTATED_AS_GUARDED_BY_THIS);
486 
487             }
488 
489             if (threadSafe) {
490                 propertySet.addProperty(InconsistentSyncWarningProperty.ANNOTATED_AS_THREAD_SAFE);
491 
492             }
493             if (!guardedByThis && locked == 0 && !stats.isServletField()) {
494                 continue;
495                 // propertySet.addProperty(InconsistentSyncWarningProperty.NEVER_LOCKED);
496             }
497 
498             if (stats.isServletField() && numWriteLocked == 0 && numWriteUnlocked == 0) {
499                 continue;
500             }
501 
502             if (DEBUG) {
503                 System.out.println("IS2: " + xfield);
504                 if (guardedByThis) {
505                     System.out.println("Guarded by this");
506                 }
507                 System.out.println("  RL: " + numReadLocked);
508                 System.out.println("  WL: " + numWriteLocked);
509                 System.out.println("  NL: " + numNullCheckLocked);
510 
511                 System.out.println("  RU: " + numReadUnlocked);
512                 System.out.println("  WU: " + numWriteUnlocked);
513                 System.out.println("  NU: " + numNullCheckUnlocked);
514             }
515             if (!EVAL && numReadUnlocked > 0 && ((int) (UNSYNC_FACTOR * (biasedUnlocked - 1))) > biasedLocked
516                     && !stats.isServletField()) {
517                 // continue;
518                 propertySet.addProperty(InconsistentSyncWarningProperty.MANY_BIASED_UNLOCKED);
519             }
520 
521             // NOTE: we ignore access to public, volatile, and final fields
522 
523             if (numWriteUnlocked + numWriteLocked == 0) {
524                 // No writes outside of constructor
525                 if (DEBUG) {
526                     System.out.println("  No writes outside of constructor");
527                 }
528                 propertySet.addProperty(InconsistentSyncWarningProperty.NEVER_WRITTEN);
529                 // continue;
530             }
531 
532             if (numReadUnlocked + numReadLocked == 0) {
533                 // No reads outside of constructor
534                 if (DEBUG) {
535                     System.out.println("  No reads outside of constructor");
536                 }
537                 // continue;
538                 propertySet.addProperty(InconsistentSyncWarningProperty.NEVER_READ);
539             }
540 
541             if (stats.getNumLocalLocks() == 0) {
542                 if (DEBUG) {
543                     System.out.println("  No local locks");
544                 }
545                 // continue;
546                 propertySet.addProperty(InconsistentSyncWarningProperty.NO_LOCAL_LOCKS);
547             }
548 
549             int freq, printFreq;
550             if (locked + unlocked > 0) {
551                 freq = (100 * locked) / (locked + unlocked);
552                 printFreq = (100 * locked) / (locked + unlocked + numNullCheckUnlocked);
553             } else {
554                 printFreq = freq = 0;
555             }
556 
557             if (freq < MIN_SYNC_PERCENT) {
558                 // continue;
559                 propertySet.addProperty(InconsistentSyncWarningProperty.BELOW_MIN_SYNC_PERCENT);
560             }
561             if (DEBUG) {
562                 System.out.println("  Sync %: " + freq);
563             }
564 
565             if (stats.getNumGetterMethodAccesses() >= unlocked) {
566                 // Unlocked accesses are only in getter method(s).
567                 propertySet.addProperty(InconsistentSyncWarningProperty.ONLY_UNSYNC_IN_GETTERS);
568             }
569 
570             // At this point, we report the field as being inconsistently
571             // synchronized
572             if (stats.isServletField()) {
573                 propertySet.addProperty(InconsistentSyncWarningProperty.MUTABLE_SERVLET_FIELD);
574             }
575 
576             BugInstance bugInstance;
577             if (stats.isServletField()) {
578                 bugInstance = new BugInstance(this, "MSF_MUTABLE_SERVLET_FIELD", Priorities.NORMAL_PRIORITY).addClass(
579                         xfield.getClassName()).addField(xfield);
580             } else {
581                 bugInstance = new BugInstance(this, guardedByThis ? "IS_FIELD_NOT_GUARDED" : "IS2_INCONSISTENT_SYNC",
582                         Priorities.NORMAL_PRIORITY).addClass(xfield.getClassName()).addField(xfield).addInt(printFreq)
583                         .describe(IntAnnotation.INT_SYNC_PERCENT);
584             }
585 
586             propertySet.decorateBugInstance(bugInstance);
587             // Add source lines for unsynchronized accesses
588             for (Iterator<SourceLineAnnotation> j = stats.unsyncAccessIterator(); j.hasNext();) {
589                 SourceLineAnnotation accessSourceLine = j.next();
590                 bugInstance.addSourceLine(accessSourceLine).describe("SOURCE_LINE_UNSYNC_ACCESS");
591             }
592 
593             if (SYNC_ACCESS) {
594                 // Add source line for synchronized accesses;
595                 // useful for figuring out what the detector is doing
596                 for (Iterator<SourceLineAnnotation> j = stats.syncAccessIterator(); j.hasNext();) {
597                     SourceLineAnnotation accessSourceLine = j.next();
598                     bugInstance.addSourceLine(accessSourceLine).describe("SOURCE_LINE_SYNC_ACCESS");
599                 }
600             }
601 
602             if (EVAL) {
603                 bugInstance.addInt(biasedLocked).describe("INT_BIASED_LOCKED");
604                 bugInstance.addInt(biasedUnlocked).describe("INT_BIASED_UNLOCKED");
605             }
606 
607             bugReporter.reportBug(bugInstance);
608 
609         }
610     }

618     private static boolean isConstructor(String methodName) {
619         return "<init>".equals(methodName) || "<clinit>".equals(methodName) || "readObject".equals(methodName)
620                 || "clone".equals(methodName) || "close".equals(methodName) || "writeObject".equals(methodName)
621                 || "toString".equals(methodName) || "init".equals(methodName) || "initialize".equals(methodName)
622                 || "dispose".equals(methodName) || "finalize".equals(methodName) || "this".equals(methodName)
623                 || "_jspInit".equals(methodName) || "_jspDestroy".equals(methodName);
624 
625     }

627     private void analyzeMethod(ClassContext classContext, Method method, Set<Method> lockedMethodSet) throws CFGBuilderException,
628     DataflowAnalysisException {
629 
630         InnerClassAccessMap icam = AnalysisContext.currentAnalysisContext().getInnerClassAccessMap();
631         ConstantPoolGen cpg = classContext.getConstantPoolGen();
632         MethodGen methodGen = classContext.getMethodGen(method);
633         if (methodGen == null) {
634             return;
635         }
636         CFG cfg = classContext.getCFG(method);
637         LockChecker lockChecker = classContext.getLockChecker(method);
638         ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
639         boolean isGetterMethod = isGetterMethod(classContext, method);
640         MethodDescriptor methodDescriptor = DescriptorFactory.instance().getMethodDescriptor(classContext.getJavaClass(), method);
641         if (DEBUG) {
642             System.out.println("**** Analyzing method " + SignatureConverter.convertMethodSignature(methodGen));
643         }
644 
645         for (Iterator<Location> i = cfg.locationIterator(); i.hasNext();) {
646             Location location = i.next();
647             try {
648                 Instruction ins = location.getHandle().getInstruction();
649                 XField xfield = null;
650                 boolean isWrite = false;
651                 boolean isLocal = false;
652                 boolean isNullCheck = false;
653 
654                 if (ins instanceof FieldInstruction) {
655                     InstructionHandle n = location.getHandle().getNext();
656                     isNullCheck = n.getInstruction() instanceof IFNONNULL || n.getInstruction() instanceof IFNULL;
657                     if (DEBUG && isNullCheck) {
658                         System.out.println("is null check");
659                     }
660                     FieldInstruction fins = (FieldInstruction) ins;
661                     xfield = Hierarchy.findXField(fins, cpg);
662                     if (xfield == null) {
663                         continue;
664                     }
665                     isWrite = ins.getOpcode() == Constants.PUTFIELD;
666                     isLocal = fins.getClassName(cpg).equals(classContext.getJavaClass().getClassName());
667                     if (DEBUG) {
668                         System.out.println("Handling field access: " + location.getHandle() + " (frame="
669                                 + vnaDataflow.getFactAtLocation(location) + ") :" + n);
670                     }
671                 } else if (ins instanceof INVOKESTATIC) {
672                     INVOKESTATIC inv = (INVOKESTATIC) ins;
673                     InnerClassAccess access = icam.getInnerClassAccess(inv, cpg);
674                     if (access != null && access.getMethodSignature().equals(inv.getSignature(cpg))) {
675                         xfield = access.getField();
676                         isWrite = !access.isLoad();
677                         isLocal = false;
678                         if (DEBUG) {
679                             System.out.println("Handling inner class access: " + location.getHandle() + " (frame="
680                                     + vnaDataflow.getFactAtLocation(location) + ")");
681                         }
682                     }
683                 }
684 
685                 if (xfield == null) {
686                     continue;
687                 }
688 
689                 // We only care about mutable nonvolatile nonpublic instance
690                 // fields.
691                 if (xfield.isStatic() || xfield.isPublic() || xfield.isVolatile() || xfield.isFinal()) {
692                     continue;
693                 }
694 
695                 // The value number frame could be invalid if the basic
696                 // block became unreachable due to edge pruning (dead code).
697                 ValueNumberFrame frame = vnaDataflow.getFactAtLocation(location);
698                 if (!frame.isValid()) {
699                     continue;
700                 }
701 
702                 // Get lock set and instance value
703                 ValueNumber thisValue = !method.isStatic() ? vnaDataflow.getAnalysis().getThisValue() : null;
704                 LockSet lockSet = lockChecker.getFactAtLocation(location);
705                 InstructionHandle handle = location.getHandle();
706                 ValueNumber instance = frame.getInstance(handle.getInstruction(), cpg);
707                 if (DEBUG) {
708                     System.out.println("Lock set: " + lockSet);
709                     System.out.println("value number: " + instance.getNumber());
710                     System.out.println("Lock count: " + lockSet.getLockCount(instance.getNumber()));
711                 }
712 
713                 // Is the instance locked?
714                 // We consider the access to be locked if either
715                 // - the object is explicitly locked, or
716                 // - the field is accessed through the "this" reference,
717                 // and the method is in the locked method set, or
718                 // - any value returned by a called method is locked;
719                 // the (conservative) assumption is that the return lock object
720                 // is correct for synchronizing the access
721                 boolean isExplicitlyLocked = lockSet.getLockCount(instance.getNumber()) > 0;
722                 boolean isAccessedThroughThis = thisValue != null && thisValue.equals(instance);
723                 boolean isLocked = isExplicitlyLocked
724                         || ((isConstructor(method.getName()) || lockedMethodSet.contains(method)) && isAccessedThroughThis)
725                         || lockSet.containsReturnValue(vnaDataflow.getAnalysis().getFactory());
726 
727                 // Adjust the field so its class name is the same
728                 // as the type of reference it is accessed through.
729                 // This helps fix false positives produced when a
730                 // threadsafe class is extended by a subclass that
731                 // doesn't care about thread safety.
732                 if (ADJUST_SUBCLASS_ACCESSES) {
733                     // Find the type of the object instance
734                     TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
735                     TypeFrame typeFrame = typeDataflow.getFactAtLocation(location);
736                     if (!typeFrame.isValid()) {
737                         continue;
738                     }
739                     Type instanceType = typeFrame.getInstance(handle.getInstruction(), cpg);
740                     if (instanceType instanceof TopType) {
741                         if (DEBUG) {
742                             System.out.println("Freaky: typeFrame is " + typeFrame);
743                         }
744                         continue;
745                     }
746                     // Note: instance type can be Null,
747                     // in which case we won't adjust the field type.
748                     if (instanceType != TypeFrame.getNullType() && instanceType != TypeFrame.getBottomType()) {
749                         if (!(instanceType instanceof ObjectType)) {
750                             throw new DataflowAnalysisException("Field accessed through non-object reference " + instanceType,
751                                     methodGen, handle);
752                         }
753                         ObjectType objType = (ObjectType) instanceType;
754 
755                         // If instance class name is not the same as that of the
756                         // field,
757                         // make it so
758                         String instanceClassName = objType.getClassName();
759                         if (!instanceClassName.equals(xfield.getClassName())) {
760                             xfield = XFactory.getExactXField(instanceClassName, xfield.getName(), xfield.getSignature(),
761                                     xfield.isStatic());
762                         }
763                     }
764                 }
765 
766                 int kind = 0;
767                 kind |= isLocked ? LOCKED : UNLOCKED;
768                 kind |= isWrite ? WRITE : isNullCheck ? NULLCHECK : READ;
769 
770                 // if (isLocked || !isConstructor(method.getName())) {
771                 if (DEBUG) {
772                     System.out.println("IS2:\t" + SignatureConverter.convertMethodSignature(methodGen) + "\t" + xfield + "\t"
773                             + ((isWrite ? "W" : "R") + "/" + (isLocked ? "L" : "U")));
774                 }
775 
776                 if (!isLocked && methodDescriptor.getClassDescriptor().isAnonymousClass()) {
777                     continue;
778                 }
779 
780                 FieldStats stats = getStats(xfield);
781 
782                 // Don't count a contructor's synchronized access
783                 // toward the field statistics because it's
784                 // trivially true and doesn't really represent the
785                 // programmer's intention
786                 if (!(isLocked && isConstructor(method.getName()))) {
787                     stats.addAccess(kind);
788                 }
789 
790                 if (isExplicitlyLocked && isLocal) {
791                     stats.addLocalLock();
792                 }
793 
794                 if (isGetterMethod && !isLocked) {
795                     stats.addGetterMethodAccess();
796                 }
797 
798                 stats.addAccess(methodDescriptor, handle, isLocked);
799                 // }
800             } catch (ClassNotFoundException e) {
801                 bugReporter.reportMissingClass(e);
802             }
803         }
804     }

806     /**
807      * Determine whether or not the the given method is a getter method. I.e.,
808      * if it just returns the value of an instance field.
809      *
810      * @param classContext
811      *            the ClassContext for the class containing the method
812      * @param method
813      *            the method
814      */
815     public static boolean isGetterMethod(ClassContext classContext, Method method) {
816         MethodGen methodGen = classContext.getMethodGen(method);
817         if (methodGen == null) {
818             return false;
819         }
820         InstructionList il = methodGen.getInstructionList();
821         // System.out.println("Checking getter method: " + method.getName());
822         if (il.getLength() > 60) {
823             return false;
824         }
825 
826         int count = 0;
827         Iterator<InstructionHandle> it = il.iterator();
828         while (it.hasNext()) {
829             InstructionHandle ih = it.next();
830             switch (ih.getInstruction().getOpcode()) {
831             case Constants.GETFIELD:
832                 count++;
833                 if (count > 1) {
834                     return false;
835                 }
836                 break;
837             case Constants.PUTFIELD:
838             case Constants.BALOAD:
839             case Constants.CALOAD:
840             case Constants.DALOAD:
841             case Constants.FALOAD:
842             case Constants.IALOAD:
843             case Constants.LALOAD:
844             case Constants.SALOAD:
845             case Constants.AALOAD:
846             case Constants.BASTORE:
847             case Constants.CASTORE:
848             case Constants.DASTORE:
849             case Constants.FASTORE:
850             case Constants.IASTORE:
851             case Constants.LASTORE:
852             case Constants.SASTORE:
853             case Constants.AASTORE:
854             case Constants.PUTSTATIC:
855                 return false;
856             case Constants.INVOKESTATIC:
857             case Constants.INVOKEVIRTUAL:
858             case Constants.INVOKEINTERFACE:
859             case Constants.INVOKESPECIAL:
860             case Constants.GETSTATIC:
861                 // no-op
862 
863             }
864         }
865         // System.out.println("Found getter method: " + method.getName());
866         return true;
867     }

872     private FieldStats getStats(XField field) {
873         FieldStats stats = statMap.get(field);
874         if (stats == null) {

881     /**
882      * Find methods that appear to never be called from an unlocked context We
883      * assume that nonpublic methods will only be called from within the class,
884      * which is not really a valid assumption.
885      */
886     private static Set<Method> findNotUnlockedMethods(ClassContext classContext, SelfCalls selfCalls, Set<CallSite> obviouslyLockedSites)
887     {
888 
889         JavaClass javaClass = classContext.getJavaClass();
890         Method[] methodList = javaClass.getMethods();
891 
892         CallGraph callGraph = selfCalls.getCallGraph();
893 
894         // Initially, assume no methods are called from an
895         // unlocked context
896         Set<Method> lockedMethodSet = new HashSet<Method>();
897         lockedMethodSet.addAll(Arrays.asList(methodList));
898 
899         // Assume all public methods are called from
900         // unlocked context
901         for (Method method : methodList) {
902             if (method.isPublic() && !isConstructor(method.getName())) {
903                 lockedMethodSet.remove(method);
904             }
905         }
906 
907         // Explore the self-call graph to find nonpublic methods
908         // that can be called from an unlocked context.
909         boolean change;
910         do {
911             change = false;
912 
913             for (Iterator<CallGraphEdge> i = callGraph.edgeIterator(); i.hasNext();) {
914                 CallGraphEdge edge = i.next();
915                 CallSite callSite = edge.getCallSite();
916 
917                 // Ignore obviously locked edges
918                 if (obviouslyLockedSites.contains(callSite)) {
919                     continue;
920                 }
921 
922                 // If the calling method is locked, ignore the edge
923                 if (lockedMethodSet.contains(callSite.getMethod())) {
924                     continue;
925                 }
926 
927                 // Calling method is unlocked, so the called method
928                 // is also unlocked.
929                 CallGraphNode target = edge.getTarget();
930                 if (lockedMethodSet.remove(target.getMethod())) {
931                     change = true;
932                 }
933             }
934         } while (change);
935 
936         if (DEBUG) {
937             System.out.println("Apparently not unlocked methods:");
938             for (Method method : lockedMethodSet) {
939                 System.out.println("\t" + method.getName());
940             }
941         }
942 
943         // We assume that any methods left in the locked set
944         // are called only from a locked context.
945         return lockedMethodSet;
946     }

948     /**
949      * Find methods that appear to always be called from a locked context. We
950      * assume that nonpublic methods will only be called from within the class,
951      * which is not really a valid assumption.
952      */
953     private static Set<Method> findLockedMethods(ClassContext classContext, SelfCalls selfCalls, Set<CallSite> obviouslyLockedSites)
954     {
955 
956         JavaClass javaClass = classContext.getJavaClass();
957         Method[] methodList = javaClass.getMethods();
958 
959         CallGraph callGraph = selfCalls.getCallGraph();
960 
961         // Initially, assume all methods are locked
962         Set<Method> lockedMethodSet = new HashSet<Method>();
963 
964         // Assume all public methods are unlocked
965         for (Method method : methodList) {
966             if (method.isSynchronized()) {
967                 lockedMethodSet.add(method);
968             }
969         }
970 
971         // Explore the self-call graph to find nonpublic methods
972         // that can be called from an unlocked context.
973         boolean change;
974         do {
975             change = false;
976 
977             for (Iterator<CallGraphEdge> i = callGraph.edgeIterator(); i.hasNext();) {
978                 CallGraphEdge edge = i.next();
979                 CallSite callSite = edge.getCallSite();
980 
981                 if (obviouslyLockedSites.contains(callSite) || lockedMethodSet.contains(callSite.getMethod())) {
982                     // Calling method is locked, so the called method
983                     // is also locked.
984                     CallGraphNode target = edge.getTarget();
985                     if (lockedMethodSet.add(target.getMethod())) {
986                         change = true;
987                     }
988                 }
989             }
990         } while (change);
991 
992         if (DEBUG) {
993             System.out.println("Apparently locked methods:");
994             for (Method method : lockedMethodSet) {
995                 System.out.println("\t" + method.getName());
996             }
997         }
998 
999         // We assume that any methods left in the locked set
1000         // are called only from a locked context.
1001         return lockedMethodSet;
1002     }

1047     /**
1048      * Find all self-call sites that are obviously locked.
1049      */
1050     private static Set<CallSite> findObviouslyLockedCallSites(ClassContext classContext, SelfCalls selfCalls)
1051             throws CFGBuilderException, DataflowAnalysisException {
1052         ConstantPoolGen cpg = classContext.getConstantPoolGen();
1053 
1054         // Find all obviously locked call sites
1055         Set<CallSite> obviouslyLockedSites = new HashSet<CallSite>();
1056         for (Iterator<CallSite> i = selfCalls.callSiteIterator(); i.hasNext();) {
1057             CallSite callSite = i.next();
1058             Method method = callSite.getMethod();
1059             Location location = callSite.getLocation();
1060             InstructionHandle handle = location.getHandle();
1061 
1062             // Only instance method calls qualify as candidates for
1063             // "obviously locked"
1064             Instruction ins = handle.getInstruction();
1065             if (ins.getOpcode() == Constants.INVOKESTATIC) {
1066                 continue;
1067             }
1068 
1069             // Get lock set for site
1070             LockChecker lockChecker = classContext.getLockChecker(method);
1071             LockSet lockSet = lockChecker.getFactAtLocation(location);
1072 
1073             // Get value number frame for site
1074             ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
1075             ValueNumberFrame frame = vnaDataflow.getFactAtLocation(location);
1076 
1077             // NOTE: if the CFG on which the value number analysis was performed
1078             // was pruned, there may be unreachable instructions. Therefore,
1079             // we can't assume the frame is valid.
1080             if (!frame.isValid()) {
1081                 continue;
1082             }
1083 
1084             // Find the ValueNumber of the receiver object
1085             int numConsumed = ins.consumeStack(cpg);
1086             MethodGen methodGen = classContext.getMethodGen(method);
1087             assert methodGen != null;
1088             if (numConsumed == Constants.UNPREDICTABLE) {
1089                 throw new DataflowAnalysisException("Unpredictable stack consumption", methodGen, handle);
1090             }
1091             // if (DEBUG) System.out.println("Getting receiver for frame: " +
1092             // frame);
1093             ValueNumber instance = frame.getStackValue(numConsumed - 1);
1094 
1095             // Is the instance locked?
1096             int lockCount = lockSet.getLockCount(instance.getNumber());
1097             if (lockCount > 0) {
1098                 // This is a locked call site
1099                 obviouslyLockedSites.add(callSite);
1100             }
1101         }
1102 
1103         return obviouslyLockedSites;
1104     }
