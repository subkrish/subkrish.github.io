182     @Override
183     public void visitClassContext(ClassContext classContext) {
184         this.classContext = classContext;
185 
186         String currentMethod = null;
187 
188         JavaClass jclass = classContext.getJavaClass();
189         String className = jclass.getClassName();
190         if (CLASS != null && !className.equals(CLASS)) {
191             return;
192         }
193 
194         List<Method> methodsInCallOrder = classContext.getMethodsInCallOrder();
195         for (Method method : methodsInCallOrder) {
196             try {
197                 if (method.isAbstract() || method.isNative() || method.getCode() == null) {
198                     continue;
199                 }
200 
201                 currentMethod = SignatureConverter.convertMethodSignature(jclass, method);
202 
203                 if (METHOD != null && !method.getName().equals(METHOD)) {
204                     continue;
205                 }
206                 if (DEBUG || DEBUG_NULLARG) {
207                     System.out.println("Checking for NP in " + currentMethod);
208                 }
209                 analyzeMethod(classContext, method);
210             } catch (MissingClassException e) {
211                 bugReporter.reportMissingClass(e.getClassNotFoundException());
212             } catch (DataflowAnalysisException e) {
213                 bugReporter.logError("While analyzing " + currentMethod + ": FindNullDeref caught dae exception", e);
214             } catch (CFGBuilderException e) {
215                 bugReporter.logError("While analyzing " + currentMethod + ": FindNullDeref caught cfgb exception", e);
216             }
217             bugAccumulator.reportAccumulatedBugs();
218         }
219     }

221     private void analyzeMethod(ClassContext classContext, Method method) throws DataflowAnalysisException, CFGBuilderException {
222         if (DEBUG || DEBUG_NULLARG) {
223             System.out.println("Pre FND ");
224         }
225 
226         if ((method.getAccessFlags() & Constants.ACC_BRIDGE) != 0) {
227             return;
228         }
229 
230         MethodGen methodGen = classContext.getMethodGen(method);
231 
232         if (methodGen == null) {
233             return;
234         }
235         if (!checkedDatabases) {
236             checkDatabases();
237             checkedDatabases = true;
238         }
239 
240         XMethod xMethod = XFactory.createXMethod(classContext.getJavaClass(), method);
241 
242         ClassDescriptor junitTestAnnotation = DescriptorFactory.createClassDescriptor("org/junit/Test");
243         AnnotationValue av = xMethod.getAnnotation(junitTestAnnotation);
244         if (av != null) {
245             Object value = av.getValue("expected");
246 
247             if (value instanceof Type) {
248                 String className = ((Type) value).getClassName();
249                 if ("java.lang.NullPointerException".equals(className)) {
250                     return;
251                 }
252             }
253         }
254 
255         // UsagesRequiringNonNullValues uses =
256         // classContext.getUsagesRequiringNonNullValues(method);
257         this.method = method;
258         this.methodAnnotation = getMethodNullnessAnnotation();
259 
260         if (DEBUG || DEBUG_NULLARG) {
261             System.out.println("FND: " + SignatureConverter.convertMethodSignature(methodGen));
262         }
263 
264         this.previouslyDeadBlocks = findPreviouslyDeadBlocks();
265 
266         // Get the IsNullValueDataflow for the method from the ClassContext
267         invDataflow = classContext.getIsNullValueDataflow(method);
268 
269         vnaDataflow = classContext.getValueNumberDataflow(method);
270 
271         // Create a NullDerefAndRedundantComparisonFinder object to do the
272         // actual
273         // work. It will call back to report null derefs and redundant null
274         // comparisons
275         // through the NullDerefAndRedundantComparisonCollector interface we
276         // implement.
277         NullDerefAndRedundantComparisonFinder worker = new NullDerefAndRedundantComparisonFinder(classContext, method, this);
278         worker.execute();
279 
280         checkCallSitesAndReturnInstructions();
281 
282     }

284     /**
285      * Find set of blocks which were known to be dead before doing the null
286      * pointer analysis.
287      *
288      * @return set of previously dead blocks, indexed by block id
289      * @throws CFGBuilderException
290      * @throws DataflowAnalysisException
291      */
292     private BitSet findPreviouslyDeadBlocks() throws DataflowAnalysisException, CFGBuilderException {
293         BitSet deadBlocks = new BitSet();
294         ValueNumberDataflow vnaDataflow = classContext.getValueNumberDataflow(method);
295         for (Iterator<BasicBlock> i = vnaDataflow.getCFG().blockIterator(); i.hasNext();) {
296             BasicBlock block = i.next();
297             ValueNumberFrame vnaFrame = vnaDataflow.getStartFact(block);
298             if (vnaFrame.isTop()) {
299                 deadBlocks.set(block.getLabel());
300             }
301         }
302 
303         return deadBlocks;
304     }

343         Profiler profiler = Global.getAnalysisCache().getProfiler();
344         profiler.start(CheckCallSitesAndReturnInstructions.class);
345         try {
346             ConstantPoolGen cpg = classContext.getConstantPoolGen();
347             TypeDataflow typeDataflow = classContext.getTypeDataflow(method);
348 
349             for (Iterator<Location> i = classContext.getCFG(method).locationIterator(); i.hasNext();) {
350                 Location location = i.next();
351                 Instruction ins = location.getHandle().getInstruction();
352                 try {
353                     ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getFactAtLocation(location);
354                     if (!vnaFrame.isValid()) {
355                         continue;
356                     }
357 
358                     if (ins instanceof InvokeInstruction) {
359                         examineCallSite(location, cpg, typeDataflow);
360                     } else if (methodAnnotation == NullnessAnnotation.NONNULL && ins.getOpcode() == Constants.ARETURN) {
361                         examineReturnInstruction(location);
362                     } else if (ins instanceof PUTFIELD) {
363                         examinePutfieldInstruction(location, (PUTFIELD) ins, cpg);
364                     }
365                 } catch (ClassNotFoundException e) {
366                     bugReporter.reportMissingClass(e);
367                 }
368             }
369         } catch (CheckedAnalysisException e) {
370             AnalysisContext.logError("error:", e);
371         } finally {
372             profiler.end(CheckCallSitesAndReturnInstructions.class);
373         }

487     private void examinePutfieldInstruction(Location location, PUTFIELD ins, ConstantPoolGen cpg)
488             throws DataflowAnalysisException {
489 
490         IsNullValueFrame frame = invDataflow.getFactAtLocation(location);
491         if (!frame.isValid()) {
492             return;
493         }
494         IsNullValue tos = frame.getTopValue();
495         if (tos.isDefinitelyNull()) {
496             XField field = XFactory.createXField(ins, cpg);
497             NullnessAnnotation annotation = AnalysisContext.currentAnalysisContext().getNullnessAnnotationDatabase()
498                     .getResolvedAnnotation(field, false);
499             if (annotation == NullnessAnnotation.NONNULL) {
500 
501                 BugAnnotation variableAnnotation = null;
502                 try {
503                     ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getFactAtLocation(location);
504                     ValueNumber valueNumber = vnaFrame.getTopValue();
505                     variableAnnotation = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, valueNumber,
506                             vnaFrame, "VALUE_OF");
507 
508                 } catch (DataflowAnalysisException e) {
509                     AnalysisContext.logError("error", e);
510                 } catch (CFGBuilderException e) {
511                     AnalysisContext.logError("error", e);
512                 }
513 
514                 BugInstance warning = new BugInstance(this, "NP_STORE_INTO_NONNULL_FIELD", tos.isDefinitelyNull() ? HIGH_PRIORITY
515                         : NORMAL_PRIORITY).addClassAndMethod(classContext.getJavaClass(), method).addField(field)
516                         .addOptionalAnnotation(variableAnnotation).addSourceLine(classContext, method, location);
517 
518                 bugReporter.reportBug(warning);
519             }
520         }

523     private void examineReturnInstruction(Location location) throws DataflowAnalysisException, CFGBuilderException {
524         if (DEBUG_NULLRETURN) {
525             System.out.println("Checking null return at " + location);
526         }
527 
528         IsNullValueDataflow invDataflow = classContext.getIsNullValueDataflow(method);
529         IsNullValueFrame frame = invDataflow.getFactAtLocation(location);
530         ValueNumberFrame vnaFrame = classContext.getValueNumberDataflow(method).getFactAtLocation(location);
531         if (!vnaFrame.isValid()) {
532             return;
533         }
534         ValueNumber valueNumber = vnaFrame.getTopValue();
535         if (!frame.isValid()) {
536             return;
537         }
538         IsNullValue tos = frame.getTopValue();
539         if (tos.isDefinitelyNull()) {
540             BugAnnotation variable = ValueNumberSourceInfo.findAnnotationFromValueNumber(method, location, valueNumber, vnaFrame,
541                     "VALUE_OF");
542 
543             String bugPattern = "NP_NONNULL_RETURN_VIOLATION";
544             int priority = NORMAL_PRIORITY;
545             if (tos.isDefinitelyNull() && !tos.isException()) {
546                 priority = HIGH_PRIORITY;
547             }
548             String methodName = method.getName();
549             if ("clone".equals(methodName)) {
550                 bugPattern = "NP_CLONE_COULD_RETURN_NULL";
551                 priority = NORMAL_PRIORITY;
552             } else if ("toString".equals(methodName)) {
553                 bugPattern = "NP_TOSTRING_COULD_RETURN_NULL";
554                 priority = NORMAL_PRIORITY;
555             }
556             BugInstance warning = new BugInstance(this, bugPattern, priority).addClassAndMethod(classContext.getJavaClass(),
557                     method).addOptionalAnnotation(variable);
558             bugAccumulator.accumulateBug(warning, SourceLineAnnotation.fromVisitedInstruction(classContext, method, location));
559         }

562     private boolean hasManyPreceedingNullTests(int pc) {
563         int ifNullTests = 0;
564         BitSet seen = new BitSet();
565         try {
566             for (Iterator<Location> i = classContext.getCFG(method).locationIterator(); i.hasNext();) {
567                 Location loc = i.next();
568                 int pc2 = loc.getHandle().getPosition();
569                 if (pc2 >= pc || pc2 < pc - 30) {
570                     continue;
571                 }
572                 Instruction ins = loc.getHandle().getInstruction();
573                 if ((ins instanceof IFNONNULL || ins instanceof IFNULL || ins instanceof NullnessConversationInstruction)
574                         && !seen.get(pc2)) {
575                     ifNullTests++;
576                     seen.set(pc2);
577                 }
578             }
579             boolean result = ifNullTests > 2;
580 
581             // System.out.println("Preceeding null tests " + ifNullTests + " " +
582             // ifNonnullTests + " " + result);
583             return result;
584         } catch (CFGBuilderException e) {
585             return false;
586         }

637     private void checkUnconditionallyDereferencedParam(Location location, ConstantPoolGen cpg, TypeDataflow typeDataflow,
638             InvokeInstruction invokeInstruction, BitSet nullArgSet, BitSet definitelyNullArgSet)
639                     throws DataflowAnalysisException, ClassNotFoundException {
640 
641         if (inExplictCatchNullBlock(location)) {
642             return;
643         }
644         boolean caught = inIndirectCatchNullBlock(location);
645         if (caught && skipIfInsideCatchNull()) {
646             return;
647         }
648 
649         // See what methods might be called here
650         XMethod calledMethod = XFactory.createXMethod(invokeInstruction, cpg);
651         if (true) {
652             // If a parameter is already marked as nonnull, don't complain about
653             // it here.
654             nullArgSet = (BitSet) nullArgSet.clone();
655             definitelyNullArgSet = (BitSet) definitelyNullArgSet.clone();
656             ClassDescriptor nonnullClassDesc = DescriptorFactory.createClassDescriptor(javax.annotation.Nonnull.class);
657             TypeQualifierValue<?> nonnullTypeQualifierValue = TypeQualifierValue.getValue(nonnullClassDesc, null);
658             for (int i = nullArgSet.nextSetBit(0); i >= 0; i = nullArgSet.nextSetBit(i + 1)) {
659                 TypeQualifierAnnotation tqa = TypeQualifierApplications.getEffectiveTypeQualifierAnnotation(calledMethod, i,
660                         nonnullTypeQualifierValue);
661                 if (tqa != null && tqa.when == When.ALWAYS) {
662                     nullArgSet.clear(i);
663                     definitelyNullArgSet.clear(i);
664                 }
665 
666             }
667         }
668         TypeFrame typeFrame = typeDataflow.getFactAtLocation(location);
669         Set<JavaClassAndMethod> targetMethodSet = Hierarchy.resolveMethodCallTargets(invokeInstruction, typeFrame, cpg);
670         if (DEBUG_NULLARG) {
671             System.out.println("Possibly called methods: " + targetMethodSet);
672         }
673 
674         // See if any call targets unconditionally dereference one of the null
675         // arguments
676         BitSet unconditionallyDereferencedNullArgSet = new BitSet();
677         List<JavaClassAndMethod> dangerousCallTargetList = new LinkedList<JavaClassAndMethod>();
678         List<JavaClassAndMethod> veryDangerousCallTargetList = new LinkedList<JavaClassAndMethod>();
679         for (JavaClassAndMethod targetMethod : targetMethodSet) {
680             if (DEBUG_NULLARG) {
681                 System.out.println("For target method " + targetMethod);
682             }
683 
684             ParameterProperty property = unconditionalDerefParamDatabase.getProperty(targetMethod.toMethodDescriptor());
685             if (property == null) {
686                 continue;
687             }
688             if (DEBUG_NULLARG) {
689                 System.out.println("\tUnconditionally dereferenced params: " + property);
690             }
691 
692             BitSet targetUnconditionallyDereferencedNullArgSet = property.getMatchingParameters(nullArgSet);
693 
694             if (targetUnconditionallyDereferencedNullArgSet.isEmpty()) {
695                 continue;
696             }
697 
698             dangerousCallTargetList.add(targetMethod);
699 
700             unconditionallyDereferencedNullArgSet.or(targetUnconditionallyDereferencedNullArgSet);
701 
702             if (!property.getMatchingParameters(definitelyNullArgSet).isEmpty()) {
703                 veryDangerousCallTargetList.add(targetMethod);
704             }
705         }
706 
707         if (dangerousCallTargetList.isEmpty()) {
708             return;
709         }
710 
711         WarningPropertySet<WarningProperty> propertySet = new WarningPropertySet<WarningProperty>();
712 
713         // See if there are any safe targets
714         Set<JavaClassAndMethod> safeCallTargetSet = new HashSet<JavaClassAndMethod>();
715         safeCallTargetSet.addAll(targetMethodSet);
716         safeCallTargetSet.removeAll(dangerousCallTargetList);
717         if (safeCallTargetSet.isEmpty()) {
718             propertySet.addProperty(NullArgumentWarningProperty.ALL_DANGEROUS_TARGETS);
719             if (dangerousCallTargetList.size() == 1) {
720                 propertySet.addProperty(NullArgumentWarningProperty.MONOMORPHIC_CALL_SITE);
721             }
722         }
723 
724         // Call to private method? In theory there should be only one possible
725         // target.
726         boolean privateCall = safeCallTargetSet.isEmpty() && dangerousCallTargetList.size() == 1
727                 && dangerousCallTargetList.get(0).getMethod().isPrivate();
728 
729         String bugType;
730         int priority;
731         if (privateCall || invokeInstruction.getOpcode() == Constants.INVOKESTATIC
732                 || invokeInstruction.getOpcode() == Constants.INVOKESPECIAL) {
733             bugType = "NP_NULL_PARAM_DEREF_NONVIRTUAL";
734             priority = HIGH_PRIORITY;
735         } else if (safeCallTargetSet.isEmpty()) {
736             bugType = "NP_NULL_PARAM_DEREF_ALL_TARGETS_DANGEROUS";
737             priority = NORMAL_PRIORITY;
738         } else {
739             return;
740         }
741 
742         if (caught) {
743             priority++;
744         }
745         if (dangerousCallTargetList.size() > veryDangerousCallTargetList.size()) {
746             priority++;
747         } else {
748             propertySet.addProperty(NullArgumentWarningProperty.ACTUAL_PARAMETER_GUARANTEED_NULL);
749         }
750         XMethod calledFrom = XFactory.createXMethod(classContext.getJavaClass(), method);
751 
752         if (safeCallToPrimateParseMethod(calledMethod, location)) {
753             return;
754         }
755         BugInstance warning = new BugInstance(this, bugType, priority).addClassAndMethod(classContext.getJavaClass(), method)
756                 .addMethod(calledMethod).describe(MethodAnnotation.METHOD_CALLED).addSourceLine(classContext, method, location);
757 
758         //        boolean uncallable = false;
759         if (!AnalysisContext.currentXFactory().isCalledDirectlyOrIndirectly(calledFrom) && calledFrom.isPrivate()) {
760 
761             propertySet.addProperty(GeneralWarningProperty.IN_UNCALLABLE_METHOD);
762             //            uncallable = true;
763         }
764         // Check which params might be null
765         addParamAnnotations(location, definitelyNullArgSet, unconditionallyDereferencedNullArgSet, propertySet, warning);
766 
767         if ("NP_NULL_PARAM_DEREF_ALL_TARGETS_DANGEROUS".equals(bugType)) {
768             // Add annotations for dangerous method call targets
769             for (JavaClassAndMethod dangerousCallTarget : veryDangerousCallTargetList) {
770                 warning.addMethod(dangerousCallTarget).describe(MethodAnnotation.METHOD_DANGEROUS_TARGET_ACTUAL_GUARANTEED_NULL);
771             }
772             dangerousCallTargetList.removeAll(veryDangerousCallTargetList);
773             if (DEBUG_NULLARG) {
774                 // Add annotations for dangerous method call targets
775                 for (JavaClassAndMethod dangerousCallTarget : dangerousCallTargetList) {
776                     warning.addMethod(dangerousCallTarget).describe(MethodAnnotation.METHOD_DANGEROUS_TARGET);
777                 }
778 
779                 // Add safe method call targets.
780                 // This is useful to see which other call targets the analysis
781                 // considered.
782                 for (JavaClassAndMethod safeMethod : safeCallTargetSet) {
783                     warning.addMethod(safeMethod).describe(MethodAnnotation.METHOD_SAFE_TARGET);
784                 }
785             }
786         }
787 
788         decorateWarning(location, propertySet, warning);
789         bugReporter.reportBug(warning);
790     }

1078         boolean isParameter = redundantBranch.firstValue.isParamValue();

1090             }

1092 
1093             if (DEBUG) {
1094                 System.out.println("Target block is  "
1095                         + (target.isExceptionThrower() ? " exception thrower" : " not exception thrower"));
1096             }
1097             // If the block is empty, it probably doesn't matter that it was
1098             // killed.
1099             // FIXME: really, we should crawl the immediately reachable blocks
1100             // starting at the target block to see if any of them are dead and
1101             // nonempty.
1102             boolean empty = !target.isExceptionThrower()
1103                     && (target.isEmpty() || isGoto(target.getFirstInstruction().getInstruction()));

1107                         if (DEBUG) {
1108                             System.out.println("Target of infeasible edge has multiple incoming edges");
1109                         }
1110                         empty = true;
1111                     }
1112                 } catch (CFGBuilderException e) {
1113                     assert true; // ignore it
1114                 }
1115             }
1116             if (DEBUG) {
1117                 System.out.println("Target block is  " + (empty ? "empty" : "not empty"));
1118             }
1119 
1120             if (!empty) {
1121                 if (isThrower(target)) {
1122                     infeasibleEdgeSimplyThrowsException = true;
1123                 }
1124 
1125             }
1126             if (!empty && !previouslyDeadBlocks.get(target.getLabel())) {
1127                 if (DEBUG) {
1128                     System.out.println("target was alive previously");
1129                 }
1130                 // Block was not dead before the null pointer analysis.
1131                 // See if it is dead now by inspecting the null value frame.
1132                 // If it's TOP, then the block became dead.
1133                 IsNullValueFrame invFrame = invDataflow.getStartFact(target);
1134                 createdDeadCode = invFrame.isTop();
1135                 if (DEBUG) {
1136                     System.out.println("target is now " + (createdDeadCode ? "dead" : "alive"));
1137                 }
1138 
1139             }
1140         }
1141 
1142         int priority;
1143         boolean valueIsNull = true;
1144         String warning;
1145         int pc = location.getHandle().getPosition();
1146         OpcodeStack stack = null;
1147         OpcodeStack.Item item1 = null;
1148 
1149         OpcodeStack.Item item2 = null;

1151             stack = OpcodeStackScanner.getStackAt(classContext.getJavaClass(), method, pc);
1152 
1153             item1 = stack.getStackItem(0);
1154         } catch (RuntimeException e) {
1155             if (SystemProperties.ASSERTIONS_ENABLED) {
1156                 AnalysisContext.logError("Error getting stack at specific PC", e);
1157             }
1158             assert true;
1159         }

1161             if (redundantBranch.firstValue.isDefinitelyNull()) {
1162                 warning = "RCN_REDUNDANT_NULLCHECK_OF_NULL_VALUE";
1163                 priority = NORMAL_PRIORITY;
1164             } else {

1166                 valueIsNull = false;
1167                 priority = isChecked ? HIGH_PRIORITY : NORMAL_PRIORITY;
1168             }
1169             if (infeasibleEdgeSimplyThrowsException) {
1170                 priority++;
1171             }
1172 
1173         } else {
1174             if (stack != null) {
1175                 item2 = stack.getStackItem(1);
1176             }
1177             boolean bothNull = redundantBranch.firstValue.isDefinitelyNull() && redundantBranch.secondValue.isDefinitelyNull();
1178             if (redundantBranch.secondValue.isChecked()) {
1179                 isChecked = true;
1180             }

1182                 wouldHaveBeenAKaboom = true;
1183                 locationOfKaBoom = redundantBranch.secondValue.getLocationOfKaBoom();
1184             }
1185             if (bothNull) {
1186                 warning = "RCN_REDUNDANT_COMPARISON_TWO_NULL_VALUES";
1187                 priority = NORMAL_PRIORITY;
1188             } else {
1189                 warning = "RCN_REDUNDANT_COMPARISON_OF_NULL_AND_NONNULL_VALUE";
1190                 priority = isChecked ? NORMAL_PRIORITY : LOW_PRIORITY;
1191             }
1192 
1193         }
1194

1405             Set<Location> assignedNullLocationSet, @Nonnull
1406             Set<Location> derefLocationSet, SortedSet<Location> doomedLocations, ValueNumberDataflow vna, ValueNumber refValue,
1407             @CheckForNull
1408             BugAnnotation variableAnnotation, NullValueUnconditionalDeref deref, boolean npeIfStatementCovered) {

1410             return;

1414             System.out.println("Found guaranteed null deref in " + method.getName());
1415             for (Location loc : doomedLocations) {
1416                 System.out.println("Doomed at " + loc);

1420         String bugType;
1421 
1422         int priority = npeIfStatementCovered ? HIGH_PRIORITY : NORMAL_PRIORITY;
1423 
1424         if (deref.isMethodReturnValue()) {
1425             if (deref.isReadlineValue()) {
1426                 bugType = "NP_DEREFERENCE_OF_READLINE_VALUE";
1427             } else {
1428                 bugType = "NP_NULL_ON_SOME_PATH_FROM_RETURN_VALUE";
1429             }
1430         } else if (derefLocationSet.size() > 1) {
1431             if (!deref.isAlwaysOnExceptionPath()) {
1432                 bugType = "NP_GUARANTEED_DEREF";
1433             } else {
1434                 bugType = "NP_GUARANTEED_DEREF_ON_EXCEPTION_PATH";
1435             }
1436         } else if (!deref.isAlwaysOnExceptionPath()) {
1437             bugType = "NP_NULL_ON_SOME_PATH";
1438         } else {
1439             bugType = "NP_NULL_ON_SOME_PATH_EXCEPTION";
1440         }
1441 
1442         boolean allCallToAssertionMethod = !doomedLocations.isEmpty();
1443         for (Location loc : doomedLocations) {
1444             if (!callToAssertionMethod(loc)) {
1445                 allCallToAssertionMethod = false;
1446             }
1447         }
1448         if (allCallToAssertionMethod) {
1449             return;
1450         }
1451 
1452         // Add Locations in the set of locations at least one of which
1453         // is guaranteed to be dereferenced
1454 
1455         SortedSet<Location> sourceLocations;

1457             sourceLocations = new TreeSet<Location>(assignedNullLocationSet);
1458         } else {
1459             sourceLocations = doomedLocations;

1463             return;

1467

1472         int distance2 = minDereferencePC - maxPC(doomedLocations);

1474         /*
1475         if (false) {
1476             System.out.printf("%9d %9d %9d RANGE %s.%s%s\n", distance, distance1, distance2, classContext.getClassDescriptor()
1477                     .toDottedClassName(), method.getName(), method.getSignature());

1479          */

1482 
1483         BitSet knownNull = new BitSet();
1484 
1485         SortedSet<SourceLineAnnotation> knownNullLocations = new TreeSet<SourceLineAnnotation>();
1486         for (Location loc : sourceLocations) {
1487             SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, method, loc);
1488             if (sourceLineAnnotation == null) {
1489                 continue;
1490             }
1491             int startLine = sourceLineAnnotation.getStartLine();
1492             if (startLine == -1) {
1493                 knownNullLocations.add(sourceLineAnnotation);
1494             } else if (!knownNull.get(startLine)) {
1495                 knownNull.set(startLine);
1496                 knownNullLocations.add(sourceLineAnnotation);

1501         MethodAnnotation invokedMethod = null;
1502         XMethod invokedXMethod = null;
1503         int parameterNumber = -1;
1504         if (derefLocationSet.size() == 1) {
1505 
1506             Location loc = derefLocationSet.iterator().next();
1507 
1508             PointerUsageRequiringNonNullValue pu = null;
1509             try {
1510                 UsagesRequiringNonNullValues usages = classContext.getUsagesRequiringNonNullValues(method);

1514             } catch (CFGBuilderException e) {
1515                 AnalysisContext.logError("Error getting UsagesRequiringNonNullValues for " + method, e);
1516             }
1517 
1518             if (pu == null) {

1521                 bugType = "NP_DEREFERENCE_OF_READLINE_VALUE";
1522                 priority = NORMAL_PRIORITY;
1523             } else if (deref.isMethodReturnValue() && !deref.isReadlineValue()) {
1524                 bugType = "NP_NULL_ON_SOME_PATH_FROM_RETURN_VALUE";
1525                 priority = NORMAL_PRIORITY;
1526             } else if (pu.isReturnFromNonNullMethod()) {

1534                     bugType = "NP_TOSTRING_COULD_RETURN_NULL";
1535                     priority = NORMAL_PRIORITY;
1536                 }
1537 
1538             } else {
1539                 XField f = pu.getNonNullField();
1540                 if (f != null) {
1541                     storedField = FieldAnnotation.fromXField(f);
1542                     bugType = "NP_STORE_INTO_NONNULL_FIELD";

1546                         invokedXMethod = mp.getMethod();
1547                         for (Location derefLoc : derefLocationSet) {
1548                             if (safeCallToPrimateParseMethod(invokedXMethod, derefLoc)) {

1551                         }

1553                         if (mp.getParameterNumber() == 0

1555                             return;

1559                     }

1561             }
1562         }

1565         for (SourceLineAnnotation sourceLineAnnotation : knownNullLocations) {
1566             if (!hasManyPreceedingNullTests(sourceLineAnnotation.getStartBytecode())) {

1568             }
1569         }

1580             assert invokedXMethod != null;

1582             if (i != invokedXMethod) {
1583                 bugInstance.addMethod(i).describe(MethodAnnotation.METHOD_CALLED);

1586                 .addParameterAnnotation(parameterNumber, "INT_MAYBE_NULL_ARG");
1587             }
1588         }
1589         if (storedField != null) {
1590             bugInstance.addField(storedField).describe("FIELD_STORED");
1591         }
1592         bugInstance.addOptionalAnnotation(variableAnnotation);
1593         if (variableAnnotation instanceof FieldAnnotation) {
1594             bugInstance.describe("FIELD_CONTAINS_VALUE");
1595         }
1596 
1597         addPropertiesForDereferenceLocations(propertySet, derefLocationSet, false);
1598 
1599         if (deref.isAlwaysOnExceptionPath()) {
1600             propertySet.addProperty(NullDerefProperty.ALWAYS_ON_EXCEPTION_PATH);
1601         }
1602 
1603         if (!assignedNullLocationSet.isEmpty() && distance > 100) {
1604             propertySet.addProperty(NullDerefProperty.LONG_RANGE_NULL_SOURCE);
1605         }
1606 
1607         propertySet.decorateBugInstance(bugInstance);

1610 
1611             int source = -9999;

1613                 source = knownNullLocations.iterator().next().getEndBytecode();
1614             }
1615             for (Location loc : derefLocationSet) {
1616                 int pos = loc.getHandle().getPosition();
1617                 if (pos != source + 3) {

1619                     bugAccumulator.accumulateBug(bugInstance,
1620                             SourceLineAnnotation.fromVisitedInstruction(classContext, method, loc));
1621                 }
1622             }
1623 
1624         } else {
1625             for (Location loc : derefLocationSet) {
1626                 bugInstance.addSourceLine(classContext, method, loc).describe(getDescription(loc, refValue));
1627             }
1628 
1629             if (sourceLocations == doomedLocations && assignedNullLocationSet.size() == 1) {
1630                 Location assignedNull = assignedNullLocationSet.iterator().next();
1631                 SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(classContext, method,
1632                         assignedNull);
1633                 if (sourceLineAnnotation != null) {
1634                     int startLine = sourceLineAnnotation.getStartLine();
1635                     if (startLine > 0 && !knownNull.get(startLine)) {
1636                         bugInstance.add(sourceLineAnnotation).describe("SOURCE_LINE_NULL_VALUE");

1639
