73      public OverridingEqualsNotSymmetrical(BugReporter bugReporter) {
74          this.bugReporter = bugReporter;
75          this.bugAccumulator = new BugAccumulator(bugReporter);
76          equalsKindSummary = AnalysisContext.currentAnalysisContext().getEqualsKindSummary();
77      }

79      @Override
80      public void visit(Code obj) {
81          if (EQUALS_NAME.equals(getMethodName()) && !getMethod().isStatic() && getMethod().isPublic()
82                  && EQUALS_SIGNATURE.equals(getMethodSig())) {
83              sawCheckedCast = sawSuperEquals = sawInstanceOf = sawGetClass = sawReturnSuper = sawCompare = sawReturnNonSuper = prevWasSuperEquals = sawGoodEqualsClass = sawBadEqualsClass = dangerDanger = sawInstanceOfSupertype = alwaysTrue = alwaysFalse = sawStaticDelegate = sawEqualsBuilder = false;
84              sawInitialIdentityCheck = obj.getCode().length == 11 || obj.getCode().length == 9;
85              equalsCalls = 0;
86              super.visit(obj);
87              EqualsKindSummary.KindOfEquals kind = EqualsKindSummary.KindOfEquals.UNKNOWN;
88              if (alwaysTrue) {
89                  kind = EqualsKindSummary.KindOfEquals.ALWAYS_TRUE;
90              } else if (alwaysFalse) {
91                  kind = EqualsKindSummary.KindOfEquals.ALWAYS_FALSE;
92              } else if (sawReturnSuper && !sawReturnNonSuper) {
93                  kind = EqualsKindSummary.KindOfEquals.RETURNS_SUPER;
94              } else if (sawSuperEquals) {
95                  kind = EqualsKindSummary.KindOfEquals.INVOKES_SUPER;
96              } else if (sawInstanceOfSupertype) {
97                  kind = EqualsKindSummary.KindOfEquals.INSTANCE_OF_SUPERCLASS_EQUALS;
98              } else if (sawInstanceOf) {
99                  kind = getThisClass().isAbstract() ? EqualsKindSummary.KindOfEquals.ABSTRACT_INSTANCE_OF
100                         : EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS;
101             } else if (sawGetClass && sawGoodEqualsClass) {
102                 kind = getThisClass().isAbstract() ? EqualsKindSummary.KindOfEquals.ABSTRACT_GETCLASS_GOOD_EQUALS
103                         : EqualsKindSummary.KindOfEquals.GETCLASS_GOOD_EQUALS;
104             } else if (sawGetClass && sawBadEqualsClass) {
105                 kind = EqualsKindSummary.KindOfEquals.GETCLASS_BAD_EQUALS;
106             } else if (equalsCalls == 1 || sawStaticDelegate || sawEqualsBuilder) {
107                 kind = EqualsKindSummary.KindOfEquals.DELEGATE_EQUALS;
108             } else if (sawInitialIdentityCheck) {
109                 kind = EqualsKindSummary.KindOfEquals.TRIVIAL_EQUALS;
110             } else if (sawCheckedCast) {
111                 kind = EqualsKindSummary.KindOfEquals.CHECKED_CAST_EQUALS;
112             } else if (sawCompare) {
113                 kind = EqualsKindSummary.KindOfEquals.COMPARE_EQUALS;
114             } else {
115                 if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
116                     bugReporter
117                     .reportBug(new BugInstance(this, "EQ_UNUSUAL", Priorities.NORMAL_PRIORITY).addClassAndMethod(this));
118                 }
119             }
120             ClassAnnotation classAnnotation = new ClassAnnotation(getDottedClassName());
121             equalsKindSummary.put(classAnnotation, kind);
122 
123             count(kind);
124             if (kind == EqualsKindSummary.KindOfEquals.GETCLASS_GOOD_EQUALS
125                     || kind == EqualsKindSummary.KindOfEquals.ABSTRACT_GETCLASS_GOOD_EQUALS
126                     || kind == EqualsKindSummary.KindOfEquals.GETCLASS_BAD_EQUALS) {
127 
128                 ClassDescriptor classDescriptor = getClassDescriptor();
129                 try {
130                     Set<ClassDescriptor> subtypes = AnalysisContext.currentAnalysisContext().getSubtypes2()
131                             .getSubtypes(classDescriptor);
132                     if (subtypes.size() > 1) {
133                         classesWithGetClassBasedEquals.put(classDescriptor, subtypes);
134                     }
135                 } catch (ClassNotFoundException e) {
136                     assert true;
137                 }
138 
139             }
140             if (kind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS
141                     || kind == EqualsKindSummary.KindOfEquals.ABSTRACT_INSTANCE_OF) {
142 
143                 ClassDescriptor classDescriptor = getClassDescriptor();
144                 try {
145                     Set<ClassDescriptor> subtypes = AnalysisContext.currentAnalysisContext().getSubtypes2()
146                             .getSubtypes(classDescriptor);
147                     if (subtypes.size() > 1) {
148                         classesWithInstanceOfBasedEquals.put(classDescriptor, subtypes);
149                     }
150                 } catch (ClassNotFoundException e) {
151                     assert true;
152                 }
153 
154             }
155 
156             String superClassName = getSuperclassName().replace('/', '.');
157             if (!"java.lang.Object".equals(superClassName)) {
158                 parentMap.put(classAnnotation, new ClassAnnotation(superClassName));
159             }
160             equalsMethod.put(classAnnotation, getMethodDescriptor());
161 
162         }
163         bugAccumulator.reportAccumulatedBugs();
164     }

194     private final EnumMap<EqualsKindSummary.KindOfEquals, Integer> count = new EnumMap<EqualsKindSummary.KindOfEquals, Integer>(
195             EqualsKindSummary.KindOfEquals.class);
196 
197     private void count(EqualsKindSummary.KindOfEquals k) {
198         Integer v = count.get(k);
199         if (v == null) {
200             count.put(k, 1);
201         } else {
202             count.put(k, v + 1);
203         }
204     }

206     @Override
207     public void sawOpcode(int seen) {
208         if (getPC() == 2 && seen != IF_ACMPEQ && seen != IF_ACMPNE) {
209             // System.out.println(OPCODE_NAMES[seen]);
210             sawInitialIdentityCheck = false;
211         }
212         if (getPC() == 2
213                 && seen == INVOKESTATIC
214                 && getCode().getCode().length == 6
215                 && (getPrevOpcode(1) == ALOAD_0 && getPrevOpcode(2) == ALOAD_1 || getPrevOpcode(1) == ALOAD_1
216                 && getPrevOpcode(2) == ALOAD_0)) {
217             sawStaticDelegate = true;
218         }
219 
220         if ((seen == INVOKESTATIC || seen == INVOKESPECIAL || seen == INVOKEVIRTUAL)
221                 && ("org/apache/commons/lang/builder/EqualsBuilder".equals(getClassConstantOperand())
222                         || "org/apache/commons/lang3/builder/EqualsBuilder".equals(getClassConstantOperand()))) {
223             sawEqualsBuilder = true;
224         }
225 
226         if (seen == IRETURN && getPC() == 1 && getPrevOpcode(1) == ICONST_0) {
227             alwaysFalse = true;
228             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
229                 bugReporter.reportBug(new BugInstance(this, "EQ_ALWAYS_FALSE", Priorities.HIGH_PRIORITY).addClassAndMethod(this)
230                         .addSourceLine(this));
231             }
232 
233         }
234         if (seen == IRETURN && getPC() == 1 && getPrevOpcode(1) == ICONST_1) {
235             alwaysTrue = true;
236             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
237                 bugReporter.reportBug(new BugInstance(this, "EQ_ALWAYS_TRUE", Priorities.HIGH_PRIORITY).addClassAndMethod(this)
238                         .addSourceLine(this));
239             }
240 
241         }
242         if (seen == IF_ACMPEQ || seen == IF_ACMPNE) {
243             checkForComparingClasses();
244         }
245         if (callToInvoke(seen)) {
246             equalsCalls++;
247             checkForComparingClasses();
248             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass()) && dangerDanger) {
249                 bugReporter.reportBug(new BugInstance(this, "EQ_COMPARING_CLASS_NAMES", Priorities.NORMAL_PRIORITY)
250                 .addClassAndMethod(this).addSourceLine(this));
251             }
252         }
253 
254         if ((seen == INVOKEINTERFACE || seen == INVOKEVIRTUAL) && "compare".equals(getNameConstantOperand())
255                 && stack.getStackDepth() >= 2) {
256             Item left = stack.getStackItem(1);
257             Item right = stack.getStackItem(0);
258             if (left.getRegisterNumber() + right.getRegisterNumber() == 1) {
259                 sawCompare = true;
260             }
261         }
262         dangerDanger = false;
263 
264         if (seen == INVOKEVIRTUAL && "java/lang/Class".equals(getClassConstantOperand())
265                 && "getName".equals(getNameConstantOperand()) && "()Ljava/lang/String;".equals(getSigConstantOperand())
266                 && stack.getStackDepth() >= 2) {
267             Item left = stack.getStackItem(1);
268             XMethod leftM = left.getReturnValueOf();
269             Item right = stack.getStackItem(0);
270             XMethod rightM = right.getReturnValueOf();
271             if (leftM != null && rightM != null && "getName".equals(leftM.getName()) && "getClass".equals(rightM.getName())) {
272                 dangerDanger = true;
273             }
274 
275         }
276         if (seen == INVOKESPECIAL && EQUALS_NAME.equals(getNameConstantOperand())
277                 && EQUALS_SIGNATURE.equals(getSigConstantOperand())) {
278             sawSuperEquals = prevWasSuperEquals = true;
279         } else {
280             if (seen == IRETURN) {
281                 if (prevWasSuperEquals) {
282                     sawReturnSuper = true;
283                 } else {
284                     sawReturnNonSuper = true;
285                 }
286             }
287             prevWasSuperEquals = false;
288         }
289 
290         if (seen == INSTANCEOF && stack.getStackDepth() > 0 && stack.getStackItem(0).getRegisterNumber() == 1) {
291             ClassDescriptor instanceOfCheck = getClassDescriptorOperand();
292             if (instanceOfCheck.equals(getClassDescriptor())) {
293                 sawInstanceOf = true;
294             } else {
295                 try {
296                     if (AnalysisContext.currentAnalysisContext().getSubtypes2().isSubtype(getClassDescriptor(), instanceOfCheck)) {
297                         sawInstanceOfSupertype = true;
298                     }
299                 } catch (ClassNotFoundException e) {
300                     sawInstanceOfSupertype = true;
301                 }
302             }
303         }
304 
305         if (seen == CHECKCAST && stack.getStackDepth() > 0 && stack.getStackItem(0).getRegisterNumber() == 1) {
306             ClassDescriptor castTo = getClassDescriptorOperand();
307             if (castTo.equals(getClassDescriptor())) {
308                 sawCheckedCast = true;
309             }
310             try {
311                 if (AnalysisContext.currentAnalysisContext().getSubtypes2().isSubtype(getClassDescriptor(), castTo)) {
312                     sawCheckedCast = true;
313                 }
314             } catch (ClassNotFoundException e) {
315                 sawCheckedCast = true;
316             }
317         }
318         if (seen == INVOKEVIRTUAL && "getClass".equals(getNameConstantOperand())
319                 && "()Ljava/lang/Class;".equals(getSigConstantOperand())) {
320             sawGetClass = true;
321         }
322 
323     }

325     private boolean callToInvoke(int seen) {
326         if (seen == INVOKEVIRTUAL || seen == INVOKEINTERFACE || seen == INVOKESPECIAL) {
327             return invokesMethodWithEqualLikeName() && EQUALS_SIGNATURE.equals(getSigConstantOperand());
328         }
329         if (seen == INVOKESTATIC) {
330             String sig = getSigConstantOperand();
331             return invokesMethodWithEqualLikeName() && sig.endsWith("Ljava/lang/Object;)Z");
332         }
333 
334         return false;
335 
336     }
337 
338     public boolean invokesMethodWithEqualLikeName() {
339         return getNameConstantOperand().toLowerCase().indexOf(EQUALS_NAME) >= 0;
340     }

345     private void checkForComparingClasses() {
346         if (stack.getStackDepth() >= 2) {
347             Item left = stack.getStackItem(1);
348             XMethod leftM = left.getReturnValueOf();
349             Item right = stack.getStackItem(0);
350             XMethod rightM = right.getReturnValueOf();
351             if ("Ljava/lang/Class;".equals(left.getSignature()) && "Ljava/lang/Class;".equals(right.getSignature())) {
352                 boolean leftMatch = leftM != null && "getClass".equals(leftM.getName());
353                 boolean rightMatch = rightM != null && "getClass".equals(rightM.getName());
354                 if (leftMatch && rightMatch) {
355                     sawGoodEqualsClass = true;
356                 } else {
357                     if (getClassName().equals(left.getConstant()) && rightMatch || leftMatch
358                             && getClassName().equals(right.getConstant())) {
359                         if (getThisClass().isFinal()) {
360                             sawGoodEqualsClass = true;
361                         } else {
362                             sawBadEqualsClass = true;
363                             if (AnalysisContext.currentAnalysisContext().isApplicationClass(getThisClass())) {
364 
365                                 int priority = Priorities.NORMAL_PRIORITY;
366 
367                                 BugInstance bug = new BugInstance(this, "EQ_GETCLASS_AND_CLASS_CONSTANT", priority)
368                                 .addClassAndMethod(this);
369 
370                                 try {
371 
372                                     Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
373                                     Set<ClassDescriptor> subtypes = subtypes2.getDirectSubtypes(getClassDescriptor());
374                                     for (ClassDescriptor c : subtypes) {
375                                         try {
376                                             Global.getAnalysisCache().getClassAnalysis(XClass.class, c);
377                                         } catch (CheckedAnalysisException e) {
378                                             continue;
379                                         }
380                                         XMethod m = Hierarchy2.findMethod(c, "equals", "(Ljava/lang/Object;)Z", false);
381                                         if (m == null) {
382                                             bug.addClass(c).describe(ClassAnnotation.SUBCLASS_ROLE);
383                                             priority--;
384                                             bug.setPriority(priority);
385                                         }
386                                     }
387 
388                                 } catch (ClassNotFoundException e) {
389                                     bugReporter.reportMissingClass(e);
390                                 }
391                                 bugAccumulator.accumulateBug(bug, this);
392                             }
393                         }
394                     }
395                 }
396             }
397 
398         }
399     }

401     @Override
402     public void report() {
403 
404         if (false) {
405             Subtypes2 subtypes2 = AnalysisContext.currentAnalysisContext().getSubtypes2();
406             for (Map.Entry<ClassDescriptor, Set<ClassDescriptor>> e : classesWithGetClassBasedEquals.entrySet()) {
407                 ClassAnnotation parentClass = ClassAnnotation.fromClassDescriptor(e.getKey());
408                 XClass xParent = AnalysisContext.currentXFactory().getXClass(e.getKey());
409                 if (xParent == null) {
410                     continue;
411                 }
412                 EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
413                 for (ClassDescriptor child : e.getValue()) {
414                     if (child.equals(e.getKey())) {
415                         continue;
416                     }
417                     XClass xChild = AnalysisContext.currentXFactory().getXClass(child);
418                     if (xChild == null) {
419                         continue;
420                     }
421                     ClassAnnotation childClass = ClassAnnotation.fromClassDescriptor(child);
422                     EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
423                     int fieldsOfInterest = 0;
424                     for (XField f : xChild.getXFields()) {
425                         if (!f.isStatic() && !f.isSynthetic()) {
426                             fieldsOfInterest++;
427                         }
428                     }
429                     int grandchildren = -1;
430                     try {
431 
432                         grandchildren = subtypes2.getSubtypes(child).size();
433                     } catch (ClassNotFoundException e1) {
434                         assert true;
435                     }
436                     System.out.println(parentKind + " " + childKind + " " + parentClass + " " + childClass + " "
437                             + fieldsOfInterest + " " + grandchildren);
438                     try {
439                         if (grandchildren >= 2) {
440                             for (ClassDescriptor g : subtypes2.getSubtypes(child)) {
441                                 if (!g.equals(child)) {
442                                     System.out.println("  " + g);
443                                 }
444                             }
445                         }
446                     } catch (ClassNotFoundException e1) {
447                         assert true;
448                     }
449 
450                 }
451 
452             }
453             int overridden = 0, total = 0;
454             for (Map.Entry<ClassDescriptor, Set<ClassDescriptor>> e : classesWithInstanceOfBasedEquals.entrySet()) {
455                 ClassAnnotation parentClass = ClassAnnotation.fromClassDescriptor(e.getKey());
456                 XClass xParent = AnalysisContext.currentXFactory().getXClass(e.getKey());
457                 if (xParent == null) {
458                     continue;
459                 }
460                 EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
461                 boolean isOverridden = false;
462                 for (ClassDescriptor child : e.getValue()) {
463                     if (child.equals(e.getKey())) {
464                         continue;
465                     }
466                     XClass xChild = AnalysisContext.currentXFactory().getXClass(child);
467                     if (xChild == null) {
468                         continue;
469                     }
470                     ClassAnnotation childClass = ClassAnnotation.fromClassDescriptor(child);
471                     EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
472                     if (childKind != null) {
473                         isOverridden = true;
474                     }
475                 }
476                 total++;
477                 if (isOverridden) {
478                     overridden++;
479                 }
480                 System.out.println("IS_OVERRIDDEN: " + e.getKey().getClassName());
481             }
482             System.out.println("Instance of equals: " + total + " subclassed, " + overridden + " overrridden");
483             for (Map.Entry<EqualsKindSummary.KindOfEquals, Integer> e : count.entrySet()) {
484                 System.out.println(e);
485             }
486 
487         }
488 
489         for (Map.Entry<ClassAnnotation, ClassAnnotation> e : parentMap.entrySet()) {
490             ClassAnnotation childClass = e.getKey();
491             EqualsKindSummary.KindOfEquals childKind = equalsKindSummary.get(childClass);
492             ClassAnnotation parentClass = e.getValue();
493             EqualsKindSummary.KindOfEquals parentKind = equalsKindSummary.get(parentClass);
494 
495             if (childKind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS
496                     && parentKind == EqualsKindSummary.KindOfEquals.INSTANCE_OF_EQUALS) {
497                 bugReporter.reportBug(new BugInstance(this, "EQ_OVERRIDING_EQUALS_NOT_SYMMETRIC", NORMAL_PRIORITY)
498                 .add(childClass).addMethod(equalsMethod.get(childClass)).addMethod(equalsMethod.get(parentClass))
499                 .describe(MethodAnnotation.METHOD_OVERRIDDEN));
500             }
501 
502         }
