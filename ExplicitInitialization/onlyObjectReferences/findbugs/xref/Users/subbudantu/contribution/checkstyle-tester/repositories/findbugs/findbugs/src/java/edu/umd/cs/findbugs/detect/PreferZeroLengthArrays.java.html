20  package edu.umd.cs.findbugs.detect;
21  
22  import java.util.Collection;
23  import java.util.LinkedList;
24  
25  import org.apache.bcel.classfile.Code;
26  
27  import edu.umd.cs.findbugs.BugInstance;
28  import edu.umd.cs.findbugs.BugReporter;
29  import edu.umd.cs.findbugs.BytecodeScanningDetector;
30  import edu.umd.cs.findbugs.SourceLineAnnotation;
31  import edu.umd.cs.findbugs.StatelessDetector;
32  
33  public class PreferZeroLengthArrays extends BytecodeScanningDetector implements StatelessDetector {
34      boolean nullOnTOS = false;
35  
36      private final BugReporter bugReporter;
37  
38      public PreferZeroLengthArrays(BugReporter bugReporter) {
39          this.bugReporter = bugReporter;
40      }
41  
42      Collection<SourceLineAnnotation> found = new LinkedList<SourceLineAnnotation>();
43  
44      @Override
45      public void visit(Code obj) {
46          found.clear();
47          // Solution to sourceforge bug 1765925; returning null is the
48          // convention used by java.io.File.listFiles()
49          if ("listFiles".equals(getMethodName())) {
50              return;
51          }
52          String returnType = getMethodSig().substring(getMethodSig().indexOf(')') + 1);
53          if (returnType.startsWith("[")) {
54              nullOnTOS = false;
55              super.visit(obj);
56              if (!found.isEmpty()) {
57                  BugInstance bug = new BugInstance(this, "PZLA_PREFER_ZERO_LENGTH_ARRAYS", LOW_PRIORITY).addClassAndMethod(this);
58                  for (SourceLineAnnotation s : found) {
59                      bug.add(s);
60                  }
61                  bugReporter.reportBug(bug);
62                  found.clear();
63              }
64          }
65      }
66  
67      @Override
68      public void sawOpcode(int seen) {
69  
70          switch (seen) {
71          case ACONST_NULL:
72              nullOnTOS = true;
73              return;
74          case ARETURN:
75              if (nullOnTOS) {
76                  SourceLineAnnotation sourceLineAnnotation = SourceLineAnnotation.fromVisitedInstruction(getClassContext(), this,
77                          getPC());
78                  if (sourceLineAnnotation != null) {
79                      found.add(sourceLineAnnotation);
80                  }
81              }
82              break;
83          default:
84              break;
85          }
86          nullOnTOS = false;
87      }
88  }
