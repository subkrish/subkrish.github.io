37   */
38  public class PrintClass {
39  
40      /**
41       * @author pugh
42       */
43      static final class ZipEntryComparator implements Comparator<ZipEntry>, Serializable {
44          private static final long serialVersionUID = 1L;
45  
46          @Override
47          public int compare(ZipEntry e1, ZipEntry e2) {
48              String s1 = e1.getName();
49              int pos1 = s1.lastIndexOf('/');
50              String p1 = "-";
51              if (pos1 >= 0) {
52                  p1 = s1.substring(0, pos1);
53              }
54  
55              String s2 = e2.getName();
56              int pos2 = s2.lastIndexOf('/');
57              String p2 = "-";
58              if (pos2 >= 0) {
59                  p2 = s2.substring(0, pos2);
60              }
61              int r = p1.compareTo(p2);
62              if (r != 0) {
63                  return r;
64              }
65              return s1.compareTo(s2);
66          }
67      }
68  
69      static boolean code = false, constants = false;
70  
71      static boolean superClasses = false;
72  
73      public static void main(String argv[]) throws IOException {
74          String[] file_name = new String[argv.length];
75          int files = 0;
76          String zip_file = null;
77  
78          /*
79           * Parse command line arguments.
80           */
81          for (int i = 0; i < argv.length; i++) {
82              if (argv[i].charAt(0) == '-') { // command line switch
83                  if ("-constants".equals(argv[i])) {
84                      constants = true;
85                  } else if ("-code".equals(argv[i])) {
86                      code = true;
87                  } else if ("-super".equals(argv[i])) {
88                      superClasses = true;
89                  } else if ("-zip".equals(argv[i])) {
90                      zip_file = argv[++i];
91                  }
92              } else if (argv[i].endsWith(".zip") || argv[i].endsWith(".jar")) {
93                  zip_file = argv[i];
94              } else { // add file name to list
95                  file_name[files++] = argv[i];
96              }
97          }
98  
99          if (!constants) {
100             code = true;
101         }
102         if (files == 0 && zip_file == null) {
103             System.err.println("list: No input files specified");
104         } else if (zip_file != null) {
105             for (int i = 0; i < files; i++) {
106                 file_name[i] = file_name[i].replace('.', '/');
107             }
108             try(ZipFile z = new ZipFile(zip_file)){
109                 TreeSet<ZipEntry> zipEntries = new TreeSet<ZipEntry>(new ZipEntryComparator());
110                 for (Enumeration<? extends ZipEntry> e = z.entries(); e.hasMoreElements();) {
111                     zipEntries.add(e.nextElement());
112                 }
113 
114                 for (ZipEntry ze : zipEntries) {
115                     String name = ze.getName();
116                     if (!name.endsWith(".class")) {
117                         continue;
118                     }
119                     checkMatch: if (files > 0) {
120                         for (int i = 0; i < files; i++) {
121                             if (name.indexOf(file_name[i]) >= 0) {
122                                 break checkMatch;
123                             }
124                         }
125                         continue;
126                     }
127                     printClass(new ClassParser(z.getInputStream(ze), name));
128 
129                 }
130             }
131         } else {
132             for (int i = 0; i < files; i++) {
133                 if (file_name[i].endsWith(".class")) {
134                     printClass(new ClassParser(file_name[i]));
135                 }
136             }
137         }
138     }

141     private static void printClass(ClassParser parser) throws IOException {
142         JavaClass java_class;
143         java_class = parser.parse();
144 
145         if (superClasses) {
146             try {
147                 while (java_class != null) {
148                     System.out.print(java_class.getClassName() + "  ");
149                     java_class = java_class.getSuperClass();
150                 }
151             } catch (ClassNotFoundException e) {
152                 System.out.println(e.getMessage());
153 
154             }
155             System.out.println();
156             return;
157         }
158         if (constants || code) {
159             System.out.println(java_class); // Dump the contents
160         }
161         if (constants) {
162             System.out.println(java_class.getConstantPool());
163         }
164 
165         if (code) {
166             printCode(java_class.getMethods());
167         }
168     }
169 
170     /**
171      * Dump the disassembled code of all methods in the class.
172      */
173     public static void printCode(Method[] methods) {
174         for (Method m : methods) {
175             System.out.println(m);
176             Code code = m.getCode();
177             if (code != null) {
178                 System.out.println(code);
179             }
180 
181         }
182     }
