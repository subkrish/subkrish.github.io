54      private static final String USAGE = "Usage: <cmd> " + " [options] [<oldData> [<newData>]]";

56      static class SetInfoCommandLine extends CommandLine {
57          String revisionName;
58  
59          String projectName;
60  
61          String exclusionFilterFile;
62  
63          String lastVersion;
64  
65          String cloudId;
66  
67          HashMap<String, String> cloudProperties = new HashMap<String, String>();
68  
69          boolean withMessages = false;
70  
71          boolean purgeStats = false;
72  
73          boolean purgeClassStats = false;
74  
75          boolean purgeMissingClasses = false;
76  
77          boolean resetSource = false;
78  
79          boolean resetProject = false;
80  
81          boolean purgeDesignations = false;
82  
83          long revisionTimestamp = 0L;
84  
85          public List<String> sourcePaths = new LinkedList<String>();
86  
87          public List<String> searchSourcePaths = new LinkedList<String>();
88  
89          SetInfoCommandLine() {
90              addOption("-name", "name", "set name for (last) revision");
91              addOption("-projectName", "name", "set name for project");
92              addOption("-timestamp", "when", "set timestamp for (last) revision");
93              addSwitch("-resetSource", "remove all source search paths");
94              addSwitch("-resetProject", "remove all source search paths, analysis and auxilary classpath entries");
95              addOption("-source", "directory", "Add this directory to the source search path");
96              addSwitch("-purgeStats", "purge/delete information about sizes of analyzed class files");
97              addSwitch("-uploadDesignations", "upload all designations to cloud");
98              addSwitch("-purgeDesignations", "purge/delete user designations (e.g., MUST_FIX or NOT_A_BUG");
99              addSwitch("-purgeClassStats", "purge/delete information about sizes of analyzed class files, but retain class stats");
100             addSwitch("-purgeMissingClasses", "purge list of missing classes");
101             addOption("-findSource", "directory", "Find and add all relevant source directions contained within this directory");
102             addOption("-suppress", "filter file", "Suppress warnings matched by this file (replaces previous suppressions)");
103             addOption("-lastVersion", "version", "Trim the history to just include just the specified version");
104             addSwitch("-withMessages", "Add bug descriptions");
105             addOption("-cloud", "id", "set cloud id");
106             addOption("-cloudProperty", "key=value", "set cloud property");
107 
108         }
109 
110         @Override
111         protected void handleOption(String option, String optionExtraPart) throws IOException {
112             if ("-withMessages".equals(option)) {
113                 withMessages = true;
114             } else if ("-resetSource".equals(option)) {
115                 resetSource = true;
116             } else if ("-resetProject".equals(option)) {
117                 resetProject = true;
118             } else if ("-purgeStats".equals(option)) {
119                 purgeStats = true;
120             } else if ("-purgeDesignations".equals(option)) {
121                 purgeDesignations = true;
122             } else if ("-purgeClassStats".equals(option)) {
123                 purgeClassStats = true;
124             } else if ("-purgeMissingClasses".equals(option)) {
125                 purgeMissingClasses = true;
126             } else {
127                 throw new IllegalArgumentException("no option " + option);
128             }
129 
130         }
131 
132         @Override
133         protected void handleOptionWithArgument(String option, String argument) throws IOException {
134             if ("-name".equals(option)) {
135                 revisionName = argument;
136             } else if ("-cloud".equals(option)) {
137                 cloudId = argument;
138             } else if ("-cloudProperty".equals(option)) {
139                 int e = argument.indexOf('=');
140                 if (e == -1) {
141                     throw new IllegalArgumentException("Bad cloud property: " + argument);
142                 }
143                 String key = argument.substring(0, e);
144                 String value = argument.substring(e + 1);
145                 cloudProperties.put(key, value);
146 
147             } else if ("-projectName".equals(option)) {
148                 projectName = argument;
149             } else if ("-suppress".equals(option)) {
150                 exclusionFilterFile = argument;
151             } else if ("-timestamp".equals(option)) {
152                 revisionTimestamp = Date.parse(argument);
153             } else if ("-source".equals(option)) {
154                 sourcePaths.add(argument);
155             } else if ("-lastVersion".equals(option)) {
156                 lastVersion = argument;
157             } else if ("-findSource".equals(option)) {
158                 searchSourcePaths.add(argument);
159             } else {
160                 throw new IllegalArgumentException("Can't handle option " + option);
161             }
162 
163         }
164 
165     }

167     public static void main(String[] args) throws IOException, DocumentException {
168         FindBugs.setNoAnalysis();
169         DetectorFactoryCollection.instance();
170         SetInfoCommandLine commandLine = new SetInfoCommandLine();
171         int argCount = commandLine.parse(args, 0, 2, USAGE);
172 
173         SortedBugCollection origCollection = new SortedBugCollection();
174 
175         if (argCount < args.length) {
176             origCollection.readXML(args[argCount++]);
177         } else {
178             origCollection.readXML(System.in);
179         }
180         Project project = origCollection.getProject();
181 
182         if (commandLine.revisionName != null) {
183             origCollection.setReleaseName(commandLine.revisionName);
184         }
185         if (commandLine.projectName != null) {
186             origCollection.getProject().setProjectName(commandLine.projectName);
187         }
188         if (commandLine.revisionTimestamp != 0) {
189             origCollection.setTimestamp(commandLine.revisionTimestamp);
190         }
191         origCollection.setWithMessages(commandLine.withMessages);
192 
193         if (commandLine.purgeDesignations) {
194             for (BugInstance b : origCollection) {
195                 b.setUserDesignation(null);
196             }
197         }
198         if (commandLine.exclusionFilterFile != null) {
199             project.setSuppressionFilter(Filter.parseFilter(commandLine.exclusionFilterFile));
200         }
201         if (commandLine.resetProject) {
202             project.getSourceDirList().clear();
203             project.getFileList().clear();
204             project.getAuxClasspathEntryList().clear();
205         }
206         boolean reinitializeCloud = false;
207         if (commandLine.cloudId != null) {
208             project.setCloudId(commandLine.cloudId);
209             reinitializeCloud = true;
210         }
211         for (Map.Entry<String, String> e : commandLine.cloudProperties.entrySet()) {
212             project.getCloudProperties().setProperty(e.getKey(), e.getValue());
213             reinitializeCloud = true;
214         }
215 
216         if (commandLine.resetSource) {
217             project.getSourceDirList().clear();
218         }
219         for (String source : commandLine.sourcePaths) {
220             project.addSourceDir(source);
221         }
222         if (commandLine.purgeStats) {
223             origCollection.getProjectStats().getPackageStats().clear();
224         }
225         if (commandLine.purgeClassStats) {
226             for (PackageStats ps : origCollection.getProjectStats().getPackageStats()) {
227                 ps.getClassStats().clear();
228             }
229         }
230         if (commandLine.purgeMissingClasses) {
231             origCollection.clearMissingClasses();
232         }
233         if (commandLine.lastVersion != null) {
234             long last = edu.umd.cs.findbugs.workflow.Filter.FilterCommandLine.getVersionNum(origCollection,
235                     commandLine.lastVersion, true);
236             if (last < origCollection.getSequenceNumber()) {
237                 String name = origCollection.getAppVersionFromSequenceNumber(last).getReleaseName();
238                 long timestamp = origCollection.getAppVersionFromSequenceNumber(last).getTimestamp();
239                 origCollection.setReleaseName(name);
240                 origCollection.setTimestamp(timestamp);
241                 origCollection.trimAppVersions(last);
242             }
243 
244         }
245 
246         Map<String, Set<String>> missingFiles = new HashMap<String, Set<String>>();
247         if (!commandLine.searchSourcePaths.isEmpty()) {
248             sourceSearcher = new SourceSearcher(project);
249             for (BugInstance bug : origCollection.getCollection()) {
250                 SourceLineAnnotation src = bug.getPrimarySourceLineAnnotation();
251                 if (!sourceSearcher.sourceNotFound.contains(src.getClassName()) && !sourceSearcher.findSource(src)) {
252                     Set<String> paths = missingFiles.get(src.getSourceFile());
253                     if (paths == null) {
254                         paths = new HashSet<String>();
255                         missingFiles.put(src.getSourceFile(), paths);
256                     }
257                     String fullPath = fullPath(src);
258                     // System.out.println("Missing " + fullPath);
259                     paths.add(fullPath);
260                 }
261             }
262             Set<String> foundPaths = new HashSet<String>();
263             for (String f : commandLine.searchSourcePaths) {
264                 for (File javaFile : RecursiveSearchForJavaFiles.search(new File(f))) {
265                     Set<String> matchingMissingClasses = missingFiles.get(javaFile.getName());
266                     if (matchingMissingClasses == null) {
267                         // System.out.println("Nothing for " + javaFile);
268                     } else {
269                         for (String sourcePath : matchingMissingClasses) {
270                             String path = javaFile.getAbsolutePath();
271                             if (path.endsWith(sourcePath)) {
272                                 String dir = path.substring(0, path.length() - sourcePath.length());
273                                 foundPaths.add(dir);
274 
275                             }
276                         }
277                     }
278 
279                 }
280             }
281 
282             Set<String> toRemove = new HashSet<String>();
283             for (String p1 : foundPaths) {
284                 for (String p2 : foundPaths) {
285                     if (!p1.equals(p2) && p1.startsWith(p2)) {
286                         toRemove.add(p1);
287                         break;
288                     }
289                 }
290             }
291             foundPaths.removeAll(toRemove);
292 
293             for (String dir : foundPaths) {
294                 project.addSourceDir(dir);
295                 if (argCount < args.length) {
296                     System.out.println("Found " + dir);
297                 }
298             }
299 
300         }
301 
302         if (reinitializeCloud)
303         {
304             origCollection.clearCloud();
305             // OK, now we know all the missing source files
306             // we also know all the .java files in the directories we were pointed
307             // to
308         }
309 
310         if (argCount < args.length) {
311             origCollection.writeXML(args[argCount++]);
312         } else {
313             origCollection.writeXML(System.out);
314         }
315 
316     }

318     static String fullPath(SourceLineAnnotation src) {
319         return src.getPackageName().replace('.', File.separatorChar) + File.separatorChar + src.getSourceFile();
320     }
