495         Collection<Integer> initialContents,

498       ConcurrentlyMutatedList<Integer> concurrentlyMutatedList =
499           newConcurrentlyMutatedList(
500               initialContents, actionsToPerformConcurrently);
501

503           ? unmodifiableIterable(concurrentlyMutatedList)

507           ImmutableList.copyOf(iterableToCopy);
508 
509       assertTrue(concurrentlyMutatedList.getAllStates()
510           .contains(copyOfIterable));
511     }

621     /**
622      * A list that mutates itself after every call to each of its {@link List}
623      * methods.
624      */
625     interface ConcurrentlyMutatedList<E> extends List<E> {
626       /**
627        * The elements of a {@link ConcurrentlyMutatedList} are added and removed
628        * over time. This method returns every state that the list has passed
629        * through at some point.
630        */
631       Set<List<E>> getAllStates();
632     }
633 
634     /**
635      * Returns a {@link ConcurrentlyMutatedList} that performs the given
636      * operations as its concurrent modifications. The mutations occur in the
637      * same thread as the triggering method call.
638      */
639     private static ConcurrentlyMutatedList<Integer> newConcurrentlyMutatedList(
640         final Collection<Integer> initialContents,
641         final Iterable<ListFrobber> actionsToPerformConcurrently) {
642       InvocationHandler invocationHandler = new InvocationHandler() {
643         final CopyOnWriteArrayList<Integer> delegate =
644             new CopyOnWriteArrayList<Integer>(initialContents);
645 
646         final Method getAllStatesMethod = getOnlyElement(asList(
647             ConcurrentlyMutatedList.class.getDeclaredMethods()));
648 
649         final Iterator<ListFrobber> remainingActions =
650             actionsToPerformConcurrently.iterator();
651 
652         final Set<List<Integer>> allStates = newHashSet();
653 
654         @Override
655         public Object invoke(Object proxy, Method method,
656             Object[] args) throws Throwable {
657           return method.equals(getAllStatesMethod)
658               ? getAllStates()
659               : invokeListMethod(method, args);
660         }
661 
662         private Set<List<Integer>> getAllStates() {
663           return allStates;
664         }
665 
666         private Object invokeListMethod(Method method, Object[] args)
667             throws Throwable {
668           try {
669             Object returnValue = method.invoke(delegate, args);
670             mutateDelegate();
671             return returnValue;
672           } catch (InvocationTargetException e) {
673             throw e.getCause();
674           } catch (IllegalAccessException e) {
675             throw new AssertionError(e);
676           }
677         }

680           allStates.add(ImmutableList.copyOf(delegate));
681           remainingActions.next().perform(delegate);
682           allStates.add(ImmutableList.copyOf(delegate));
683         }
684       };
