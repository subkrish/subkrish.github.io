61      TestSuite suite = new TestSuite();
62      suite.addTestSuite(ImmutableMultisetTest.class);

64      suite.addTest(MultisetTestSuiteBuilder.using(
65          new TestStringMultisetGenerator() {
66            @Override protected Multiset<String> create(String[] elements) {
67              return ImmutableMultiset.copyOf(elements);
68            }
69          })
70          .named("ImmutableMultiset")
71          .withFeatures(CollectionSize.ANY,
72              CollectionFeature.SERIALIZABLE_INCLUDING_VIEWS,
73              CollectionFeature.ALLOWS_NULL_QUERIES)
74          .createTestSuite());
75  
76      suite.addTest(SetTestSuiteBuilder.using(new TestStringSetGenerator() {
77            @Override protected Set<String> create(String[] elements) {
78              return ImmutableMultiset.copyOf(elements).elementSet();
79            }
80          })
81          .named("ImmutableMultiset, element set")
82          .withFeatures(CollectionSize.ANY,
83              CollectionFeature.SERIALIZABLE,
84              CollectionFeature.ALLOWS_NULL_QUERIES)
85          .createTestSuite());
86  
87      suite.addTest(ListTestSuiteBuilder.using(new TestStringListGenerator() {
88            @Override protected List<String> create(String[] elements) {
89              return ImmutableMultiset.copyOf(elements).asList();
90            }
91  
92            @Override
93            public List<String> order(List<String> insertionOrder) {
94              List<String> order = new ArrayList<String>();
95              for (String s : insertionOrder) {
96                int index = order.indexOf(s);
97                if (index == -1) {
98                  order.add(s);
99                } else {
100                 order.add(index, s);
101               }
102             }
103             return order;
104           }
105         })
106         .named("ImmutableMultiset.asList")
107         .withFeatures(CollectionSize.ANY,
108             CollectionFeature.SERIALIZABLE,
109             CollectionFeature.ALLOWS_NULL_QUERIES)
110         .createTestSuite());
111 
112     suite.addTest(ListTestSuiteBuilder.using(new TestStringListGenerator() {
113           @Override protected List<String> create(String[] elements) {
114             Set<String> set = new HashSet<String>();
115             ImmutableMultiset.Builder<String> builder = ImmutableMultiset.builder();
116             for (String s : elements) {
117               checkArgument(set.add(s));
118               builder.addCopies(s, 2);
119             }
120             ImmutableSet<String> elementSet = (ImmutableSet<String>) builder.build().elementSet();
121             return elementSet.asList();
122           }
123         })
124         .named("ImmutableMultiset.elementSet.asList")
125         .withFeatures(CollectionSize.ANY,
126             CollectionFeature.REJECTS_DUPLICATES_AT_CREATION,
127             CollectionFeature.SERIALIZABLE,
128             CollectionFeature.ALLOWS_NULL_QUERIES)
129         .createTestSuite());
130 
131     return suite;

312   public void testToImmutableMultiset_duplicates() {
313     class TypeWithDuplicates {
314       final int a;
315       final int b;
316 
317       TypeWithDuplicates(int a, int b) {
318         this.a = a;
319         this.b = b;
320       }
321 
322       @Override
323       public int hashCode() {
324         return a;
325       }
326 
327       @Override
328       public boolean equals(Object obj) {
329         return obj instanceof TypeWithDuplicates && ((TypeWithDuplicates) obj).a == a;
330       }
331 
332       public boolean fullEquals(TypeWithDuplicates other) {
333         return other != null && a == other.a && b == other.b;
334       }
335     }
336 
337     Collector<TypeWithDuplicates, ?, ImmutableMultiset<TypeWithDuplicates>> collector =
338         ImmutableMultiset.toImmutableMultiset();
339     BiPredicate<ImmutableMultiset<TypeWithDuplicates>, ImmutableMultiset<TypeWithDuplicates>>
340         equivalence =
341             (ms1, ms2)
342                 -> {
343                   if (!ms1.equals(ms2)) {
344                     return false;
345                   }
346                   List<TypeWithDuplicates> elements1 = ImmutableList.copyOf(ms1.elementSet());
347                   List<TypeWithDuplicates> elements2 = ImmutableList.copyOf(ms2.elementSet());
348                   for (int i = 0; i < ms1.elementSet().size(); i++) {
349                     if (!elements1.get(i).fullEquals(elements2.get(i))) {
350                       return false;
351                     }
352                   }
353                   return true;
354                 };
355     TypeWithDuplicates a = new TypeWithDuplicates(1, 1);
356     TypeWithDuplicates b1 = new TypeWithDuplicates(2, 1);
357     TypeWithDuplicates b2 = new TypeWithDuplicates(2, 2);
358     TypeWithDuplicates c = new TypeWithDuplicates(3, 1);
359     CollectorTester.of(collector, equivalence)
360         .expectCollects(
361             ImmutableMultiset.<TypeWithDuplicates>builder().add(a).addCopies(b1, 2).add(c).build(),
362             a,
363             b1,
364             c,
365             b2);
366     collector = ImmutableMultiset.toImmutableMultiset(e -> e, e -> 1);
367     CollectorTester.of(collector, equivalence)
368         .expectCollects(
369             ImmutableMultiset.<TypeWithDuplicates>builder().add(a).addCopies(b1, 2).add(c).build(),
370             a,
371             b1,
372             c,
373             b2);
374   }
