252     RateLimiter limiter = RateLimiter.create(stopwatch, 2.0, 4000, MILLISECONDS, 3.0);
253     for (int i = 0; i < 8; i++) {

256     stopwatch.sleepMillis(4500); // #2: back to cold state (warmup period + repay last acquire)

258       limiter.acquire(); // #3
259     }
260 
261     limiter.setRate(4.0); // double the rate!
262     limiter.acquire(); // #4, we repay the debt of the last acquire (imposed by the old rate)
263     for (int i = 0; i < 4; i++) {
264       limiter.acquire(); // #5
265     }
266     stopwatch.sleepMillis(4250); // #6, back to cold state (warmup period + repay last acquire)
267     for (int i = 0; i < 11; i++) {
268       limiter.acquire(); // #7, showing off the warmup starting from totally cold
269     }
270 
271     // make sure the areas (times) remain the same, while permits are different
272     assertEvents(
273         "R0.00, R1.38, R1.13, R0.88, R0.63, R0.50, R0.50, R0.50", // #1
274         "U4.50", // #2
275         "R0.00, R1.38, R1.13", // #3, after that the rate changes
276         "R0.88", // #4, this is what the throttling would be with the old rate
277         "R0.34, R0.28, R0.25, R0.25", // #5
278         "U4.25", // #6

448   /**
449    * Make sure that bursts can never go above 1-second-worth-of-work for the current
450    * rate, even when we change the rate.
451    */
452   public void testWeNeverGetABurstMoreThanOneSec() {
453     RateLimiter limiter = RateLimiter.create(stopwatch, 1.0);
454     int[] rates = { 1000, 1, 10, 1000000, 10, 1};
455     for (int rate : rates) {
456       int oneSecWorthOfWork = rate;
457       stopwatch.sleepMillis(rate * 1000);
458       limiter.setRate(rate);
459       long burst = measureTotalTimeMillis(limiter, oneSecWorthOfWork, new Random());
460       // we allow one second worth of work to go in a burst (i.e. take less than a second)
461       assertTrue(burst <= 1000);
462       long afterBurst = measureTotalTimeMillis(limiter, oneSecWorthOfWork, new Random());
463       // but work beyond that must take at least one second
464       assertTrue(afterBurst >= 1000);
465     }
466   }

468   /**
469    * This neat test shows that no matter what weights we use in our requests, if we push X
470    * amount of permits in a cool state, where X = rate * timeToCoolDown, and we have
471    * specified a timeToWarmUp() period, it will cost as the prescribed amount of time. E.g.,
472    * calling [acquire(5), acquire(1)] takes exactly the same time as
473    * [acquire(2), acquire(3), acquire(1)].
474    */
475   public void testTimeToWarmUpIsHonouredEvenWithWeights() {
476     Random random = new Random();
477     int warmupPermits = 10;
478     double[] coldFactorsToTest = { 2.0, 3.0, 10.0 };
479     double[] qpsToTest = { 4.0, 2.0, 1.0, 0.5, 0.1 };
480     for (int trial = 0; trial < 100; trial++) {
481       for (double coldFactor : coldFactorsToTest) {
482         for (double qps : qpsToTest) {
483           // If warmupPermits = maxPermits - thresholdPermits then
484           // warmupPeriod = (1 + coldFactor) * warmupPermits * stableInterval / 2
485           long warmupMillis = (long) ((1 + coldFactor) * warmupPermits / (2.0 * qps) * 1000.0);
486           RateLimiter rateLimiter = RateLimiter.create(
487               stopwatch, qps, warmupMillis, MILLISECONDS, coldFactor);
488           assertEquals(warmupMillis, measureTotalTimeMillis(rateLimiter, warmupPermits, random));
489         }
490       }
491     }
492   }

512   private long measureTotalTimeMillis(RateLimiter rateLimiter, int permits, Random random) {
513     long startTime = stopwatch.instant;
514     while (permits > 0) {
515       int nextPermitsToAcquire = Math.max(1, random.nextInt(permits));
516       permits -= nextPermitsToAcquire;
517       rateLimiter.acquire(nextPermitsToAcquire);
518     }
519     rateLimiter.acquire(1); // to repay for any pending debt
520     return NANOSECONDS.toMillis(stopwatch.instant - startTime);
521   }

529    * R0.6 means a delay of 0.6 seconds caused by the (R)ateLimiter
530    * U1.0 means the (U)ser caused the stopwatch to sleep for a second.
531    */
532   static class FakeStopwatch extends SleepingStopwatch {
533     long instant = 0L;
534     final List<String> events = Lists.newArrayList();
535 
536     @Override
537     public long readMicros() {
538       return NANOSECONDS.toMicros(instant);
539     }
540 
541     void sleepMillis(int millis) {
542       sleepMicros("U", MILLISECONDS.toMicros(millis));
543     }
544 
545     void sleepMicros(String caption, long micros) {
546       instant += MICROSECONDS.toNanos(micros);
547       events.add(caption + String.format(Locale.ROOT, "%3.2f", (micros / 1000000.0)));
548     }
549 
550     @Override
551     protected void sleepMicrosUninterruptibly(long micros) {
552       sleepMicros("R", micros);
553     }
554 
555     String readEventsAndClear() {
556       try {
557         return events.toString();
558       } finally {
559         events.clear();
560       }
561     }
562 
563     @Override
564     public String toString() {
565       return events.toString();
566     }
567   }

586   private static void doTestMocking(RateLimiter mock) throws Exception {
587     for (Method method : RateLimiter.class.getMethods()) {
588       if (!isStatic(method.getModifiers())
589           && !NOT_WORKING_ON_MOCKS.contains(method.getName())
590           && !method.getDeclaringClass().equals(Object.class)) {
591         method.invoke(mock, arbitraryParameters(method));
592       }
593     }
594   }
595 
596   private static Object[] arbitraryParameters(Method method) {
597     Class<?>[] parameterTypes = method.getParameterTypes();
598     Object[] params = new Object[parameterTypes.length];
599     for (int i = 0; i < parameterTypes.length; i++) {
600       params[i] = PARAMETER_VALUES.get(parameterTypes[i]);
601     }
602     return params;
603   }
604 
605   private static final ImmutableSet<String> NOT_WORKING_ON_MOCKS =
606       ImmutableSet.of("latestPermitAgeSec", "setRate", "getAvailablePermits");
607 
608   // We would use ArbitraryInstances, but it returns 0, invalid for many RateLimiter methods.
609   private static final ImmutableClassToInstanceMap<Object> PARAMETER_VALUES =
610       ImmutableClassToInstanceMap.builder()
611           .put(int.class, 1)
612           .put(long.class, 1L)
613           .put(double.class, 1.0)
614           .put(TimeUnit.class, SECONDS)
615           .build();
