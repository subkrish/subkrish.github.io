17  /**

109     private int crc = 0;
110 
111     @Override
112     public void update(byte b) {
113       crc ^= 0xFFFFFFFF;
114       // See Hacker's Delight 2nd Edition, Figure 14-7.
115       crc = ~((crc >>> 8) ^ CRC_TABLE[(crc ^ b) & 0xFF]);
116     }
117 
118     @Override
119     public HashCode hash() {
120       return HashCode.fromInt(crc);
121     }
