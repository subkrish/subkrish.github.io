17  /**

110     private int crc = 0;
111 
112     @Override
113     public void update(byte b) {
114       crc ^= 0xFFFFFFFF;
115       // See Hacker's Delight 2nd Edition, Figure 14-7.
116       crc = ~((crc >>> 8) ^ CRC_TABLE[(crc ^ b) & 0xFF]);
117     }
118 
119     @Override
120     public HashCode hash() {
121       return HashCode.fromInt(crc);
122     }
