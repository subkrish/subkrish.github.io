25  package org.hibernate.engine.jdbc.internal;

27  import java.util.HashSet;
28  import java.util.LinkedList;
29  import java.util.Set;
30  import java.util.StringTokenizer;
31  
32  import org.hibernate.internal.util.StringHelper;
33  
34  /**
35   * Performs formatting of basic SQL statements (DML + query).
36   *
37   * @author Gavin King
38   * @author Steve Ebersole
39   */
40  public class BasicFormatterImpl implements Formatter {
41  
42  	private static final Set<String> BEGIN_CLAUSES = new HashSet<String>();
43  	private static final Set<String> END_CLAUSES = new HashSet<String>();
44  	private static final Set<String> LOGICAL = new HashSet<String>();
45  	private static final Set<String> QUANTIFIERS = new HashSet<String>();
46  	private static final Set<String> DML = new HashSet<String>();
47  	private static final Set<String> MISC = new HashSet<String>();
48  
49  	static {
50  		BEGIN_CLAUSES.add( "left" );
51  		BEGIN_CLAUSES.add( "right" );
52  		BEGIN_CLAUSES.add( "inner" );
53  		BEGIN_CLAUSES.add( "outer" );
54  		BEGIN_CLAUSES.add( "group" );
55  		BEGIN_CLAUSES.add( "order" );
56  
57  		END_CLAUSES.add( "where" );
58  		END_CLAUSES.add( "set" );
59  		END_CLAUSES.add( "having" );
60  		END_CLAUSES.add( "join" );
61  		END_CLAUSES.add( "from" );
62  		END_CLAUSES.add( "by" );
63  		END_CLAUSES.add( "join" );
64  		END_CLAUSES.add( "into" );
65  		END_CLAUSES.add( "union" );
66  
67  		LOGICAL.add( "and" );
68  		LOGICAL.add( "or" );
69  		LOGICAL.add( "when" );
70  		LOGICAL.add( "else" );
71  		LOGICAL.add( "end" );
72  
73  		QUANTIFIERS.add( "in" );
74  		QUANTIFIERS.add( "all" );
75  		QUANTIFIERS.add( "exists" );
76  		QUANTIFIERS.add( "some" );
77  		QUANTIFIERS.add( "any" );
78  
79  		DML.add( "insert" );
80  		DML.add( "update" );
81  		DML.add( "delete" );
82  
83  		MISC.add( "select" );
84  		MISC.add( "on" );
85  	}
86  
87  	static final String indentString = "    ";
88  	static final String initial = "\n    ";
89  
90  	public String format(String source) {
91  		return new FormatProcess( source ).perform();
92  	}
93  
94  	private static class FormatProcess {
95  		boolean beginLine = true;
96  		boolean afterBeginBeforeEnd = false;
97  		boolean afterByOrSetOrFromOrSelect = false;
98  		boolean afterValues = false;
99  		boolean afterOn = false;
100 		boolean afterBetween = false;
101 		boolean afterInsert = false;
102 		int inFunction = 0;
103 		int parensSinceSelect = 0;
104 		private LinkedList<Integer> parenCounts = new LinkedList<Integer>();
105 		private LinkedList<Boolean> afterByOrFromOrSelects = new LinkedList<Boolean>();
106 
107 		int indent = 1;
108 
109 		StringBuilder result = new StringBuilder();
110 		StringTokenizer tokens;
111 		String lastToken;
112 		String token;
113 		String lcToken;
114 
115 		public FormatProcess(String sql) {
116 			tokens = new StringTokenizer(
117 					sql,
118 					"()+*/-=<>'`\"[]," + StringHelper.WHITESPACE,
119 					true
120 			);
121 		}
122 
123 		public String perform() {
124 
125 			result.append( initial );
126 
127 			while ( tokens.hasMoreTokens() ) {
128 				token = tokens.nextToken();
129 				lcToken = token.toLowerCase();
130 
131 				if ( "'".equals( token ) ) {
132 					String t;
133 					do {
134 						t = tokens.nextToken();
135 						token += t;
136 					}
137 					while ( !"'".equals( t ) && tokens.hasMoreTokens() ); // cannot handle single quotes
138 				}
139 				else if ( "\"".equals( token ) ) {
140 					String t;
141 					do {
142 						t = tokens.nextToken();
143 						token += t;
144 					}
145 					while ( !"\"".equals( t ) );
146 				}
147 
148 				if ( afterByOrSetOrFromOrSelect && ",".equals( token ) ) {
149 					commaAfterByOrFromOrSelect();
150 				}
151 				else if ( afterOn && ",".equals( token ) ) {
152 					commaAfterOn();
153 				}
154 
155 				else if ( "(".equals( token ) ) {
156 					openParen();
157 				}
158 				else if ( ")".equals( token ) ) {
159 					closeParen();
160 				}
161 
162 				else if ( BEGIN_CLAUSES.contains( lcToken ) ) {
163 					beginNewClause();
164 				}
165 
166 				else if ( END_CLAUSES.contains( lcToken ) ) {
167 					endNewClause();
168 				}
169 
170 				else if ( "select".equals( lcToken ) ) {
171 					select();
172 				}
173 
174 				else if ( DML.contains( lcToken ) ) {
175 					updateOrInsertOrDelete();
176 				}
177 
178 				else if ( "values".equals( lcToken ) ) {
179 					values();
180 				}
181 
182 				else if ( "on".equals( lcToken ) ) {
183 					on();
184 				}
185 
186 				else if ( afterBetween && lcToken.equals( "and" ) ) {
187 					misc();
188 					afterBetween = false;
189 				}
190 
191 				else if ( LOGICAL.contains( lcToken ) ) {
192 					logical();
193 				}
194 
195 				else if ( isWhitespace( token ) ) {
196 					white();
197 				}
198 
199 				else {
200 					misc();
201 				}
202 
203 				if ( !isWhitespace( token ) ) {
204 					lastToken = lcToken;
205 				}
206 
207 			}
208 			return result.toString();
209 		}
210 
211 		private void commaAfterOn() {
212 			out();
213 			indent--;
214 			newline();
215 			afterOn = false;
216 			afterByOrSetOrFromOrSelect = true;
217 		}
218 
219 		private void commaAfterByOrFromOrSelect() {
220 			out();
221 			newline();
222 		}
223 
224 		private void logical() {
225 			if ( "end".equals( lcToken ) ) {
226 				indent--;
227 			}
228 			newline();
229 			out();
230 			beginLine = false;
231 		}
232 
233 		private void on() {
234 			indent++;
235 			afterOn = true;
236 			newline();
237 			out();
238 			beginLine = false;
239 		}
240 
241 		private void misc() {
242 			out();
243 			if ( "between".equals( lcToken ) ) {
244 				afterBetween = true;
245 			}
246 			if ( afterInsert ) {
247 				newline();
248 				afterInsert = false;
249 			}
250 			else {
251 				beginLine = false;
252 				if ( "case".equals( lcToken ) ) {
253 					indent++;
254 				}
255 			}
256 		}
257 
258 		private void white() {
259 			if ( !beginLine ) {
260 				result.append( " " );
261 			}
262 		}
263 
264 		private void updateOrInsertOrDelete() {
265 			out();
266 			indent++;
267 			beginLine = false;
268 			if ( "update".equals( lcToken ) ) {
269 				newline();
270 			}
271 			if ( "insert".equals( lcToken ) ) {
272 				afterInsert = true;
273 			}
274 		}
275 
276 		@SuppressWarnings( {"UnnecessaryBoxing"})
277 		private void select() {
278 			out();
279 			indent++;
280 			newline();
281 			parenCounts.addLast( Integer.valueOf( parensSinceSelect ) );
282 			afterByOrFromOrSelects.addLast( Boolean.valueOf( afterByOrSetOrFromOrSelect ) );
283 			parensSinceSelect = 0;
284 			afterByOrSetOrFromOrSelect = true;
285 		}
286 
287 		private void out() {
288 			result.append( token );
289 		}
290 
291 		private void endNewClause() {
292 			if ( !afterBeginBeforeEnd ) {
293 				indent--;
294 				if ( afterOn ) {
295 					indent--;
296 					afterOn = false;
297 				}
298 				newline();
299 			}
300 			out();
301 			if ( !"union".equals( lcToken ) ) {
302 				indent++;
303 			}
304 			newline();
305 			afterBeginBeforeEnd = false;
306 			afterByOrSetOrFromOrSelect = "by".equals( lcToken )
307 					|| "set".equals( lcToken )
308 					|| "from".equals( lcToken );
309 		}
310 
311 		private void beginNewClause() {
312 			if ( !afterBeginBeforeEnd ) {
313 				if ( afterOn ) {
314 					indent--;
315 					afterOn = false;
316 				}
317 				indent--;
318 				newline();
319 			}
320 			out();
321 			beginLine = false;
322 			afterBeginBeforeEnd = true;
323 		}
324 
325 		private void values() {
326 			indent--;
327 			newline();
328 			out();
329 			indent++;
330 			newline();
331 			afterValues = true;
332 		}
333 
334 		@SuppressWarnings( {"UnnecessaryUnboxing"})
335 		private void closeParen() {
336 			parensSinceSelect--;
337 			if ( parensSinceSelect < 0 ) {
338 				indent--;
339 				parensSinceSelect = parenCounts.removeLast().intValue();
340 				afterByOrSetOrFromOrSelect = afterByOrFromOrSelects.removeLast().booleanValue();
341 			}
342 			if ( inFunction > 0 ) {
343 				inFunction--;
344 				out();
345 			}
346 			else {
347 				if ( !afterByOrSetOrFromOrSelect ) {
348 					indent--;
349 					newline();
350 				}
351 				out();
352 			}
353 			beginLine = false;
354 		}
355 
356 		private void openParen() {
357 			if ( isFunctionName( lastToken ) || inFunction > 0 ) {
358 				inFunction++;
359 			}
360 			beginLine = false;
361 			if ( inFunction > 0 ) {
362 				out();
363 			}
364 			else {
365 				out();
366 				if ( !afterByOrSetOrFromOrSelect ) {
367 					indent++;
368 					newline();
369 					beginLine = true;
370 				}
371 			}
372 			parensSinceSelect++;
373 		}
374 
375 		private static boolean isFunctionName(String tok) {
376 			final char begin = tok.charAt( 0 );
377 			final boolean isIdentifier = Character.isJavaIdentifierStart( begin ) || '"' == begin;
378 			return isIdentifier &&
379 					!LOGICAL.contains( tok ) &&
380 					!END_CLAUSES.contains( tok ) &&
381 					!QUANTIFIERS.contains( tok ) &&
382 					!DML.contains( tok ) &&
383 					!MISC.contains( tok );
384 		}
385 
386 		private static boolean isWhitespace(String token) {
387 			return StringHelper.WHITESPACE.indexOf( token ) >= 0;
388 		}
389 
390 		private void newline() {
391 			result.append( "\n" );
392 			for ( int i = 0; i < indent; i++ ) {
393 				result.append( indentString );
394 			}
395 			beginLine = true;
396 		}
397 	}
