121 	/**
122 	 * Constructor version used during shallow copy
123 	 *
124 	 * @param registeredTypes The type map to copy over
125 	 */
126 	@SuppressWarnings({ "UnusedDeclaration" })
127 	private BasicTypeRegistry(Map<String, BasicType> registeredTypes) {
128 		registry.putAll( registeredTypes );
129 		locked = true;
130 	}
131 
132 	public void register(BasicType type) {
133 		if ( locked ) {
134 			throw new HibernateException( "Can not alter TypeRegistry at this time" );
135 		}
136 
137 		if ( type == null ) {
138 			throw new HibernateException( "Type to register cannot be null" );
139 		}
140 
141 		if ( type.getRegistrationKeys() == null || type.getRegistrationKeys().length == 0 ) {
142 			LOG.typeDefinedNoRegistrationKeys( type );
143 		}
144 
145 		for ( String key : type.getRegistrationKeys() ) {
146 			// be safe...
147             if (key == null) continue;
148             LOG.debugf("Adding type registration %s -> %s", key, type);
149 			final Type old = registry.put( key, type );
150             if (old != null && old != type) LOG.typeRegistrationOverridesPrevious(key, old);
151 		}
152 	}
153 
154 	public void register(UserType type, String[] keys) {
155 		register( new CustomType( type, keys ) );
156 	}
157 
158 	public void register(CompositeUserType type, String[] keys) {
159 		register( new CompositeCustomType( type, keys ) );
160 	}
161 
162 	public BasicType getRegisteredType(String key) {
163 		return registry.get( key );
164 	}
165 
166 	public BasicTypeRegistry shallowCopy() {
167 		return new BasicTypeRegistry( this.registry );
168 	}
