24  package org.hibernate.test.common;
25  
26  import java.sql.Connection;
27  
28  import org.hibernate.engine.jdbc.spi.ConnectionObserver;
29  
30  /**
31   * @author Steve Ebersole
32   */
33  public class JournalingConnectionObserver implements ConnectionObserver {
34  	private int physicalConnectionObtainedCount = 0;
35  	private int physicalConnectionReleasedCount = 0;
36  	private int logicalConnectionClosedCount = 0;
37  	private int statementPreparedCount = 0;
38  
39  	@Override
40  	public void physicalConnectionObtained(Connection connection) {
41  		physicalConnectionObtainedCount++;
42  	}
43  
44  	@Override
45  	public void physicalConnectionReleased() {
46  		physicalConnectionReleasedCount++;
47  	}
48  
49  	@Override
50  	public void logicalConnectionClosed() {
51  		logicalConnectionClosedCount++;
52  	}
53  
54  	@Override
55  	public void statementPrepared() {
56  		statementPreparedCount++;
57  	}
58  
59  	public int getPhysicalConnectionObtainedCount() {
60  		return physicalConnectionObtainedCount;
61  	}
62  
63  	public int getPhysicalConnectionReleasedCount() {
64  		return physicalConnectionReleasedCount;
65  	}
66  
67  	public int getLogicalConnectionClosedCount() {
68  		return logicalConnectionClosedCount;
69  	}
70  
71  	public int getStatementPreparedCount() {
72  		return statementPreparedCount;
73  	}
74  }
