21  package org.hibernate.internal.util;
22  
23  /**
24   * This exists purely to allow custom ClassLoaders to be injected and used
25   * prior to ServiceRegistry and ClassLoadingService existence.  This should be
26   * replaced in Hibernate 5.
27   * 
28   * @author Brett Meyer
29   */
30  public class ClassLoaderHelper {
31  	
32  	public static ClassLoader overridenClassLoader = null;
33  	
34  	public static ClassLoader getContextClassLoader() {
35  		return overridenClassLoader != null ?
36  			overridenClassLoader : Thread.currentThread().getContextClassLoader();
37  	}
38  }
