24  package org.hibernate.ejb.test.metadata;
25  import java.util.ArrayList;
26  import java.util.List;
27  import javax.persistence.Basic;
28  import javax.persistence.CascadeType;
29  import javax.persistence.Column;
30  import javax.persistence.Entity;
31  import javax.persistence.Id;
32  import javax.persistence.JoinColumn;
33  import javax.persistence.ManyToOne;
34  import javax.persistence.OneToMany;
35  import javax.persistence.Table;
36  
37  /**
38  * This class has a List of mapped entity objects that are themselves parameterized.
39  * This class was added for JIRA issue #HHH-
40  *
41  * @author Kahli Burke
42  */
43  @Entity
44  @Table(name = "WITH_GENERIC_COLLECTION")
45  public class WithGenericCollection<T> implements java.io.Serializable {
46      @Id
47      @Column(name = "ID")
48      private String id;
49  
50      @Basic(optional=false)
51      private double d;
52  
53      @ManyToOne(optional=false)
54      @JoinColumn(name="PARENT_ID", insertable=false, updatable=false)
55      private WithGenericCollection<? extends Object> parent = null;
56  
57      @OneToMany(cascade = CascadeType.ALL)
58      @JoinColumn(name="PARENT_ID")
59      private List<WithGenericCollection<? extends Object>> children = new ArrayList<WithGenericCollection<? extends Object>>();
60  
61      public WithGenericCollection() {
62      }
63  
64      //====================================================================
65      // getters and setters for State fields
66  
67      public String getId() {
68          return id;
69      }
70  
71      public void setId(String id) {
72          this.id = id;
73      }
74  
75      public void setD(double d) {
76          this.d = d;
77      }
78  
79      public double getD() {
80          return d;
81      }
82  
83      public List<WithGenericCollection<? extends Object>> getChildren() {
84          return children;
85      }
86  
87      public void setChildren(List<WithGenericCollection<? extends Object>> children) {
88          this.children = children;
89      }
90  
91  
92  }
