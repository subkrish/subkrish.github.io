18  @TestForIssue(jiraKey = "HHH-3843")
19  public class ParentReferencingChildTest extends BaseEnversJPAFunctionalTestCase {
20      Person expLukaszRev1 = null;
21      Role expAdminRev1 = null;
22  
23      public void configure(Ejb3Configuration cfg) {
24          cfg.addAnnotatedClass(Person.class);
25          cfg.addAnnotatedClass(Role.class);
26          cfg.addAnnotatedClass(RightsSubject.class);
27      }
28  
29      @Test
30      @Priority(10)
31      public void initData() {
32          EntityManager em = getEntityManager();
33  
34          // Revision 1
35          em.getTransaction().begin();
36          Person lukasz = new Person();
37          lukasz.setName("Lukasz");
38          lukasz.setGroup("IT");
39          em.persist(lukasz);
40          Role admin = new Role();
41          admin.setName("Admin");
42          admin.setGroup("Confidential");
43          lukasz.getRoles().add(admin);
44          admin.getMembers().add(lukasz);
45          em.persist(admin);
46          em.getTransaction().commit();
47  
48          expLukaszRev1 = new Person(lukasz.getId(), "IT", "Lukasz");
49          expAdminRev1 = new Role(admin.getId(), "Confidential", "Admin");
50      }
51  
52      @Test
53      public void testRevisionsCounts() {
54          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(Person.class, expLukaszRev1.getId()));
55          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(RightsSubject.class, expLukaszRev1.getId()));
56  
57          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(Role.class, expAdminRev1.getId()));
58          Assert.assertEquals(Arrays.asList(1), getAuditReader().getRevisions(RightsSubject.class, expAdminRev1.getId()));
59      }
60  
61      @Test
62      public void testHistoryOfLukasz() {
63          Person lukaszRev1 = getAuditReader().find(Person.class, expLukaszRev1.getId(), 1);
64          RightsSubject rightsSubjectLukaszRev1 = getAuditReader().find(RightsSubject.class, expLukaszRev1.getId(), 1);
65  
66          Assert.assertEquals(expLukaszRev1, lukaszRev1);
67          Assert.assertEquals(TestTools.makeSet(expAdminRev1), lukaszRev1.getRoles());
68          Assert.assertEquals(TestTools.makeSet(expAdminRev1), rightsSubjectLukaszRev1.getRoles());
69      }
70  
71      @Test
72      public void testHistoryOfAdmin() {
73          Role adminRev1 = getAuditReader().find(Role.class, expAdminRev1.getId(), 1);
74  
75          Assert.assertEquals(expAdminRev1, adminRev1);
76          Assert.assertEquals(TestTools.makeSet(expLukaszRev1), adminRev1.getMembers());
77      }
78  }
