27      }
28  
29      @Test
30      @Priority(10)
31      public void initData() {
32          EntityManager em = getEntityManager();
33  
34          // Revision 1
35          em.getTransaction().begin();
36          Parent parent = new Parent("parent");
37          Child child = new Child("child");
38          parent.getCollection().add(child);
39          em.persist(child);
40          em.persist(parent);
41          em.getTransaction().commit();
42  
43          parentId = parent.getId();
44          childId = child.getId();
45      }
46  
47      @Test
48      public void testAuditChildTableAlias() {
49          Parent parent = new Parent("parent", parentId);
50          Child child = new Child("child", childId);
51  
52          Parent ver1 = getAuditReader().find(Parent.class, parentId, 1);
53  
54          Assert.assertEquals(parent, ver1);
55          Assert.assertEquals(TestTools.makeSet(child), ver1.getCollection());
56      }
