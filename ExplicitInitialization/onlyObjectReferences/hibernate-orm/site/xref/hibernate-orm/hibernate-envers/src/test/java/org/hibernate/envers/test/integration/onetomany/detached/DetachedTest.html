15  /**
16   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
17   */
18  public class DetachedTest extends BaseEnversFunctionalTestCase {
19  	private Integer parentId = null;
20  	private Integer childId = null;
21  
22  	@Override
23  	protected Class<?>[] getAnnotatedClasses() {
24  		return new Class[] { ListRefCollEntity.class, StrTestEntity.class };
25  	}
26  
27  	@Test
28  	@Priority(10)
29  	@TestForIssue(jiraKey = "HHH-7543")
30  	public void testUpdatingDetachedEntityWithRelation() {
31  		Session session = getSession();
32  
33  		// Revision 1
34  		session.getTransaction().begin();
35  		ListRefCollEntity parent = new ListRefCollEntity( 1, "initial data" );
36  		StrTestEntity child = new StrTestEntity( "data" );
37  		session.save( child );
38  		parent.setCollection( Arrays.asList( child ) );
39  		session.save( parent );
40  		session.getTransaction().commit();
41  
42  		session.close();
43  		session = getSession();
44  
45  		// Revision 2 - updating detached entity
46  		session.getTransaction().begin();
47  		parent.setData( "modified data" );
48  		session.update( parent );
49  		session.getTransaction().commit();
50  
51  		session.close();
52  
53  		parentId = parent.getId();
54  		childId = child.getId();
55  	}
56  
57  	@Test
58  	public void testRevisionsCounts() {
59  		Assert.assertEquals( Arrays.asList( 1, 2 ), getAuditReader().getRevisions( ListRefCollEntity.class, parentId ) );
60  		Assert.assertEquals( Arrays.asList( 1 ), getAuditReader().getRevisions( StrTestEntity.class, childId ) );
61  	}
62  
63  	@Test
64  	public void testHistoryOfParent() {
65  		ListRefCollEntity parent = new ListRefCollEntity( parentId, "initial data" );
66  		parent.setCollection( Arrays.asList( new StrTestEntity( "data", childId ) ) );
67  
68  		ListRefCollEntity ver1 = getAuditReader().find( ListRefCollEntity.class, parentId, 1 );
69  
70  		Assert.assertEquals( parent, ver1 );
71  		Assert.assertEquals( parent.getCollection(), ver1.getCollection() );
72  
73  		parent.setData( "modified data" );
74  
75  		ListRefCollEntity ver2 = getAuditReader().find( ListRefCollEntity.class, parentId, 2 );
76  
77  		Assert.assertEquals( parent, ver2 );
78  		Assert.assertEquals( parent.getCollection(), ver2.getCollection() );
79  	}
80  }
