24  package org.hibernate.envers.test.integration.onetoone.bidirectional;

26  import javax.persistence.EntityManager;
27  
28  import org.junit.Test;
29  
30  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
31  import org.hibernate.envers.test.Priority;
32  import org.hibernate.envers.test.entities.onetoone.BidirectionalEagerAnnotationRefEdOneToOne;
33  import org.hibernate.envers.test.entities.onetoone.BidirectionalEagerAnnotationRefIngOneToOne;
34  import org.hibernate.testing.TestForIssue;
35  
36  import static org.junit.Assert.assertNotNull;
37  
38  /**
39   * @author Erik-Berndt Scheper
40   */
41  @TestForIssue(jiraKey = "HHH-3854")
42  public class BidirectionalEagerAnnotationTest extends BaseEnversJPAFunctionalTestCase {
43  	private Integer refIngId1 = null;
44  
45  	@Override
46  	protected Class<?>[] getAnnotatedClasses() {
47  		return new Class<?>[] {
48  				BidirectionalEagerAnnotationRefEdOneToOne.class,
49  				BidirectionalEagerAnnotationRefIngOneToOne.class
50  		};
51  	}
52  
53  	@Test
54  	@Priority(10)
55  	public void initData() {
56  		EntityManager em = getEntityManager();
57  
58  		// Revision 1
59  		em.getTransaction().begin();
60  		BidirectionalEagerAnnotationRefEdOneToOne ed1 = new BidirectionalEagerAnnotationRefEdOneToOne();
61  		BidirectionalEagerAnnotationRefIngOneToOne ing1 = new BidirectionalEagerAnnotationRefIngOneToOne();
62  		ed1.setData( "referredEntity1" );
63  		ed1.setRefIng( ing1 );
64  		ing1.setData( "referringEntity" );
65  		ing1.setRefedOne( ed1 );
66  		em.persist( ed1 );
67  		em.persist( ing1 );
68  		em.getTransaction().commit();
69  
70  		refIngId1 = ing1.getId();
71  
72  		em.close();
73  	}
74  
75  	@Test
76  	public void testNonProxyObjectTraversing() {
77  		BidirectionalEagerAnnotationRefIngOneToOne referencing =
78  				getAuditReader().find( BidirectionalEagerAnnotationRefIngOneToOne.class, refIngId1, 1 );
79  		assertNotNull( referencing.getRefedOne().getData() );
80  	}
81  }
