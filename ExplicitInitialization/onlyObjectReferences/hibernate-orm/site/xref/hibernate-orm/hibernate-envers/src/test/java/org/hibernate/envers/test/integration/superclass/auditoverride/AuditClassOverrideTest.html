18  @TestForIssue(jiraKey = "HHH-4439")
19  public class AuditClassOverrideTest extends BaseEnversJPAFunctionalTestCase {
20      private Integer classAuditedEntityId = null;
21      private Integer classNotAuditedEntityId = null;
22      private Table classAuditedTable = null;
23      private Table classNotAuditedTable = null;
24  
25      @Override
26      public void configure(Ejb3Configuration cfg) {
27          cfg.addAnnotatedClass(ClassOverrideAuditedEntity.class);
28          cfg.addAnnotatedClass(ClassOverrideNotAuditedEntity.class);
29      }
30  
31      @Test
32      @Priority(10)
33      public void initData() {
34          EntityManager em = getEntityManager();
35  
36          // Revision 1
37          em.getTransaction().begin();
38          ClassOverrideAuditedEntity classOverrideAuditedEntity = new ClassOverrideAuditedEntity("data 1", 1, "data 2");
39          em.persist(classOverrideAuditedEntity);
40          em.getTransaction().commit();
41          classAuditedEntityId = classOverrideAuditedEntity.getId();
42  
43          // Revision 2
44          em.getTransaction().begin();
45          ClassOverrideNotAuditedEntity classOverrideNotAuditedEntity = new ClassOverrideNotAuditedEntity("data 1", 1, "data 2");
46          em.persist(classOverrideNotAuditedEntity);
47          em.getTransaction().commit();
48          classNotAuditedEntityId = classOverrideNotAuditedEntity.getId();
49  
50          classAuditedTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.superclass.auditoverride.ClassOverrideAuditedEntity_AUD").getTable();
51          classNotAuditedTable = getCfg().getClassMapping("org.hibernate.envers.test.integration.superclass.auditoverride.ClassOverrideNotAuditedEntity_AUD").getTable();
52      }
53  
54      @Test
55      public void testAuditedProperty() {
56          Assert.assertNotNull(classAuditedTable.getColumn(new Column("number1")));
57          Assert.assertNotNull(classAuditedTable.getColumn(new Column("str1")));
58          Assert.assertNotNull(classAuditedTable.getColumn(new Column("str2")));
59          Assert.assertNotNull(classNotAuditedTable.getColumn(new Column("str2")));
60      }
61  
62      @Test
63      public void testNotAuditedProperty() {
64          Assert.assertNull(classNotAuditedTable.getColumn(new Column("number1")));
65          Assert.assertNull(classNotAuditedTable.getColumn(new Column("str1")));
66      }
67  
68      @Test
69      public void testHistoryOfClassOverrideAuditedEntity() {
70          ClassOverrideAuditedEntity ver1 = new ClassOverrideAuditedEntity("data 1", 1, classAuditedEntityId, "data 2");
71          Assert.assertEquals(ver1, getAuditReader().find(ClassOverrideAuditedEntity.class, classAuditedEntityId, 1));
72      }
73  
74      @Test
75      public void testHistoryOfClassOverrideNotAuditedEntity() {
76          ClassOverrideNotAuditedEntity ver1 = new ClassOverrideNotAuditedEntity(null, null, classNotAuditedEntityId, "data 2");
77          Assert.assertEquals(ver1, getAuditReader().find(ClassOverrideNotAuditedEntity.class, classNotAuditedEntityId, 2));
78      }
79  }
