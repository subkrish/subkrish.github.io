42  /**
43   * A factory of bulk accessors.
44   *
45   * @author Muga Nishizawa
46   * @author modified by Shigeru Chiba
47   */
48  class BulkAccessorFactory {
49  	private static final String PACKAGE_NAME_PREFIX = "org.javassist.tmp.";
50  	private static final String BULKACESSOR_CLASS_NAME = BulkAccessor.class.getName();
51  	private static final String OBJECT_CLASS_NAME = Object.class.getName();
52  	private static final String GENERATED_GETTER_NAME = "getPropertyValues";
53  	private static final String GENERATED_SETTER_NAME = "setPropertyValues";
54  	private static final String GET_SETTER_DESC = "(Ljava/lang/Object;[Ljava/lang/Object;)V";
55  	private static final String THROWABLE_CLASS_NAME = Throwable.class.getName();
56  	private static final String BULKEXCEPTION_CLASS_NAME = BulkAccessorException.class.getName();
57  	private static int counter = 0;
58  
59  	private Class targetBean;
60  	private String[] getterNames;
61  	private String[] setterNames;
62  	private Class[] types;
63  	public String writeDirectory;
64  
65  	BulkAccessorFactory(
66  			Class target,
67  	        String[] getterNames,
68  	        String[] setterNames,
69  	        Class[] types) {
70  		this.targetBean = target;
71  		this.getterNames = getterNames;
72  		this.setterNames = setterNames;
73  		this.types = types;
74  		this.writeDirectory = null;
75  	}
76  
77  	BulkAccessor create() {
78  		Method[] getters = new Method[getterNames.length];
79  		Method[] setters = new Method[setterNames.length];
80  		findAccessors( targetBean, getterNames, setterNames, types, getters, setters );
81  
82  		Class beanClass;
83  		try {
84  			ClassFile classfile = make( getters, setters );
85  			ClassLoader loader = this.getClassLoader();
86  			if ( writeDirectory != null ) {
87  				FactoryHelper.writeFile( classfile, writeDirectory );
88  			}
89  
90  			beanClass = FactoryHelper.toClass( classfile, loader, getDomain() );
91  			return ( BulkAccessor ) this.newInstance( beanClass );
92  		}
93  		catch ( Exception e ) {
94  			throw new BulkAccessorException( e.getMessage(), e );
95  		}
96  	}
97  
98  	private ProtectionDomain getDomain() {
99  		Class cl;
100 		if ( this.targetBean != null ) {
101 			cl = this.targetBean;
102 		}
103 		else {
104 			cl = this.getClass();
105 		}
106 		return cl.getProtectionDomain();
107 	}
108 
109 	private ClassFile make(Method[] getters, Method[] setters) throws CannotCompileException {
110 		String className = targetBean.getName();
111 		// set the name of bulk accessor.
112 		className = className + "_$$_bulkaccess_" + counter++;
113 		if ( className.startsWith( "java." ) ) {
114 			className = "org.javassist.tmp." + className;
115 		}
116 
117 		ClassFile classfile = new ClassFile( false, className, BULKACESSOR_CLASS_NAME );
118 		classfile.setAccessFlags( AccessFlag.PUBLIC );
119 		addDefaultConstructor( classfile );
120 		addGetter( classfile, getters );
121 		addSetter( classfile, setters );
122 		return classfile;
123 	}
124 
125 	private ClassLoader getClassLoader() {
126 		if ( targetBean != null && targetBean.getName().equals( OBJECT_CLASS_NAME ) ) {
127 			return targetBean.getClassLoader();
128 		}
129 		else {
130 			return getClass().getClassLoader();
131 		}
132 	}

134 	private Object newInstance(Class type) throws Exception {
135 		BulkAccessor instance = ( BulkAccessor ) type.newInstance();
136 		instance.target = targetBean;
137 		int len = getterNames.length;
138 		instance.getters = new String[len];
139 		instance.setters = new String[len];
140 		instance.types = new Class[len];
141 		for ( int i = 0; i < len; i++ ) {
142 			instance.getters[i] = getterNames[i];
143 			instance.setters[i] = setterNames[i];
144 			instance.types[i] = types[i];
145 		}
146 
147 		return instance;
148 	}

150 	/**
151 	 * Declares a constructor that takes no parameter.
152 	 *
153 	 * @param classfile
154 	 * @throws CannotCompileException
155 	 */
156 	private void addDefaultConstructor(ClassFile classfile) throws CannotCompileException {
157 		ConstPool cp = classfile.getConstPool();
158 		String cons_desc = "()V";
159 		MethodInfo mi = new MethodInfo( cp, MethodInfo.nameInit, cons_desc );
160 
161 		Bytecode code = new Bytecode( cp, 0, 1 );
162 		// aload_0
163 		code.addAload( 0 );
164 		// invokespecial
165 		code.addInvokespecial( BulkAccessor.class.getName(), MethodInfo.nameInit, cons_desc );
166 		// return
167 		code.addOpcode( Opcode.RETURN );
168 
169 		mi.setCodeAttribute( code.toCodeAttribute() );
170 		mi.setAccessFlags( AccessFlag.PUBLIC );
171 		classfile.addMethod( mi );
172 	}

174 	private void addGetter(ClassFile classfile, final Method[] getters) throws CannotCompileException {
175 		ConstPool cp = classfile.getConstPool();
176 		int target_type_index = cp.addClassInfo( this.targetBean.getName() );
177 		String desc = GET_SETTER_DESC;
178 		MethodInfo mi = new MethodInfo( cp, GENERATED_GETTER_NAME, desc );
179 
180 		Bytecode code = new Bytecode( cp, 6, 4 );
181 		/* | this | bean | args | raw bean | */
182 		if ( getters.length >= 0 ) {
183 			// aload_1 // load bean
184 			code.addAload( 1 );
185 			// checkcast // cast bean
186 			code.addCheckcast( this.targetBean.getName() );
187 			// astore_3 // store bean
188 			code.addAstore( 3 );
189 			for ( int i = 0; i < getters.length; ++i ) {
190 				if ( getters[i] != null ) {
191 					Method getter = getters[i];
192 					// aload_2 // args
193 					code.addAload( 2 );
194 					// iconst_i // continue to aastore
195 					code.addIconst( i ); // growing stack is 1
196 					Class returnType = getter.getReturnType();
197 					int typeIndex = -1;
198 					if ( returnType.isPrimitive() ) {
199 						typeIndex = FactoryHelper.typeIndex( returnType );
200 						// new
201 						code.addNew( FactoryHelper.wrapperTypes[typeIndex] );
202 						// dup
203 						code.addOpcode( Opcode.DUP );
204 					}
205 
206 					// aload_3 // load the raw bean
207 					code.addAload( 3 );
208 					String getter_desc = RuntimeSupport.makeDescriptor( getter );
209 					String getterName = getter.getName();
210 					if ( this.targetBean.isInterface() ) {
211 						// invokeinterface
212 						code.addInvokeinterface( target_type_index, getterName, getter_desc, 1 );
213 					}
214 					else {
215 						// invokevirtual
216 						code.addInvokevirtual( target_type_index, getterName, getter_desc );
217 					}
218 
219 					if ( typeIndex >= 0 ) {       // is a primitive type
220 						// invokespecial
221 						code.addInvokespecial(
222 								FactoryHelper.wrapperTypes[typeIndex],
223 						        MethodInfo.nameInit,
224 						        FactoryHelper.wrapperDesc[typeIndex]
225 						);
226 					}
227 
228 					// aastore // args
229 					code.add( Opcode.AASTORE );
230 					code.growStack( -3 );
231 				}
232 			}
233 		}
234 		// return
235 		code.addOpcode( Opcode.RETURN );
236 
237 		mi.setCodeAttribute( code.toCodeAttribute() );
238 		mi.setAccessFlags( AccessFlag.PUBLIC );
239 		classfile.addMethod( mi );
240 	}

242 	private void addSetter(ClassFile classfile, final Method[] setters) throws CannotCompileException {
243 		ConstPool cp = classfile.getConstPool();
244 		int target_type_index = cp.addClassInfo( this.targetBean.getName() );
245 		String desc = GET_SETTER_DESC;
246 		MethodInfo mi = new MethodInfo( cp, GENERATED_SETTER_NAME, desc );
247 
248 		Bytecode code = new Bytecode( cp, 4, 6 );
249 		StackMapTable stackmap = null;
250 		/* | this | bean | args | i | raw bean | exception | */
251 		if ( setters.length > 0 ) {
252 			int start, end; // required to exception table
253 			// iconst_0 // i
254 			code.addIconst( 0 );
255 			// istore_3 // store i
256 			code.addIstore( 3 );
257 			// aload_1 // load the bean
258 			code.addAload( 1 );
259 			// checkcast // cast the bean into a raw bean
260 			code.addCheckcast( this.targetBean.getName() );
261 			// astore 4 // store the raw bean
262 			code.addAstore( 4 );
263 			/* current stack len = 0 */
264 			// start region to handling exception (BulkAccessorException)
265 			start = code.currentPc();
266 			int lastIndex = 0;
267 			for ( int i = 0; i < setters.length; ++i ) {
268 				if ( setters[i] != null ) {
269 					int diff = i - lastIndex;
270 					if ( diff > 0 ) {
271 						// iinc 3, 1
272 						code.addOpcode( Opcode.IINC );
273 						code.add( 3 );
274 						code.add( diff );
275 						lastIndex = i;
276 					}
277 				}
278 				/* current stack len = 0 */
279 				// aload 4 // load the raw bean
280 				code.addAload( 4 );
281 				// aload_2 // load the args
282 				code.addAload( 2 );
283 				// iconst_i
284 				code.addIconst( i );
285 				// aaload
286 				code.addOpcode( Opcode.AALOAD );
287 				// checkcast
288 				Class[] setterParamTypes = setters[i].getParameterTypes();
289 				Class setterParamType = setterParamTypes[0];
290 				if ( setterParamType.isPrimitive() ) {
291 					// checkcast (case of primitive type)
292 					// invokevirtual (case of primitive type)
293 					this.addUnwrapper( classfile, code, setterParamType );
294 				}
295 				else {
296 					// checkcast (case of reference type)
297 					code.addCheckcast( setterParamType.getName() );
298 				}
299 				/* current stack len = 2 */
300 				String rawSetterMethod_desc = RuntimeSupport.makeDescriptor( setters[i] );
301 				if ( !this.targetBean.isInterface() ) {
302 					// invokevirtual
303 					code.addInvokevirtual( target_type_index, setters[i].getName(), rawSetterMethod_desc );
304 				}
305 				else {
306 					// invokeinterface
307 					Class[] params = setters[i].getParameterTypes();
308 					int size;
309 					if ( params[0].equals( Double.TYPE ) || params[0].equals( Long.TYPE ) ) {
310 						size = 3;
311 					}
312 					else {
313 						size = 2;
314 					}
315 
316 					code.addInvokeinterface( target_type_index, setters[i].getName(), rawSetterMethod_desc, size );
317 				}
318 			}
319 
320 			// end region to handling exception (BulkAccessorException)
321 			end = code.currentPc();
322 			// return
323 			code.addOpcode( Opcode.RETURN );
324 			/* current stack len = 0 */
325 			// register in exception table
326 			int throwableType_index = cp.addClassInfo( THROWABLE_CLASS_NAME );
327 			int handler_pc = code.currentPc();
328 			code.addExceptionHandler( start, end, handler_pc, throwableType_index );
329 			// astore 5 // store exception
330 			code.addAstore( 5 );
331 			// new // BulkAccessorException
332 			code.addNew( BULKEXCEPTION_CLASS_NAME );
333 			// dup
334 			code.addOpcode( Opcode.DUP );
335 			// aload 5 // load exception
336 			code.addAload( 5 );
337 			// iload_3 // i
338 			code.addIload( 3 );
339 			// invokespecial // BulkAccessorException.<init>
340 			String cons_desc = "(Ljava/lang/Throwable;I)V";
341 			code.addInvokespecial( BULKEXCEPTION_CLASS_NAME, MethodInfo.nameInit, cons_desc );
342 			// athrow
343 			code.addOpcode( Opcode.ATHROW );
344 			StackMapTable.Writer writer = new StackMapTable.Writer(32);
345 			int[] localTags = { StackMapTable.OBJECT, StackMapTable.OBJECT, StackMapTable.OBJECT, StackMapTable.INTEGER };
346 			int[] localData = { cp.getThisClassInfo(), cp.addClassInfo("java/lang/Object"),
347                         	cp.addClassInfo("[Ljava/lang/Object;"), 0};
348 			int[] stackTags = { StackMapTable.OBJECT };
349 			int[] stackData = { throwableType_index };
350 			writer.fullFrame(handler_pc, localTags, localData, stackTags, stackData);
351 			stackmap = writer.toStackMapTable(cp);
352 		}
353 		else {
354 			// return
355 			code.addOpcode( Opcode.RETURN );
356 		}
357 		CodeAttribute ca = code.toCodeAttribute();
358 		if (stackmap != null) {
359 			ca.setAttribute(stackmap);
360 		}
361 		mi.setCodeAttribute( ca );
362 		mi.setAccessFlags( AccessFlag.PUBLIC );
363 		classfile.addMethod( mi );
364 	}

366 	private void addUnwrapper(
367 			ClassFile classfile,
368 	        Bytecode code,
369 	        Class type) {
370 		int index = FactoryHelper.typeIndex( type );
371 		String wrapperType = FactoryHelper.wrapperTypes[index];
372 		// checkcast
373 		code.addCheckcast( wrapperType );
374 		// invokevirtual
375 		code.addInvokevirtual( wrapperType, FactoryHelper.unwarpMethods[index], FactoryHelper.unwrapDesc[index] );
376 	}

378 	private static void findAccessors(
379 			Class clazz,
380 	        String[] getterNames,
381 	        String[] setterNames,
382 	        Class[] types,
383 	        Method[] getters,
384 	        Method[] setters) {
385 		int length = types.length;
386 		if ( setterNames.length != length || getterNames.length != length ) {
387 			throw new BulkAccessorException( "bad number of accessors" );
388 		}
389 
390 		Class[] getParam = new Class[0];
391 		Class[] setParam = new Class[1];
392 		for ( int i = 0; i < length; i++ ) {
393 			if ( getterNames[i] != null ) {
394 				Method getter = findAccessor( clazz, getterNames[i], getParam, i );
395 				if ( getter.getReturnType() != types[i] ) {
396 					throw new BulkAccessorException( "wrong return type: " + getterNames[i], i );
397 				}
398 
399 				getters[i] = getter;
400 			}
401 
402 			if ( setterNames[i] != null ) {
403 				setParam[0] = types[i];
404 				setters[i] = findAccessor( clazz, setterNames[i], setParam, i );
405 			}
406 		}
407 	}
408 
409 	private static Method findAccessor(
410 			Class clazz,
411 	        String name,
412 	        Class[] params,
413 	        int index) throws BulkAccessorException {
414 		try {
415 			Method method = clazz.getDeclaredMethod( name, params );
416 			if ( Modifier.isPrivate( method.getModifiers() ) ) {
417 				throw new BulkAccessorException( "private property", index );
418 			}
419 
420 			return method;
421 		}
422 		catch ( NoSuchMethodException e ) {
423 			throw new BulkAccessorException( "cannot find an accessor", index );
424 		}
425 	}
