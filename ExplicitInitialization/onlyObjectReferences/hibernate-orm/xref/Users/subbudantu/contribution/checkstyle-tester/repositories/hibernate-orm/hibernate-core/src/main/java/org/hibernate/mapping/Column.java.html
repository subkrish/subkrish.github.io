85  	public void setName(String name) {
86  		if (
87  			StringHelper.isNotEmpty( name ) &&
88  			Dialect.QUOTE.indexOf( name.charAt(0) ) > -1 //TODO: deprecated, remove eventually
89  		) {
90  			quoted=true;
91  			this.name=name.substring( 1, name.length()-1 );
92  		}
93  		else {
94  			this.name = name;
95  		}
96  	}

111 	@Override
112 	public String getAlias(Dialect dialect) {
113 		final int lastLetter = StringHelper.lastIndexOfLetter( name );
114 		final String suffix = Integer.toString(uniqueInteger) + '_';
115 
116 		String alias = name;
117 		if ( lastLetter == -1 ) {
118 			alias = "column";
119 		}
120 		else if ( name.length() > lastLetter + 1 ) {
121 			alias = name.substring( 0, lastLetter + 1 );
122 		}
123 
124 		boolean useRawName = name.length() + suffix.length() <= dialect.getMaxAliasLength()
125 				&& !quoted && !name.toLowerCase().equals( "rowid" );
126 		if ( !useRawName ) {
127 			if ( suffix.length() >= dialect.getMaxAliasLength() ) {
128 				throw new MappingException( String.format(
129 						"Unique suffix [%s] length must be less than maximum [%d]",
130 						suffix, dialect.getMaxAliasLength() ) );
131 			}
132 			if ( alias.length() + suffix.length() > dialect.getMaxAliasLength() ) {
133 				alias = alias.substring( 0, dialect.getMaxAliasLength() - suffix.length() );
134 			}
135 		}
136 		return alias + suffix;
137 	}

177 		if (null == column) return false;

182 			name.equalsIgnoreCase(column.name);

185     public int getSqlTypeCode(Mapping mapping) throws MappingException {
186         org.hibernate.type.Type type = getValue().getType();
187         try {
188             int sqlTypeCode = type.sqlTypes( mapping )[getTypeIndex()];
189             if ( getSqlTypeCode() != null && getSqlTypeCode() != sqlTypeCode ) {
190                 throw new MappingException( "SQLType code's does not match. mapped as " + sqlTypeCode + " but is " + getSqlTypeCode() );
191             }
192             return sqlTypeCode;
193         }
194         catch ( Exception e ) {
195             throw new MappingException(
196                     "Could not determine type for column " +
197                             name +
198                             " of type " +
199                             type.getClass().getName() +
200                             ": " +
201                             e.getClass().getName(),
202                     e
203             );
204         }
205     }

224     public String getSqlType(Dialect dialect, Mapping mapping) throws HibernateException {
225         if ( sqlType == null ) {
226             sqlType = dialect.getTypeName( getSqlTypeCode( mapping ), getLength(), getPrecision(), getScale() );
227         }
228         return sqlType;
229     }

343 	 * Shallow copy, the value is not copied
344 	 */
345 	@Override
346 	public Column clone() {
347 		Column copy = new Column();
348 		copy.setLength( length );
349 		copy.setScale( scale );
350 		copy.setValue( value );
351 		copy.setTypeIndex( typeIndex );
352 		copy.setName( getQuotedName() );
353 		copy.setNullable( nullable );
354 		copy.setPrecision( precision );
355 		copy.setUnique( unique );
356 		copy.setSqlType( sqlType );
357 		copy.setSqlTypeCode( sqlTypeCode );
358 		copy.uniqueInteger = uniqueInteger; //usually useless
359 		copy.setCheckConstraint( checkConstraint );
360 		copy.setComment( comment );
361 		copy.setDefaultValue( defaultValue );
362 		copy.setCustomRead( customRead );
363 		copy.setCustomWrite( customWrite );
364 		return copy;
365 	}
