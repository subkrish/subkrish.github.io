24  package org.hibernate.metamodel.relational;
25  
26  /**
27   * Models a value within a {@link ValueContainer}.  This will generally be either a {@link Column column} or a
28   * {@link DerivedValue derived value}, but we also allow the notion of {@link Tuple} at this level
29   *
30   * @author Steve Ebersole
31   */
32  public interface Value {
33  	/**
34  	 * Retrieve the table that owns this value.
35  	 *
36  	 * @return The owning table.
37  	 */
38  	public TableSpecification getTable();
39  
40  	/**
41  	 * Obtain the string representation of this value usable in log statements.
42  	 *
43  	 * @return The loggable representation
44  	 */
45  	public String toLoggableString();
46  
47  	/**
48  	 * Used to track JDBC type usage throughout a series of potential recursive calls to component
49  	 * values since we do not know ahead of time which values correspond to which indexes of the
50  	 * jdbc type array.
51  	 */
52  	public static class JdbcCodes {
53  		private final int[] typeCodes;
54  		private int index = 0;
55  
56  		public JdbcCodes(int[] typeCodes) {
57  			this.typeCodes = typeCodes;
58  		}
59  
60  		public int nextJdbcCde() {
61  			return typeCodes[index++];
62  		}
63  
64  		public int getIndex() {
65  			return index;
66  		}
67  	}
68  
69  	/**
70  	 * Validate the value against the incoming JDBC type code array, both in terms of number of types
71  	 * and compatibility of types.
72  	 *
73  	 * @param typeCodes The type codes.
74  	 *
75  	 * @throws org.hibernate.metamodel.ValidationException if validaton fails.
76  	 */
77  	public void validateJdbcTypes(JdbcCodes typeCodes);
78  
79  }
