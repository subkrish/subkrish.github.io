53  	@Test
54  	public void testShouldIndexFieldInPath() throws Exception {
55  		List<Human> result = search( session, "parents.parents.parents.name", "Philippa" );
56  
57  		Assert.assertEquals( 1, result.size() );
58  		Assert.assertEquals(
59  				"Should be able to index a field in path regarding the depth",
60  				"John of England", result.get( 0 ).getFullname() );
61  	}
62  
63  	@Test
64  	public void testIndexFieldIfInsideDepth() throws Exception {
65  		List<Human> result = search( session, "parents.parents.name", "Empress" );
66  
67  		Assert.assertEquals( 1, result.size() );
68  		Assert.assertEquals(
69  				"Should be able to index field inside depth and in path",
70  				"John of England", result.get( 0 ).getFullname() );
71  	}

73  	@Test
74  	public void testShouldNotIndexFieldOutsidePathAndOverDepth() throws Exception {
75  		try {
76  			search( session, "parents.parents.parents.surname", "de Montfort" );
77  			fail( "Shoudl not index a field if not in path and over the depth threshold" );
78  		}
79  		catch (SearchException e) {
80  		}

83  	@Test
84  	public void testShouldIndexFieldNotInPathButInsideDepthThreshold() throws Exception {
85  		List<Human> result = search( session, "parents.parents.surname", "de Châtellerault" );
86  
87  		Assert.assertEquals( 1, result.size() );
88  		Assert.assertEquals( "Should be able to index a field if is inside the depth threshold even if not in path",
89  				"John of England", result.get( 0 ).getFullname() );

145 	private List<Human> search(Session s, String field, String value) {
146 		FullTextSession session = Search.getFullTextSession( s );
147 		QueryBuilder queryBuilder = session.getSearchFactory().buildQueryBuilder().forEntity( Human.class ).get();
148 		Query query = queryBuilder.keyword().onField( field ).matching( value ).createQuery();
149 		@SuppressWarnings("unchecked")
150 		List<Human> result = session.createFullTextQuery( query ).list();
151 		return result;
152 	}
153 
154 	private void deleteAll(Session s, Class<?>... classes) {
155 		Transaction tx = s.beginTransaction();
156 		for ( Class<?> each : classes ) {
157 			List<?> list = s.createCriteria( each ).list();
158 			for ( Object object : list ) {
159 				s.delete( object );
160 			}
161 		}
162 		tx.commit();
163 	}
