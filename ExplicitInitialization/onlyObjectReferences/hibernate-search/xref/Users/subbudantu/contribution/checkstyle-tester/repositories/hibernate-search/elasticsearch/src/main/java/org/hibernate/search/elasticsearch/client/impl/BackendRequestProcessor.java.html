34  /**
35   * Executes single or multiple {@link BackendRequest}s against the Elasticsearch server. When processing multiple
36   * requests, bulk requests will be formed and executed as far as possible.
37   * <p>
38   * Requests can be processed synchronously or asynchronously. In the latter case, incoming requests are added to a queue
39   * via {@link AsyncBackendRequestProcessor} from where a worker runnable will process them in bulks.
40   *
41   * @author Gunnar Morling
42   */
43  public class BackendRequestProcessor implements Service, Startable, Stoppable {
44  
45  	private static final Log LOG = LoggerFactory.make( Log.class );
46  
47  	/**
48  	 * Maximum number of requests sent in a single bulk. Could be made an option if needed.
49  	 */
50  	private static final int MAX_BULK_SIZE = 250;
51  
52  	private final AsyncBackendRequestProcessor asyncProcessor;
53  	private ErrorHandler errorHandler;
54  	private ServiceManager serviceManager;
55  	private JestClient jestClient;
56  
57  	public BackendRequestProcessor() {
58  		asyncProcessor = new AsyncBackendRequestProcessor();
59  	}
60  
61  	@Override
62  	public void start(Properties properties, BuildContext context) {
63  		this.errorHandler = context.getErrorHandler();
64  		this.serviceManager = context.getServiceManager();
65  		this.jestClient = serviceManager.requestService( JestClient.class );
66  	}
67  
68  	@Override
69  	public void stop() {
70  		awaitAsyncProcessingCompletion();
71  		asyncProcessor.shutdown();
72  		serviceManager.releaseService( JestClient.class );
73  	}
74  
75  	public void executeSync(Iterable<BackendRequest<?>> requests) {
76  		doExecute( requests );
77  	}
78  
79  	public void executeAsync(BackendRequest<?> request) {
80  		asyncProcessor.submitRequest( request );
81  	}
82  
83  	/**
84  	 * Blocks until the queue of requests scheduled for asynchronous processing has been fully processed.
85  	 * N.B. if more work is added to the queue in the meantime, this might delay the wait.
86  	 */
87  	public void awaitAsyncProcessingCompletion() {
88  		asyncProcessor.awaitCompletion();
89  	}
90  
91  	/**
92  	 * Groups the given work list into executable bulks and executes them. For each bulk, the error handler - if
93  	 * registered - will be invoked with the items of that bulk.
94  	 */
95  	private void doExecute(Iterable<BackendRequest<?>> requests) {
96  		ExecutableRequest nextBulk = null;
97  		Set<String> indexesNeedingRefresh = new HashSet<>();
98  
99  		for ( ExecutableRequest backendRequestGroup : createRequestGroups( requests, true ) ) {
100 			nextBulk = backendRequestGroup;
101 
102 			if ( LOG.isTraceEnabled() ) {
103 				LOG.tracef( "Processing bulk of %s items on index(es) %s", (Integer) nextBulk.getSize(), nextBulk.getTouchedIndexes() );
104 			}
105 
106 			nextBulk.execute();
107 			indexesNeedingRefresh.addAll( backendRequestGroup.getIndexesNeedingRefresh() );
108 		}
109 
110 		refresh( indexesNeedingRefresh );
111 	}
112 
113 	/**
114 	 * Organizes the given work list into {@link ExecutableRequest}s to be executed.
115 	 */
116 	private List<ExecutableRequest> createRequestGroups(Iterable<BackendRequest<?>> requests, boolean refreshAtEnd) {
117 		List<ExecutableRequest> groups = new ArrayList<>();
118 		BulkRequestBuilder bulkBuilder = new BulkRequestBuilder();
119 
120 		for ( BackendRequest<?> request : requests ) {
121 			boolean currentRequestBulkable = request.getAction() instanceof BulkableAction;
122 			boolean currentBulkNeedsFinishing = ( !bulkBuilder.canAddMore() || !currentRequestBulkable ) && !bulkBuilder.isEmpty();
123 
124 			// finish up current bulk
125 			if ( currentBulkNeedsFinishing ) {
126 				groups.add( bulkBuilder.build( false ) );
127 				bulkBuilder = new BulkRequestBuilder();
128 			}
129 
130 			// either add to current bulk...
131 			if ( currentRequestBulkable ) {
132 				bulkBuilder.add( request );
133 			}
134 			// ...  or add single request for non-bulkable request
135 			else {
136 				groups.add( new SingleRequest( jestClient, errorHandler, request ) );
137 			}
138 		}
139 
140 		// finish up last bulk
141 		if ( !bulkBuilder.isEmpty() ) {
142 			groups.add( bulkBuilder.build( refreshAtEnd ) );
143 		}
144 
145 		return groups;
146 	}
147 
148 	/**
149 	 * Performs an explicit refresh of the given index(es).
150 	 */
151 	private void refresh(Set<String> indexesNeedingRefresh) {
152 		if ( indexesNeedingRefresh.isEmpty() ) {
153 			return;
154 		}
155 
156 		if ( LOG.isTraceEnabled() ) {
157 			LOG.tracef( "Refreshing index(es) %s", indexesNeedingRefresh );
158 		}
159 
160 		Refresh.Builder refreshBuilder = new Refresh.Builder();
161 
162 		for ( String index : indexesNeedingRefresh ) {
163 			refreshBuilder.addIndex( index );
164 		}
165 
166 		try {
167 			jestClient.executeRequest( refreshBuilder.build() );
168 		}
169 		catch (SearchException e) {
170 			errorHandler.handleException( "Refresh failed", e );
171 		}
172 	}

174 	/**
175 	 * Processes requests asynchronously.
176 	 * <p>
177 	 * Incoming messages are submitted to a queue. A worker runnable takes all messages in the queue at a given time and
178 	 * processes them as a bulk as far as possible. The worker is started upon first message arrival after the queue has
179 	 * been emptied and remains active until the queue is empty again.
180 	 *
181 	 * @author Gunnar Morling
182 	 */
183 	private class AsyncBackendRequestProcessor {
184 
185 		private final ScheduledExecutorService scheduler;
186 		private final MultiWriteDrainableLinkedList<BackendRequest<?>> asyncRequestQueue;
187 		private final AtomicBoolean asyncWorkerWasStarted;
188 
189 		private volatile CountDownLatch lastAsyncWorkLatch;
190 
191 		private AsyncBackendRequestProcessor() {
192 			asyncRequestQueue = new MultiWriteDrainableLinkedList<>();
193 			scheduler = Executors.newScheduledThreadPool( "Elasticsearch AsyncBackendRequestProcessor" );
194 			asyncWorkerWasStarted = new AtomicBoolean( false );
195 		}
196 
197 		public void submitRequest(BackendRequest<?> request) {
198 			asyncRequestQueue.add( request );
199 
200 			// Set up worker if needed
201 			if ( !asyncWorkerWasStarted.get() ) {
202 				synchronized ( AsyncBackendRequestProcessor.this ) {
203 					if ( asyncWorkerWasStarted.compareAndSet( false, true ) ) {
204 						try {
205 							RequestProcessingRunnable runnable = new RequestProcessingRunnable( this );
206 							scheduler.schedule( runnable, 100, TimeUnit.MILLISECONDS );
207 							//only assign this when the job was successfully scheduled:
208 							lastAsyncWorkLatch = runnable.latch;
209 						}
210 						catch (Exception e) {
211 							// Make sure a failure to setup the worker doesn't leave other threads waiting indefinitely:
212 							asyncWorkerWasStarted.set( false );
213 							final CountDownLatch latch = lastAsyncWorkLatch;
214 							if ( latch != null ) {
215 								latch.countDown();
216 							}
217 							throw e;
218 						}
219 					}
220 				}
221 			}
222 		}
223 
224 		public void awaitCompletion() {
225 			final CountDownLatch localLatch = lastAsyncWorkLatch;
226 			if ( localLatch != null ) {
227 				try {
228 					localLatch.await();
229 				}
230 				catch (InterruptedException e) {
231 					Thread.currentThread().interrupt();
232 					throw LOG.interruptedWhileWaitingForRequestCompletion( e );
233 				}
234 			}
235 		}
236 
237 		public void shutdown() {
238 			scheduler.shutdown();
239 			try {
240 				scheduler.awaitTermination( Long.MAX_VALUE, TimeUnit.SECONDS );
241 			}
242 			catch (InterruptedException e) {
243 				LOG.interruptedWhileWaitingForIndexActivity( e );
244 			}
245 			finally {
246 				final CountDownLatch localLatch = lastAsyncWorkLatch;
247 				if ( localLatch != null ) {
248 					//It's possible that a task was successfully scheduled but had no chance to run,
249 					//so we need to release waiting threads:
250 					localLatch.countDown();
251 				}
252 			}
253 		}
254 	}
255 
256 	/**
257 	 * Takes requests from the queue and processes them.
258 	 */
259 	private class RequestProcessingRunnable implements Runnable {
260 
261 		private final AsyncBackendRequestProcessor asyncProcessor;
262 		private final CountDownLatch latch = new CountDownLatch( 1 );
263 
264 		public RequestProcessingRunnable(AsyncBackendRequestProcessor asyncProcessor) {
265 			this.asyncProcessor = asyncProcessor;
266 		}
267 
268 		@Override
269 		public void run() {
270 			try {
271 				processAsyncWork();
272 			}
273 			finally {
274 				latch.countDown();
275 			}
276 		}
277 
278 		private void processAsyncWork() {
279 			Set<String> indexesNeedingFlush = new HashSet<>();
280 			synchronized ( asyncProcessor ) {
281 				while ( true ) {
282 					Iterable<BackendRequest<?>> requests = asyncProcessor.asyncRequestQueue.drainToDetachedIterable();
283 					if ( requests == null ) {
284 						// Allow other async processors to be setup already as we're on our way to termination:
285 						asyncProcessor.asyncWorkerWasStarted.set( false );
286 						// Nothing more to do, flush and terminate:
287 						refresh( indexesNeedingFlush );
288 						return;
289 					}
290 					for ( ExecutableRequest backendRequestGroup : createRequestGroups( requests, false ) ) {
291 						backendRequestGroup.execute();
292 						indexesNeedingFlush.addAll( backendRequestGroup.getIndexesNeedingRefresh() );
293 					}
294 				}
295 			}
296 		}
297 	}
298 
299 	private class BulkRequestBuilder {
300 
301 		private final List<BackendRequest<?>> bulk = new ArrayList<>();
302 		private final Set<String> indexNames = new HashSet<>();
303 		private final Set<String> indexesNeedingRefresh = new HashSet<>();
304 		private int size = 0;
305 
306 		private void add(BackendRequest<?> request) {
307 			bulk.add( request );
308 			indexNames.add( request.getIndexName() );
309 			if ( request.needsRefreshAfterWrite() ) {
310 				indexesNeedingRefresh.add( request.getIndexName() );
311 			}
312 			size++;
313 		}
314 		private boolean canAddMore() {
315 			return size < MAX_BULK_SIZE;
316 		}
317 
318 		private boolean isEmpty() {
319 			return size == 0;
320 		}
321 
322 		private ExecutableRequest build(boolean refresh) {
323 			if ( size > 1 ) {
324 				return new BulkRequest( jestClient, errorHandler, bulk, indexNames, indexesNeedingRefresh, refresh );
325 			}
326 			else {
327 				return new SingleRequest( jestClient, errorHandler, bulk.iterator().next() );
328 			}
329 		}
330 	}
