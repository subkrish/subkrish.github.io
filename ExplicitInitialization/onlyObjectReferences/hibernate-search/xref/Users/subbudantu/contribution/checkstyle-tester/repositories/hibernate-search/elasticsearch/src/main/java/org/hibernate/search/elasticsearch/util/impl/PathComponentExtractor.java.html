7   package org.hibernate.search.elasticsearch.util.impl;
8   
9   import org.hibernate.search.exception.AssertionFailure;
10  
11  /**
12   * A stateful helper to extract path components from a string, one
13   * component at a time.
14   *
15   * @author Yoann Rodiere
16   */
17  public class PathComponentExtractor implements Cloneable {
18  
19  	private static final String PATH_COMPONENT_SEPARATOR = ".";
20  
21  	/*
22  	 * Non-final for cloning only.
23  	 */
24  	private StringBuilder path = new StringBuilder();
25  	private int currentIndexInPath = 0;
26  
27  	/**
28  	 * Append a string to the path for later consumption through {@link #next(ConsumptionLimit)}.
29  	 * @param pathPart A string that may or may not include path component
30  	 * separators (dots).
31  	 */
32  	public void append(String pathPart) {
33  		path.append( pathPart );
34  	}
35  
36  	/**
37  	 * Append to the path the part of {@code otherPath} that is relative to the current path.
38  	 * <p>In other words, replace the current path with {@code otherPath} provided {@code otherPath}
39  	 * denotes a child element of the current path, while preserving the memory of the previously
40  	 * consumed path components.
41  	 * @param otherPath A path that must start with the current path.
42  	 * @throws ParentPathMismatchException If {@code otherPath} is not contained with the current path.
43  	 */
44  	public void appendRelativePart(String otherPath) throws ParentPathMismatchException {
45  		String pathAsString = path.toString();
46  		if ( !otherPath.startsWith( pathAsString ) ) {
47  			throw new ParentPathMismatchException( pathAsString, otherPath );
48  		}
49  
50  		path.append( otherPath, path.length(), otherPath.length() );
51  	}
52  
53  	public enum ConsumptionLimit {
54  		/**
55  		 * Consume up to the second-but-last path component, i.e. up to the last path separator
56  		 * in the path string.
57  		 * <p>Don't consume anything if there is no path separator in the path string.
58  		 */
59  		SECOND_BUT_LAST,
60  		/**
61  		 * Consume up to the last path component, i.e. up to the end of the path string.
62  		 */
63  		LAST;
64  	}
65  
66  	/**
67  	 * Consume one more component in the current path (provided it's not beyond the given limit) and returns this component.
68  	 *
69  	 *<p>If this method reaches a incompletely qualified path component, i.e one that is not
70  	 *followed by a dot but by the end of the path, it will return it only if {@code consumptionLimit}
71  	 *is {@link ConsumptionLimit#LAST}.
72  	 *
73  	 * @param consumeLimit The consumption limit, i.e. the definition of the last component
74  	 * to consume in the path.
75  	 * @return The next path component, or {@code null} if the consumption limit has been reached.
76  	 */
77  	public String next(ConsumptionLimit consumeLimit) {
78  		int nextSeparatorIndex = path.indexOf( PATH_COMPONENT_SEPARATOR, currentIndexInPath );
79  		if ( nextSeparatorIndex >= 0 ) {
80  			String childName = path.substring( currentIndexInPath, nextSeparatorIndex );
81  			currentIndexInPath = nextSeparatorIndex + 1 /* skip the dot */;
82  			return childName;
83  		}
84  		else if ( ConsumptionLimit.LAST.equals( consumeLimit ) && currentIndexInPath < path.length() ) {
85  			String lastComponent = path.substring( currentIndexInPath );
86  			currentIndexInPath = path.length();
87  			return lastComponent;
88  		}
89  		else {
90  			return null;
91  		}
92  	}
93  
94  	/**
95  	 * Consume all components in the current path up to the given limit.
96  	 *
97  	 *<p>This is equivalent to calling {@code next(consumeLimit)} repeatedly until {@code null} is returned.
98  	 *
99  	 * @param consumeLimit The consumption limit, i.e. the definition of the last component
100 	 * to consume in the path.
101 	 */
102 	public void flushTo(ConsumptionLimit consumeLimit) {
103 		if ( ConsumptionLimit.LAST.equals( consumeLimit ) ) {
104 			currentIndexInPath = path.length();
105 		}
106 		else {
107 			int nextSeparatorIndex = path.lastIndexOf( PATH_COMPONENT_SEPARATOR );
108 			if ( nextSeparatorIndex >= 0 ) {
109 				currentIndexInPath = nextSeparatorIndex + 1;
110 			}
111 		}
112 	}
113 
114 	public void reset() {
115 		path.delete( 0, path.length() );
116 		currentIndexInPath = 0;
117 	}
118 
119 	/**
120 	 * @return The absolute path of the last non-null path component returned by {@link #next(ConsumptionLimit)},
121 	 * or {@code null} if {@link #next(ConsumptionLimit)} hasn't returned such a component yet.
122 	 */
123 	public String getLastComponentAbsolutePath() {
124 		return currentIndexInPath == 0 ? null : path.substring( 0, currentIndexInPath - 1 );
125 	}
126 
127 	@Override
128 	public PathComponentExtractor clone() {
129 		try {
130 			PathComponentExtractor clone = (PathComponentExtractor) super.clone();
131 			clone.path = new StringBuilder( path );
132 			return clone;
133 		}
134 		catch (CloneNotSupportedException e) {
135 			throw new AssertionFailure( "Unexpected clone() failure", e );
136 		}
137 	}
138 
139 	@Override
140 	public String toString() {
141 		return new StringBuilder( path ).insert( currentIndexInPath, "[]" ).toString();
142 	}
143 
144 }
