96  	private void testDetectConflictDuringSchemaGeneration(Class<?> entityClass) {
97  		Assume.assumeTrue( "The strategy " + strategy + " does not involve schema generation."
98  				+ " No point running this test.",
99  				generatesSchema( strategy ) );
100 
101 		thrown.expect(
102 				isException( SearchException.class )
103 						.withMessage( COMPOSITE_CONCRETE_CONFLICT_MESSAGE_ID )
104 						.withMessage( CONFLICTING_FIELD_NAME + "'" )
105 						.withMessage( entityClass.getName() )
106 				.build()
107 		);
108 
109 		elasticSearchClient.index( entityClass ).registerForCleanup();
110 		init( strategy, entityClass );
111 	}

125 	private void testDetectConflictDuringIndexing(Class<?> entityClass) throws Exception {
126 		Assume.assumeFalse( "The strategy " + strategy + " involves schema generation,"
127 				+ " which means conflicts prevent search factory initialization"
128 				+ " and thus prevent indexing. No point running this test.",
129 				generatesSchema( strategy ) );
130 
131 		elasticSearchClient.index( entityClass ).deleteAndCreate();
132 		init( strategy, entityClass );
133 
134 		thrown.expect(
135 				isException( AssertionFailure.class )
136 				.causedBy( HibernateException.class )
137 				.causedBy( SearchException.class )
138 						.withMessage( COMPOSITE_CONCRETE_CONFLICT_MESSAGE_ID )
139 						.withMessage( CONFLICTING_FIELD_NAME + "'" )
140 						.withMessage( entityClass.getName() )
141 				.build()
142 		);
143 
144 		Object newEntity = entityClass.newInstance();
145 		try ( Session session = getTestResourceManager().openSession() ) {
146 			Transaction tx = session.beginTransaction();
147 			session.save( newEntity );
148 			tx.commit();
149 		}
150 	}

154 	public void detectIndexedEmbeddedPrefixBypass_schemaGeneration() {
155 		Assume.assumeTrue( "The strategy " + strategy + " does not involve schema generation."
156 				+ " No point running this test.",
157 				generatesSchema( strategy ) );
158 
159 		Class<?> entityClass = BypassingIndexedEmbeddedPrefixEntity.class;
160 
161 		thrown.expect(
162 				isException( SearchException.class )
163 						.withMessage( INDEXED_EMBEDDED_BYPASS_MESSAGE_ID )

166 						.withMessage( entityClass.getName() )
167 				.build()
168 		);
169 
170 		elasticSearchClient.index( entityClass ).registerForCleanup();
171 		init( strategy, entityClass );
172 	}

174 	@Test
175 	@TestForIssue(jiraKey = "HSEARCH-2488")
176 	public void detectIndexedEmbeddedPrefixBypass_indexing() throws Exception {
177 		Assume.assumeFalse( "The strategy " + strategy + " involves schema generation,"
178 				+ " which means @IndexedEmbedded.prefix bypasses prevent search factory initialization"
179 				+ " and thus prevent indexing. No point running this test.",
180 				generatesSchema( strategy ) );
181 
182 		Class<?> entityClass = BypassingIndexedEmbeddedPrefixEntity.class;
183 
184 		elasticSearchClient.index( entityClass ).deleteAndCreate();
185 		init( strategy, entityClass );
186 
187 		thrown.expect(
188 				isException( AssertionFailure.class )
189 				.causedBy( HibernateException.class )
190 				.causedBy( SearchException.class )
191 						.withMessage( INDEXED_EMBEDDED_BYPASS_MESSAGE_ID )
192 						.withMessage( "'" + BYPASSING_FIELD_NAME + "'" )
193 						.withMessage( "'" + BYPASSING_FIELD_INDEXED_EMBEDDED_PREFIX + "'" )
194 						.withMessage( entityClass.getName() )
195 				.build()
196 		);
197 
198 		Object newEntity = entityClass.newInstance();
199 		try ( Session session = getTestResourceManager().openSession() ) {
200 			Transaction tx = session.beginTransaction();
201 			session.save( newEntity );
202 			tx.commit();
203 		}
204 	}

206 	private boolean generatesSchema(IndexSchemaManagementStrategy strategy) {
207 		return !IndexSchemaManagementStrategy.NONE.equals( strategy );
208 	}

210 	private void init(IndexSchemaManagementStrategy strategy, Class<?> ... entityClasses) {
211 		Map<String, Object> settings = new HashMap<>();
212 		settings.put(
213 				"hibernate.search.default." + ElasticsearchEnvironment.INDEX_SCHEMA_MANAGEMENT_STRATEGY,
214 				strategy.name()
215 		);
216 
217 		init( new ImmutableTestConfiguration( settings, entityClasses ) );
218 	}

232 	/**
233 	 * A class for which Hibernate Search will first handle the mapping for the "concrete"
234 	 * (non-composite) field, and then the mapping for the composite field.
235 	 */
236 	@Entity
237 	@Indexed
238 	static class CompositeOnConcreteEntity {
239 		@DocumentId
240 		@Id
241 		@GeneratedValue
242 		Long id;
243 
244 		@IndexedEmbedded(prefix = CONFLICTING_FIELD_NAME + ".")
245 		@Field(name = CONFLICTING_FIELD_NAME, bridge = @FieldBridge(impl = SimpleToStringBridge.class))
246 		@Embedded
247 		EmbeddedTypeWithNonConflictingField embedded =
248 				new EmbeddedTypeWithNonConflictingField();
249 
250 		public static class SimpleToStringBridge implements MetadataProvidingFieldBridge {
251 
252 			@Override
253 			public void set(String name, Object value, Document document, LuceneOptions luceneOptions) {
254 				luceneOptions.addFieldToDocument( name, String.valueOf( value ), document );
255 			}
256 
257 			@Override
258 			public void configureFieldMetadata(String name, FieldMetadataBuilder builder) {
259 				builder.field( name, FieldType.STRING );
260 			}
261 		}
262 	}

278 				new EmbeddedTypeWithNonConflictingField();

280 		// Hack: methods are handled first, so take advantage of that.
281 		@IndexedEmbedded(prefix = CONFLICTING_FIELD_NAME + ".")
282 		public EmbeddedTypeWithNonConflictingField getEmbedded() {
283 			return embedded;
284 		}

287 		@Embedded

308 		private String field = "fieldValue";

311 	public static class BypassingIndexedEmbeddedPrefixFieldBridge implements MetadataProvidingFieldBridge {
312 		@Override
313 		public void configureFieldMetadata(String name, FieldMetadataBuilder builder) {
314 			builder.field( BYPASSING_FIELD_NAME, FieldType.STRING );
315 		}
316 
317 		@Override
318 		public void set(String name, Object value, Document document, LuceneOptions luceneOptions) {
319 			luceneOptions.addFieldToDocument( BYPASSING_FIELD_NAME, (String) value, document );
320 		}
