7   package org.hibernate.search.backend.impl.lucene;
8   
9   import org.hibernate.search.backend.IndexingMonitor;
10  import org.hibernate.search.backend.LuceneWork;
11  import org.hibernate.search.util.logging.impl.Log;
12  import org.hibernate.search.util.logging.impl.LoggerFactory;
13  
14  import java.io.PrintWriter;
15  import java.io.StringWriter;
16  import java.util.List;
17  import java.util.concurrent.CountDownLatch;
18  import java.util.concurrent.TimeUnit;
19  import java.util.concurrent.locks.LockSupport;
20  
21  /**
22   * Processes changesets in batches, maintaining sync guarantees.
23   *<p>
24   * Multiple threads produce one or more {@link org.hibernate.search.backend.LuceneWork}
25   * by calling {@link #submit(java.util.List, org.hibernate.search.backend.IndexingMonitor)},
26   * and get blocked until their changes are applied to the index;</p>
27   * The {@link org.hibernate.search.backend.impl.lucene.SyncWorkProcessor.Consumer} thread will
28   * coalesce changes from multiple threads and apply them in the index, releasing the waiting threads
29   * at the end.
30   * <p>
31   * In the absence of work to be applied, the Consumer thread is parked to avoid busy waiting.</p>
32   *
33   * @author gustavonalle
34   */
35  final class SyncWorkProcessor implements WorkProcessor {
36  
37  	private static final Log log = LoggerFactory.make();
38  
39  	private final MultiWriteDrainableLinkedList<Changeset> transferQueue = new MultiWriteDrainableLinkedList<>();
40  
41  	private volatile LuceneBackendResources resources;
42  
43  	//To allow others to wait on actual shutdown of the internal threads
44  	private final CountDownLatch shutdownLatch = new CountDownLatch( 1 );
45  
46  	private final String indexName;
47  	private volatile boolean stop = false;
48  	final Thread consumerThread;
49  
50  	/**
51  	 * Constructor
52  	 * @param resources LuceneResources to obtain the workspace
53  	 * @param indexName for debugging purposes
54  	 */
55  	public SyncWorkProcessor(LuceneBackendResources resources, String indexName) {
56  		this.resources = resources;
57  		this.indexName = indexName;
58  		consumerThread = new Thread( new Consumer(), "Hibernate Search sync consumer thread for index " + indexName );
59  		consumerThread.setDaemon( true );
60  	}
61  
62  	/**
63  	 * Start processing
64  	 */
65  	public void start() {
66  		log.startingSyncConsumerThread( indexName );
67  		consumerThread.start();
68  	}
69  
70  	/**
71  	 * Submit work and wait for it to be applied to the index
72  	 * @param workList list of work
73  	 * @param monitor for statistics collection
74  	 */
75  	@Override
76  	public void submit(List<LuceneWork> workList, IndexingMonitor monitor) {
77  		//avoid empty work lists as workaround for HSEARCH-1769
78  		if ( workList.isEmpty() ) {
79  			// only log this error at trace level until we properly fix HSEARCH-1769
80  			if ( log.isTraceEnabled() ) {
81  				StringWriter stackTraceStringWriter = new StringWriter();
82  				PrintWriter stackTracePrintWriter = new PrintWriter( stackTraceStringWriter );
83  				new Throwable().printStackTrace( stackTracePrintWriter );
84  				log.workListShouldNeverBeEmpty( stackTraceStringWriter.toString() );
85  			}
86  			// skip that work
87  			return;
88  		}
89  		Changeset changeset = new Changeset( workList, Thread.currentThread(), monitor );
90  		transferQueue.add( changeset );
91  		wakeUpConsumer();
92  		boolean interrupted = false;
93  		while ( ! changeset.isProcessed() && ! interrupted ) {
94  			parkCurrentThread();
95  			if ( Thread.interrupted() ) {
96  				interrupted = true;
97  			}
98  		}
99  		if ( interrupted ) {
100 			Thread.currentThread().interrupt();
101 		}
102 	}
103 
104 	/**
105 	 * Wakes up consumer thread if necessary
106 	 */
107 	private void wakeUpConsumer() {
108 		LockSupport.unpark( consumerThread );
109 	}
110 
111 	/**
112 	 * Dispose resources
113 	 */
114 	@Override
115 	public void shutdown() {
116 		stop = true;
117 		LockSupport.unpark( consumerThread );
118 		try {
119 			shutdownLatch.await( Long.MAX_VALUE, TimeUnit.SECONDS );
120 		}
121 		catch (InterruptedException e) {
122 			log.timedOutWaitingShutdown( indexName );
123 		}
124 	}
125 
126 	/**
127 	 * Handle on the fly rebuilds
128 	 * @param resources new instance of {@link org.hibernate.search.backend.impl.lucene.LuceneBackendResources}
129 	 */
130 	@Override
131 	public void updateResources(LuceneBackendResources resources) {
132 		this.resources = resources;
133 	}
134 
135 	/**
136 	 * Consumer thread
137 	 */
138 	private class Consumer implements Runnable {
139 		@Override
140 		public void run() {
141 			Iterable<Changeset> changesets;
142 			try {
143 				while ( !stop ) {
144 					changesets = transferQueue.drainToDetachedIterable();
145 					while ( changesets == null && !stop ) {
146 						// Avoid busy wait
147 						parkCurrentThread();
148 						changesets = transferQueue.drainToDetachedIterable();
149 					}
150 					if ( changesets != null ) {
151 						applyChangesets( changesets );
152 					}
153 				}
154 				log.stoppingSyncConsumerThread( indexName );
155 			}
156 			finally {
157 				shutdownLatch.countDown();
158 			}
159 		}
160 
161 		private void applyChangesets(Iterable<Changeset> changesets) {
162 			ChangesetList changesetList = new ChangesetList( changesets );
163 			try {
164 				LuceneBackendQueueTask luceneBackendQueueTask = new LuceneBackendQueueTask( changesetList, resources, null );
165 				luceneBackendQueueTask.run();
166 			}
167 			finally {
168 				changesetList.markProcessed();
169 			}
170 		}
171 	}
172 
173 	private void parkCurrentThread() {
174 		//Always use some safety margin when parking threads:
175 		LockSupport.parkNanos( 1_000_000_000 );
176 	}
177 
178 }
