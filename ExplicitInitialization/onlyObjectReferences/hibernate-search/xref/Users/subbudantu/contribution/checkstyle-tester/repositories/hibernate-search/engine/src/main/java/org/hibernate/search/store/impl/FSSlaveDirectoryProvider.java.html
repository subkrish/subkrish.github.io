35  /**
36   * File based directory provider that takes care of getting a version of the index
37   * from a given source.
38   * The base directory is represented by hibernate.search.{@literal <index>}.indexBase
39   * The index is created in {@literal <base directory>/<index name>}
40   * The source (aka copy) directory is built from {@literal <sourceBase>/<index name>}
41   * <p>
42   * A copy is triggered every refresh seconds
43   *
44   * @author Emmanuel Bernard
45   * @author Sanne Grinovero
46   * @author Oliver Siegmar
47   */
48  public class FSSlaveDirectoryProvider implements DirectoryProvider<Directory> {
49  
50  	private static final Log log = LoggerFactory.make();
51  	private final Timer timer = new Timer( true ); //daemon thread, the copy algorithm is robust
52  
53  	private volatile boolean initialized = false;
54  	private volatile boolean started = false;
55  
56  	private volatile int current; //used also as memory barrier of all other values, which are set once.
57  
58  	//variables having visibility granted by a read of "current"
59  	private volatile Directory dummyDirectory;
60  	private FSDirectory directory1;
61  	private FSDirectory directory2;
62  	private String indexName;
63  	private long copyChunkSize;
64  
65  	//variables needed between initialize and start (used by same thread: no special care needed)
66  	private File sourceIndexDir;
67  	private File indexDir;
68  	private String directoryProviderName;
69  	private Properties properties;
70  	private UpdateTask updateTask;
71  	private ServiceManager serviceManager;

73  	@Override
74  	public void initialize(String directoryProviderName, Properties properties, BuildContext context) {
75  		this.properties = properties;
76  		this.directoryProviderName = directoryProviderName;
77  		this.serviceManager = context.getServiceManager();
78  		//source guessing
79  		sourceIndexDir = DirectoryProviderHelper.getSourceDirectory( directoryProviderName, properties, false );
80  		log.debugf( "Source directory: %s", sourceIndexDir.getPath() );
81  		indexDir = DirectoryHelper.getVerifiedIndexDir( directoryProviderName, properties, true );
82  		log.debugf( "Index directory: %s", indexDir.getPath() );
83  		try {
84  			indexName = indexDir.getCanonicalPath();
85  		}
86  		catch (IOException e) {
87  			throw new SearchException( "Unable to initialize index: " + directoryProviderName, e );
88  		}
89  		copyChunkSize = DirectoryProviderHelper.getCopyBufferSize( directoryProviderName, properties );
90  		current = 0; //publish all state to other threads
91  	}

93  	private boolean currentMarkerIsInSource() {
94  		int retry = ConfigurationParseHelper.getIntValue( properties, Environment.RETRY_MARKER_LOOKUP, 0 );
95  		if ( retry < 0 ) {
96  			throw new SearchException( Environment.RETRY_MARKER_LOOKUP +
97  					" option must be a positive integer, but was \"" + retry + "\"" );
98  		}
99  		boolean currentMarkerInSource = false;
100 		for ( int tried = 0 ; tried <= retry ; tried++ ) {
101 			//we try right away the first time
102 			if ( tried > 0 ) {
103 				try {
104 					Thread.sleep( TimeUnit.SECONDS.toMillis( 5 ) );
105 				}
106 				catch (InterruptedException e) {
107 					//continue
108 					Thread.currentThread().interrupt();
109 				}
110 			}
111 			currentMarkerInSource =
112 					new File( sourceIndexDir, "current1" ).exists()
113 					|| new File( sourceIndexDir, "current2" ).exists();
114 			if ( currentMarkerInSource ) {
115 				break;
116 			}
117 		}
118 		return currentMarkerInSource;
119 	}

121 	@Override
122 	public void start(DirectoryBasedIndexManager indexManager) {
123 		if ( ! attemptInitializeAndStart() ) {
124 			// if we failed to initialize and/or start, we'll try again later: setup a timer
125 			long period = DirectoryProviderHelper.getRetryInitializePeriod( properties, directoryProviderName );
126 			if ( period != 0 ) {
127 				scheduleTask( new InitTask(), period );
128 			}
129 			else {
130 				throw new SearchException( "Failed to initialize DirectoryProvider \""
131 						+ directoryProviderName + "\": could not find marker file in index source" );
132 			}
133 		}
134 	}
135 
136 	private void startIt() {
137 		@SuppressWarnings("unused")
138 		int readCurrentState = current; //Unneeded value, but ensure visibility of state protected by memory barrier
139 		int currentToBe = 0;
140 		try {
141 			directory1 = DirectoryProviderHelper.createFSIndex( new File( indexDir, "1" ), properties, serviceManager );
142 			directory2 = DirectoryProviderHelper.createFSIndex( new File( indexDir, "2" ), properties, serviceManager );
143 			File currentMarker = new File( indexDir, "current1" );
144 			File current2Marker = new File( indexDir, "current2" );
145 			if ( currentMarker.exists() ) {
146 				currentToBe = 1;
147 				if ( current2Marker.exists() ) {
148 					current2Marker.delete(); //TODO or throw an exception?
149 				}
150 			}
151 			else if ( current2Marker.exists() ) {
152 				currentToBe = 2;
153 			}
154 			else {
155 				//no default
156 				log.debug( "Setting directory 1 as current" );
157 				currentToBe = 1;
158 				File destinationFile = new File( indexDir, Integer.valueOf( currentToBe ).toString() );
159 				int sourceCurrent;
160 				if ( new File( sourceIndexDir, "current1" ).exists() ) {
161 					sourceCurrent = 1;
162 				}
163 				else if ( new File( sourceIndexDir, "current2" ).exists() ) {
164 					sourceCurrent = 2;
165 				}
166 				else {
167 					throw new SearchException( "No current file marker found in source directory: " + sourceIndexDir.getPath() );
168 				}
169 				try {
170 					FileHelper.synchronize(
171 							new File( sourceIndexDir, String.valueOf( sourceCurrent ) ),
172 							destinationFile, true, copyChunkSize
173 					);
174 				}
175 				catch (IOException e) {
176 					throw new SearchException( "Unable to synchronize directory: " + indexName, e );
177 				}
178 				if ( !currentMarker.createNewFile() ) {
179 					throw new SearchException( "Unable to create the directory marker file: " + indexName );
180 				}
181 			}
182 			log.debugf( "Current directory: %d", (Integer) currentToBe );
183 		}
184 		catch (IOException e) {
185 			throw new SearchException( "Unable to initialize index: " + directoryProviderName, e );
186 		}
187 		updateTask = new UpdateTask( sourceIndexDir, indexDir );
188 		long period = DirectoryProviderHelper.getRefreshPeriod( properties, directoryProviderName );
189 		scheduleTask( updateTask, period );
190 		this.current = currentToBe;
191 		started = true;
192 	}

194 	@Override
195 	public Directory getDirectory() {
196 		if ( !started ) {
197 			if ( dummyDirectory == null ) {
198 				RAMDirectory directory = new RAMDirectory();
199 				DirectoryHelper.initializeIndexIfNeeded( directory );
200 				dummyDirectory = directory;
201 			}
202 			return dummyDirectory;
203 		}
204 
205 		int readState = current;// to have the read consistent in the next two "if"s.
206 		if ( readState == 1 ) {
207 			return directory1;
208 		}
209 		else if ( readState == 2 ) {
210 			return directory2;
211 		}
212 		else {
213 			throw new AssertionFailure( "Illegal current directory: " + readState );
214 		}
215 	}

217 	@Override
218 	public boolean equals(Object obj) {
219 		// this code is actually broken since the value change after initialize call
220 		// but from a practical POV this is fine since we only call this method
221 		// after initialize call
222 		if ( obj == this ) {
223 			return true;
224 		}
225 		if ( obj == null || !( obj instanceof FSSlaveDirectoryProvider ) ) {
226 			return false;
227 		}
228 		FSSlaveDirectoryProvider other = (FSSlaveDirectoryProvider) obj;
229 		//need to break memory barriers on both instances:
230 		@SuppressWarnings("unused")
231 		int readCurrentState = this.current; //unneeded value, but ensure visibility of indexName
232 		readCurrentState = other.current; //another unneeded value, but ensure visibility of indexName
233 		return indexName.equals( other.indexName );
234 	}

238 		// this code is actually broken since the value change after initialize call
239 		// but from a practical POV this is fine since we only call this method
240 		// after initialize call
241 		@SuppressWarnings("unused")
242 		int readCurrentState = current; //unneeded value, but ensure visibility of indexName
243 		int hash = 11;
244 		return 37 * hash + indexName.hashCode();

247 	class InitTask extends TimerTask {
248 
249 		@Override
250 		public void run() {
251 			try {
252 				if ( attemptInitializeAndStart() ) {
253 					// then this task is no longer needed
254 					cancel();
255 				}
256 			}
257 			catch (RuntimeException re) {
258 				// we need this to make sure the error is logged somewhere,
259 				// as we're executing it in the timer thread
260 				log.failedSlaveDirectoryProviderInitialization( indexName, re );
261 			}
262 		}
263 	}
264 
265 	/**
266 	 * @return true if both initialize and start succeeded
267 	 */
268 	protected synchronized boolean attemptInitializeAndStart() {
269 		if ( !initialized ) {
270 			if ( currentMarkerIsInSource() ) {
271 				initialized = true;
272 				log.foundCurrentMarker();
273 			}
274 			else {
275 				log.noCurrentMarkerInSourceDirectory();
276 			}
277 		}
278 		if ( initialized ) {
279 			startIt();
280 		}
281 		return this.started;
282 	}

284 	class UpdateTask extends TimerTask {
285 
286 		private final ExecutorService executor;
287 		private final CopyDirectory copyTask;
288 
289 		public UpdateTask(File sourceIndexDir, File destination) {
290 			executor = Executors.newSingleThreadExecutor();
291 			copyTask = new CopyDirectory( sourceIndexDir, destination );
292 		}
293 
294 		@Override
295 		public void run() {
296 			if ( copyTask.inProgress.compareAndSet( false, true ) ) {
297 				executor.execute( copyTask );
298 			}
299 			else {
300 				if ( log.isTraceEnabled() ) {
301 					@SuppressWarnings("unused")
302 					int unneeded = current;//ensure visibility of indexName in Timer threads.
303 					log.tracef( "Skipping directory synchronization, previous work still in progress: %s", indexName );
304 				}
305 			}
306 		}
307 
308 		public void stop() {
309 			executor.shutdownNow();
310 		}
311 	}

313 	class CopyDirectory implements Runnable {
314 		private final File source;
315 		private final File destination;
316 		private final AtomicBoolean inProgress = new AtomicBoolean( false );
317 
318 		public CopyDirectory(File sourceIndexDir, File destination) {
319 			this.source = sourceIndexDir;
320 			this.destination = destination;
321 		}
322 
323 		@Override
324 		public void run() {
325 			long start = System.nanoTime();
326 			try {
327 				File sourceFile = determineCurrentSourceFile();
328 				if ( sourceFile == null ) {
329 					log.unableToDetermineCurrentInSourceDirectory();
330 					return;
331 				}
332 
333 				// check whether a copy is needed at all
334 				File currentDestinationFile = new File( destination, Integer.valueOf( current ).toString() );
335 				try {
336 					if ( FileHelper.areInSync( sourceFile, currentDestinationFile ) ) {
337 						if ( log.isTraceEnabled() ) {
338 							log.trace( "Source and destination directory are in sync. No copying required." );
339 						}
340 						return;
341 					}
342 				}
343 				catch (IOException ioe) {
344 					log.unableToCompareSourceWithDestinationDirectory( sourceFile.getName(), currentDestinationFile.getName() );
345 				}
346 
347 				// copy is required
348 				int oldIndex = current;
349 				int index = oldIndex == 1 ? 2 : 1;
350 				File destinationFile = new File( destination, Integer.valueOf( index ).toString() );
351 				try {
352 					log.tracef( "Copying %s into %s", sourceFile, destinationFile );
353 					FileHelper.synchronize( sourceFile, destinationFile, true, copyChunkSize );
354 					current = index;
355 					log.tracef( "Copy for %s took %d ms", indexName, TimeUnit.NANOSECONDS.toMillis( System.nanoTime() - start ) );
356 				}
357 				catch (IOException e) {
358 					//don't change current
359 					log.unableToSynchronizeSource( indexName, e );
360 					return;
361 				}
362 				if ( !new File( indexName, "current" + oldIndex ).delete() ) {
363 					log.unableToRemovePreviousMarket( indexName );
364 				}
365 				try {
366 					new File( indexName, "current" + index ).createNewFile();
367 				}
368 				catch (IOException e) {
369 					log.unableToCreateCurrentMarker( indexName, e );
370 				}
371 			}
372 			finally {
373 				inProgress.set( false );
374 			}
375 		}
376 
377 		/**
378 		 * @return Return a file to the currently active source directory. Tests for the files "current1" and
379 		 *         "current2" in order to determine which is the current directory. If the marker file does not exist
380 		 *         <code>null</code> is returned.
381 		 */
382 		private File determineCurrentSourceFile() {
383 			File sourceFile = null;
384 			if ( new File( source, "current1" ).exists() ) {
385 				sourceFile = new File( source, "1" );
386 			}
387 			else if ( new File( source, "current2" ).exists() ) {
388 				sourceFile = new File( source, "2" );
389 			}
390 			return sourceFile;
391 		}
392 	}
393 
394 	@Override
395 	public void stop() {
396 		@SuppressWarnings("unused")
397 		int readCurrentState = current; //unneeded value, but ensure visibility of state protected by memory barrier
398 		timer.cancel();
399 		if ( updateTask != null ) {
400 			updateTask.stop();
401 		}
402 		closeDirectory( directory1 );
403 		closeDirectory( directory2 );
404 	}
405 
406 	private void closeDirectory(Directory directory) {
407 		if ( directory != null ) {
408 			try {
409 				directory.close();
410 			}
411 			catch (Exception e) {
412 				log.unableToCloseLuceneDirectory( directory, e );
413 			}
414 		}
415 	}

417 	protected void scheduleTask(TimerTask task, long period) {
418 		timer.schedule( task, period, period );
