7   package org.hibernate.search.store.optimization.impl;
8   
9   import java.util.Properties;
10  
11  import org.apache.lucene.index.IndexWriter;
12  
13  import org.hibernate.search.store.Workspace;
14  import org.hibernate.search.store.optimization.OptimizerStrategy;
15  import org.hibernate.search.util.configuration.impl.ConfigurationParseHelper;
16  import org.hibernate.search.indexes.spi.IndexManager;
17  import org.hibernate.search.util.logging.impl.Log;
18  import org.hibernate.search.util.logging.impl.LoggerFactory;
19  
20  /**
21   * Optimization strategy triggered after a certain amount of operations
22   *
23   * @author Emmanuel Bernard
24   */
25  public class IncrementalOptimizerStrategy extends ExplicitOnlyOptimizerStrategy implements OptimizerStrategy {
26  
27  	private static final Log log = LoggerFactory.make();
28  
29  	private int operationMax = -1;
30  	private int transactionMax = -1;
31  	private volatile long operations = 0;
32  	private volatile long transactions = 0;
33  	private volatile long optimizationsPerformed = 0;
34  	private final Object lockOnCounters = new Object();
35  
36  	@Override
37  	public boolean performOptimization(IndexWriter writer) {
38  		boolean done = super.performOptimization( writer );
39  		if ( done ) {
40  			synchronized ( lockOnCounters ) {
41  				operations = 0;
42  				transactions = 0;
43  				optimizationsPerformed++;
44  			}
45  		}
46  		return done;
47  	}
48  
49  	private boolean needOptimization() {
50  		return (operationMax != -1 && operations >= operationMax)
51  				|| (transactionMax != -1 && transactions >= transactionMax);
52  	}
53  
54  	@Override
55  	public void addOperationWithinTransactionCount(long operations) {
56  		synchronized ( lockOnCounters ) {
57  			this.operations += operations;
58  			this.transactions++;
59  		}
60  	}
61  
62  	@Override
63  	public void optimize(Workspace workspace) {
64  		if ( needOptimization() ) {
65  			log.debugv( "Optimize {0} after {1} operations and {2} transactions",
66  						indexName, operations, transactions );
67  			IndexWriter writer = workspace.getIndexWriter();
68  			try {
69  				performOptimization( writer );
70  			}
71  			finally {
72  				workspace.afterTransactionApplied( false, false );
73  			}
74  		}
75  	}
76  
77  	@Override
78  	public void initialize(IndexManager indexManager, Properties indexProperties) {
79  		super.initialize( indexManager, indexProperties );
80  		operationMax = ConfigurationParseHelper.getIntValue( indexProperties, "operation_limit.max", -1 );
81  		transactionMax = ConfigurationParseHelper.getIntValue( indexProperties, "transaction_limit.max", -1 );
82  	}
83  
84  	public long getOptimizationsPerformed() {
85  		return optimizationsPerformed;
86  	}
87  
88  }
