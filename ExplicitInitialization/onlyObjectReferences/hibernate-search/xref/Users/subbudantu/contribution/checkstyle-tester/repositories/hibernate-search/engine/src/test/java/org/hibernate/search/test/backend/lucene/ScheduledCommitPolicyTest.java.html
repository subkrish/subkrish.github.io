34  /**
35   * Tests for the scheduled commit policy
36   *
37   * @author gustavonalle
38   */
39  @RunWith(BMUnitRunner.class)
40  @Category(SkipOnElasticsearch.class) // Commit policies are specific to the Lucene backend
41  public class ScheduledCommitPolicyTest {
42  
43  	private static final int NUMBER_ENTITIES = 1000;
44  
45  	private int globalIdCounter = 0;
46  
47  	@Rule
48  	public SearchFactoryHolder sfAsyncExclusiveIndex = new SearchFactoryHolder( Quote.class )
49  			.withProperty( "hibernate.search.default.index_flush_interval", "1" )
50  			.withProperty( "hibernate.search.default.worker.execution", "async" )
51  			.withProperty( "hibernate.search.default.exclusive_index_use", "true" )
52  			.withProperty( "hibernate.search.error_handler", CountingErrorHandler.class.getName() );
53  
54  	@Test
55  	public void testScheduledCommits() throws Exception {
56  		writeData( sfAsyncExclusiveIndex, NUMBER_ENTITIES );
57  		AbstractWorkspaceImpl workspace = sfAsyncExclusiveIndex.extractWorkspace( Quote.class );
58  		CommitPolicy commitPolicy = workspace.getCommitPolicy();
59  
60  		assertTrue( commitPolicy instanceof ScheduledCommitPolicy );
61  
62  		ScheduledCommitPolicy scheduledCommitPolicy = (ScheduledCommitPolicy) commitPolicy;
63  		ScheduledThreadPoolExecutor scheduledExecutor = (ScheduledThreadPoolExecutor) scheduledCommitPolicy.getScheduledExecutorService();
64  
65  		assertConditionMet( new TaskExecutedCondition( scheduledExecutor, 1 ) );
66  	}
67  
68  	@Test
69  	@BMRule(targetClass = "org.apache.lucene.index.IndexWriter",
70  			targetMethod = "commit",
71  			action = "throw new IOException(\"File not found!\")",
72  			name = "commitError")
73  	public void testErrorHandlingDuringCommit() throws Exception {
74  		writeData( sfAsyncExclusiveIndex, 2 );
75  		final CountingErrorHandler errorHandler = (CountingErrorHandler) sfAsyncExclusiveIndex.getSearchFactory().getErrorHandler();
76  		assertConditionMet( new Condition() {
77  			@Override
78  			public boolean evaluate() {
79  				return errorHandler.getCountFor( IOException.class ) >= 2;
80  			}
81  		} );
82  	}
83  
84  	@Test
85  	@BMRule(targetClass = "org.hibernate.search.backend.impl.lucene.IndexWriterHolder",
86  			targetMethod = "commitIndexWriter()",
87  			action = "throw new NullPointerException(\"Fake internal error\")",
88  			name = "timerDisruptingError")
89  	public void testErrorHandlingOnBackgroundThread() throws Exception {
90  		writeData( sfAsyncExclusiveIndex, 2 );
91  		final CountingErrorHandler errorHandler = (CountingErrorHandler) sfAsyncExclusiveIndex.getSearchFactory().getErrorHandler();
92  		assertConditionMet( new Condition() {
93  			@Override
94  			public boolean evaluate() {
95  				// It's going to commit once each millisecond, and produce a failure each time.
96  				// So "4" is just a random number higher than 0, but high enough to
97  				// verify that the scheduled task is not being killed at the first failure,
98  				// and will keep trying.
99  				return errorHandler.getCountFor( NullPointerException.class ) >= 4;
100 			}
101 		} );
102 	}
103 
104 	@Test
105 	public void testDocVisibility() throws Exception {
106 		writeData( sfAsyncExclusiveIndex, NUMBER_ENTITIES );
107 		assertConditionMet( new IndexingFinishedCondition( sfAsyncExclusiveIndex, NUMBER_ENTITIES ) );
108 
109 		writeData( sfAsyncExclusiveIndex, 10 );
110 		assertConditionMet( new IndexingFinishedCondition( sfAsyncExclusiveIndex, NUMBER_ENTITIES + 10 ) );
111 
112 		writeData( sfAsyncExclusiveIndex, 1 );
113 		assertConditionMet( new IndexingFinishedCondition( sfAsyncExclusiveIndex, NUMBER_ENTITIES + 10 + 1 ) );
114 	}
115 
116 	private class IndexingFinishedCondition implements Condition {
117 		private final int expectedDocsCount;
118 		private final ExtendedSearchIntegrator searchFactory;
119 		private IndexingFinishedCondition(SearchFactoryHolder searchFactoryHolder, int expectedDocsCount) {
120 			this.searchFactory = searchFactoryHolder.getSearchFactory();
121 			this.expectedDocsCount = expectedDocsCount;
122 		}
123 
124 		private HSQuery matchAllQuery() {
125 			return searchFactory
126 					.createHSQuery( new MatchAllDocsQuery(), Quote.class );
127 		}
128 
129 		@Override
130 		public boolean evaluate() {
131 			return expectedDocsCount == matchAllQuery().queryResultSize();
132 		}
133 	}
134 
135 	private class TaskExecutedCondition implements Condition {
136 
137 		private final ScheduledThreadPoolExecutor executor;
138 		private final int taskCount;
139 
140 		private TaskExecutedCondition(ScheduledThreadPoolExecutor executor, int taskCount) {
141 			this.executor = executor;
142 			this.taskCount = taskCount;
143 		}
144 		@Override
145 		public boolean evaluate() {
146 			return executor.getCompletedTaskCount() >= taskCount;
147 		}
148 	}
149 
150 	private void writeData(SearchFactoryHolder sfHolder, int numberEntities) {
151 		for ( int i = 0; i < numberEntities; i++ ) {
152 			Integer id = Integer.valueOf( globalIdCounter++ );
153 			Quote quote = new Quote( id, "description" );
154 			Work work = new Work( quote, id, WorkType.ADD, false );
155 			TransactionContextForTest tc = new TransactionContextForTest();
156 			sfHolder.getSearchFactory().getWorker().performWork( work, tc );
157 			tc.end();
158 		}
159 	}
