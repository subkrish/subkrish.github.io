7   package org.hibernate.search.test.integration.jbossjta.infra;
8   
9   import java.sql.SQLException;
10  import javax.sql.DataSource;
11  import javax.sql.XADataSource;
12  
13  import com.arjuna.ats.arjuna.coordinator.TxControl;
14  import com.arjuna.ats.jdbc.TransactionalDriver;
15  
16  public class JBossTADataSourceBuilder {
17  	private String user;
18  	private String password;
19  	//by default no timeout
20  	private int timeout = 0;
21  	private XADataSource xaDataSource;
22  
23  	public JBossTADataSourceBuilder setUser(String user) {
24  		this.user = user;
25  		return this;
26  	}
27  
28  	public JBossTADataSourceBuilder setPassword(String password) {
29  		this.password = password;
30  		return this;
31  	}
32  
33  	public JBossTADataSourceBuilder setTimeout(int timeout) {
34  		this.timeout = timeout;
35  		return this;
36  	}
37  
38  	public JBossTADataSourceBuilder setXADataSource(XADataSource xaDataSource) {
39  		this.xaDataSource = xaDataSource;
40  		return this;
41  	}
42  
43  	public DataSource createDataSource() throws SQLException {
44  		TxControl.setDefaultTimeout( timeout );
45  		DataSourceProvider.initialize( xaDataSource );
46  		DataSourceProvider dsProvider = new DataSourceProvider();
47  		final XADataSource dataSource = dsProvider.getDataSource( dsProvider.getDataSourceName() );
48  		XADataSourceWrapper dsw = new XADataSourceWrapper(
49  				dsProvider.getDataSourceName(),
50  				dataSource
51  		);
52  		dsw.setProperty( TransactionalDriver.dynamicClass, DataSourceProvider.class.getName() );
53  		dsw.setProperty( TransactionalDriver.userName, user );
54  		dsw.setProperty( TransactionalDriver.password, password );
55  		return dsw;
56  	}
57  }
