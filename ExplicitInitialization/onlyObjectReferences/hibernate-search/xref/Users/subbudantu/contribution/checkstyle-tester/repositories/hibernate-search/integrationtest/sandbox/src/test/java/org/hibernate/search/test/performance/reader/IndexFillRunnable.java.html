7   package org.hibernate.search.test.performance.reader;
8   
9   import java.io.IOException;
10  
11  import org.apache.lucene.document.Document;
12  import org.apache.lucene.document.Field;
13  import org.apache.lucene.document.Field.Index;
14  import org.apache.lucene.document.Field.Store;
15  import org.apache.lucene.index.IndexWriter;
16  
17  /**
18   * @author Sanne Grinovero
19   */
20  public class IndexFillRunnable implements Runnable {
21  
22  	private volatile int jobSeed = 0;
23  	private final IndexWriter iw;
24  
25  	public IndexFillRunnable(IndexWriter iw) {
26  		super();
27  		this.iw = iw;
28  	}
29  
30  	@Override
31  	public void run() {
32  		Field f1 = new Field( "name", "Some One " + jobSeed++, Store.NO, Index.ANALYZED );
33  		Field f2 = new Field(
34  				"physicalDescription",
35  				" just more people sitting around and filling my index... ",
36  				Store.NO,
37  				Index.ANALYZED
38  		);
39  		Document d = new Document();
40  		d.add( f1 );
41  		d.add( f2 );
42  		try {
43  			iw.addDocument( d );
44  		}
45  		catch (IOException e) {
46  			e.printStackTrace();
47  		}
48  	}
49  
50  }
