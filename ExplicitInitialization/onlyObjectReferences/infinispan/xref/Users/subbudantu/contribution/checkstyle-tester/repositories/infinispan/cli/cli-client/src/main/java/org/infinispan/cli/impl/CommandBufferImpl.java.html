1   package org.infinispan.cli.impl;
2   
3   import org.infinispan.cli.CommandBuffer;
4   import org.infinispan.cli.commands.ProcessedCommand;
5   
6   import java.util.ArrayList;
7   import java.util.List;
8   
9   
10  public class CommandBufferImpl implements CommandBuffer {
11     private int nesting = 0;
12     StringBuilder commandsString = new StringBuilder(1024);
13     List<ProcessedCommand> commands = new ArrayList<ProcessedCommand>(1);
14  
15     @Override
16     public void reset() {
17        nesting = 0;
18        commandsString.setLength(0);
19        commands.clear();
20     }
21  
22     @Override
23     public boolean addCommand(ProcessedCommand commandLine, int nesting) {
24        this.nesting += nesting;
25        commandsString.append(commandLine.getCommandLine()).append("\n");
26        commands.add(commandLine);
27        return this.nesting == 0;
28     }
29  
30     @Override
31     public List<ProcessedCommand> getBufferedCommands() {
32        return commands;
33     }
34  
35     @Override
36     public String toString() {
37        return commandsString.toString();
38     }
39  }
