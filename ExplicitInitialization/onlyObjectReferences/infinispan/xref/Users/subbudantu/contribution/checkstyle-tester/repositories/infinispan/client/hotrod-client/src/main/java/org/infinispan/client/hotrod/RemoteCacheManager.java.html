39  /**
40   * Factory for {@link org.infinispan.client.hotrod.RemoteCache}s. <p/> <p> <b>Lifecycle:</b> </p> In order to be able to
41   * use an {@link org.infinispan.client.hotrod.RemoteCache}, the {@link org.infinispan.client.hotrod.RemoteCacheManager}
42   * must be started first: beside other things, this instantiates connections to Hot Rod server(s). Starting the {@link
43   * org.infinispan.client.hotrod.RemoteCacheManager} can be done either at creation by passing start==true to constructor
44   * or by using a constructor that does that for you (see C-tor documentation); or after construction by calling {@link
45   * #start()}.
46   * <p/>
47   * This is an "expensive" object, as it manages a set of persistent TCP connections to the Hot Rod servers. It is recommended
48   * to only have one instance of this per JVM, and to cache it between calls to the server (i.e. remoteCache
49   * operations).
50   * <p/>
51   * {@link #stop()} needs to be called explicitly in order to release all the resources (e.g. threads, TCP connections).
52   * <p/>
53   * <p/>
54   * <b>Configuration:</b>
55   * <p/>
56   * The cache manager is configured through a {@link Configuration} object passed to the constructor (there are also
57   * "simplified" constructors that rely on default values). If migrating from a previous version of Infinispan, where {@link Properties}
58   * were used to configure the RemoteCacheManager, please use the {@link ConfigurationBuilder#withProperties(Properties)} method.
59   * <p/>
60   * Below is the list of supported configuration elements:
61   * <ul>
62   * <li><tt>infinispan.client.hotrod.request_balancing_strategy</tt>, default = org.infinispan.client.hotrod.impl.transport.tcp.RoundRobinBalancingStrategy.  For replicated (vs distributed) Hot Rod server clusters, the client balances requests to the
63   * servers according to this strategy.</li>
64   * <li><tt>infinispan.client.hotrod.server_list</tt>, default = 127.0.0.1:11222.  This is the initial list of Hot Rod servers to connect to, specified in the following format: host1:port1;host2:port2...
65   * At least one host:port must be specified.</li>
66   * <li><tt>infinispan.client.hotrod.force_return_values</tt>, default = false.  Whether or not to implicitly {@link org.infinispan.client.hotrod.Flag#FORCE_RETURN_VALUE} for all calls.</li>
67   * <li><tt>infinispan.client.hotrod.tcp_no_delay</tt>, default = true.  Affects TCP NODELAY on the TCP stack.</li>
68   * <li><tt>infinispan.client.hotrod.tcp_keep_alive</tt>, default = false.  Affects TCP KEEPALIVE on the TCP stack.</li>
69   * <li><tt>infinispan.client.hotrod.ping_on_startup</tt>, default = true.  If true, a ping request is sent to a back end server in order to fetch cluster's topology.</li>
70   * <li><tt>infinispan.client.hotrod.transport_factory</tt>, default = org.infinispan.client.hotrod.impl.transport.tcp.TcpTransportFactory - controls which transport to use.  Currently only the TcpTransport is supported.</li>
71   * <li><tt>infinispan.client.hotrod.marshaller</tt>, default = org.infinispan.marshall.jboss.GenericJBossMarshaller.  Allows you to specify a custom {@link org.infinispan.marshall.Marshaller} implementation to serialize and deserialize user objects. For portable serialization payloads, you should configure the marshaller to be {@link org.infinispan.client.hotrod.marshall.ProtoStreamMarshaller}</li>
72   * <li><tt>infinispan.client.hotrod.async_executor_factory</tt>, default = org.infinispan.client.hotrod.impl.async.DefaultAsyncExecutorFactory.  Allows you to specify a custom asynchroous executor for async calls.</li>
73   * <li><tt>infinispan.client.hotrod.default_executor_factory.pool_size</tt>, default = 99.  If the default executor is used, this configures the number of threads to initialize the executor with.</li>
74   * <li><tt>infinispan.client.hotrod.default_executor_factory.queue_size</tt>, default = 100000.  If the default executor is used, this configures the queue size to initialize the executor with.</li>
75   * <li><tt>infinispan.client.hotrod.hash_function_impl.1</tt>, default = It uses the hash function specified by the server in the responses as indicated in {@link org.infinispan.client.hotrod.impl.consistenthash.ConsistentHashFactory}.  This specifies the version of the hash function and consistent hash algorithm in use, and is closely tied with the HotRod server version used.</li>
76   * <li><tt>infinispan.client.hotrod.key_size_estimate</tt>, default = 64.  This hint allows sizing of byte buffers when serializing and deserializing keys, to minimize array resizing.</li>
77   * <li><tt>infinispan.client.hotrod.value_size_estimate</tt>, default = 512.  This hint allows sizing of byte buffers when serializing and deserializing values, to minimize array resizing.</li>
78   * <li><tt>infinispan.client.hotrod.socket_timeout</tt>, default = 60000 (60 seconds).  This property defines the maximum socket read timeout before giving up waiting for bytes from the server.</li>
79   * <li><tt>infinispan.client.hotrod.protocol_version</tt>, default = 2.0 .This property defines the protocol version that this client should use. Other valid values include 1.0.</li>
80   * <li><tt>infinispan.client.hotrod.connect_timeout</tt>, default = 60000 (60 seconds).  This property defines the maximum socket connect timeout before giving up connecting to the server.</li>
81   * <li><tt>infinispan.client.hotrod.max_retries</tt>, default = 10.  This property defines the maximum number of retries in case of a recoverable error. A valid value should be greater or equals to 0 (zero). Zero mean no retry.</li>
82   * </ul>
83   * <br/>
84   * <i>The following properties are related to connection pooling</i>:
85   * <p/>
86   * <ul>
87   * <li><tt>maxActive</tt> - controls the maximum number of connections per server that are allocated (checked out to client threads, or idle in
88   * the pool) at one time. When non-positive, there is no limit to the number of connections per server. When maxActive
89   * is reached, the connection pool for that server is said to be exhausted. The default setting for this parameter is
90   * -1, i.e. there is no limit.</li>
91   * <li><tt>maxTotal</tt> - sets a global limit on the number persistent connections that can be in circulation within the combined set of
92   * servers. When non-positive, there is no limit to the total number of persistent connections in circulation. When
93   * maxTotal is exceeded, all connections pools are exhausted. The default setting for this parameter is -1 (no limit).
94   * </li>
95   * <p/>
96   * <li><tt>maxIdle</tt> - controls the maximum number of idle persistent connections, per server, at any time. When negative, there is no limit
97   * to the number of connections that may be idle per server. The default setting for this parameter is -1.</li>
98   * <p/>
99   * <li>
100  * <tt>whenExhaustedAction</tt> - specifies what happens when asking for a connection from a server's pool, and that pool is exhausted. Possible values:
101  * <ul>
102  * <li> <tt>0</tt> - an exception will be thrown to the calling user</li>
103  * <li> <tt>1</tt> - the caller will block (invoke waits until a new or idle connections is available.
104  * <li> <tt>2</tt> - a new persistent connection will be created and returned (essentially making maxActive meaningless.) </li>
105  * </ul>
106  * The default whenExhaustedAction setting is 1.
107  * </li>
108  * <p/>
109  * <li>
110  * Optionally, one may configure the pool to examine and possibly evict connections as they sit idle in the pool and to
111  * ensure that a minimum number of idle connections is maintained for each server. This is performed by an "idle connection
112  * eviction" thread, which runs asynchronously. The idle object evictor does not lock the pool
113  * throughout its execution.  The idle connection eviction thread may be configured using the following attributes:
114  * <ul>
115  * <li><tt>timeBetweenEvictionRunsMillis</tt> - indicates how long the eviction thread should sleep before "runs" of examining idle
116  * connections. When non-positive, no eviction thread will be launched. The default setting for this parameter is
117  * 2 minutes </li>
118  * <li> <tt>minEvictableIdleTimeMillis</tt> - specifies the minimum amount of time that an connection may sit idle in the pool before it
119  * is eligible for eviction due to idle time. When non-positive, no connection will be dropped from the pool due to
120  * idle time alone. This setting has no effect unless timeBetweenEvictionRunsMillis > 0. The default setting for this
121  * parameter is 1800000(30 minutes). </li>
122  * <li> <tt>testWhileIdle</tt> - indicates whether or not idle connections should be validated by sending an TCP packet to the server,
123  * during idle connection eviction runs.  Connections that fail to validate will be dropped from the pool. This setting
124  * has no effect unless timeBetweenEvictionRunsMillis > 0.  The default setting for this parameter is true.
125  * </li>
126  * <li><tt>minIdle</tt> - sets a target value for the minimum number of idle connections (per server) that should always be available.
127  * If this parameter is set to a positive number and timeBetweenEvictionRunsMillis > 0, each time the idle connection
128  * eviction thread runs, it will try to create enough idle instances so that there will be minIdle idle instances
129  * available for each server.  The default setting for this parameter is 1. </li>
130  * </ul>
131  * </li>
132  * <li>
133  * </ul>
134  *
135  * @author Mircea.Markus@jboss.com
136  * @since 4.1
137  */

356    /**
357     * Same as {@link #RemoteCacheManager(java.util.Properties)}, but it will try to lookup the config properties in the
358     * classpath, in a file named <tt>hotrod-client.properties</tt>. If no properties can be found in the classpath, the
359     * server tries to connect to "127.0.0.1:11222" in start.
360     *
361     * @param start whether or not to start the RemoteCacheManager
362     * @throws HotRodClientException if such a file cannot be found in the classpath
363     */
364    public RemoteCacheManager(boolean start) {
365       ConfigurationBuilder builder = new ConfigurationBuilder();
366       ClassLoader cl = Thread.currentThread().getContextClassLoader();
367       builder.classLoader(cl);
368       InputStream stream = FileLookupFactory.newInstance().lookupFile(HOTROD_CLIENT_PROPERTIES, cl);
369       if (stream == null) {
370          log.couldNotFindPropertiesFile(HOTROD_CLIENT_PROPERTIES);
371       } else {
372          try {
373             builder.withProperties(loadFromStream(stream));
374          } finally {
375             Util.close(stream);
376          }
377       }
378       this.configuration = builder.build();
379       if (start) start();
380    }

553    @Override
554    public void start() {
555       // Workaround for JDK6 NPE: http://bugs.sun.com/view_bug.do?bug_id=6427854
556       SecurityActions.setProperty("sun.nio.ch.bugLevel", "\"\"");
557 
558       codec = CodecFactory.getCodec(configuration.protocolVersion());
559 
560       transportFactory = Util.getInstance(configuration.transportFactory());
561 
562       if (marshaller == null) {
563          marshaller = configuration.marshaller();
564          if (marshaller == null) {
565             marshaller = Util.getInstance(configuration.marshallerClass());
566          }
567       }
568 
569       if (asyncExecutorService == null) {
570          ExecutorFactory executorFactory = configuration.asyncExecutorFactory().factory();
571          if (executorFactory == null) {
572             executorFactory = Util.getInstance(configuration.asyncExecutorFactory().factoryClass());
573          }
574          asyncExecutorService = executorFactory.getExecutor(configuration.asyncExecutorFactory().properties());
575       }
576 
577       listenerNotifier = ClientListenerNotifier.create(codec, marshaller);
578       transportFactory.start(codec, configuration, defaultCacheTopologyId, listenerNotifier);
579 
580       synchronized (cacheName2RemoteCache) {
581          for (RemoteCacheHolder rcc : cacheName2RemoteCache.values()) {
582             startRemoteCache(rcc, defaultCacheTopologyId);
583          }
584       }
585 
586       // Print version to help figure client version run
587       log.version(RemoteCacheManager.class.getPackage().getImplementationVersion());
588 
589       started = true;
590    }

594     * As part of the disconnection, all registered client cache listeners will
595     * be removed since client no longer can receive callbacks.

621    @SuppressWarnings("unchecked")
622    private <K, V> RemoteCache<K, V> createRemoteCache(String cacheName, Boolean forceReturnValueOverride) {
623       synchronized (cacheName2RemoteCache) {
624          RemoteCacheKey key = new RemoteCacheKey(cacheName, forceReturnValueOverride);
625          if (!cacheName2RemoteCache.containsKey(key)) {
626             RemoteCacheImpl<K, V> result = createRemoteCache(cacheName);
627             RemoteCacheHolder rcc = new RemoteCacheHolder(result, forceReturnValueOverride == null ? configuration.forceReturnValues() : forceReturnValueOverride);
628             AtomicInteger topologyId = cacheName.isEmpty() ? defaultCacheTopologyId : new AtomicInteger(-1);
629             startRemoteCache(rcc, topologyId);
630             if (configuration.pingOnStartup()) {
631                // If ping not successful assume that the cache does not exist
632                // Default cache is always started, so don't do for it
633                if (!cacheName.equals(RemoteCacheManager.DEFAULT_CACHE_NAME) &&
634                      ping(result) == PingResult.CACHE_DOES_NOT_EXIST) {
635                   return null;
636                }
637             }
638             result.start();
639             // If ping on startup is disabled, or cache is defined in server
640             cacheName2RemoteCache.put(key, rcc);
641             return result;
642          } else {
643             return (RemoteCache<K, V>) cacheName2RemoteCache.get(key).remoteCache;
644          }
645       }
646    }
647 
648    private <K, V> RemoteCacheImpl<K, V> createRemoteCache(String cacheName) {
649       switch (configuration.nearCache().mode()) {
650          case LAZY:
651             NearCacheService<K, V> invalidateNearCache = createNearCacheService(configuration.nearCache());
652             return new InvalidatedNearRemoteCache<>(this, cacheName, invalidateNearCache);
653          case EAGER:
654             NearCacheService<K, V> eagerNearCache = createNearCacheService(configuration.nearCache());
655             return new EagerNearRemoteCache<>(this, cacheName, eagerNearCache);
656          case DISABLED:
657          default:
658             return new RemoteCacheImpl<>(this, cacheName);
659       }
660    }
661 
662    protected <K, V> NearCacheService<K, V> createNearCacheService(NearCacheConfiguration cfg) {
663       return NearCacheService.create(cfg, listenerNotifier);
664    }
665 
666    private <K, V> PingResult ping(RemoteCacheImpl<K, V> cache) {
667       if (transportFactory == null) {
668          return PingResult.FAIL;
669       }
670 
671       return cache.ping();
672    }

674    private void startRemoteCache(RemoteCacheHolder remoteCacheHolder, AtomicInteger topologyId) {
675       RemoteCacheImpl<?, ?> remoteCache = remoteCacheHolder.remoteCache;
676       OperationsFactory operationsFactory = new OperationsFactory(
677             transportFactory, remoteCache.getName(), topologyId, remoteCacheHolder.forceReturnValue,
678             codec, listenerNotifier);
679       remoteCache.init(marshaller, asyncExecutorService, operationsFactory, configuration.keySizeEstimate(), configuration.valueSizeEstimate());
680    }

698 class RemoteCacheKey {
699 
700    final String cacheName;
701    final boolean forceReturnValue;
702 
703    RemoteCacheKey(String cacheName, boolean forceReturnValue) {
704       this.cacheName = cacheName;
705       this.forceReturnValue = forceReturnValue;
706    }
707 
708    @Override
709    public boolean equals(Object o) {
710       if (this == o) return true;
711       if (!(o instanceof RemoteCacheKey)) return false;
712 
713       RemoteCacheKey that = (RemoteCacheKey) o;
714 
715       if (forceReturnValue != that.forceReturnValue) return false;
716       return !(cacheName != null ? !cacheName.equals(that.cacheName) : that.cacheName != null);
717    }
718 
719    @Override
720    public int hashCode() {
721       int result = cacheName != null ? cacheName.hashCode() : 0;
722       result = 31 * result + (forceReturnValue ? 1 : 0);
723       return result;
724    }
