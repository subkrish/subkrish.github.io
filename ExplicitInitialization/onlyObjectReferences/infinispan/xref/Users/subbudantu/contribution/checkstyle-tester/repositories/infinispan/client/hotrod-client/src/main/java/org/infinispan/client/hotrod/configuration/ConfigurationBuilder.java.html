39     // Match IPv4 (host:port) or IPv6 ([host]:port) addresses
40     private static final Pattern ADDRESS_PATTERN = Pattern
41           .compile("(\\[([0-9A-Fa-f:]+)\\]|([^:/?#]*))(?::(\\d*))?");

83     @Override
84     public ConfigurationBuilder addServers(String servers) {
85        for (String server : servers.split(";")) {
86           Matcher matcher = ADDRESS_PATTERN.matcher(server);
87           if (matcher.matches()) {
88              String v6host = matcher.group(2);
89              String v4host = matcher.group(3);
90              String host = v6host != null ? v6host : v4host;
91              String portString = matcher.group(4);
92              int port = portString == null
93                    ? ConfigurationProperties.DEFAULT_HOTROD_PORT
94                    : Integer.parseInt(portString);
95              this.addServer().host(host).port(port);
96           } else {
97              throw log.parseErrorServerAddress(server);
98           }
99  
100       }
101       return this;
102    }

255    @Override
256    public ConfigurationBuilder withProperties(Properties properties) {
257       TypedProperties typed = TypedProperties.toTypedProperties(properties);
258 
259       if (typed.containsKey(ConfigurationProperties.ASYNC_EXECUTOR_FACTORY)) {
260          this.asyncExecutorFactory().factoryClass(typed.getProperty(ConfigurationProperties.ASYNC_EXECUTOR_FACTORY));
261       }
262       this.asyncExecutorFactory().withExecutorProperties(typed);
263       this.balancingStrategy(typed.getProperty(ConfigurationProperties.REQUEST_BALANCING_STRATEGY, balancingStrategy.getName()));
264       this.connectionPool.withPoolProperties(typed);
265       this.connectionTimeout(typed.getIntProperty(ConfigurationProperties.CONNECT_TIMEOUT, connectionTimeout));
266       for (int i = 1; i <= consistentHashImpl.length; i++) {
267          this.consistentHashImpl(i, typed.getProperty(ConfigurationProperties.HASH_FUNCTION_PREFIX + "." + i, consistentHashImpl[i - 1].getName()));
268       }
269       this.forceReturnValues(typed.getBooleanProperty(ConfigurationProperties.FORCE_RETURN_VALUES, forceReturnValues));
270       this.keySizeEstimate(typed.getIntProperty(ConfigurationProperties.KEY_SIZE_ESTIMATE, keySizeEstimate));
271       if (typed.containsKey(ConfigurationProperties.MARSHALLER)) {
272          this.marshaller(typed.getProperty(ConfigurationProperties.MARSHALLER));
273       }
274       this.pingOnStartup(typed.getBooleanProperty(ConfigurationProperties.PING_ON_STARTUP, pingOnStartup));
275       this.protocolVersion(typed.getProperty(ConfigurationProperties.PROTOCOL_VERSION, protocolVersion));
276       this.servers.clear();
277       this.addServers(typed.getProperty(ConfigurationProperties.SERVER_LIST, ""));
278       this.socketTimeout(typed.getIntProperty(ConfigurationProperties.SO_TIMEOUT, socketTimeout));
279       this.tcpNoDelay(typed.getBooleanProperty(ConfigurationProperties.TCP_NO_DELAY, tcpNoDelay));
280       this.tcpKeepAlive(typed.getBooleanProperty(ConfigurationProperties.TCP_KEEP_ALIVE, tcpKeepAlive));
281       if (typed.containsKey(ConfigurationProperties.TRANSPORT_FACTORY)) {
282          this.transportFactory(typed.getProperty(ConfigurationProperties.TRANSPORT_FACTORY));
283       }
284       this.valueSizeEstimate(typed.getIntProperty(ConfigurationProperties.VALUE_SIZE_ESTIMATE, valueSizeEstimate));
285       this.maxRetries(typed.getIntProperty(ConfigurationProperties.MAX_RETRIES, maxRetries));
286       return this;
287    }

289    @Override
290    public void validate() {
291       connectionPool.validate();
292       asyncExecutorFactory.validate();
293       security.validate();
294       nearCache.validate();
295       if (maxRetries < 0) {
296          throw log.invalidMaxRetries(maxRetries);
297       }
298    }

300    @Override
301    public Configuration create() {
302       List<ServerConfiguration> servers = new ArrayList<ServerConfiguration>();
303       if (this.servers.size() > 0)
304          for (ServerConfigurationBuilder server : this.servers) {
305             servers.add(server.create());
306          }
307       else {
308          servers.add(new ServerConfiguration("127.0.0.1", ConfigurationProperties.DEFAULT_HOTROD_PORT));
309       }
310       if (marshaller == null) {
311          return new Configuration(asyncExecutorFactory.create(), balancingStrategy, classLoader == null ? null : classLoader.get(), connectionPool.create(), connectionTimeout,
312                consistentHashImpl, forceReturnValues, keySizeEstimate, marshallerClass, pingOnStartup, protocolVersion, servers, socketTimeout, security.create(), tcpNoDelay, tcpKeepAlive, transportFactory,
313                valueSizeEstimate, maxRetries, nearCache.create());
314       } else {
315          return new Configuration(asyncExecutorFactory.create(), balancingStrategy, classLoader == null ? null : classLoader.get(), connectionPool.create(), connectionTimeout,
316                consistentHashImpl, forceReturnValues, keySizeEstimate, marshaller, pingOnStartup, protocolVersion, servers, socketTimeout, security.create(), tcpNoDelay, tcpKeepAlive, transportFactory,
317                valueSizeEstimate, maxRetries, nearCache.create());
318       }
319    }

333    @Override
334    public ConfigurationBuilder read(Configuration template) {
335       this.classLoader = new WeakReference<ClassLoader>(template.classLoader());
336       this.asyncExecutorFactory.read(template.asyncExecutorFactory());
337       this.balancingStrategy = template.balancingStrategy();
338       this.connectionPool.read(template.connectionPool());
339       this.connectionTimeout = template.connectionTimeout();
340       for (int i = 0; i < consistentHashImpl.length; i++) {
341          this.consistentHashImpl[i] = template.consistentHashImpl()[i];
342       }
343       this.forceReturnValues = template.forceReturnValues();
344       this.keySizeEstimate = template.keySizeEstimate();
345       this.marshaller = template.marshaller();
346       this.marshallerClass = template.marshallerClass();
347       this.pingOnStartup = template.pingOnStartup();
348       this.protocolVersion = template.protocolVersion();
349       this.servers.clear();
350       for (ServerConfiguration server : template.servers()) {
351          this.addServer().host(server.host()).port(server.port());
352       }
353       this.socketTimeout = template.socketTimeout();
354       this.security.read(template.security());
355       this.tcpNoDelay = template.tcpNoDelay();
356       this.tcpKeepAlive = template.tcpKeepAlive();
357       this.transportFactory = template.transportFactory();
358       this.valueSizeEstimate = template.valueSizeEstimate();
359       this.maxRetries = template.maxRetries();
360       this.nearCache.read(template.nearCache());
361       return this;
362    }
