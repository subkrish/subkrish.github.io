20  /**
21   * Base class for all the operations that need retry logic: if the operation fails due to connection problems, try with 
22   * another available connection.
23   *
24   * @author Mircea.Markus@jboss.com
25   * @since 4.1
26   * @param T the return type of this operation
27   */
28  @Immutable
29  public abstract class RetryOnFailureOperation<T> extends HotRodOperation {
30  
31     private static final Log log = LogFactory.getLog(RetryOnFailureOperation.class, Log.class);
32  
33     protected final TransportFactory transportFactory;
34  
35     private boolean triedCompleteRestart = false;
36  
37     protected RetryOnFailureOperation(Codec codec, TransportFactory transportFactory,
38              byte[] cacheName, AtomicInteger topologyId, Flag[] flags) {
39        super(codec, flags, cacheName, topologyId);
40        this.transportFactory = transportFactory;
41     }
42  
43     @Override
44     public T execute() {
45        int retryCount = 0;
46        Set<SocketAddress> failedServers = null;
47        while (shouldRetry(retryCount)) {
48           Transport transport = null;
49           try {
50              // Transport retrieval should be retried
51              transport = getTransport(retryCount, failedServers);
52              return executeOperation(transport);
53           } catch (TransportException te) {
54              if (failedServers == null) {
55                 failedServers = new HashSet<SocketAddress>();
56              }
57              failedServers.add(te.getServerAddress());
58              // Invalidate transport since this exception means that this
59              // instance is no longer usable and should be destroyed.
60              if (transport != null) {
61                 if (log.isTraceEnabled())
62                    log.tracef("Invalidating transport %s as a result of transport exception", transport);
63  
64                 transportFactory.invalidateTransport(
65                       te.getServerAddress(), transport);
66              }
67              retryCount = logTransportErrorAndThrowExceptionIfNeeded(retryCount, te);
68           } catch (RemoteNodeSuspectException | RemoteIllegalLifecycleStateException e) {
69              // Do not invalidate transport because this exception is caused
70              // as a result of a server finding out that another node has
71              // been suspected, so there's nothing really wrong with the server
72              // from which this node was received.
73              logErrorAndThrowExceptionIfNeeded(retryCount, e);
74           } finally {
75              releaseTransport(transport);
76           }
77  
78           retryCount++;
79        }
80        throw new IllegalStateException("We should not reach here!");
81     }
82  
83     protected boolean shouldRetry(int retryCount) {
84        return retryCount <= transportFactory.getMaxRetries();
85     }
86  
87     protected int logTransportErrorAndThrowExceptionIfNeeded(int i, TransportException e) {
88        String message = "Exception encountered. Retry %d out of %d";
89        if (i >= transportFactory.getMaxRetries() || transportFactory.getMaxRetries() < 0) {
90           if (!triedCompleteRestart) {
91              log.debug("Cluster might have completely shut down, try resetting transport layer and topology id", e);
92              transportFactory.reset(cacheName);
93              triedCompleteRestart = true;
94              return -1; // reset retry count
95           } else {
96              log.exceptionAndNoRetriesLeft(i,transportFactory.getMaxRetries(), e);
97              throw e;
98           }
99        } else {
100          log.tracef(e, message, i, transportFactory.getMaxRetries());
101          return i;
102       }
103    }
104 
105    protected void logErrorAndThrowExceptionIfNeeded(int i, HotRodClientException e) {
106       String message = "Exception encountered. Retry %d out of %d";
107       if (i >= transportFactory.getMaxRetries() || transportFactory.getMaxRetries() < 0) {
108          log.exceptionAndNoRetriesLeft(i,transportFactory.getMaxRetries(), e);
109          throw e;
110       } else {
111          log.tracef(e, message, i, transportFactory.getMaxRetries());
112       }
113    }
114 
115    protected void releaseTransport(Transport transport) {
116       if (transport != null)
117          transportFactory.releaseTransport(transport);
118    }
119 
120    protected abstract Transport getTransport(int retryCount, Set<SocketAddress> failedServers);
121 
122    protected abstract T executeOperation(Transport transport);
123 }
