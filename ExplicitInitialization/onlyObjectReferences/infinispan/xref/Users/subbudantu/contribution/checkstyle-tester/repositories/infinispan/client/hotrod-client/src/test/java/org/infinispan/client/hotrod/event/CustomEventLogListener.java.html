27  @ClientListener(converterFactoryName = "test-converter-factory")
28  public abstract class CustomEventLogListener<E> {
29     BlockingQueue<E> createdCustomEvents = new ArrayBlockingQueue<>(128);
30     BlockingQueue<E> modifiedCustomEvents = new ArrayBlockingQueue<>(128);
31     BlockingQueue<E> removedCustomEvents = new ArrayBlockingQueue<>(128);
32  
33     public E pollEvent(ClientEvent.Type type) {
34        try {
35           E event = queue(type).poll(10, TimeUnit.SECONDS);
36           assertNotNull(event);
37           return event;
38        } catch (InterruptedException e) {
39           throw new AssertionError(e);
40        }
41     }
42  
43     protected BlockingQueue<E> queue(ClientEvent.Type type) {
44        switch (type) {
45           case CLIENT_CACHE_ENTRY_CREATED: return createdCustomEvents;
46           case CLIENT_CACHE_ENTRY_MODIFIED: return modifiedCustomEvents;
47           case CLIENT_CACHE_ENTRY_REMOVED: return removedCustomEvents;
48           default: throw new IllegalArgumentException("Unknown event type: " + type);
49        }
50     }
51  
52     public void expectNoEvents(ClientEvent.Type type) {
53        assertEquals(0, queue(type).size());
54     }
55  
56     public void expectNoEvents() {
57        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_CREATED);
58        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_MODIFIED);
59        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_REMOVED);
60     }
61  
62     public void expectSingleCustomEvent(ClientEvent.Type type, E expected) {
63        E event = pollEvent(type);
64        assertAnyEquals(expected, event);
65     }
66     
67     public void expectCreatedEvent(E expected) {
68        expectSingleCustomEvent(ClientEvent.Type.CLIENT_CACHE_ENTRY_CREATED, expected);
69        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_MODIFIED);
70        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_REMOVED);
71     }
72  
73     public void expectModifiedEvent(E expected) {
74        expectSingleCustomEvent(ClientEvent.Type.CLIENT_CACHE_ENTRY_MODIFIED, expected);
75        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_CREATED);
76        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_REMOVED);
77     }
78  
79     public void expectRemovedEvent(E expected) {
80        expectSingleCustomEvent(ClientEvent.Type.CLIENT_CACHE_ENTRY_REMOVED, expected);
81        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_CREATED);
82        expectNoEvents(ClientEvent.Type.CLIENT_CACHE_ENTRY_MODIFIED);
83     }

106    public static final class CustomEvent implements Serializable {
107       final Integer key;
108       final String value;
109       final long timestamp;
110       final int counter;
111 
112       public CustomEvent(Integer key, String value, int counter) {
113          this.key = key;
114          this.value = value;
115          this.timestamp = System.nanoTime();
116          this.counter = counter;
117       }
118 
119       @Override
120       public boolean equals(Object o) {
121          if (this == o) return true;
122          if (o == null || getClass() != o.getClass()) return false;
123 
124          CustomEvent that = (CustomEvent) o;
125 
126          if (counter != that.counter) return false;
127          if (!key.equals(that.key)) return false;
128          return !(value != null ? !value.equals(that.value) : that.value != null);
129       }
130 
131       @Override
132       public int hashCode() {
133          int result = key.hashCode();
134          result = 31 * result + (value != null ? value.hashCode() : 0);
135          result = 31 * result + counter;
136          return result;
137       }
138 
139       @Override
140       public String toString() {
141          return "CustomEvent{" +
142                "key=" + key +
143                ", value='" + value + '\'' +
144                ", timestamp=" + timestamp +
145                ", counter=" + counter +
146                '}';
147       }
148    }

150    @ClientListener(converterFactoryName = "static-converter-factory")
151    public static class StaticCustomEventLogListener extends CustomEventLogListener<CustomEvent> {
152 
153       @Override
154       public void expectSingleCustomEvent(ClientEvent.Type type, CustomEvent expected) {
155          CustomEvent event = pollEvent(type);
156          assertNotNull(event.key);
157          assertNotNull(event.timestamp); // check only custom field, value can be null
158          assertAnyEquals(expected, event);
159       }
160       
161       public void expectOrderedEventQueue(ClientEvent.Type type) {
162          BlockingQueue<CustomEvent> queue = queue(type);
163          if (queue.size() < 2) 
164             return;
165          
166          try {
167             CustomEvent before = queue.poll(10, TimeUnit.SECONDS);
168             Iterator<CustomEvent> iter = queue.iterator();
169             while (iter.hasNext()) {
170                CustomEvent after = iter.next();
171                expectTimeOrdered(before, after);
172                before = after;
173             }
174          } catch (InterruptedException e) {
175             throw new AssertionError(e);
176          }
177       }
178       
179       private void expectTimeOrdered(CustomEvent before, CustomEvent after) {
180          assertTrue("Before timestamp=" + before.timestamp + ", after timestamp=" + after.timestamp,
181             before.timestamp < after.timestamp);
182       }
183    }

260    public static final class NumericCallbackCounter implements CallbackCounter {
261       int count = 0;

265          count++;
266       }

280    public static class FilterConverterFactory implements CacheEventFilterConverterFactory {
281 
282       @Override
283       public CacheEventFilterConverter<Integer, String, CustomEvent> getFilterConverter(Object[] params) {
284          return new FilterConverter(params);
285       }
286 
287       static class FilterConverter extends AbstractCacheEventFilterConverter<Integer, String, CustomEvent>
288          implements Serializable {
289          private final Object[] params;
290          private final CallbackCounter counter = new NumericCallbackCounter();
291 
292          public FilterConverter(Object[] params) {
293             this.params = params;
294          }
295 
296          @Override
297          public CustomEvent filterAndConvert(Integer key, String oldValue, Metadata oldMetadata,
298             String newValue, Metadata newMetadata, EventType eventType) {
299             counter.incr();
300             if (params[0].equals(key))
301                return new CustomEvent(key, null, counter.get());
302 
303             return new CustomEvent(key, newValue, counter.get());
304          }
305       }

311    static byte[] concat(byte[] a, byte[] b) {
312       int aLen = a.length;
313       int bLen = b.length;
314       byte[] ret = new byte[aLen + bLen];
315       System.arraycopy(a, 0, ret, 0, aLen);
316       System.arraycopy(b, 0, ret, aLen, bLen);
317       return ret;
318    }
