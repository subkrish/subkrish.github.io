31   * Common parent for both embedded and standalone JBoss Marshalling-based marshallers.
32   *
33   * @author Galder Zamarre√±o

38  public abstract class AbstractJBossMarshaller extends AbstractMarshaller implements StreamingMarshaller {

40     protected static final Log log = LogFactory.getLog(AbstractJBossMarshaller.class);
41     protected static final boolean trace = log.isTraceEnabled();
42     protected static final JBossMarshallerFactory factory = new JBossMarshallerFactory();
43     protected static final int DEF_INSTANCE_COUNT = 16;
44     protected static final int DEF_CLASS_COUNT = 8;
45     private static final int PER_THREAD_REUSABLE_INSTANCES = 6;
46     private static final int RIVER_INTERNAL_BUFFER = 512;
47  
48     protected final MarshallingConfiguration baseCfg;
49  
50     /**
51      * Marshaller thread local. In non-internal marshaller usages, such as Java
52      * Hot Rod client, this is a singleton shared by all so no urgent need for
53      * static here. JBMAR clears pretty much any state during finish(), so no
54      * urgent need to clear the thread local since it shouldn't be leaking.
55      * It might take a long time to warmup and pre-initialize all needed instances!
56      */
57     private final ConcurrentWeakKeyHashMap<Thread, PerThreadInstanceHolder> marshallerTL =
58           new ConcurrentWeakKeyHashMap<Thread, PerThreadInstanceHolder>();
59  
60     public AbstractJBossMarshaller() {
61        // Class resolver now set when marshaller/unmarshaller will be created
62        baseCfg = new MarshallingConfiguration();
63        baseCfg.setExceptionListener(new DebuggingExceptionListener());
64        baseCfg.setClassExternalizerFactory(new SerializeWithExtFactory());
65        baseCfg.setInstanceCount(DEF_INSTANCE_COUNT);
66        baseCfg.setClassCount(DEF_CLASS_COUNT);
67        baseCfg.setVersion(3);
68     }

70     @Override
71     final public void objectToObjectStream(final Object obj, final ObjectOutput out) throws IOException {
72        out.writeObject(obj);
73     }
74  
75     @Override
76     final protected ByteBuffer objectToBuffer(final Object o, final int estimatedSize) throws IOException {
77        ExposedByteArrayOutputStream baos = new ExposedByteArrayOutputStream(estimatedSize);
78        ObjectOutput marshaller = startObjectOutput(baos, false, estimatedSize);
79        try {
80           objectToObjectStream(o, marshaller);
81        } finally {
82           finishObjectOutput(marshaller);
83        }
84        return new ByteBufferImpl(baos.getRawBuffer(), 0, baos.size());
85     }
86  
87     @Override
88     final public ObjectOutput startObjectOutput(final OutputStream os, final boolean isReentrant, final int estimatedSize) throws IOException {
89        PerThreadInstanceHolder instanceHolder = getPerThreadInstanceHolder();
90        org.jboss.marshalling.Marshaller marshaller = instanceHolder.getMarshaller(estimatedSize);
91        marshaller.start(Marshalling.createByteOutput(os));
92        return marshaller;
93     }
94  
95     @Override
96     final public void finishObjectOutput(final ObjectOutput oo) {
97        try {
98           if (trace) log.trace("Stop marshaller");
99  
100          ((org.jboss.marshalling.Marshaller) oo).finish();
101       } catch (IOException ignored) {
102       }
103    }
104 
105    @Override
106    final public Object objectFromByteBuffer(final byte[] buf, final int offset, final int length) throws IOException,
107            ClassNotFoundException {
108       ByteArrayInputStream is = new ByteArrayInputStream(buf, offset, length);
109       ObjectInput unmarshaller = startObjectInput(is, false);
110       Object o = null;
111       try {
112          o = objectFromObjectStream(unmarshaller);
113       } finally {
114          finishObjectInput(unmarshaller);
115       }
116       return o;
117    }
118 
119    @Override
120    final public ObjectInput startObjectInput(final InputStream is, final boolean isReentrant) throws IOException {
121       PerThreadInstanceHolder instanceHolder = getPerThreadInstanceHolder();
122       Unmarshaller unmarshaller = instanceHolder.getUnmarshaller();
123 
124       if (trace)
125          log.tracef("Start unmarshaller after retrieving marshaller from %s",
126                    isReentrant ? "factory" : "thread local");
127 
128       unmarshaller.start(Marshalling.createByteInput(is));
129       return unmarshaller;
130    }
131 
132    @Override
133    final public Object objectFromObjectStream(final ObjectInput in) throws IOException, ClassNotFoundException {
134       return in.readObject();
135    }
136 
137    @Override
138    final public void finishObjectInput(final ObjectInput oi) {
139       try {
140          if (trace)
141             log.trace("Stop unmarshaller");
142 
143          if (oi != null) ((Unmarshaller) oi).finish();
144       } catch (IOException ignored) {
145       }
146    }

148    @Override
149    public boolean isMarshallable(Object o) throws Exception {
150       Class<?> clazz = o.getClass();
151       boolean containsMarshallable = marshallableTypeHints.isKnownMarshallable(clazz);
152       if (containsMarshallable) {
153          boolean marshallable = marshallableTypeHints.isMarshallable(clazz);
154          if (trace)
155             log.tracef("Marshallable type '%s' known and is marshallable=%b",
156                clazz.getName(), marshallable);
157 
158          return marshallable;
159       } else {
160          if (isMarshallableCandidate(o)) {
161             boolean isMarshallable = true;
162             try {
163                objectToBuffer(o);
164             } catch (Exception e) {
165                isMarshallable = false;
166                throw e;
167             } finally {
168                marshallableTypeHints.markMarshallable(clazz, isMarshallable);
169             }
170             return isMarshallable;
171          }
172          return false;
173       }
174    }

177    public void start() {
178       // No-op

182    public void stop() {
183        // Clear class cache
184       marshallableTypeHints.clear();
185       marshallerTL.clear();

188    protected boolean isMarshallableCandidate(Object o) {
189       return o instanceof Serializable;
190    }

192    private PerThreadInstanceHolder getPerThreadInstanceHolder() {
193       final Thread thread = Thread.currentThread();
194       PerThreadInstanceHolder holder = marshallerTL.get(thread);
195       if (holder == null) {
196          holder = new PerThreadInstanceHolder(baseCfg.clone());
197          if (marshallerTL.put(thread, holder) != null) {
198             throw new IllegalStateException();
199          }
200       }
201       return holder;
202    }

204    protected static final class DebuggingExceptionListener implements ExceptionListener {
205       private static final URL[] EMPTY_URLS = {};
206 
207       @Override
208       public void handleMarshallingException(final Throwable problem, final Object subject) {
209          if (log.isDebugEnabled()) {
210             TraceInformation.addUserInformation(problem, "toString = " + subject.toString());
211          }
212       }
213 
214       @Override
215       public void handleUnmarshallingException(final Throwable problem, final Class<?> subjectClass) {
216          if (log.isDebugEnabled()) {
217             StringBuilder builder = new StringBuilder();
218             ClassLoader cl = subjectClass.getClassLoader();
219             builder.append("classloader hierarchy:");
220             ClassLoader parent = cl;
221             while (parent != null) {
222                if (parent.equals(cl)) {
223                   builder.append("\n\t\t-> type classloader = ").append(parent);
224                } else {
225                   builder.append("\n\t\t-> parent classloader = ").append(parent);
226                }
227                URL[] urls = getClassLoaderURLs(parent);
228 
229                if (urls != null) {
230                   for (URL u : urls) builder.append("\n\t\t->...").append(u);
231                }
232 
233                parent = parent.getParent();
234             }
235             TraceInformation.addUserInformation(problem, builder.toString());
236          }
237       }
238 
239       @Override
240       public void handleUnmarshallingException(Throwable problem) {
241          // no-op
242       }
243 
244       private static URL[] getClassLoaderURLs(final ClassLoader cl) {
245          URL[] urls = EMPTY_URLS;
246          try {
247             Class<?> returnType = urls.getClass();
248             Class<?>[] parameterTypes = EMPTY_CLASS_ARRAY;
249             Method getURLs = cl.getClass().getMethod("getURLs", parameterTypes);
250             if (returnType.isAssignableFrom(getURLs.getReturnType())) {
251                Object[] args = EMPTY_OBJECT_ARRAY;
252                urls = (URL[]) getURLs.invoke(cl, args);
253             }
254          } catch (Exception ignore) {
255          }
256          return urls;
257       }

261    private static final class PerThreadInstanceHolder implements RiverCloseListener {
262 
263       final MarshallingConfiguration configuration;
264       final ExtendedRiverMarshaller[] reusableMarshaller = new ExtendedRiverMarshaller[PER_THREAD_REUSABLE_INSTANCES];
265       int availableMarshallerIndex = 0;
266       final ExtendedRiverUnmarshaller[] reusableUnMarshaller = new ExtendedRiverUnmarshaller[PER_THREAD_REUSABLE_INSTANCES];
267       int availableUnMarshallerIndex = 0;
268 
269       PerThreadInstanceHolder(final MarshallingConfiguration threadDedicatedConfiguration) {
270          this.configuration = threadDedicatedConfiguration;
271       }
272 
273       Unmarshaller getUnmarshaller() throws IOException {
274          //as opposing to getMarshaller(int), in this case we don't have a good hint about initial buffer sizing
275          if (availableUnMarshallerIndex == PER_THREAD_REUSABLE_INSTANCES) {
276             //we're above the pool threshold: make a throw-away-after usage Marshaller
277             configuration.setBufferSize(512);//reset to default as it might be changed by getMarshaller
278             return factory.createUnmarshaller(configuration);
279          }
280          else {
281             ExtendedRiverUnmarshaller unMarshaller = reusableUnMarshaller[availableUnMarshallerIndex];
282             if (unMarshaller != null) {
283                availableUnMarshallerIndex++;
284                return unMarshaller;
285             }
286             else {
287                configuration.setBufferSize(RIVER_INTERNAL_BUFFER);//reset to default as it might be changed by getMarshaller
288                unMarshaller = factory.createUnmarshaller(configuration);
289                unMarshaller.setCloseListener(this);
290                reusableUnMarshaller[availableUnMarshallerIndex] = unMarshaller;
291                availableUnMarshallerIndex++;
292                return unMarshaller;
293             }
294          }
295       }
296 
297       ExtendedRiverMarshaller getMarshaller(int estimatedSize) throws IOException {
298          if (availableMarshallerIndex == PER_THREAD_REUSABLE_INSTANCES) {
299             //we're above the pool threshold: make a throw-away-after usage Marshaller
300             //setting the buffer as cheap as possible:
301             configuration.setBufferSize(estimatedSize);
302             return factory.createMarshaller(configuration);
303          }
304          else {
305             ExtendedRiverMarshaller marshaller = reusableMarshaller[availableMarshallerIndex];
306             if (marshaller != null) {
307                availableMarshallerIndex++;
308                return marshaller;
309             }
310             else {
311                //we're going to pool this one, make sure the buffer size is set to a reasonable value
312                //as we might have changed it previously:
313                configuration.setBufferSize(RIVER_INTERNAL_BUFFER);
314                marshaller = factory.createMarshaller(configuration);
315                marshaller.setCloseListener(this);
316                reusableMarshaller[availableMarshallerIndex] = marshaller;
317                availableMarshallerIndex++;
318                return marshaller;
319             }
320          }
321       }
322 
323       @Override
324       public void closeMarshaller() {
325          availableMarshallerIndex--;
326       }
327 
328       @Override
329       public void closeUnmarshaller() {
330          availableUnMarshallerIndex--;
331       }
332    }
