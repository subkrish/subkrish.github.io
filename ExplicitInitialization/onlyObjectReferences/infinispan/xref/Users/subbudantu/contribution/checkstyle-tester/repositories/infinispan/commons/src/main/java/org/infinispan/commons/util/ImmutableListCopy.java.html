23  /**
24   * A lightweight, read-only copy of a List.  Typically used in place of the common idiom: <code> return
25   * Collections.unmodifiableList(new ArrayList( myInternalList )); </code>
26   * <p/>
27   * a it is far more efficient than making a defensive copy and then wrapping the defensive copy in a read-only wrapper.
28   * <p/>
29   * Also used whenever a read-only reference List is needed.
30   * <p/>
31   *
32   * @author Manik Surtani (<a href="mailto:manik@jboss.org">manik@jboss.org</a>)
33   * @since 4.0
34   */
35  @Immutable
36  public class ImmutableListCopy<E> extends AbstractList<E> implements Externalizable, Immutables.Immutable {
37     private static final long serialVersionUID = 10929568968966L;
38     private E[] elements;
39  
40     /**
41      * Constructs a new ImmutableListCopy. Required by Serialization.
42      */
43     public ImmutableListCopy() {
44     }
45  
46     /**
47      * Only one copy constructor since the list is immutable.
48      *
49      * @param c collection to copy from
50      */
51     @SuppressWarnings("unchecked")
52     public ImmutableListCopy(Collection<? extends E> c) {
53        int size = c.size();
54        Object[] el = new Object[size]; // no room for growth;
55        el = c.toArray(el);
56        elements = (E[]) el;
57     }
58  
59     /**
60      * Assumes that the array passed in is "safe", i.e., is not referenced from elsewhere.  Use with care!
61      *
62      * @param array to reference
63      */
64     public ImmutableListCopy(E[] array) {
65        elements = array;
66     }
67  
68     /**
69      * Utility constructors to allow combining collections
70      *
71      * @param collection1 collection to copy from
72      * @param collection2 collection to copy from
73      */
74     @SuppressWarnings("unchecked")
75     public ImmutableListCopy(Collection<? extends E> collection1, Collection<? extends E> collection2) {
76        int size = collection1.size() + collection2.size();
77        elements = (E[]) new Object[size]; // no room for growth;
78        Object[] c1 = new Object[collection1.size()];
79        Object[] c2 = new Object[collection2.size()];
80        c1 = collection1.toArray(c1);
81        c2 = collection2.toArray(c2);
82        System.arraycopy(c1, 0, elements, 0, c1.length);
83        System.arraycopy(c2, 0, elements, c1.length, c2.length);
84     }

107    public final Object[] toArray() {
108       int size = elements.length;

114    @Override
115    @SuppressWarnings("unchecked")
116    public final <T> T[] toArray(T[] a) {
117       int size = elements.length;
118       if (a.length < size) {
119          a = (T[]) Array.newInstance(a.getClass().getComponentType(), size);
120       }
121       System.arraycopy(elements, 0, a, 0, size);
122       if (a.length > size) a[size] = null;
123       return a;
124    }

161    @Override
162    public final int indexOf(Object o) {
163       int size = elements.length;
164       if (o == null) {
165          for (int i = 0; i < size; i++) {
166             if (elements[i] == null) return i;
167          }
168       } else {
169          for (int i = 0; i < size; i++) {
170             if (o.equals(elements[i])) return i;
171          }
172       }
173       return -1;
174    }
175 
176    @Override
177    public final int lastIndexOf(Object o) {
178       int size = elements.length;
179       if (o == null) {
180          for (int i = size - 1; i >= 0; i--) {
181             if (elements[i] == null) return i;
182          }
183       } else {
184          for (int i = size - 1; i >= 0; i--) {
185             if (o.equals(elements[i])) return i;
186          }
187       }
188       return -1;
189    }

206    @Override
207    public boolean equals(Object o) {
208       if (this == o)
209          return true;
210       if (!(o instanceof ImmutableListCopy))
211          return super.equals(o);
212 
213       ImmutableListCopy<?> that = (ImmutableListCopy<?>) o;
214       return Arrays.equals(elements, that.elements);
215 
216    }

252    private class ImmutableIterator implements ListIterator<E> {
253       int cursor = 0;
254 
255       ImmutableIterator(int index) {
256          if (index < 0 || index > size()) throw new IndexOutOfBoundsException("Index: " + index);
257          cursor = index;
258       }
259 
260       ImmutableIterator() {
261       }
262 
263       @Override
264       public boolean hasNext() {
265          return cursor != elements.length;
266       }
267 
268       @Override
269       public E next() {
270          try {
271             return get(cursor++);
272          }
273          catch (IndexOutOfBoundsException e) {
274             throw new NoSuchElementException();
275          }
276       }
277 
278       @Override
279       public void remove() {
280          throw new UnsupportedOperationException();
281       }
282 
283       @Override
284       public boolean hasPrevious() {
285          return cursor != 0;
286       }
287 
288       @Override
289       public E previous() {
290          try {
291             return get(--cursor);
292          }
293          catch (IndexOutOfBoundsException e) {
294             throw new NoSuchElementException();
295          }
296       }
297 
298       @Override
299       public int nextIndex() {
300          return cursor;
301       }
302 
303       @Override
304       public int previousIndex() {
305          return cursor - 1;
306       }
307 
308       @Override
309       public void set(E o) {
310          throw new UnsupportedOperationException();
311       }
312 
313       @Override
314       public void add(E o) {
315          throw new UnsupportedOperationException();
316       }
317    }

319    private class ImmutableSubList<E> extends AbstractList<E> {
320       private int offset;
321       private int size;
322 
323       ImmutableSubList(int fromIndex, int toIndex) {
324          int size = ImmutableListCopy.this.elements.length;
325          if (fromIndex < 0 || toIndex > size || fromIndex > toIndex)
326             throw new IllegalArgumentException("fromIndex(" + fromIndex + "), toIndex(" + toIndex + "), size (" + size + "), List=" + ImmutableListCopy.this.toString());
327          offset = fromIndex;
328          this.size = toIndex - fromIndex;
329       }
330 
331       @Override
332       @SuppressWarnings("unchecked")
333       public final E get(int index) {
334          if (index < 0 || index >= size) throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
335          return (E) ImmutableListCopy.this.get(index + offset);
336       }
337 
338       @Override
339       public final int size() {
340          return size;
341       }
342 
343       @Override
344       protected final void removeRange(int fromIndex, int toIndex) {
345          throw new UnsupportedOperationException();
346       }
347 
348       @Override
349       public final boolean addAll(Collection<? extends E> c) {
350          throw new UnsupportedOperationException();
351       }
352 
353       @Override
354       public final boolean addAll(int index, Collection<? extends E> c) {
355          throw new UnsupportedOperationException();
356       }
357 
358       @Override
359       public final Iterator<E> iterator() {
360          return super.listIterator();
361       }
362 
363       @Override
364       public final ListIterator<E> listIterator(final int index) {
365          if (index < 0 || (index != 0 && index >= size))
366             throw new IndexOutOfBoundsException("Index: " + index + ", Size: " + size);
367 
368          return new ListIterator<E>() {
369             private ListIterator<?> i = ImmutableListCopy.this.listIterator(index + offset);
370 
371             @Override
372             public boolean hasNext() {
373                return nextIndex() < size;
374             }
375 
376             @Override
377             @SuppressWarnings("unchecked")
378             public E next() {
379                if (hasNext())
380                   return (E) i.next();
381                else
382                   throw new NoSuchElementException();
383             }
384 
385             @Override
386             public boolean hasPrevious() {
387                return previousIndex() >= 0;
388             }
389 
390             @Override
391             @SuppressWarnings("unchecked")
392             public E previous() {
393                if (hasPrevious())
394                   return (E) i.previous();
395                else
396                   throw new NoSuchElementException();
397             }
398 
399             @Override
400             public int nextIndex() {
401                return i.nextIndex() - offset;
402             }
403 
404             @Override
405             public int previousIndex() {
406                return i.previousIndex() - offset;
407             }
408 
409             @Override
410             public void remove() {
411                throw new UnsupportedOperationException();
412             }
413 
414             @Override
415             public void set(E o) {
416                throw new UnsupportedOperationException();
417             }
418 
419             @Override
420             public void add(E o) {
421                throw new UnsupportedOperationException();
422             }
423          };
424       }
425 
426       @Override
427       public final List<E> subList(int fromIndex, int toIndex) {
428          return new ImmutableSubList<E>(offset + fromIndex, offset + toIndex);
429       }
430    }

432    @Immutable
433    public static class Externalizer extends AbstractExternalizer<List> {
434 
435       @Override
436       public void writeObject(ObjectOutput output, List list) throws IOException {
437          int size = list.size();
438          UnsignedNumeric.writeUnsignedInt(output, size);
439          for (int i = 0; i < size; i++) {
440             output.writeObject(list.get(i));
441          }
442       }
443 
444       @Override
445       public List readObject(ObjectInput input) throws IOException, ClassNotFoundException {
446          int size = UnsignedNumeric.readUnsignedInt(input);
447          Object[] elements = new Object[size];
448          for (int i = 0; i < size; i++)
449             elements[i] = input.readObject();
450 
451          return new ImmutableListCopy(elements);
452       }
453 
454       @Override
455       public Integer getId() {
456          return Ids.IMMUTABLE_LIST;
457       }
458 
459       @Override
460       public Set<Class<? extends List>> getTypeClasses() {
461          return Util.<Class<? extends List>>asSet(ImmutableListCopy.class, ImmutableListCopy.ImmutableSubList.class);
462       }
