41  @ThreadSafe
42  public class KeyAffinityServiceImpl<K> implements KeyAffinityService<K> {
43  
44     // TODO During state transfer, we should try to assign keys to a node only if they are owners in both CHs
45     public final static float THRESHOLD = 0.5f;
46  
47     //interval between key/queue poll
48     private static final int POLL_INTERVAL_MILLIS = 50;
49  
50     private static final Log log = LogFactory.getLog(KeyAffinityServiceImpl.class);
51  
52     private final Set<Address> filter;
53  
54     @GuardedBy("maxNumberInvariant")
55     private final Map<Address, BlockingQueue<K>> address2key = CollectionFactory.makeConcurrentMap();
56     private final Executor executor;
57     private final Cache<? extends K, ?> cache;
58     private final KeyGenerator<? extends K> keyGenerator;
59     private final int bufferSize;
60     private final AtomicInteger maxNumberOfKeys = new AtomicInteger(); //(nr. of addresses) * bufferSize;
61     final AtomicInteger existingKeyCount = new AtomicInteger();
62  
63     private volatile boolean started;
64  
65     /**
66      * Guards and make sure the following invariant stands:  maxNumberOfKeys ==  address2key.keys().size() * bufferSize
67      */
68     private final ReadWriteLock maxNumberInvariant = new ReentrantReadWriteLock();
69  
70     /**
71      * Used for coordinating between the KeyGeneratorWorker and consumers.
72      */
73     private final ReclosableLatch keyProducerStartLatch = new ReclosableLatch();
74     private volatile KeyGeneratorWorker keyGenWorker;
75     private volatile ListenerRegistration listenerRegistration;
76  
77  
78     public KeyAffinityServiceImpl(Executor executor, Cache<? extends K, ?> cache, KeyGenerator<? extends K> keyGenerator,
79                                   int bufferSize, Collection<Address> filter, boolean start) {
80        this.executor = executor;
81        this.cache = cache;
82        this.keyGenerator = keyGenerator;
83        this.bufferSize = bufferSize;
84        if (filter != null) {
85           this.filter = new ConcurrentHashSet<Address>();
86           for (Address address : filter) {
87              this.filter.add(address);
88           }
89        } else {
90           this.filter = null;
91        }
92        if (start)
93           start();
94     }

96     @Override
97     public K getCollocatedKey(K otherKey) {
98        Address address = getAddressForKey(otherKey);
99        return getKeyForAddress(address);
100    }

102    @Override
103    public K getKeyForAddress(Address address) {
104       if (!started) {
105          throw new IllegalStateException("You have to start the service first!");
106       }
107       if (address == null)
108          throw new NullPointerException("Null address not supported!");
109 
110       BlockingQueue<K> queue = null;
111       try {
112          K result = null;
113          while (result == null && !keyGenWorker.isStopped()) {
114             // obtain the read lock inside the loop, otherwise a topology change will never be able
115             // to obtain the write lock
116             maxNumberInvariant.readLock().lock();
117             try {
118                queue = address2key.get(address);
119                if (queue == null)
120                   throw new IllegalStateException("Address " + address + " is no longer in the cluster");
121 
122                // first try to take an element without waiting
123                result = queue.poll();
124                if (result == null) {
125                   // there are no elements in the queue, make sure the producer is started
126                   keyProducerStartLatch.open();
127                   // our address might have been removed from the consistent hash
128                   if (!isNodeInConsistentHash(address))
129                      throw new IllegalStateException("Address " + address + " is no longer in the cluster");
130                }
131             } finally {
132                maxNumberInvariant.readLock().unlock();
133             }
134 
135             if (result == null) {
136                // Now wait for a new key. If there's a topology change, poll() will time out and we'll retry
137                // on the new queue.
138                try {
139                   result = queue.poll(POLL_INTERVAL_MILLIS, TimeUnit.MILLISECONDS);
140                } catch (InterruptedException e) {
141                   // Ignore and restore interruption status
142                   Thread.currentThread().interrupt();
143                }
144             }
145          }
146          existingKeyCount.decrementAndGet();
147          log.tracef("Returning key %s for address %s", result, address);
148          return result;
149       } finally {
150          if (queue != null && queue.size() < bufferSize * THRESHOLD + 1) {
151             keyProducerStartLatch.open();
152          }
153       }
154    }

156    @Override
157    public void start() {
158       if (started) {
159          log.debug("Service already started, ignoring call to start!");
160          return;
161       }
162       List<Address> existingNodes = getExistingNodes();
163       maxNumberInvariant.writeLock().lock();
164       try {
165          addQueuesForAddresses(existingNodes);
166          resetNumberOfKeys();
167       } finally {
168          maxNumberInvariant.writeLock().unlock();
169       }
170       keyGenWorker = new KeyGeneratorWorker();
171       executor.execute(keyGenWorker);
172       listenerRegistration = new ListenerRegistration(this);
173       cache.getCacheManager().addListener(listenerRegistration);
174       cache.addListener(listenerRegistration);
175       keyProducerStartLatch.open();
176       started = true;
177    }
178 
179    @Override
180    public void stop() {
181       if (!started) {
182          log.debug("Ignoring call to stop as service is not started.");
183          return;
184       }
185       started = false;
186       EmbeddedCacheManager cacheManager = cache.getCacheManager();
187       if (cacheManager.getListeners().contains(listenerRegistration)) {
188          cacheManager.removeListener(listenerRegistration);
189       } else {
190          throw new IllegalStateException("Listener must have been registered!");
191       }
192       //most likely the listeners collection is shared between CacheManager and the Cache
193       if (cache.getListeners().contains(listenerRegistration)) {
194          cache.removeListener(listenerRegistration);
195       }
196       keyGenWorker.stop();
197    }

199    public void handleViewChange(TopologyChangedEvent<?, ?> vce) {
200       log.tracef("TopologyChangedEvent received: %s", vce);
201       maxNumberInvariant.writeLock().lock();
202       try {
203          address2key.clear(); //we need to drop everything as key-mapping data is stale due to view change
204          addQueuesForAddresses(vce.getConsistentHashAtEnd().getMembers());
205          resetNumberOfKeys();
206          keyProducerStartLatch.open();
207       } finally {
208          maxNumberInvariant.writeLock().unlock();
209       }
210    }
211 
212    public boolean isKeyGeneratorThreadAlive() {
213       return !keyGenWorker.isStopped();
214    }
215 
216    public void handleCacheStopped(CacheStoppedEvent cse) {
217       if (this.cache.getName().equals(cse.getCacheName())) {
218          log.tracef("Cache stopped, stopping the service: %s", cse);
219          stop();
220       }
221    }

224    private class KeyGeneratorWorker implements Runnable {
225 
226       private volatile boolean isActive;
227       private volatile boolean isStopped = false;
228 
229       @Override
230       public void run() {
231          try {
232             while (isStopped == false) {
233                keyProducerStartLatch.await();
234                if (isStopped == false) {
235                   isActive = true;
236                   log.trace("KeyGeneratorWorker marked as ACTIVE");
237                   generateKeys();
238 
239                   isActive = false;
240                   log.trace("KeyGeneratorWorker marked as INACTIVE");
241                }
242             }
243          } catch (InterruptedException e) {
244             Thread.currentThread().interrupt();
245          }
246          finally {
247             log.debugf("Shutting down KeyAffinity service for key set: %s", filter);
248          }
249       }
250 
251       public boolean isStopped() {
252          return isStopped;
253       }
254 
255       private void generateKeys() {
256          maxNumberInvariant.readLock().lock();
257          try {
258             // if there's a topology change, some queues will stop receiving keys
259             // so we want to establish an upper bound on how many extra keys to generate
260             // in order to fill all the queues
261             int maxMisses = maxNumberOfKeys.get();
262             int missCount = 0;
263             while (existingKeyCount.get() < maxNumberOfKeys.get() && missCount < maxMisses) {
264                K key = keyGenerator.getKey();
265                Address addressForKey = getAddressForKey(key);
266                boolean added = false;
267                if (interestedInAddress(addressForKey)) {
268                   added = tryAddKey(addressForKey, key);
269                }
270                if (!added) missCount++;
271             }
272 
273             // if we had too many misses, just release the lock and try again
274             if (missCount < maxMisses) {
275                keyProducerStartLatch.close();
276             }
277          } finally {
278             maxNumberInvariant.readLock().unlock();
279          }
280       }
281 
282       private boolean tryAddKey(Address address, K key) {
283          BlockingQueue<K> queue = address2key.get(address);
284          // on node stop the distribution manager might still return the dead server for a while after we have already removed its queue
285          if (queue == null)
286             return false;
287 
288          boolean added = queue.offer(key);
289          if (added) {
290             existingKeyCount.incrementAndGet();
291             log.tracef("Added key %s for address %s", key, address);
292          }
293          return added;
294       }
295 
296       public boolean isActive() {
297          return isActive;
298       }
299 
300       public void stop() {
301          isStopped = true;
302          keyProducerStartLatch.open();
303       }
304    }

306    /**
307     * Important: this *MUST* be called with WL on {@link #address2key}.
308     */
309    private void resetNumberOfKeys() {
310       maxNumberOfKeys.set(address2key.keySet().size() * bufferSize);
311       existingKeyCount.set(0);
312       if (log.isTraceEnabled()) {
313          log.tracef("resetNumberOfKeys ends with: maxNumberOfKeys=%s, existingKeyCount=%s",
314                     maxNumberOfKeys.get(), existingKeyCount.get());
315       }
316    }
317 
318    /**
319     * Important: this *MUST* be called with WL on {@link #address2key}.
320     */
321    private void addQueuesForAddresses(Collection<Address> addresses) {
322       for (Address address : addresses) {
323          if (interestedInAddress(address)) {
324             address2key.put(address, new ArrayBlockingQueue<K>(bufferSize));
325          } else {
326             log.tracef("Skipping address: %s", address);
327          }
328       }
329    }
330 
331    private boolean interestedInAddress(Address address) {
332       return filter == null || filter.contains(address);
333    }
334 
335    private List<Address> getExistingNodes() {
336       return cache.getAdvancedCache().getRpcManager().getTransport().getMembers();
337    }
338 
339    private Address getAddressForKey(Object key) {
340       DistributionManager distributionManager = getDistributionManager();
341       ConsistentHash hash = distributionManager.getConsistentHash();
342       return hash.locatePrimaryOwner(key);
343    }
344 
345    private boolean isNodeInConsistentHash(Address address) {
346       DistributionManager distributionManager = getDistributionManager();
347       ConsistentHash hash = distributionManager.getConsistentHash();
348       return hash.getMembers().contains(address);
349    }
350    private DistributionManager getDistributionManager() {
351       DistributionManager distributionManager = cache.getAdvancedCache().getDistributionManager();
352       if (distributionManager == null) {
353          throw new IllegalStateException("Null distribution manager. Is this an distributed(v.s. replicated) cache?");
354       }
355       return distributionManager;
356    }
