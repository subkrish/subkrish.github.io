27  /**
28   * If {@link org.infinispan.configuration.cache.Configuration#jmxStatistics()} is enabled, then class will register all
29   * the MBeans from cache local's ConfigurationRegistry to the MBean server.
30   *
31   * @author Mircea.Markus@jboss.com
32   * @author Galder Zamarre√±o
33   * @see java.lang.management.ManagementFactory#getPlatformMBeanServer()
34   * @since 4.0
35   */
36  @SurvivesRestarts
37  public class CacheJmxRegistration extends AbstractJmxRegistration {
38     private static final Log log = LogFactory.getLog(CacheJmxRegistration.class);
39     public static final String CACHE_JMX_GROUP = "type=Cache";
40  
41     private AdvancedCache<?, ?> cache;
42     private Set<Component> nonCacheComponents;
43     private boolean needToUnregister = false;
44  
45     private volatile boolean unregisterCacheMBean;
46  
47     @Inject
48     public void initialize(Cache<?, ?> cache, GlobalConfiguration globalConfig) {
49        this.cache = cache.getAdvancedCache();
50        this.globalConfig = globalConfig;
51     }
52  
53     /**
54      * Here is where the registration is being performed.
55      */
56     @Start(priority = 14)
57     public void start() {
58        if (cache == null)
59           throw new IllegalStateException("The cache should had been injected before a call to this method");
60        Set<Component> components = cache.getComponentRegistry().getRegisteredComponents();
61        nonCacheComponents = getNonCacheComponents(components);
62        if (registerMBeans(components, cache.getCacheManager().getCacheManagerConfiguration())) {
63           needToUnregister = true;
64           log.mbeansSuccessfullyRegistered();
65        } else {
66           if (cache.getName().equals(BasicCacheContainer.DEFAULT_CACHE_NAME)) {
67              log.unableToRegisterMBeans();
68           } else {
69              log.unableToRegisterMBeans(cache.getName());
70           }
71        }
72     }
73  
74     /**
75      * Unregister when the cache is being stopped.
76      */
77     @Stop
78     public void stop() {
79        // This method might get called several times.
80        // After the first call the cache will become null, so we guard this
81        if (cache == null) return;
82        if (needToUnregister) {
83           // Only unregister the non cache MBean so that it can be restarted
84           try {
85              unregisterMBeans(nonCacheComponents);
86              needToUnregister = false;
87           } catch (Exception e) {
88              log.problemsUnregisteringMBeans(e);
89           }
90        }
91  
92        // If removing cache, also remove cache MBean
93        if (unregisterCacheMBean)
94           unregisterCacheMBean();
95  
96        // make sure we don't set cache to null, in case it needs to be restarted via JMX.
97     }
98  
99     public void unregisterCacheMBean() {
100       if (mBeanServer != null) {
101          String groupName = CACHE_JMX_GROUP + "," + getCacheJmxName() + ",manager=" + ObjectName.quote(globalConfig.globalJmxStatistics().cacheManagerName());
102          String pattern = jmxDomain + ":" + groupName + ",*";
103          try {
104             Set<ObjectName> names = mBeanServer.queryNames(new ObjectName(pattern), null);
105             for (ObjectName name : names) {
106                mBeanServer.unregisterMBean(name);
107             }
108          } catch (MBeanRegistrationException e) {
109             log.unableToUnregisterMBeanWithPattern(pattern, e);
110          } catch (InstanceNotFoundException e) {
111             // Ignore if Cache MBeans not present
112          } catch (MalformedObjectNameException e) {
113             String message = "Malformed pattern " + pattern;
114             throw new CacheException(message, e);
115          }
116       }
117    }
118 
119    public void setUnregisterCacheMBean(boolean unregisterCacheMBean) {
120       this.unregisterCacheMBean = unregisterCacheMBean;
121    }

123    @Override
124    protected ComponentsJmxRegistration buildRegistrar(Set<AbstractComponentRegistry.Component> components) {
125       // Quote group name, to handle invalid ObjectName characters
126       String groupName = CACHE_JMX_GROUP + "," + getCacheJmxName() + ",manager=" + ObjectName.quote(globalConfig.globalJmxStatistics().cacheManagerName());
127       ComponentsJmxRegistration registrar = new ComponentsJmxRegistration(mBeanServer, components, groupName);
128       updateDomain(registrar, cache.getComponentRegistry().getGlobalComponentRegistry(), mBeanServer, groupName);
129       return registrar;
130    }
131 
132    private String getCacheJmxName() {
133       return ComponentsJmxRegistration.NAME_KEY + "="
134                   + ObjectName.quote(cache.getName() + "(" + cache.getCacheConfiguration().clustering().cacheModeString().toLowerCase() + ")");
135    }
136 
137    protected void updateDomain(ComponentsJmxRegistration registrar, GlobalComponentRegistry componentRegistry,
138                                MBeanServer mBeanServer, String groupName) {
139       CacheManagerJmxRegistration managerJmxReg = componentRegistry.getComponent(CacheManagerJmxRegistration.class);
140       if (!globalConfig.globalJmxStatistics().enabled() && jmxDomain == null) {
141          String tmpJmxDomain = JmxUtil.buildJmxDomain(globalConfig, mBeanServer, groupName);
142          synchronized (managerJmxReg) {
143             if (managerJmxReg.jmxDomain == null) {
144                if (!tmpJmxDomain.equals(globalConfig.globalJmxStatistics().domain()) && !globalConfig.globalJmxStatistics().allowDuplicateDomains()) {
145                   throw log.jmxMBeanAlreadyRegistered(tmpJmxDomain, globalConfig.globalJmxStatistics().domain());
146                }
147                // Set manager component's jmx domain so that other caches under same manager
148                // can see it, particularly important when jmx is only enabled at the cache level
149                managerJmxReg.jmxDomain = tmpJmxDomain;
150             }
151             // So that all caches share the same domain, regardless of whether dups are
152             // allowed or not, simply assign the manager's calculated jmxDomain
153             jmxDomain = managerJmxReg.jmxDomain;
154          }
155       } else {
156          // If global stats were enabled, manager's jmxDomain would have been populated
157          // when cache manager was started, so no need for synchronization here.
158          jmxDomain = managerJmxReg.jmxDomain == null ? globalConfig.globalJmxStatistics().domain() : managerJmxReg.jmxDomain;
159       }
160       registrar.setJmxDomain(jmxDomain);
161    }
162 
163    protected Set<Component> getNonCacheComponents(Set<Component> components) {
164       Set<Component> componentsExceptCache = new HashSet<AbstractComponentRegistry.Component>(64);
165       for (AbstractComponentRegistry.Component component : components) {
166          String name = component.getName();
167          if (!name.equals(Cache.class.getName()) && !name.equals(AdvancedCache.class.getName())) {
168             componentsExceptCache.add(component);
169          }
170       }
171       return componentsExceptCache;
172    }
