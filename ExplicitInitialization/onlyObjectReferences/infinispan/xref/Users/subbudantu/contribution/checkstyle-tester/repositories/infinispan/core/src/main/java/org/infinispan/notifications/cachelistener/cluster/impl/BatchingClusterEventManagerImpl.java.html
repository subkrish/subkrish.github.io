20  public class BatchingClusterEventManagerImpl<K, V> implements ClusterEventManager<K, V>{
21     private final Cache<K, V> cache;
22     
23     private DistributedExecutorService distExecService;
24     
25     private final ThreadLocal<EventContext<K, V>> localContext = new ThreadLocal<>();
26     
27     public BatchingClusterEventManagerImpl(Cache<K, V> cache) {
28        this.cache = cache;
29     }
30     
31     @Start
32     public void start() {
33        distExecService = SecurityActions.getDefaultExecutorService(cache);
34     }
35     
36     @Override
37     public void addEvents(Address target, UUID identifier, Collection<ClusterEvent<K, V>> events, boolean sync) {
38        EventContext<K, V> ctx = localContext.get();
39        if (ctx == null) {
40           ctx = new UnicastEventContext<K, V>();
41           localContext.set(ctx);
42        }
43        ctx.addTargets(target, identifier, events, sync);
44     }
45  
46     @Override
47     public void sendEvents() {
48        EventContext<K, V> ctx = localContext.get();
49        if (ctx != null) {
50           ctx.sendToTargets(distExecService);
51           localContext.remove();
52        }
53     }
54     
55     @Override
56     public void dropEvents() {
57        localContext.remove();
58     }
59     
60     private static interface EventContext<K, V> {
61        public void addTargets(Address address, UUID identifier, Collection<ClusterEvent<K, V>> events, boolean sync);
62        
63        public void sendToTargets(DistributedExecutorService service);
64     }
65     
66     protected static class UnicastEventContext<K, V> implements EventContext<K, V> {
67        protected final Map<Address, TargetEvents<K, V>> targets = new HashMap<>();
68  
69        @Override
70        public void addTargets(Address address, UUID identifier, Collection<ClusterEvent<K, V>> events, boolean sync) {
71           TargetEvents<K, V> targetEvents = targets.get(address);
72           if (targetEvents == null) {
73              targetEvents = new TargetEvents<>();
74              targets.put(address, targetEvents);
75           }
76           
77           Map<UUID, Collection<ClusterEvent<K, V>>> listenerEvents = targetEvents.events;
78           // This shouldn't be set before, so do put instead of doing get then put
79           Collection<ClusterEvent<K, V>> prevEvents = listenerEvents.put(identifier, events);
80           if (prevEvents != null) {
81              // If we have multiple events to the same node for the same uuid condense them.  This shouldn't really happen...
82              events.addAll(prevEvents);
83           }
84           if (sync) {
85              targetEvents.sync = true;
86           }
87        }
88        
89        @Override
90        public void sendToTargets(DistributedExecutorService service) {
91           DistributedExecutionCompletionService<Void> completion = new DistributedExecutionCompletionService<Void>(service);
92           int syncCount = 0;
93           for (Entry<Address, TargetEvents<K, V>> entry : targets.entrySet()) {
94              TargetEvents<K, V> value = entry.getValue();
95              if (value.events.size() > 1) {
96                 if (value.sync) {
97                    completion.submit(entry.getKey(), new MultiClusterEventCallable<>(value.events));
98                    syncCount++;
99                 } else {
100                   service.submit(entry.getKey(), new MultiClusterEventCallable<>(value.events));
101                }
102             } else if (value.events.size() == 1) {
103                Entry<UUID, Collection<ClusterEvent<K, V>>> entryValue = value.events.entrySet().iterator().next();
104                if (value.sync) {
105                   completion.submit(entry.getKey(), new ClusterEventCallable<K, V>(entryValue.getKey(), entryValue.getValue()));
106                   syncCount++;
107                } else {
108                   service.submit(entry.getKey(), new ClusterEventCallable<K, V>(entryValue.getKey(), entryValue.getValue()));
109                }
110             }
111          }
112          
113          try {
114             for (int i = 0; i < syncCount; ++i) {
115                completion.take();
116             }
117          }
118          catch (InterruptedException e) {
119             throw new CacheException("Interrupted while waiting for event notifications to complete.", e);
120          }
121       }
122    }
123    
124    private static class TargetEvents<K, V> {
125       final Map<UUID, Collection<ClusterEvent<K, V>>> events = new HashMap<>();
126       boolean sync = false;
127    }
128 }
