32     private boolean statisticsEnabled = false;

35     public CacheContainerStatsImpl(EmbeddedCacheManager cm) {
36        this.cm = cm;
37        cm.getGlobalComponentRegistry().registerComponent(this, CacheContainerStats.class);
38        boolean globalJmxStatsEnabled = cm.getCacheManagerConfiguration().globalJmxStatistics().enabled();
39        setStatisticsEnabled(globalJmxStatsEnabled);
40     }

44        this.timeService = timeService;
45     }

47     @Override
48     public void setStatisticsEnabled(boolean enabled) {
49        this.statisticsEnabled = enabled;
50        if (enabled) {
51           //yes technically we do not reset stats but we initialize them
52           resetNanoseconds.set(timeService.time());
53        }
54     }

57     @Override
58     public boolean getStatisticsEnabled() {
59        return statisticsEnabled;
60     }
61  
62     @Override
63     public void resetStatistics() {
64        if (getStatisticsEnabled()) {
65           for (String cn : cm.getCacheNames()) {
66              if (cm.cacheExists(cn)) {
67                 cm.getCache(cn).getAdvancedCache().getStats().reset();
68              }
69           }
70           resetNanoseconds.set(timeService.time());
71        }
72     }

82     @ManagedAttribute(description = "Cache container total average number of milliseconds for all read operation in this cache container",
83           displayName = "Cache container total average read time",
84           units = Units.MILLISECONDS,
85           displayType = DisplayType.SUMMARY)
86     @Override
87     public long getAverageReadTime() {
88        long result = -1;
89        if (getStatisticsEnabled()) {
90           result = calculateAverageReadTime();
91        }
92        return result;
93     }
94  
95     protected long calculateAverageReadTime() {
96        long totalAverageReadTime = 0;
97        int includedCacheCounter = 0;
98        for (String cn : cm.getCacheNames()) {
99           if (cm.cacheExists(cn)) {
100             long averageReadTime = cm.getCache(cn).getAdvancedCache().getStats().getAverageReadTime();
101             if (averageReadTime > 0) {
102                includedCacheCounter++;
103                totalAverageReadTime += averageReadTime;
104             }
105          }
106       }
107       if (includedCacheCounter > 0) {
108          totalAverageReadTime = totalAverageReadTime / includedCacheCounter;
109       }
110       return totalAverageReadTime;
111    }

230    @ManagedAttribute(
231          description = "Cache container total percentage hit/(hit+miss) ratio for this cache",
232          displayName = "Cache container total hit ratio",
233          units = Units.PERCENTAGE,
234          displayType = DisplayType.SUMMARY
235    )
236    @Override

238       double result = -1d;
239       if (getStatisticsEnabled()) {
240          result = calculateHitRatio();
241       }
242       return result;
243    }
244 
245    protected double calculateHitRatio() {
246       long totalHits = 0;
247       double totalRequests = 0;
248       double rwRatio = 0;
249       for (String cn : cm.getCacheNames()) {
250          if (cm.cacheExists(cn)) {
251             Stats stats = cm.getCache(cn).getAdvancedCache().getStats();
252             long requests = stats.getRetrievals();
253             if (requests > 0) {
254                totalHits += stats.getHits();
255                totalRequests += requests;
256             }
257          }
258       }
259       if (totalRequests > 0) {
260          rwRatio = totalHits / totalRequests;
261       }
262       return rwRatio;
263    }

319    @ManagedAttribute(
320          description = "Cache container read/writes ratio in all caches from this cache container",
321          displayName = "Cache container read/write ratio",
322          units = Units.PERCENTAGE,
323          displayType = DisplayType.SUMMARY
324    )
325    @Override
326    public double getReadWriteRatio() {
327       double result = -1d;
328       if (getStatisticsEnabled()) {
329          result = calculateReadWriteRatio();
330       }
331       return result;
332    }
333 
334    protected double calculateReadWriteRatio() {
335       long sumOfAllReads = 0;
336       long sumOfAllWrites = 0;
337       double rwRatio = 0;
338       for (String cn : cm.getCacheNames()) {
339          if (cm.cacheExists(cn)) {
340             Stats stats = cm.getCache(cn).getAdvancedCache().getStats();
341             long stores = stats.getStores();
342             if (stores > 0) {
343                sumOfAllReads += stats.getRetrievals();
344                sumOfAllWrites += stores;
345             }
346          }
347       }
348       if (sumOfAllWrites > 0) {
349          rwRatio = (double) sumOfAllReads / sumOfAllWrites;
350       }
351       return rwRatio;
352    }

462    protected long calculateTimeSinceStart() {
463       long longestRunning = 0;
464       for (String cn : cm.getCacheNames()) {
465          if (cm.cacheExists(cn)) {
466             long runningTime = cm.getCache(cn).getAdvancedCache().getStats().getTimeSinceStart();
467             if (runningTime > longestRunning) {
468                longestRunning = runningTime;
469             }
470          }
471       }
