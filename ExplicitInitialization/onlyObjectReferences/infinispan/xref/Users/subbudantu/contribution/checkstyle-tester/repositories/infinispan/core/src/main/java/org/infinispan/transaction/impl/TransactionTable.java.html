129    @Start(priority = 9) // Start before cache loader manager
130    @SuppressWarnings("unused")
131    public void start() {
132       final int concurrencyLevel = configuration.locking().concurrencyLevel();
133       //use the IdentityEquivalence because some Transaction implementation does not have a stable hash code function
134       //and it can cause some leaks in the concurrent map.
135       localTransactions = CollectionFactory.makeConcurrentMap(concurrencyLevel, 0.75f, concurrencyLevel,
136             new IdentityEquivalence<Transaction>(),
137             AnyEquivalence.getInstance());
138       globalToLocalTransactions = CollectionFactory.makeConcurrentMap(concurrencyLevel, 0.75f, concurrencyLevel);
139 
140       boolean transactional = configuration.transaction().transactionMode().isTransactional();
141       if (clustered && transactional) {
142          minTopologyRecalculationLock = new ReentrantLock();
143          remoteTransactions = CollectionFactory.makeConcurrentMap(concurrencyLevel, 0.75f, concurrencyLevel);
144 
145          ThreadFactory tf = new ThreadFactory() {
146             @Override
147             public Thread newThread(Runnable r) {
148                String address = rpcManager != null ? rpcManager.getTransport().getAddress().toString() : "local";
149                Thread th = new Thread(r, "TxCleanupService," + cacheName + "," + address);
150                th.setDaemon(true);
151                return th;
152             }
153          };
154          executorService = Executors.newSingleThreadScheduledExecutor(tf);
155 
156          notifier.addListener(this);
157          cacheManagerNotifier.addListener(this);
158 
159          boolean totalOrder = configuration.transaction().transactionProtocol().isTotalOrder();
160          if (!totalOrder) {
161             completedTransactionsInfo = new CompletedTransactionsInfo();
162 
163             // Periodically run a task to cleanup the transaction table of completed transactions.
164             long interval = configuration.transaction().reaperWakeUpInterval();
165             executorService.scheduleAtFixedRate(new Runnable() {
166                @Override
167                public void run() {
168                   completedTransactionsInfo.cleanupCompletedTransactions();
169                }
170             }, interval, interval, TimeUnit.MILLISECONDS);
171 
172             executorService.scheduleAtFixedRate(new Runnable() {
173                @Override
174                public void run() {
175                   cleanupTimedOutTransactions();
176                }
177             }, interval, interval, TimeUnit.MILLISECONDS);
178          }
179       }
180    }

231       if (!localTransaction.isEnlisted()) {
232          SynchronizationAdapter sync = new SynchronizationAdapter(
233                 localTransaction, txCoordinator, commandsFactory, rpcManager,
234                 this, clusteringLogic, configuration);
235          if (transactionSynchronizationRegistry != null) {
236             try {
237                transactionSynchronizationRegistry.registerInterposedSynchronization(sync);
238             } catch (Exception e) {
239                log.failedSynchronizationRegistration(e);
240                throw new CacheException(e);
241             }
242 
243          } else {
244 
245             try {
246                transaction.registerSynchronization(sync);
247             } catch (Exception e) {
248                log.failedSynchronizationRegistration(e);
249                throw new CacheException(e);
250             }
251          }
252          ((SyncLocalTransaction) localTransaction).setEnlisted(true);
253       }
254    }

276    public void cleanupLeaverTransactions(List<Address> members) {
277       // Can happen if the cache is non-transactional
278       if (remoteTransactions == null)
279          return;
280 
281       if (trace) log.tracef("Checking for transactions originated on leavers. Current cache members are %s, remote transactions: %d",
282             members, remoteTransactions.size());
283       HashSet<Address> membersSet = new HashSet<>(members);
284       List<GlobalTransaction> toKill = new ArrayList<GlobalTransaction>();
285       for (Map.Entry<GlobalTransaction, RemoteTransaction> e : remoteTransactions.entrySet()) {
286          GlobalTransaction gt = e.getKey();
287          RemoteTransaction remoteTx = e.getValue();
288          if (trace) log.tracef("Checking transaction %s", gt);
289          if (!membersSet.contains(gt.getAddress())) {
290             toKill.add(gt);
291          }
292       }
293 
294       if (toKill.isEmpty()) {
295          if (trace) log.tracef("No remote transactions pertain to originator(s) who have left the cluster.");
296       } else {
297          log.debugf("The originating node left the cluster for %d remote transactions", toKill.size());
298       }
299 
300       for (GlobalTransaction gtx : toKill) {
301          log.debugf("Rolling back transaction %s because originator %s left the cluster", gtx, gtx.getAddress());
302          killTransaction(gtx);
303       }
304 
305       if (trace) log.tracef("Completed cleaning transactions originating on leavers. Remote transactions remaining: %d",
306             remoteTransactions.size());
307    }

309    public void cleanupTimedOutTransactions() {
310       if (trace) log.tracef("About to cleanup remote transactions older than %d ms", configuration.transaction().completedTxTimeout());
311       long beginning = timeService.time();
312       long cutoffCreationTime = beginning - TimeUnit.MILLISECONDS.toNanos(configuration.transaction().completedTxTimeout());
313       List<GlobalTransaction> toKill = new ArrayList<GlobalTransaction>();
314 
315       // Check remote transactions.
316       for(Map.Entry<GlobalTransaction, RemoteTransaction> e : remoteTransactions.entrySet()) {
317          GlobalTransaction gtx = e.getKey();
318          RemoteTransaction remoteTx = e.getValue();
319          if(remoteTx != null) {
320             if (trace) log.tracef("Checking transaction %s", gtx);
321             // Check the time.
322             if (remoteTx.getCreationTime() - cutoffCreationTime < 0) {
323                long duration = timeService.timeDuration(remoteTx.getCreationTime(), beginning, TimeUnit.MILLISECONDS);
324                log.remoteTransactionTimeout(gtx, duration);
325                toKill.add(gtx);
326             }
327          }
328       }
329 
330       // Rollback the orphaned transactions and release any held locks.
331       for (GlobalTransaction gtx : toKill) {
332          killTransaction(gtx);
333       }
334    }

336    private void killTransaction(GlobalTransaction gtx) {
337       RollbackCommand rc = new RollbackCommand(cacheName, gtx);
338       rc.init(invoker, icf, TransactionTable.this);
339       try {
340          rc.perform(null);
341          if (trace) log.tracef("Rollback of transaction %s complete.", gtx);
342       } catch (Throwable e) {
343          log.unableToRollbackGlobalTx(gtx, e);
344       }
345    }

369    private RemoteTransaction getOrCreateRemoteTransaction(GlobalTransaction globalTx, WriteCommand[] modifications, int topologyId) {
370       RemoteTransaction remoteTransaction = remoteTransactions.get(globalTx);
371       if (remoteTransaction != null)
372          return remoteTransaction;
373       remoteTransaction = modifications == null ? txFactory.newRemoteTransaction(globalTx, topologyId)
374             : txFactory.newRemoteTransaction(modifications, globalTx, topologyId);
375       RemoteTransaction existing = remoteTransactions.putIfAbsent(globalTx, remoteTransaction);
376       if (existing != null) {
377          if (trace) log.tracef("Remote transaction already registered: %s", existing);
378          return existing;
379       } else {
380          if (trace) log.tracef("Created and registered remote transaction %s", remoteTransaction);
381          if (remoteTransaction.getTopologyId() < minTxTopologyId) {
382             if (trace) log.tracef("Changing minimum topology ID from %d to %d", minTxTopologyId, remoteTransaction.getTopologyId());
383             minTxTopologyId = remoteTransaction.getTopologyId();
384          }
385          return remoteTransaction;
386       }
387    }

390     * Returns the {@link org.infinispan.transaction.xa.TransactionXaAdapter} corresponding to the supplied transaction.
391     * If none exists, will be created first.
392     */
393    public LocalTransaction getOrCreateLocalTransaction(Transaction transaction, boolean implicitTransaction) {
394       LocalTransaction current = localTransactions.get(transaction);
395       if (current == null) {
396          Address localAddress = rpcManager != null ? rpcManager.getTransport().getAddress() : null;
397          GlobalTransaction tx = txFactory.newGlobalTransaction(localAddress, false);
398          current = txFactory.newLocalTransaction(transaction, tx, implicitTransaction, currentTopologyId);
399          if (trace) log.tracef("Created a new local transaction: %s", current);
400          localTransactions.put(transaction, current);
401          globalToLocalTransactions.put(current.getGlobalTransaction(), current);
402          notifier.notifyTransactionRegistered(tx, true);
403       }
404       return current;
405    }

415    protected final LocalTransaction removeLocalTransactionInternal(Transaction tx) {
416       LocalTransaction localTx = localTransactions.get(tx);
417       if (localTx != null) {
418          globalToLocalTransactions.remove(localTx.getGlobalTransaction());
419          localTransactions.remove(tx);
420          releaseResources(localTx);
421       }
422       return localTx;
423    }

428             recalculateMinTopologyIdIfNeeded(cacheTransaction);
429          }
430          if (trace) log.tracef("Removed %s from transaction table.", cacheTransaction);
431          cacheTransaction.notifyOnTransactionFinished();
432       }

489    protected final void recalculateMinTopologyIdIfNeeded(CacheTransaction removedTransaction) {
490       if (removedTransaction == null) throw new IllegalArgumentException("Transaction cannot be null!");
491       if (currentTopologyId != CACHE_STOPPED_TOPOLOGY_ID) {
492 
493          // Assume that we only get here if we are clustered.
494          int removedTransactionTopologyId = removedTransaction.getTopologyId();
495          if (removedTransactionTopologyId < minTxTopologyId) {
496             if (trace) log.tracef("A transaction has a topology ID (%s) that is smaller than the smallest transaction topology ID (%s) this node knows about!  This can happen if a concurrent thread recalculates the minimum topology ID after the current transaction has been removed from the transaction table.", removedTransactionTopologyId, minTxTopologyId);
497          } else if (removedTransactionTopologyId == minTxTopologyId && removedTransactionTopologyId < currentTopologyId) {
498             // We should only need to re-calculate the minimum topology ID if the transaction being completed
499             // has the same ID as the smallest known transaction ID, to check what the new smallest is, and this is
500             // not the current topology ID.
501             calculateMinTopologyId(removedTransactionTopologyId);
502          }
503       }
504    }

508    public void onTopologyChange(TopologyChangedEvent<?, ?> tce) {
509       // don't do anything if this cache is not clustered
510       if (clustered) {
511          if (!tce.isPre()) {
512             // The topology id must be updated after the topology was updated in StateConsumerImpl,
513             // as state transfer requires transactions not sent to the new owners to have a smaller topology id.
514             currentTopologyId = tce.getNewTopologyId();
515             log.debugf("Topology changed, recalculating minTopologyId");
516             calculateMinTopologyId(-1);
517          }
518       }
519    }

531    /**
532     * This method calculates the minimum topology ID known by the current node.  This method is only used in a clustered
533     * cache, and only invoked when either a topology change is detected, or a transaction whose topology ID is not the same as
534     * the current topology ID.
535     * <p/>
536     * This method is guarded by minTopologyRecalculationLock to prevent concurrent updates to the minimum topology ID field.
537     *
538     * @param idOfRemovedTransaction the topology ID associated with the transaction that triggered this recalculation, or -1
539     *                               if triggered by a topology change event.
540     */
541    @GuardedBy("minTopologyRecalculationLock")
542    private void calculateMinTopologyId(int idOfRemovedTransaction) {
543       minTopologyRecalculationLock.lock();
544       try {
545          // We should only need to re-calculate the minimum topology ID if the transaction being completed
546          // has the same ID as the smallest known transaction ID, to check what the new smallest is.  We do this check
547          // again here, since this is now within a synchronized method.
548          if (idOfRemovedTransaction == -1 ||
549                (idOfRemovedTransaction == minTxTopologyId && idOfRemovedTransaction < currentTopologyId)) {
550             int minTopologyIdFound = currentTopologyId;
551 
552             for (CacheTransaction ct : localTransactions.values()) {
553                int topologyId = ct.getTopologyId();
554                if (topologyId < minTopologyIdFound) minTopologyIdFound = topologyId;
555             }
556             for (CacheTransaction ct : remoteTransactions.values()) {
557                int topologyId = ct.getTopologyId();
558                if (topologyId < minTopologyIdFound) minTopologyIdFound = topologyId;
559             }
560             if (minTopologyIdFound != minTxTopologyId) {
561                if (trace) log.tracef("Changing minimum topology ID from %s to %s", minTxTopologyId, minTopologyIdFound);
562                minTxTopologyId = minTopologyIdFound;
563             } else {
564                if (trace) log.tracef("Minimum topology ID still is %s; nothing to change", minTopologyIdFound);
565             }
566          }
567       } finally {
568          minTopologyRecalculationLock.unlock();
569       }
570    }

576    private void shutDownGracefully() {
577       if (log.isDebugEnabled())
578          log.debugf("Wait for on-going transactions to finish for %s.", Util.prettyPrintTime(configuration.transaction().cacheStopTimeout(), TimeUnit.MILLISECONDS));
579       long failTime = timeService.expectedEndTime(configuration.transaction().cacheStopTimeout(), TimeUnit.MILLISECONDS);
580       boolean txsOnGoing = areTxsOnGoing();
581       while (txsOnGoing && !timeService.isTimeExpired(failTime)) {
582          try {
583             Thread.sleep(30);
584             txsOnGoing = areTxsOnGoing();
585          } catch (InterruptedException e) {
586             Thread.currentThread().interrupt();
587             if (clustered) {
588                log.debugf("Interrupted waiting for on-going transactions to finish. %s local transactions and %s remote transactions", localTransactions.size(), remoteTransactions.size());
589             } else {
590                log.debugf("Interrupted waiting for %s on-going transactions to finish.", localTransactions.size());
591             }
592          }
593       }
594 
595       if (txsOnGoing) {
596          log.unfinishedTransactionsRemain(localTransactions == null ? 0 : localTransactions.size(),
597                                           remoteTransactions == null ? 0 : remoteTransactions.size());
598       } else {
599          log.debug("All transactions terminated");
600       }
601    }

619          return false;
620 
621       return completedTransactionsInfo.isTransactionCompleted(gtx);
622    }

624    /**
625     * @see #markTransactionCompleted(org.infinispan.transaction.xa.GlobalTransaction, boolean)
626     */
627    public CompletedTransactionStatus getCompletedTransactionStatus(GlobalTransaction gtx) {
628       if (completedTransactionsInfo == null)
629          return CompletedTransactionStatus.NOT_COMPLETED;
630 
631       return completedTransactionsInfo.getTransactionStatus(gtx);
632    }

634    private class CompletedTransactionsInfo {
635       final EquivalentConcurrentHashMapV8<GlobalTransaction, CompletedTransactionInfo> completedTransactions;
636       // The highest transaction id previously cleared, one per originator
637       final EquivalentConcurrentHashMapV8<Address, Long> nodeMaxPrunedTxIds;
638       // The highest transaction id previously cleared, with any originator
639       volatile long globalMaxPrunedTxId;
640 
641       public CompletedTransactionsInfo() {
642          nodeMaxPrunedTxIds = new EquivalentConcurrentHashMapV8<>(AnyEquivalence.getInstance(), AnyEquivalence.getInstance());
643          completedTransactions = new EquivalentConcurrentHashMapV8<>(AnyEquivalence.getInstance(), AnyEquivalence.getInstance());
644          globalMaxPrunedTxId = -1;
645       }
646 
647 
648       /**
649        * With the current state transfer implementation it is possible for a transaction to be prepared several times
650        * on a remote node. This might cause leaks, e.g. if the transaction is prepared, committed and prepared again.
651        * Once marked as completed (because of commit or rollback) any further prepare received on that transaction are discarded.
652        */
653       public void markTransactionCompleted(GlobalTransaction globalTx, boolean successful) {
654          if (trace) log.tracef("Marking transaction %s as completed", globalTx);
655          completedTransactions.put(globalTx, new CompletedTransactionInfo(timeService.time(), successful));
656       }
657 
658       /**
659        * @see #markTransactionCompleted(GlobalTransaction, boolean)
660        */
661       public boolean isTransactionCompleted(GlobalTransaction gtx) {
662          if (completedTransactions.containsKey(gtx))
663             return true;
664 
665          // Transaction ids are allocated in sequence, so any transaction with a smaller id must have been started
666          // before a transaction that was already removed from the completed transactions map because it was too old.
667          // We assume that the transaction was either committed, or it was rolled back (e.g. because the prepare
668          // RPC timed out.
669          // Note: We must check the id *after* verifying that the tx doesn't exist in the map.
670          if (gtx.getId() > globalMaxPrunedTxId)
671             return false;
672          Long nodeMaxPrunedTxId = nodeMaxPrunedTxIds.get(gtx.getAddress());
673          return nodeMaxPrunedTxId != null && gtx.getId() <= nodeMaxPrunedTxId;
674       }
675 
676       public CompletedTransactionStatus getTransactionStatus(GlobalTransaction gtx) {
677          CompletedTransactionInfo completedTx = completedTransactions.get(gtx);
678          if (completedTx != null) {
679             return completedTx.successful ? CompletedTransactionStatus.COMMITTED : CompletedTransactionStatus.ABORTED;
680          }
681 
682          // Transaction ids are allocated in sequence, so any transaction with a smaller id must have been started
683          // before a transaction that was already removed from the completed transactions map because it was too old.
684          // We assume that the transaction was either committed, or it was rolled back (e.g. because the prepare
685          // RPC timed out.
686          // Note: We must check the id *after* verifying that the tx doesn't exist in the map.
687          if (gtx.getId() > globalMaxPrunedTxId)
688             return CompletedTransactionStatus.NOT_COMPLETED;
689          Long nodeMaxPrunedTxId = nodeMaxPrunedTxIds.get(gtx.getAddress());
690          if (nodeMaxPrunedTxId == null) {
691             // We haven't removed any transaction for this node
692             return CompletedTransactionStatus.NOT_COMPLETED;
693          } else if (gtx.getId() > nodeMaxPrunedTxId) {
694             // We haven't removed this particular transaction yet
695             return CompletedTransactionStatus.NOT_COMPLETED;
696          } else {
697             // We already removed the status of this transaction from the completed transactions map
698             return CompletedTransactionStatus.EXPIRED;
699          }
700       }
701 
702       public void cleanupCompletedTransactions() {
703          if (completedTransactions.isEmpty())
704             return;
705 
706          try {
707             if (trace) log.tracef("About to cleanup completed transaction. Initial size is %d", completedTransactions.size());
708             long beginning = timeService.time();
709             long minCompleteTimestamp = timeService.time() - TimeUnit.MILLISECONDS.toNanos(configuration.transaction().completedTxTimeout());
710             int removedEntries = 0;
711 
712             // Collect the leavers. They will be removed at the end.
713             Set<Address> leavers = new HashSet<>();
714             for (Map.Entry<Address, Long> e : nodeMaxPrunedTxIds.entrySet()) {
715                if (!rpcManager.getMembers().contains(e.getKey())) {
716                   leavers.add(e.getKey());
717                }
718             }
719 
720             // Remove stale completed transactions.
721             Iterator<Map.Entry<GlobalTransaction, CompletedTransactionInfo>> txIterator = completedTransactions.entrySet().iterator();
722             while (txIterator.hasNext()) {
723                Map.Entry<GlobalTransaction, CompletedTransactionInfo> e = txIterator.next();
724                CompletedTransactionInfo completedTx = e.getValue();
725                if (minCompleteTimestamp - completedTx.timestamp > 0) {
726                   // Need to update lastPrunedTxId *before* removing the tx from the map
727                   // Don't need atomic operations, there can't be more than one thread updating lastPrunedTxId.
728                   final long txId = e.getKey().getId();
729                   final Address address = e.getKey().getAddress();
730                   updateLastPrunedTxId(txId, address);
731 
732                   txIterator.remove();
733                   removedEntries++;
734                } else {
735                   // Nodes with "active" completed transactions are not removed..
736                   leavers.remove(e.getKey().getAddress());
737                }
738             }
739 
740             // Finally, remove nodes that are no longer members and don't have any "active" completed transactions.
741             for (Address e : leavers) {
742                nodeMaxPrunedTxIds.remove(e);
743             }
744 
745             long duration = timeService.timeDuration(beginning, TimeUnit.MILLISECONDS);
746 
747             if (trace) log.tracef("Finished cleaning up completed transactions in %d millis, %d transactions were removed, " +
748                   "current number of completed transactions is %d",
749                   removedEntries, duration, completedTransactions.size());
750             if (trace) log.tracef("Last pruned transaction ids were updated: %d, %s", globalMaxPrunedTxId, nodeMaxPrunedTxIds);
751          } catch (Exception e) {
752             log.errorf(e, "Failed to cleanup completed transactions: %s", e.getMessage());
753          }
754       }
755 
756       private void updateLastPrunedTxId(final long txId, Address address) {
757          if (txId > globalMaxPrunedTxId) {
758             globalMaxPrunedTxId = txId;
759          }
760          nodeMaxPrunedTxIds.compute(address, new EquivalentConcurrentHashMapV8.BiFun<Address, Long, Long>() {
761             @Override
762             public Long apply(Address address, Long nodeMaxPrunedTxId) {
763                if (nodeMaxPrunedTxId != null && txId <= nodeMaxPrunedTxId) {
764                   return nodeMaxPrunedTxId;
765                }
766                return txId;
767             }
768          });
769       }
770    }
