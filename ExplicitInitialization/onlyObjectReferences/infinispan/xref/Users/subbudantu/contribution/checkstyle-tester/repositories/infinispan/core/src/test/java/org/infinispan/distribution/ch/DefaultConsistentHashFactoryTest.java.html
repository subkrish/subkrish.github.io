43     public void testConsistentHashDistribution() {
44        int[] numSegments = {1, 2, 4, 8, 16, 50, 100, 500};
45        int[] numNodes = {1, 2, 3, 4, 5, 7, 10, 100};
46        int[] numOwners = {1, 2, 3, 5};
47        // Since the number of nodes changes, the capacity factors are repeated
48        float[][] capacityFactors = {null, {1}, {2}, {1, 100}, {2, 0, 1}};
49  
50        ConsistentHashFactory <DefaultConsistentHash> chf = createConsistentHashFactory();
51        Hash hashFunction = MurmurHash3.getInstance();
52  
53        for (int nn : numNodes) {
54           List<Address> nodes = new ArrayList<Address>(nn);
55           for (int j = 0; j < nn; j++) {
56              nodes.add(new TestAddress(j, "TA"));
57           }
58  
59           for (int ns : numSegments) {
60              if (nn < ns) {
61                 for (int no : numOwners) {
62                    for (float[] lf : capacityFactors) {
63                       Map<Address, Float> lfMap = null;
64                       if (lf != null) {
65                          lfMap = new HashMap<Address, Float>();
66                          for (int i = 0; i < nn; i++) {
67                             lfMap.put(nodes.get(i), lf[i % lf.length]);
68                          }
69                       }
70                       testConsistentHashModifications(chf, hashFunction, nodes, ns, no, lfMap);
71                    }
72                 }
73              }
74           }
75        }

78     private void testConsistentHashModifications(ConsistentHashFactory<DefaultConsistentHash> chf, Hash hashFunction, List<Address> nodes, int ns, int no, Map<Address, Float> lfMap) {
79        DefaultConsistentHash baseCH = chf.create(hashFunction, no, ns, nodes, lfMap);
80        assertEquals(lfMap, baseCH.getCapacityFactors());
81        checkDistribution(baseCH, lfMap, false);
82  
83        // each element in the array is a pair of numbers: the first is the number of nodes to add
84        // the second is the number of nodes to remove (the index of the removed nodes are pseudo-random)
85        int[][] nodeChanges = {{1, 0}, {2, 0}, {0, 1}, {0, 2}, {1, 1}, {1, 2}, {2, 1}, {10, 0}, {0, 10}};
86  
87        // check that the base CH is already balanced
88        List<Address> baseMembers = baseCH.getMembers();
89        assertSame(baseCH, chf.updateMembers(baseCH, baseMembers, lfMap));
90        assertSame(baseCH, chf.rebalance(baseCH));
91  
92        // starting point, so that we don't confuse nodes
93        int nodeIndex = baseMembers.size();
94  
95        for (int i = 0; i < nodeChanges.length; i++) {
96           int nodesToAdd = nodeChanges[i][0];
97           int nodesToRemove = nodeChanges[i][1];
98           if (nodesToRemove > baseMembers.size())
99              break;
100          if (nodesToRemove == baseMembers.size() && nodesToAdd == 0)
101             break;
102 
103          List<Address> newMembers = new ArrayList<Address>(baseMembers);
104          HashMap<Address, Float> newCapacityFactors = lfMap != null ?
105                new HashMap<Address, Float>(lfMap) : null;
106          for (int k = 0; k < nodesToRemove; k++) {
107             int indexToRemove = Math.abs(baseCH.getHashFunction().hash(k) % newMembers.size());
108             if (newCapacityFactors != null) {
109                newCapacityFactors.remove(newMembers.get(indexToRemove));
110             }
111             newMembers.remove(indexToRemove);
112          }
113          for (int k = 0; k < nodesToAdd; k++) {
114             TestAddress address = new TestAddress(nodeIndex++, "TA");
115             newMembers.add(address);
116             if (newCapacityFactors != null) {
117                newCapacityFactors.put(address, lfMap.get(baseMembers.get(k % baseMembers.size())));
118             }
119          }
120 
121          log.tracef("Testing consistent hash modifications iteration %d. Initial CH is %s. New members are %s",
122                iterationCount, baseCH, newMembers);
123          baseCH = checkModificationsIteration(chf, baseCH, nodesToAdd, nodesToRemove, newMembers, newCapacityFactors);
124 
125          iterationCount++;
126       }
127    }

129    private DefaultConsistentHash checkModificationsIteration(ConsistentHashFactory<DefaultConsistentHash> chf,
130                                                              DefaultConsistentHash baseCH, int nodesToAdd,
131                                                              int nodesToRemove, List<Address> newMembers,
132                                                              Map<Address, Float> lfMap) {
133       int actualNumOwners = computeActualNumOwners(baseCH.getNumOwners(), newMembers, lfMap);
134 
135       // first phase: just update the members list, removing the leavers
136       // and adding new owners, but not necessarily assigning segments to them
137       DefaultConsistentHash updatedMembersCH = chf.updateMembers(baseCH, newMembers, lfMap);
138       assertEquals(lfMap, updatedMembersCH.getCapacityFactors());
139       if (nodesToRemove > 0) {
140          for (int l = 0; l < updatedMembersCH.getNumSegments(); l++) {
141             assertTrue(updatedMembersCH.locateOwnersForSegment(l).size() > 0);
142             assertTrue(updatedMembersCH.locateOwnersForSegment(l).size() <= actualNumOwners);
143          }
144       }
145 
146       // second phase: rebalance with the new members list
147       DefaultConsistentHash rebalancedCH = chf.rebalance(updatedMembersCH);
148       checkDistribution(rebalancedCH, lfMap, false);
149 
150       for (int l = 0; l < rebalancedCH.getNumSegments(); l++) {
151          assertTrue(rebalancedCH.locateOwnersForSegment(l).size() >= actualNumOwners);
152       }
153 
154       checkMovedSegments(baseCH, rebalancedCH);
155 
156       // union doesn't have to keep the CH balanced, but it does have to include owners from both CHs
157       DefaultConsistentHash unionCH = chf.union(updatedMembersCH, rebalancedCH);
158       for (int l = 0; l < updatedMembersCH.getNumSegments(); l++) {
159          assertTrue(unionCH.locateOwnersForSegment(l).containsAll(updatedMembersCH.locateOwnersForSegment(l)));
160          assertTrue(unionCH.locateOwnersForSegment(l).containsAll(rebalancedCH.locateOwnersForSegment(l)));
161       }
162 
163       // switch to the new CH in the next iteration
164       assertEquals(rebalancedCH.getNumSegments(), baseCH.getNumSegments());
165       assertEquals(rebalancedCH.getNumOwners(), baseCH.getNumOwners());
166       assertEquals(rebalancedCH.getMembers(), newMembers);
167       baseCH = rebalancedCH;
168       return baseCH;
169    }

171    private void checkDistribution(ConsistentHash ch, Map<Address, Float> lfMap, boolean allowExtraOwners) {
172       int numSegments = ch.getNumSegments();
173       List<Address> nodes = ch.getMembers();
174       int numNodes = nodes.size();
175       int actualNumOwners = computeActualNumOwners(ch.getNumOwners(), nodes, lfMap);
176 
177       OwnershipStatistics stats = new OwnershipStatistics(nodes);
178       for (int i = 0; i < numSegments; i++) {
179          List<Address> owners = ch.locateOwnersForSegment(i);
180          if (!allowExtraOwners) {
181             assertEquals(owners.size(), actualNumOwners);
182          } else {
183             assertTrue(owners.size() >= actualNumOwners);
184          }
185          stats.incPrimaryOwned(owners.get(0));
186          for (int j = 0; j < owners.size(); j++) {
187             Address owner = owners.get(j);
188             stats.incOwned(owner);
189             assertEquals(owners.indexOf(owner), j, "Found the same owner twice in the owners list");
190          }
191       }
192 
193       float totalCapacity = computeTotalCapacity(nodes, lfMap);
194       float maxCapacityFactor = computeMaxCapacityFactor(nodes, lfMap);
195       Map<Address, Float> expectedOwnedMap = computeExpectedOwned(numSegments, numNodes, actualNumOwners, nodes, lfMap);
196       for (Address node : nodes) {
197          float capacityFactor = lfMap != null ? lfMap.get(node) : 1;
198          float expectedPrimaryOwned = expectedPrimaryOwned(numSegments, numNodes, totalCapacity, capacityFactor);
199          float deviationPrimaryOwned = allowedDeviationPrimaryOwned(numSegments, numNodes, totalCapacity, maxCapacityFactor);
200          int minPrimaryOwned = (int) Math.floor(expectedPrimaryOwned - deviationPrimaryOwned);
201          int maxPrimaryOwned = (int) Math.ceil(expectedPrimaryOwned + deviationPrimaryOwned);
202          if (!allowExtraOwners) {
203             int primaryOwned = stats.getPrimaryOwned(node);
204             assertTrue(minPrimaryOwned <= primaryOwned);
205             assertTrue(primaryOwned <= maxPrimaryOwned);
206          }
207 
208          float expectedOwned = expectedOwnedMap.get(node);
209          float deviationOwned = allowedDeviationOwned(numSegments, actualNumOwners, numNodes, totalCapacity, maxCapacityFactor);
210          int minOwned = (int) Math.floor(expectedOwned - deviationOwned);
211          int maxOwned = (int) Math.ceil(expectedOwned + deviationOwned);
212          int owned = stats.getOwned(node);
213          assertTrue(Math.floor(minOwned) <= owned);
214          if (!allowExtraOwners) {
215             assertTrue(owned <= Math.ceil(maxOwned));
216          }
217       }
218    }

220    public int computeActualNumOwners(int numOwners, List<Address> members, Map<Address, Float> capacityFactors) {
221       if (capacityFactors == null)
222          return Math.min(numOwners, members.size());
223 
224       int nodesWithLoad = 0;
225       for (Address node : members) {
226          if (capacityFactors.get(node) != 0) {
227             nodesWithLoad++;
228          }
229       }
230       return Math.min(numOwners, nodesWithLoad);
231    }

233    protected float expectedPrimaryOwned(int numSegments, int numNodes, float totalCapacity, float nodeLoad) {
234       return Math.min(numSegments * nodeLoad / totalCapacity, numSegments);
235    }

237    protected float allowedDeviationPrimaryOwned(int numSegments, int numNodes, float totalCapacity, float maxCapacityFactor) {
238       return numNodes * maxCapacityFactor / totalCapacity;
239    }

241    protected Map<Address, Float> computeExpectedOwned(int numSegments, int numNodes, int actualNumOwners,
242                                                        Collection<Address> nodes,
243                                                        final Map<Address, Float> capacityFactors) {
244       Map<Address, Float> expectedOwned = new HashMap<Address, Float>();
245       if (capacityFactors == null) {
246          float expected = Math.min(numSegments, (float) numSegments * actualNumOwners / numNodes);
247          for (Address node : nodes) {
248             expectedOwned.put(node, expected);
249          }
250          return expectedOwned;
251       }
252 
253       List<Address> sortedNodes = new ArrayList<Address>(nodes);
254       Collections.sort(sortedNodes, new Comparator<Address>() {
255          @Override
256          public int compare(Address o1, Address o2) {
257             // Reverse order
258             return (int) Math.signum(capacityFactors.get(o2) - capacityFactors.get(o1));
259          }
260       });
261 
262       float totalCapacity = computeTotalCapacity(nodes, capacityFactors);
263 
264       int remainingCopies = actualNumOwners * numSegments;
265       for (Address node : sortedNodes) {
266          float nodeLoad = capacityFactors.get(node);
267          float nodeSegments;
268          if (remainingCopies * nodeLoad / totalCapacity > numSegments) {
269             nodeSegments = numSegments;
270             totalCapacity -= nodeLoad;
271             remainingCopies -= nodeSegments;
272          } else {
273             nodeSegments = nodeLoad != 0 ? remainingCopies * nodeLoad / totalCapacity : 0;
274          }
275          expectedOwned.put(node, nodeSegments);
276       }
277       return expectedOwned;
278    }

280    protected float allowedDeviationOwned(int numSegments, int actualNumOwners, int numNodes, float totalCapacity,
281                                           float maxCapacityFactor) {
282       return numNodes * maxCapacityFactor / totalCapacity;
283    }

285    private float computeTotalCapacity(Collection<Address> nodes, Map<Address, Float> capacityFactors) {
286       if (capacityFactors == null)
287          return nodes.size();
288 
289       float totalCapacity = 0;
290       for (Address node : nodes) {
291          totalCapacity += capacityFactors.get(node);
292       }
293       return totalCapacity;
294    }
295 
296    private float computeMaxCapacityFactor(Collection<Address> nodes, Map<Address, Float> capacityFactors) {
297       if (capacityFactors == null)
298          return 1;
299 
300       float maxCapacityFactor = 0;
301       for (Address node : nodes) {
302          Float capacityFactor = capacityFactors.get(node);
303          if (capacityFactor > maxCapacityFactor) {
304             maxCapacityFactor = capacityFactor;
305          }
306       }
307       return maxCapacityFactor;
308    }

310    protected int allowedExtraMoves(DefaultConsistentHash oldCH, DefaultConsistentHash newCH,
311                                    int leaverSegments) {
312       return (int) Math.ceil(0.25 * oldCH.getNumSegments());
313    }

315    private void checkMovedSegments(DefaultConsistentHash oldCH, DefaultConsistentHash newCH) {
316       int numSegments = oldCH.getNumSegments();
317       int numOwners = oldCH.getNumOwners();
318       Set<Address> oldMembers = new HashSet<Address>(oldCH.getMembers());
319       Set<Address> newMembers = new HashSet<Address>(newCH.getMembers());
320 
321       // Compute the number of segments owned by members that left
322       int leaverSegments = 0;
323       for (Address node : oldMembers) {
324          if (!newMembers.contains(node)) {
325             leaverSegments += oldCH.getSegmentsForOwner(node).size();
326          }
327       }
328 
329       // Compute the number of segments where an old node became an owner
330       int oldMembersAddedSegments = 0;
331       for (int segment = 0; segment < numSegments; segment++) {
332          ArrayList<Address> oldMembersAdded = new ArrayList<Address>(newCH.locateOwnersForSegment(segment));
333          oldMembersAdded.removeAll(oldCH.locateOwnersForSegment(segment));
334          oldMembersAdded.retainAll(oldMembers);
335          oldMembersAddedSegments += oldMembersAdded.size();
336       }
337 
338       int movedSegments = oldMembersAddedSegments - leaverSegments;
339       int expectedExtraMoves = allowedExtraMoves(oldCH, newCH, leaverSegments);
340       if (movedSegments > expectedExtraMoves) {
341          log.debugf("%d of %d segments moved, %d (%fx) more than expected (%d)", movedSegments, numSegments,
342                movedSegments - expectedExtraMoves, (float) movedSegments / expectedExtraMoves, expectedExtraMoves);
343       }
344       assert movedSegments <= expectedExtraMoves
345                : String.format("Two many moved segments between %s and %s: expected %d, got %d",
346             oldCH, newCH, expectedExtraMoves, oldMembersAddedSegments);
347    }

349    protected <T> Set<T> symmetricalDiff(Collection <T> set1, Collection<T> set2) {
350       HashSet<T> commonMembers = new HashSet<T>(set1);
351       commonMembers.retainAll(set2);
352       HashSet<T> symDiffMembers = new HashSet<T>(set1);
353       symDiffMembers.addAll(set2);
354       symDiffMembers.removeAll(commonMembers);
355       return symDiffMembers;
356    }
