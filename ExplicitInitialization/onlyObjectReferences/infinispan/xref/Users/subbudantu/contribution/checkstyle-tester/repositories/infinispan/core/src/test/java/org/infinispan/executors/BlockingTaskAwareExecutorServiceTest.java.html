23  @Test(groups = "functional", testName = "executors.BlockingTaskAwareExecutorServiceTest")
24  public class BlockingTaskAwareExecutorServiceTest extends AbstractInfinispanTest {
25  
26     public void testSimpleExecution() throws Exception {
27        BlockingTaskAwareExecutorService executorService = createExecutorService();
28        try {
29           final DoSomething doSomething = new DoSomething();
30           executorService.execute(doSomething);
31  
32           Thread.sleep(100);
33  
34           assert !doSomething.isReady();
35           assert !doSomething.isExecuted();
36  
37           doSomething.markReady();
38           executorService.checkForReadyTasks();
39  
40           assert doSomething.isReady();
41  
42           eventually(new Condition() {
43              @Override
44              public boolean isSatisfied() throws Exception {
45                 return doSomething.isExecuted();
46              }
47           });
48        } finally {
49           executorService.shutdownNow();
50        }
51     }
52  
53     public void testMultipleExecutions() throws Exception {
54        BlockingTaskAwareExecutorServiceImpl executorService = createExecutorService();
55        try {
56           List<DoSomething> tasks = new LinkedList<DoSomething>();
57  
58           for (int i = 0; i < 30; ++i) {
59              tasks.add(new DoSomething());
60           }
61  
62           for (DoSomething doSomething : tasks) {
63              executorService.execute(doSomething);
64           }
65  
66           for (DoSomething doSomething : tasks) {
67              assert !doSomething.isReady();
68              assert !doSomething.isExecuted();
69           }
70  
71           for (DoSomething doSomething : tasks) {
72              doSomething.markReady();
73           }
74           executorService.checkForReadyTasks();
75  
76           for (final DoSomething doSomething : tasks) {
77              eventually(new Condition() {
78                 @Override
79                 public boolean isSatisfied() throws Exception {
80                    return doSomething.isExecuted();
81                 }
82              });
83           }
84  
85        } finally {
86           executorService.shutdownNow();
87        }
88     }
89  
90     private BlockingTaskAwareExecutorServiceImpl createExecutorService() {
91        return new BlockingTaskAwareExecutorServiceImpl(new ThreadPoolExecutor(1, 2, 60, TimeUnit.SECONDS,
92                                                                               new LinkedBlockingQueue<Runnable>(1000),
93                                                                               new DummyThreadFactory()),
94                                                        TIME_SERVICE);
95     }
96  
97     public static class DummyThreadFactory implements ThreadFactory {
98  
99        @Override
100       public Thread newThread(Runnable runnable) {
101          return new Thread(runnable);
102       }
103    }
104 
105    public static class DoSomething implements BlockingRunnable {
106 
107       private volatile boolean ready = false;
108       private volatile boolean executed = false;
109 
110       @Override
111       public synchronized final boolean isReady() {
112          return ready;
113       }
114 
115       @Override
116       public synchronized final void run() {
117          executed = true;
118       }
119 
120       public synchronized final void markReady() {
121          ready = true;
122       }
123 
124       public synchronized final boolean isExecuted() {
125          return executed;
126       }
127    }
