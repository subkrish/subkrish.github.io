211       Cache cache2 = cache(1, "replSync");

213       Pojo key1 = new Pojo(1), value1 = new Pojo(11), key2 = new Pojo(2), value2 = new Pojo(22);
214       String key3 = "3", value3 = "three";
215       cache1.put(key1, value1);
216       cache1.put(key2, value2);
217       cache1.put(key3, value3);
218 
219       Set expKeys = new HashSet();
220       expKeys.add(key1);
221       expKeys.add(key2);
222       expKeys.add(key3);
223 
224       Set expValues = new HashSet();
225       expValues.add(value1);
226       expValues.add(value2);
227       expValues.add(value3);
228 
229       Set expKeyEntries = ObjectDuplicator.duplicateSet(expKeys);
230       Set expValueEntries = ObjectDuplicator.duplicateSet(expValues);
231 
232       Set keys = cache2.keySet();
233       for (Object key : keys) assertTrue(expKeys.remove(key));
234       assertTrue("Did not see keys " + expKeys + " in iterator!", expKeys.isEmpty());
235 
236       Collection values = cache2.values();
237       for (Object key : values) assertTrue(expValues.remove(key));
238       assertTrue("Did not see keys " + expValues + " in iterator!", expValues.isEmpty());
239 
240       Set<Map.Entry> entries = cache2.entrySet();
241       for (Map.Entry entry : entries) {
242          assertTrue(expKeyEntries.remove(entry.getKey()));
243          assertTrue(expValueEntries.remove(entry.getValue()));
244       }
245       assertTrue("Did not see keys " + expKeyEntries + " in iterator!", expKeyEntries.isEmpty());
246       assertTrue("Did not see keys " + expValueEntries + " in iterator!", expValueEntries.isEmpty());
