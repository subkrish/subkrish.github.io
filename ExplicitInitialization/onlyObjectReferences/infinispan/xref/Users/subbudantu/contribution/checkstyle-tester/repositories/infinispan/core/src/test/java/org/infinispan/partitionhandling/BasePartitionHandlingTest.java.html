113       private void disableDiscovery() {
114          for (Channel c : channels) {
115             for (Protocol p : c.getProtocolStack().getProtocols()) {
116                if (p instanceof Discovery) {
117                   if (!(p instanceof TEST_PING)) throw new IllegalStateException("TEST_PING required for this test.");
118                   ((TEST_PING) p).suspend();
119                }
120             }
121          }
122       }

124       private void assertPartitionFormed() {
125          final List<Address> viewMembers = new ArrayList<>();
126          for (Channel ac : channels) viewMembers.add(ac.getAddress());
127          for (Channel c : channels) {
128             List<Address> members = c.getView().getMembers();
129             if (!members.equals(viewMembers)) throw new AssertionError();
130          }
131       }

133       private List<Address> installNewView() {
134          final List<Address> viewMembers = new ArrayList<>();
135          for (Channel c : channels) viewMembers.add(c.getAddress());
136          View view = View.create(channels.get(0).getAddress(), viewId.incrementAndGet(), (Address[]) viewMembers.toArray(new Address[viewMembers.size()]));
137 
138          log.trace("Before installing new view...");
139          for (Channel c : channels)
140             ((GMS) c.getProtocolStack().findProtocol(GMS.class)).installView(view);
141          return viewMembers;
142       }

144       private List<Address> installMergeView(ArrayList<Channel> view1, ArrayList<Channel> view2) {
145          List<Address> allAddresses = new ArrayList<>();
146          for (Channel c: view1) allAddresses.add(c.getAddress());
147          for (Channel c: view2) allAddresses.add(c.getAddress());
148 
149          View v1 = toView(view1);
150          View v2 = toView(view2);
151          List<View> allViews = new ArrayList<>();
152          allViews.add(v1);
153          allViews.add(v2);
154 
155 //         log.trace("Before installing new view: " + viewMembers);
156 //         System.out.println("Before installing new view: " + viewMembers);
157          MergeView mv = new MergeView(view1.get(0).getAddress(), (long)viewId.incrementAndGet(), allAddresses, allViews);
158          for (Channel c : channels)
159             ((GMS) c.getProtocolStack().findProtocol(GMS.class)).installView(mv);
160          return allMembers;
161       }

163       private View toView(ArrayList<Channel> channels) {
164          final List<Address> viewMembers = new ArrayList<>();
165          for (Channel c : channels) viewMembers.add(c.getAddress());
166          return View.create(channels.get(0).getAddress(), viewId.incrementAndGet(), (Address[]) viewMembers.toArray(new Address[viewMembers.size()]));
167       }

169       private void discardOtherMembers() {
170          List<Address> outsideMembers = new ArrayList<Address>();
171          for (Address a : allMembers) {
172             boolean inThisPartition = false;
173             for (Channel c : channels) {
174                if (c.getAddress().equals(a)) inThisPartition = true;
175             }
176             if (!inThisPartition) outsideMembers.add(a);
177          }
178          for (Channel c : channels) {
179             DISCARD discard = new DISCARD();
180             for (Address a : outsideMembers) discard.addIgnoreMember(a);
181             try {
182                c.getProtocolStack().insertProtocol(discard, ProtocolStack.ABOVE, TP.class);
183             } catch (Exception e) {
184                throw new RuntimeException(e);
185             }
186          }
187       }

196       public void merge(Partition partition) {
197          observeMembers(partition);
198          partition.observeMembers(this);
199          ArrayList<Channel> view1 = new ArrayList<>(channels);
200          ArrayList<Channel> view2 = new ArrayList<>(partition.channels);
201 //         System.out.println("view1 = " + printView(view1));
202 //         System.out.println("view2 = " + printView(view2));
203          channels.addAll(partition.channels);
204          installMergeView(view1, view2);
205          waitForPartitionToForm();
206          List<Partition> tmp = new ArrayList<>(Arrays.asList(BasePartitionHandlingTest.this.partitions));
207          if (!tmp.remove(partition)) throw new AssertionError();
208          BasePartitionHandlingTest.this.partitions = tmp.toArray(new Partition[tmp.size()]);
209       }

217       private void waitForPartitionToForm() {
218          List<Cache<Object, Object>> caches = new ArrayList<>(getCaches(null));
219          Iterator<Cache<Object, Object>> i = caches.iterator();
220          while (i.hasNext()) {
221             if (!channels.contains(channel(i.next())))
222                i.remove();
223          }
224          Cache<Object, Object> cache = caches.get(0);
225          TestingUtil.blockUntilViewsReceived(10000, caches);
226          if (cache.getCacheConfiguration().clustering().cacheMode().isClustered()) {
227             TestingUtil.waitForRehashToComplete(caches);
228          }
229       }

231       public void enableDiscovery() {
232          for (Channel c : channels) {
233             for (Protocol p : c.getProtocolStack().getProtocols()) {
234                if (p instanceof Discovery) {
235                   try {
236                      log.tracef("About to start discovery: %s", p);
237                      p.start();

239                      throw new RuntimeException(e);
240                   }
241                }
242             }
243          }
244          log.trace("Discovery started.");
245       }

247       private void observeMembers(Partition partition) {
248          for (Channel c : channels) {
249             List<Protocol> protocols = c.getProtocolStack().getProtocols();
250             for (Protocol p : protocols) {
251                if (p instanceof DISCARD) {
252                   for (Channel oc : partition.channels) {
253                      ((DISCARD) p).removeIgnoredMember(oc.getAddress());
254                   }
255                }
256             }
257          }
258       }

331    protected void splitCluster(int[]... parts) {
332       List<Address> allMembers = channel(0).getView().getMembers();
333       partitions = new Partition[parts.length];
334       for (int i = 0; i < parts.length; i++) {
335          Partition p = new Partition(allMembers);
336          for (int j : parts[i]) {
337             p.addNode(channel(j));
338          }
339          partitions[i] = p;
340          p.partition();
341       }
342    }
