77     public static class DelayTransfer implements Serializable {
78  
79        private static final long serialVersionUID = 6361429803359702822L;
80        
81        private volatile boolean doDelay = false;
82  
83        private void readObject(ObjectInputStream in) throws IOException, ClassNotFoundException {
84           in.defaultReadObject();
85        }
86  
87        private void writeObject(ObjectOutputStream out) throws IOException {
88           out.defaultWriteObject();
89  
90           if (doDelay) {
91              try {
92                 // Delay state transfer
93                 Thread.sleep(2000);
94              }
95              catch (InterruptedException e) {
96                 Thread.currentThread().interrupt();
97              }
98           }
99        }
100 
101       public void enableDelay() {
102          doDelay = true;
103       }
104    }

106    private static class WritingTask implements Callable<Integer> {
107       private final Cache<Object, Object> cache;
108       private final boolean tx;
109       private volatile boolean stop;
110       private TransactionManager tm;
111 
112       WritingTask(Cache<Object, Object> cache, boolean tx) {
113          this.cache = cache;
114          this.tx = tx;
115          if (tx) tm = TestingUtil.getTransactionManager(cache);
116       }
117 
118       @Override
119       public Integer call() throws Exception {
120          int c = 0;
121          while (!stop) {
122             boolean success = false;
123             try {
124                if (tx)
125                   tm.begin();
126                cache.put("test" + c, c);
127                if (tx)
128                   tm.commit();
129                success = true;
130                c++;
131 
132                // Without this, the writing thread would occupy 1 core completely before the 2nd node joins.
133                Thread.sleep(1);
134             } catch (Exception e) {
135                log.errorf(e, "Error writing key test%s", c);
136                stop();
137             } finally {
138                if (tx && !success) {
139                   try {
140                      tm.rollback();
141                   } catch (SystemException e) {
142                      log.error(e);
143                   }
144                }
145             }
146          }
147          return c;
148       }
149 
150       public void stop() {
151          stop = true;
152       }
153    }

189    public void testConcurrentStateTransfer(Method m) throws Exception {
190       testCount++;
191       logTestStart(m);
192       Cache<Object, Object> cache1, cache2, cache3, cache4;
193       cache1 = createCacheManager().getCache(cacheName);
194       writeInitialData(cache1);
195 
196       JoiningNode node2 = new JoiningNode(createCacheManager());
197       cache2 = node2.getCache(cacheName);
198 
199       cache1.put("delay", new DelayTransfer());
200 
201       node2.waitForJoin(60000, cache1, cache2);
202       node2.verifyStateTransfer(new CacheVerifier(cache2));
203 
204       final JoiningNode node3 = new JoiningNode(createCacheManager());
205       final JoiningNode node4 = new JoiningNode(createCacheManager());
206 
207       Thread t1 = new Thread(new Runnable() {
208          public void run() {
209             node3.getCache(cacheName);
210          }
211       });
212       t1.setName("CacheStarter-Cache3");
213       t1.start();
214 
215       Thread t2 = new Thread(new Runnable() {
216          public void run() {
217             node4.getCache(cacheName);
218          }
219       });
220       t2.setName("CacheStarter-Cache4");
221       t2.start();
222 
223       t1.join();
224       t2.join();
225 
226       cache3 = node3.getCache(cacheName);
227       cache4 = node4.getCache(cacheName);
228 
229       node3.waitForJoin(120000, cache1, cache2, cache3, cache4);
230       node4.waitForJoin(120000, cache1, cache2, cache3, cache4);
231 
232       node3.verifyStateTransfer(new CacheVerifier(cache3));
233       node4.verifyStateTransfer(new CacheVerifier(cache4));
234 
235       logTestEnd(m);
236    }

299    private void thirdWritingCacheTest(boolean tx) throws Exception {
300       Cache<Object, Object> cache1, cache2, cache3;
301       cache1 = createCacheManager().getCache(cacheName);
302       cache3 = createCacheManager().getCache(cacheName);
303       TestingUtil.blockUntilViewsReceived(60000, cache1, cache3);
304 
305       writeInitialData(cache1);
306 
307       // Delay the transient copy, so that we get a more thorough log test
308       DelayTransfer value = new DelayTransfer();
309       cache1.put("delay", value);
310       value.enableDelay();
311 
312       WritingTask writingTask = new WritingTask(cache3, tx);
313       Future<Integer> future = fork(writingTask);
314 
315       JoiningNode node2 = new JoiningNode(createCacheManager());
316       cache2 = node2.getCache(cacheName);
317 
318       node2.waitForJoin(60000, cache1, cache2, cache3);
319 
320       writingTask.stop();
321       int count = future.get(60, SECONDS);
322 
323       node2.verifyStateTransfer(new CacheVerifier(cache2));
324 
325       for (int c = 0; c < count; c++) {
326          assertEquals(c, cache2.get("test" + c));
327       }
328    }

330    protected void verifyInitialData(Cache<Object, Object> c) {
331       Address address = c.getAdvancedCache().getRpcManager().getAddress();
332       log.debugf("Checking values on cache " + address);
333       assertEquals("Incorrect value for key " + A_B_NAME, JOE, c.get(A_B_NAME));
334       assertEquals("Incorrect value for key " + A_B_AGE, TWENTY, c.get(A_B_AGE));
335       assertEquals("Incorrect value for key " + A_C_NAME, BOB, c.get(A_C_NAME));
336       assertEquals("Incorrect value for key " + A_C_AGE, FORTY, c.get(A_C_AGE));
337    }
