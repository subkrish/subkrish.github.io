25  /**
26   * A listener that listens for replication events on a cache it is watching.  Typical usage: <code> ReplListener r =
27   * attachReplicationListener(cache); r.expect(RemoveCommand.class); // ... r.waitForRPC(); </code>
28   */
29  public class ReplListener {
30     Cache<?, ?> c;
31     volatile List<Class<? extends VisitableCommand>> expectedCommands;
32     List<Class<? extends VisitableCommand>> eagerCommands = new LinkedList<Class<? extends VisitableCommand>>();
33     boolean recordCommandsEagerly;
34     boolean watchLocal;
35     final Lock expectationSetupLock = new ReentrantLock();
36     CountDownLatch latch = new CountDownLatch(1);
37     volatile boolean sawAtLeastOneInvocation = false;
38     boolean expectAny = false;
39     private Log log = LogFactory.getLog(ReplListener.class);
40  
41     /**
42      * This listener attaches itself to a cache and when {@link #expect(Class[])} is invoked, will start checking for
43      * invocations of the command on the cache, waiting for all expected commands to be received in {@link
44      * #waitForRpc()}.
45      *
46      * @param c cache on which to attach listener
47      */
48     public ReplListener(Cache<?, ?> c) {
49        this(c, false);
50     }
51  
52     /**
53      * As {@link #ReplListener(org.infinispan.Cache)} except that you can optionally configure whether command recording
54      * is eager (false by default).
55      * <p/>
56      * If <tt>recordCommandsEagerly</tt> is true, then commands are recorded from the moment the listener is attached to
57      * the cache, even before {@link #expect(Class[])} is invoked.  As such, when {@link #expect(Class[])} is called, the
58      * list of commands to wait for will take into account commands already seen thanks to eager recording.
59      *
60      * @param c                     cache on which to attach listener
61      * @param recordCommandsEagerly whether to record commands eagerly
62      */
63     public ReplListener(Cache<?, ?> c, boolean recordCommandsEagerly) {
64        this(c, recordCommandsEagerly, false);
65     }
66  
67     /**
68      * Same as {@link #ReplListener(org.infinispan.Cache, boolean)} except that this constructor allows you to set the
69      * watchLocal parameter.  If true, even local events are recorded (not just ones that originate remotely).
70      *
71      * @param c                     cache on which to attach listener
72      * @param recordCommandsEagerly whether to record commands eagerly
73      * @param watchLocal            if true, local events are watched for as well
74      */
75     public ReplListener(Cache<?, ?> c, boolean recordCommandsEagerly, boolean watchLocal) {
76        this.c = c;
77        this.recordCommandsEagerly = recordCommandsEagerly;
78        this.watchLocal = watchLocal;
79        this.c.getAdvancedCache().addInterceptor(new ReplListenerInterceptor(), 1);
80     }

82     /**
83      * Expects any commands.  The moment a single command is detected, the {@link #waitForRpc()} command will be
84      * unblocked.
85      */
86     public void expectAny() {
87        expectAny = true;
88        expect();
89     }

91     /**
92      * Expects a specific set of commands, within transactional scope (i.e., as a payload to a PrepareCommand).  If the
93      * cache mode is synchronous, a CommitCommand is expected as well.
94      *
95      * @param commands commands to expect (not counting transaction boundary commands like PrepareCommand and
96      *                 CommitCommand)
97      */
98     @SuppressWarnings("unchecked")
99     public void expectWithTx(Class<? extends VisitableCommand>... commands) {
100       List<Class<? extends VisitableCommand>> cmdsToExpect = new ArrayList<Class<? extends VisitableCommand>>();
101       cmdsToExpect.add(PrepareCommand.class);
102       if (commands != null) cmdsToExpect.addAll(Arrays.asList(commands));
103       //this is because for async replication we have an 1pc transaction
104       if (c.getCacheConfiguration().clustering().cacheMode().isSynchronous()) cmdsToExpect.add(CommitCommand.class);
105 
106       expect(cmdsToExpect.toArray(new Class[cmdsToExpect.size()]));
107    }

109    /**
110     * Expects any commands, within transactional scope (i.e., as a payload to a PrepareCommand).  If the cache mode is
111     * synchronous, a CommitCommand is expected as well.
112     */
113    @SuppressWarnings("unchecked")
114    public void expectAnyWithTx() {
115       List<Class<? extends VisitableCommand>> cmdsToExpect = new ArrayList<Class<? extends VisitableCommand>>(2);
116       cmdsToExpect.add(PrepareCommand.class);
117       //this is because for async replication we have an 1pc transaction
118       if (c.getCacheConfiguration().clustering().cacheMode().isSynchronous()) cmdsToExpect.add(CommitCommand.class);
119 
120       expect(cmdsToExpect.toArray(new Class[cmdsToExpect.size()]));
121    }

123    /**
124     * Expects a specific set of commands.  {@link #waitForRpc()} will block until all of these commands are detected.
125     *
126     * @param expectedCommands commands to expect
127     */
128    public void expect(Class<? extends VisitableCommand>... expectedCommands) {
129       expectationSetupLock.lock();
130       try {
131          if (this.expectedCommands == null) {
132             this.expectedCommands = new CopyOnWriteArrayList<Class<? extends VisitableCommand>>();
133          }
134          this.expectedCommands.addAll(Arrays.asList(expectedCommands));
135          info("Setting expected commands to " + this.expectedCommands);
136          info("Record eagerly is " + recordCommandsEagerly + ", and eager commands are " + eagerCommands);
137          if (recordCommandsEagerly) {
138             this.expectedCommands.removeAll(eagerCommands);
139             if (!eagerCommands.isEmpty()) sawAtLeastOneInvocation = true;
140             eagerCommands.clear();
141          }
142       } finally {
143          expectationSetupLock.unlock();
144       }
145    }
146 
147    private void info(String str) {
148       log.info(" [" + c + "] " + str);
149    }

151    /**
152     * Blocks for a predefined amount of time (120 Seconds) until commands defined in any of the expect*() methods have
153     * been detected.  If the commands have not been detected by this time, an exception is thrown.
154     */
155    public void waitForRpc() {
156       waitForRpc(30, TimeUnit.SECONDS);
157    }
158 
159    /**
160     * The same as {@link #waitForRpc()} except that you are allowed to specify the max wait time.
161     */
162    public void waitForRpc(long time, TimeUnit unit) {
163       assert expectedCommands != null : "there are no replication expectations; please use ReplListener.expect() before calling this method";
164       try {
165          boolean successful = (expectAny && sawAtLeastOneInvocation) || (!expectAny && expectedCommands.isEmpty());
166          info("Expect Any is " + expectAny + ", saw at least one? " + sawAtLeastOneInvocation + " Successful? " + successful + " Expected commands " + expectedCommands);
167          if (!successful && !latch.await(time, unit)) {
168             EmbeddedCacheManager cacheManager = c.getCacheManager();
169             assert false : "Waiting for more than " + time + " " + unit + " and following commands did not replicate: " + expectedCommands + " on cache [" + cacheManager.getAddress() + "]";
170          } else {
171             info("Exiting wait for rpc with expected commands " + expectedCommands);
172          }
173       }
174       catch (InterruptedException e) {
175          throw new IllegalStateException("unexpected", e);
176       }
177       finally {
178          expectationSetupLock.lock();
179          expectedCommands = null;
180          expectationSetupLock.unlock();
181          expectAny = false;
182          sawAtLeastOneInvocation = false;
183          latch = new CountDownLatch(1);
184          eagerCommands.clear();
185       }
186    }

188    public Cache<?, ?> getCache() {
189       return c;

193       eagerCommands.clear();

201    protected class ReplListenerInterceptor extends CommandInterceptor {
202       @Override
203       protected Object handleDefault(InvocationContext ctx, VisitableCommand cmd) throws Throwable {
204          // first pass up chain
205          Object o;
206          try {
207             o = invokeNextInterceptor(ctx, cmd);
208          } finally {//make sure we do mark this command as received even in the case of exceptions(e.g. timeouts)
209             info("Checking whether command " + cmd.getClass().getSimpleName() + " should be marked as local with watch local set to " + watchLocal);
210             if (!ctx.isOriginLocal() || watchLocal) markAsVisited(cmd);
211          }
212          return o;
213       }
214 
215       @Override
216       public Object visitPrepareCommand(TxInvocationContext ctx, PrepareCommand cmd) throws Throwable {
217          // first pass up chain
218          Object o = invokeNextInterceptor(ctx, cmd);
219          if (!ctx.isOriginLocal() || watchLocal) {
220             markAsVisited(cmd);
221             for (WriteCommand mod : cmd.getModifications()) markAsVisited(mod);
222          }
223          return o;
224       }
225 
226       private void markAsVisited(VisitableCommand cmd) {
227          expectationSetupLock.lock();
228          try {
229             info("ReplListener saw command " + cmd);
230             if (expectedCommands != null) {
231                if (expectedCommands.remove(cmd.getClass())) {
232                   info("Successfully removed command: " + cmd.getClass());
233                }
234                else {
235                   if (recordCommandsEagerly) eagerCommands.add(cmd.getClass());
236                }
237                sawAtLeastOneInvocation = true;
238                if (expectedCommands.isEmpty()) {
239                   info("Nothing to wait for, releasing latch");
240                   latch.countDown();
241                }
242             } else {
243                if (recordCommandsEagerly) eagerCommands.add(cmd.getClass());
244             }
245          } finally {
246             expectationSetupLock.unlock();
247          }
248       }
249    }
