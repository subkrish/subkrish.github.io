13  public enum ExtendedStatistic {

22     //Read-only transaction related statistics

28     //Write transaction access

31     NUM_GETS_WR_TX(true, false), //number of gets on local keys per write transaction
32     NUM_REMOTE_GETS_WR_TX(true, false), //number of gets on remote keys per write transaction
33  
34     //Read-Only transaction access

36     NUM_REMOTE_GETS_RO_TX(true, false), //number of gets on remote keys per read-only transaction
37  
38     //Accesses (used during runtime. when the transaction ends, they are copied to the access above)
39     NUM_REMOTE_GET(true, false), //number of gets on remote keys in the transaction
40     NUM_GET(true, false), //number of gets on local keys in the transaction
41     NUM_REMOTE_PUT(true, false), //number of gets on remote keys in the transaction
42     NUM_PUT(true, false), //number of gets on local keys in the transaction
43  
44     //Per operation execution time
45     REMOTE_GET_EXECUTION(true, false), //get execution time for a remote keys
46     LOCAL_GET_EXECUTION(true, false), //get execution time for a local keys
47     ALL_GET_EXECUTION(true, false), //TODO: fixme
48     REMOTE_PUT_EXECUTION(true, false), //put execution time for a remote key
49     LOCAL_PUT_EXECUTION(true, false), //put execution time for a local key
50  
51     //Transaction counter
52     //TODO add the remaining
53     NUM_COMMITTED_TX(false, false), //the total number of committed remote and local transactions
54     NUM_LOCAL_COMMITTED_TX(false, false), //the total number of committed local transactions
55  
56     //Lock related
57     LOCK_HOLD_TIME(true, true), //the average lock hold time
58     NUM_HELD_LOCKS(true, true), //the total number of locks acquired
59     LOCK_HOLD_TIME_SUCCESS_LOCAL_TX(true, false), //the average lock hold time
60     NUM_HELD_LOCKS_SUCCESS_LOCAL_TX(true, false), //number of locks held by a successful local transaction
61     NUM_LOCK_FAILED_TIMEOUT(true, false), //number of locks failed to acquire due to timeout
62     NUM_LOCK_FAILED_DEADLOCK(true, false), //number of locks failed to acquire due to deadlock (deadlock exception)
63     NUM_WAITED_FOR_LOCKS(true, true), //number of lock that were contented and the transaction have to wait
64     LOCK_WAITING_TIME(true, true), //the waiting time before acquire the lock
65  
66     LOCK_HOLD_TIME_LOCAL(false, false), //query only: average lock hold time by local transactions
67     LOCK_HOLD_TIME_REMOTE(false, false), //query only: average lock hold time by remote transaction
68     NUM_LOCK_PER_LOCAL_TX(false, false), //query only: average number of locks per all local transaction
69     NUM_LOCK_PER_REMOTE_TX(false, false), //query only: average number of locks per all remote transaction
70  
71     //other transactional stuff
72     WRITE_TX_PERCENTAGE(false, false), //query only: write transaction percentage
73     SUCCESSFUL_WRITE_TX_PERCENTAGE(false, false), //query only: same as above but only for successful transactions
74  
75     //write skew
76     NUM_WRITE_SKEW(true, false), //number of write skew detected
77     WRITE_SKEW_PROBABILITY(false, false), //query only: write skew probability
78  
79     //Abort rate, arrival rate and throughput
80     ABORT_RATE(false, false), //percentage of aborted transactions
81     ARRIVAL_RATE(false, false), // number of local and remote transactions processed per second
82     THROUGHPUT(false, false), //number of local transactions processed per second
83  
84     //commands execution time
85     PREPARE_EXECUTION_TIME(true, true), //prepare command execution time (includes communication)
86     NUM_PREPARE_COMMAND(true, true), //number of prepare commands
87     LOCAL_PREPARE_EXECUTION_TIME(false, false), //query only: average prepare command execution time for local transactions
88     REMOTE_PREPARE_EXECUTION_TIME(false, false), //query only: average prepare command execution time for remote transactions
89  
90     COMMIT_EXECUTION_TIME(true, true), //commit command execution time (includes communication)
91     NUM_COMMIT_COMMAND(true, true), //number of commit commands
92     LOCAL_COMMIT_EXECUTION_TIME(false, false), //query only: average commit command execution time for local transactions
93     REMOTE_COMMIT_EXECUTION_TIME(false, false), //query only: average commit command execution time for remote transactions
94  
95     ROLLBACK_EXECUTION_TIME(true, true), //rollback command execution time (includes communication)
96     NUM_ROLLBACK_COMMAND(true, true), //number of rollback commands
97     LOCAL_ROLLBACK_EXECUTION_TIME(false, false), //query only: average rollback command execution time for local transactions
98     REMOTE_ROLLBACK_EXECUTION_TIME(false, false), //query only: average rollback command execution time for remote transactions
99

101    PREPARE_COMMAND_SIZE(true, false), //prepare command size in bytes
102    COMMIT_COMMAND_SIZE(true, false), //commit command size in bytes
103    CLUSTERED_GET_COMMAND_SIZE(true, false), //cluster get command size in bytes
104 
105    //RTT: local && synchronous communication
106    NUM_SYNC_PREPARE(true, false), //number of synchronous prepare command sent
107    SYNC_PREPARE_TIME(true, false), //response time of synchronous prepare command
108    NUM_SYNC_COMMIT(true, false), //number of synchronous commit command sent
109    SYNC_COMMIT_TIME(true, false), //response time of synchronous commit command
110    NUM_SYNC_ROLLBACK(true, false), //number of synchronous rollback command sent
111    SYNC_ROLLBACK_TIME(true, false), //response time of synchronous rollback command
112    NUM_SYNC_GET(true, false), //number of synchronous cluster get command sent
113    SYNC_GET_TIME(true, false), //response time of synchronous cluster get command
114 
115    //ASYNC: local && asynchronous communication
116    ASYNC_PREPARE_TIME(true, false), //send-and-forget time for prepare command
117    NUM_ASYNC_PREPARE(true, false), //number of asynchronous prepare command sent
118    ASYNC_COMMIT_TIME(true, false), //send-and-forget time for commit command
119    NUM_ASYNC_COMMIT(true, false), //number of asynchronous commit command sent
120    ASYNC_ROLLBACK_TIME(true, false), //send-and-forget time for rollback command
121    NUM_ASYNC_ROLLBACK(true, false), //number of asynchronous rollback command sent
122    ASYNC_COMPLETE_NOTIFY_TIME(true, false), //send-and-forget time for tx completion notify command
123    NUM_ASYNC_COMPLETE_NOTIFY(true, false), //number of asynchronous tx completion notify command sent
124 
125    //Number of nodes in remote communication
126    NUM_NODES_PREPARE(true, false), //average number of nodes involved in a prepare command
127    NUM_NODES_COMMIT(true, false), //average number of nodes involved in a commit command
128    NUM_NODES_ROLLBACK(true, false), //average number of nodes involved in a rollback command
129    NUM_NODES_COMPLETE_NOTIFY(true, false), //average number of nodes involved in a tx completion notify command
130    NUM_NODES_GET(true, false), //average number of nodes involved in a tx completion notify command
131    RESPONSE_TIME(false, false); //average transaction response time
132    public static final int NO_INDEX = -1;
133    private static short localStatsSize = 0;
134    private static short remoteStatsSize = 0;
135    private final boolean local;
136    private final boolean remote;
137    private short localIndex = NO_INDEX;
138    private short remoteIndex = NO_INDEX;
139 
140    ExtendedStatistic(boolean local, boolean remote) {
141       this.local = local;
142       this.remote = remote;
143    }
144 
145    public final int getLocalIndex() {
146       return localIndex;
147    }
148 
149    public final int getRemoteIndex() {
150       return remoteIndex;
151    }
152 
153    public final boolean isLocal() {
154       return local;
155    }
156 
157    public final boolean isRemote() {
158       return remote;
159    }
160 
161    public static int getRemoteStatsSize() {
162       return remoteStatsSize;
163    }
164 
165    public static int getLocalStatsSize() {
166       return localStatsSize;
167    }
168 
169    static {
170       for (ExtendedStatistic stat : values()) {
171          if (stat.local) {
172             stat.localIndex = localStatsSize++;
173          }
174          if (stat.remote) {
175             stat.remoteIndex = remoteStatsSize++;
176          }
177       }
178    }
