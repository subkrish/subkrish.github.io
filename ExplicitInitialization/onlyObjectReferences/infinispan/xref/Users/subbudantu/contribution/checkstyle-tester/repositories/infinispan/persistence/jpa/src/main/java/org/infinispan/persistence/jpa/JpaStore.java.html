77     @Override
78     public void start() {
79        try {
80           this.emf = emfRegistry.getEntityManagerFactory(configuration.persistenceUnitName());
81        } catch (PersistenceException e) {
82           throw new JpaStoreException("Persistence Unit [" + this.configuration.persistenceUnitName() + "] not found", e);
83        }
84  
85        ManagedType<?> mt;
86        try {
87           mt = emf.getMetamodel().entity(this.configuration.entityClass());
88        } catch (IllegalArgumentException e) {
89           throw new JpaStoreException("Entity class [" + this.configuration.entityClass().getName() + " specified in configuration is not recognized by the EntityManagerFactory with Persistence Unit [" + this.configuration.persistenceUnitName() + "]", e);
90        }
91  
92        if (!(mt instanceof IdentifiableType)) {
93           throw new JpaStoreException(
94                 "Entity class must have one and only one identifier (@Id or @EmbeddedId)");
95        }
96        IdentifiableType<?> it = (IdentifiableType<?>) mt;
97        if (!it.hasSingleIdAttribute()) {
98           throw new JpaStoreException(
99                 "Entity class has more than one identifier.  It must have only one identifier.");
100       }
101 
102       Type<?> idType = it.getIdType();
103       Class<?> idJavaType = idType.getJavaType();
104 
105       if (idJavaType.isAnnotationPresent(GeneratedValue.class)) {
106          throw new JpaStoreException(
107                "Entity class has one identifier, but it must not have @GeneratedValue annotation");
108       }
109 
110       // Hack: MySQL needs to have fetchSize set to Integer.MIN_VALUE in order to do streaming
111       SessionFactory sessionFactory = emf.createEntityManager().unwrap(Session.class).getSessionFactory();
112       if (sessionFactory instanceof SessionFactoryImplementor) {
113          Dialect dialect = ((SessionFactoryImplementor) sessionFactory).getDialect();
114          if (dialect instanceof MySQLDialect) {
115             setFetchSizeMinInteger = true;
116          }
117       }
118    }

193    @Override
194    public void clear() {
195       EntityManager emStream = emf.createEntityManager();
196       try {
197          ScrollableResults results = null;
198          ArrayList<Object> batch = new ArrayList<Object>((int) configuration.batchSize());
199          EntityTransaction txStream = emStream.getTransaction();
200          txStream.begin();
201          try {
202             log.trace("Clearing JPA Store");
203             Session session = emStream.unwrap(Session.class);
204             Criteria criteria = session.createCriteria(configuration.entityClass()).setReadOnly(true)
205                   .setProjection(Projections.id());
206             if (setFetchSizeMinInteger) {
207                criteria.setFetchSize(Integer.MIN_VALUE);
208             }
209             results = criteria.scroll(ScrollMode.FORWARD_ONLY);
210 
211             while (results.next()) {
212                Object o = results.get(0);
213                batch.add(o);
214                if (batch.size() == configuration.batchSize()) {
215                   session.clear();
216                   removeBatch(batch);
217                }
218             }
219             if (configuration.storeMetadata()) {
220                /* We have to close the stream before executing further request */
221                results.close();
222                results = null;
223                String metadataTable = emStream.getMetamodel().entity(MetadataEntity.class).getName();
224                Query clearMetadata = emStream.createQuery("DELETE FROM " + metadataTable);
225                clearMetadata.executeUpdate();
226             }
227             txStream.commit();
228          } finally {
229             removeBatch(batch);
230             if (results != null) {
231                results.close();
232             }
233             if (txStream != null && txStream.isActive()) {
234                txStream.rollback();
235             }
236          }
237       } catch (RuntimeException e) {
238          log.error("Error in clear", e);
239          throw e;
240       } finally {
241          emStream.close();
242       }
243    }

245    private void removeBatch(ArrayList<Object> batch) {
246       for (int i = 10; i >= 0; --i) {
247          EntityManager emExec = emf.createEntityManager();
248          EntityTransaction txn = null;
249          try {
250             txn = emExec.getTransaction();
251             txn.begin();
252             try {
253                for (Object key : batch) {
254                   try {
255                      Object entity = emExec.getReference(configuration.entityClass(), key);
256                      removeEntity(emExec, entity);
257                   } catch (EntityNotFoundException e) {
258                      log.trace("Cleared entity with key " + key + " not found", e);
259                   }
260                }
261                txn.commit();
262                batch.clear();
263                break;
264             } catch (Exception e) {
265                if (i != 0) {
266                   log.trace("Remove batch failed once", e);
267                } else {
268                   throw new JpaStoreException("Remove batch failing", e);
269                }
270             }
271          } finally {
272             if (txn != null && txn.isActive())
273                txn.rollback();
274             emExec.close();
275          }
276       }

279    public boolean delete(Object key) {
280       if (!isValidKeyType(key)) {
281          return false;
282       }
283 
284       EntityManager em = emf.createEntityManager();
285       try {
286          Object entity = findEntity(em, key);
287          if (entity == null) {
288             return false;
289          }
290          MetadataEntity metadata = null;
291          if (configuration.storeMetadata()) {
292             byte[] keyBytes;
293             try {
294                keyBytes = marshaller.objectToByteBuffer(key);
295             } catch (Exception e) {
296                throw new JpaStoreException("Failed to marshall key", e);
297             }
298             metadata = findMetadata(em, new MetadataEntityKey(keyBytes));
299          }
300 
301          EntityTransaction txn = em.getTransaction();
302          if (trace) log.trace("Removing " + entity + "(" + toString(metadata) + ")");
303          long txnBegin = timeService.time();
304          txn.begin();
305          try {
306             removeEntity(em, entity);
307             if (metadata != null) {
308                removeMetadata(em, metadata);
309             }
310             txn.commit();
311             stats.addRemoveTxCommitted(timeService.time() - txnBegin);
312             return true;
313          } catch (Exception e) {
314             stats.addRemoveTxFailed(timeService.time() - txnBegin);
315             throw new JpaStoreException("Exception caught in delete()", e);
316          } finally {
317             if (txn != null && txn.isActive())
318                txn.rollback();
319          }
320       } finally {
321          em.close();
322       }
323    }

325    @Override
326    public void write(MarshalledEntry entry) {
327       EntityManager em = emf.createEntityManager();
328 
329       Object entity = entry.getValue();
330       MetadataEntity metadata = configuration.storeMetadata() ?
331             new MetadataEntity(entry.getKeyBytes(), entry.getMetadataBytes(),
332                   entry.getMetadata() == null ? Long.MAX_VALUE : entry.getMetadata().expiryTime()) : null;
333       try {
334          if (!configuration.entityClass().isAssignableFrom(entity.getClass())) {
335             throw new JpaStoreException(String.format(
336                   "This cache is configured with JPA CacheStore to only store values of type %s - cannot write %s = %s",
337                   configuration.entityClass().getName(), entity, entity.getClass().getName()));
338          } else {
339             EntityTransaction txn = em.getTransaction();
340             Object id = emf.getPersistenceUnitUtil().getIdentifier(entity);
341             if (!entry.getKey().equals(id)) {
342                throw new JpaStoreException(
343                      "Entity id value must equal to key of cache entry: "
344                            + "key = [" + entry.getKey() + "], id = ["
345                            + id + "]");
346             }
347             long txnBegin = timeService.time();
348             try {
349                if (trace) log.trace("Writing " + entity + "(" + toString(metadata) + ")");
350                txn.begin();
351 
352                mergeEntity(em, entity);
353                if (metadata != null && metadata.hasBytes()) {
354                   mergeMetadata(em, metadata);
355                }
356 
357                txn.commit();
358                stats.addWriteTxCommited(timeService.time() - txnBegin);
359             } catch (Exception e) {
360                stats.addWriteTxFailed(timeService.time() - txnBegin);
361                throw new JpaStoreException("Exception caught in write()", e);
362             } finally {
363                if (txn != null && txn.isActive())
364                   txn.rollback();
365             }
366          }
367       } finally {
368          em.close();
369       }

373    @Override
374    public boolean contains(Object key) {
375       if (!isValidKeyType(key)) {
376          return false;
377       }
378 
379       EntityManager em = emf.createEntityManager();
380       try {
381          EntityTransaction txn = em.getTransaction();
382          long txnBegin = timeService.time();
383          txn.begin();
384          try {
385             Object entity = findEntity(em, key);
386             if (trace) log.trace("Entity " + key + " -> " + entity);
387             try {
388                if (entity == null) return false;
389                if (configuration.storeMetadata()) {
390                   byte[] keyBytes;
391                   try {
392                      keyBytes = marshaller.objectToByteBuffer(key);
393                   } catch (Exception e) {
394                      throw new JpaStoreException("Cannot marshall key", e);
395                   }
396                   MetadataEntity metadata = findMetadata(em, new MetadataEntityKey(keyBytes));
397                   if (trace) log.trace("Metadata " + key + " -> " + toString(metadata));
398                   return metadata == null || metadata.getExpiration() > timeService.wallClockTime();
399                } else {
400                   return true;
401                }
402             } finally {
403                txn.commit();
404                stats.addReadTxCommitted(timeService.time() - txnBegin);
405             }
406          } catch (RuntimeException e) {
407             stats.addReadTxFailed(timeService.time() - txnBegin);
408             throw e;
409          } finally {
410             if (txn != null && txn.isActive())
411                txn.rollback();
412          }
413       } finally {
414          em.close();
415       }
416    }

418    @Override
419    public MarshalledEntry load(Object key) {
420       if (!isValidKeyType(key)) {
421          return null;
422       }
423 
424       EntityManager em = emf.createEntityManager();
425       try {
426          EntityTransaction txn = em.getTransaction();
427          long txnBegin = timeService.time();
428          txn.begin();
429          try {
430             Object entity = findEntity(em, key);
431             try {
432                if (entity == null)
433                   return null;
434                InternalMetadata m = null;
435                if (configuration.storeMetadata()) {
436                   byte[] keyBytes;
437                   try {
438                      keyBytes = marshaller.objectToByteBuffer(key);
439                   } catch (Exception e) {
440                      throw new JpaStoreException("Failed to marshall key", e);
441                   }
442                   MetadataEntity metadata = findMetadata(em, new MetadataEntityKey(keyBytes));
443                   if (metadata != null && metadata.getMetadata() != null) {
444                      try {
445                         m = (InternalMetadata) marshaller.objectFromByteBuffer(metadata.getMetadata());
446                      } catch (Exception e) {
447                         throw new JpaStoreException("Failed to unmarshall metadata", e);
448                      }
449                      if (m.isExpired(timeService.wallClockTime())) {
450                         return null;
451                      }
452                   }
453                }
454                if (trace) log.trace("Loaded " + entity + " (" + m + ")");
455                return marshallerEntryFactory.newMarshalledEntry(key, entity, m);
456             } finally {
457                try {
458                   txn.commit();
459                   stats.addReadTxCommitted(timeService.time() - txnBegin);
460                } catch (Exception e) {
461                   stats.addReadTxFailed(timeService.time() - txnBegin);
462                   throw new JpaStoreException("Failed to load entry", e);
463                }
464             }
465          } finally {
466             if (txn != null && txn.isActive())
467                txn.rollback();
468          }
469       } finally {
470          em.close();
471       }
472    }

474    @Override
475    public void process(KeyFilter filter, CacheLoaderTask task, Executor executor, boolean fetchValue, boolean fetchMetadata) {
476       if (fetchMetadata && !configuration.storeMetadata()) {
477          log.debug("Metadata cannot be retrieved as JPA Store is not configured to persist metadata.");
478          fetchMetadata = false;
479       }
480       // We cannot stream entities as a full table as the entity can contain collections in another tables.
481       // Then, Hibernate uses left outer joins which give us several rows of results for each entity.
482       // We cannot use DISTINCT_ROOT_ENTITY transformer when streaming, that's only available for list()
483       // and this is prone to OOMEs. Theoretically, custom join is possible but it is out of scope
484       // of current JpaStore implementation.
485       // We also can't switch JOINs to SELECTs as some DBs (e.g. MySQL) fails when another command is executed
486       // on the connection when streaming.
487       // Therefore, we can only query IDs and do a findEntity for each fetch.
488 
489       // Another problem: even for fetchValue=false and fetchMetadata=true, we cannot iterate over metadata
490       // table, because this table does not have records for keys without metadata. With such iteration,
491       // we wouldn't iterate over all keys - therefore, we can iterate only over entity table IDs and we
492       // have to request the metadata in separate connection.
493       if (fetchValue || fetchMetadata) {
494          final boolean fv = fetchValue;
495          final boolean fm = fetchMetadata;
496          process(filter, task, executor, new ProcessStrategy() {
497             @Override
498             public Criteria getCriteria(Session session) {
499                return session.createCriteria(configuration.entityClass()).setProjection(Projections.id());
500             }
501 
502             @Override
503             public Object getKey(Object scrollResult) {
504                return scrollResult;
505             }
506 
507             @Override
508             public Callable<Void> getTask(CacheLoaderTask task, TaskContext taskContext, Object scrollResult, Object key) {
509                return new LoadingProcessTask(task, taskContext, key, fv, fm);
510             }
511          });
512       } else {
513          process(filter, task, executor, new ProcessStrategy() {
514             @Override
515             public Criteria getCriteria(Session session) {
516                return session.createCriteria(configuration.entityClass()).setProjection(Projections.id());
517             }
518 
519             @Override
520             public Object getKey(Object scrollResult) {
521                return scrollResult;
522             }
523 
524             @Override
525             public Callable<Void> getTask(CacheLoaderTask task, TaskContext taskContext, Object scrollResult, Object key) {
526                return new ProcessTask(task, taskContext, key, null, null);
527             }
528          });
529       }
530    }

532    private void process(KeyFilter filter, final CacheLoaderTask task, Executor executor, ProcessStrategy strategy) {
533       ExecutorAllCompletionService eacs = new ExecutorAllCompletionService(executor);
534       TaskContextImpl taskContext = new TaskContextImpl();
535       EntityManager emStream = emf.createEntityManager();
536       try {
537          EntityTransaction txStream = emStream.getTransaction();
538          ScrollableResults results = null;
539          txStream.begin();
540          try {
541             Session session = emStream.unwrap(Session.class);
542             Criteria criteria = strategy.getCriteria(session).setReadOnly(true);
543             if (setFetchSizeMinInteger) {
544                criteria.setFetchSize(Integer.MIN_VALUE);
545             }
546             results = criteria.scroll(ScrollMode.FORWARD_ONLY);
547             try {
548                while (results.next()) {
549                   if (taskContext.isStopped())
550                      break;
551                   Object result = results.get(0);
552                   Object key = strategy.getKey(result);
553                   if (filter != null && !filter.accept(key)) {
554                      if (trace) log.trace("Key " + key + " filtered");
555                      continue;
556                   }
557                   eacs.submit(strategy.getTask(task, taskContext, result, key));
558                }
559             } finally {
560                if (results != null) results.close();
561             }
562             txStream.commit();
563          } finally {
564             if (txStream != null && txStream.isActive()) {
565                txStream.rollback();
566             }
567          }
568       } finally {
569          emStream.close();
570       }
571       eacs.waitUntilAllCompleted();
572       if (eacs.isExceptionThrown()) {
573          throw new org.infinispan.persistence.spi.PersistenceException("Execution exception!", eacs.getFirstException());
574       }
575    }

577    private InternalMetadata getMetadata(EntityManager em, Object key) {
578       byte[] keyBytes;
579       try {
580          keyBytes = marshaller.objectToByteBuffer(key);
581       } catch (Exception e) {
582          throw new JpaStoreException("Failed to marshall key", e);
583       }
584       MetadataEntity m = findMetadata(em, new MetadataEntityKey(keyBytes));
585       if (m == null) return null;
586 
587       try {
588          return (InternalMetadata) marshaller.objectFromByteBuffer(m.getMetadata());
589       } catch (Exception e) {
590          throw new JpaStoreException("Failed to unmarshall metadata", e);
591       }
592    }

594    @Override
595    public int size() {
596       EntityManager em = emf.createEntityManager();
597       EntityTransaction txn = em.getTransaction();
598       txn.begin();
599       try {
600          CriteriaBuilder builder = em.getCriteriaBuilder();
601          CriteriaQuery<Long> cq = builder.createQuery(Long.class);
602          cq.select(builder.count(cq.from(configuration.entityClass())));
603          return em.createQuery(cq).getSingleResult().intValue();
604       } finally {
605          try {
606             txn.commit();
607          } finally {
608             if (txn != null && txn.isActive()) {
609                txn.rollback();
610             }
611          }
612          em.close();
613       }
614    }

616    @Override
617    public void purge(Executor threadPool, final PurgeListener listener) {
618       if (!configuration.storeMetadata()) {
619          log.debug("JPA Store cannot be purged as metadata holding expirations are not available");
620          return;
621       }
622       ExecutorAllCompletionService eacs = new ExecutorAllCompletionService(threadPool);
623       EntityManager emStream = emf.createEntityManager();
624       try {
625          EntityTransaction txStream = emStream.getTransaction();
626          ScrollableResults metadataKeys = null;
627          txStream.begin();
628          try {
629             long currentTime = timeService.wallClockTime();
630             Session session = emStream.unwrap(Session.class);
631             Criteria criteria = session.createCriteria(MetadataEntity.class).setReadOnly(true)
632                   .add(Restrictions.le(MetadataEntity.EXPIRATION, currentTime)).setProjection(Projections.id());
633             if (setFetchSizeMinInteger) {
634                criteria.setFetchSize(Integer.MIN_VALUE);
635             }
636 
637             metadataKeys = criteria.scroll(ScrollMode.FORWARD_ONLY);
638             ArrayList<MetadataEntityKey> batch = new ArrayList<MetadataEntityKey>((int) configuration.batchSize());
639             while (metadataKeys.next()) {
640                MetadataEntityKey mKey = (MetadataEntityKey) metadataKeys.get(0);
641                batch.add(mKey);
642                if (batch.size() == configuration.batchSize()) {
643                   purgeBatch(batch, listener, eacs, currentTime);
644                   batch.clear();
645                }
646             }
647             purgeBatch(batch, listener, eacs, currentTime);
648             txStream.commit();
649          } finally {
650             if (metadataKeys != null) metadataKeys.close();
651             if (txStream != null && txStream.isActive()) {
652                txStream.rollback();
653             }
654          }
655       } finally {
656          emStream.close();
657       }
658       eacs.waitUntilAllCompleted();
659       if (eacs.isExceptionThrown()) {
660          throw new JpaStoreException(eacs.getFirstException());
661       }
662    }

664    private void purgeBatch(List<MetadataEntityKey> batch, final PurgeListener listener, ExecutorAllCompletionService eacs, long currentTime) {
665       EntityManager emExec = emf.createEntityManager();
666       try {
667          EntityTransaction txn = emExec.getTransaction();
668          txn.begin();
669          try {
670             for (MetadataEntityKey metadataKey : batch) {
671                MetadataEntity metadata = findMetadata(emExec, metadataKey);
672                // check for transaction - I hope write skew check is done here
673                if (metadata.getExpiration() > currentTime) {
674                   continue;
675                }
676 
677                final Object key;
678                try {
679                   key = marshaller.objectFromByteBuffer(metadata.getKeyBytes());
680                } catch (Exception e) {
681                   throw new JpaStoreException("Cannot unmarshall key", e);
682                }
683                Object entity = null;
684                try {
685                   entity = emExec.getReference(configuration.entityClass(), key);
686                   removeEntity(emExec, entity);
687                } catch (EntityNotFoundException e) {
688                   log.trace("Expired entity with key " + key + " not found", e);
689                }
690                removeMetadata(emExec, metadata);
691 
692                if (trace) log.trace("Expired " + key + " -> " + entity + "(" + toString(metadata) + ")");
693                if (listener != null) {
694                   eacs.submit(new Runnable() {
695                      @Override
696                      public void run() {
697                         listener.entryPurged(key);
698                      }
699                   }, null);
700                }
701             }
702             txn.commit();
703          } finally {
704             if (txn != null && txn.isActive()) {
705                txn.rollback();
706             }
707          }
708       } finally {
709          emExec.close();
710       }
711    }
712 
713    private String toString(MetadataEntity metadata) {
714       if (metadata == null || !metadata.hasBytes()) return "<no metadata>";
715       try {
716          return marshaller.objectFromByteBuffer(metadata.getMetadata()).toString();
717       } catch (Exception e) {
718          log.trace("Failed to unmarshall metadata", e);
719          return "<metadata: " + e + ">";
720       }
721    }

729    private class ProcessTask implements Callable<Void> {
730       private final CacheLoaderTask task;
731       private final TaskContext taskContext;
732       private final Object key;
733       private final Object entity;
734       private final InternalMetadata metadata;
735 
736       private ProcessTask(CacheLoaderTask task, TaskContext taskContext, Object key, Object entity, InternalMetadata metadata) {
737          this.task = task;
738          this.taskContext = taskContext;
739          this.key = key;
740          this.entity = entity;
741          this.metadata = metadata;
742          if (trace) {
743             log.tracef("Created process task with key=%s, value=%s, metadata=%s", key, entity, metadata);
744          }
745       }
746 
747       @Override
748       public Void call() throws Exception {
749          try {
750             final MarshalledEntry marshalledEntry = marshallerEntryFactory.newMarshalledEntry(key, entity, metadata);
751             if (marshalledEntry != null) {
752                task.processEntry(marshalledEntry, taskContext);
753             }
754             return null;
755          } catch (Exception e) {
756             log.errorExecutingParallelStoreTask(e);
757             throw e;
758          }
759       }

762    private class LoadingProcessTask implements Callable<Void> {
763       private final CacheLoaderTask task;
764       private final TaskContext taskContext;
765       private final Object key;
766       private final boolean fetchValue;
767       private final boolean fetchMetadata;
768 
769       private LoadingProcessTask(CacheLoaderTask task, TaskContext taskContext, Object key, boolean fetchValue, boolean fetchMetadata) {
770          this.task = task;
771          this.taskContext = taskContext;
772          this.key = key;
773          this.fetchValue = fetchValue;
774          this.fetchMetadata = fetchMetadata;
775          if (trace) {
776             log.tracef("Created process task with key=%s, fetchMetadata=%s", key, fetchMetadata);
777          }
778       }
779 
780       @Override
781       public Void call() throws Exception {
782          boolean loaded = false;
783          Object entity;
784          InternalMetadata metadata;
785 
786          // The loading of entries and metadata is offloaded to another thread.
787          // We need second entity manager anyway because with MySQL we can't do streaming
788          // in parallel with other queries using single connection
789          EntityManager emExec = emf.createEntityManager();
790          try {
791             EntityTransaction txExec = emExec.getTransaction();
792             txExec.begin();
793             try {
794                do {
795                   try {
796                      metadata = fetchMetadata ? getMetadata(emExec, key) : null;
797                      if (trace) {
798                         log.tracef("Fetched metadata (fetching? %s) %s", fetchMetadata, metadata);
799                      }
800                      if (metadata != null && metadata.isExpired(timeService.wallClockTime())) {
801                         return null;
802                      }
803                      if (fetchValue) {
804                         entity = findEntity(emExec, key);
805                         if (trace) {
806                            log.tracef("Fetched value %s", entity);
807                         }
808                      } else {
809                         entity = null;
810                      }
811                   } finally {
812                      try {
813                         txExec.commit();
814                         loaded = true;
815                      } catch (Exception e) {
816                         log.trace("Failed to load once", e);
817                      }
818                   }
819                } while (!loaded);
820             } finally {
821                if (txExec != null && txExec.isActive()) {
822                   txExec.rollback();
823                }
824 
825             }
826          } finally {
827             if (emExec != null) {
828                emExec.close();
829             }
830          }
831          try {
832             final MarshalledEntry marshalledEntry = marshallerEntryFactory.newMarshalledEntry(key, entity, metadata);
833             if (marshalledEntry != null) {
834                task.processEntry(marshalledEntry, taskContext);
835             }
836             return null;
837          } catch (Exception e) {
838             log.errorExecutingParallelStoreTask(e);
839             throw e;
840          }
841       }
842    }
