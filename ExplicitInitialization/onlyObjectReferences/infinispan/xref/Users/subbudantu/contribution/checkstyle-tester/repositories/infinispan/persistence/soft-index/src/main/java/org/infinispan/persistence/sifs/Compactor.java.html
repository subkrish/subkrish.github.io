17  /**
18   * Component keeping the data about log file usage - as soon as entries from some file are overwritten so that the file
19   * becomes cluttered with old records, the valid records are moved to another file and the old ones are dropped.
20   * Expired records are moved as tombstones without values (records of entry removal).
21   *
22   * @author Radim Vansa &lt;rvansa@redhat.com&gt;
23   */
24  class Compactor extends Thread {
25     private static final Log log = LogFactory.getLog(Compactor.class);
26     private static final boolean trace = log.isTraceEnabled();
27  
28     private final ConcurrentMap<Integer, Stats> fileStats = new ConcurrentHashMap<Integer, Stats>();
29     private final BlockingQueue<Integer> scheduledCompaction = new LinkedBlockingQueue<Integer>();
30     private final BlockingQueue<IndexRequest> indexQueue;
31     private final FileProvider fileProvider;
32     private final TemporaryTable temporaryTable;
33     private final Marshaller marshaller;
34     private final TimeService timeService;
35     private final int maxFileSize;
36     private final double compactionThreshold;
37  
38     private Index index;
39     // as processing single scheduled compaction takes a lot of time, we don't use the queue to signalize
40     private volatile boolean clearSignal = false;
41     private volatile boolean terminateSignal = false;
42     private volatile CountDownLatch compactorResume;
43     private volatile CountDownLatch compactorStop;
44  
45  
46     public Compactor(FileProvider fileProvider,
47                      TemporaryTable temporaryTable,
48                      BlockingQueue<IndexRequest> indexQueue,
49                      Marshaller marshaller, TimeService timeService, int maxFileSize, double compactionThreshold) {
50        super("BCS-Compactor");
51        this.fileProvider = fileProvider;
52        this.temporaryTable = temporaryTable;
53        this.indexQueue = indexQueue;
54        this.marshaller = marshaller;
55        this.timeService = timeService;
56        this.maxFileSize = maxFileSize;
57        this.compactionThreshold = compactionThreshold;
58        this.start();
59     }
60  
61     public void setIndex(Index index) {
62        this.index = index;
63     }
64  
65     public void releaseStats(int file) {
66        fileStats.remove(file);
67     }
68  
69     public void free(int file, int size) {
70        // entries expired from compacted file are reported with file = -1
71        if (file < 0) return;
72        recordFreeSpace(getStats(file), file, size);
73     }
74  
75     public void completeFile(int file) {
76        Stats stats = getStats(file);
77        stats.setCompleted();
78        if (stats.readyToBeScheduled(compactionThreshold, stats.getFree())) {
79           schedule(file, stats);
80        }
81     }
82  
83     private Stats getStats(int file) {
84        Stats stats = fileStats.get(file);
85        if (stats == null) {
86           int fileSize = (int) fileProvider.getFileSize(file);
87           stats = new Stats(fileSize, 0);
88           Stats other = fileStats.putIfAbsent(file, stats);
89           if (other != null) {
90              if (fileSize > other.getTotal()) {
91                 other.setTotal(fileSize);
92              }
93              return other;
94           }
95        }
96        if (stats.getTotal() < 0) {
97           int fileSize = (int) fileProvider.getFileSize(file);
98           if (fileSize >= 0) {
99              stats.setTotal(fileSize);
100          }
101       }
102       return stats;
103    }
104 
105    private void recordFreeSpace(Stats stats, int file, int size) {
106       if (stats.addFree(size, compactionThreshold)) {
107          schedule(file, stats);
108       }
109    }
110 
111    private void schedule(int file, Stats stats) {
112       try {
113          synchronized (stats) {
114             if (!stats.isScheduled()) {
115                log.debug(String.format("Scheduling file %d for compaction: %d/%d free", file, stats.free.get(), stats.total));
116                stats.setScheduled();
117                scheduledCompaction.put(file);
118             }
119          }
120       } catch (InterruptedException e) {
121          throw new RuntimeException(e);
122       }
123    }
124 
125    @Override
126    public void run() {
127       try {
128          FileProvider.Log logFile = null;
129          int currentOffset = 0;
130          for(;;) {
131             Integer scheduledFile = null;
132             try {
133                scheduledFile = scheduledCompaction.poll(1, TimeUnit.MINUTES);
134             } catch (InterruptedException e) {
135             }
136             if (terminateSignal) {
137                if (logFile != null) {
138                   logFile.close();
139                   completeFile(logFile.fileId);
140                }
141                break;
142             }
143             if (clearSignal) {
144                pauseCompactor(logFile);
145                logFile = null;
146                continue;
147             }
148             if (scheduledFile == null) {
149                if (logFile != null) {
150                   logFile.close();
151                   completeFile(logFile.fileId);
152                   logFile = null;
153                }
154                continue;
155             }
156 
157             if (log.isDebugEnabled()) {
158                log.debugf("Compacting file %d", scheduledFile);
159             }
160             int scheduledOffset = 0;
161             FileProvider.Handle handle = fileProvider.getFile(scheduledFile);
162             if (handle == null) {
163                throw new IllegalStateException("Compactor should not get deleted file for compaction!");
164             }
165             try {
166                while (!clearSignal && !terminateSignal) {
167                   EntryHeader header = EntryRecord.readEntryHeader(handle, scheduledOffset);
168                   if (header == null) {
169                      break;
170                   }
171                   byte[] serializedKey = EntryRecord.readKey(handle, header, scheduledOffset);
172                   Object key = marshaller.objectFromByteBuffer(serializedKey);
173 
174                   boolean drop = true;
175                   EntryPosition entry = temporaryTable.get(key);
176                   if (entry != null) {
177                      synchronized (entry) {
178                         if (entry.file == scheduledFile && entry.offset == scheduledOffset) {
179                            drop = false;
180                         } else if (trace) {
181                            log.tracef("Key for %d:%d was found in temporary table on %d:%d",
182                                  scheduledFile, scheduledOffset, entry.file, entry.offset);
183                         }
184                      }
185                   } else {
186                      EntryPosition position = index.getPosition(key, serializedKey);
187                      if (position != null && position.file == scheduledFile && position.offset == scheduledOffset) {
188                         drop = false;
189                      } else if (trace) {
190                         if (position != null) {
191                            log.tracef("Key for %d:%d was found in index on %d:%d",
192                                  scheduledFile, scheduledOffset, position.file, position.offset);
193                         } else {
194                            log.tracef("Key for %d:%d was not found in index!", scheduledFile, (Object)scheduledOffset);
195                         }
196                      }
197                   }
198                   if (drop) {
199                      if (trace) {
200                         log.tracef("Drop %d:%d (%s)", scheduledFile, (Object)scheduledOffset,
201                               header.valueLength() > 0 ? "record" : "tombstone");
202                      }
203                      scheduledOffset += header.totalLength();
204                      continue;
205                   }
206 
207                   if (logFile == null || currentOffset + header.totalLength() > maxFileSize) {
208                      if (logFile != null) {
209                         logFile.close();
210                         completeFile(logFile.fileId);
211                      }
212                      currentOffset = 0;
213                      logFile = fileProvider.getFileForLog();
214                      if (log.isDebugEnabled()) {
215                         log.debugf("Compacting to %d", (Object)logFile.fileId);
216                      }
217                   }
218 
219                   byte[] serializedValue;
220                   byte[] serializedMetadata;
221                   int entryOffset;
222                   // we have to keep track of expired entries but only as tombstones - do not copy the value nor metadata
223                   if (header.valueLength() > 0 && (header.expiryTime() < 0 || header.expiryTime() > timeService.wallClockTime())) {
224                      serializedMetadata = EntryRecord.readMetadata(handle, header, scheduledOffset);
225                      serializedValue = EntryRecord.readValue(handle, header, scheduledOffset);
226                      entryOffset = currentOffset;
227                   } else {
228                      serializedMetadata = null;
229                      serializedValue = null;
230                      entryOffset = ~currentOffset;
231                   }
232 
233                   EntryRecord.writeEntry(logFile.fileChannel, serializedKey, serializedMetadata, serializedValue, header.seqId(), header.expiryTime());
234                   temporaryTable.setConditionally(key, logFile.fileId, entryOffset, scheduledFile, scheduledOffset);
235                   if (trace) {
236                      log.tracef("Update %d:%d -> %d:%d | %d,%d", scheduledFile, scheduledOffset, logFile.fileId, entryOffset, logFile.fileChannel.position(), logFile.fileChannel.size());
237                   }
238                   // entryFile cannot be used as we have to report the file due to free space statistics
239                   indexQueue.put(new IndexRequest(key, serializedKey,
240                         logFile.fileId, entryOffset,
241                         header.totalLength(), scheduledFile, scheduledOffset));
242 
243                   currentOffset += header.totalLength();
244                   scheduledOffset += header.totalLength();
245                }
246             } finally {
247                handle.close();
248             }
249             if (terminateSignal) {
250                if (logFile != null) {
251                   logFile.close();
252                   completeFile(logFile.fileId);
253                }
254                return;
255             } else if (clearSignal) {
256                pauseCompactor(logFile);
257                logFile = null;
258             } else {
259                // The deletion must be executed only after the index is fully updated.
260                log.debugf("Finished compacting %d, scheduling delete", scheduledFile);
261                indexQueue.put(IndexRequest.deleteFileRequest(scheduledFile));
262             }
263          }
264       } catch (IOException e) {
265          throw new RuntimeException(e);
266       } catch (ClassNotFoundException e) {
267          throw new RuntimeException(e);
268       } catch (InterruptedException e) {
269          throw new RuntimeException(e);
270       }
271    }
272 
273    private void pauseCompactor(FileProvider.Log logFile) throws IOException, InterruptedException {
274       if (logFile != null) {
275          logFile.close();
276          completeFile(logFile.fileId);
277       }
278       compactorStop.countDown();
279       compactorResume.await();
280    }
281 
282    public void clearAndPause() throws InterruptedException {
283       compactorResume = new CountDownLatch(1);
284       compactorStop = new CountDownLatch(1);
285       clearSignal = true;
286       scheduledCompaction.put(-1);
287       compactorStop.await();
288       scheduledCompaction.clear();
289       fileStats.clear();
290    }
291 
292    public void resumeAfterPause() {
293       clearSignal = false;
294       compactorResume.countDown();
295    }
296 
297    public void stopOperations() throws InterruptedException {
298       terminateSignal = true;
299       scheduledCompaction.put(-1);
300       this.join();
301    }

303    private static class Stats {
304       private final AtomicInteger free;
305       private volatile int total;
306       /* File is not 'completed' when we have not loaded that yet completely.
307          Files created by log appender/compactor are completed as soon as it closes them.
308          File cannot be scheduled for compaction until it's completed.
309          */
310       private volatile boolean completed = false;
311       private volatile boolean scheduled = false;
312 
313       private Stats(int total, int free) {
314          this.free = new AtomicInteger(free);
315          this.total = total;
316       }
317 
318       public int getTotal() {
319          return total;
320       }
321 
322       public void setTotal(int total) {
323          this.total = total;
324       }
325 
326       public boolean addFree(int size, double compactionThreshold) {
327          int free = this.free.addAndGet(size);
328          return readyToBeScheduled(compactionThreshold, free);
329       }
330 
331       public int getFree() {
332          return free.get();
333       }
334 
335       public boolean readyToBeScheduled(double compactionThreshold, int free) {
336          int total = this.total;
337          return completed && !scheduled && total >= 0 && free > total * compactionThreshold;
338       }
339 
340       public boolean isScheduled() {
341          return scheduled;
342       }
343 
344       public void setScheduled() {
345          scheduled = true;
346       }
347 
348       public boolean isCompleted() {
349          return completed;
350       }
351 
352       public void setCompleted() {
353          this.completed = true;
354       }
355    }
