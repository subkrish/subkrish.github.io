1   package org.infinispan.query.impl;
2   
3   import org.hibernate.search.spi.SearchIntegrator;
4   import org.infinispan.commons.util.concurrent.jdk8backported.LongAdder;
5   
6   import java.util.Map;
7   import java.util.Set;
8   
9   /**
10   * This MBean accumulates query statistics from the Hibernate Search statistics object. The only difference is this
11   * statistics continue to accumulate and are not reset when the Search Factory is reconfigured. They are still reset if
12   * {@code clear()} is explicitly called though.
13   *
14   * @author anistor@redhat.com
15   * @since 6.1
16   */
17  public class InfinispanQueryStatisticsInfo implements InfinispanQueryStatisticsInfoMBean {
18  
19     private final SearchIntegrator sf;
20     private final LongAdder searchQueryExecutionCount = new LongAdder();
21     private final LongAdder searchQueryTotalTime = new LongAdder();
22     private volatile long searchQueryExecutionMaxTime = 0;
23     private volatile String searchQueryExecutionMaxTimeQueryString = null;
24     private final LongAdder objectLoadingTotalTime = new LongAdder();
25     private volatile long objectLoadingExecutionMaxTime = 0;
26     private final LongAdder objectLoadedCount = new LongAdder();
27  
28     public InfinispanQueryStatisticsInfo(SearchIntegrator sf) {
29        this.sf = sf;
30     }
31  
32     @Override
33     public void clear() {
34        searchQueryExecutionCount.reset();
35        searchQueryTotalTime.reset();
36        searchQueryExecutionMaxTime = 0;
37        searchQueryExecutionMaxTimeQueryString = null;
38        objectLoadingTotalTime.reset();
39        objectLoadingExecutionMaxTime = 0;
40        objectLoadedCount.reset();
41        sf.getStatistics().clear();
42     }
43  
44     @Override
45     public long getSearchQueryExecutionCount() {
46        searchQueryExecutionCount.add(sf.getStatistics().getSearchQueryExecutionCount());
47        return searchQueryExecutionCount.sum();
48     }
49  
50     @Override
51     public long getSearchQueryTotalTime() {
52        searchQueryTotalTime.add(sf.getStatistics().getSearchQueryTotalTime());
53        return searchQueryTotalTime.sum();
54     }
55  
56     @Override
57     public long getSearchQueryExecutionMaxTime() {
58        long temp = sf.getStatistics().getSearchQueryExecutionMaxTime();
59        if (searchQueryExecutionMaxTime < temp) {
60           searchQueryExecutionMaxTime = temp;
61        }
62        return searchQueryExecutionMaxTime;
63     }
64  
65     @Override
66     public long getSearchQueryExecutionAvgTime() {
67        long count = getSearchQueryExecutionCount();
68        if (count == 0) {
69           return 0;
70        }
71        return getSearchQueryTotalTime() / count;
72     }
73  
74     @Override
75     public String getSearchQueryExecutionMaxTimeQueryString() {
76        String temp = sf.getStatistics().getSearchQueryExecutionMaxTimeQueryString();
77        if (temp != null) {
78           searchQueryExecutionMaxTimeQueryString = temp;
79        }
80        return searchQueryExecutionMaxTimeQueryString;
81     }
82  
83     @Override
84     public long getObjectLoadingTotalTime() {
85        objectLoadingTotalTime.add(sf.getStatistics().getObjectLoadingTotalTime());
86        return objectLoadingTotalTime.sum();
87     }
88  
89     @Override
90     public long getObjectLoadingExecutionMaxTime() {
91        long temp = sf.getStatistics().getObjectLoadingExecutionMaxTime();
92        if (objectLoadingExecutionMaxTime < temp) {
93           objectLoadingExecutionMaxTime = temp;
94        }
95        return objectLoadingExecutionMaxTime;
96     }
97  
98     @Override
99     public long getObjectLoadingExecutionAvgTime() {
100       long count = getObjectsLoadedCount();
101       if (count == 0) {
102          return 0;
103       }
104       return getObjectLoadingTotalTime() / count;
105    }
106 
107    @Override
108    public long getObjectsLoadedCount() {
109       objectLoadedCount.add(sf.getStatistics().getObjectsLoadedCount());
110       return objectLoadedCount.sum();
111    }
112 
113    @Override
114    public boolean isStatisticsEnabled() {
115       return sf.getStatistics().isStatisticsEnabled();
116    }
117 
118    @Override
119    public void setStatisticsEnabled(boolean isStatisticsEnabled) {
120       sf.getStatistics().setStatisticsEnabled(isStatisticsEnabled);
121    }
122 
123    @Override
124    public String getSearchVersion() {
125       return sf.getStatistics().getSearchVersion();
126    }
127 
128    @Override
129    public Set<String> getIndexedClassNames() {
130       return sf.getStatistics().getIndexedClassNames();
131    }
132 
133    @Override
134    public int getNumberOfIndexedEntities(String entity) {
135       return sf.getStatistics().getNumberOfIndexedEntities(entity);
136    }
137 
138    @Override
139    public Map<String, Integer> indexedEntitiesCount() {
140       return sf.getStatistics().indexedEntitiesCount();
141    }
142 }
