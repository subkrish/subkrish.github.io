9   public class MBeanOperation extends JmxComponent {

11     public String signature = "";
12     int n = 0;
13  
14     @Override
15     public String toString() {
16        return "Operation(name = " + name + ", desc = " + desc + ", sig = " + signature + ", retType = " + returnType + ")";
17     }
18  
19     void addParam(String paramType) {
20        if (signature.length() != 0) signature += ", ";
21        signature += paramType + " p" + n++;
22     }
23  }
