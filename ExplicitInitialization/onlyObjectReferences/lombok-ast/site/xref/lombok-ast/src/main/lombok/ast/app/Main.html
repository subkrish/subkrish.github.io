137 	public static void main(String[] rawArgs) throws Exception {
138 		CmdArgs args;
139 		CmdReader<CmdArgs> reader = CmdReader.of(CmdArgs.class);
140 		
141 		try {
142 			args = reader.make(rawArgs);
143 		} catch (InvalidCommandLineException e) {
144 			System.err.println(e.getMessage());
145 			System.err.println(reader.generateCommandLineHelp("java -jar lombok.ast.jar"));
146 			System.exit(1);
147 			return;
148 		}
149 		
150 		if (args.help) {
151 			System.out.println("lombok.ast java AST tool " + Version.getVersion());
152 			System.out.println(reader.generateCommandLineHelp("java -jar lombok.ast.jar"));
153 			System.exit(0);
154 			return;
155 		}
156 		
157 		if (args.version) {
158 			System.out.println(Version.getVersion());
159 			System.exit(0);
160 			return;
161 		}
162 		
163 		try {
164 			Charset charset = args.encoding == null ? Charset.defaultCharset() : Charset.forName(args.encoding);
165 			Main main = new Main(charset, args.verbose, args.normalize, !args.noPositions, args.saveIntermediate);
166 			main.compile(args.program);
167 			if (!args.print) {
168 				File targetDir = new File(args.target);
169 				if (!targetDir.exists()) targetDir.mkdirs();
170 				if (!targetDir.isDirectory()) {
171 					System.err.printf("%s is not a directory or cannot be created\n", targetDir.getCanonicalPath());
172 					System.exit(1);
173 					return;
174 				}
175 				main.setOutputDir(targetDir);
176 			}
177 			
178 			for (String input : args.input) {
179 				main.addToQueue(input);
180 			}
181 			
182 			main.go();
183 		} catch (IllegalArgumentException e) {
184 			System.err.println(e.getMessage());
185 			System.exit(1);
186 			return;
187 		}
188 	}

190 	private void go() throws IOException {
191 		for (Plan p : files) {
192 			process(p.getFile(), outDir, p.getRelativeName());
193 		}
194 		if (errors > 0) {
195 			System.err.printf("%d errors\n", errors);
196 		}
197 		System.exit(errors > 0 ? 2 : 0);
198 	}

208 	private void addToQueue0(File f, String pathSoFar) throws IOException {
209 		pathSoFar += (pathSoFar.isEmpty() ? "" : "/") + f.getName();
210 		if (f.isFile()) {
211 			if (f.getName().endsWith(".java")) {
212 				files.add(new Plan(f, pathSoFar));
213 			}
214 		} else if (f.isDirectory()) {
215 			for (File inner : f.listFiles()) {
216 				addToQueue0(inner, pathSoFar);
217 			}
218 		} else {
219 			throw new IllegalArgumentException("Unknown file: " + f.getCanonicalPath());
220 		}
221 	}

229 	private void process(File in, File outDir, String relativeName) throws IOException {
230 		File out = outDir == null ? null : new File(outDir, relativeName);
231 		
232 		if (verbose && !saveIntermediate) {
233 			System.out.printf("Processing: %s to %s\n", in.getCanonicalPath(), out == null ? "sysout" : out.getCanonicalPath());
234 		}
235 		
236 		Source source = new Source(Files.toString(in, charset), in.getCanonicalPath());
237 		Object transfer = null;
238 		String chain = "/";
239 		
240 		try {
241 			for (Operation<Object, Object> programElem : program) {
242 				transfer = programElem.process(source, transfer);
243 				if (saveIntermediate) {
244 					if (!"/".equals(chain)) {
245 						chain += "-";
246 					}
247 					chain += getDestinationType(programElem);
248 					File intermediate = new File(outDir.getCanonicalPath() + chain + "/" + relativeName);
249 					intermediate.getParentFile().mkdirs();
250 					
251 					if (verbose) {
252 						System.out.printf("Processing: %s to %s\n", in.getCanonicalPath(), intermediate.getCanonicalPath());
253 					}
254 					
255 					if (TO_JAVAC.contains(programElem)) {
256 						Files.write(javacToText.process(source, (JCCompilationUnit) transfer).toString(), intermediate, charset);
257 					}
258 					else if (TO_ECJ.contains(programElem)) {
259 						Files.write(ecjToText.process(source, (CompilationUnitDeclaration) transfer).toString(), intermediate, charset);
260 					}
261 					else if (TO_LOMBOK.contains(programElem)) {
262 						Files.write(lombokToText.process(source, (Node) transfer).toString(), intermediate, charset);
263 					}
264 				}
265 			}
266 			
267 			if (out == null) {
268 				System.out.println(transfer);
269 			} else if (!saveIntermediate) {
270 				out.getParentFile().mkdirs();
271 				Files.write(transfer.toString(), out, charset);
272 			}
273 		} catch (ConversionProblem cp) {
274 			System.err.printf("Can't convert: %s due to %s\n", in.getCanonicalPath(), cp.getMessage());
275 			errors++;
276 		} catch (RuntimeException e) {
277 			System.err.printf("Error during convert: %s\n%s\n", in.getCanonicalPath(), printEx(e));
278 			errors++;
279 		}
280 	}

282 	private String getDestinationType(Operation<Object, Object> operation) {
283 		if (TO_LOMBOK.contains(operation)) return "lombok";
284 		else if (TO_ECJ.contains(operation)) return "ecj";
285 		else if (TO_JAVAC.contains(operation)) return "javac";
286 		else if (TO_TEXT.contains(operation)) return "text";
287 		else return null;
288 	}

303 	private static final class ChainElement {
304 		private final String type, subtype;
305 		
306 		@Override public String toString() {
307 			return subtype.length() == 0 ? type : String.format("%s:%s", type, subtype);
308 		}
309 		
310 		public boolean hasSubtype() {
311 			return subtype.length() > 0;
312 		}
313 	}

315 	private List<ChainElement> toChainElements(String program) {
316 		val out = new ArrayList<ChainElement>();
317 		for (String part : program.split("\\s*,\\s*")) {
318 			int idx = part.indexOf(':');
319 			if (idx == -1) out.add(new ChainElement(part.trim(), ""));
320 			else out.add(new ChainElement(part.substring(0, idx).trim(), part.substring(idx+1).trim()));
321 		}
322 		return out;
323 	}

325 	@SuppressWarnings("unchecked")
326 	private void addNormalization(List<Operation<Object, Object>> list, ChainElement element) {
327 		if (!element.hasSubtype()) return;
328 		Operation<?, ?> operation = NORMALIZATION.get(element.toString());
329 		if (operation == null) {
330 			List<String> normalizations = Lists.newArrayList();
331 			for (String n : NORMALIZATION.keySet()) if (n.startsWith(element.getType() + ":")) normalizations.add(n);
332 			throw new IllegalArgumentException(String.format(
333 					"Illegal normalization operation: %s. Valid normalizations: %s", element, Joiner.on(",").join(normalizations)));
334 		}
335 		list.add((Operation<Object, Object>) operation);
336 	}

338 	@SuppressWarnings("unchecked")
339 	private List<Operation<Object, Object>> compile0(String program) {
340 		List<ChainElement> parts = toChainElements(program);
341 		List<Operation<Object, Object>> out = Lists.newArrayList();
342 		if (parts.isEmpty()) throw new IllegalArgumentException("No operations");
343 		Operation<?, ?> initialOp = CONVERSIONS.get("_," + parts.get(0).getType());
344 		if (initialOp == null) {
345 			List<String> initialOps = Lists.newArrayList();
346 			for (String key : CONVERSIONS.keySet()) {
347 				if (key.startsWith("_,")) initialOps.add(key.substring(2));
348 			}
349 			throw new IllegalArgumentException(String.format(
350 					"Illegal initial operation: %s\nLegal initial operations: %s",
351 					parts.get(0), Joiner.on(",").join(initialOps)));
352 		}
353 		
354 		out.add((Operation<Object, Object>) initialOp);
355 		addNormalization(out, parts.get(0));
356 		for (int i = 0; i < parts.size() - 1; i++) {
357 			String convKey = String.format("%s,%s", parts.get(i).getType(), parts.get(i + 1).getType());
358 			Operation<?, ?> convOp = CONVERSIONS.get(convKey);
359 			if (convOp == null) {
360 				List<String> convOps = Lists.newArrayList();
361 				for (String key : CONVERSIONS.keySet()) {
362 					if (key.startsWith(parts.get(i).getType() + ",")) convOps.add(key.substring(parts.get(i).getType().length() + 1));
363 				}
364 				throw new IllegalArgumentException(String.format(
365 						"Illegal conversion operation: %s\nLegal conversion operations from %s: %s",
366 						convKey, parts.get(i), Joiner.on(",").join(convOps)));
367 			}
368 			out.add((Operation<Object, Object>) convOp);
369 			addNormalization(out, parts.get(i + 1));
370 		}
371 		
372 		String lastPart = parts.get(parts.size() - 1).getType();
373 		if (!LEGAL_FINAL.contains(lastPart) && !saveIntermediate) {
374 			throw new IllegalArgumentException(String.format(
375 					"Illegal final operation: %s\nLegal final operations: %s",
376 					lastPart, Joiner.on(",").join(LEGAL_FINAL)));
377 		}
378 		
379 		return out;
380 	}

411 	private final Operation<Void, Node> parseWithLombok = new Operation<Void, Node>() {
412 		@Override public Node process(Source in, Void irrelevant) throws ConversionProblem {
413 			List<Node> nodes = in.getNodes();
414 			List<ParseProblem> problems = in.getProblems();
415 			if (problems.size() > 0) throw new ConversionProblem(String.format("Can't read file %s due to parse error: %s", in.getName(), problems.get(0)));
416 			if (nodes.size() == 1) return nodes.get(0);
417 			if (nodes.size() == 0) throw new ConversionProblem("No nodes parsed by lombok.ast");
418 			throw new ConversionProblem("More than 1 node parsed by lombok.ast");
419 		}
420 	};

422 	private final Operation<Void, ASTNode> parseWithEcj = new Operation<Void, ASTNode>() {
423 		@Override public ASTNode process(Source in, Void irrelevant) throws ConversionProblem {
424 			CompilerOptions compilerOptions = ecjCompilerOptions();
425 			Parser parser = new Parser(new ProblemReporter(
426 					DefaultErrorHandlingPolicies.proceedWithAllProblems(),
427 					compilerOptions,
428 					new DefaultProblemFactory()
429 				), compilerOptions.parseLiteralExpressionsAsConstants);
430 			parser.javadocParser.checkDocComment = true;
431 			CompilationUnit sourceUnit = new CompilationUnit(in.getRawInput().toCharArray(), in.getName(), charset.name());
432 			CompilationResult compilationResult = new CompilationResult(sourceUnit, 0, 0, 0);
433 			CompilationUnitDeclaration cud = parser.parse(sourceUnit, compilationResult);
434 			
435 			if (cud.hasErrors()) {
436 				throw new ConversionProblem(String.format("Can't read file %s due to parse error: %s", in.getName(), compilationResult.getErrors()[0]));
437 			}
438 			
439 			return cud;
440 		}
441 	};

542 	private final Operation<CompilationUnitDeclaration, String> ecjToText = new Operation<CompilationUnitDeclaration, String>() {
543 		@Override public String process(Source source, CompilationUnitDeclaration in) throws ConversionProblem {
544 			if (normalize) {
545 				return positions ? EcjTreeOperations.convertToString(in) : EcjTreeOperations.convertToStringNoPositions(in);
546 			} else {
547 				EcjTreePrinter printer = positions ? EcjTreePrinter.printerWithPositions() : EcjTreePrinter.printerWithoutPositions();
548 				printer.visit(in);
549 				return printer.getContent();
