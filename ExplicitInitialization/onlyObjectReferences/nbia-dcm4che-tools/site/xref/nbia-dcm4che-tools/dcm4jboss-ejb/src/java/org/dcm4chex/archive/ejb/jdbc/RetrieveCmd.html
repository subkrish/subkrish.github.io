90                          Entry<String, SeriesAttrsCacheEntry> eldest) {

92                          if (log.isDebugEnabled()) {

97                          return true;
98                      }
99                      return false;
100                 }

139     public static void setSeriesAttrsCacheMaxSize(int maxSize) {
140         int toRemove = seriesAttrsCache.size() - maxSize;
141         if (toRemove > 0) {
142             for (Iterator iterator = seriesAttrsCache.entrySet().iterator();
143                     toRemove > 0; toRemove--) {
144                 iterator.next();
145                 iterator.remove();
146             }
147         }
148         seriesAttrsCacheMaxSize = maxSize;

167         if ("IMAGE".equals(qrLevel))
168             return createInstanceRetrieve(keys);
169         if ("SERIES".equals(qrLevel))
170             return createSeriesRetrieve(keys);
171         if ("STUDY".equals(qrLevel))
172             return createStudyRetrieve(keys);
173         if ("PATIENT".equals(qrLevel))
174             return createPatientRetrieve(keys);
175         throw new IllegalArgumentException("QueryRetrieveLevel=" + qrLevel);

252     public FileInfo[][] getFileInfos() throws SQLException {
253         Map result = map();
254         try {
255             PreparedStatement pstmt = ((PreparedStatement) stmt);
256             if (fetchSize > 0)
257                 pstmt.setFetchSize(fetchSize);
258             int start = 0;
259             String[] fixParams = sqlCmd.getFixParams();
260             for (int i = 0; i < fixParams.length; i++) {
261                 pstmt.setString(i + 1, fixParams[i]);
262             }
263             int firstListIdx = fixParams.length;
264             String[] params = sqlCmd.getParams();
265             if (params != null) {
266                 int len = sqlCmd.getNumberOfParams();
267                 while (start < params.length) {
268                     if (start + len > params.length) { // we need a new
269                                                         // statement for the
270                                                         // remaining parameter
271                                                         // values
272                         len = params.length - start;
273                         sqlCmd.updateUIDMatch(len);
274                         pstmt = con.prepareStatement(sqlCmd.getSql(),
275                                 resultSetType, resultSetConcurrency);
276                         if (firstListIdx > 0) { // we need to set the fix params
277                                                 // for the new statement!
278                             for (int i = 0; i < fixParams.length; i++) {
279                                 pstmt.setString(i + 1, fixParams[i]);
280                             }
281                         }
282                     }
283                     for (int i = 1; i <= len; i++) {// set the values for the
284                                                     // uid list match
285                         pstmt.setString(firstListIdx + i, params[start++]);
286                     }
287                     rs = pstmt.executeQuery();
288                     addFileInfos(result);
289                 }
290             } else {
291                 rs = pstmt.executeQuery();
292                 addFileInfos(result);
293             }
294             log.trace("getFileInfos finished!");
295         } finally {
296             log.trace("getFileInfos: Close Connection.");
297             close();
298             log.trace("getFileInfos: Connection closed.");
299         }
300         return toArray(result);
301     }

303     private void addFileInfos(Map result) throws SQLException {
304         while (next()) {
305             if (lazyFetchSeriesAttrs) {
306                 addFileInfoWithLazyFetchSeriesAttrs(result);
307             } else {
308                 addFileInfoWithEagerFetchSeriesAttrs(result);
309             }
310         }
311     }

313     private void addFileInfoWithLazyFetchSeriesAttrs(Map result)
314             throws SQLException {
315         FileInfo info = new FileInfo();
316         info.instAttrs = rs.getBytes(1);
317         info.seriesIUID = rs.getString(2);
318         long instPk = rs.getLong(3);
319         info.sopIUID = rs.getString(4);
320         info.sopCUID = rs.getString(5);
321         info.extRetrieveAET = rs.getString(6);
322         info.pk = rs.getLong(7);
323         info.fileRetrieveAET = rs.getString(8);
324         info.availability = rs.getInt(9);
325         info.basedir = rs.getString(10);
326         info.fsGroupID = rs.getString(11);
327         info.fileID = rs.getString(12);
328         info.tsUID = rs.getString(13);
329         info.md5 = rs.getString(14);
330         info.size = rs.getInt(15);
331         info.status = rs.getInt(16);
332         FileInfo seriesFileInfo = getCachedSeriesAttrs(info.seriesIUID);
333         if (seriesFileInfo == null) {
334             if (log.isDebugEnabled()) {
335                 log.debug("Lazy fetch Series attributes for Series "
336                         + info.seriesIUID);
337             }
338             QuerySeriesAttrsForRetrieveCmd seriesQuery =
339                 new QuerySeriesAttrsForRetrieveCmd(
340                         QueryCmd.transactionIsolationLevel,
341                         QueryCmd.seriesBlobAccessType,
342                         info.seriesIUID);
343             try {
344                 seriesQuery.execute();
345                 seriesQuery.next();
346                 seriesFileInfo = seriesQuery.getFileInfo();
347             } finally {
348                 seriesQuery.close();
349             }
350             seriesAttrsCache.put(info.seriesIUID,
351                     new SeriesAttrsCacheEntry(seriesFileInfo));
352         }
353         info.patAttrs = seriesFileInfo.patAttrs;
354         info.studyAttrs = seriesFileInfo.studyAttrs;
355         info.seriesAttrs = seriesFileInfo.seriesAttrs;
356         info.patID = seriesFileInfo.patID;
357         info.patName = seriesFileInfo.patName;
358         info.studyIUID = seriesFileInfo.studyIUID;
359         addFileInfo(result, instPk, info.sopIUID, info);
360     }

362     private static FileInfo getCachedSeriesAttrs(String seriesIUID) {
363         SeriesAttrsCacheEntry cacheEntry = seriesAttrsCache.get(seriesIUID);
364         if (cacheEntry == null) {
365             return null;
366         }
367         if (cacheEntry.timestamp + seriesAttrsCacheCurrencyTimeLimit 
368                 < System.currentTimeMillis()) {
369             if (log.isDebugEnabled()) {
370                 log.debug("Remove stale Series attributes for Series "
371                         + seriesIUID + " from cache");
372             }
373             seriesAttrsCache.remove(seriesIUID);
374             return null;
375         }
376         return cacheEntry.fileInfo;
377     }

379     public static void removeCachedSeriesAttrs(String seriesIUID) {
380         SeriesAttrsCacheEntry cacheEntry = seriesAttrsCache.get(seriesIUID);
381         if (cacheEntry == null) {
382             return;
383         }
384         if ( log.isDebugEnabled()) {
385         	log.debug("Remove Series attributes for Series "
386                     + seriesIUID + " from cache");
387         }
388         seriesAttrsCache.remove(seriesIUID);
389     }

398     private void addFileInfoWithEagerFetchSeriesAttrs(Map result)
399             throws SQLException {
400         FileInfo info = new FileInfo();
401         info.instAttrs = rs.getBytes(1);
402         info.seriesIUID = rs.getString(2);
403         if (cacheSeriesAttrs) {
404             FileInfo seriesFileInfo = getCachedSeriesAttrs(info.seriesIUID);
405             if (seriesFileInfo == null) {
406                 if (log.isDebugEnabled()) {
407                     log.debug("Cache Series attributes for Series "
408                             + info.seriesIUID);
409                 }
410                 info.patAttrs = rs.getBytes(3);
411                 info.studyAttrs = rs.getBytes(4);
412                 info.seriesAttrs = rs.getBytes(5);
413                 info.patID = rs.getString(6);
414                 info.patName = rs.getString(7);
415                 info.studyIUID = rs.getString(8);
416                 seriesAttrsCache.put(info.seriesIUID,
417                         new SeriesAttrsCacheEntry(info));
418             } else {
419                 if (log.isDebugEnabled()) {
420                     log.debug("Use cached Series attributes for Series "
421                             + info.seriesIUID);
422                 }
423                 info.patAttrs = seriesFileInfo.patAttrs;
424                 info.studyAttrs = seriesFileInfo.studyAttrs;
425                 info.seriesAttrs = seriesFileInfo.seriesAttrs;
426                 info.patID = seriesFileInfo.patID;
427                 info.patName = seriesFileInfo.patName;
428                 info.studyIUID = seriesFileInfo.studyIUID;
429             }
430         } else {
431             info.patAttrs = rs.getBytes(3);
432             info.studyAttrs = rs.getBytes(4);
433             info.seriesAttrs = rs.getBytes(5);
434             info.patID = rs.getString(6);
435             info.patName = rs.getString(7);
436             info.studyIUID = rs.getString(8);
437         }
438         long instPk = rs.getLong(9);
439         info.sopIUID = rs.getString(10);
440         info.sopCUID = rs.getString(11);
441         info.extRetrieveAET = rs.getString(12);
442         info.pk = rs.getLong(13);
443         info.fileRetrieveAET = rs.getString(14);
444         info.availability = rs.getInt(15);
445         info.basedir = rs.getString(16);
446         info.fsGroupID = rs.getString(17);
447         info.fileID = rs.getString(18);
448         info.tsUID = rs.getString(19);
449         info.md5 = rs.getString(20);
450         info.size = rs.getInt(21);
451         info.status = rs.getInt(22);
452         addFileInfo(result, instPk, info.sopIUID, info);
453     }

455     private void addFileInfo(Map result, long instPk, String sopIUID,
456             FileInfo info) {
457         Object key = selectKey(instPk, sopIUID);
458         ArrayList list = (ArrayList) result.get(key);
459         if (list == null) {
460             result.put(key, list = new ArrayList());
461         }
462         list.add(info);
463     }

473     protected FileInfo[][] toArray(Map result) {
474         FileInfo[][] array = new FileInfo[result.size()][];
475         ArrayList list;
476         Iterator it = result.values().iterator();
477         for (int i = 0; i < array.length; i++) {
478             list = (ArrayList) it.next();
479             array[i] = (FileInfo[]) list.toArray(new FileInfo[list.size()]);
480             Arrays.sort(array[i]);
481         }
482         return array;
483     }

587         public boolean updateUIDMatch(int len) {
588             if (uidMatch == null)
589                 return false;
590             uidMatch.setLiteral(getUIDMatchLiteral(len));
591             return true;
592         }
593 
594         protected void addUidMatch(String column, String[] uid) {
595             if (uid.length <= maxElementsInUIDMatch) {
596                 sqlBuilder.addLiteralMatch(null, column, SqlBuilder.TYPE1,
597                         getUIDMatchLiteral(uid.length));
598                 for (int i = 0; i < uid.length; i++) {
599                     fixValues.add(uid[i]);
600                 }
601             } else {
602                 if (params != null)
603                     throw new IllegalArgumentException(
604                             "Only one UID list > maxElementsInUIDMatch ("
605                                     + maxElementsInUIDMatch
606                                     + ") is allowed in RetrieveCmd!");
607                 params = uid;
608                 numberOfParams = uid.length < maxElementsInUIDMatch ? uid.length
609                         : maxElementsInUIDMatch;
610                 uidMatch = (Match.AppendLiteral) sqlBuilder.addLiteralMatch(
611                         null, column, SqlBuilder.TYPE1,
612                         getUIDMatchLiteral(numberOfParams));
613             }
614         }
615 
616         /**
617          * @param uid
618          * @return
619          */
620         private String getUIDMatchLiteral(int len) {
621             if (len == 1)
622                 return "=?";
623             StringBuffer sb = new StringBuffer();
624             sb.append(" IN (?");
625             for (int i = 1; i < len; i++) {
626                 sb.append(", ?");
627             }
628             sb.append(")");
629             return sb.toString();
630         }
