65  public class FileCopyOrder extends BaseJmsOrder {

67      private static final long serialVersionUID = 3258409538737550129L;
68  
69      protected List<FileInfo> fileInfos = null;
70  
71      protected final String dstFsPath;
72  
73      protected final String retrieveAETs;
74  
75      protected Dataset ian = null;
76      
77      protected int fetchSize;
78      
79      private static Logger log = Logger.getLogger(FileCopyOrder.class);
80  
81      public FileCopyOrder(Dataset ian, String dstFsPath, String retrieveAETs, int fetchSize) {
82          this.ian = ian;
83          this.dstFsPath = dstFsPath;
84          this.retrieveAETs = retrieveAETs;
85          this.fetchSize = fetchSize;
86      }
87  
88      public List<FileInfo> getFileInfos() throws Exception {
89          if (fileInfos == null)
90              convertFromIAN();
91  
92          return fileInfos;
93      }
94  
95      /**
96       * Convert to fileInfos
97       * 
98       * @throws Exception
99       */
100     protected void convertFromIAN() throws Exception {
101         Dataset refSeriesSeq = ian.getItem(Tags.RefSeriesSeq);
102         DcmElement refSOPSeq = refSeriesSeq.get(Tags.RefSOPSeq);
103         RetrieveCmd cmd = RetrieveCmd.create(refSOPSeq);
104         cmd.setFetchSize(fetchSize);
105         FileInfo[][] aa = cmd.getFileInfos();
106         fileInfos = new ArrayList<FileInfo>(aa.length);
107         HashMap<String, FileInfo> fiCopy = new HashMap<String, FileInfo>();
108         HashMap<String, FileInfo> fi2Copy = new HashMap<String, FileInfo>();
109         for (FileInfo[] a : aa) {
110             fiCopy.clear();
111             fi2Copy.clear();
112             for (FileInfo fi : a) {
113                 if (fi.basedir.equals(dstFsPath)) {
114                     log.info("Instance "+fi.sopIUID+" md5:"+fi.md5+" has already a copy on dstFsPath:"+dstFsPath);
115                     if (fi.status != FileStatus.MD5_CHECK_FAILED && fi.status != FileStatus.QUERY_HSM_FAILED) {
116                         fiCopy.put(fi.md5, fi);
117                     } else {
118                         log.info("Copy has file status "+FileStatus.toString(fi.status)+"! Retry copy!");
119                     }
120                 } else if (fi.availability == Availability.ONLINE && isLocalRetrieveAET(fi.fileRetrieveAET) &&
121                         !fi.basedir.startsWith("tar:")) {
122                     fi2Copy.put(fi.md5, fi);
123                 }
124             }
125             for (Map.Entry<String, FileInfo> e : fi2Copy.entrySet()) {
126                 if (!fiCopy.containsKey(e.getKey())) {
127                     fileInfos.add(e.getValue());
128                 }
129             }
130         }
131     }
132 
133     private boolean isLocalRetrieveAET(String aet) {
134         int pos = retrieveAETs.indexOf(aet);
135         int end = pos + aet.length();
136         return pos != -1 && (pos == 0 || retrieveAETs.charAt(pos-1) == '\\')
137             && (end == retrieveAETs.length() || retrieveAETs.charAt(end) == '\\');
138     }
139 
140     public final String getDestinationFileSystemPath() {
141         return dstFsPath;
142     }
143 
144     public String toString() {
145         StringBuffer sb = new StringBuffer();
146         sb.append(super.toString());
147         sb.append("\tRetrieveAETs: ").append(retrieveAETs).append("\n");
148         sb.append("\tDestination: ").append(dstFsPath).append("\n");
149         if (fileInfos != null) {
150             sb.append("\n\tSource files: \n");
151             for (Iterator iter = fileInfos.iterator(); iter.hasNext();) {
152                 FileInfo fi = (FileInfo) iter.next();
153                 sb.append("\t\t").append(((FileInfo) fi).basedir).append(",")
154                         .append(((FileInfo) fi).fileID).append("\n");
155             }
156         } else if (ian != null) {
157             if (log.isDebugEnabled()) {
158                 sb.append("\n\tIAN Dataset: \n");
159                 StringWriter sw = new StringWriter();
160                 try {
161                     ian.dumpDataset(sw, null);
162                     sb.append(sw.toString());
163                 } catch (Throwable e) {
164                     sb.append("Failed to dump dataset due to: " + e.getMessage());
165                 }
166             } else {
167                 sb.append("\n\tIAN ").append(ian);
168             }
169         }
170         sb.append("\n");
171         return sb.toString();
172     }

174     /**
175      * Processes order attributes based on the {@code Dataset} set in the {@code ctor}.
176      * @see BaseJmsOrder#processOrderProperties(Object...)
177      */
178     @Override
179     public void processOrderProperties(Object... properties) {
180         this.setOrderProperty(JmsOrderProperties.STUDY_INSTANCE_UID, ian.getString(Tags.StudyInstanceUID));
181         
182         List<String> seriesUIDList = new ArrayList<String>();
183         DcmElement refSeriesSeq = ian.get(Tags.RefSeriesSeq);
184         if ( refSeriesSeq != null ) {
185             for ( int i = 0; i < refSeriesSeq.countItems(); i++ ) {
186                 Dataset refSeriesDS = refSeriesSeq.getItem(i);
187                 seriesUIDList.add(refSeriesDS.getString(Tags.SeriesInstanceUID));
188             }
189         }
190         
191         this.setOrderMultiProperty(JmsOrderProperties.SERIES_INSTANCE_UID, seriesUIDList.toArray(new String[0]));
192     }
