179     public File retrieveFileFromTAR(String fsID, String fileID)
180             throws IOException, VerifyTarException {
181         if (!fsID.startsWith("tar:")) {
182             throw new IllegalArgumentException(
183                     "Not a tar file system: " + fsID);
184         }
185         int tarEnd = fileID.indexOf('!');
186         if (tarEnd == -1) {
187             throw new IllegalArgumentException("Missing ! in " + fileID);
188         }
189         String tarPath = fileID.substring(0, tarEnd);
190         File cacheDir = new File(journal.getDataRootDir(),
191                 tarPath.replace('/', File.separatorChar));
192         String fpath = fileID.substring(tarEnd + 1)
193                             .replace('/', File.separatorChar);
194         File f = new File(cacheDir, fpath);
195         if (f.exists()) {
196             journal.record(cacheDir, true);
197             return f;
198         } else {
199             boolean extracted = false;
200             if (extracting.add(tarPath)) {
201                 try {
202                     fetchAndExtractTar(fsID, tarPath, cacheDir);
203                     extracted = true;
204                 } finally {
205                     synchronized (extracting) {
206                         extracting.remove(tarPath);
207                         extracting.notifyAll();
208                     }
209                 }
210             } else {
211                 if (log.isDebugEnabled())
212                     log.debug("Wait for concurrent fetch and extract of tar: "
213                             + tarPath);
214                 synchronized (extracting) {
215                     while (extracting.contains(tarPath))
216                         try {
217                             extracting.wait();
218                         } catch (InterruptedException e) {
219                             log.warn("Wait for concurrent fetch and extract of tar: "
220                             + tarPath + " interrupted:", e);
221                         }
222                 }
223             }
224             journal.record(cacheDir, !extracted);
225         }
226         if (!f.exists()) {
227             log.error("Tar file "+tarPath+" doesn't contain file "+fpath+"!");
228             throw new FileNotFoundException(f.getPath());
229         }
230         return f;
231     }

233     private void fetchAndExtractTar(String fsID, String tarPath, File cacheDir) throws IOException,
234             VerifyTarException {
235         File tarFile = fetchTarFile(fsID, tarPath);
236         try {
237             extractTar(tarFile, cacheDir);
238         } finally {
239             fetchHSMFileFinished(fsID, tarPath, tarFile);
240         }
241     }

243     public File fetchTarFile(String fsID, String tarPath) throws IOException {
244         return hsmModuleServicename == null ? 
245                 FileUtils.toFile(fsID.substring(4), tarPath) : fetchHSMFile(fsID, tarPath);
246     }

248     private File fetchHSMFile(String fsID, String tarPath) throws IOException {
249         try {
250             return (File) server.invoke(hsmModuleServicename, "fetchHSMFile", new Object[]{fsID, tarPath}, 
251                 new String[]{String.class.getName(),String.class.getName()});
252         } catch (Exception x) {
253             log.error("Fetch of HSMFile failed! fsID:"+fsID+" tarPath:"+tarPath, x);
254             IOException iox = new IOException("Fetch of HSMFile failed!");
255             iox.initCause(x);
256             throw iox;
257         }
258     }

260     private void fetchHSMFileFinished(String fsID, String tarPath, File tarFile) throws IOException {
261         if (hsmModuleServicename != null) {
262             try {
263                 server.invoke(hsmModuleServicename, "fetchHSMFileFinished", new Object[]{fsID, tarPath, tarFile}, 
264                     new String[]{String.class.getName(),String.class.getName(),File.class.getName()});
265             } catch (Exception x) {
266                 log.warn("fetchHSMFileFinished! fsID:"+fsID+" tarPath:"+tarPath+" tarFile:"+tarFile, x);
267             }
268         }
269     }

271     private void extractTar(File tarFile, File cacheDir)
272             throws IOException, VerifyTarException {
273         int count = 0;
274         long totalSize = 0;
275         long free = FileSystemUtils.freeSpace(
276                 journal.getDataRootDir().getPath());
277         long fsize = tarFile.length();
278         long toDelete = fsize + minFreeDiskSpace - free;
279         if (toDelete > 0)
280             free += free(toDelete);
281         byte[] buf = new byte[bufferSize];
282         TarInputStream tar = new TarInputStream(new FileInputStream(tarFile));
283         InputStream in = tar;
284         try {
285             TarEntry entry = skipDirectoryEntries(tar);
286             if (entry == null)
287                 throw new IOException("No entries in " + tarFile);
288             String entryName = entry.getName();
289             Map<String, byte[]> md5sums = null;
290             MessageDigest digest = null;
291             if ("MD5SUM".equals(entryName)) {
292                 if (checkMD5) {
293                     try {
294                         digest = MessageDigest.getInstance("MD5");
295                     } catch (NoSuchAlgorithmException e) {
296                         throw new RuntimeException(e);
297                     }
298                     md5sums = new HashMap<String, byte[]>();
299                     BufferedReader lineReader = new BufferedReader(
300                             new InputStreamReader(tar));
301                     String line;
302                     while ((line = lineReader.readLine()) != null) {
303                         md5sums.put(line.substring(34), 
304                                 MD5Utils.toBytes(line.substring(0, 32)));
305                     }
306                 }
307                 entry = skipDirectoryEntries(tar);
308             } else if (checkMD5 ) {
309                 getLog().warn("Missing MD5SUM entry in " + tarFile);
310             }
311             for (; entry != null; entry = skipDirectoryEntries(tar)) {
312                 entryName = entry.getName();
313                 // Retrieve saved MD5 checksum
314                 byte[] md5sum = null;
315                 if (md5sums != null && digest != null) {
316                     md5sum = md5sums.remove(entryName);
317                     if (md5sum == null)
318                         throw new VerifyTarException("Unexpected TAR entry: "
319                                 + entryName + " in " + tarFile);
320                     digest.reset();
321                     in = new DigestInputStream(tar, digest);
322                 }
323 
324                 File fOri = new File(cacheDir, 
325                         entryName.replace('/', File.separatorChar));
326                 File f = new File(fOri.getAbsolutePath()+".tmp");
327                 File dir = f.getParentFile();
328                 if (dir.mkdirs()) {
329                     log.info("M-WRITE " + dir);
330                 }
331                 log.info("M-WRITE " + f);
332                 // Write the stream to file
333                 FileOutputStream out = new FileOutputStream(f);
334                 boolean cleanup = true;
335                 try {
336                     int len;
337                     while ((len = in.read(buf)) > 0) {
338                         out.write(buf, 0, len);
339                     }
340                     cleanup = false;
341                 } finally {
342                     try {
343                         out.close();
344                     } catch (Exception ignore) {
345                     }
346                     if (cleanup) {
347                         log.info("M-DELETE " + f);
348                         f.delete();
349                     }
350                 }
351 
352                 // Verify MD5
353                 if (md5sums != null && digest != null) {
354                     if (!Arrays.equals(digest.digest(), md5sum)) {
355                         log.info("M-DELETE " + f);
356                         f.delete();
357                         throw new VerifyTarException(
358                                 "Failed MD5 check of TAR entry: " + entryName
359                                         + " in " + tarFile);
360                     } else
361                         log.info("MD5 check is successful for " + entryName
362                                 + " in " + tarFile);
363                 }
364                 free -= f.length();
365                 count++;
366                 totalSize += f.length();
367                 if (f.exists())
368                     f.renameTo(fOri);
369             }
370         } finally {
371             tar.close();
372         }
373         toDelete = prefFreeDiskSpace - free;
374         if (toDelete > 0) {
375             freeNonBlocking(toDelete);
376         }
377     }
378 
379     private TarEntry skipDirectoryEntries(TarInputStream tar)
380             throws IOException {
381         for (TarEntry entry = tar.getNextEntry(); entry != null;
382             entry = tar.getNextEntry()) { 
383             if (!entry.isDirectory())
384                 return entry;
385         }
386         return null;
387     }
388 
389     private void freeNonBlocking(final long toDelete) {
390         new Thread(new Runnable() {
391 
392             public void run() {
393                 try {
394                     free(toDelete);
395                 } catch (IOException e) {
396                     log.error(e.getMessage(), e);
397                 }
398             }
399         }).start();
400     }

402     public long free(long size) throws IOException {
403         log.info("Start deleting LRU directories of at least " + size
404                 + " bytes from TAR cache");
405         long deleted = journal.free(size);
406         log.info("Finished deleting LRU directories with " + deleted
407                 + " bytes from TAR cache");
408         return deleted;
409     }
