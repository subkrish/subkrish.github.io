40  package org.dcm4chex.archive.web.maverick.gppps;
41  
42  import java.text.ParseException;
43  
44  import javax.servlet.http.HttpServletRequest;
45  
46  import org.dcm4chex.archive.web.maverick.Dcm4cheeFormController;
47  import org.dcm4chex.archive.web.maverick.gppps.model.GPPPSEntry;
48  import org.dcm4chex.archive.web.maverick.gppps.model.GPPPSModel;
49  
50  /**
51   * @author franz.willer
52   *
53   * The Maverick controller for Media Creation Manager.
54   */
55  public class GPPPSConsoleCtrl extends Dcm4cheeFormController {
56  
57  
58      /** the view model. */
59      private GPPPSModel model;
60  
61      private static GPPPSDelegate delegate = null;
62  
63      /**
64       * Get the model for the view.
65       */
66      protected Object makeFormBean() {
67          if ( delegate == null ) {
68              delegate = new GPPPSDelegate();
69              delegate.init( getCtx().getServletConfig() );
70          }
71          model =  GPPPSModel.getModel(getCtx().getRequest());
72          return model;
73      }
74  
75  
76  
77      protected String perform() throws Exception {
78          try {
79              HttpServletRequest request = getCtx().getRequest();
80              model = GPPPSModel.getModel(request);
81              model.clearPopupMsg();
82              model.setGpppsIUIDs( request.getParameterValues("gpppsIUID"), false );
83              if ( request.getParameter("filter.x") != null ) {//action from filter button
84                  try {
85                      checkFilter( request );
86                      model.filterWorkList( true );
87                  } catch ( Exception x ) {
88                      model.setPopupMsg("folder.err_datetime", "yyyy/MM/dd" );
89                  }
90              } else if ( request.getParameter("prev.x") != null ) { 
91                  model.performPrevious();
92              } else if ( request.getParameter("next.x") != null ) { 
93                  model.performNext();
94              } else if ( request.getParameter("link.x") != null ) {//action from link button. (sticky support;redirect to gpwl ctrl.)
95                  model.setGpppsIUIDs( request.getParameterValues( "gpppsIUID" ), true );
96                  model.getFilter().setPatientName( request.getParameter("patientName"));
97                  return "link";
98              } else if ( request.getParameter("del.x") != null ) {//action from delete button.
99                  if ( model.getGpppsIUIDs() != null ) {
100                     delegate.deleteGPPPSEntries(model.getGpppsIUIDs());
101                     model.setGpppsIUIDs(null, false);
102                     model.filterWorkList( true );
103                 } else {
104                     model.setPopupMsg("gppps.err_delete_selection","");
105                 }
106             } else {
107                 String action = request.getParameter("action");
108                 if ( action != null ) {
109                     return performAction( action, request );
110                 }
111             }
112             return SUCCESS;
113         } catch (Exception e) {
114             model.setPopupMsg("gppps.err", e.getMessage());
115             return "error";
116         }
117     }
118 
119     /**
120      * @param action
121      * @param request
122      */
123     private String performAction(String action, HttpServletRequest request) {
124         if ( "linkDone".equals(action) ) {
125             model.filterWorkList( true );
126         } else if ( "inspect".equals(action) ) {
127             return inspect(request.getParameter("gpppsIUID"));
128         }
129         return SUCCESS;
130     }
131 
132 
133     private String inspect(String iuid) {
134         if ( iuid != null ) {
135             GPPPSEntry entry = model.getGPPPSEntry(iuid);
136             if ( entry != null ) {
137                 this.getCtx().getRequest().getSession().setAttribute("dataset2view", entry.toDataset());
138                 return INSPECT;
139             } else {
140                 model.setPopupMsg("gppps.err_inspect",iuid);
141             }
142         }
143         return SUCCESS;   
144     }
145 
146     /**
147      * Checks the http parameters for filter params and update the filter.
148      * 
149      * @param rq The http request.
150      * 
151      * @throws ParseException
152      * 
153      */
154     private void checkFilter(HttpServletRequest rq) throws ParseException {
155         PPSFilter filter = model.getFilter();
156         if ( rq.getParameter("patientName") != null ) filter.setPatientName(rq.getParameter("patientName") );
157         if ( rq.getParameter("startDate") != null ) filter.setStartDate(rq.getParameter("startDate") );
158         if ( rq.getParameter("status") != null ) filter.setStatus(rq.getParameter("status") );
159     }
160 
161     /**
162      * Returns the delegater that is used to query the MWLSCP or delete an MWL Entry (only if MWLSCP AET is local)
163      * 
164      * @return The delegator.
165      */
166     public static GPPPSDelegate getGPPPSDelegate() {
167         return delegate;
168     }
169 
170     protected String getCtrlName() {
171         return "gppps_console";
172     }
173 
174 }
