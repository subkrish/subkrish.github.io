98      /**
99       * Get the model for an http request.
100      * <p>
101      * Look in the session for an associated model via <code>MPPS_MODEL_ATTR_NAME</code><br>
102      * If there is no model stored in session (first request) a new model is created and stored in session.
103      * 
104      * @param request A http request.
105      * 
106      * @return The model for given request.
107      */
108     public static final MPPSModel getModel( ControllerContext ctx ) {
109         HttpServletRequest request = ctx.getRequest();
110         MPPSModel model = (MPPSModel) request.getSession().getAttribute(MPPS_MODEL_ATTR_NAME);
111         if (model == null) {
112             model = new MPPSModel(request);
113             model.getFilter().setEmptyAccNo(ctx.getServletConfig().getInitParameter("MPPS_unscheduled"));
114             request.getSession().setAttribute(MPPS_MODEL_ATTR_NAME, model);
115             if ( !"true".equals(ctx.getServletConfig().getInitParameter("MPPS_startWithoutQuery"))) {
116                 model.filterWorkList( true );
117             }
118         }
119         return model;
120     }

142     /**
143      * @param stickies The stickies to set.
144      * @param check
145      */
146     public void setMppsIUIDs(String[] stickies, boolean check) {
147         this.mppsIUIDs = stickies;
148         stickyList = new HashMap();
149         if ( mppsEntries.isEmpty() || mppsIUIDs == null || mppsIUIDs.length < 1) return;
150         MPPSEntry stickyEntry = (MPPSEntry) mppsEntries.get(mppsIUIDs[0]);
151         String patID = stickyEntry.getPatientID(); 
152         stickyList.put( mppsIUIDs[0], stickyEntry );
153         for ( int i = 1; i < mppsIUIDs.length ; i++ ) {
154             stickyEntry = (MPPSEntry) mppsEntries.get(mppsIUIDs[i]);
155             if ( check && ! patID.equals( stickyEntry.getPatientID() )) {
156                 throw new IllegalArgumentException("All selected MPPS must have the same patient!");
157             }
158             stickyList.put( mppsIUIDs[i], stickyEntry );
159         }
160     }

174     /**
175      * Update the list of MPPSEntries for the view.
176      * <p>
177      * The query use the search criteria values from the filter and use offset and limit for paging.
178      * <p>
179      * if <code>newSearch is true</code> will reset paging (set <code>offset</code> to 0!)
180      * @param newSearch
181      */
182     public void filterWorkList(boolean newSearch) {
183         int total = 0;
184         int limit = getLimit();
185         int offset = getOffset();
186         if ( newSearch ) {
187             setOffset(0);
188             total = MPPSConsoleCtrl.getMppsDelegate().countMppsEntries(this.mppsFilter);
189             setTotal(total);
190         }
191         List<Dataset> l = MPPSConsoleCtrl.getMppsDelegate().findMppsEntries( this.mppsFilter, offset, limit );
192         if ( l == null ) {
193             this.setPopupMsg("mpps.err", "findMppsEntries");
194             return;
195         }
196         if ( l.size() == limit ) {
197             offset += limit;
198         }
199         mppsEntries.clear();
200         if ( stickyList != null ) {
201             mppsEntries.putAll(stickyList);
202         }
203         MPPSEntry entry;
204         int countNull = 0;
205         for ( Dataset ds : l ){
206             if ( ds != null ) {
207                 entry = new MPPSEntry( ds );
208                 mppsEntries.put( entry.getMppsIUID(), entry );
209             } else {
210                 countNull++;
211             }
212         }
213         if (countNull > 0) {
214             log.warn("List of MPPS entries contains null entries! countNull:"+countNull); 
215         }
216     }

218     public String getPatientOfSelectedMpps(String mppsIUID) {
219         if ( stickyList.isEmpty() ) return null;
220         if ( mppsIUID == null ) {
221             return ((MPPSEntry) stickyList.values().iterator().next()).getPatientName();
222         } else {
223             MPPSEntry entry =(MPPSEntry) stickyList.get(mppsIUID);
224             return entry == null ? null : entry.getPatientName();
225         }
226     }
227     public String getStartDateOfSelectedMpps(String mppsIUID) {
228         if ( stickyList.isEmpty() ) return null;
229         String startDate = null;
230         if ( mppsIUID == null ) {
231             startDate = ((MPPSEntry) stickyList.values().iterator().next()).getPpsStartDateTime();
232         } else {
233             MPPSEntry entry =(MPPSEntry) stickyList.get(mppsIUID);
234             if ( entry != null ) {
235                 startDate = entry.getPpsStartDateTime();
236             }
237         }
238         if ( startDate != null && startDate.length() > 10) //shrink to yyyy/mm/dd
239             startDate = startDate.substring(0,10);
240         return startDate;
241     }
242     public String getModalityOfSelectedMpps(String mppsIUID) {
243         if ( stickyList.isEmpty() ) return null;
244         if ( mppsIUID == null ) {
245             return ((MPPSEntry) stickyList.values().iterator().next()).getModality();
246         } else {
247             MPPSEntry entry =(MPPSEntry) stickyList.get(mppsIUID);
248             return entry == null ? null : entry.getModality();
249         }
250     }
