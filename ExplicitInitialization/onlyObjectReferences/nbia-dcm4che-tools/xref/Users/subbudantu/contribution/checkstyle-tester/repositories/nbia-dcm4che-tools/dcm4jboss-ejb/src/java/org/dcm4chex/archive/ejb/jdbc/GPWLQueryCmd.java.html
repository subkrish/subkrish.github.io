63  public class GPWLQueryCmd extends BaseDSQueryCmd {
64  
65      public static int transactionIsolationLevel = 0;
66      public static int blobAccessType = Types.LONGVARBINARY;
67  
68      private static final String[] FROM = { "Patient", "GPSPS"};
69  
70      private static final String[] SELECT = { "Patient.encodedAttributes",
71              "GPSPS.encodedAttributes"};
72  
73      private static final String[] RELATIONS = { "Patient.pk",
74              "GPSPS.patient_fk"};
75      
76      private static final String ITEM_CODE = "item_code";
77      private static final String APP_CODE = "app_code";
78      private static final String DEVNAME_CODE = "devname_code";
79      private static final String DEVCLASS_CODE = "devclass_code";
80      private static final String DEVLOC_CODE = "devloc_code";
81      private static final String PERF_CODE = "perf_code";
82      
83      public GPWLQueryCmd(Dataset keys, boolean fuzzyMatchingOfPN)
84              throws SQLException, DcmServiceException {
85          super(keys, true, false, transactionIsolationLevel);
86          try {
87              AttributeFilter patAttrFilter = AttributeFilter.getPatientAttributeFilter();
88              defineColumnTypes(new int[] { blobAccessType, blobAccessType });
89              String s;
90              // ensure keys contains (8,0005) for use as result filter
91              if (!keys.contains(Tags.SpecificCharacterSet)) {
92                  keys.putCS(Tags.SpecificCharacterSet);
93              }
94              sqlBuilder.setSelect(SELECT);
95              sqlBuilder.setFrom(FROM);
96              sqlBuilder.setRelations(RELATIONS);
97              sqlBuilder.setLeftJoin(getLeftJoin());
98              sqlBuilder.addListOfUidMatch(null, "GPSPS.sopIuid",
99                      SqlBuilder.TYPE1,
100                     keys.getStrings(Tags.SOPInstanceUID));
101             if ((s = keys.getString(Tags.GPSPSStatus)) != null) {
102                 sqlBuilder.addIntValueMatch(null, "GPSPS.gpspsStatusAsInt",
103                         SqlBuilder.TYPE1,
104                         GPSPSStatus.toInt(s));
105             }
106             if ((s = keys.getString(Tags.InputAvailabilityFlag)) != null) {
107                 sqlBuilder.addIntValueMatch(null, "GPSPS.inputAvailabilityFlagAsInt",
108                         SqlBuilder.TYPE1,
109                         InputAvailabilityFlag.toInt(s));
110             }
111             s = keys.getString(Tags.GPSPSPriority);
112             if (s != null) {
113                 sqlBuilder.addIntValueMatch(null, "GPSPS.gpspsPriorityAsInt",
114                         SqlBuilder.TYPE1,
115                         Priority.toInt(s));
116             }
117             sqlBuilder.addRangeMatch(null, "GPSPS.spsStartDateTime",
118                     SqlBuilder.TYPE1,
119                     keys.getDateRange(Tags.SPSStartDateAndTime));
120             sqlBuilder.addRangeMatch(null, "GPSPS.expectedCompletionDateTime",
121                     SqlBuilder.TYPE2,
122                     keys.getDateRange(Tags.ExpectedCompletionDateAndTime));        
123             addCodeMatch(Tags.ScheduledWorkitemCodeSeq, ITEM_CODE);
124             addCodeMatch(Tags.ScheduledProcessingApplicationsCodeSeq, APP_CODE);
125             addCodeMatch(Tags.ScheduledStationNameCodeSeq, DEVNAME_CODE);
126             addCodeMatch(Tags.ScheduledStationClassCodeSeq, DEVCLASS_CODE);
127             addCodeMatch(Tags.ScheduledStationGeographicLocationCodeSeq, DEVLOC_CODE);
128             Dataset item = keys.getItem(Tags.ScheduledHumanPerformersSeq);
129             if (item != null) {
130                 if (fuzzyMatchingOfPN)
131                     try {
132                         sqlBuilder.addPNFuzzyMatch(
133                                 new String[] {
134                                     "GPSPSPerformer.humanPerformerFamilyNameSoundex",
135                                     "GPSPSPerformer.humanPerformerGivenNameSoundex" },
136                                 SqlBuilder.TYPE2,
137                                 keys.getString(Tags.HumanPerformerName));
138                     } catch (IllegalArgumentException ex) {
139                         throw new DcmServiceException(
140                                 Status.IdentifierDoesNotMatchSOPClass,
141                                 ex.getMessage() + ": " + keys.get(Tags.HumanPerformerName));
142                     }
143                 else
144                     sqlBuilder.addPNMatch(
145                             new String[] {
146                                 "GPSPSPerformer.humanPerformerName",
147                                 "GPSPSPerformer.humanPerformerIdeographicName",
148                                 "GPSPSPerformer.humanPerformerPhoneticName"},
149                             SqlBuilder.TYPE2,
150                             true, // TODO make ICASE configurable
151                             item.getString(Tags.HumanPerformerName));
152                 addCodeMatch(item.getItem(Tags.HumanPerformerCodeSeq), PERF_CODE);
153             }
154             item = keys.getItem(Tags.RefRequestSeq);
155             if (item != null) {
156                 sqlBuilder.addListOfStringMatch(null,
157                         "GPSPSRequest.requestedProcedureId",
158                         SqlBuilder.TYPE2,
159                         item.getStrings(Tags.RequestedProcedureID));
160                 sqlBuilder.addListOfStringMatch(null,
161                         "GPSPSRequest.accessionNumber",
162                         SqlBuilder.TYPE2,
163                         item.getStrings(Tags.AccessionNumber));
164             }        
165             if (sqlBuilder.addListOfStringMatch(null, "Patient.patientId",
166                     SqlBuilder.TYPE1,
167                     patAttrFilter.getStrings(keys, Tags.PatientID)) != null)
168                 sqlBuilder.addSingleValueMatch(null, "Patient.issuerOfPatientId",
169                         SqlBuilder.TYPE2,
170                         patAttrFilter.getString(keys, Tags.IssuerOfPatientID));
171             if (fuzzyMatchingOfPN)
172                 try {
173                     sqlBuilder.addPNFuzzyMatch(
174                             new String[] {
175                                     "Patient.patientFamilyNameSoundex",
176                                     "Patient.patientGivenNameSoundex" },
177                             SqlBuilder.TYPE2,
178                             keys.getString(Tags.PatientName));
179                 } catch (IllegalArgumentException ex) {
180                     throw new DcmServiceException(
181                             Status.IdentifierDoesNotMatchSOPClass,
182                             ex.getMessage() + ": " + keys.get(Tags.PatientName));
183                 }
184             else
185                 sqlBuilder.addPNMatch(
186                         new String[] {
187                             "Patient.patientName",
188                             "Patient.patientIdeographicName",
189                             "Patient.patientPhoneticName" },
190                         SqlBuilder.TYPE2,
191                         patAttrFilter.isICase(Tags.PatientName),
192                         keys.getString(Tags.PatientName));
193         } catch (SQLException x) {
194             close();
195             throw x;
196         } catch (DcmServiceException x) {
197             close();
198             throw x;
199         } catch (RuntimeException x) {
200             close();
201             throw x;
202         }
203     }
204 
205     private void addCodeMatch(int tag, String alias) {
206         addCodeMatch(keys.getItem(tag), alias);
207     }
208 
209     private void addCodeMatch(Dataset item, String alias) {
210         if (item != null) {
211             sqlBuilder.addSingleValueMatch(alias, "Code.codeValue",
212                     SqlBuilder.TYPE2,
213                     item.getString(Tags.CodeValue));
214             sqlBuilder.addSingleValueMatch(alias, "Code.codingSchemeDesignator",
215                     SqlBuilder.TYPE2,
216                     item.getString(Tags.CodingSchemeDesignator));
217         }
218     }
219 
220     private boolean isMatchCode(int tag) {
221         return isMatchCode(keys.getItem(tag));
222     }
223 
224     private boolean isMatchCode(Dataset code) {
225         return code != null
226                 && (code.containsValue(Tags.CodeValue) 
227                         || code.containsValue(Tags.CodingSchemeDesignator));
228     }
229 
230     private boolean isMatchRefRequest() {
231         Dataset refrq = keys.getItem(Tags.RefRequestSeq);
232         return refrq != null
233                 && (refrq.containsValue(Tags.RequestedProcedureID)
234                         || refrq.containsValue(Tags.AccessionNumber));
235     }
236     
237     private String[] getLeftJoin() {
238         ArrayList list = new ArrayList();
239         if (isMatchCode(Tags.ScheduledWorkitemCodeSeq)) {
240             list.add("Code");
241             list.add(ITEM_CODE);
242             list.add("GPSPS.code_fk");
243             list.add("Code.pk");
244         }
245         if (isMatchCode(Tags.ScheduledProcessingApplicationsCodeSeq)) {
246             sqlBuilder.setDistinct(true);
247             list.add("rel_gpsps_appcode");
248             list.add(null);
249             list.add("GPSPS.pk");
250             list.add("rel_gpsps_appcode.gpsps_fk");
251             list.add("Code");
252             list.add(APP_CODE);
253             list.add("rel_gpsps_appcode.appcode_fk");
254             list.add("Code.pk");
255         }
256         if (isMatchCode(Tags.ScheduledStationNameCodeSeq)) {
257             sqlBuilder.setDistinct(true);
258             list.add("rel_gpsps_devname");
259             list.add(null);
260             list.add("GPSPS.pk");
261             list.add("rel_gpsps_devname.gpsps_fk");
262             list.add("Code");
263             list.add(DEVNAME_CODE);
264             list.add("rel_gpsps_devname.devname_fk");
265             list.add("Code.pk");
266         }
267         if (isMatchCode(Tags.ScheduledStationClassCodeSeq)) {
268             sqlBuilder.setDistinct(true);
269             list.add("rel_gpsps_devclass");
270             list.add(null);
271             list.add("GPSPS.pk");
272             list.add("rel_gpsps_devclass.gpsps_fk");
273             list.add("Code");
274             list.add(DEVCLASS_CODE);
275             list.add("rel_gpsps_devclass.devclass_fk");
276             list.add("Code.pk");
277         }
278         if (isMatchCode(Tags.ScheduledStationGeographicLocationCodeSeq)) {
279             sqlBuilder.setDistinct(true);
280             list.add("rel_gpsps_devloc");
281             list.add(null);
282             list.add("GPSPS.pk");
283             list.add("rel_gpsps_devloc.gpsps_fk");
284             list.add("Code");
285             list.add(DEVLOC_CODE);
286             list.add("rel_gpsps_devloc.devloc_fk");
287             list.add("Code.pk");
288         }
289         Dataset item = keys.getItem(Tags.ScheduledHumanPerformersSeq);
290         if (item != null) {
291             boolean matchCode = isMatchCode(item.getItem(Tags.HumanPerformerCodeSeq));
292             if (matchCode || item.containsValue(Tags.HumanPerformerName)) {
293                 sqlBuilder.setDistinct(true);
294                 list.add("GPSPSPerformer");
295                 list.add(null);
296                 list.add("GPSPS.pk");
297                 list.add("GPSPSPerformer.gpsps_fk");
298                 if (matchCode) {
299                     list.add("Code");
300                     list.add(PERF_CODE);
301                     list.add("GPSPSPerformer.code_fk");
302                     list.add("Code.pk");
303                 }
304             }
305         }
306         if (isMatchRefRequest()) {
307             sqlBuilder.setDistinct(true);
308             list.add("GPSPSRequest");
309             list.add(null);
310             list.add("GPSPS.pk");
311             list.add("GPSPSRequest.gpsps_fk");
312         }
313         return (String[]) (list.isEmpty() ? null
314                 : list.toArray(new String[list.size()]));
315     }

317     public Dataset getDataset() throws SQLException {
318         Dataset ds = DcmObjectFactory.getInstance().newDataset();
319         DatasetUtils.fromByteArray( rs.getBytes(1), ds);
320         DatasetUtils.fromByteArray( rs.getBytes(2), ds);
321         adjustDataset(ds, keys);
322         return ds.subSet(keys);
323     }
