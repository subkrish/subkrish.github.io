42  import java.sql.SQLException;
43  import java.sql.Types;
44  import java.util.ArrayList;
45  import java.util.List;
46  
47  import org.dcm4che.data.Dataset;
48  import org.dcm4che.dict.Tags;
49  import org.dcm4che.net.DcmServiceException;
50  import org.dcm4chex.archive.common.DatasetUtils;
51  
52  /**
53   * @author gunter.zeilinger@tiani.com
54   * @version $Revision: 16583 $ $Date: 2012-02-08 13:36:32 +0100 (Mi, 08 Feb 2012) $
55   * @since Aug 22, 2005
56   */
57  public class HPRetrieveCmd extends BaseReadCmd {
58  
59      public static int transactionIsolationLevel = 0;
60      public static int blobAccessType = Types.LONGVARBINARY;
61  
62      private static final String[] FROM = { "HP" };
63  
64      private static final String[] SELECT = { "HP.encodedAttributes" };
65  
66      private final SqlBuilder sqlBuilder = new SqlBuilder();
67  
68      public HPRetrieveCmd(Dataset keys) throws SQLException {
69          super(JdbcProperties.getInstance().getDataSource(),
70                  transactionIsolationLevel);
71          try {
72              defineColumnTypes(new int[] { blobAccessType });
73              sqlBuilder.setSelect(SELECT);
74              sqlBuilder.setFrom(FROM);
75              sqlBuilder.addListOfUidMatch(null, "HP.sopIuid", SqlBuilder.TYPE1,
76                      keys.getStrings(Tags.SOPInstanceUID));
77          } catch (SQLException x) {
78              close();
79              throw x;
80          } catch (RuntimeException x) {
81              close();
82              throw x;
83          }
84      }
85  
86      public List getDatasets() throws SQLException {
87          ArrayList result = new ArrayList();
88          try {
89              execute(sqlBuilder.getSql());
90              while (next()) {
91                  result.add(DatasetUtils.fromByteArray(rs.getBytes(1)));			
92              }
93          } finally {
94              close();
95          }
96          return result;
97      }
98  
99  }
