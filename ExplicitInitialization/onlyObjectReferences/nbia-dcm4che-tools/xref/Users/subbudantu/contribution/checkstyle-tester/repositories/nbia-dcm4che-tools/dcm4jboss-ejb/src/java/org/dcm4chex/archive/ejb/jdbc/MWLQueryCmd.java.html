59  public class MWLQueryCmd extends BaseDSQueryCmd {
60      
61      private static final String[] FROM = { "Patient", "MWLItem" };
62      private static final String[] SELECT = { "Patient.encodedAttributes",
63              "MWLItem.encodedAttributes" };
64      private static final String[] RELATIONS = { "Patient.pk",
65              "MWLItem.patient_fk" };
66  
67      private static final int[] MATCHING_KEYS = new int[] {
68          Tags.RequestedProcedureID,
69  		Tags.AccessionNumber,
70  		Tags.StudyInstanceUID,
71  		Tags.PatientID,
72  		Tags.IssuerOfPatientID,
73  		Tags.PatientName,
74  		Tags.SPSSeq
75  		};
76      private static final int[] MATCHING_SPS_SQ_KEYS = new int[] {
77          Tags.SPSStatus,
78  		Tags.SPSID,
79  		Tags.SPSStartDate, Tags.SPSStartTime,
80  		Tags.Modality,
81  		Tags.ScheduledPerformingPhysicianName,
82  		Tags.ScheduledStationAET,
83  		Tags.ScheduledStationName,
84  		};
85      
86      public static int transactionIsolationLevel = 0;
87      public static int blobAccessType = Types.LONGVARBINARY;
88  
89      protected static final DcmObjectFactory dof = DcmObjectFactory.getInstance();
90      
91      /**
92       * @param ds
93       * @throws SQLException
94       * @throws DcmServiceException 
95       */
96      public MWLQueryCmd(Dataset keys, boolean fuzzyMatchingOfPN,
97              boolean noMatchForNoValue) throws SQLException, DcmServiceException {
98          super(keys, true, noMatchForNoValue, transactionIsolationLevel);
99          try {
100             AttributeFilter patAttrFilter = AttributeFilter.getPatientAttributeFilter();
101             defineColumnTypes(new int[] { blobAccessType, blobAccessType });
102             // ensure keys contains (8,0005) for use as result filter
103             if (!keys.contains(Tags.SpecificCharacterSet)) {
104                 keys.putCS(Tags.SpecificCharacterSet);
105             }
106             sqlBuilder.setSelect(SELECT);
107             sqlBuilder.setFrom(FROM);
108             sqlBuilder.setRelations(RELATIONS);
109             Dataset spsItem = keys.getItem(Tags.SPSSeq);
110             if (spsItem != null) {
111                 sqlBuilder.addListOfIntMatch(null, "MWLItem.spsStatusAsInt",
112                             SqlBuilder.TYPE1, 
113                             SPSStatus.toInts(spsItem.getStrings(Tags.SPSStatus)));
114                 sqlBuilder.addWildCardMatch(null, "MWLItem.spsId",
115                         SqlBuilder.TYPE1,
116                         spsItem.getStrings(Tags.SPSID));
117                 sqlBuilder.addRangeMatch(null, "MWLItem.spsStartDateTime",
118                         SqlBuilder.TYPE1,
119                         spsItem.getDateTimeRange(Tags.SPSStartDate,
120                                 Tags.SPSStartTime));
121                 sqlBuilder.addWildCardMatch(null, "MWLItem.modality",
122                         SqlBuilder.TYPE1,
123                         spsItem.getStrings(Tags.Modality));
124                 sqlBuilder.addWildCardMatch(null, "MWLItem.scheduledStationAET",
125                         SqlBuilder.TYPE1,
126                         spsItem.getStrings(Tags.ScheduledStationAET));
127                 sqlBuilder.addWildCardMatch(null, "MWLItem.scheduledStationName",
128                         type2,
129                         spsItem.getStrings(Tags.ScheduledStationName));
130                 if (fuzzyMatchingOfPN)
131                     try {
132                         sqlBuilder.addPNFuzzyMatch(
133                                 new String[] {
134                                     "MWLItem.performingPhysicianFamilyNameSoundex",
135                                     "MWLItem.performingPhysicianGivenNameSoundex" },
136                                 type2,
137                                 keys.getString(Tags.ScheduledPerformingPhysicianName));
138                     } catch (IllegalArgumentException ex) {
139                         throw new DcmServiceException(
140                                 Status.IdentifierDoesNotMatchSOPClass,
141                                 ex.getMessage() + ": " + keys.get(Tags.ScheduledPerformingPhysicianName));
142                     }
143                 else
144                     sqlBuilder.addPNMatch(
145                             new String[] {
146                                 "MWLItem.performingPhysicianName",
147                                 "MWLItem.performingPhysicianIdeographicName",
148                                 "MWLItem.performingPhysicianPhoneticName"},
149                             true, // TODO make ICASE configurable
150                             type2,
151                             spsItem.getString(Tags.ScheduledPerformingPhysicianName));
152             }
153             sqlBuilder.addWildCardMatch(null, "MWLItem.requestedProcedureId",
154                     SqlBuilder.TYPE1,
155                     keys.getStrings(Tags.RequestedProcedureID));
156             sqlBuilder.addWildCardMatch(null, "MWLItem.accessionNumber",
157                     type2,
158                     keys.getStrings(Tags.AccessionNumber));
159             sqlBuilder.addListOfStringMatch(null, "MWLItem.studyIuid",
160                     SqlBuilder.TYPE1,
161                     keys.getStrings(Tags.StudyInstanceUID));
162             if (sqlBuilder.addWildCardMatch(null, "Patient.patientId",
163                     SqlBuilder.TYPE1,
164                     patAttrFilter.getStrings(keys, Tags.PatientID)) != null)
165                 sqlBuilder.addSingleValueMatch(null, "Patient.issuerOfPatientId",
166                         type2,
167                         patAttrFilter.getString(keys, Tags.IssuerOfPatientID));
168             if (fuzzyMatchingOfPN)
169                 try {
170                     sqlBuilder.addPNFuzzyMatch(
171                             new String[] {
172                                 "Patient.patientFamilyNameSoundex",
173                                 "Patient.patientGivenNameSoundex" },
174                             type2,
175                             keys.getString(Tags.PatientName));
176                 } catch (IllegalArgumentException ex) {
177                     throw new DcmServiceException(
178                             Status.IdentifierDoesNotMatchSOPClass,
179                             ex.getMessage() + ": " + keys.get(Tags.PatientName));
180                 }
181             else
182                 sqlBuilder.addPNMatch(
183                         new String[] {
184                             "Patient.patientName",
185                             "Patient.patientIdeographicName",
186                             "Patient.patientPhoneticName"},
187                         type2,
188                         patAttrFilter.isICase(Tags.PatientName),
189                         keys.getString(Tags.PatientName));
190         } catch (SQLException x) {
191             close();
192             throw x;
193         } catch (DcmServiceException x) {
194             close();
195             throw x;
196         } catch (RuntimeException x) {
197             close();
198             throw x;
199         }
200     }
201 
202     public Dataset getDataset() throws SQLException {
203         Dataset ds = DcmObjectFactory.getInstance().newDataset();    
204         Dataset dsPat = DcmObjectFactory.getInstance().newDataset(); //It seems that Oracle has problems to read 1st BLOB after 2nd! 
205         DatasetUtils.fromByteArray(rs.getBytes(1), dsPat); //patient
206         DatasetUtils.fromByteArray(rs.getBytes(2), ds); //mwl item
207         ds.putAll(dsPat);
208         adjustDataset(ds, keys);
209         return ds.subSet(keys);
210     }
