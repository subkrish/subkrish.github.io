38  package org.dcm4chex.archive.ejb.jdbc;
39  
40  import java.sql.PreparedStatement;
41  import java.sql.SQLException;
42  import java.util.ArrayList;
43  import java.util.Collection;
44  
45  /**
46   * @author Gunter Zeilinger <gunterze@gmail.com>
47   * @version $Revision$ $Date$
48   * @since May 6, 2009
49   */
50  public class QueryExternalRetrieveAETsOfSeriesCmd extends BaseReadCmd {
51  
52      private static final String SQL = "SELECT i.sop_iuid, i.ext_retr_aet "
53          + "FROM instance i, series s "
54          + "WHERE i.series_fk = s.pk AND s.series_iuid=?";
55  
56      public static int transactionIsolationLevel = 0;
57  
58      public QueryExternalRetrieveAETsOfSeriesCmd(String seriuid, int fetchSize)
59              throws SQLException {
60          super(JdbcProperties.getInstance().getDataSource(),
61                  transactionIsolationLevel, SQL);
62          ((PreparedStatement) stmt).setString(1, seriuid);
63          setFetchSize(fetchSize);
64          execute();
65      }
66  
67      public Collection<String[]> getRetrieveAETs() throws SQLException {
68          Collection<String[]> c = new ArrayList<String[]>();
69          try {
70              while (next()) {
71                  c.add(new String[] { rs.getString(1), rs.getString(2) });
72              }
73          } finally {
74              close();
75          }
76          return c;
77      }
78  }
