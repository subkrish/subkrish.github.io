68      public Set<String> getTarFilenamesToMigrate(long fsPk, int[] fileStati, 
69              boolean lastPksFirst, int limit, int offset) throws SQLException {
70          sqlBuilder.setFrom( new String[] {"File"} );
71          sqlBuilder.setFieldNamesForSelect( new String[] { getSelectTarFilename("files") });
72          sqlBuilder.addIntValueMatch(null, "File.filesystem_fk", false, (int)fsPk);
73          addFileStatiMatch(fileStati);
74          sqlBuilder.addOrderBy("File.pk", lastPksFirst ? " DESC" : " ASC");
75          sqlBuilder.setDistinct(true);
76          sqlBuilder.setLimit(limit);
77          sqlBuilder.setOffset(offset);
78          HashSet<String> result = new HashSet<String>();
79          try {
80              execute(sqlBuilder.getSql());
81              while (next()) {
82                  result.add(toTarFn(rs.getString(1)));
83              }
84              return result;
85          } finally {
86              close();
87          }
88      }
89  
90  
91      private String getSelectTarFilename(String alias) {
92          switch(JdbcProperties.getInstance().getDatabase()) {
93              case JdbcProperties.PSQL:
94              case JdbcProperties.FIREBIRD:
95                  return "SUBSTRING("+alias+".filepath FROM 1 FOR POSITION('!' IN "+alias+".filepath))";
96              case JdbcProperties.ORACLE:
97                  return "SUBSTR("+alias+".filepath, 1, INSTR("+alias+".filepath,'!'))";
98              case JdbcProperties.DB2:
99                  return "SUBSTR("+alias+".filepath, 1, LOCATE('!', "+alias+".filepath))";
100             case JdbcProperties.MSSQL:
101                 return "SUBSTRING("+alias+".filepath, 1, PATINDEX('%!%', "+alias+".filepath))";
102             default:
103                 return "SUBSTRING("+alias+".filepath FROM 1 FOR LOCATE('!',"+alias+".filepath))";
104         }
105     }

108     /**
109      * Return list of tar filenames and file status for source FS and target FS of instances
110      * String[]:
111      * [0]..source tar filename
112      * [1]..source file status
113      * [2]..target tar filename
114      * [3]..target file status
115      * @param lastFilePkForRemove 
116      * @param srcFsId
117      * @param targetFsId
118      * @param lastPksFirst
119      * @param limit
120      * @return
121      * @throws SQLException
122      */
123     public List<String[]> getTarFilenamesAndStatus(long srcFsPk, long targetFsPk, int[] targetFileStati, 
124             int limit, int offset) throws SQLException {
125         sqlBuilder.setFrom(new String[] {"File", "File"});
126         sqlBuilder.setAliases(new String[] {ALIAS_SRC, null});
127         sqlBuilder.setFieldNamesForSelect( new String[] { 
128                 getSelectTarFilename(ALIAS_SRC),
129                 "src.file_status",
130                 getSelectTarFilename("files"),
131                 "files.file_status"
132                 });
133         sqlBuilder.addIntValueMatch(ALIAS_SRC, "File.filesystem_fk", false, (int)srcFsPk);
134         sqlBuilder.addIntValueMatch(null, "File.filesystem_fk", false, (int)targetFsPk);
135         sqlBuilder.addFieldValueMatch(ALIAS_SRC, "File.instance_fk", false, null, "File.instance_fk");
136         addFileStatiMatch(targetFileStati);
137         sqlBuilder.setDistinct(true);
138         sqlBuilder.addOrderBy("File.pk", " ASC");
139         sqlBuilder.setLimit(limit);
140         sqlBuilder.setOffset(offset);
141         ArrayList<String[]> result = new ArrayList<String[]>();
142         try {
143             execute(sqlBuilder.getSql());
144             while (next()) {
145                 result.add(new String[]{toTarFn(rs.getString(1)), FileStatus.toString(rs.getInt(2)), 
146                         toTarFn(rs.getString(3)), FileStatus.toString(rs.getInt(4))});
147             }
148             return result;
149         } finally {
150             close();
151         }
152     }

154     public List<int[]> countFilesPerStatus(long srcFsPk) throws SQLException {
155         sqlBuilder.setFrom(new String[] {"File"});
156         sqlBuilder.setSelect(new String[]{"File.fileStatus",
157                 sqlBuilder.getCountOf(null, false)});
158         sqlBuilder.addIntValueMatch(null, "File.filesystem_fk", false, (int)srcFsPk);
159         sqlBuilder.setGroupBy("File.fileStatus");
160         ArrayList<int[]> result = new ArrayList<int[]>();
161         try {
162             execute(sqlBuilder.getSql());
163             while (next()) {
164                 result.add(new int[]{rs.getInt(1), rs.getInt(2)});
165             }
166             return result;
167         } finally {
168             close();
169         }
170     }

172     public int countFileCopiesOfTarFile(long srcFsPk, String tarFile, long targetFsPk) throws SQLException {
173         sqlBuilder.setFrom(new String[] {"File", "File"});
174         sqlBuilder.setAliases(new String[] {ALIAS_SRC, null});
175         sqlBuilder.setSelect(new String[]{sqlBuilder.getCountOf(null, false)});
176         sqlBuilder.addIntValueMatch(ALIAS_SRC, "File.filesystem_fk", false, (int)srcFsPk);
177         sqlBuilder.addIntValueMatch(null, "File.filesystem_fk", false, (int)targetFsPk);
178         sqlBuilder.addFieldValueMatch(ALIAS_SRC, "File.instance_fk", false, null, "File.instance_fk");
179         sqlBuilder.addWildCardMatch(ALIAS_SRC, "File.filePath", false, tarFile+"!*");
180         sqlBuilder.addOrderBy("File.pk", " ASC");
181         try {
182             execute(sqlBuilder.getSql());
183             next();
184             return rs.getInt(1); 
185         } finally {
186             close();
187         }
188     }
189     
190     private void addFileStatiMatch(int[] fileStati) {
191         if (fileStati != null) {
192             if (fileStati.length == 1) {
193                 sqlBuilder.addIntValueMatch(null, "File.fileStatus", false, fileStati[0]);
194             } else {
195                 sqlBuilder.addListOfIntMatch(null, "File.fileStatus", false, fileStati);
196             }
197         }
198     }
199 
200     private String toTarFn(String s) {
201         return s.endsWith("!") ? s.substring(0,s.length()-1) : s;
202     }
