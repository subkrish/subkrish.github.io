61  public class QueryPrivateStudiesCmd extends BaseReadCmd {
62  
63      public static int transactionIsolationLevel = 0;
64      public static int blobAccessType = Types.LONGVARBINARY;
65  
66      private static final DcmObjectFactory dof = DcmObjectFactory.getInstance();
67  
68      private static final String[] SELECT_ATTRIBUTE = { 
69          "PrivatePatient.encodedAttributes",
70          "PrivateStudy.encodedAttributes",
71          "PrivatePatient.pk",
72      "PrivateStudy.pk"};
73  
74      private static final String[] ENTITY = {"PrivatePatient"};
75  
76      private static final String[] LEFT_JOIN = { 
77          "PrivateStudy", null, "PrivatePatient.pk", "PrivateStudy.patient_fk",};
78  
79      private boolean hideMissingStudies = false;
80  
81      private final SqlBuilder sqlBuilder = new SqlBuilder();
82  
83      public QueryPrivateStudiesCmd(Dataset filter, int privateType, boolean hideMissingStudies)
84      throws SQLException {
85          super(JdbcProperties.getInstance().getDataSource(),
86                  transactionIsolationLevel);
87          try {
88              this.hideMissingStudies = hideMissingStudies;
89              sqlBuilder.setFrom(ENTITY);
90              sqlBuilder.setLeftJoin(LEFT_JOIN);
91              sqlBuilder.addIntValueMatch(null, "PrivatePatient.privateType",
92                      SqlBuilder.TYPE1, privateType);
93              if ( filter != null ) {
94                  sqlBuilder.addWildCardMatch(null, "PrivatePatient.patientId",
95                          SqlBuilder.TYPE2,
96                          filter.getStrings(Tags.PatientID));
97                  sqlBuilder.addWildCardMatch(null, "PrivatePatient.patientName",
98                          SqlBuilder.TYPE2,
99                          toWildcardMatchString( filter.getString(Tags.PatientName)) );
100                 sqlBuilder.addWildCardMatch(null, "PrivateStudy.accessionNumber",
101                         SqlBuilder.TYPE2,
102                         filter.getStrings(Tags.AccessionNumber));
103                 sqlBuilder.addListOfStringMatch(null, "PrivateStudy.studyIuid",
104                         SqlBuilder.TYPE1, filter.getStrings( Tags.StudyInstanceUID));
105                 filter.setPrivateCreatorID(PrivateTags.CreatorID);
106                 sqlBuilder.addCallingAETsNestedMatch(true,
107                         filter.getStrings(PrivateTags.CallingAET));
108                 filter.setPrivateCreatorID(null);
109             }
110             if ( this.hideMissingStudies ) {
111                 sqlBuilder.addNULLValueMatch(null,"PrivateStudy.encodedAttributes", true);
112             }
113         } catch (RuntimeException x) {
114             close();
115             throw x;
116         }
117 
118     }
119 
120     private static String toWildcardMatchString(String patientName) {
121         if ( patientName != null ) {
122             patientName = patientName.toUpperCase();
123             if ( patientName.length() > 0 && 
124                     patientName.indexOf('*') == -1 &&
125                     patientName.indexOf('?') == -1) patientName+="*";
126         }
127         return patientName;
128     }
129 
130     public int count() throws SQLException {
131         try {
132             sqlBuilder.setSelectCount(new String[]{"PrivateStudy.pk"}, true);
133             execute( sqlBuilder.getSql() );
134             next();
135             if (hideMissingStudies) return rs.getInt(1);
136             //we have to add number of studies and number of patients without studies.
137             int studies = rs.getInt(1);
138             rs.close();
139             rs = null;
140             sqlBuilder.setSelectCount(new String[]{"PrivatePatient.pk"}, true);
141             sqlBuilder.addNULLValueMatch(null,"PrivateStudy.pk", false);
142             execute( sqlBuilder.getSql() );
143             next();
144             int emptyPatients = rs.getInt(1);
145             List matches = sqlBuilder.getMatches();
146             matches.remove( matches.size() - 1);//removes the Study.pk NULLValue match!
147             return studies + emptyPatients;
148         } finally {
149             close();
150         }
151     }
152 
153 
154     public List list(int offset, int limit) throws SQLException {
155         defineColumnTypes(new int[] {
156                 blobAccessType,
157                 blobAccessType,
158                 Types.BIGINT,
159                 Types.BIGINT});
160         sqlBuilder.setSelect(SELECT_ATTRIBUTE);
161         sqlBuilder.addOrderBy("PrivatePatient.pk", SqlBuilder.ASC);
162         sqlBuilder.setOffset(offset);
163         sqlBuilder.setLimit(limit);
164         try {
165             setFetchSize(limit);
166             execute(sqlBuilder.getSql());
167             ArrayList result = new ArrayList();
168 
169             while (next()) {
170                 final byte[] patAttrs = rs.getBytes(1);
171                 final byte[] styAttrs = rs.getBytes(2);
172                 Dataset ds = dof.newDataset();
173                 ds.setPrivateCreatorID(PrivateTags.CreatorID);
174                 ds.putOB(PrivateTags.PatientPk, Convert.toBytes(rs.getLong(3)) );
175                 ds.setPrivateCreatorID(null);
176                 long studyPk = rs.getLong(4);
177                 DatasetUtils.fromByteArray(patAttrs, ds);
178                 if (styAttrs != null) {
179                     ds.setPrivateCreatorID(PrivateTags.CreatorID);
180                     ds.putOB(PrivateTags.StudyPk, Convert.toBytes(studyPk) );
181                     ds.setPrivateCreatorID(null);
182                     DatasetUtils.fromByteArray(styAttrs, ds);
183                 } 
184                 ds.setPrivateCreatorID(null);
185                 result.add(ds);
186             }
187             return result;
188         } finally {
189             close();
190         }
191     }
