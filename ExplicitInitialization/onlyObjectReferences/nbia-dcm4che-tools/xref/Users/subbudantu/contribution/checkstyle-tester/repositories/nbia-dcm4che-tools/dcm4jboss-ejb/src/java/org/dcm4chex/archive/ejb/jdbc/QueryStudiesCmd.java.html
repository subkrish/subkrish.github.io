100     /**
101      * Creates a new QueryStudiesCmd object with given filter.
102      * <p>
103      * If parameter <code>noMatchForNoValue=true</code> all Type2 Matches are forced to Type1 matches and therefore all
104      * 'empty field' matches will be hidden.
105      * <p>
106      * Dont use this feature for DICOM queries!
107      *   
108      * @param keys                    Filter Dataset.
109      * @param hideMissingStudies        Hide patients without studies.
110      * @param noMatchForNoValue         disable type2 matches.Hide results where object does not have a value for a query attribute (NOT DICOM conform)
111      * @param queryHasIssuerOfPID          Enable query if issuer of patient is set. Only effective if IssuerOfPatientID isn't a query attribute!
112      *                                  <dl><dd><code>null</code> disabled (default: contains</dd> 
113      *                                  <dd>TRUE...only objects of patients with issuer</dd>
114      *                                  <dd>FALSE..only objects of patients without issuer</dd></dl>
115      * @param subject                   Security subject to apply study permissions for current user. If <code>null</code> studyPermission check is disabled!
116      * 
117      * @throws SQLException
118      */
119     public QueryStudiesCmd(Dataset keys, boolean hideMissingStudies, boolean noMatchForNoValue, Boolean queryHasIssuerOfPID, Subject subject)
120     throws SQLException {
121         super(JdbcProperties.getInstance().getDataSource(),
122                 transactionIsolationLevel);
123         try {
124             AttributeFilter patAttrFilter = AttributeFilter.getPatientAttributeFilter();
125             AttributeFilter studyAttrFilter = AttributeFilter.getStudyAttributeFilter();
126             AttributeFilter seriesAttrFilter = AttributeFilter.getSeriesAttributeFilter();
127             checkPermissions = subject != null;
128             boolean type2 = noMatchForNoValue ? SqlBuilder.TYPE1 : SqlBuilder.TYPE2;
129             sqlBuilder.setFrom(getTables());
130             sqlBuilder.setLeftJoin( getLeftJoin(keys.containsValue(Tags.SeriesInstanceUID) || 
131                     keys.containsValue(Tags.RequestAttributesSeq)));
132             sqlBuilder.setRelations(getRelations());
133             sqlBuilder.addLiteralMatch(null, "Patient.merge_fk", false, "IS NULL");
134             sqlBuilder.addWildCardMatch(null, "Patient.patientId",
135                     type2,
136                     patAttrFilter.getStrings(keys, Tags.PatientID));
137             String issuer = patAttrFilter.getString(keys, Tags.IssuerOfPatientID);
138             if ( issuer != null ) {
139                 sqlBuilder.addSingleValueMatch(null, "Patient.issuerOfPatientId", type2, issuer);
140             } else if ( queryHasIssuerOfPID != null ) {
141                 sqlBuilder.addNULLValueMatch(null,"Patient.issuerOfPatientId", queryHasIssuerOfPID.booleanValue() );
142             }
143             sqlBuilder.addPNMatch(new String[] {
144                     "Patient.patientName",
145                     "Patient.patientIdeographicName",
146                     "Patient.patientPhoneticName"},
147                     type2,
148                     patAttrFilter.isICase(Tags.PatientName),
149                     keys.getString(Tags.PatientName));
150             sqlBuilder.addRangeMatch(null, "Patient.patientBirthDate", type2,
151                     keys.getString(Tags.PatientBirthDate));
152             
153             sqlBuilder.addWildCardMatch(null, "Study.studyId", type2,
154                     studyAttrFilter.getStrings(keys, Tags.StudyID));
155             if (keys.containsValue(Tags.RequestAttributesSeq) && 
156                     keys.getItem(Tags.RequestAttributesSeq).containsValue(Tags.StudyInstanceUID)) {
157                 Match.Node node0 = sqlBuilder.addNodeMatch("OR", false);
158                 node0.addMatch(new Match.ListOfString(null, "Study.studyIuid", SqlBuilder.TYPE1, keys.getStrings(Tags.StudyInstanceUID)));
159                 Dataset rqAttrs = keys.getItem(Tags.RequestAttributesSeq);
160         
161                 SqlBuilder subQuery = new SqlBuilder();
162                 subQuery.setSelect(new String[] { "SeriesRequest.pk" });
163                 subQuery.setFrom(new String[] { "SeriesRequest" });
164                 subQuery.addFieldValueMatch(null, "Series.pk", SqlBuilder.TYPE1, null,
165                         "SeriesRequest.series_fk");
166                 subQuery.addListOfUidMatch(null, "SeriesRequest.studyIuid", type2,
167                         rqAttrs.getStrings(Tags.StudyInstanceUID));
168                 node0.addMatch(new Match.Subquery(subQuery, null, null));
169             } else {
170                 sqlBuilder.addListOfStringMatch(null, "Study.studyIuid",
171                         SqlBuilder.TYPE1, keys.getStrings( Tags.StudyInstanceUID));
172             }
173             
174             sqlBuilder.addListOfStringMatch(null, "Series.seriesIuid",
175                     SqlBuilder.TYPE1, keys.getStrings( Tags.SeriesInstanceUID));
176             sqlBuilder.addRangeMatch(null, "Study.studyDateTime", type2,
177                     keys.getDateTimeRange(Tags.StudyDate, Tags.StudyTime));
178             sqlBuilder.addWildCardMatch(null, "Study.accessionNumber", type2,
179                     studyAttrFilter.getStrings(keys, Tags.AccessionNumber));
180             sqlBuilder.addModalitiesInStudyNestedMatch(null,
181                     seriesAttrFilter.getStrings(keys, Tags.ModalitiesInStudy, Tags.Modality));
182             keys.setPrivateCreatorID(PrivateTags.CreatorID);
183             sqlBuilder.addCallingAETsNestedMatch(false,
184                     keys.getStrings(PrivateTags.CallingAET));
185             keys.setPrivateCreatorID(null);
186             this.hideMissingStudies = hideMissingStudies;	
187             if ( this.hideMissingStudies ) {
188                 sqlBuilder.addNULLValueMatch(null,"Study.pk", true);
189             }
190             if ( checkPermissions ) {
191                 String[] roles = SecurityUtils.rolesOf(subject);
192                 if ( roles.length < 1 ) {
193                     throw new IllegalArgumentException("User is not in a StudyPermission relevant role");
194                 }
195                 addStudyPermissionMatch(subject);
196             }
197         } catch (RuntimeException x) {
198             close();
199             throw x;
200         }
201     }
202     
203     private void addStudyPermissionMatch(Subject subject) {
204         if (subject != null) {
205             sqlBuilder.addQueryPermissionNestedMatch(false,
206                     !hideMissingStudies, SecurityUtils.rolesOf(subject));
207         }
208     }

224     public int count() throws SQLException {
225         try {
226             sqlBuilder.setSelectCount(new String[]{"Study.pk"}, true);
227             execute( sqlBuilder.getSql() );
228             next();
229             if (hideMissingStudies) return rs.getInt(1);
230             //we have to add number of studies and number of patients without studies.
231             int studies = rs.getInt(1);
232             rs.close();
233             rs = null;
234             sqlBuilder.setSelectCount(new String[]{"Patient.pk"}, true);
235             sqlBuilder.addNULLValueMatch(null,"Study.pk", false);
236             execute( sqlBuilder.getSql() );
237             next();
238             int emptyPatients = rs.getInt(1);
239             List matches = sqlBuilder.getMatches();
240             matches.remove( matches.size() - 1);//removes the Study.pk NULLValue match!
241             return studies + emptyPatients;
242         } finally {
243             close();
244         }
245     }

248     public List list(int offset, int limit, boolean latestStudiesFirst) throws SQLException {
249         defineColumnTypes(new int[] { 
250                 blobAccessType, // Patient.encodedAttributes
251                 blobAccessType, // Study.encodedAttributes
252                 Types.BIGINT,   // Patient.pk
253                 Types.BIGINT,   // Study.pk
254                 Types.VARCHAR,  // Study.modalitiesInStudy
255                 Types.INTEGER,  // Study.numberOfStudyRelatedSeries
256                 Types.INTEGER,  // Study.numberOfStudyRelatedInstances
257                 Types.VARCHAR,  // Study.retrieveAETs
258                 Types.INTEGER,  // Study.availability
259                 Types.VARCHAR,  // Study.filesetId
260                 Types.VARCHAR,  // Study.studyStatusId
261         });
262         sqlBuilder.setSelect(SELECT_ATTRIBUTE);
263         sqlBuilder.addOrderBy("Patient.patientName", SqlBuilder.ASC);
264         sqlBuilder.addOrderBy("Patient.pk", SqlBuilder.ASC);
265         sqlBuilder.addOrderBy("Study.studyDateTime", 
266                 latestStudiesFirst ? SqlBuilder.DESC : SqlBuilder.ASC);
267         sqlBuilder.setOffset(offset);
268         sqlBuilder.setLimit(limit);
269         try {
270             setFetchSize(limit);
271             execute(sqlBuilder.getSql());
272             ArrayList result = new ArrayList();
273 
274             while (next()) {
275                 final byte[] patAttrs = rs.getBytes(1);
276                 final byte[] styAttrs = rs.getBytes(2);
277                 Dataset ds = dof.newDataset();
278                 ds.setPrivateCreatorID(PrivateTags.CreatorID);
279                 ds.putOB(PrivateTags.PatientPk, Convert.toBytes(rs.getLong(3)) );
280                 ds.setPrivateCreatorID(null);
281                 long studyPk = rs.getLong(4);
282                 DatasetUtils.fromByteArray(patAttrs, ds);
283                 if (styAttrs != null) {
284                     ds.setPrivateCreatorID(PrivateTags.CreatorID);
285                     ds.putOB(PrivateTags.StudyPk, Convert.toBytes(studyPk) );
286                     ds.setPrivateCreatorID(null);
287                     DatasetUtils.fromByteArray(styAttrs, ds);
288                     ds.putCS(Tags.ModalitiesInStudy, StringUtils.split(rs
289                             .getString(5), '\\'));
290                     ds.putIS(Tags.NumberOfStudyRelatedSeries, rs.getInt(6));
291                     ds.putIS(Tags.NumberOfStudyRelatedInstances, rs.getInt(7));
292                     ds.putAE(Tags.RetrieveAET, StringUtils.split(rs
293                             .getString(8), '\\'));
294                     ds.putCS(Tags.InstanceAvailability, Availability
295                             .toString(rs.getInt(9)));
296                     ds.putSH(Tags.StorageMediaFileSetID, rs.getString(10));
297                     ds.putCS(Tags.StudyStatusID, rs.getString(11) );
298                 }
299                 result.add(ds);
300             }
301             return result;
302         } finally {
303             close();
304         }
305     }
