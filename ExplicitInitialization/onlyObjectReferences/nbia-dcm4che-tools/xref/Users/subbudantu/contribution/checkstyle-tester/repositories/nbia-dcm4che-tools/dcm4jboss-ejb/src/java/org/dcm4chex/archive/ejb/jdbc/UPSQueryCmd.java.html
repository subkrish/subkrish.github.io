72      public UPSQueryCmd(Dataset keys, boolean fuzzyMatchingOfPN,
73              boolean noMatchForNoValue) throws SQLException, DcmServiceException {
74          super(keys, true, noMatchForNoValue, transactionIsolationLevel);
75          try {
76              AttributeFilter patAttrFilter = AttributeFilter.getPatientAttributeFilter();
77              defineColumnTypes(new int[] { blobAccessType, blobAccessType });
78              String s;
79              // ensure keys contains (8,0005) for use as result filter
80              if (!keys.contains(Tags.SpecificCharacterSet)) {
81                  keys.putCS(Tags.SpecificCharacterSet);
82              }
83              sqlBuilder.setSelect(SELECT);
84              sqlBuilder.setFrom(FROM);
85              sqlBuilder.setLeftJoin(getLeftJoin());
86              sqlBuilder.addListOfUidMatch(null, "UPS.sopInstanceUID",
87                      SqlBuilder.TYPE1,
88                      keys.getStrings(Tags.SOPInstanceUID));
89              if ((s = keys.getString(Tags.UPSState)) != null) {
90                  sqlBuilder.addIntValueMatch(null, "UPS.stateAsInt",
91                          SqlBuilder.TYPE1,
92                          UPSState.toInt(s));
93              }
94              s = keys.getString(Tags.SPSPriority);
95              if (s != null) {
96                  sqlBuilder.addIntValueMatch(null, "UPS.priorityAsInt",
97                          SqlBuilder.TYPE1,
98                          Priority.toInt(s));
99              }
100             sqlBuilder.addWildCardMatch(null, "UPS.procedureStepLabel",
101                     SqlBuilder.TYPE1,
102                     keys.getStrings(Tags.ProcedureStepLabel));
103             sqlBuilder.addWildCardMatch(null, "UPS.worklistLabel",
104                     SqlBuilder.TYPE1,
105                     keys.getStrings(Tags.WorklistLabel));
106             sqlBuilder.addRangeMatch(null, "UPS.scheduledStartDateTime",
107                     SqlBuilder.TYPE1,
108                     keys.getDateRange(Tags.SPSStartDateAndTime));
109             sqlBuilder.addRangeMatch(null, "UPS.expectedCompletionDateTime",
110                     type2,
111                     keys.getDateRange(Tags.ExpectedCompletionDateAndTime));
112             sqlBuilder.addCodeMatch(ITEM_CODE,
113                     keys.getItem(Tags.ScheduledWorkitemCodeSeq));
114             addNestedCodeMatch(Tags.ScheduledProcessingApplicationsCodeSeq,
115                     new String[]{ "UPS.pk", "rel_ups_appcode.ups_fk"},
116                     new String[]{ "rel_ups_appcode", "Code" },
117                     new String[]{ "rel_ups_appcode.appcode_fk", "Code.pk"});
118             addNestedCodeMatch(Tags.ScheduledStationNameCodeSeq,
119                     new String[]{ "UPS.pk", "rel_ups_devname.ups_fk"},
120                     new String[]{ "rel_ups_devname", "Code" },
121                     new String[]{ "rel_ups_devname.devname_fk", "Code.pk"});
122             addNestedCodeMatch(Tags.ScheduledStationClassCodeSeq,
123                     new String[]{ "UPS.pk", "rel_ups_devclass.ups_fk"},
124                     new String[]{ "rel_ups_devclass", "Code" },
125                     new String[]{ "rel_ups_devclass.devclass_fk", "Code.pk"});
126             addNestedCodeMatch(Tags.ScheduledStationGeographicLocationCodeSeq,
127                     new String[]{ "UPS.pk", "rel_ups_devloc.ups_fk"},
128                     new String[]{ "rel_ups_devloc", "Code" },
129                     new String[]{ "rel_ups_devloc.devloc_fk", "Code.pk"});
130             addNestedCodeMatch(Tags.ScheduledHumanPerformersSeq,
131                     Tags.HumanPerformerCodeSeq, 
132                     new String[] { "UPS.pk", "rel_ups_performer.ups_fk" },
133                     new String[] { "rel_ups_performer", "Code" },
134                     new String[] { "rel_ups_performer.performer_fk", "Code.pk" });
135             addRefRequestMatch();
136             sqlBuilder.addRefSOPMatch(
137                     new String[]{ "UPS.pk", "UPSRelatedPS.ups_fk"},
138                     "UPSRelatedPS", "UPSRelatedPS.refSOPClassUID", 
139                     "UPSRelatedPS.refSOPInstanceUID",
140                     keys.getItem(Tags.RelatedProcedureStepSeq), type2);
141     // TODO Tags.ReplacedProcedureStepSeq not yet defined
142     //        sqlBuilder.addRefSOPMatch(
143     //                new String[]{ "UPS.pk", "UPSReplacedPS.ups_fk"},
144     //                "UPSReplacedPS", "UPSReplacedPS.refSOPClassUID", 
145     //                "UPSReplacedPS.refSOPInstanceUID",
146     //                keys.getItem(Tags.Tags.ReplacedProcedureStepSeq), type2);
147     
148             sqlBuilder.addWildCardMatch(null, "UPS.admissionID",
149                     type2,
150                     keys.getStrings(Tags.AdmissionID));
151             Dataset issuer = keys.getItem(Tags.IssuerOfAdmissionIDSeq);
152             if (issuer != null) {
153                 sqlBuilder.addSingleValueMatch(null,
154                         "UPS.issuerOfAdmissionIDLocalNamespaceEntityID",
155                         type2,
156                         issuer.getString(Tags.LocalNamespaceEntityID));
157                 sqlBuilder.addSingleValueMatch(null,
158                         "UPS.issuerOfAdmissionIDUniversalEntityID",
159                         type2,
160                         issuer.getString(Tags.UniversalEntityID));
161             }
162             if (sqlBuilder.addWildCardMatch(null, "Patient.patientId",
163                     type2,
164                     patAttrFilter.getStrings(keys, Tags.PatientID)) != null)
165                 sqlBuilder.addSingleValueMatch(null, "Patient.issuerOfPatientId",
166                         type2,
167                         patAttrFilter.getString(keys, Tags.IssuerOfPatientID));
168             if (fuzzyMatchingOfPN)
169                 try {
170                     sqlBuilder.addPNFuzzyMatch(
171                             new String[] {
172                                 "Patient.patientFamilyNameSoundex",
173                                 "Patient.patientGivenNameSoundex" },
174                             type2,
175                             keys.getString(Tags.PatientName));
176                 } catch (IllegalArgumentException ex) {
177                     throw new DcmServiceException(
178                             Status.IdentifierDoesNotMatchSOPClass,
179                             ex.getMessage() + ": " + keys.get(Tags.PatientName));
180                 }
181             else
182                 sqlBuilder.addPNMatch(
183                         new String[] {
184                             "Patient.patientName",
185                             "Patient.patientIdeographicName",
186                             "Patient.patientPhoneticName"},
187                         type2,
188                         patAttrFilter.isICase(Tags.PatientName),
189                         keys.getString(Tags.PatientName));
190             sqlBuilder.addRangeMatch(null, "Patient.patientBirthDate", type2,
191                     keys.getString(Tags.PatientBirthDate));
192             sqlBuilder.addWildCardMatch(null, "Patient.patientSex", type2,
193                     patAttrFilter.getStrings(keys, Tags.PatientSex));
194         } catch (SQLException x) {
195             close();
196             throw x;
197         } catch (DcmServiceException x) {
198             close();
199             throw x;
200         } catch (RuntimeException x) {
201             close();
202             throw x;
203         }
204     }

206     private void addRefRequestMatch() {
207         Dataset item = keys.getItem(Tags.RefRequestSeq);
208         if (item == null || item.isEmpty())
209             return;
210         
211         SqlBuilder subQuery = new SqlBuilder();
212         subQuery.setSelect1();
213         String[] from = { "UPSRequest" };
214         subQuery.setFrom(from);
215         subQuery.addFieldValueMatch(null, "UPS.pk", false, null,
216                 "UPSRequest.ups_fk");
217         boolean universalMatch = subQuery.addWildCardMatch(null,
218                 "UPSRequest.requestedProcedureId", type2,
219                 item.getStrings(Tags.RequestedProcedureID)) == null;
220         universalMatch = subQuery.addWildCardMatch(null,
221                 "UPSRequest.accessionNumber", type2,
222                 item.getStrings(Tags.AccessionNumber)) == null
223                 && universalMatch;
224         universalMatch = subQuery.addWildCardMatch(null,
225                 "UPSRequest.confidentialityCode", type2,
226                 item.getStrings(Tags.ConfidentialityCode)) == null
227                 && universalMatch;
228         universalMatch = subQuery.addWildCardMatch(null,
229                 "UPSRequest.requestingService", type2,
230                 item.getStrings(Tags.RequestingService)) == null
231                 && universalMatch;
232 
233         if (universalMatch)
234             return;
235 
236         if (!type2) {
237             sqlBuilder.addCorrelatedSubquery(subQuery);
238         } else {
239             SqlBuilder subQuery2 = new SqlBuilder();
240             subQuery2.setSelect1();
241             subQuery2.setFrom(from);
242             subQuery2.addFieldValueMatch(null, "UPS.pk", false, null, 
243                     "UPSRequest.ups_fk");
244             Match match2 = new Match.Subquery(subQuery2, null, null);
245             Node notNode = new  Match.Node(null, true);
246             notNode.addMatch(match2);
247             Node orMatch = sqlBuilder.addNodeMatch(" OR", false);
248             orMatch.addMatch(new Match.Subquery(subQuery, null, null));
249             orMatch.addMatch(notNode);
250         }
251     }

253     private void addNestedCodeMatch(int tag, String[] parentRelation,
254             String[] tables, String[] relations) {
255         addNestedCodeMatch(keys.getItem(tag), parentRelation, tables, relations);
256     }

258     private void addNestedCodeMatch(int tag1, int tag2, String[] parentRelation,
259             String[] tables, String[] relations) {
260         Dataset item = keys.getItem(tag1);
261         if (item != null)
262             addNestedCodeMatch(item.getItem(tag2), parentRelation, tables, relations);
263     }
264 
265     private void addNestedCodeMatch(Dataset item, String[] parentRelation,
266             String[] tables, String[] relations) {
267         sqlBuilder.addNestedCodeMatch(parentRelation, tables, relations, item, 
268                 type2);
269     }

271     private String[] getLeftJoin() {
272         boolean workitem;
273         int index = 4;
274         if (workitem = SqlBuilder.isCodeMatch(
275                 keys.getItem(Tags.ScheduledWorkitemCodeSeq)))
276             index += 4;
277         String[] leftJoin = new String[index];
278         leftJoin[0] = "Patient";
279         leftJoin[1] = null;
280         leftJoin[2] = "UPS.patient_fk";
281         leftJoin[3] = "Patient.pk";
282         index = 4;
283         if (workitem) {
284             leftJoin[index++] = "Code";
285             leftJoin[index++] = ITEM_CODE;
286             leftJoin[index++] = "UPS.code_fk";
287             leftJoin[index++] = "Code.pk";
288         }
289         return leftJoin;
290     }

293         Dataset ds = DcmObjectFactory.getInstance().newDataset();
294         DatasetUtils.fromByteArray( rs.getBytes(1), ds);
295         DatasetUtils.fromByteArray( rs.getBytes(2), ds);
296         adjustDataset(ds, keys);
297         return ds.subSet(keys);
