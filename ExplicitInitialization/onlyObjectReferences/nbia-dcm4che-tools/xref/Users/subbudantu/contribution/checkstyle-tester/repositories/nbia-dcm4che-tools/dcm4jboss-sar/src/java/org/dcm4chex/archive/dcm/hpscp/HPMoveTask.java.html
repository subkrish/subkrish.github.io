71  class HPMoveTask implements Runnable {
72  
73      private static final int STORE_PCID = 1;
74  
75      private static final String[] NATIVE_LE_TS = { UIDs.ExplicitVRLittleEndian,
76              UIDs.ImplicitVRLittleEndian, };
77  
78      private final HPScpService service;
79  
80      private final List hpList;

100     private int priority;

102     private int msgID;
103 
104     private int failed = 0;
105 
106     private int warnings = 0;
107 
108     private int completed = 0;
109 
110     private int remaining = 0;
111 
112     private boolean canceled = false;

114     private ActiveAssociation storeAssoc;
115 
116     private final ArrayList failedIUIDs = new ArrayList();
117 
118     public HPMoveTask(HPScpService service, ActiveAssociation moveAssoc,
119             int movePcid, Command moveRqCmd, Dataset moveRqData, List hpList,
120             AEDTO aeData, String moveDest) throws DcmServiceException {
121         this.service = service;
122         this.hpList = hpList;
123         this.log = service.getLog();
124         this.moveAssoc = moveAssoc;
125         this.movePcid = movePcid;
126         this.moveRqCmd = moveRqCmd;
127         this.moveRqData = moveRqData;
128         this.aeData = aeData;
129         this.moveDest = moveDest;
130         this.moveOriginatorAET = moveAssoc.getAssociation().getCallingAET();
131         this.moveCalledAET = moveAssoc.getAssociation().getCalledAET();
132         this.priority = moveRqCmd.getInt(Tags.Priority, Command.MEDIUM);
133         this.msgID = moveRqCmd.getMessageID();
134         if (hpList.isEmpty())
135             return;
136         openAssociation();
137         moveAssoc.addCancelListener(msgID, new DimseListener() {
138             public void dimseReceived(Association assoc, Dimse dimse) {
139                 canceled = true;
140             }
141         });
142     }
143 
144     private void openAssociation() throws DcmServiceException {
145         PDU ac = null;
146         Association a = null;
147         AssociationFactory asf = AssociationFactory.getInstance();
148         try {
149             a = asf.newRequestor(service.createSocket(moveCalledAET, aeData));
150             a.setAcTimeout(service.getAcTimeout());
151             AAssociateRQ rq = asf.newAAssociateRQ();
152             rq.setCalledAET(moveDest);
153             rq.setCallingAET(moveCalledAET);
154             rq.addPresContext(asf.newPresContext(STORE_PCID,
155                     UIDs.HangingProtocolStorage, NATIVE_LE_TS));
156             ac = a.connect(rq);
157         } catch (Exception e) {
158             final String prompt = "Failed to connect " + moveDest;
159             log.error(prompt, e);
160             throw new DcmServiceException(Status.UnableToPerformSuboperations,
161                     prompt, e);
162         }
163         if (!(ac instanceof AAssociateAC)) {
164             final String prompt = "Association not accepted by " + moveDest
165                     + ":\n" + ac;
166             log.error(prompt);
167             throw new DcmServiceException(Status.UnableToPerformSuboperations,
168                     prompt);
169         }
170         storeAssoc = asf.newActiveAssociation(a, null);
171         storeAssoc.start();
172         if (a.listAcceptedPresContext(UIDs.HangingProtocolStorage).isEmpty()) {
173             final String prompt = "No Presentation Context for Hanging Protocol Storage not negotiated by "
174                     + moveDest;
175             log.error(prompt);
176             throw new DcmServiceException(Status.UnableToPerformSuboperations,
177                     prompt);
178 
179         }
180     }
181 
182     public void run() {
183         Association a = storeAssoc.getAssociation();
184         for (int i = 0, n = hpList.size(); i < n
185                 && a.getState() == Association.ASSOCIATION_ESTABLISHED
186                 && !canceled; ++i) {
187             Dataset hp = (Dataset) hpList.get(i);
188             final String iuid = hp.getString(Tags.SOPInstanceUID);
189             DimseListener storeScpListener = new DimseListener() {
190 
191                 public void dimseReceived(Association assoc, Dimse dimse) {
192                     switch (dimse.getCommand().getStatus()) {
193                     case Status.Success:
194                         ++completed;
195                         break;
196                     case Status.CoercionOfDataElements:
197                     case Status.DataSetDoesNotMatchSOPClassWarning:
198                     case Status.ElementsDiscarded:
199                         ++warnings;
200                         break;
201                     default:
202                         ++failed;
203                         failedIUIDs.add(iuid);
204                         break;
205                     }
206                     if (--remaining > 0) {
207                         notifyMovePending();
208                     }
209                 }
210             };
211             try {
212                 storeAssoc.invoke(makeCStoreRQ(hp), storeScpListener);
213             } catch (Exception e) {
214                 log.error("Exception during move of " + iuid, e);
215             }
216         }
217         if (a.getState() == Association.ASSOCIATION_ESTABLISHED) {
218             try {
219                 storeAssoc.release(true);
220                 // workaround to ensure that last STORE-RSP is processed before
221                 // finally MOVE-RSP is sent
222                 Thread.sleep(10);
223             } catch (Exception e) {
224                 log.error("Exception during release:", e);
225             }
226         } else {
227             try {
228                 a.abort(AssociationFactory.getInstance().newAAbort(
229                         AAbort.SERVICE_PROVIDER, AAbort.REASON_NOT_SPECIFIED));
230             } catch (IOException ignore) {
231             }
232         }
233         if (!canceled) {
234             for (int i = completed + warnings + failed, n = hpList.size(); i < n; ++i) {
235                 Dataset hp = (Dataset) hpList.get(i);
236                 failedIUIDs.add(hp.getString(Tags.SOPInstanceUID));
237             }
238             remaining = 0;
239             failed = failedIUIDs.size();
240         }
241         notifyMoveFinished();
242     }
243 
244     private Dimse makeCStoreRQ(Dataset hp) throws NoPresContextException {
245         Association assoc = storeAssoc.getAssociation();
246         Command storeRqCmd = DcmObjectFactory.getInstance().newCommand();
247         storeRqCmd.initCStoreRQ(assoc.nextMsgID(), hp
248                 .getString(Tags.SOPClassUID),
249                 hp.getString(Tags.SOPInstanceUID), priority);
250         storeRqCmd.putUS(Tags.MoveOriginatorMessageID, msgID);
251         storeRqCmd.putAE(Tags.MoveOriginatorAET, moveOriginatorAET);
252         return AssociationFactory.getInstance().newDimse(STORE_PCID,
253                 storeRqCmd, hp);
254     }
255 
256     private void notifyMovePending() {
257         if (service.isSendPendingMoveRSP()) {
258             notifyMoveSCU(Status.Pending, null);
259         }
260     }
261 
262     private void notifyMoveFinished() {
263         if (moveAssoc != null)
264             moveAssoc.removeCancelListener(msgID);
265         notifyMoveSCU(canceled ? Status.Cancel
266                 : failed > 0 ? Status.SubOpsOneOrMoreFailures : Status.Success,
267                 makeMoveRspIdentifier());
268     }
269 
270     private void notifyMoveSCU(int status, Dataset ds) {
271         if (moveAssoc == null)
272             return;
273         try {
274             moveAssoc.getAssociation().write(
275                     AssociationFactory.getInstance().newDimse(movePcid,
276                             makeMoveRsp(status), ds));
277         } catch (Exception e) {
278             log.info("Failed to send Move RSP to Move Originator:", e);
279             moveAssoc = null;
280         }
281     }
282 
283     private Dataset makeMoveRspIdentifier() {
284         if (failed == 0)
285             return null;
286         Dataset ds = DcmObjectFactory.getInstance().newDataset();
287         if (failed == failedIUIDs.size()) {
288             String[] a = (String[]) failedIUIDs.toArray(new String[failedIUIDs
289                     .size()]);
290             ds.putUI(Tags.FailedSOPInstanceUIDList, a);
291             // check if 64k limit for UI attribute is reached
292             if (ds.get(Tags.FailedSOPInstanceUIDList).length() < 0x10000)
293                 return ds;
294             log.warn("Failed SOP InstanceUID List exceeds 64KB limit "
295                     + "- send empty attribute instead");
296         }
297         ds.putUI(Tags.FailedSOPInstanceUIDList);
298         return ds;
299     }
300 
301     private Command makeMoveRsp(int status) {
302         Command rspCmd = DcmObjectFactory.getInstance().newCommand();
303         rspCmd.initCMoveRSP(moveRqCmd.getMessageID(), moveRqCmd
304                 .getAffectedSOPClassUID(), status);
305         if (remaining > 0) {
306             rspCmd.putUS(Tags.NumberOfRemainingSubOperations, remaining);
307         } else {
308             rspCmd.remove(Tags.NumberOfRemainingSubOperations);
309         }
310         rspCmd.putUS(Tags.NumberOfCompletedSubOperations, completed);
311         rspCmd.putUS(Tags.NumberOfWarningSubOperations, warnings);
312         rspCmd.putUS(Tags.NumberOfFailedSubOperations, failed);
313         return rspCmd;
314     }
