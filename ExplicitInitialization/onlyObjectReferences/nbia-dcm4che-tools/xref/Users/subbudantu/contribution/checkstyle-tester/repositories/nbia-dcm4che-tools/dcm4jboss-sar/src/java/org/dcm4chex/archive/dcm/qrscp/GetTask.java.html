102     private DimseListener cancelListener = new DimseListener() {
103         public void dimseReceived(Association assoc, Dimse dimse) {
104             canceled = true;
105             MoveScu tmp = moveScu;
106             if (tmp != null) {
107                 tmp.forwardCancelRQ(dimse);
108             }
109         }
110     };

112     private TimerTask sendPendingRsp = new TimerTask() {
113         public void run() {
114             if (!canceled && !remainingIUIDs.isEmpty()) {
115                 sendGetRsp(Status.Pending, null);
116             }
117         }};

119     public GetTask(QueryRetrieveScpService service, ActiveAssociation assoc,
120             int pcid, Command rqCmd, Dataset rqData, FileInfo[][] fileInfos) {
121         this.service = service;
122         this.log = service.getLog();
123         this.assoc = assoc;
124         this.callingAET = assoc.getAssociation().getCallingAET();
125         this.calledAET = assoc.getAssociation().getCalledAET();
126         this.pcid = pcid;
127         this.rqData = rqData;
128         this.sopClassUID = rqCmd.getAffectedSOPClassUID();
129         this.priority = rqCmd.getInt(Tags.Priority, Command.MEDIUM);
130         this.msgID = rqCmd.getMessageID();
131         this.total = fileInfos.length;
132         this.retrieveInfo = new RetrieveInfo(service, fileInfos);
133         this.remainingIUIDs = retrieveInfo.getAvailableIUIDs();
134         this.failedIUIDs = retrieveInfo.getNotAvailableIUIDs();
135         assoc.addCancelListener(msgID, cancelListener);
136         this.pixQueryResults = service.isAdjustPatientIDOnRetrieval()
137                 ? new HashMap<PIDWithIssuer, Set<PIDWithIssuer>>()
138                 : null;
139     }

141     private void removeInstancesOfUnsupportedStorageSOPClasses() {
142         Association a = assoc.getAssociation();
143         Iterator<String> it = retrieveInfo.getCUIDs();
144         String cuid;
145         Collection<String> iuids;
146         while (it.hasNext()) {
147             cuid = it.next();
148             if (a.listAcceptedPresContext(cuid).isEmpty()
149                     && a.getAcceptedPresContext(cuid, UIDs.NoPixelData) == null
150                     && a.getAcceptedPresContext(cuid, UIDs.NoPixelDataDeflate) == null) {
151                 iuids = retrieveInfo.removeInstancesOfClass(cuid);
152                 it.remove(); // Use Iterator itself to remove the current
153                                 // item to avoid ConcurrentModificationException
154                 final String prompt = "No Presentation Context for "
155                         + QueryRetrieveScpService.uidDict.toString(cuid)
156                         + " offered by " + callingAET
157                         + "\n\tCannot send " + iuids.size()
158                         + " instances of this class";
159                 noPresentationContext(cuid, iuids, prompt);
160             } else {
161                 Set<String> tsuids = new HashSet<String>(
162                         service.notDecompressTsuidSet());
163                 tsuids.retainAll(retrieveInfo.getTransferSyntaxesOfClass(cuid));
164                 for (String tsuid : tsuids) {
165                     if (a.getAcceptedPresContext(cuid, tsuid) == null) {
166                         iuids = retrieveInfo.removeLocalFilesOfClassWithTransferSyntax(cuid, tsuid);
167                         if (!iuids.isEmpty()) {
168                             noPresentationContext(cuid, iuids, "No Presentation Context for "
169                                     + QueryRetrieveScpService.uidDict.toString(cuid)
170                                     + " with " + QueryRetrieveScpService.uidDict.toString(tsuid)
171                                     + " accepted by " + callingAET
172                                     + "\n\tCannot send " + iuids.size()
173                                     + " instances of this class");
174                         }
175                     }
176                 }
177             }
178         }
179     }
180 
181     private void noPresentationContext(String cuid, Collection<String> iuids,
182             final String prompt) {
183         if (!service.isIgnorableSOPClass(cuid, callingAET)) {
184             failedIUIDs.addAll(iuids);
185             log.warn(prompt);
186         } else {
187             completed += iuids.size();
188             log.info(prompt);
189         }
190         remainingIUIDs.removeAll(iuids);
191     }

193     public void run() {
194         Set<String> localUIDs = retrieveInfo.removeLocalIUIDs();
195         boolean updateLocalUIDs = false;
196         if (retrieveInfo.nextMoveForward()) {
197             service.scheduleSendPendingCGetRsp(sendPendingRsp);
198             try {
199                 do {
200                     String retrieveAET = retrieveInfo.getMoveForwardAET();
201                     Collection<String> iuids = retrieveInfo.getMoveForwardUIDs();
202                     moveScu = new MoveScu(service, calledAET, calledAET,
203                             retrieveAET, 0);
204                     if (iuids.size() == total) {
205                         MoveScu.addStudySeriesIUIDs(rqData,
206                                 retrieveInfo.getStudyIUIDs(),
207                                 retrieveInfo.getSeriesIUIDs());
208                         moveScu.forwardMoveRQ(pcid, msgID, priority,
209                                 service.getLocalStorageAET(), rqData, iuids);
210                     } else {
211                         moveScu.splitAndForwardMoveRQ(pcid, msgID, priority,
212                                 service.getLocalStorageAET(),
213                                 retrieveInfo.getStudyIUIDs(),
214                                 retrieveInfo.getSeriesIUIDs(), iuids);
215                     }
216                     if (moveScu.completed() > 0 || moveScu.warnings() > 0) {
217                         updateLocalUIDs = true;
218                     }
219                     moveScu = null;
220                 } while (!canceled && retrieveInfo.nextMoveForward());
221             } finally {
222                 sendPendingRsp.cancel();
223             }
224         }
225         if (!canceled) {
226             try {
227                 if (updateLocalUIDs) {
228                     RetrieveCmd cmd = RetrieveCmd.create(rqData);
229                     cmd.setFetchSize(service.getFetchSize());
230                     FileInfo[][] fileInfo = cmd.getFileInfos();
231                     retrieveInfo = new RetrieveInfo(service, fileInfo);
232                     localUIDs = retrieveInfo.removeLocalIUIDs();
233                 }
234                 if (!localUIDs.isEmpty()) {
235                     service.prefetchTars(retrieveInfo.getLocalFiles());
236                     removeInstancesOfUnsupportedStorageSOPClasses();
237                     retrieveInstances();
238                 }
239             } catch (Exception e) {
240                 log.error(e.getMessage(), e);
241             }
242         }
243 
244         if (!canceled) {
245             failedIUIDs.addAll(remainingIUIDs);
246             remainingIUIDs.clear();
247         }
248         assoc.removeCancelListener(msgID);
249         sendGetRsp(status(), service.makeRetrieveRspIdentifier(failedIUIDs));
250     }

252     private void retrieveInstances() {
253         Set<StudyInstanceUIDAndDirPath> studyInfos =
254             new HashSet<StudyInstanceUIDAndDirPath>();
255         Association a = assoc.getAssociation();
256         Collection<List<FileInfo>> localFiles = retrieveInfo.getLocalFiles();
257         for (List<FileInfo> list : localFiles) {
258             if (service.isSendPendingCGetRSP()) {
259                 sendGetRsp(Status.Pending, null);
260             }
261             final FileInfo fileInfo = list.get(0);
262             final String iuid = fileInfo.sopIUID;
263             DimseListener storeScpListener = new DimseListener() {
264 
265                 public void dimseReceived(Association assoc, Dimse dimse) {
266                     switch (dimse.getCommand().getStatus()) {
267                     case Status.Success:
268                         ++completed;
269                         transferred.add(fileInfo);
270                         break;
271                     case Status.CoercionOfDataElements:
272                     case Status.DataSetDoesNotMatchSOPClassWarning:
273                     case Status.ElementsDiscarded:
274                         ++warnings;
275                         transferred.add(fileInfo);
276                         break;
277                     default:
278                         failedIUIDs.add(iuid);
279                     break;
280                     }
281                     remainingIUIDs.remove(iuid);
282                 }
283             };
284             Dimse rq;
285             try {
286                 rq = service.makeCStoreRQ(assoc, fileInfo, null, priority, 
287                         null, 0, null, pixQueryResults);
288             } catch (NoPresContextException e) {
289                 if (!service.isIgnorableSOPClass(fileInfo.sopCUID, 
290                         a.getCallingAET())) {
291                     failedIUIDs.add(fileInfo.sopIUID);
292                     remainingIUIDs.remove(iuid);
293                     log.warn(e.getMessage());
294                 } else {
295                     log.info(e.getMessage());
296                 }
297                 continue;
298             } catch (Exception e) {
299                 log.error(e.getMessage(), e);
300                 failedIUIDs.add(iuid);
301                 remainingIUIDs.remove(iuid);
302                 continue;
303             }
304 
305             try {
306                 assoc.invoke(rq, storeScpListener);
307             } catch (Exception e) {
308                 log.error("Exception during retrieve of " + iuid, e);
309             }
310             // track access on ONLINE FS
311             if (fileInfo.availability == Availability.ONLINE) {
312                 studyInfos.add(new StudyInstanceUIDAndDirPath(fileInfo));
313             }
314             if (canceled 
315                     || a.getState() != Association.ASSOCIATION_ESTABLISHED) {
316                 break;
317             }
318         }
319         if (a.getState() == Association.ASSOCIATION_ESTABLISHED) {
320             try {
321                 assoc.waitForPendingRSP();
322             } catch (InterruptedException e) {
323                 log.warn("Exception during wait for pending C-STORE RSP:", e);
324             }
325         }
326         if (!transferred.isEmpty()) {
327             service.logInstancesSent(a, a, transferred);
328         }
329         service.updateStudyAccessTime(studyInfos);
330     }
331 
332     private int status() {
333         return canceled ? Status.Cancel 
334                 : failedIUIDs.isEmpty() ? Status.Success
335                 : completed == 0 && warnings == 0 
336                         ? Status.UnableToPerformSuboperations
337                         : Status.SubOpsOneOrMoreFailures;
338     }
339 
340     private void sendGetRsp(int status, Dataset ds) {
341         Command rspCmd = DcmObjectFactory.getInstance().newCommand();
342         rspCmd.initCGetRSP(msgID, sopClassUID, status);
343         if (!remainingIUIDs.isEmpty()) {
344             rspCmd.putUS(Tags.NumberOfRemainingSubOperations,
345                     remainingIUIDs.size());
346         } else {
347             rspCmd.remove(Tags.NumberOfRemainingSubOperations);
348         }
349         rspCmd.putUS(Tags.NumberOfCompletedSubOperations, completed);
350         rspCmd.putUS(Tags.NumberOfWarningSubOperations, warnings);
351         rspCmd.putUS(Tags.NumberOfFailedSubOperations, failedIUIDs.size());
352         try {
353             assoc.getAssociation().write(
354                     AssociationFactory.getInstance().newDimse(pcid, rspCmd, ds));
355         } catch (Exception e) {
356             log.info("Failed to send C-GET RSP to " + callingAET, e);
357         }
358     }
