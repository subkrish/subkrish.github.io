150     public final void setConcurrency(int concurrency) throws Exception {
151         if (concurrency <= 0)
152             throw new IllegalArgumentException("Concurrency: " + concurrency);
153         if (this.concurrency != concurrency) {
154             final boolean restart = getState() == STARTED;
155             if (restart)
156                 stop();
157             this.concurrency = concurrency;
158             if (restart)
159                 start();
160         }
161     }

163     public String getRequestStgCmtFromAETs() {
164         if (requestStgCmtFromAETs.isEmpty())
165             return NONE;
166 
167         StringBuilder sb = new StringBuilder();
168         for (Map.Entry<String,String> entry : requestStgCmtFromAETs.entrySet()) {
169             final String key = entry.getKey();
170             final String value = entry.getValue();
171             sb.append(key);
172             if (!key.equals(value))
173                 sb.append(':').append(value);
174             sb.append('\\');
175         }
176         sb.setLength(sb.length() - 1);
177         return sb.toString();
178     }

180     public void setRequestStgCmtFromAETs(String aets) {
181         requestStgCmtFromAETs.clear();
182         if (aets != null && aets.length() > 0 && !aets.equalsIgnoreCase(NONE)) {
183             String[] a = StringUtils.split(aets, '\\');
184             String s;
185             int c;
186             for (int i = 0; i < a.length; i++) {
187                 s = a[i];
188                 c = s.indexOf(':');
189                 if (c == -1)
190                     requestStgCmtFromAETs.put(s, s);
191                 else if (c > 0 && c < s.length() - 1)
192                     requestStgCmtFromAETs.put(s.substring(0, c), s
193                             .substring(c + 1));
194             }
195         }
196     }

373     private int scheduleStgCmtForRetrRq(Dataset retrRq, String called, String calling) throws Exception {
374         if (called == null || called.trim().length() == 0)
375             throw new IllegalArgumentException("Missing CalledAET");
376         RetrieveCmd cmd = RetrieveCmd.create(retrRq);
377         cmd.setFetchSize(fetchSize);
378         FileInfo[][] fileInfos =cmd.getFileInfos();
379         if (fileInfos.length > 0) {
380             Dataset actionInfo = DcmObjectFactory.getInstance().newDataset();
381             DcmElement refSOPSeq = actionInfo.putSQ(Tags.RefSOPSeq);
382             for (FileInfo[] fileInfo : fileInfos) {
383                 Dataset item = refSOPSeq.addNewItem();
384                 item.putUI(Tags.RefSOPClassUID, fileInfo[0].sopCUID);
385                 item.putUI(Tags.RefSOPInstanceUID, fileInfo[0].sopIUID);
386                 
387             }
388             queueStgCmtOrder(calling, called, actionInfo, false);
389         }
390         return fileInfos.length;
391     }

436     public void onMessage(Message message) {
437         ObjectMessage om = (ObjectMessage) message;
438         try {
439             StgCmtOrder order = (StgCmtOrder) om.getObject();
440             log.info("Start processing " + order);
441             try {
442                 process(order);
443                 log.info("Finished processing " + order);
444             } catch (Exception e) {
445                 order.setThrowable(e);
446                 final int failureCount = order.getFailureCount() + 1;
447                 order.setFailureCount(failureCount);
448                 final RetryIntervalls retryIntervalls = order.isScpRole() ? scpRetryIntervalls
449                         : scuRetryIntervalls;
450                 final long delay = retryIntervalls.getIntervall(failureCount);
451                 if (delay == -1L) {
452                     log.error("Give up to process " + order, e);
453                     jmsDelegate.fail(queueName, order);
454                 } else {
455                     log.warn("Failed to process " + order
456                             + ". Scheduling retry.", e);
457                     jmsDelegate.queue(queueName, order, 0, System
458                             .currentTimeMillis()
459                             + delay);
460                 }
461             }
462         } catch (JMSException e) {
463             log.error("jms error during processing message: " + message, e);
464         } catch (Throwable e) {
465             log.error("unexpected error during processing message: " + message,
466                     e);
467         }

471     private Dataset commit(StgCmtOrder order) {
472         int failureReason = Status.ProcessingFailure;
473         Storage storage = null;
474         try {
475             StorageHome home = (StorageHome) EJBHomeFactory.getFactory()
476                     .lookup(StorageHome.class, StorageHome.JNDI_NAME);
477             storage = home.create();
478         } catch (Exception e) {
479             log.error("Failed to access Storage EJB", e);
480         }
481         Dataset actionInfo = order.getActionInfo();
482         log.trace("StgCmt actionInfo:");log.trace(actionInfo);
483         DcmElement refSOPSeq = actionInfo.get(Tags.RefSOPSeq);
484         Map fileInfos = null;
485         if (storage != null) {
486             try {
487                 FileInfo[][] aa = ((RetrieveCmd)RetrieveCmd.create(refSOPSeq).setFetchSize(fetchSize)).getFileInfos();
488                 fileInfos = new HashMap();
489                 for (int i = 0; i < aa.length; i++) {
490                     fileInfos.put(aa[i][0].sopIUID, aa[i]);
491                 }
492             } catch (SQLException e) {
493                 log.error("Failed to query DB", e);
494             }
495         }
496         Dataset eventInfo = DcmObjectFactory.getInstance().newDataset();
497         eventInfo.putUI(Tags.TransactionUID, actionInfo
498                 .getString(Tags.TransactionUID));
499         DcmElement successSOPSeq = eventInfo.putSQ(Tags.RefSOPSeq);
500         DcmElement failedSOPSeq = eventInfo.putSQ(Tags.FailedSOPSeq);
501         for (int i = 0, n = refSOPSeq.countItems(); i < n; ++i) {
502             Dataset refSOP = refSOPSeq.getItem(i);
503             if (storage != null
504                     && fileInfos != null
505                     && (failureReason = commit(storage, refSOP, fileInfos)) == Status.Success) {
506                 successSOPSeq.addItem(refSOP);
507             } else {
508                 refSOP.putUS(Tags.FailureReason, failureReason);
509                 failedSOPSeq.addItem(refSOP);
510             }
511         }
512         if (failedSOPSeq.isEmpty()) {
513             eventInfo.remove(Tags.FailedSOPSeq);
514         }
515         return eventInfo;
516     }

518     private void process(StgCmtOrder order) throws Exception {
519         AEManager aeMgr = aeMgr();
520         String aet = order.getCalledAET();
521         String callingAET = getCallingAET(order);
522         AEDTO callingAE = aeMgr.findByAET(callingAET);
523         AEDTO remoteAE = aeMgr.findByAET(aet);
524         Dataset ds = order.isScpRole() ? commit(order) : order.getActionInfo();
525         AssociationFactory af = AssociationFactory.getInstance();
526         Association a = af.newRequestor(tlsConfig.createSocket(callingAE, remoteAE));
527         a.setAcTimeout(acTimeout);
528         a.setDimseTimeout(dimseTimeout);
529         a.setSoCloseDelay(soCloseDelay);
530         AAssociateRQ rq = af.newAAssociateRQ();
531         rq.setCalledAET(aet);
532         rq.setCallingAET(callingAET);
533         rq.addPresContext(af.newPresContext(PCID_STGCMT,
534                         UIDs.StorageCommitmentPushModel,
535                         valuesToStringArray(tsuidMap)));
536         if (order.isScpRole()) {
537             rq.addRoleSelection(af.newRoleSelection(
538                     UIDs.StorageCommitmentPushModel, false, true));
539         }
540         PDU ac = a.connect(rq);
541         if (!(ac instanceof AAssociateAC)) {
542             throw new DcmServiceException(ERR_ASSOC_RJ,
543                     "Association not accepted by " + aet + ": " + ac);
544         }
545         ActiveAssociation aa = af.newActiveAssociation(a, null);
546         aa.start();
547         try {
548             if (a.getAcceptedTransferSyntaxUID(PCID_STGCMT) == null)
549                 throw new DcmServiceException(ERR_STGCMT_RJ,
550                         "StgCmt not supported by remote AE: " + aet);
551             Command cmd = DcmObjectFactory.getInstance().newCommand();
552             if (order.isScpRole()) {
553                 RoleSelection rs = ((AAssociateAC) ac)
554                         .getRoleSelection(UIDs.StorageCommitmentPushModel);
555                 if (rs == null || !rs.scp()) {
556                     log.warn("SCU Role of Storage Commitment Service rejected by "
557                                     + aet + " - try to send N_EVENT_REPORT anyway");
558                 }
559                 cmd.initNEventReportRQ(1, UIDs.StorageCommitmentPushModel,
560                         UIDs.StorageCommitmentPushModelSOPInstance,
561                         ds.contains(Tags.FailedSOPSeq) ? 2 : 1);
562                 invokeDimse(aa, cmd, ds, "StgCmt Result:\n");
563             } else {
564                 cmd.initNActionRQ(MSG_ID, UIDs.StorageCommitmentPushModel,
565                         UIDs.StorageCommitmentPushModelSOPInstance, 1);
566                 ds.putUI(Tags.TransactionUID,
567                         UIDGenerator.getInstance().createUID());
568                 invokeDimse(aa, cmd, ds, "StgCmt Request:\n");
569                 Thread.sleep(receiveResultInSameAssocTimeout);
570             }
571         } finally {
572             try {
573                 aa.release(true);
574             } catch (Exception e) {
575                 log.warn(
576                         "Failed to release association " + aa.getAssociation(),
577                         e);
578             }
579         }
580     }

587     private void invokeDimse(ActiveAssociation aa, Command cmd, Dataset ds,
588             String prompt) throws InterruptedException, IOException,
589             DcmServiceException {
590         log.debug(prompt);
591         log.debug(ds);
592         final AssociationFactory af = AssociationFactory.getInstance();
593         Dimse rsp = aa.invoke(af.newDimse(PCID_STGCMT, cmd, ds)).get();
594         final Command cmdRsp = rsp.getCommand();
595         int status = cmdRsp.getStatus();
596         if (status != 0)
597             throw new DcmServiceException(status, cmdRsp
598                     .getString(Tags.ErrorComment));
599     }

601     private int commit(Storage storage, Dataset refSOP, Map fileInfos) {
602         final String iuid = refSOP.getString(Tags.RefSOPInstanceUID);
603         final String cuid = refSOP.getString(Tags.RefSOPClassUID);
604         FileInfo[] fileInfo = (FileInfo[]) fileInfos.get(iuid);
605         if (fileInfo == null) {
606             log.warn("Failed Storage Commitment of Instance[uid=" + iuid
607                     + "]: no such object");
608             return Status.NoSuchObjectInstance;
609         }
610         if (!fileInfo[0].sopCUID.equals(cuid)) {
611             log.warn("Failed Storage Commitment of Instance[uid=" + iuid
612                     + "]: SOP Class in request[" + cuid
613                     + "] does not match SOP Class in stored object["
614                     + fileInfo[0].sopCUID + "]");
615             return Status.ClassInstanceConflict;
616         }
617         try {
618             LinkedHashSet retrieveAETs = new LinkedHashSet();
619             for (int i = 0; i < fileInfo.length; i++) {
620                 retrieveAETs.add(fileInfo[i].fileRetrieveAET);
621                 checkFile(fileInfo[i]);
622             }
623             storage.commit(iuid);
624             retrieveAETs.add(fileInfo[0].extRetrieveAET);
625             retrieveAETs.remove(null);
626             if (!retrieveAETs.isEmpty())
627                 refSOP.putAE(Tags.RetrieveAET, (String[]) retrieveAETs
628                         .toArray(new String[retrieveAETs.size()]));
629             return Status.Success;
630         } catch (Exception e) {
631             log.error("Failed Storage Commitment of Instance[uid="
632                     + fileInfo[0].sopIUID + "]:", e);
633             return Status.ProcessingFailure;
634         }
635     }

638     	if (isLocalRetrieveAET(info.fileRetrieveAET)) {
639     		FileUtils.verifyMD5(info);

643     void commited(Dataset stgcmtResult) throws DcmServiceException {
644         Storage storage = null;
645         try {
646             StorageHome home = (StorageHome) EJBHomeFactory.getFactory()
647                     .lookup(StorageHome.class, StorageHome.JNDI_NAME);
648             storage = home.create();
649             storage.commited(stgcmtResult);
650         } catch (Exception e) {
651             log.error("Failed update External AETs in DB records", e);
652             throw new DcmServiceException(Status.ProcessingFailure, e);
653         } finally {
654             if (storage != null)
655                 try {
656                     storage.remove();
657                 } catch (Exception ignore) {
658                 }
659         }
660     }
