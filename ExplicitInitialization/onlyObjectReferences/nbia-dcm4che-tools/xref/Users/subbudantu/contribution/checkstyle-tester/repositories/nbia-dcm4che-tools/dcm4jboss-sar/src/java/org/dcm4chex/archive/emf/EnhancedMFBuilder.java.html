101     public EnhancedMFBuilder(UpgradeToEnhancedMFService service,
102             Dataset filter, int frameTypeTag, int numFrames) {
103         this.log = service.getLog();
104         this.noPixelData = service.isNoPixelData();
105         this.deflate = service.isDeflate();
106         this.filter = filter;
107         this.fgFilters = filter.getItem(Tags.SharedFunctionalGroupsSeq);
108         this.dataset = DcmObjectFactory.getInstance().newDataset();
109         this.sharedFGs = dataset.putSQ(Tags.SharedFunctionalGroupsSeq).addNewItem();
110         this.frameTypeTag = frameTypeTag;
111         this.pixelDataOffsets = new long[numFrames];
112         this.pixelDataLengths = new int[numFrames];
113      }

132     public void add(File f) throws UpgradeToEnhancedMFException, IOException {
133         if (curFrame == pixelDataOffsets.length) {
134             throw new IllegalStateException("curFrame: " + curFrame
135                     + " == numFrame: " + pixelDataOffsets.length);
136         }
137         if (log.isDebugEnabled()) {
138             log.debug("M-READ " + f);
139         }
140         Dataset source = DcmObjectFactory.getInstance().newDataset();
141         BufferedInputStream bis = new BufferedInputStream(
142                 new FileInputStream(f));
143         DcmParser parser = DcmParserFactory.getInstance().newDcmParser(bis);
144         try {
145             parser.setDcmHandler(source.getDcmHandler());
146             parser.parseDcmFile(FileFormat.DICOM_FILE, Tags.PixelData);
147             if (parser.getReadTag() != Tags.PixelData) {
148                 throw new UpgradeToEnhancedMFException("No Pixel Data in " + f);
149             }
150             if (curFrame == 0) {
151                 if (log.isDebugEnabled()) {
152                     log.debug("Create new Enhanced MF from " + f);
153                 }
154                 tsuid = source.getFileMetaInfo().getTransferSyntaxUID();
155                 pixelDataVR = parser.getReadVR();
156                 pixelDataLength = parser.getReadLength();
157                 init(source);
158                 f0 = f;
159             } else {
160                 if (log.isDebugEnabled()) {
161                     log.debug("Merge Functional Groups to Enhanced MF from " + f);
162                 }
163                 if (pixelDataVR != parser.getReadVR()) {
164                     throw new UpgradeToEnhancedMFException("VR of Pixel Data in "
165                             + f + " differs from " + f0);
166                 }
167                 if (pixelDataLength != parser.getReadLength()) {
168                     throw new UpgradeToEnhancedMFException("Pixel Data Length in "
169                             + f + " differs from " + f0);
170                 }
171                 if (!noPixelData && !tsuid.equals(source.getFileMetaInfo()
172                                                 .getTransferSyntaxUID())) {
173                     throw new UpgradeToEnhancedMFException("Transfer Syntax of "
174                             + f + " differs from " + f0);
175                 }
176                 if (!dataset.exclude(FG_SEQ_TAGS).equals(source.subSet(filter))) {
177                     throw new UpgradeToEnhancedMFException(
178                             "Non Functional Groups elements of "  + f 
179                             + " differs from " + f0);
180                 }
181                 mergeFunctionalGroups(dataset, source);
182             }
183             if (pixelDataLength == -1) {
184                 // skip frame offset table
185                 parser.parseHeader();
186                 if (parser.getReadLength() != 0) {
187                     throw new UpgradeToEnhancedMFException(
188                             "Non-empty Frame Offset table in " + f);
189                 }
190                 parser.parseHeader();
191             }
192             pixelDataOffsets[curFrame] = parser.getStreamPosition();
193             pixelDataLengths[curFrame] = parser.getReadLength();
194         } finally {
195             try { bis.close(); } catch (IOException ignore) {}
196         }
197         ++curFrame;
198     }

200     private void init(Dataset source) {
201         dataset.putAll(source.subSet(filter));
202         for (Iterator it = fgFilters.iterator(); it.hasNext();) {
203             DcmElement el = (DcmElement) it.next();
204             Dataset fgFilter = el.getItem();
205             Dataset fg = source.subSet(fgFilter);
206             if (!fg.isEmpty()) {
207                 sharedFGs.putSQ(el.tag()).addNewItem().putAll(fg);
208             }
209         }
210         sharedFGs.putSQ(frameTypeTag).addNewItem().putCS(Tags.FrameType,
211                 imageType = source.getStrings(Tags.ImageType));
212         Dataset perFrameFGs = dataset.putSQ(
213                 Tags.PerFrameFunctionalGroupsSeq).addNewItem();
214         addReferencedImageFG(perFrameFGs, source);
215         addFrameContentFG(perFrameFGs, source);
216     }

218     private void addReferencedImageFG(Dataset perFrameFGs, Dataset source) {
219         DcmElement refImageSeq = perFrameFGs.putSQ(Tags.RefImageSeq);
220         Dataset refSource = refImageSeq.addNewItem();
221         refSource.putUI(Tags.RefSOPClassUID,
222                 source.getString(Tags.SOPClassUID));
223         refSource.putUI(Tags.RefSOPInstanceUID,
224                 source.getString(Tags.SOPInstanceUID));
225         refSource.putSQ(Tags.PurposeOfReferenceCodeSeq)
226                 .addItem(ALTERNATE_SOP_CLASS_INSTANCE);
227         
228         DcmElement srcRefImageSeq = source.get(Tags.RefImageSeq);
229         if (srcRefImageSeq != null) {
230             for (int i = 0, n = srcRefImageSeq.countItems(); i < n; i++) {
231                 refImageSeq.addItem(srcRefImageSeq.getItem(i));
232             }
233         }
234     }

236     private void addFrameContentFG(Dataset perFrameFGs, Dataset source) {
237         Dataset fg = perFrameFGs.putSQ(Tags.FrameContentSeq).addNewItem();
238         Date d = source.getDate(Tags.AcquisitionDatetime);
239         if (d == null) {
240             d = source.getDateTime(Tags.AcquisitionDate, Tags.AcquisitionTime);
241         }
242         if (d != null) {
243             fg.putDT(Tags.FrameAcquisitionDatetime, d);
244             if (acquisitionDatetime == null
245                     || acquisitionDatetime.compareTo(d) > 0) {
246                 acquisitionDatetime = d;
247             }
248         }
249         String s;
250         if ((s = source.getString(Tags.AcquisitionNumber)) != null) {
251             try {
252                 fg.putUS(Tags.FrameAcquisitionNumber, Integer.parseInt(s));
253             } catch (NumberFormatException e) {
254                 log.info("Ignore non-numeric Acquisition Number: " + s
255                         + " of Instance: "
256                         + source.getString(Tags.SOPInstanceUID));
257             }
258         }
259         if ((s = source.getString(Tags.ImageComments)) != null) {
260             fg.putLT(Tags.FrameComments, s);
261         }
262         if ((s = source.getString(Tags.InstanceCreationDate)) != null) {
263             fg.putDA(Tags.InstanceCreationDate, s);
264         }
265         if ((s = source.getString(Tags.InstanceCreationTime)) != null) {
266             fg.putTM(Tags.InstanceCreationTime, s);
267         }
268         if ((s = source.getString(Tags.ContentDate)) != null) {
269             fg.putDA(Tags.ContentDate, s);
270         }
271         if ((s = source.getString(Tags.ContentTime)) != null) {
272             fg.putTM(Tags.ContentTime, s);
273         }
274         if ((s = source.getString(Tags.InstanceNumber)) != null) {
275             fg.putIS(Tags.InstanceNumber, s);
276         }
277         d = source.getDateTime(Tags.ContentDate, Tags.ContentTime);
278         if (d != null) {
279             if (contentDatetime == null
280                     || contentDatetime.compareTo(d) > 0) {
281                 contentDatetime = d;
282                 instanceNumber = s;
283             }
284         }
285     }

287     private void mergeFunctionalGroups(Dataset dataset, Dataset source) {
288         DcmElement perFrameFGSeq =
289                 dataset.get(Tags.PerFrameFunctionalGroupsSeq);
290         Dataset perFrameFGs = perFrameFGSeq.addNewItem();
291         addReferencedImageFG(perFrameFGs, source);
292         addFrameContentFG(perFrameFGs, source);
293         int[] noLongerSharedFGTags = {};
294         for (Iterator it = fgFilters.iterator(); it.hasNext();) {
295             DcmElement el = (DcmElement) it.next();
296             int fgTag = el.tag();
297             Dataset fgFilter = el.getItem();
298             Dataset fg = source.subSet(fgFilter);
299             Dataset sharedFG = sharedFGs.getItem(fgTag);
300             if (!fg.equals(sharedFG)) {
301                 if (sharedFG != null) {
302                     noLongerSharedFGTags = append(noLongerSharedFGTags, fgTag);
303                 }
304                 if (!fg.isEmpty()) {
305                     perFrameFGs.putSQ(fgTag).addNewItem().putAll(fg);
306                 }
307             }
308         }
309         String[] frameType = source.getStrings(Tags.ImageType);
310         if (!Arrays.equals(this.imageType, frameType)
311                 && sharedFGs.contains(frameTypeTag)) {
312             noLongerSharedFGTags = append(noLongerSharedFGTags, frameTypeTag);
313         }
314         if (noLongerSharedFGTags.length != 0) {
315             Dataset noLongerSharedFGs = sharedFGs.subSet(noLongerSharedFGTags);
316             for (int i = 0, n = perFrameFGSeq.countItems() - 1; i < n; i++) {
317                 perFrameFGSeq.getItem(i).putAll(noLongerSharedFGs);
318             }
319             noLongerSharedFGs.clear();
320         }
321         if (!sharedFGs.contains(frameTypeTag)) {
322             perFrameFGs.putSQ(frameTypeTag).addNewItem()
323                     .putCS(Tags.FrameType, frameType);
324         }
325         this.imageType = mergeImageType(this.imageType, frameType);
326     }

328     private int[] append(int[] a, int i) {
329         int[] tmp = new int[a.length + 1];
330         System.arraycopy(a, 0, tmp, 0, a.length);
331         tmp[a.length] = i;
332         return tmp;
333     }

335     public Dataset build() throws UpgradeToEnhancedMFException {
336         if (curFrame < pixelDataOffsets.length) {
337             throw new IllegalStateException("curFrame: " + curFrame
338                     + " < numFrame: " + pixelDataOffsets.length);
339         }
340         dataset.putCS(Tags.ImageType, imageType);
341         dataset.putCS(Tags.InstanceNumber, instanceNumber);
342         dataset.putDA(Tags.ContentDate, contentDatetime);
343         dataset.putTM(Tags.ContentTime, contentDatetime);
344         dataset.putDT(Tags.AcquisitionDatetime, acquisitionDatetime);
345         dataset.putIS(Tags.NumberOfFrames, pixelDataOffsets.length);
346         UIDGenerator uidGen = UIDGenerator.getInstance();
347         if (!dataset.containsValue(Tags.SeriesInstanceUID)) {
348             dataset.putUI(Tags.SeriesInstanceUID, uidGen.createUID());
349         }
350         dataset.putUI(Tags.SOPClassUID, filter.getString(Tags.SOPClassUID));
351         dataset.putUI(Tags.SOPInstanceUID, uidGen.createUID());
352         try {
353             dataset.setFileMetaInfo(
354                     DcmObjectFactory.getInstance().newFileMetaInfo(dataset,
355                             noPixelData
356                                     ? (deflate ? UIDs.NoPixelDataDeflate
357                                                : UIDs.NoPixelData)
358                                     : tsuid));
359         } catch (DcmValueException e) {
360             throw new UpgradeToEnhancedMFException(e);
361         }
362         return dataset;
363     }

365     private String[] mergeImageType(String[] t1, String[] t2) {
366         if (t1.length < t2.length) {
367             String[] tmp = t1;
368             t1 = t2;
369             t2 = tmp;
370         }
371         for (int i = 0; i < t1.length; i++) {
372             if (!(i < t2.length && t1[i].equals(t2[i]))) {
373                 t1[i] = "MIXED";
374             }
375         }
376         return t1;
377     }
