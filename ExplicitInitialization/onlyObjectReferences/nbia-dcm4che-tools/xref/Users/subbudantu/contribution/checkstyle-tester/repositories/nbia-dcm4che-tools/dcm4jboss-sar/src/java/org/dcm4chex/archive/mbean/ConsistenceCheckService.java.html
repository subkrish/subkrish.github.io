87      private final NotificationListener consistentCheckListener = new NotificationListener() {
88          public void handleNotification(Notification notif, Object handback) {
89              Calendar cal = Calendar.getInstance();
90              int hour = cal.get(Calendar.HOUR_OF_DAY);
91              if (isDisabled(hour)) {
92                  if (log.isDebugEnabled())
93                      log.debug("ConsistentCheck ignored in time between "
94                              + disabledStartHour + " and " + disabledEndHour
95                              + " !");
96              } else {
97                  try {
98                      check();
99                  } catch (Exception e) {
100                     log.error("Consistant check failed!", e);
101                 }
102             }
103         }
104     };

123     public final String getTaskInterval() {
124         String s = RetryIntervalls.formatIntervalZeroAsNever(taskInterval);
125         return (disabledEndHour == -1) ? s : s + "!" + disabledStartHour + "-"
126                 + disabledEndHour;
127     }
128 
129     public void setTaskInterval(String interval) throws Exception {
130         long oldInterval = taskInterval;
131         int pos = interval.indexOf('!');
132         if (pos == -1) {
133             taskInterval = RetryIntervalls.parseIntervalOrNever(interval);
134             disabledEndHour = -1;
135         } else {
136             taskInterval = RetryIntervalls.parseIntervalOrNever(interval
137                     .substring(0, pos));
138             int pos1 = interval.indexOf('-', pos);
139             disabledStartHour = Integer.parseInt(interval.substring(pos + 1,
140                     pos1));
141             disabledEndHour = Integer.parseInt(interval.substring(pos1 + 1));
142         }
143         if (getState() == STARTED && oldInterval != taskInterval) {
144             scheduler.stopScheduler(timerIDCheckStudyConsistency, listenerID,
145                     consistentCheckListener);
146             listenerID = scheduler.startScheduler(timerIDCheckStudyConsistency,
147                     taskInterval, consistentCheckListener);
148         }
149     }

230     public String check() throws RemoteException, FinderException {
231         int updated = 0;
232         long l = System.currentTimeMillis();
233         Timestamp createdBefore = new Timestamp(l - this.minStudyAge);
234         Timestamp createdAfter = new Timestamp(l - this.maxStudyAge);
235         Timestamp checkedBefore = new Timestamp(l - this.maxCheckedBefore);
236         ConsistencyCheck checker = newConsistencyCheck();
237         if (log.isDebugEnabled())
238             log.debug("call findStudiesToCheck: createdAfter:" + createdAfter
239                     + " createdBefore:" + createdBefore + " checkedBefore:"
240                     + checkedBefore);
241         long[] studyPks = checker.findStudiesToCheck(createdAfter,
242                 createdBefore, checkedBefore, this.limitNumberOfStudiesPerTask);
243         for (int i = 0, len = studyPks.length; i < len; i++) {
244             if (checker.updateStudy(studyPks[i],
245                     availabilityOfExternalRetrieveable)) {
246                 updated++;
247             }
248         }
249         return updated + " of " + studyPks.length + " studies updated!";
250     }
251 
252     private boolean isDisabled(int hour) {
253         if (disabledEndHour == -1)
254             return false;
255         boolean sameday = disabledStartHour <= disabledEndHour;
256         boolean inside = hour >= disabledStartHour && hour < disabledEndHour;
257         return sameday ? inside : !inside;
258     }

260     protected void startService() throws Exception {
261         listenerID = scheduler.startScheduler(timerIDCheckStudyConsistency,
262                 taskInterval, consistentCheckListener);
263     }
264 
265     protected void stopService() throws Exception {
266         scheduler.stopScheduler(timerIDCheckStudyConsistency, listenerID,
267                 consistentCheckListener);
268         super.stopService();
269     }

271     private ConsistencyCheck newConsistencyCheck() {
272         try {
273             ConsistencyCheckHome home = (ConsistencyCheckHome) EJBHomeFactory
274                     .getFactory().lookup(ConsistencyCheckHome.class,
275                             ConsistencyCheckHome.JNDI_NAME);
276             return home.create();
277         } catch (Exception e) {
278             throw new RuntimeException(
279                     "Failed to access ConsistencyCheck EJB:", e);
280         }
281     }
