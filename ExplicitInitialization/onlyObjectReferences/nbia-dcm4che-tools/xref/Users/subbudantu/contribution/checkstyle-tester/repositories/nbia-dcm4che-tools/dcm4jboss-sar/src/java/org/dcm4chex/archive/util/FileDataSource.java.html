213     public final void setSimpleFrameList(int[] simpleFrameList) {
214         if (simpleFrameList != null) {
215             if (calculatedFrameList != null) {
216                 throw new IllegalStateException();
217             }
218             if (simpleFrameList.length == 0) {
219                 throw new IllegalArgumentException();
220             }
221             for (int i = 0; i < simpleFrameList.length; i++) {
222                 if (simpleFrameList[i] <= 0) {
223                     throw new IllegalArgumentException();
224                 }
225                 if (i != 0 && simpleFrameList[i]
226                            <= simpleFrameList[i-1]) {
227                     throw new IllegalArgumentException();
228                 }
229             }
230         }
231         this.simpleFrameList = simpleFrameList;
232     }
233 
234     public final void setCalculatedFrameList(int[] calculatedFrameList) {
235         if (calculatedFrameList != null) {
236             if (simpleFrameList != null) {
237                 throw new IllegalStateException();
238             }
239             if (calculatedFrameList.length == 0) {
240                 throw new IllegalArgumentException();
241             }
242             if (calculatedFrameList.length % 3 != 0) {
243                 throw new IllegalArgumentException();
244             }
245             for (int i = 0; i < calculatedFrameList.length; i++) {
246                 if (calculatedFrameList[i] <= 0) {
247                     throw new IllegalArgumentException();
248                 }
249                 switch (i % 3) {
250                 case 0:
251                     if (i != 0 && calculatedFrameList[i]
252                                <= calculatedFrameList[i-2]) {
253                         throw new IllegalArgumentException();
254                     }
255                     break;
256                 case 1:
257                     if (i != 0 && calculatedFrameList[i]
258                                < calculatedFrameList[i-1]) {
259                         throw new IllegalArgumentException();
260                     }
261                     break;
262                 }
263             }
264         }
265         this.calculatedFrameList = calculatedFrameList;
266     }

304     public void writeTo(OutputStream out, String tsUID) throws IOException {
305         log.info("M-READ file:" + file);
306         boolean withoutPixeldata1 = withoutPixeldata 
307                 || UIDs.NoPixelData.equals(tsUID)
308                 || UIDs.NoPixelDataDeflate.equals(tsUID);
309         DataInputStream dis = new DataInputStream(
310                 new BufferedInputStream(new FileInputStream(file)));
311         FileImageInputStream fiis = null;
312         try {
313             Dataset ds = DcmObjectFactory.getInstance().newDataset();
314             DcmParser parser = DcmParserFactory.getInstance().newDcmParser(dis);
315             parser.setDcmHandler(ds.getDcmHandler());
316             parser.parseDcmFile(null, Tags.PixelData);
317             boolean hasPixelData = parser.getReadTag() == Tags.PixelData;
318             DcmDecodeParam dcmDecodeParam = parser.getDcmDecodeParam();
319             if (!hasPixelData && !parser.hasSeenEOF()) {
320                 parser.unreadHeader();
321                 parser.parseDataset(dcmDecodeParam, -1);
322             }
323             ds.putAll(mergeAttrs);
324             // hook to perform any other dataset updates after the file has been read from the file system
325             if (datasetUpdater != null) {
326             	ds = datasetUpdater.updateDataset(ds);
327             }            
328             FileMetaInfo fmi = ds.getFileMetaInfo();
329             if (fmi == null) {
330                 fmi = DcmObjectFactory.getInstance()
331                         .newFileMetaInfo(ds, UIDs.ImplicitVRLittleEndian);
332                 ds.setFileMetaInfo(fmi);
333             }
334             String tsOrig = fmi.getTransferSyntaxUID();
335             if (tsUID == null)
336                 tsUID = tsOrig;
337             boolean patchJpegLS = this.patchJpegLS
338                     && !withoutPixeldata1
339                     && tsUID.equals(UIDs.JPEGLSLossless)
340                     && (patchJpegLSImplCUID == null
341                             || patchJpegLSImplCUID.equals(
342                             fmi.getImplementationClassUID()))
343                     && ds.getInt(Tags.BitsAllocated, 0) == 16;
344             if (writeFile) {
345                 if (!(withoutPixeldata1 
346                         || tsUID.equals(tsOrig)
347                         || tsUID.equals(UIDs.ImplicitVRLittleEndian))) {
348                     tsUID = UIDs.ExplicitVRLittleEndian;
349                     patchJpegLS = false;
350                 }
351                 fmi.putUI(Tags.TransferSyntaxUID, tsUID);
352 
353                 if (patchJpegLS && patchJpegLSNewImplCUID != null)
354                     fmi.putUI(Tags.ImplementationClassUID,
355                             patchJpegLSNewImplCUID);
356             }
357             log.debug("using transfersyntx:" + tsUID);
358             DcmEncodeParam enc = DcmEncodeParam.valueOf(tsUID);
359             if (!hasPixelData) {
360                 log.debug("Dataset:\n");
361                 log.debug(ds);
362                 write(ds, out, enc);
363                 return;
364             }
365             int pixelDataLen = parser.getReadLength();
366             boolean encapsulated = pixelDataLen == -1;
367             int framesInFile = ds.getInt(Tags.NumberOfFrames, 1);
368             if (simpleFrameList != null) {
369                 if (simpleFrameList[simpleFrameList.length - 1] > framesInFile) {
370                     throw new RequestedFrameNumbersOutOfRangeException();
371                 }
372             } else if (calculatedFrameList != null) {
373                 if (calculatedFrameList[0] > framesInFile) {
374                     throw new RequestedFrameNumbersOutOfRangeException();
375                 }
376                 simpleFrameList = calculateFrameList(framesInFile);
377             }
378             if (framesInFile == 1) {
379                 simpleFrameList = null;
380             }
381             if (simpleFrameList != null) {
382                 addFrameExtractionSeq(ds);
383                 addContributingEquipmentSeq(ds);
384                 adjustNumberOfFrames(ds);
385                 ds.putUI(Tags.SOPInstanceUID,
386                         UIDGenerator.getInstance().createUID());
387             }
388             if (withoutPixeldata1) {
389                 // skip Pixel Data
390                 if (!encapsulated) {
391                     dis.skipBytes(pixelDataLen);
392                 } else {
393                     do {
394                         parser.parseHeader();
395                         dis.skipBytes(parser.getReadLength());
396                     } while (parser.getReadTag() == Tags.Item);
397                 }
398                 // parse attributes after Pixel Data
399                 parser.parseDataset(dcmDecodeParam, -1);
400                 log.debug("Dataset:\n");
401                 log.debug(ds);
402                 write(ds, out, enc);
403                 return;
404             }
405             if (encapsulated && !enc.encapsulated) {
406                 DecompressCmd.adjustPhotometricInterpretation(ds, tsOrig);
407             }
408             log.debug("Dataset:\n");
409             log.debug(ds);
410             write(ds, out, enc);
411             if (!encapsulated) {
412                 // copy native Pixel Data
413                 if (simpleFrameList == null) {
414                     ds.writeHeader(out, enc, Tags.PixelData, VRs.OW,
415                             pixelDataLen);
416                     copyBytes(dis, out, pixelDataLen, buffer);
417                 } else {
418                     int frameLength = pixelDataLen / framesInFile;
419                     int newPixelDataLength =
420                         frameLength * simpleFrameList.length;
421                     ds.writeHeader(out, enc, Tags.PixelData, VRs.OW,
422                             (newPixelDataLength+1)&~1);
423                     int frameIndex = 0;
424                     for (int i = 0; i < simpleFrameList.length; i++) {
425                         while (++frameIndex < simpleFrameList[i]) {
426                             dis.skipBytes(frameLength);
427                         }
428                         copyBytes(dis, out, frameLength, buffer);
429                     }
430                     if ((newPixelDataLength & 1) != 0)
431                         out.write(0);
432                     // ignore attributes after Pixel Data
433                     return;
434                 }
435             } else if (enc.encapsulated) {
436                 // copy encapsulated Pixel Data
437                 ds.writeHeader(out, enc, Tags.PixelData, VRs.OB, -1);
438                 parser.parseHeader();
439                 int itemlen = parser.getReadLength();
440                 if (simpleFrameList == null && !patchJpegLS) {
441                     // copy Basic Offset Table
442                     ds.writeHeader(out, enc, parser.getReadTag(),
443                             VRs.NONE, itemlen);
444                     copyBytes(dis, out, itemlen, buffer);
445                 } else {
446                     // write empty Basic Offset Table
447                     ds.writeHeader(out, enc, Tags.Item, VRs.NONE, 0);
448                     // skip Basic Offset Table
449                     dis.skipBytes(itemlen);
450                 }
451                 if (simpleFrameList == null) {
452                     parser.parseHeader();
453                     while (parser.getReadTag() == Tags.Item) {
454                         itemlen = parser.getReadLength();
455                         copyItem(patchJpegLS, dis, ds, out, enc, itemlen);
456                         parser.parseHeader();
457                     };
458                     ds.writeHeader(out, enc, Tags.SeqDelimitationItem,
459                             VRs.NONE, 0);
460                     dis.skipBytes(parser.getReadLength());
461                 } else {
462                     // WARN frames spanning multiple data fragments not supported
463                     // assume one item per frame
464                     int frameIndex = 0;
465                     for (int i = 0; i < simpleFrameList.length; i++) {
466                         parser.parseHeader();
467                         itemlen = parser.getReadLength();
468                         while (++frameIndex < simpleFrameList[i]) {
469                             dis.skipBytes(itemlen);
470                             parser.parseHeader();
471                             itemlen = parser.getReadLength();
472                         }
473                         copyItem(patchJpegLS, dis, ds, out, enc, itemlen);
474                     }
475                     ds.writeHeader(out, enc, Tags.SeqDelimitationItem,
476                             VRs.NONE, 0);
477                     // ignore attributes after Pixel Data
478                     return;
479                 }
480             } else {
481                 // decompress encapsulated Pixel Data
482                 dis.close();
483                 dis = null;
484                 fiis = new FileImageInputStream(file);
485                 fiis.seek(parser.getStreamPosition());
486                 parser = DcmParserFactory.getInstance().newDcmParser(fiis);
487                 parser.setDcmHandler(ds.getDcmHandler());
488                 DecompressCmd cmd = createDecompressCmd(ds, tsOrig, parser);
489                 cmd.setSimpleFrameList(simpleFrameList);
490                 int newPixelDataLen = cmd.getPixelDataLength();
491                 ds.writeHeader(out, enc, Tags.PixelData, VRs.OW,
492                         (newPixelDataLen+1)&~1);
493                 try {
494                     cmd.decompress(enc.byteOrder, out);
495                 } catch (IOException e) {
496                     throw e;
497                 } catch (Throwable e) {
498                     throw new RuntimeException("Decompression failed:", e);
499                 }
500                 if ((newPixelDataLen&1) != 0)
501                     out.write(0);
502             }
503             // parse attributes after Pixel Data
504             parser.parseDataset(dcmDecodeParam, -1);
505             ds.subSet(Tags.PixelData, -1).writeDataset(out, enc);
506         } catch (JMException jme) {
507             throw new RuntimeException("Unable to update dataset:", jme);
508         } finally {
509             try {
510                 if (dis != null)
511                     dis.close();
512             } catch (IOException ignore) {
513             }
514             try {
515                 if (fiis != null) {
516                     fiis.close();
517                 }
518             } catch (IOException ignore) {
519             }
520         }
521     }

524     private void copyItem(boolean patchJpegLS, DataInputStream dis,
525             Dataset ds, OutputStream out, DcmEncodeParam enc, int itemlen)
526             throws IOException {
527         if (patchJpegLS) {
528             byte[] jpegheader = buffer;
529             dis.readFully(jpegheader, 0, 17);
530             byte[] lse = selectLSE(jpegheader);
531             if (lse == null) {
532                 ds.writeHeader(out, enc, Tags.Item, VRs.NONE, itemlen);
533                 out.write(jpegheader, 0, 17);
534                 copyBytes(dis, out, itemlen - 17, buffer);
535             } else {
536                 ds.writeHeader(out, enc, Tags.Item, VRs.NONE, itemlen + 16);
537                 out.write(jpegheader, 0, 15);
538                 out.write(lse);
539                 copyBytes(dis, out, itemlen - 17, buffer);
540                 out.write(0);
541             }
542         } else {
543             ds.writeHeader(out, enc, Tags.Item, VRs.NONE, itemlen);
544             copyBytes(dis, out, itemlen, buffer);
545         }
546     }

548     private static int toInt(byte[] b, int off) {
549         return (b[off] & 0xff) << 8 | (b[off+1] & 0xff);
550     }

552     private static byte[] selectLSE(byte[] jpegheader) {
553         if (toInt(jpegheader, 0) != SOI) {
554             log.warn("SOI marker is missing - do not patch JPEG LS");
555             return null;
556         }
557         if (toInt(jpegheader, 2) != SOF55) {
558             log.warn("SOI marker is not followed by JPEG-LS SOF marker "
559                     + "- do not patch JPEG LS");
560             return null;
561         }
562         if (toInt(jpegheader, 4) != 11) {
563             log.warn("unexpected length of JPEG-LS SOF marker segment "
564                     + "- do not patch JPEG LS");
565             return null;
566         }
567         int marker = toInt(jpegheader, 15);
568         if (marker != SOS) {
569             log.warn(marker == LSE
570                 ? "contains already LSE marker segment "
571                     + "- do not patch JPEG LS"
572                 : "JPEG-LS SOF marker segment is not followed by SOS marker "
573                     + "- do not patch JPEG LS");
574             return null;
575         }
576         switch (jpegheader[6]) {
577         case 13:
578             log.info("Patch JPEG LS 13-bit with "
579                     + "LSE segment(T1=34, T2=131, T3=548)");
580             return LSE_13;
581         case 14:
582             log.info("Patch JPEG LS 14-bit with "
583                     + "LSE segment(T1=66, T2=259, T3=1092)");
584             return LSE_14;
585         case 15:
586             log.info("Patch JPEG LS 15-bit with "
587                     + "LSE segment(T1=130, T2=515, T3=2180)");
588             return LSE_15;
589         case 16:
590             log.info("Patch JPEG LS 16-bit with "
591                     + "LSE segment(T1=258, T2=1027, T3=4356)");
592             return LSE_16;
593         }
594         return null;
595     }

597     private void adjustNumberOfFrames(Dataset ds) {
598         ds.putIS(Tags.NumberOfFrames, simpleFrameList.length);
599         DcmElement src = ds.remove(Tags.PerFrameFunctionalGroupsSeq);
600         if (src != null) {
601             DcmElement dest = ds.putSQ(Tags.PerFrameFunctionalGroupsSeq);
602             for (int i = 0; i < simpleFrameList.length; i++) {
603                 dest.addItem(src.getItem(simpleFrameList[i]-1));
604             }
605         }
606     }

615     private void addFrameExtractionSeq(Dataset ds) {
616         DcmElement seq = getOrPutSQ(ds, Tags.FrameExtractionSeq);
617         Dataset item = seq.addNewItem();
618         item.putUI(Tags.MultiFrameSourceSOPInstanceUID,
619                 ds.getString(Tags.SOPInstanceUID));
620         if (calculatedFrameList != null) {
621             item.putUL(Tags.CalculatedFrameList, calculatedFrameList);
622         } else {
623             item.putUL(Tags.SimpleFrameList, simpleFrameList);
624         }
625      }

632     private int[] calculateFrameList(int frames) {
633         int[] src = new int[frames];
634         int length = 0;
635         addFrame:
636         for (int i = 0; i < calculatedFrameList.length;) {
637             for (int f = calculatedFrameList[i++],
638                     last = calculatedFrameList[i++],
639                     step = calculatedFrameList[i++];
640                     f <= last; f += step) {
641                 if (f > frames) {
642                     break addFrame;
643                 }
644                 src[length++] = f;
645             }
646         }
647         int[] dest = new int[length];
648         System.arraycopy(src, 0, dest, 0, length);
649         return dest;
650     }

652     private void write(Dataset ds, OutputStream out, DcmEncodeParam enc)
653             throws IOException {
654     	Dataset dsOut = filterDataset(ds);
655     	
656         if (writeFile) {
657 			dsOut.writeFile(out, enc);
658         } else {
659             dsOut.writeDataset(out, enc);
660         }
661     }

663     protected Dataset filterDataset(Dataset dataset) {
664     	Dataset filteredDataset;
665     	
666     	if (excludePrivate) {
667     		filteredDataset = dataset.excludePrivate();
668     		
669     		if (writeFile) {
670     			filteredDataset.setFileMetaInfo(dataset.getFileMetaInfo());
671     		}
672     	} else {
673     		filteredDataset = dataset;
674     	}
675     	
676     	return filteredDataset;
677     }

679     private void copyBytes(InputStream is, OutputStream out, int totLen,
680             byte[] buffer) throws IOException {
681         for (int len, toRead = totLen; toRead > 0; toRead -= len) {
682             len = is.read(buffer, 0, Math.min(toRead, buffer.length));
683             if (len == -1) {
684                 throw new EOFException();
685             }
686             out.write(buffer, 0, len);
687         }
688     }
