283      * Set either a fix authorPerson or a mapping user to authorPerson.
284      * <p>
285      * Mapping format: &lt;user&gt;=&lt;authorPerson&gt;<br>
286      * Use either newline or semicolon to seperate mappings.
287      * <p>
288      * If '=' is ommited, a fixed autorPerson is set in <code>metadataProps</code>
289      * 
290      * @param s The authorPerson(-mapping) to set.
291      */
292     public void setAuthorPersonMapping(String s) {
293         if ( s == null || s.trim().length() < 1) return;
294         usr2author.clear();
295         if ( s.indexOf('=') == -1) {
296             metadataProps.setProperty(AUTHOR_PERSON, s); //NO mapping user -> authorPerson; use fix authorPerson instead
297         } else {
298             this.addMappingString(s, usr2author);
299         }
300     }

518     /**
519      * Adds a 'mappingString' (format:&lt;key&gt;=&lt;value&gt;...) to a map.
520      * 
521      * @param s
522      */
523     private void addMappingString(String s, Map map) {
524         StringTokenizer st = new StringTokenizer( s, ",;\n\r\t ");
525         String t;
526         int pos;
527         while ( st.hasMoreTokens() ) {
528             t = st.nextToken();
529             pos = t.indexOf('=');
530             if ( pos == -1) {
531                 map.put(t,t);
532             } else {
533                 map.put(t.substring(0,pos), t.substring(++pos));
534             }
535         }
536     }
537     /**
538      * Returns the String representation of a map
539      * @return
540      */
541     private String getMappingString(Map map) {
542         if ( map == null || map.isEmpty() ) return null;
543         StringBuffer sb = new StringBuffer();
544         String key;
545         for ( Iterator iter = map.keySet().iterator() ; iter.hasNext() ; ) {
546             key = iter.next().toString();
547             sb.append(key).append('=').append(map.get(key)).append( System.getProperty("line.separator", "\n"));
548         }
549         return sb.toString();
550     }

552     private List setListString(String s) {
553         List l = new ArrayList();
554         if ( NONE.equals(s) ) return l;
555         StringTokenizer st = new StringTokenizer( s, ";\n\r");
556         while ( st.hasMoreTokens() ) {
557             l.add(st.nextToken());
558         }
559         return l;
560     }

562     private String getListString(List l) {
563         if ( l == null || l.isEmpty() ) return NONE;
564         StringBuffer sb = new StringBuffer();
565         for ( Iterator iter = l.iterator() ; iter.hasNext() ; ) {
566             sb.append(iter.next()).append( System.getProperty("line.separator", "\n"));
567         }
568         return sb.toString();
569     }

644     /**
645      * @throws IOException
646      * 
647      */
648     public List readCodeFile(File codeFile) throws IOException {
649         if ( codeFile == null ) return new ArrayList();
650         List l = (List) mapCodeLists.get(codeFile);
651         if ( l == null ) {
652             l = new ArrayList();
653             File file = FileUtils.resolve(codeFile);
654             if ( file.exists() ) {
655                 BufferedReader r = new BufferedReader( new FileReader(file));
656                 String line;
657                 while ( (line = r.readLine()) != null ) {
658                     if ( ! (line.charAt(0) == '#') ) {
659                         l.add( line );
660                     }
661                 }
662                 log.debug("Codes read from code file "+codeFile);
663                 log.debug("Codes:"+l);
664                 mapCodeLists.put(codeFile,l);
665             } else {
666                 log.warn("Code File "+file+" does not exist! return empty code list!");
667             }
668         }
669         return l;
670     }

676     public boolean sendSOAP( String metaDataFilename, String docNames, String url ) {
677         File metaDataFile = new File( metaDataFilename);
678 
679         XDSIDocument[] docFiles = null;
680         if ( docNames != null && docNames.trim().length() > 0) {
681             StringTokenizer st = new StringTokenizer( docNames, "," );
682             docFiles = new XDSIDocument[ st.countTokens() ];
683             for ( int i=0; st.hasMoreTokens(); i++ ) {
684                 docFiles[i] = XDSIFileDocument.valueOf( st.nextToken() );

690     public boolean sendSOAP( Document metaData, XDSIDocument[] docs, String url ) {
691         if ( isUseXDSb() ) {
692             if ( xdsbSourceServiceName != null ) {
693                 return exportXDSb(metaData, docs);
694             } else {
695                 log.warn("UseXDSb is enabled but XdsbSourceServiceName is not configured! Use XDS.a instead!");
696             }
697         }
698         if ( url == null ) url = this.docRepositoryURI;
699         log.info("Send 'Provide and Register Document Set' request to "+url);
700         SOAPConnection conn = null;
701         try {
702             configTLS(url);
703             MessageFactory messageFactory = MessageFactory.newInstance();
704             SOAPMessage message = messageFactory.createMessage();
705             MimeHeaders hd = message.getMimeHeaders();
706             hd.addHeader("SOAPAction", "urn:ihe:iti:2007:ProvideAndRegisterDocumentSet");
707             
708             SOAPEnvelope envelope = message.getSOAPPart().getEnvelope();
709             SOAPBody soapBody = envelope.getBody();
710             SOAPElement bodyElement = soapBody.addDocument(metaData);
711             if ( docs != null ) {
712                 for (int i = 0; i < docs.length; i++) {
713                     DataHandler dhAttachment = docs[i].getDataHandler();
714                     AttachmentPart part = message.createAttachmentPart(dhAttachment);
715                     part.setMimeHeader("Content-Type", docs[i].getMimeType());
716                     String docId = docs[i].getDocumentID();
717                     if ( docId.charAt(0) != '<' ) {//Wrap with < >
718                         docId = "<"+docId+">";
719                     }
720 
721                     part.setContentId(docId);
722                     if ( log.isDebugEnabled()){
723                         log.debug("Add Attachment Part ("+(i+1)+"/"+docs.length+")! Document ID:"+part.getContentId()+" mime:"+docs[i].getMimeType());
724                     }
725                     message.addAttachmentPart(part);
726                 }
727             }
728             SOAPConnectionFactory connFactory = SOAPConnectionFactory.newInstance();
729 
730             conn = connFactory.createConnection();
731             log.info("++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++");
732             log.info("send request to "+url);
733             if ( logSOAPMessage ) log.info("-------------------------------- request  ----------------------------------");
734             dumpSOAPMessage(message);
735             SOAPMessage response = conn.call(message, url);
736             if ( ! logSOAPMessage ) log.info("-------------------------------- response ----------------------------------");
737             dumpSOAPMessage(response);
738             if ( logSOAPMessage ) log.info("++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++");
739             return checkResponse( response );
740         } catch ( Throwable x ) {
741             log.error("Cant send SOAP message! Reason:", x);
742             return false;
743         } finally {
744             if ( conn != null ) try {
745                 conn.close();
746             } catch (SOAPException ignore) {}
747         }

1026             if ( affinityDomain.length() == 1 ) {

1033                 return patID+"^^^"+affinityDomain.substring(2);
1034             } else {
1035                 log.info("PIX Query disabled: replace configured patient ID! :"+affinityDomain.substring(1));
1036                 return affinityDomain.substring(1);
1037             }
1038         }
1039         if ( this.pixQueryServiceName == null ) {
1040             log.info("PIX Query disabled: use source patient ID!");
1041             patID+="^^^";
1042             if ( issuer == null ) return patID;
1043             return patID+issuer;
1044         } else {
1045             try {
1046                 if ( localDomain != null ) {
1047                     if ( localDomain.charAt(0) == '=') {
1048                         String oldIssuer = issuer;
1049                         issuer = localDomain.substring(1);
1050                         log.info("PIX Query: Local affinity domain changed from "+oldIssuer+" to "+issuer);
1051                     } else if ( issuer == null ) {
1052                         log.info("PIX Query: Unknown local affinity domain changed to "+issuer);
1053                         issuer = localDomain;
1054                     }
1055                 } else if ( issuer == null ) {
1056                     issuer = "";
1057                 }
1058                 List pids = (List) server.invoke(this.pixQueryServiceName,
1059                         "queryCorrespondingPIDs",
1060                         new Object[] { patID, issuer, new String[]{affinityDomain} },
1061                         new String[] { String.class.getName(), String.class.getName(), String[].class.getName() });
1062                 String pid, affPid;
1063                 for ( Iterator iter = pids.iterator() ; iter.hasNext() ; ) {
1064                     pid = toPIDString((String[]) iter.next());
1065                     log.debug("Check if from domain! PatientID:"+pid);
1066                     if ( isFromDomain(pid) ) {
1067                         return pid;
1068                     }
1069                 }
1070                 log.error("Patient ID is not known in Affinity domain:"+affinityDomain);
1071                 return null;
1072             } catch (Exception e) {
1073                 log.error("Failed to get patientID for Affinity Domain:", e);
1074                 return null;
1075             }
1076         }

1099     protected boolean isFromDomain(String pid) {
1100         String assAuth = getAssigningAuthority(pid);
1101         if (assAuth == null || assAuth.indexOf('&') == -1)
1102             return false;
1103         if (assAuth.charAt(0) != '&') { //is namespace id subcomponent not empty?
1104             //we only compare <universal ID> and <universal ID type
1105             //leave subcomponent delimiter! (affinityDomain will almost always have no namespace id)
1106             assAuth = assAuth.substring(assAuth.indexOf('&'));
1107         }
1108         if (affinityDomain.charAt(0) == '&') {
1109             return assAuth.equals(affinityDomain);
1110         } else { //affinityDomain has namespace id but we ignore that!
1111             return assAuth.equals(affinityDomain.substring(affinityDomain.indexOf('&')));
1112         }
1113     }

1118             pos = pid.indexOf('^', pos);

1121                 return null;

1126         return end == -1 ? pid.substring(pos) : pid.substring(pos, end);

1300     private Properties getAutoPublishMetadataProperties(Dataset mpps) {
1301         Properties props = new Properties();
1302         BufferedInputStream bis = null;
1303         try {
1304             if (autoPublishPropertyFile == null ) return props;
1305             File propFile = FileUtils.resolve(this.autoPublishPropertyFile);
1306             bis= new BufferedInputStream( new FileInputStream( propFile ));
1307             props.load(bis);
1308             if ( sourceID != null ) {
1309                 props.setProperty(SOURCE_ID, sourceID);
1310             }
1311         } catch (IOException x) {
1312             log.error("Cant read Metadata Properties for AutoPublish!",x);
1313         } finally {
1314             if (bis != null) {
1315                 try {
1316                     bis.close();
1317                 } catch (IOException ignore) {}
1318             }
1319         }
1320         return props;
1321     }
