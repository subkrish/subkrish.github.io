138     private void initTransformHandler() {
139         try {
140             SAXTransformerFactory tf = (SAXTransformerFactory) TransformerFactory.newInstance();
141             if (xslt != null) {
142                 try {
143                     th = tf.newTransformerHandler(new StreamSource(xslt.openStream(),
144                             xslt.toExternalForm()));
145                 } catch ( IOException x ) {
146                     log.error("Cant open xsl file:"+xslt, x );
147                 }
148             } else {
149                 th = tf.newTransformerHandler();
150                 th.getTransformer().setOutputProperty(OutputKeys.INDENT, "yes");
151             }
152         } catch ( Throwable t ) {
153             log.warn("Could not initialize SAX transformer", t);
154         }
155     }

164             log.error("Wrong XSLT URL for XDS Metadata! Additional transformation disabled!", x);
165         }
166     }

185     public Document getMetadata() {
186         initTransformHandler();
187         DOMResult result = new DOMResult();
188         th.setResult( result );
189         try {
190             th.startDocument();
191             addSubmitObjectRequest();
192             th.endDocument();
193         } catch (SAXException x) {
194             log.error( "Cant build metadata!",x);
195             return null;
196         }
197         return (Document) result.getNode();
198     }

200     private void addSubmitObjectRequest() throws SAXException {
201         AttributesImpl attr = new AttributesImpl();
202         //attr.addAttribute("", XMLNS_XSI, XMLNS_XSI, "", "http://www.w3.org/2001/XMLSchema-instance");		
203         th.startPrefixMapping(XMLNS_RS,URN_RS);
204         th.startPrefixMapping(XMLNS_DEFAULT,URN_RIM);
205         th.startElement(URN_RS, TAG_SUBMITOBJECTSREQUEST, TAG_SUBMITOBJECTSREQUEST, attr );
206         addLeafRegistryObjectList();
207         th.endElement(URN_RS, TAG_SUBMITOBJECTSREQUEST, TAG_SUBMITOBJECTSREQUEST );
208         th.endPrefixMapping(XMLNS_RS);
209         th.endPrefixMapping(XMLNS_DEFAULT);
210     }

212     private void addLeafRegistryObjectList() throws SAXException {
213         th.startElement(URN_RIM, TAG_LEAFREGISTRYOBJECTLIST, TAG_LEAFREGISTRYOBJECTLIST, EMPTY_ATTRIBUTES );
214         addObjectRef(UUID.XDSSubmissionSet);
215         addObjectRef(UUID.XDSSubmissionSet_authorPerson);
216         addObjectRef(UUID.XDSSubmissionSet_uniqueId);
217         addObjectRef(UUID.XDSSubmissionSet_sourceId);
218         addObjectRef(UUID.XDSSubmissionSet_patientId); 
219         addObjectRef(UUID.XDSSubmissionSet_contentTypeCode); 
220         addObjectRef(UUID.XDSDocumentEntry_authorPerson);
221         addObjectRef(UUID.XDSDocumentEntry_uniqueId);
222         addObjectRef(UUID.XDSDocumentEntry_patientId); 
223         addObjectRef(UUID.XDSDocumentEntry_classCode); 
224         addObjectRef(UUID.XDSDocumentEntry_confidentialityCode); 
225         addObjectRef(UUID.XDSDocumentEntry_formatCode); 
226         addObjectRef(UUID.XDSDocumentEntry_eventCodeList); 
227         addObjectRef(UUID.XDSDocumentEntry_healthCareFacilityTypeCode); 
228         addObjectRef(UUID.XDSDocumentEntry_practiceSettingCode); 
229         addObjectRef(UUID.XDSDocumentEntry_typeCode); 
230         addClassification(SUBMISSION_SET, UUID.XDSSubmissionSet);
231         if ( newFolder ) {
232             addObjectRef(UUID.XDSFolder); 
233             addObjectRef(UUID.XDSFolder_uniqueId); 
234             addObjectRef(UUID.XDSFolder_patientId);
235             addObjectRef(UUID.XDSFolder_codeList);
236             
237             addClassification("Folder", UUID.XDSFolder);
238         }
239         String title;
240         if ( this.manifest != null ) {
241             for ( int i = 0 ; i < docs.length ; i++ ) {
242                 addExtrinsicObject(docs[i]);
243             }
244             Dataset cnDS = manifest.getItem(Tags.ConceptNameCodeSeq);
245             title = cnDS != null ? cnDS.getString(Tags.CodeMeaning) : mdValues.getProperty("docTitle","Title?");
246         } else {
247             title = mdValues.getProperty("submissionSetTitle","TITLE");
248         }
249         addSubmissionRegistryPackage(SUBMISSION_SET, title);
250         int fldrDocIdx = 0;
251         String folderDocAssocID;
252         if ( newFolder ) {
253             addFolderRegistryPackage("Folder", mdValues.getProperty("folder.name","FOLDER"));
254             addAssociation(null, SUBMISSION_SET,"Folder", HAS_MEMBER, null);
255             if ( docs != null ) {
256                 for ( int i = 0 ; i < docs.length ; i++ ) {
257                     folderDocAssocID = FOLDER_DOC_ASSOC+fldrDocIdx++;
258                     addAssociation( folderDocAssocID, "Folder", docs[i].getDocumentID(),HAS_MEMBER, null);
259                     addAssociation( null, SUBMISSION_SET, folderDocAssocID, HAS_MEMBER, null );
260                 }
261             }
262         }
263         if ( docs != null ) {
264             for ( int i = 0 ; i < docs.length ; i++ ) {
265                 addAssociation( null, SUBMISSION_SET,docs[i].getDocumentID(), HAS_MEMBER, ORIGINAL );
266             }
267 
268             String folders = mdValues.getProperty("folder_assoc.uniqueId");
269             if ( folders != null) {
270                 StringTokenizer st = new StringTokenizer(folders, "|");
271                 String uuid;
272                 while ( st.hasMoreTokens() ) {
273                     uuid = st.nextToken();
274                     for ( int i = 0 ; i < docs.length ; i++ ) {
275                         folderDocAssocID = FOLDER_DOC_ASSOC+fldrDocIdx++;
276                         addAssociation(folderDocAssocID, uuid,docs[i].getDocumentID(), HAS_MEMBER, null);
277                         addAssociation( null, SUBMISSION_SET, folderDocAssocID, HAS_MEMBER, null );
278                     }
279                     addObjectRef(uuid);
280                 }
281             }
282         }
283         addDocRefs();
284         th.endElement(URN_RIM, TAG_LEAFREGISTRYOBJECTLIST, TAG_LEAFREGISTRYOBJECTLIST );
285 
286     }

288     private void addDocRefs() throws SAXException {
289         if ( docs != null ) {
290             XDSIDocument doc;
291             List assocs;
292             XDSIDocument.Association assoc;
293             String uuid;
294             List objectRefs = new ArrayList();
295             for ( int i = 0 ; i < docs.length ; i++ ) {
296                 assocs = docs[i].getAssociations();
297                 if ( assocs != null ) {
298                     for ( Iterator iter = assocs.iterator() ; iter.hasNext() ; ) {
299                         assoc = (XDSIDocument.Association) iter.next();
300                         uuid = assoc.getUUID();
301                         addAssociation(null, docs[i].getDocumentID(),uuid, assoc.getType(), assoc.getStatus());
302                         if ( !objectRefs.contains(uuid)){
303                             addObjectRef(uuid);
304                             objectRefs.add(uuid);
305                         }
306                     }
307                 }
308             }
309         }
310     }

316     private void addAssociation(String id, String sourceObject, String targetObject, String type, String status) throws SAXException {
317         AttributesImpl attr = new AttributesImpl();
318         if ( id != null )
319             attr.addAttribute("", "id", "id", "", id);          
320         attr.addAttribute("", "associationType", "associationType", "", type);          
321         attr.addAttribute("", "sourceObject", "sourceObject", "", sourceObject);		
322         attr.addAttribute("", "targetObject", "targetObject", "", targetObject);
323         status = checkStatus(status, type);
324         th.startElement("", TAG_ASSOCIATION, TAG_ASSOCIATION, attr );
325         if ( status != null ) addSlot("SubmissionSetStatus", status);
326         th.endElement("", TAG_ASSOCIATION, TAG_ASSOCIATION );
327     }

329     private String checkStatus(String status, String type) {
330         if ( status == null ) return null;
331         if ( type == null || !type.endsWith("HasMember") ) 
332             return null;
333         return status;
334     }

405         th.endElement("", TAG_EXTRINSICOBJECT, TAG_EXTRINSICOBJECT );

480     private void addEventCodeList(XDSIDocument doc) throws SAXException {
481         String eventCodeList = mdValues.getProperty("eventCodeList");
482         if ( eventCodeList == null || eventCodeList.trim().length() < 1 ) return;
483 
484         StringTokenizer st = new StringTokenizer( eventCodeList, "|");
485         String[] code;//0..codeValue, 1.. codeMeaning (2..codeDesignator)
486         while ( st.hasMoreTokens() ) {
487             code = StringUtils.split(st.nextToken(), '^');
488             addClassification(UUID.XDSDocumentEntry_eventCodeList, doc.getDocumentID(), 
489                     code[0],
490                     code.length > 1 ? code[1] : code[0],
491                             code.length > 2 ? code[2] : mdValues.getProperty("eventCodeListCodingSchemeOID","Connect-a-thon confidentialityCodes"));
492         }

494     private String getTime(int dateTag, int timeTag, int alternateDateTag, int alternateTimeTag) {
495         String date = manifest.getString(dateTag);
496         String time;
497         if ( date == null ) {
498             if ( alternateDateTag > 0) {
499                 date = manifest.getString(alternateDateTag);
500             }
501             if (date == null ) {
502                 return formatter.format(new Date());
503             } else if (alternateTimeTag != -1){
504                 time = manifest.getString(alternateTimeTag);
505                 if ( time == null ) time =DEFAULT_TIME_STRING;
506             } else {
507                 time =DEFAULT_TIME_STRING;
508             }
509         } else if (timeTag != -1) {
510             time = manifest.getString(timeTag);
511             if ( time == null ) time =DEFAULT_TIME_STRING;
512 
513         } else {
514             time = DEFAULT_TIME_STRING;
515         }
516         return date+time.substring(0,6);//cut fraction part of time
517     }

521     private String[] getPatientInfo(String patId) {
522         String[] values = new String[5];
523         values[0] = "PID-3|"+patId;
524         values[1] = "PID-5|"+manifest.getString(Tags.PatientName);
525         String birthdate = manifest.getString(Tags.PatientBirthDate);
526         if (birthdate == null ) {
527             values[2] = null;
528         } else {
529             String birthtime = manifest.getString(Tags.PatientBirthTime);
530             values[2] = "PID-7|"+birthdate+(birthtime != null ? birthtime : DEFAULT_TIME_STRING);
531         }
532         String sex = manifest.getString(Tags.PatientSex);
