206     public AEDTO getAE() {
207         if (cipherSuites == null || cipherSuites.length() < 1) {
208             StringBuffer sb = new StringBuffer();
209             if (cipher1 != null && cipher1.length() > 0)
210                 sb.append(cipher1);
211             if (cipher2 != null && cipher2.length() > 0)
212                 sb.append(",").append(cipher2);
213             if (cipher3 != null && cipher3.length() > 0)
214                 sb.append(",").append(cipher3);
215             cipherSuites = sb.toString();
216         }
217         AEDTO dto = new AEDTO();
218         dto.setPk(pk);
219         dto.setTitle(title);
220         dto.setHostName(hostName);
221         dto.setPort(port);
222         dto.setCipherSuitesAsString(cipherSuites);
223         dto.setIssuerOfPatientID(issuerOfPatientID);
224         dto.setIssuerOfAccessionNumberAsString(issuerOfAccessionNumber());
225         dto.setUserID(userID);
226         dto.setPassword(passwd);
227         dto.setFileSystemGroupID(fileSystemGroupID);
228         dto.setGroup(group);
229         dto.setDescription(description);
230         dto.setWadoURL(wadoUrl);
231         dto.setStationName(stationName);
232         dto.setInstitution(institution);
233         dto.setDepartment(department);
234         dto.setInstalled(installed);
235         return dto;
236     }

238     private String issuerOfAccessionNumber() {
239         String id = trim(issuerOfAccessionNumberID);
240         String uid = trim(issuerOfAccessionNumberUID);
241         String uidType = trim(issuerOfAccessionNumberUIDType);
242         int len = id.length() + uid.length() + uidType.length();
243         if (len == 0)
244             return null;
245         
246         return new StringBuilder(len+2)
247                 .append(id)
248                 .append('^')
249                 .append(uid)
250                 .append('^')
251                 .append(uidType)
252                 .toString();
253     }
254 
255     private String trim(String s) {
256         return s != null ? s.trim() : "";
257     }

259     public void setAE(AEDTO ae) {
260         pk = ae.getPk();
261         title = ae.getTitle();
262         hostName = ae.getHostName();
263         port = ae.getPort();
264         cipherSuites = ae.getCipherSuitesAsString();
265         issuerOfPatientID = ae.getIssuerOfPatientID();
266         String[] a = ae.getIssuerOfAccessionNumber();
267         issuerOfAccessionNumberID = a != null && a.length > 0 ? a[0] : null;
268         issuerOfAccessionNumberUID = a != null && a.length > 1 ? a[1] : null;
269         issuerOfAccessionNumberUIDType = a != null && a.length > 2 ? a[2] : null;
270         userID = ae.getUserID();
271         passwd = ae.getPassword();
272         fileSystemGroupID = ae.getFileSystemGroupID();
273         group = ae.getGroup();
274         description = ae.getDescription();
275         wadoUrl = ae.getWadoURL();
276         stationName = ae.getStationName();
277         institution = ae.getInstitution();
278         department = ae.getDepartment();
279         installed = ae.isInstalled();
280     }

317     /**
318      * Get the model for an http request.
319      * <p>
320      * Look in the session for an associated model via <code>AE_ATTR_NAME</code><br>
321      * If there is no model stored in session (first request) a new model is
322      * created and stored in session.
323      * 
324      * @param request
325      *            A http request.
326      * 
327      * @return The model for given request.
328      */
329     public static final AEModel getModel(HttpServletRequest request) {
330         AEModel model = (AEModel) request.getSession().getAttribute(
331                 AE_ATTR_NAME);
332         if (model == null) {
333             model = new AEModel(request.getUserPrincipal().getName(), request);
334             request.getSession().setAttribute(AE_ATTR_NAME, model);
335         }
336         return model;
337     }
