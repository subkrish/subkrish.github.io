91      /**
92       * Get the model for an http request.
93       * <p>
94       * Look in the session for an associated model via <code>GPWLMODEL_ATTR_NAME</code><br>
95       * If there is no model stored in session (first request) a new model is created and stored in session.
96       * 
97       * @param request A http request.
98       * 
99       * @return The model for given request.
100      */
101     public static final GPWLModel getModel( HttpServletRequest request ) {
102         GPWLModel model = (GPWLModel) request.getSession().getAttribute(GPWLMODEL_ATTR_NAME);
103         if (model == null) {
104             model = new GPWLModel(request);
105             request.getSession().setAttribute(GPWLMODEL_ATTR_NAME, model);
106             model.filterWorkList( true );
107         }
108         return model;
109     }

151     /**
152      * Update the list of GPWLEntries for the view.
153      * <p>
154      * The query use the search criteria values from the filter and use offset and limit for paging.
155      * <p>
156      * if <code>newSearch is true</code> will reset paging (set <code>offset</code> to 0!)
157      * @param newSearch
158      */
159     public void filterWorkList(boolean newSearch) {
160 
161         if ( newSearch ) setOffset(0);
162         Dataset searchDS = gpwlFilter.toDataset();
163         isLocal = GPWLConsoleCtrl.getGPWLScuDelegate().isLocal();
164         List l = GPWLConsoleCtrl.getGPWLScuDelegate().findGPWLEntries( searchDS );
165         Collections.sort( l, comparator );
166         int total = l.size();
167         int offset = getOffset();
168         int limit = getLimit();
169         int end;
170         if ( offset >= total ) {
171             offset = 0;
172             end = limit < total ? limit : total;
173         } else {
174             end = offset + limit;
175             if ( end > total ) end = total;
176         }
177         Dataset ds;
178         gpwlEntries.clear();
179         int countNull = 0;
180         GPWLEntry entry;
181         for ( int i = offset ; i < end ; i++ ){
182             ds = (Dataset) l.get( i );
183             if ( ds != null ) {
184                 entry = new GPWLEntry( ds );
185                 gpwlEntries.put( entry.getIUID(), entry );
186             } else {
187                 countNull++;
188             }
189         }
190         setTotal(total - countNull); // the real total (without null entries!)	
191     }

193     /**
194      * Inner class that compares two datasets for sorting Scheduled Procedure Steps 
195      * according scheduled Procedure step start date.
196      * 
197      * @author franz.willer
198      *
199      * TODO To change the template for this generated type comment go to
200      * Window - Preferences - Java - Code Style - Code Templates
201      */
202     public class SpsDSComparator implements Comparator {
203 
204         private final Date DATE_0 = new Date(0l);
205         public SpsDSComparator() {
206 
207         }
208 
209         /**
210          * Compares the scheduled procedure step start date and time of two Dataset objects.
211          * <p>
212          * USe either SPSStartDateAndTime or SPSStartDate and SPSStartTime to get the date.
213          * <p>
214          * Use the '0' Date (new Date(0l)) if the date is not in the Dataset.
215          * <p>
216          * Compares its two arguments for order. Returns a negative integer, zero, or a positive integer 
217          * as the first argument is less than, equal to, or greater than the second.
218          * <p>
219          * Throws an Exception if one of the arguments is null or not a Dataset object.
220          * 
221          * @param arg0 	First argument
222          * @param arg1	Second argument
223          * 
224          * @return <0 if arg0<arg1, 0 if equal and >0 if arg0>arg1
225          */
226         public int compare(Object arg0, Object arg1) {
227             Dataset ds1 = (Dataset) arg0;
228             Dataset ds2 = (Dataset) arg1;
229             Date d1 = _getStartDateAsLong( ds1 );
230             return d1.compareTo( _getStartDateAsLong( ds2 ) );
231         }
232 
233         /**
234          * @param ds1 The dataset
235          * 
236          * @return the date of this SPS Dataset.
237          */
238         private Date _getStartDateAsLong(Dataset ds) {
239             if ( ds == null ) return DATE_0;
240             Date d = ds.getDate( Tags.SPSStartDateAndTime );
241             if ( d == null ) d = DATE_0;
242             return d;
243         }
244     }
