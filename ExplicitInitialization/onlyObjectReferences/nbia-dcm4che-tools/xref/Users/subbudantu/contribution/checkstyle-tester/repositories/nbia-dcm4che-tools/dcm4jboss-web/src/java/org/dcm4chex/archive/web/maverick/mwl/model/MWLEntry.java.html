40  package org.dcm4chex.archive.web.maverick.mwl.model;
41  
42  import java.text.SimpleDateFormat;
43  import java.util.Date;
44  
45  import org.dcm4che.data.Dataset;
46  import org.dcm4che.data.DcmElement;
47  import org.dcm4che.dict.Tags;
48  
49  /**
50   * @author franz.willer
51   *
52   * Holds a single entry of the MWLList from the model.
53   * <p>
54   * This class is a wrapper over a Dataset to get values with getter methods for the view. 
55   */
56  public class MWLEntry {
57  
58      private Dataset ds;
59      private Dataset spsItem;
60  
61      private static int internalIdCounter=0;
62      private int internalId;
63  
64      /** The Date/Time formatter to format date/time values. */
65      private static final SimpleDateFormat dtformatter = new SimpleDateFormat("yyyy/MM/dd HH:mm:ss");
66      /** The Date/Time formatter to format date/time values. */
67      private static final SimpleDateFormat shortDTformatter = new SimpleDateFormat("yyyy/MM/dd HH:mm");
68      /** The Date formatter to format date values. */
69      private static final SimpleDateFormat dformatter = new SimpleDateFormat("yyyy/MM/dd");
70  
71  
72      public MWLEntry( Dataset ds ) {
73          internalId = internalIdCounter++;
74          this.ds = ds;
75          spsItem = ds.get( Tags.SPSSeq ).getItem();//scheduled procedure step sequence item.
76      }
77      
78      public int getInternalId() {
79          return internalId;
80      }
81  
82      public Dataset toDataset() {
83          return ds;
84      }
85  
86      public String getSpsID() {
87          return spsItem.getString( Tags.SPSID );
88      }
89      /**
90       * @return Returns the modality.
91       */
92      public String getModality() {
93          return spsItem.getString( Tags.Modality );
94      }
95      /**
96       * @return Returns the physiciansName.
97       */
98      public String getPhysiciansName() {
99          return spsItem.getString( Tags.ScheduledPerformingPhysicianName );
100     }
101     /**
102      * @return Returns the reqProcedureID.
103      */
104     public String getReqProcedureID() {
105         return ds.getString( Tags.RequestedProcedureID );
106     }
107 
108 
109     public String getRqSpsID() {
110         return getReqProcedureID() + '\\' + getSpsID();
111     }
112 
113     /**
114      * @return Returns the spsStartDateTime.
115      */
116     public String getSpsStartDateTime() {
117         Date d = spsItem.getDate( Tags.SPSStartDateAndTime );
118         if ( d == null ) {
119             d = spsItem.getDateTime( Tags.SPSStartDate, Tags.SPSStartTime );
120         }
121         if ( d == null ) return "";
122 
123         return shortDTformatter.format( d );
124     }
125 
126     /**
127      * @return Returns the stationAET.
128      */
129     public String getStationAET() {
130         return spsItem.getString( Tags.ScheduledStationAET );
131     }
132     /**
133      * @return Returns the stationName.
134      */
135     public String getStationName() {
136         return spsItem.getString( Tags.ScheduledStationName );
137     }
138 
139     public String getAccessionNumber() {
140         return ds.getString( Tags.AccessionNumber );
141     }
142 
143     public String getPatientName() {
144         return ds.getString( Tags.PatientName );
145     }
146     public String getPatientID() {
147         return ds.getString( Tags.PatientID );
148     }
149 
150     public String getStudyUID() {
151         return ds.getString( Tags.StudyInstanceUID );
152     }
153     public String getImgServiceReqComments() {
154         return ds.getString( Tags.ImagingServiceRequestComments );
155     }
156     public String getRequestingPhysician() {
157         return ds.getString( Tags.RequestingPhysician );
158     }
159     public String getReferringPhysicianName() {
160         return ds.getString( Tags.ReferringPhysicianName );
161     }
162     public String getPlacerOrderNumber() {
163         return ds.getString( Tags.PlacerOrderNumber );
164     }
165     public String getFillerOrderNumber() {
166         return ds.getString( Tags.FillerOrderNumber );
167     }
168     //Visit Identification
169     public String getAdmissionID() {
170         return ds.getString( Tags.AdmissionID );
171     }
172     //Patient demographic
173     public String getPatientBirthDate() {
174         Date d = ds.getDate( Tags.PatientBirthDate );
175         if ( d == null ) return "";
176         return dformatter.format( d );
177     }
178 
179     public String getPatientSex() {
180         return ds.getString( Tags.PatientSex );
181     }
182 
183     public String getSPSDescription() {
184         String desc = null;
185         DcmElement spcs = spsItem.get( Tags.ScheduledProtocolCodeSeq );
186         if ( spcs != null ) {
187             desc = getCodeValues( spcs );
188         } else {
189             desc = spsItem.getString( Tags.SPSDescription );
190         }
191         return desc;
192     }
193 
194     private String getCodeValues( DcmElement elem ) {
195         int len = elem.countItems();
196         Dataset dsCode;
197         StringBuffer sb = new StringBuffer();
198         for ( int i = 0 ; i < len ; i++ ) {
199             dsCode = elem.getItem( i );
200             if ( i > 0 ) sb.append("/");
201             sb.append( dsCode.getString( Tags.CodeMeaning ) ).append("[");
202             sb.append( dsCode.getString( Tags.CodeValue ) ).append("]");
203         }
204         return sb.toString();
205 
206     }
207 
208     public String getReqProcedureDescription() {
209         String desc = null;
210         DcmElement rpcs = ds.get( Tags.RequestedProcedureCodeSeq );
211         if ( rpcs != null ) {
212             desc = getCodeValues( rpcs );
213         } else {
214             desc = ds.getString( Tags.RequestedProcedureDescription );
215         }
216         return desc;
217     }
218 
219     public String getSpsStatus() {
220         return spsItem.getString(Tags.SPSStatus);
221     }
222 
223 }
