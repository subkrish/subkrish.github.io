40  class UIStyle {
41      public static enum CacheMode {
42          NO_CACHING, FIXED_SIZES, NINE_SQUARE_SCALE
43      }
44  
45      @XmlElement private UIColor textForeground = null;
46      @XmlElement(name="inherit-textForeground")
47      private boolean textForegroundInherited = true;
48  
49      @XmlElement private UIColor textBackground = null;
50      @XmlElement(name="inherit-textBackground")
51      private boolean textBackgroundInherited = true;
52  
53      @XmlElement private UIColor background = null;
54      @XmlElement(name="inherit-background")
55      private boolean backgroundInherited = true;
56  
57      @XmlElement private boolean cacheSettingsInherited = true;
58      @XmlElement CacheMode cacheMode = CacheMode.FIXED_SIZES;
59      @XmlElement String maxHozCachedImgScaling = "1.0";
60      @XmlElement String maxVertCachedImgScaling = "1.0";
61  
62      @XmlElement(name="uiProperty")
63      @XmlElementWrapper(name="uiproperties")
64      private List<UIProperty> uiProperties = new ArrayList<UIProperty>();
65  
66      private UIStyle parentStyle = null;
67      public void setParentStyle(UIStyle parentStyle) {
68          this.parentStyle = parentStyle;
69      }
70  
71      public CacheMode getCacheMode() {
72          if (cacheSettingsInherited) {
73              return (parentStyle == null ?
74                  CacheMode.FIXED_SIZES : parentStyle.getCacheMode());
75          } else {
76              return cacheMode;
77          }
78      }
79  
80      public String getMaxHozCachedImgScaling() {
81          if (cacheSettingsInherited) {
82              return (parentStyle == null ?
83                  "1.0" : parentStyle.getMaxHozCachedImgScaling());
84          } else {
85              return maxHozCachedImgScaling;
86          }
87      }
88  
89      public String getMaxVertCachedImgScaling() {
90          if (cacheSettingsInherited) {
91              return (parentStyle == null ?
92                  "1.0" : parentStyle.getMaxVertCachedImgScaling());
93          } else {
94              return maxVertCachedImgScaling;
95          }
96      }
97  
98      public String write(String prefix) {
99          StringBuilder sb = new StringBuilder();
100         if (! textForegroundInherited) {
101             sb.append(String.format("        addColor(d, \"%s%s\", %s);\n",
102                     prefix, "textForeground", textForeground.getValue().write()));
103         }
104         if (! textBackgroundInherited) {
105             sb.append(String.format("        addColor(d, \"%s%s\", %s);\n",
106                     prefix, "textBackground", textBackground.getValue().write()));
107         }
108         if (! backgroundInherited) {
109             sb.append(String.format("        addColor(d, \"%s%s\", %s);\n",
110                     prefix, "background", background.getValue().write()));
111         }
112         for (UIProperty property : uiProperties) {
113             sb.append(property.write(prefix));
114         }
115         return sb.toString();
116     }
117 }

119 class UIRegion {
120     @XmlAttribute protected String name;
121     @XmlAttribute protected String key;
122     @XmlAttribute private boolean opaque = false;
123 
124     @XmlElement private Insets contentMargins = new Insets(0, 0, 0, 0);
125 
126     @XmlElement(name="state")
127     @XmlElementWrapper(name="backgroundStates")
128     protected List<UIState> backgroundStates = new ArrayList<UIState>();
129     public List<UIState> getBackgroundStates() { return backgroundStates; }
130 
131     @XmlElement(name="state")
132     @XmlElementWrapper(name="foregroundStates")
133     protected List<UIState> foregroundStates = new ArrayList<UIState>();
134     public List<UIState> getForegroundStates() { return foregroundStates; }
135 
136     @XmlElement(name="state")
137     @XmlElementWrapper(name="borderStates")
138     protected List<UIState> borderStates = new ArrayList<UIState>();
139     public List<UIState> getBorderStates() { return borderStates; }
140 
141     @XmlElement private UIStyle style = new UIStyle();
142 
143     @XmlElements({
144         @XmlElement(name = "region", type = UIRegion.class),
145         @XmlElement(name = "uiComponent", type = UIComponent.class),
146         @XmlElement(name = "uiIconRegion", type = UIIconRegion.class)
147     })
148     @XmlElementWrapper(name="regions")
149     private List<UIRegion> subRegions = new ArrayList<UIRegion>();
150     public List<UIRegion> getSubRegions() { return subRegions; }
151 
152     protected void initStyles(UIStyle parentStyle) {
153         style.setParentStyle(parentStyle);
154         for (UIState state: backgroundStates) {
155             state.getStyle().setParentStyle(this.style);
156         }
157         for (UIState state: foregroundStates) {
158             state.getStyle().setParentStyle(this.style);
159         }
160         for (UIState state: borderStates) {
161             state.getStyle().setParentStyle(this.style);
162         }
163         for (UIRegion region: subRegions) {
164             region.initStyles(this.style);
165         }
166     }
167 
168     public String getKey() {
169         return key == null || "".equals(key) ? name : key;
170     }
171 
172     private boolean hasCanvas() {
173         for (UIState s : backgroundStates) {
174             if (s.hasCanvas()) return true;
175         }
176         for (UIState s : borderStates) {
177             if (s.hasCanvas()) return true;
178         }
179         for (UIState s : foregroundStates) {
180             if (s.hasCanvas()) return true;
181         }
182         for (UIRegion r: subRegions) {
183             if (r.hasCanvas()) return true;
184         }
185         return false;
186     }
187 
188     public void write(StringBuilder sb, StringBuilder styleBuffer,
189                       UIComponent comp, String prefix, String pkg) {
190         // write content margins
191         sb.append(String.format("        d.put(\"%s.contentMargins\", %s);\n",
192                                 prefix, contentMargins.write(true)));
193         // write opaque if true
194         if (opaque) {
195             sb.append(String.format("        d.put(\"%s.opaque\", Boolean.TRUE);\n", prefix));
196         }
197 
198         // register component with LAF
199         String regionCode = "Region." + Utils.regionNameToCaps(name);
200         styleBuffer.append(String.format("        register(%s, \"%s\");\n",
201                                          regionCode, prefix));
202 
203         //write the State, if necessary
204         StringBuffer regString = new StringBuffer();
205         List<UIStateType> types = comp.getStateTypes();
206         if (types != null && types.size() > 0) {
207             for (UIStateType type : types) {
208                 regString.append(type.getKey());
209                 regString.append(",");
210             }
211             //remove the last ","
212             regString.deleteCharAt(regString.length() - 1);
213         }
214 
215         if (! regString.equals("Enabled,MouseOver,Pressed,Disabled,Focused,Selected,Default") && types.size() > 0) {
216             //there were either custom states, or the normal states were in a custom order
217             //so go ahead and write out prefix.State
218             sb.append(String.format("        d.put(\"%s.States\", \"%s\");\n",
219                                     prefix, regString));
220         }
221 
222         // write out any custom states, if necessary
223         for (UIStateType type : types) {
224             String synthState = type.getKey();
225             if (! "Enabled".equals(synthState) &&
226                 ! "MouseOver".equals(synthState) &&
227                 ! "Pressed".equals(synthState) &&
228                 ! "Disabled".equals(synthState) &&
229                 ! "Focused".equals(synthState) &&
230                 ! "Selected".equals(synthState) &&
231                 ! "Default".equals(synthState)) {
232 
233                 //what we have here, gentlemen, is a bona-fide custom state.
234                 //if the type is not one of the standard types, then construct a name for
235                 //the new type, and write out a new subclass of State.
236                 String className = Utils.normalize(prefix) + synthState + "State";
237                 sb.append(String.format("        d.put(\"%s.%s\", new %s());\n",
238                                         prefix, synthState, className));
239 
240                 String body = type.getCodeSnippet();
241                 Map<String, String> variables = Generator.getVariables();
242                 variables.put("STATE_NAME", className);
243                 variables.put("STATE_KEY", synthState);
244                 variables.put("BODY", body);
245 
246                 Generator.writeSrcFile("StateImpl", variables, className);
247             }
248         }
249 
250         // write style
251         sb.append(style.write(prefix + '.'));
252 
253         String fileName = Utils.normalize(prefix) + "Painter";
254         boolean hasCanvas = hasCanvas();
255         if (hasCanvas) {
256             PainterGenerator.writePainter(this, fileName);
257         }
258         // write states ui defaults
259         for (UIState state : backgroundStates) {
260             state.write(sb, prefix, pkg, fileName, "background");
261         }
262         for (UIState state : foregroundStates) {
263             state.write(sb, prefix, pkg, fileName, "foreground");
264         }
265         for (UIState state : borderStates) {
266             state.write(sb, prefix, pkg, fileName, "border");
267         }
268 
269         // handle sub regions
270         for (UIRegion subreg : subRegions) {
271             String p = prefix;
272             if (! (subreg instanceof UIIconRegion)) {
273                 p = prefix + ":" + Utils.escape(subreg.getKey());
274             }
275             UIComponent c = comp;
276             if (subreg instanceof UIComponent) {
277                 c = (UIComponent) subreg;
278             }
279             subreg.write(sb, styleBuffer, c, p, pkg);
280         }
281     }
282 }

284 class UIIconRegion extends UIRegion {
285     @XmlAttribute private String basicKey;
286 
287     @Override public void write(StringBuilder sb, StringBuilder styleBuffer, UIComponent comp, String prefix, String pkg) {
288         Dimension size = null;
289         String fileNamePrefix = Utils.normalize(prefix) + "Painter";
290         // write states ui defaults
291         for (UIState state : backgroundStates) {
292             Canvas canvas = state.getCanvas();
293             if (!canvas.isBlank()) {
294                 state.write(sb, prefix, pkg, fileNamePrefix, getKey());
295                 size = canvas.getSize();
296             }
297         }
298 
299         if (size != null) {
300             // Put SynthIconImpl wrapper in UiDefaults
301             String k = (basicKey == null ? prefix + "." + getKey() : basicKey);
302             sb.append(String.format(
303                     "        d.put(\"%s\", new NimbusIcon(\"%s\", \"%sPainter\", %d, %d));\n",
304                     k, prefix, getKey(), size.width, size.height));
305         }
306     }
307 }

309 class UIComponent extends UIRegion {
310     @XmlAttribute private String componentName;
311 
312     @XmlElement(name="stateType")
313     @XmlElementWrapper(name="stateTypes")
314     private List<UIStateType> stateTypes = new ArrayList<UIStateType>();
315     public List<UIStateType> getStateTypes() { return stateTypes; }
316 
317     @Override public String getKey() {
318         if (key == null || "".equals(key)) {
319             if (componentName == null || "".equals(componentName)) {

323             }
324         } else {
325             return key;
326         }
327     }

330 class UIState {
331     @XmlAttribute private String stateKeys;
332     public String getStateKeys() { return stateKeys; }
333 
334     /** Indicates whether to invert the meaning of the 9-square stretching insets */
335     @XmlAttribute private boolean inverted;
336 
337     /** A cached string representing the list of stateKeys deliminated with "+" */
338     private String cachedName = null;
339 
340     @XmlElement private Canvas canvas;
341     public Canvas getCanvas() { return canvas; }
342 
343     @XmlElement private UIStyle style;
344     public UIStyle getStyle() { return style; }
345 
346     public boolean hasCanvas() {
347         return ! canvas.isBlank();
348     }
349 
350     public static List<String> stringToKeys(String keysString) {
351         return Arrays.asList(keysString.split("\\+"));
352     }
353 
354     public String getName() {
355         if (cachedName == null) {
356             StringBuilder buf = new StringBuilder();
357             List<String> keys = stringToKeys(stateKeys);
358             Collections.sort(keys);
359             for (Iterator<String> iter = keys.iterator(); iter.hasNext();) {
360                 buf.append(iter.next());
361                 if (iter.hasNext()) {
362                     buf.append('+');
363                 }
364             }
365             cachedName = buf.toString();
366         }
367         return cachedName;
368     }
369 
370     public void write(StringBuilder sb, String prefix, String pkg, String fileNamePrefix, String painterPrefix) {
371         String statePrefix = prefix + "[" + getName() + "]";
372         // write state style
373         sb.append(style.write(statePrefix + '.'));
374         // write painter
375         if (hasCanvas()) {
376             writeLazyPainter(sb, statePrefix, pkg, fileNamePrefix, painterPrefix);
377         }
378     }
379 
380     private void writeLazyPainter(StringBuilder sb, String statePrefix, String packageNamePrefix, String fileNamePrefix, String painterPrefix) {
381         String cacheModeString = "AbstractRegionPainter.PaintContext.CacheMode." + style.getCacheMode();
382         String stateConstant = Utils.statesToConstantName(painterPrefix + "_" + stateKeys);
383         sb.append(String.format(
384                 "        d.put(\"%s.%sPainter\", new LazyPainter(\"%s.%s\", %s.%s, %s, %s, %b, %s, %s, %s));\n",
385                 statePrefix, painterPrefix, packageNamePrefix, fileNamePrefix,
386                 fileNamePrefix, stateConstant, canvas.getStretchingInsets().write(false),
387                 canvas.getSize().write(false), inverted, cacheModeString,
388                 Utils.formatDouble(style.getMaxHozCachedImgScaling()),
389                 Utils.formatDouble(style.getMaxVertCachedImgScaling())));
390     }
391 }
