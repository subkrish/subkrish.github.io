139    /**
140     * Canonicalizes a Subtree node.
141     * @param input the root of the subtree to canicalize
142     * @return The canonicalize stream.
143     * @throws CanonicalizationException
144     */
145     public byte[] engineCanonicalize(XMLSignatureInput input)
146     throws CanonicalizationException {
147         try {
148                 if (input.isExcludeComments())
149                         _includeComments = false;
150                         byte[] bytes;
151                         if (input.isOctetStream()) {
152                                 return engineCanonicalize(input.getBytes());
153                         }
154                         if (input.isElement()) {
155                                 bytes = engineCanonicalizeSubTree(input.getSubNode(), input
156                                                 .getExcludeNode());
157                                 return bytes;
158                         } else if (input.isNodeSet()) {
159                                 nodeFilter=input.getNodeFilters();
160 
161                 circumventBugIfNeeded(input);
162 
163                                 if (input.getSubNode() != null) {
164                                     bytes = engineCanonicalizeXPathNodeSetInternal(input.getSubNode());
165                                 } else {
166                                     bytes = engineCanonicalizeXPathNodeSet(input.getNodeSet());
167                                 }
168                                 return bytes;
169 
170                         }
171                         return null;
172                 } catch (CanonicalizationException ex) {
173                         throw new CanonicalizationException("empty", ex);
174                 } catch (ParserConfigurationException ex) {
175                         throw new CanonicalizationException("empty", ex);
176                 } catch (IOException ex) {
177                         throw new CanonicalizationException("empty", ex);
178                 } catch (SAXException ex) {
179                         throw new CanonicalizationException("empty", ex);
180                 }
181    }

189     /**
190          * Canonicalizes a Subtree node.
191          *
192          * @param rootNode
193          *            the root of the subtree to canicalize
194          * @param excludeNode
195          *            a node to be excluded from the canicalize operation
196          * @return The canonicalize stream.
197          * @throws CanonicalizationException
198          */
199     byte[] engineCanonicalizeSubTree(Node rootNode,Node excludeNode)
200     throws CanonicalizationException {
201         this._excludeNode = excludeNode;
202         try {
203          NameSpaceSymbTable ns=new NameSpaceSymbTable();
204          int nodeLevel=NODE_BEFORE_DOCUMENT_ELEMENT;
205          if (rootNode instanceof Element) {
206                 //Fills the nssymbtable with the definitions of the parent of the root subnode
207                 getParentNameSpaces((Element)rootNode,ns);
208                 nodeLevel=NODE_NOT_BEFORE_OR_AFTER_DOCUMENT_ELEMENT;
209          }
210          this.canonicalizeSubTree(rootNode,ns,rootNode,nodeLevel);
211          this._writer.close();
212          if (this._writer instanceof ByteArrayOutputStream) {
213             byte []result=((ByteArrayOutputStream)this._writer).toByteArray();
214             if (reset) {
215                 ((ByteArrayOutputStream)this._writer).reset();
216             }
217                 return result;
218          }  else if (this._writer instanceof UnsyncByteArrayOutputStream) {
219                  byte []result=((UnsyncByteArrayOutputStream)this._writer).toByteArray();
220              if (reset) {
221                  ((UnsyncByteArrayOutputStream)this._writer).reset();
222              }
223                  return result;
224          }
225          return null;
226 
227       } catch (UnsupportedEncodingException ex) {
228          throw new CanonicalizationException("empty", ex);
229       } catch (IOException ex) {
230          throw new CanonicalizationException("empty", ex);
231       }
232    }

235    /**
236     * Method canonicalizeSubTree, this function is a recursive one.
237     *
238     * @param currentNode
239     * @param ns
240     * @param endnode
241     * @throws CanonicalizationException
242     * @throws IOException
243     */
244     final void canonicalizeSubTree(Node currentNode, NameSpaceSymbTable ns,Node endnode,
245                 int documentLevel)
246     throws CanonicalizationException, IOException {
247         if (isVisibleInt(currentNode)==-1)
248                 return;
249         Node sibling=null;
250         Node parentNode=null;
251         final OutputStream writer=this._writer;
252         final Node excludeNode=this._excludeNode;
253         final boolean includeComments=this._includeComments;
254         Map cache=new HashMap();
255         do {
256                 switch (currentNode.getNodeType()) {
257 
258                 case Node.DOCUMENT_TYPE_NODE :
259                 default :
260                         break;
261 
262                 case Node.ENTITY_NODE :
263                 case Node.NOTATION_NODE :
264                 case Node.ATTRIBUTE_NODE :
265                         // illegal node type during traversal
266                         throw new CanonicalizationException("empty");
267 
268             case Node.DOCUMENT_FRAGMENT_NODE :
269                 case Node.DOCUMENT_NODE :
270                         ns.outputNodePush();
271                         sibling= currentNode.getFirstChild();
272                         break;
273 
274                 case Node.COMMENT_NODE :
275                         if (includeComments) {
276                                 outputCommentToWriter((Comment) currentNode, writer, documentLevel);
277                         }
278                         break;
279 
280                 case Node.PROCESSING_INSTRUCTION_NODE :
281                         outputPItoWriter((ProcessingInstruction) currentNode, writer, documentLevel);
282                         break;
283 
284                 case Node.TEXT_NODE :
285                 case Node.CDATA_SECTION_NODE :
286                         outputTextToWriter(currentNode.getNodeValue(), writer);
287                         break;
288 
289                 case Node.ELEMENT_NODE :
290                         documentLevel=NODE_NOT_BEFORE_OR_AFTER_DOCUMENT_ELEMENT;
291                         if (currentNode==excludeNode) {
292                                 break;
293                         }
294                         Element currentElement = (Element) currentNode;
295                         //Add a level to the nssymbtable. So latter can be pop-back.
296                         ns.outputNodePush();
297                         writer.write('<');
298                         String name=currentElement.getTagName();
299                         UtfHelpper.writeByte(name,writer,cache);
300 
301                         Iterator attrs = this.handleAttributesSubtree(currentElement,ns);
302                         if (attrs!=null) {
303                                 //we output all Attrs which are available
304                                 while (attrs.hasNext()) {
305                                         Attr attr = (Attr) attrs.next();
306                                         outputAttrToWriter(attr.getNodeName(),attr.getNodeValue(), writer,cache);
307                                 }
308                         }
309                         writer.write('>');
310                         sibling= currentNode.getFirstChild();
311                         if (sibling==null) {
312                                 writer.write(_END_TAG);
313                                 UtfHelpper.writeStringToUtf8(name,writer);
314                                 writer.write('>');
315                                 //We fineshed with this level, pop to the previous definitions.
316                                 ns.outputNodePop();
317                                     if (parentNode != null) {
318                                         sibling= currentNode.getNextSibling();
319                                     }
320                         } else {
321                                 parentNode=currentElement;
322                         }
323                         break;
324                 }
325                 while (sibling==null  && parentNode!=null) {
326                         writer.write(_END_TAG);
327                         UtfHelpper.writeByte(((Element)parentNode).getTagName(),writer,cache);
328                         writer.write('>');
329                         //We fineshed with this level, pop to the previous definitions.
330                         ns.outputNodePop();
331                         if (parentNode==endnode)
332                                 return;
333                         sibling=parentNode.getNextSibling();
334                         parentNode=parentNode.getParentNode();
335                         if (!(parentNode instanceof Element)) {
336                                 documentLevel=NODE_AFTER_DOCUMENT_ELEMENT;
337                                 parentNode=null;
338                         }
339                 }
340                 if (sibling==null)
341                         return;
342                 currentNode=sibling;
343                 sibling=currentNode.getNextSibling();
344         } while(true);
345     }

376    /**
377     * Canoicalizes all the nodes included in the currentNode and contained in the
378         * _xpathNodeSet field.
379     *
380     * @param currentNode
381         * @param endnode
382     * @throws CanonicalizationException
383     * @throws IOException
384     */
385    final void canonicalizeXPathNodeSet(Node currentNode,Node endnode )
386            throws CanonicalizationException, IOException {
387         if (isVisibleInt(currentNode)==-1)
388                 return;
389         boolean currentNodeIsVisible = false;
390         NameSpaceSymbTable ns=new  NameSpaceSymbTable();
391         if (currentNode instanceof Element)
392                 getParentNameSpaces((Element)currentNode,ns);
393         Node sibling=null;
394         Node parentNode=null;
395         OutputStream writer=this._writer;
396         int documentLevel=NODE_BEFORE_DOCUMENT_ELEMENT;
397         Map cache=new HashMap();
398         do {
399                 switch (currentNode.getNodeType()) {
400 
401                 case Node.DOCUMENT_TYPE_NODE :
402                 default :
403                         break;
404 
405                 case Node.ENTITY_NODE :
406                 case Node.NOTATION_NODE :
407                 case Node.ATTRIBUTE_NODE :
408                         // illegal node type during traversal
409                         throw new CanonicalizationException("empty");
410 
411         case Node.DOCUMENT_FRAGMENT_NODE :
412                 case Node.DOCUMENT_NODE :
413                         ns.outputNodePush();
414                         //currentNode = currentNode.getFirstChild();
415                         sibling= currentNode.getFirstChild();
416                         break;
417 
418                 case Node.COMMENT_NODE :
419                         if (this._includeComments && (isVisibleDO(currentNode,ns.getLevel())==1)) {
420                                 outputCommentToWriter((Comment) currentNode, writer, documentLevel);
421                         }
422                         break;
423 
424                 case Node.PROCESSING_INSTRUCTION_NODE :
425                         if (isVisible(currentNode))
426                                 outputPItoWriter((ProcessingInstruction) currentNode, writer, documentLevel);
427                         break;
428 
429                 case Node.TEXT_NODE :
430                 case Node.CDATA_SECTION_NODE :
431                         if (isVisible(currentNode)) {
432                         outputTextToWriter(currentNode.getNodeValue(), writer);
433                         for (Node nextSibling = currentNode.getNextSibling();
434                     (nextSibling != null)
435                     && ((nextSibling.getNodeType() == Node.TEXT_NODE)
436                         || (nextSibling.getNodeType()
437                             == Node.CDATA_SECTION_NODE));
438                     nextSibling = nextSibling.getNextSibling()) {
439                outputTextToWriter(nextSibling.getNodeValue(), writer);
440                currentNode=nextSibling;
441                sibling=currentNode.getNextSibling();
442             }
443 
444                         }
445                         break;
446 
447                 case Node.ELEMENT_NODE :
448                         documentLevel=NODE_NOT_BEFORE_OR_AFTER_DOCUMENT_ELEMENT;
449                         Element currentElement = (Element) currentNode;
450                         //Add a level to the nssymbtable. So latter can be pop-back.
451                         String name=null;
452                         int i=isVisibleDO(currentNode,ns.getLevel());
453                         if (i==-1) {
454                                 sibling= currentNode.getNextSibling();
455                                 break;
456                         }
457                         currentNodeIsVisible=(i==1);
458                         if (currentNodeIsVisible) {
459                                 ns.outputNodePush();
460                                 writer.write('<');
461                                 name=currentElement.getTagName();
462                                 UtfHelpper.writeByte(name,writer,cache);
463                         } else {
464                                 ns.push();
465                         }
466 
467                         Iterator attrs = handleAttributes(currentElement,ns);
468                         if (attrs!=null) {
469                                 //we output all Attrs which are available
470                                 while (attrs.hasNext()) {
471                                         Attr attr = (Attr) attrs.next();
472                                         outputAttrToWriter(attr.getNodeName(),attr.getNodeValue(), writer,cache);
473                                 }
474                         }
475                         if (currentNodeIsVisible) {
476                                 writer.write('>');
477                         }
478                         sibling= currentNode.getFirstChild();
479 
480                         if (sibling==null) {
481                                 if (currentNodeIsVisible) {
482                                         writer.write(_END_TAG);
483                                         UtfHelpper.writeByte(name,writer,cache);
484                                         writer.write('>');
485                                         //We fineshed with this level, pop to the previous definitions.
486                                         ns.outputNodePop();
487                                 } else {
488                                         ns.pop();
489                                 }
490                                 if (parentNode != null) {
491                                         sibling= currentNode.getNextSibling();
492                                 }
493                         } else {
494                                 parentNode=currentElement;
495                         }
496                         break;
497                 }
498                 while (sibling==null  && parentNode!=null) {
499                         if (isVisible(parentNode)) {
500                                 writer.write(_END_TAG);
501                                 UtfHelpper.writeByte(((Element)parentNode).getTagName(),writer,cache);
502                                 writer.write('>');
503                                 //We fineshed with this level, pop to the previous definitions.
504                                 ns.outputNodePop();
505                         } else {
506                                 ns.pop();
507                         }
508                         if (parentNode==endnode)
509                                 return;
510                         sibling=parentNode.getNextSibling();
511                         parentNode=parentNode.getParentNode();
512                         if (!(parentNode instanceof Element)) {
513                                 parentNode=null;
514                                 documentLevel=NODE_AFTER_DOCUMENT_ELEMENT;
515                         }
516                 }
517                 if (sibling==null)
518                         return;
519                 currentNode=sibling;
520                 sibling=currentNode.getNextSibling();
521         } while(true);
522    }
523    int isVisibleDO(Node currentNode,int level) {
524            if (nodeFilter!=null) {
525                         Iterator it=nodeFilter.iterator();
526                         while (it.hasNext()) {
527                                 int i=((NodeFilter)it.next()).isNodeIncludeDO(currentNode,level);
528                                 if (i!=1)
529                                         return i;
530                         }
531                    }
532            if ((this._xpathNodeSet!=null) && !this._xpathNodeSet.contains(currentNode))
533                         return 0;
534            return 1;
535    }
536    int isVisibleInt(Node currentNode) {
537            if (nodeFilter!=null) {
538                 Iterator it=nodeFilter.iterator();
539                 while (it.hasNext()) {
540                         int i=((NodeFilter)it.next()).isNodeInclude(currentNode);
541                         if (i!=1)
542                                 return i;
543                 }
544            }
545                 if ((this._xpathNodeSet!=null) && !this._xpathNodeSet.contains(currentNode))
546                         return 0;
547                 return 1;
548         }

550    boolean isVisible(Node currentNode) {
551            if (nodeFilter!=null) {
552                 Iterator it=nodeFilter.iterator();
553                 while (it.hasNext()) {
554                         if (((NodeFilter)it.next()).isNodeInclude(currentNode)!=1)
555                                 return false;
556                 }
557            }
558                 if ((this._xpathNodeSet!=null) && !this._xpathNodeSet.contains(currentNode))
559                         return false;
560                 return true;
561         }

563         void handleParent(Element e,NameSpaceSymbTable ns) {
564            if (!e.hasAttributes()) {
565                                 return;
566                 }
567                 NamedNodeMap attrs = e.getAttributes();
568                 int attrsLength = attrs.getLength();
569                 for (int i = 0; i < attrsLength; i++) {
570                         Attr N = (Attr) attrs.item(i);
571                         if (Constants.NamespaceSpecNS!=N.getNamespaceURI()) {
572                                 //Not a namespace definition, ignore.
573                                 continue;
574                         }
575 
576                         String NName=N.getLocalName();
577                         String NValue=N.getNodeValue();
578                         if (XML.equals(NName)
579                && Constants.XML_LANG_SPACE_SpecNS.equals(NValue)) {
580                                         continue;
581                         }
582                         ns.addMapping(NName,NValue,N);
583                 }
584    }

586         /**
587          * Adds to ns the definitons from the parent elements of el
588          * @param el
589          * @param ns
590          */
591         final void getParentNameSpaces(Element el,NameSpaceSymbTable ns)  {
592                 List parents=new ArrayList(10);
593                 Node n1=el.getParentNode();
594                 if (!(n1 instanceof Element)) {
595                         return;
596                 }
597                 //Obtain all the parents of the elemnt
598                 Element parent=(Element) n1;
599                 while (parent!=null) {
600                         parents.add(parent);
601                         Node n=parent.getParentNode();
602                         if (!(n instanceof Element )) {
603                                 break;
604                         }
605                         parent=(Element)n;
606                 }
607                 //Visit them in reverse order.
608                 ListIterator it=parents.listIterator(parents.size());
609                 while (it.hasPrevious()) {
610                         Element ele=(Element)it.previous();
611                         handleParent(ele, ns);
612         }
613         Attr nsprefix;
614         if (((nsprefix=ns.getMappingWithoutRendered("xmlns"))!=null)
615                 && "".equals(nsprefix.getValue())) {
616              ns.addMappingAndRender("xmlns","",nullNode);
617         }
618         }

643    /**
644             * Outputs an Attribute to the internal Writer.
645             *
646             * The string value of the node is modified by replacing
647             * <UL>
648             * <LI>all ampersands (&) with <CODE>&amp;amp;</CODE></LI>
649             * <LI>all open angle brackets (<) with <CODE>&amp;lt;</CODE></LI>
650             * <LI>all quotation mark characters with <CODE>&amp;quot;</CODE></LI>
651             * <LI>and the whitespace characters <CODE>#x9</CODE>, #xA, and #xD, with character
652             * references. The character references are written in uppercase
653             * hexadecimal with no leading zeroes (for example, <CODE>#xD</CODE> is represented
654             * by the character reference <CODE>&amp;#xD;</CODE>)</LI>
655             * </UL>
656             *
657             * @param name
658             * @param value
659             * @param writer
660             * @throws IOException
661             */
662            static final void outputAttrToWriter(final String name, final String value, final OutputStream writer,
663                                 final Map cache) throws IOException {
664               writer.write(' ');
665               UtfHelpper.writeByte(name,writer,cache);
666               writer.write(equalsStr);
667               byte  []toWrite;
668               final int length = value.length();
669               int i=0;
670               while (i < length) {
671                  char c = value.charAt(i++);
672 
673                  switch (c) {
674 
675                  case '&' :
676                         toWrite=_AMP_;
677                     break;
678 
679                  case '<' :
680                         toWrite=_LT_;
681                     break;
682 
683                  case '"' :
684                         toWrite=_QUOT_;
685                     break;
686 
687                  case 0x09 :    // '\t'
688                         toWrite=__X9_;
689                     break;
690 
691                  case 0x0A :    // '\n'
692                         toWrite=__XA_;
693                     break;
694 
695                  case 0x0D :    // '\r'
696                         toWrite=__XD_;
697                     break;
698 
699                  default :
700                         if (c < 0x80 ) {
701                                 writer.write(c);
702                         } else {

809               final int length = text.length();
810               byte []toWrite;
811               for (int i = 0; i < length; i++) {
