86      // Implementation of ServiceTag.getJavaServiceTag(String) method
87      static ServiceTag getJavaServiceTag(String source) throws IOException {
88          String vendor = System.getProperty("java.vendor", "");
89          if (!vendor.startsWith(SUN) && !vendor.startsWith(ORACLE)) {
90              // Products bundling this implementation may run on
91              // Mac OS which is not a Sun/Oracle JDK
92              return null;
93          }
94          boolean cleanup = false;
95          try {
96              // Check if we have the swordfish entries for this JRE version
97              if (loadSwordfishEntries() == null) {
98                  return null;
99              }
100 
101             ServiceTag st = getJavaServiceTag();
102             // Check if the service tag created by this bundle owner
103             if (st != null && st.getSource().equals(source)) {
104                 // Install the system service tag if supported
105                 // stclient may be installed after the service tag creation
106                 if (Registry.isSupported()) {
107                     installSystemServiceTag();
108                 }
109                 return st;
110             }
111 
112             // in case any exception thrown during the cleanup
113             cleanup = true;
114 
115             // re-create a new one for this bundle owner
116             // first delete the registration data
117             deleteRegistrationData();
118             cleanup = false;
119 
120             // create service tag and generate new register.html pages
121             return createServiceTag(source);
122         } finally {
123             if (cleanup) {
124                 if (regXmlFile.exists()) {
125                     regXmlFile.delete();
126                 }
127                 if (serviceTagFile.exists()) {
128                     serviceTagFile.delete();
129                 }
130             }
131         }
132     }

134     /**
135      * Returns the Java SE registration data located in
136      * the <JRE>/lib/servicetag/registration.xml by default.
137      *
138      * @throws IllegalArgumentException if the registration data
139      *         is of invalid format.
140      */
141     private static synchronized RegistrationData getRegistrationData()
142             throws IOException {
143         if (registration != null) {
144             return registration;
145         }
146         if (regXmlFile.exists()) {
147             BufferedInputStream in = null;
148             try {
149                 in = new BufferedInputStream(new FileInputStream(regXmlFile));
150                 registration = RegistrationData.loadFromXML(in);
151             } catch (IllegalArgumentException ex) {
152                 System.err.println("Error: Bad registration data \"" +
153                                     regXmlFile + "\":" + ex.getMessage());
154                 throw ex;
155             } finally {
156                 if (in != null) {
157                     in.close();
158                 }
159             }
160         } else {
161             registration = new RegistrationData();
162         }
163         return registration;
164     }

166     /**
167      * Write the registration data to the registration.xml file.
168      *
169      * The offline registration page has to be regenerated with
170      * the new registration data.
171      *
172      * @throws java.io.IOException
173      */
174     private static synchronized void writeRegistrationXml()
175             throws IOException {
176         if (!svcTagDir.exists()) {
177             // This check is for NetBeans or other products that
178             // bundles this com.sun.servicetag implementation for
179             // pre-6u5 release.
180             if (!svcTagDir.mkdir()) {
181                 throw new IOException("Failed to create directory: " + svcTagDir);
182             }
183         }
184 
185         // regenerate the new offline registration page
186         deleteRegistrationHtmlPage();
187         getRegistrationHtmlPage();
188 
189         BufferedOutputStream out = null;
190         try {
191             out = new BufferedOutputStream(new FileOutputStream(regXmlFile));
192             getRegistrationData().storeToXML(out);
193         } catch (IllegalArgumentException ex) {
194             System.err.println("Error: Bad registration data \"" +
195                                 regXmlFile + "\":" + ex.getMessage());
196             throw ex;
197         } finally {
198             if (out != null) {
199                 out.close();
200             }
201         }
202     }

204     /**
205      * Returns the instance urn(s) stored in the servicetag file
206      * or empty set if file not exists.
207      */
208     private static Set<String> getInstalledURNs() throws IOException {
209         Set<String> urnSet = new HashSet<String>();
210         if (serviceTagFile.exists()) {
211             BufferedReader in = null;
212             try {
213                 in = new BufferedReader(new FileReader(serviceTagFile));
214                 String urn;
215                 while ((urn = in.readLine()) != null) {
216                     urn = urn.trim();
217                     if (urn.length() > 0) {
218                         urnSet.add(urn);
219                     }
220                 }
221             } finally {
222                 if (in != null) {
223                     in.close();
224                 }
225             }
226         }
227         return urnSet;
228     }

230     /**
231      * Return the Java SE service tag(s) if it exists.
232      * Typically only one Java SE service tag but it could have two for
233      * Solaris 32-bit and 64-bit on the same install directory.
234      *
235      * @return the service tag(s) for Java SE
236      */
237     private static ServiceTag[] getJavaServiceTagArray() throws IOException {
238         RegistrationData regData = getRegistrationData();
239         Set<ServiceTag> svcTags = regData.getServiceTags();
240         Set<ServiceTag> result = new HashSet<ServiceTag>();
241 
242         Properties props = loadSwordfishEntries();
243         String jdkUrn = props.getProperty("servicetag.jdk.urn");
244         String jreUrn = props.getProperty("servicetag.jre.urn");
245         for (ServiceTag st : svcTags) {
246             if (st.getProductURN().equals(jdkUrn) ||
247                 st.getProductURN().equals(jreUrn)) {
248                 result.add(st);
249             }
250         }
251         return result.toArray(new ServiceTag[0]);
252     }

254     /**
255      * Returns the Java SE service tag for this running platform;
256      * or null if not exist.
257      * This method will return the 64-bit service tag if the JDK
258      * supports both 32-bit and 64-bit if already created.
259      */
260     private static ServiceTag getJavaServiceTag() throws IOException {
261         String definedId = getProductDefinedId();
262         for (ServiceTag st : getJavaServiceTagArray()) {
263             if (st.getProductDefinedInstanceID().equals(definedId)) {
264                 return st;
265             }
266         }
267         return null;
268     }

270     /**
271      * Create a service tag for Java SE and install in the system
272      * service tag registry if supported.
273      *
274      * A registration data <JRE>/lib/servicetag/registration.xml
275      * will be created to storeToXML the XML entry for Java SE service tag.
276      * If the system supports service tags, this method will install
277      * the Java SE service tag in the system service tag registry and
278      * its <tt>instance_urn</tt> will be stored to <JRE>/lib/servicetag/servicetag.
279      *
280      * If <JRE>/lib/servicetag/registration.xml exists but is not installed
281      * in the system service tag registry (i.e. servicetag doesn't exist),
282      * this method will install it as described above.
283      *
284      * If the system supports service tag, stclient will be used
285      * to create the Java SE service tag.
286      *
287      * A Solaris 32-bit and 64-bit JDK will be installed in the same
288      * directory but the registration.xml will have 2 service tags.
289      * The servicetag file will also contain 2 instance_urns for that case.
290      */
291     private static ServiceTag createServiceTag(String svcTagSource)
292             throws IOException {
293         // determine if a new service tag is needed to be created
294         ServiceTag newSvcTag = null;
295         if (getJavaServiceTag() == null) {
296             newSvcTag = newServiceTag(svcTagSource);
297         }
298 
299         // Add the new service tag in the registration data
300         if (newSvcTag != null) {
301             RegistrationData regData = getRegistrationData();
302 
303             // Add the service tag to the registration data in JDK/JRE
304             newSvcTag = regData.addServiceTag(newSvcTag);
305 
306             // add if there is a service tag for the OS
307             ServiceTag osTag = SolarisServiceTag.getServiceTag();
308             if (osTag != null && regData.getServiceTag(osTag.getInstanceURN()) == null) {
309                 regData.addServiceTag(osTag);
310             }
311             // write to the registration.xml
312             writeRegistrationXml();
313         }
314 
315         // Install the system service tag if supported
316         if (Registry.isSupported()) {
317             installSystemServiceTag();
318         }
319         return newSvcTag;
320     }

322     private static void installSystemServiceTag() throws IOException {
323         // only install the service tag in the registry if
324         // it has permission to write the servicetag file.
325         if ((!serviceTagFile.exists() && !svcTagDir.canWrite()) ||
326                 (serviceTagFile.exists() && !serviceTagFile.canWrite())) {
327             return;
328         }
329 
330         Set<String> urns = getInstalledURNs();
331         ServiceTag[] javaSvcTags = getJavaServiceTagArray();
332         if (urns.size() < javaSvcTags.length) {
333             for (ServiceTag st : javaSvcTags) {
334                 // Add the service tag in the system service tag registry
335                 // if not installed
336                 String instanceURN = st.getInstanceURN();
337                 if (!urns.contains(instanceURN)) {
338                     Registry.getSystemRegistry().addServiceTag(st);
339                 }
340             }
341         }
342         writeInstalledUrns();
343     }

345     private static ServiceTag newServiceTag(String svcTagSource) throws IOException {
346         // Load the swoRDFish information for the service tag creation
347         Properties props = loadSwordfishEntries();
348 
349         // Determine the product URN and name
350         String productURN;
351         String productName;
352 
353         if (isJdk()) {
354             // <HOME>/jre exists which implies it's a JDK
355             productURN = props.getProperty("servicetag.jdk.urn");
356             productName = props.getProperty("servicetag.jdk.name");
357         } else {
358             // Otherwise, it's a JRE
359             productURN = props.getProperty("servicetag.jre.urn");
360             productName = props.getProperty("servicetag.jre.name");
361         }
362 
363         return ServiceTag.newInstance(ServiceTag.generateInstanceURN(),
364                                       productName,
365                                       System.getProperty("java.version"),
366                                       productURN,
367                                       props.getProperty("servicetag.parent.name"),
368                                       props.getProperty("servicetag.parent.urn"),
369                                       getProductDefinedId(),
370                                       System.getProperty("java.vendor"),
371                                       System.getProperty("os.arch"),
372                                       getZoneName(),
373                                       svcTagSource);
374     }

376     /**
377      * Delete the registration data, the offline registration pages and
378      * the service tags in the system service tag registry if installed.
379      *
380      * The registration.xml and servicetag file will be removed.
381      */
382     private static synchronized void deleteRegistrationData()
383             throws IOException {
384         try {
385             // delete the offline registration page
386             deleteRegistrationHtmlPage();
387 
388             // Remove the service tag from the system ST registry if exists
389             Set<String> urns = getInstalledURNs();
390             if (urns.size() > 0 && Registry.isSupported()) {
391                 for (String u : urns) {
392                     Registry.getSystemRegistry().removeServiceTag(u);
393                 }
394             }
395             registration = null;
396         } finally {
397             // Delete the registration.xml and servicetag files if exists
398             if (regXmlFile.exists()) {
399                 if (!regXmlFile.delete()) {
400                     throw new IOException("Failed to delete " + regXmlFile);
401                 }
402             }
403             if (serviceTagFile.exists()) {
404                 if (!serviceTagFile.delete()) {
405                     throw new IOException("Failed to delete " + serviceTagFile);
406                 }
407             }
408         }
409     }

411     /**
412      * Updates the registration data to contain one single service tag
413      * for the running Java runtime.
414      */
415     private static synchronized void updateRegistrationData(String svcTagSource)
416             throws IOException {
417         RegistrationData regData = getRegistrationData();
418         ServiceTag curSvcTag = newServiceTag(svcTagSource);
419 
420         ServiceTag[] javaSvcTags = getJavaServiceTagArray();
421         Set<String> urns = getInstalledURNs();
422         for (ServiceTag st : javaSvcTags) {
423             if (!st.getProductDefinedInstanceID().equals(curSvcTag.getProductDefinedInstanceID())) {
424                 String instanceURN = st.getInstanceURN();
425                 regData.removeServiceTag(instanceURN);
426 
427                 // remove it from the system service tag registry if exists
428                 if (urns.contains(instanceURN) && Registry.isSupported()) {
429                     Registry.getSystemRegistry().removeServiceTag(instanceURN);
430                 }
431             }
432         }
433         writeRegistrationXml();
434         writeInstalledUrns();
435     }

437     private static void writeInstalledUrns() throws IOException {
438         // if the Registry is not supported,
439         // remove the servicetag file
440         if (!Registry.isSupported() && serviceTagFile.exists()) {
441             serviceTagFile.delete();
442             return;
443         }
444 
445         PrintWriter out = null;
446         try {
447             out = new PrintWriter(serviceTagFile);
448 
449             ServiceTag[] javaSvcTags = getJavaServiceTagArray();
450             for (ServiceTag st : javaSvcTags) {
451                 // Write the instance_run to the servicetag file
452                 String instanceURN = st.getInstanceURN();
453                 out.println(instanceURN);
454             }
455         } finally {
456             if (out != null) {
457                 out.close();
458             }
459         }
460     }

462     /**
463      * Load the values associated with the swoRDFish metadata entries
464      * for Java SE. The swoRDFish metadata entries are different for
465      * different release.
466      *
467      * @param version Version of Java SE
468      */
469     private static synchronized Properties loadSwordfishEntries() throws IOException {
470         if (swordfishProps != null) {
471             return swordfishProps;
472         }
473 
474         // The version string for Java SE 6 is 1.6.0
475         // We just need the minor number in the version string
476         int version = Util.getJdkVersion();
477 
478         String filename = "/com/sun/servicetag/resources/javase_" +
479                 version + "_swordfish.properties";
480         InputStream in = Installer.class.getResourceAsStream(filename);
481         if (in == null) {
482             return null;
483         }
484         swordfishProps = new Properties();
485         try {
486             swordfishProps.load(in);
487         } finally {
488             in.close();
489         }
490         return swordfishProps;
491     }

493     /**
494      * Returns the product defined instance ID for Java SE.
495      * It is a list of comma-separated name/value pairs:
496      *    "id=<full-version>  <arch> [<arch>]*"
497      *    "dir=<java.home system property value>"
498      *
499      * where <full-version> is the full version string of the JRE,
500      *       <arch> is the architecture that the runtime supports
501      *       (i.e. "sparc", "sparcv9", "i386", "amd64" (ISA list))
502      *
503      * For Solaris, it can be dual mode that can support both
504      * 32-bit and 64-bit. the "id" will be set to
505      *     "1.6.0_03-b02 sparc sparcv9"
506      *
507      * The "dir" property is included in the service tag to enable
508      * the Service Tag software to determine if a service tag for
509      * Java SE is invalid and perform appropriate service tag
510      * cleanup if necessary.  See RFE# 6574781 Service Tags Enhancement.
511      *
512      */
513     private static String getProductDefinedId() {
514         StringBuilder definedId = new StringBuilder();
515         definedId.append("id=");
516         definedId.append(System.getProperty("java.runtime.version"));
517 
518         String[] archs = getJreArchs();
519         for (String name : archs) {
520             definedId.append(" " + name);
521         }
522 
523         String location = ",dir=" + javaHome;
524         if ((definedId.length() + location.length()) < 256) {
525             definedId.append(",dir=");
526             definedId.append(javaHome);
527         } else {
528             // if it exceeds the limit, we will not include the location
529             if (isVerbose()) {
530                 System.err.println("Warning: Product defined instance ID exceeds the field limit:");
531             }
532         }
533 
534         return definedId.toString();
535     }

537     /**
538      * Returns the architectures that the runtime supports
539      *  (i.e. "sparc", "sparcv9", "i386", "amd64" (ISA list))
540      * The directory name where libjava.so is located.
541      *
542      * On Windows, returns the "os.arch" system property value.
543      */
544     private synchronized static String[] getJreArchs() {
545         if (jreArchs != null) {
546             return jreArchs;
547         }
548 
549         Set<String> archs = new HashSet<String>();
550 
551         String os = System.getProperty("os.name");
552         if (os.equals("SunOS") || os.equals("Linux")) {
553             // Traverse the directories under <JRE>/lib.
554             // If <JRE>/lib/<arch>/libjava.so exists, add <arch>
555             // to the product defined ID
556             File dir = new File(getJrePath() + File.separator + "lib");
557             if (dir.isDirectory()) {
558                 String[] children = dir.list();
559                 for (String name : children) {
560                     File f = new File(dir, name + File.separator + "libjava.so");
561                     if (f.exists()) {
562                         archs.add(name);
563                     }
564                 }
565             }
566         } else {
567             // Windows - append the os.arch
568             archs.add(System.getProperty("os.arch"));
569         }
570         jreArchs = archs.toArray(new String[0]);
571         return jreArchs;
572     }

574     /**
575      * Return the zonename if zone is supported; otherwise, return
576      * "global".
577      */
578     private static String getZoneName() throws IOException {
579         String zonename = "global";
580 
581         String command = "/usr/bin/zonename";
582         File f = new File(command);
583         // com.sun.servicetag package has to be compiled with JDK 5 as well
584         // JDK 5 doesn't support the File.canExecute() method.
585         // Risk not checking isExecute() for the zonename command is very low.
586         if (f.exists()) {
587             ProcessBuilder pb = new ProcessBuilder(command);
588             Process p = pb.start();
589             String output = commandOutput(p);
590             if (p.exitValue() == 0) {
591                 zonename = output.trim();
592             }
593 
594         }
595         return zonename;
596     }

598     private synchronized static String getRegisterHtmlParent() throws IOException {
599         if (registerHtmlParent == null) {
600             File htmlDir;    // register.html is put under the JDK directory
601             if (getJrePath().endsWith(File.separator + "jre")) {
602                 htmlDir = new File(getJrePath(), "..");
603             } else {
604                 // j2se non-image build
605                 htmlDir = new File(getJrePath());
606             }
607 
608             // initialize the supported locales
609             initSupportedLocales(htmlDir);
610 
611             // Determine the location of the offline registration page
612             String path = System.getProperty(SVCTAG_DIR_PATH);
613             if (path == null) {
614                 // Default is <JDK>/register.html
615                 registerHtmlParent = htmlDir.getCanonicalPath();
616             } else {
617                 File f = new File(path);
618                 registerHtmlParent = f.getCanonicalPath();
619                 if (!f.isDirectory()) {
620                     throw new InternalError("Path " + path + " set in \"" +
621                             SVCTAG_DIR_PATH + "\" property is not a directory");
622                 }
623             }
624         }
625         return registerHtmlParent;
626     }

628     /**
629      * Returns the File object of the offline registration page localized
630      * for the default locale in the JDK directory.
631      */
632     static synchronized File getRegistrationHtmlPage() throws IOException {
633         if (!supportRegistration) {
634             // No register.html page generated if JRE
635             return null;
636         }
637 
638         String parent = getRegisterHtmlParent();
639 
640         // check if the offline registration page is already generated
641         File f = new File(parent, REGISTRATION_HTML_NAME + ".html");
642         if (!f.exists()) {
643             // Generate the localized version of the offline registration Page
644             generateRegisterHtml(parent);
645         }
646 
647         String name = REGISTRATION_HTML_NAME;
648         Locale locale = getDefaultLocale();
649         if (!locale.equals(Locale.ENGLISH) && supportedLocales.contains(locale)) {
650             // if the locale is not English and is supported by JDK
651             // set to the appropriate offline registration page;
652             // otherwise,set to register.html.
653             name = REGISTRATION_HTML_NAME + "_" + locale.toString();
654         }
655         File htmlFile = new File(parent, name + ".html");
656         if (isVerbose()) {
657             System.out.print("Offline registration page: " + htmlFile);
658             System.out.println((htmlFile.exists() ?
659                                "" : " not exist. Use register.html"));
660         }
661         if (htmlFile.exists()) {
662             return htmlFile;
663         } else {
664             return new File(parent,
665                             REGISTRATION_HTML_NAME + ".html");
666         }
667     }

669     private static Locale getDefaultLocale() {
670         List<Locale> candidateLocales = getCandidateLocales(Locale.getDefault());
671         for (Locale l : candidateLocales) {
672             if (supportedLocales.contains(l)) {
673                 return l;
674             }
675         }
676         return Locale.getDefault();
677     }

679     private static List<Locale> getCandidateLocales(Locale locale) {
680         String language = locale.getLanguage();
681         String country = locale.getCountry();
682         String variant = locale.getVariant();
683 
684         List<Locale> locales = new ArrayList<Locale>(3);
685         if (variant.length() > 0) {
686             locales.add(locale);
687         }
688         if (country.length() > 0) {
689             locales.add((locales.size() == 0) ?
690                         locale : new Locale(language, country, ""));
691         }
692         if (language.length() > 0) {
693             locales.add((locales.size() == 0) ?
694                         locale : new Locale(language, "", ""));
695         }
696         return locales;
697     }

699     // Remove the offline registration pages
700     private static void deleteRegistrationHtmlPage() throws IOException {
701         String parent = getRegisterHtmlParent();
702         if (parent == null) {
703             return;
704         }
705 
706         for (Locale locale : supportedLocales) {
707             String name = REGISTRATION_HTML_NAME;
708             if (!locale.equals(Locale.ENGLISH)) {
709                 name += "_" + locale.toString();
710             }
711             File f = new File(parent, name + ".html");
712             if (f.exists()) {
713                 if (!f.delete()) {
714                     throw new IOException("Failed to delete " + f);
715                 }
716             }

720     private static void initSupportedLocales(File jdkDir) {
721         if (supportedLocales.isEmpty()) {
722             // initialize with the known supported locales
723             for (Locale l : knownSupportedLocales) {
724                 supportedLocales.add(l);
725             }
726         }
727 
728         // Determine unknown supported locales if any
729         // by finding the localized version of README.html
730         // This prepares if a new locale in JDK is supported in
731         // e.g. in the OpenSource world
732         FilenameFilter ff = new FilenameFilter() {
733             public boolean accept(File dir, String name) {
734                 String fname = name.toLowerCase();
735                 if (fname.startsWith("readme") && fname.endsWith(".html")) {
736                     return true;
737                 }
738                 return false;
739             }
740         };
741 
742         String[] readmes = jdkDir.list(ff);
743         for (String name : readmes) {
744             String basename = name.substring(0, name.length() - ".html".length());
745             String[] ss = basename.split("_");
746             switch (ss.length) {
747                 case 1:
748                     // English version
749                     break;
750                 case 2:
751                     supportedLocales.add(new Locale(ss[1]));
752                     break;
753                 case 3:
754                     supportedLocales.add(new Locale(ss[1], ss[2]));
755                     break;
756                 default:
757                     // ignore
758                     break;
759             }
760         }
761         if (isVerbose()) {
762             System.out.println("Supported locales: ");
763             for (Locale l : supportedLocales) {
764                 System.out.println(l);
765             }
766         }
767     }

769     private static final String JDK_HEADER_PNG_KEY = "@@JDK_HEADER_PNG@@";
770     private static final String JDK_VERSION_KEY = "@@JDK_VERSION@@";
771     private static final String REGISTRATION_URL_KEY = "@@REGISTRATION_URL@@";
772     private static final String REGISTRATION_PAYLOAD_KEY = "@@REGISTRATION_PAYLOAD@@";
773 
774     @SuppressWarnings("unchecked")
775     private static void generateRegisterHtml(String parent) throws IOException {
776         int version = Util.getJdkVersion();
777         int update = Util.getUpdateVersion();
778         String jdkVersion = "Version " + version;
779         if (update > 0) {
780             // product name is not translated
781             jdkVersion += " Update " + update;
782         }
783         RegistrationData regData = getRegistrationData();
784         // Make sure it uses the canonical path before getting the URI.
785         File img = new File(svcTagDir.getCanonicalPath(), "jdk_header.png");
786         String headerImageSrc = img.toURI().toString();
787 
788         // Format the registration data in one single line
789         StringBuilder payload = new StringBuilder();
790         String xml = regData.toString().replaceAll("\"", "%22");
791         BufferedReader reader = new BufferedReader(new StringReader(xml));
792         try {
793             String line = null;
794             while ((line = reader.readLine()) != null) {
795                 payload.append(line.trim());
796             }
797         } finally {
798             reader.close();
799         }
800 
801         String resourceFilename = "/com/sun/servicetag/resources/register";
802         for (Locale locale : supportedLocales) {
803             String name = REGISTRATION_HTML_NAME;
804             String resource = resourceFilename;
805             if (!locale.equals(Locale.ENGLISH)) {
806                 name += "_" + locale.toString();
807                 resource += "_" + locale.toString();
808             }
809             File f = new File(parent, name + ".html");
810             InputStream in = null;
811             BufferedReader br = null;
812             PrintWriter pw = null;
813             String registerURL = SunConnection.
814                 getRegistrationURL(regData.getRegistrationURN(),
815                                    locale,
816                                    String.valueOf(version)).toString();
817             try {
818                 in = Installer.class.getResourceAsStream(resource + ".html");
819                 if (in == null) {
820                     // if the resource file is missing
821                     if (isVerbose()) {
822                         System.out.println("Missing resouce file: " + resource + ".html");
823                     }
824                     continue;
825                 }
826                 if (isVerbose()) {
827                     System.out.println("Generating " + f + " from " + resource + ".html");
828                 }
829 
830                 try {
831                     br = new BufferedReader(new InputStreamReader(in, "UTF-8"));
832                     pw = new PrintWriter(f, "UTF-8");
833                     String line = null;
834                     while ((line = br.readLine()) != null) {
835                         String output = line;
836                         if (line.contains(JDK_VERSION_KEY)) {
837                             output = line.replace(JDK_VERSION_KEY, jdkVersion);
838                         } else if (line.contains(JDK_HEADER_PNG_KEY)) {
839                             output = line.replace(JDK_HEADER_PNG_KEY, headerImageSrc);
840                         } else if (line.contains(REGISTRATION_URL_KEY)) {
841                             output = line.replace(REGISTRATION_URL_KEY, registerURL);
842                         } else if (line.contains(REGISTRATION_PAYLOAD_KEY)) {
843                             output = line.replace(REGISTRATION_PAYLOAD_KEY, payload.toString());
844                         }
845                         pw.println(output);
846                     }
847                     f.setReadOnly();
848                     pw.flush();
849                 } finally {
850                     // It's safe for this finally block to have two close statements
851                     // consecutively as PrintWriter.close doesn't throw IOException.
852                     if (pw != null) {
853                         pw.close();
854                     }
855                     if (br!= null) {
856                         br.close();
857                     }
858                 }
859             } finally {
860                 if (in != null) {
861                     in.close();
862                 }
863             }
864         }
865     }

867     private static final int MAX_SOURCE_LEN = 63;
868 
869     /**
870      * A utility class to create a service tag for Java SE.
871      * <p>
872      * <b>Usage:</b><br>
873      * <blockquote><tt>
874      * &lt;JAVA_HOME&gt;/bin/java com.sun.servicetag.Installer
875      * </tt></blockquote>
876      * <p>
877      */
878     public static void main(String[] args) {
879         String source = "Manual ";
880         String runtimeName = System.getProperty("java.runtime.name");
881         if (runtimeName.startsWith("OpenJDK")) {
882             source = "OpenJDK ";
883         }
884         source += System.getProperty("java.runtime.version");
885         if (source.length() > MAX_SOURCE_LEN) {
886             source = source.substring(0, MAX_SOURCE_LEN);
887         }
888 
889         // Parse the options (arguments starting with "-" )
890         boolean delete = false;
891         boolean update = false;
892         boolean register = false;
893         int count = 0;
894         while (count < args.length) {
895             String arg = args[count];
896             if (arg.trim().length() == 0) {
897                 // skip empty arguments
898                 count++;
899                 continue;
900             }
901 
902             if (arg.equals("-source")) {
903                 source = args[++count];
904             } else if (arg.equals("-delete")) {
905                 delete = true;
906             } else if (arg.equals("-register")) {
907                 register = true;
908             } else {
909                 usage();
910                 return;
911             }
912             count++;
913         }
914         try {
915             if (delete) {
916                 deleteRegistrationData();
917             } else {
918                 ServiceTag[] javaSvcTags = getJavaServiceTagArray();
919                 String[] archs = getJreArchs();
920                 if (javaSvcTags.length > archs.length) {
921                     // 64-bit has been uninstalled
922                     // so remove the service tag
923                     updateRegistrationData(source);
924                 } else {
925                     // create the service tag
926                     createServiceTag(source);
927                 }
928             }
929 
930             if (register) {
931                 // Registration is only supported by JDK
932                 // For testing purpose, override with a "servicetag.enable.registration" property
933 
934                 RegistrationData regData = getRegistrationData();
935                 if (supportRegistration && !regData.getServiceTags().isEmpty()) {
936                     SunConnection.register(regData,
937                                            getDefaultLocale(),
938                                            String.valueOf(Util.getJdkVersion()));
939                 }
940             }
941             System.exit(0);
942         } catch (IOException e) {
943             System.err.println("I/O Error: " + e.getMessage());
944             if (isVerbose()) {
945                 e.printStackTrace();
946             }
947         } catch (IllegalArgumentException ex) {
948             if (isVerbose()) {
949                 ex.printStackTrace();
950             }
951         } catch (Exception e) {
952             System.err.println("Error: " + e.getMessage());
953             if (isVerbose()) {
954                 e.printStackTrace();
955             }
956         }
957         System.exit(1);
958     }
