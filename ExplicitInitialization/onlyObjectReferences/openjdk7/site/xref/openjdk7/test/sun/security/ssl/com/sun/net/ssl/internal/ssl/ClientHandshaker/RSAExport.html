326      */
327     void doServerSide() throws Exception {
328         SSLServerSocketFactory sslssf =
329                 getSSLContext(true).getServerSocketFactory();
330         SSLServerSocket sslServerSocket =
331                 (SSLServerSocket) sslssf.createServerSocket(serverPort);
332 
333         serverPort = sslServerSocket.getLocalPort();
334 
335         /*
336          * Signal Client, we're ready for this connect.
337          */
338         serverReady = true;
339 
340         // Enable RSA_EXPORT cipher suites only.
341         try {
342             String enabledSuites[] = {
343                 "SSL_RSA_EXPORT_WITH_RC4_40_MD5",
344                 "SSL_RSA_EXPORT_WITH_DES40_CBC_SHA"};
345             sslServerSocket.setEnabledCipherSuites(enabledSuites);
346         } catch (IllegalArgumentException iae) {
347             // ignore the exception a cipher suite is unsupported.
348         }
349 
350         SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
351         InputStream sslIS = sslSocket.getInputStream();
352         OutputStream sslOS = sslSocket.getOutputStream();
353 
354         sslIS.read();
355         sslOS.write(85);
356         sslOS.flush();
357 
358 
359         sslSocket.close();
360     }

423     Thread clientThread = null;
424     Thread serverThread = null;

426     /*
427      * Primary constructor, used to drive remainder of the test.
428      *
429      * Fork off the other side, then do your work.
430      */
431     RSAExport() throws Exception {
432         if (separateServerThread) {
433             startServer(true);
434             startClient(false);
435         } else {
436             startClient(true);
437             startServer(false);
438         }
439 
440         /*
441          * Wait for other side to close down.
442          */
443         if (separateServerThread) {
444             serverThread.join();
445         } else {
446             clientThread.join();
447         }
448 
449         /*
450          * When we get here, the test is pretty much over.
451          *
452          * If the main thread excepted, that propagates back
453          * immediately.  If the other thread threw an exception, we
454          * should report back.
455          */
456         if (serverException != null)
457             throw serverException;
458         if (clientException != null)
459             throw clientException;
460     }

462     void startServer(boolean newThread) throws Exception {
463         if (newThread) {
464             serverThread = new Thread() {
465                 public void run() {
466                     try {
467                         doServerSide();
468                     } catch (Exception e) {
469                         /*
470                          * Our server thread just died.
471                          *
472                          * Release the client, if not active already...
473                          */
474                         System.err.println("Server died..." + e);
475                         serverReady = true;
476                         serverException = e;
477                     }
478                 }
479             };
480             serverThread.start();
481         } else {
482             doServerSide();
483         }
484     }

487         if (newThread) {
488             clientThread = new Thread() {
489                 public void run() {
490                     try {
491                         doClientSide();
492                     } catch (Exception e) {
493                         /*
494                          * Our client thread just died.
495                          */
496                         System.err.println("Client died...");
497                         clientException = e;
498                     }
499                 }
500             };
501             clientThread.start();
502         } else {
503             doClientSide();
504         }
505     }

507     // Get the SSL context
508     private SSLContext getSSLContext(boolean authnRequired) throws Exception {
509         // generate certificate from cert string
510         CertificateFactory cf = CertificateFactory.getInstance("X.509");
511 
512         ByteArrayInputStream is =
513                     new ByteArrayInputStream(trusedCertStr.getBytes());
514         Certificate trustedCert = cf.generateCertificate(is);
515 
516         // create a key store
517         KeyStore ks = KeyStore.getInstance("JKS");
518         ks.load(null, null);
519 
520         // import the trusted cert
521         ks.setCertificateEntry("RSA Export Signer", trustedCert);
522 
523         if (authnRequired) {
524             // generate the private key.
525             RSAPrivateKeySpec priKeySpec = new RSAPrivateKeySpec(
526                                             new BigInteger(modulus),
527                                             new BigInteger(privateExponent));
528             KeyFactory kf = KeyFactory.getInstance("RSA");
529             RSAPrivateKey priKey =
530                     (RSAPrivateKey)kf.generatePrivate(priKeySpec);
531 
532             // generate certificate chain
533             is = new ByteArrayInputStream(serverCertStr.getBytes());
534             Certificate serverCert = cf.generateCertificate(is);
535 
536             Certificate[] chain = new Certificate[2];
537             chain[0] = serverCert;
538             chain[1] = trustedCert;
539 
540             // import the key entry.
541             ks.setKeyEntry("RSA Export", priKey, passphrase, chain);
542         }
543 
544         // create SSL context
545         TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
546         tmf.init(ks);
547 
548         SSLContext ctx = SSLContext.getInstance("TLS");
549         if (authnRequired) {
550             KeyManagerFactory kmf = KeyManagerFactory.getInstance("SunX509");
551             kmf.init(ks, passphrase);
552 
553             ctx.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
554         } else {
555             ctx.init(null, tmf.getTrustManagers(), null);
556         }
557 
558         return ctx;
559     }
