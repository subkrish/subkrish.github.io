34  /**
35   * Collection of relocatable constant pool references.
36   * It operates with respect to a particular byte array,
37   * and stores some of its state in the bytes themselves.
38   * <p>
39   * As a Collection, it can be iterated over, but it is not a List,
40   * since it does not natively support indexed access.
41   * <p>
42   *
43   * @author John Rose
44   */
45  final class Fixups extends AbstractCollection {
46      byte[] bytes;    // the subject of the relocations
47      int head;        // desc locating first reloc
48      int tail;        // desc locating last reloc
49      int size;        // number of relocations
50      Entry[] entries; // [0..size-1] relocations
51      int[] bigDescs;  // descs which cannot be stored in the bytes
52  
53      // A "desc" (descriptor) is a bit-encoded pair of a location
54      // and format.  Every fixup occurs at a "desc".  Until final
55      // patching, bytes addressed by descs may also be used to
56      // link this data structure together.  If the bytes are missing,
57      // or if the "desc" is too large to encode in the bytes,
58      // it is kept in the bigDescs array.
59  
60      Fixups(byte[] bytes) {
61          this.bytes = bytes;
62          entries = new Entry[3];
63          bigDescs = noBigDescs;
64      }
65      Fixups() {
66          // If there are no bytes, all descs are kept in bigDescs.
67          this((byte[])null);
68      }
69      Fixups(byte[] bytes, Collection fixups) {
70          this(bytes);
71          addAll(fixups);
72      }
73      Fixups(Collection fixups) {
74          this((byte[])null);
75          addAll(fixups);
76      }

78      private static final int MINBIGSIZE = 1;
79      // cleverly share empty bigDescs:
80      private static int[] noBigDescs = {MINBIGSIZE};

82      public int size() {
83          return size;
84      }

86      public void trimToSize() {
87          if (size != entries.length) {
88              Entry[] oldEntries = entries;
89              entries = new Entry[size];
90              System.arraycopy(oldEntries, 0, entries, 0, size);
91          }
92          int bigSize = bigDescs[BIGSIZE];
93          if (bigSize == MINBIGSIZE) {
94              bigDescs = noBigDescs;
95          } else if (bigSize != bigDescs.length) {
96              int[] oldBigDescs = bigDescs;
97              bigDescs = new int[bigSize];
98              System.arraycopy(oldBigDescs, 0, bigDescs, 0, bigSize);
99          }
100     }

102     public void visitRefs(Collection<Entry> refs) {
103         for (int i = 0; i < size; i++) {
104             refs.add(entries[i]);
105         }
106     }

108     public void clear() {
109         if (bytes != null) {
110             // Clean the bytes:
111             for (Iterator i = iterator(); i.hasNext(); ) {
112                 Fixup fx = (Fixup) i.next();
113                 //System.out.println("clean "+fx);
114                 storeIndex(fx.location(), fx.format(), 0);
115             }
116         }
117         size = 0;
118         if (bigDescs != noBigDescs)
119             bigDescs[BIGSIZE] = MINBIGSIZE;
120         // do not trim to size, however
121     }

127     @SuppressWarnings("unchecked")
128     public void setBytes(byte[] newBytes) {
129         if (bytes == newBytes)  return;
130         ArrayList old = null;
131         assert((old = new ArrayList(this)) != null);
132         if (bytes == null || newBytes == null) {
133             // One or the other representations is deficient.
134             // Construct a checkpoint.
135             ArrayList save = new ArrayList(this);
136             clear();
137             bytes = newBytes;
138             addAll(save);
139         } else {
140             // assume newBytes is some sort of bitwise copy of the old bytes
141             bytes = newBytes;
142         }
143         assert(old.equals(new ArrayList(this)));
144     }

146     static final int LOC_SHIFT = 1;
147     static final int FMT_MASK = 0x1;
148     static final byte UNUSED_BYTE = 0;
149     static final byte OVERFLOW_BYTE = -1;
150     // fill pointer of bigDescs array is in element [0]
151     static final int BIGSIZE = 0;
152 
153     // Format values:
154     public static final int U2_FORMAT = 0;
155     public static final int U1_FORMAT = 1;
156 
157     // Special values for the static methods.
158     private static final int SPECIAL_LOC = 0;
159     private static final int SPECIAL_FMT = U2_FORMAT;
160 
161     static int fmtLen(int fmt) { return 1+(fmt-U1_FORMAT)/(U2_FORMAT-U1_FORMAT); }
162     static int descLoc(int desc) { return desc >>> LOC_SHIFT; }
163     static int descFmt(int desc) { return desc  &  FMT_MASK; }
164     static int descEnd(int desc) { return descLoc(desc) + fmtLen(descFmt(desc)); }
165     static int makeDesc(int loc, int fmt) {
166         int desc = (loc << LOC_SHIFT) | fmt;
167         assert(descLoc(desc) == loc);
168         assert(descFmt(desc) == fmt);
169         return desc;
170     }
171     int fetchDesc(int loc, int fmt) {
172         byte b1 = bytes[loc];
173         assert(b1 != OVERFLOW_BYTE);
174         int value;
175         if (fmt == U2_FORMAT) {
176             byte b2 = bytes[loc+1];
177             value = ((b1 & 0xFF) << 8) + (b2 & 0xFF);
178         } else {
179             value = (b1 & 0xFF);
180         }
181         // Stored loc field is difference between its own loc and next loc.
182         return value + (loc << LOC_SHIFT);
183     }
184     boolean storeDesc(int loc, int fmt, int desc) {
185         if (bytes == null)
186             return false;
187         int value = desc - (loc << LOC_SHIFT);
188         byte b1, b2;
189         switch (fmt) {
190         case U2_FORMAT:
191             assert(bytes[loc+0] == UNUSED_BYTE);
192             assert(bytes[loc+1] == UNUSED_BYTE);
193             b1 = (byte)(value >> 8);
194             b2 = (byte)(value >> 0);
195             if (value == (value & 0xFFFF) && b1 != OVERFLOW_BYTE) {
196                 bytes[loc+0] = b1;
197                 bytes[loc+1] = b2;
198                 assert(fetchDesc(loc, fmt) == desc);
199                 return true;
200             }
201             break;
202         case U1_FORMAT:
203             assert(bytes[loc] == UNUSED_BYTE);
204             b1 = (byte)value;
205             if (value == (value & 0xFF) && b1 != OVERFLOW_BYTE) {
206                 bytes[loc] = b1;
207                 assert(fetchDesc(loc, fmt) == desc);
208                 return true;
209             }
210             break;
211         default: assert(false);
212         }
213         // Failure.  Caller must allocate a bigDesc.
214         bytes[loc] = OVERFLOW_BYTE;
215         assert(fmt==U1_FORMAT || (bytes[loc+1]=(byte)bigDescs[BIGSIZE])!=999);
216         return false;
217     }
218     void storeIndex(int loc, int fmt, int value) {
219         storeIndex(bytes, loc, fmt, value);
220     }
221     static
222     void storeIndex(byte[] bytes, int loc, int fmt, int value) {
223         switch (fmt) {
224         case U2_FORMAT:
225             assert(value == (value & 0xFFFF)) : (value);
226             bytes[loc+0] = (byte)(value >> 8);
227             bytes[loc+1] = (byte)(value >> 0);
228             break;
229         case U1_FORMAT:
230             assert(value == (value & 0xFF)) : (value);
231             bytes[loc] = (byte)value;
232             break;
233         default: assert(false);
234         }
235     }

237     /** Simple and necessary tuple to present each fixup. */
238     public static
239     class Fixup implements Comparable {
240         int desc;         // location and format of reloc
241         Entry entry;      // which entry to plug into the bytes
242         Fixup(int desc, Entry entry) {
243             this.desc = desc;
244             this.entry = entry;
245         }
246         public Fixup(int loc, int fmt, Entry entry) {
247             this.desc = makeDesc(loc, fmt);
248             this.entry = entry;
249         }
250         public int location() { return descLoc(desc); }
251         public int format() { return descFmt(desc); }
252         public Entry entry() { return entry; }
253         public int compareTo(Fixup that) {
254             // Ordering depends only on location.
255             return this.location() - that.location();
256         }
257         public int compareTo(Object that) {
258             return compareTo((Fixup)that);
259         }
260         public boolean equals(Object x) {
261             if (!(x instanceof Fixup))  return false;
262             Fixup that = (Fixup) x;
263             return this.desc == that.desc && this.entry == that.entry;
264         }
265         public String toString() {
266             return "@"+location()+(format()==U1_FORMAT?".1":"")+"="+entry;
267         }
268     }

270     private
271     class Itr implements Iterator {
272         int index = 0;               // index into entries
273         int bigIndex = BIGSIZE+1;    // index into bigDescs
274         int next = head;             // desc pointing to next fixup
275         public boolean hasNext() { return index < size; }
276         public void remove() { throw new UnsupportedOperationException(); }
277         public Object next() {
278             int thisIndex = index;
279             return new Fixup(nextDesc(), entries[thisIndex]);
280         }
281         int nextDesc() {
282             index++;
283             int thisDesc = next;
284             if (index < size) {
285                 // Fetch next desc eagerly, in case this fixup gets finalized.
286                 int loc = descLoc(thisDesc);
287                 int fmt = descFmt(thisDesc);
288                 if (bytes != null && bytes[loc] != OVERFLOW_BYTE) {
289                     next = fetchDesc(loc, fmt);
290                 } else {
291                     // The unused extra byte is "asserted" to be equal to BI.
292                     // This helps keep the overflow descs in sync.
293                     assert(fmt==U1_FORMAT || bytes == null || bytes[loc+1]==(byte)bigIndex);
294                     next = bigDescs[bigIndex++];
295                 }
296             }
297             return thisDesc;
298         }
299     }

314     @SuppressWarnings("unchecked")
315     public boolean addAll(Collection c) {
316         if (c instanceof Fixups) {
317             // Use knowledge of Itr structure to avoid building little structs.
318             Fixups that = (Fixups) c;
319             if (that.size == 0)  return false;
320             if (this.size == 0 && entries.length < that.size)
321                 growEntries(that.size);  // presize exactly
322             Entry[] thatEntries = that.entries;
323             for (Itr i = that.new Itr(); i.hasNext(); ) {
324                 int ni = i.index;
325                 addDesc(i.nextDesc(), thatEntries[ni]);
326             }
327             return true;
328         } else {
329             return super.addAll(c);
330         }
331     }
332     // Here is how things get added:
333     private void addDesc(int thisDesc, Entry entry) {
334         if (entries.length == size)
335             growEntries(size * 2);
336         entries[size] = entry;
337         if (size == 0) {
338             head = tail = thisDesc;
339         } else {
340             int prevDesc = tail;
341             // Store new desc in previous tail.
342             int prevLoc = descLoc(prevDesc);
343             int prevFmt = descFmt(prevDesc);
344             int prevLen = fmtLen(prevFmt);
345             int thisLoc = descLoc(thisDesc);
346             // The collection must go in ascending order, and not overlap.
347             if (thisLoc < prevLoc + prevLen)
348                 badOverlap(thisLoc);
349             tail = thisDesc;
350             if (!storeDesc(prevLoc, prevFmt, thisDesc)) {
351                 // overflow
352                 int bigSize = bigDescs[BIGSIZE];
353                 if (bigDescs.length == bigSize)
354                     growBigDescs();
355                 //System.out.println("bigDescs["+bigSize+"] = "+thisDesc);
356                 bigDescs[bigSize++] = thisDesc;
357                 bigDescs[BIGSIZE] = bigSize;
358             }
359         }
360         size += 1;
361     }
362     private void badOverlap(int thisLoc) {
363         throw new IllegalArgumentException("locs must be ascending and must not overlap:  "+thisLoc+" >> "+this);
364     }
365 
366     private void growEntries(int newSize) {
367         Entry[] oldEntries = entries;
368         entries = new Entry[Math.max(3, newSize)];
369         System.arraycopy(oldEntries, 0, entries, 0, oldEntries.length);
370     }
371     private void growBigDescs() {
372         int[] oldBigDescs = bigDescs;
373         bigDescs = new int[oldBigDescs.length * 2];
374         System.arraycopy(oldBigDescs, 0, bigDescs, 0, oldBigDescs.length);
375     }

377     /// Static methods that optimize the use of this class.
378     public static
379     Object add(Object prevFixups,
380                byte[] bytes, int loc, int fmt,
381                Entry e) {
382         Fixups f;
383         if (prevFixups == null) {
384             if (loc == SPECIAL_LOC && fmt == SPECIAL_FMT) {
385                 // Special convention:  If the attribute has a
386                 // U2 relocation at position zero, store the Entry
387                 // rather than building a Fixups structure.
388                 return e;
389             }
390             f = new Fixups(bytes);
391         } else if (!(prevFixups instanceof Fixups)) {
392             // Recognize the special convention:
393             Entry firstEntry = (Entry) prevFixups;
394             f = new Fixups(bytes);
395             f.add(SPECIAL_LOC, SPECIAL_FMT, firstEntry);
396         } else {
397             f = (Fixups) prevFixups;
398             assert(f.bytes == bytes);
399         }
400         f.add(loc, fmt, e);
401         return f;
402     }

436     // Clear out this set of fixups by replacing each reference
437     // by a hardcoded coding of its reference, drawn from ix.
438     public static
439     void finishRefs(Object fixups, byte[] bytes, ConstantPool.Index ix) {
440         if (fixups == null)
441             return;
442         if (!(fixups instanceof Fixups)) {
443             // Special convention; see above.
444             int index = ix.indexOf((Entry) fixups);
445             storeIndex(bytes, SPECIAL_LOC, SPECIAL_FMT, index);
446             return;
447         }
448         Fixups f = (Fixups) fixups;
449         assert(f.bytes == bytes);
450         f.finishRefs(ix);
451     }
452 
453     void finishRefs(ConstantPool.Index ix) {
454         if (isEmpty())
455             return;
456         for (Iterator i = iterator(); i.hasNext(); ) {
457             Fixup fx = (Fixup) i.next();
458             int index = ix.indexOf(fx.entry);
459             //System.out.println("finish "+fx+" = "+index);
460             // Note that the iterator has already fetched the
461             // bytes we are about to overwrite.
462             storeIndex(fx.location(), fx.format(), index);
463         }
464         // Further iterations should do nothing:
465         bytes = null;  // do not clean them
466         clear();
467     }
