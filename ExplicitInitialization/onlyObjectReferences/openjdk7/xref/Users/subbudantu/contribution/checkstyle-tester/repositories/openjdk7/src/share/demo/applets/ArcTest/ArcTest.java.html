38  /**
39   * An interactive test of the Graphics.drawArc and Graphics.fillArc
40   * routines. Can be run either as a standalone application by
41   * typing "java ArcTest" or as an applet in the AppletViewer.
42   */
43  @SuppressWarnings("serial")
44  public class ArcTest extends Applet {
45  
46      ArcControls controls;   // The controls for marking and filling arcs
47      ArcCanvas canvas;       // The drawing area to display arcs
48  
49      @Override
50      public void init() {
51          setLayout(new BorderLayout());
52          canvas = new ArcCanvas();
53          add("Center", canvas);
54          add("South", controls = new ArcControls(canvas));
55      }
56  
57      @Override
58      public void destroy() {
59          remove(controls);
60          remove(canvas);
61      }
62  
63      @Override
64      public void start() {
65          controls.setEnabled(true);
66      }
67  
68      @Override
69      public void stop() {
70          controls.setEnabled(false);
71      }
72  
73      @Override
74      public void processEvent(AWTEvent e) {
75          if (e.getID() == Event.WINDOW_DESTROY) {
76              System.exit(0);
77          }
78      }
79  
80      public static void main(String args[]) {
81          Frame f = new Frame("ArcTest");
82          ArcTest arcTest = new ArcTest();
83  
84          arcTest.init();
85          arcTest.start();
86  
87          f.add("Center", arcTest);
88          f.setSize(300, 300);
89          f.setVisible(true);
90      }
91  
92      @Override
93      public String getAppletInfo() {
94          return "An interactive test of the Graphics.drawArc and \nGraphics."
95                  + "fillArc routines. Can be run \neither as a standalone "
96                  + "application by typing 'java ArcTest' \nor as an applet in "
97                  + "the AppletViewer.";
98      }
99  }

102 @SuppressWarnings("serial")
103 class ArcCanvas extends Canvas {
104 
105     int startAngle = 0;
106     int extent = 45;
107     boolean filled = false;
108     Font font = new java.awt.Font("SansSerif", Font.PLAIN, 12);
109 
110     @Override
111     public void paint(Graphics g) {
112         Rectangle r = getBounds();
113         int hlines = r.height / 10;
114         int vlines = r.width / 10;
115 
116         g.setColor(Color.pink);
117         for (int i = 1; i <= hlines; i++) {
118             g.drawLine(0, i * 10, r.width, i * 10);
119         }
120         for (int i = 1; i <= vlines; i++) {
121             g.drawLine(i * 10, 0, i * 10, r.height);
122         }
123 
124         g.setColor(Color.red);
125         if (filled) {
126             g.fillArc(0, 0, r.width - 1, r.height - 1, startAngle, extent);
127         } else {
128             g.drawArc(0, 0, r.width - 1, r.height - 1, startAngle, extent);
129         }
130 
131         g.setColor(Color.black);
132         g.setFont(font);
133         g.drawLine(0, r.height / 2, r.width, r.height / 2);
134         g.drawLine(r.width / 2, 0, r.width / 2, r.height);
135         g.drawLine(0, 0, r.width, r.height);
136         g.drawLine(r.width, 0, 0, r.height);
137         int sx = 10;
138         int sy = r.height - 28;
139         g.drawString("Start = " + startAngle, sx, sy);
140         g.drawString("Extent = " + extent, sx, sy + 14);
141     }
142 
143     public void redraw(boolean filled, int start, int extent) {
144         this.filled = filled;
145         this.startAngle = start;
146         this.extent = extent;
147         repaint();
148     }
149 }

152 @SuppressWarnings("serial")
153 class ArcControls extends Panel
154         implements ActionListener {
155 
156     TextField startTF;
157     TextField extentTF;
158     ArcCanvas canvas;
159 
160     @SuppressWarnings("LeakingThisInConstructor")
161     public ArcControls(ArcCanvas canvas) {
162         Button b = null;
163 
164         this.canvas = canvas;
165         add(startTF = new IntegerTextField("0", 4));
166         add(extentTF = new IntegerTextField("45", 4));
167         b = new Button("Fill");
168         b.addActionListener(this);
169         add(b);
170         b = new Button("Draw");
171         b.addActionListener(this);
172         add(b);
173     }
174 
175     @Override
176     public void actionPerformed(ActionEvent ev) {
177         String label = ev.getActionCommand();
178 
179         int start, extent;
180         try {
181             start = Integer.parseInt(startTF.getText().trim());
182         } catch (NumberFormatException ignored) {
183             start = 0;
184         }
185         try {
186             extent = Integer.parseInt(extentTF.getText().trim());
187         } catch (NumberFormatException ignored) {
188             extent = 0;
189         }
190 
191         canvas.redraw(label.equals("Fill"), start, extent);
192     }
193 }

196 @SuppressWarnings("serial")
197 class IntegerTextField extends TextField {
198 
199     String oldText = null;
200 
201     public IntegerTextField(String text, int columns) {
202         super(text, columns);
203         enableEvents(AWTEvent.KEY_EVENT_MASK | AWTEvent.TEXT_EVENT_MASK);
204         oldText = getText();
205     }
206 
207     // Consume non-digit KeyTyped events
208     // Note that processTextEvent kind of eliminates the need for this
209     // function, but this is neater, since ideally, it would prevent
210     // the text from appearing at all.  Sigh.  See bugid 4100317/4114565.
211     //
212     @Override
213     protected void processEvent(AWTEvent evt) {
214         int id = evt.getID();
215         if (id != KeyEvent.KEY_TYPED) {
216             super.processEvent(evt);
217             return;
218         }
219 
220         KeyEvent kevt = (KeyEvent) evt;
221         char c = kevt.getKeyChar();
222 
223         // Digits, backspace, and delete are okay
224         // Note that the minus sign is allowed, but not the decimal
225         if (Character.isDigit(c) || (c == '\b') || (c == '\u007f') || (c
226                 == '\u002d')) {
227             super.processEvent(evt);
228             return;
229         }
230 
231         Toolkit.getDefaultToolkit().beep();
232         kevt.consume();
233     }
234 
235     // Should consume TextEvents for non-integer Strings
236     // Store away the text in the tf for every TextEvent
237     // so we can revert to it on a TextEvent (paste, or
238     // legal key in the wrong location) with bad text
239     //
240     @Override
241     protected void processTextEvent(TextEvent te) {
242         // The empty string is okay, too
243         String newText = getText();
244         if (newText.equals("") || textIsInteger(newText)) {
245             oldText = newText;
246             super.processTextEvent(te);
247             return;
248         }
249 
250         Toolkit.getDefaultToolkit().beep();
251         setText(oldText);
252     }
253 
254     // Returns true for Integers (zero and negative
255     // values are allowed).
256     // Note that the empty string is not allowed.
257     //
258     private boolean textIsInteger(String textToCheck) {
259 
260         try {
261             Integer.parseInt(textToCheck, 10);
262             return true;
263         } catch (NumberFormatException ignored) {
264             return false;
265         }
266     }
