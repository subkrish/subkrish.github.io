46  /**
47   * RSA signature implementation. Supports RSA signing using PKCS#1 v1.5 padding.
48   *
49   * Objects should be instantiated by calling Signature.getInstance() using the
50   * following algorithm names:
51   *
52   *  . "NONEwithRSA"
53   *  . "SHA1withRSA"
54   *  . "SHA256withRSA"
55   *  . "SHA384withRSA"
56   *  . "SHA512withRSA"
57   *  . "MD5withRSA"
58   *  . "MD2withRSA"
59   *
60   * NOTE: RSA keys must be at least 512 bits long.
61   *
62   * NOTE: NONEwithRSA must be supplied with a pre-computed message digest.
63   *       Only the following digest algorithms are supported: MD5, SHA-1,
64   *       SHA-256, SHA-384, SHA-512 and a special-purpose digest algorithm
65   *       which is a concatenation of SHA-1 and MD5 digests.
66   *
67   * @since   1.6
68   * @author  Stanley Man-Kit Ho
69   */
70  abstract class RSASignature extends java.security.SignatureSpi
71  {
72      // message digest implementation we use
73      private final MessageDigest messageDigest;
74  
75      // message digest name
76      private String messageDigestAlgorithm;
77  
78      // flag indicating whether the digest has been reset
79      private boolean needsReset;
80  
81      // the signing key
82      private Key privateKey = null;
83  
84      // the verification key
85      private Key publicKey = null;
86  
87      /**
88       * Constructs a new RSASignature. Used by Raw subclass.
89       */
90      RSASignature() {
91          messageDigest = null;
92          messageDigestAlgorithm = null;
93      }
94  
95      /**
96       * Constructs a new RSASignature. Used by subclasses.
97       */
98      RSASignature(String digestName) {
99  
100         try {
101             messageDigest = MessageDigest.getInstance(digestName);
102             // Get the digest's canonical name
103             messageDigestAlgorithm = messageDigest.getAlgorithm();
104 
105         } catch (NoSuchAlgorithmException e) {
106            throw new ProviderException(e);
107         }
108 
109         needsReset = false;
110     }

112     // Nested class for NONEwithRSA signatures
113     public static final class Raw extends RSASignature {
114 
115         // the longest supported digest is 512 bits (SHA-512)
116         private static final int RAW_RSA_MAX = 64;
117 
118         private final byte[] precomputedDigest;
119         private int offset = 0;
120 
121         public Raw() {
122             precomputedDigest = new byte[RAW_RSA_MAX];
123         }
124 
125         // Stores the precomputed message digest value.
126         @Override
127         protected void engineUpdate(byte b) throws SignatureException {
128             if (offset >= precomputedDigest.length) {
129                 offset = RAW_RSA_MAX + 1;
130                 return;
131             }
132             precomputedDigest[offset++] = b;
133         }
134 
135         // Stores the precomputed message digest value.
136         @Override
137         protected void engineUpdate(byte[] b, int off, int len)
138                 throws SignatureException {
139             if (offset + len > precomputedDigest.length) {
140                 offset = RAW_RSA_MAX + 1;
141                 return;
142             }
143             System.arraycopy(b, off, precomputedDigest, offset, len);
144             offset += len;
145         }
146 
147         // Stores the precomputed message digest value.
148         @Override
149         protected void engineUpdate(ByteBuffer byteBuffer) {
150             int len = byteBuffer.remaining();
151             if (len <= 0) {
152                 return;
153             }
154             if (offset + len > precomputedDigest.length) {
155                 offset = RAW_RSA_MAX + 1;
156                 return;
157             }
158             byteBuffer.get(precomputedDigest, offset, len);
159             offset += len;
160         }
161 
162         @Override
163         protected void resetDigest(){
164             offset = 0;
165         }
166 
167         // Returns the precomputed message digest value.
168         @Override
169         protected byte[] getDigestValue() throws SignatureException {
170             if (offset > RAW_RSA_MAX) {
171                 throw new SignatureException("Message digest is too long");
172             }
173 
174             // Determine the digest algorithm from the digest length
175             if (offset == 20) {
176                 setDigestName("SHA1");
177             } else if (offset == 36) {
178                 setDigestName("SHA1+MD5");
179             } else if (offset == 32) {
180                 setDigestName("SHA-256");
181             } else if (offset == 48) {
182                 setDigestName("SHA-384");
183             } else if (offset == 64) {
184                 setDigestName("SHA-512");
185             } else if (offset == 16) {
186                 setDigestName("MD5");
187             } else {
188                 throw new SignatureException(
189                     "Message digest length is not supported");
190             }
191 
192             byte[] result = new byte[offset];
193             System.arraycopy(precomputedDigest, 0, result, 0, offset);
194             offset = 0;
195 
196             return result;
197         }
198     }

236     // initialize for signing. See JCA doc
237     protected void engineInitVerify(PublicKey key)
238         throws InvalidKeyException
239     {
240         // This signature accepts only RSAPublicKey
241         if ((key instanceof java.security.interfaces.RSAPublicKey) == false) {
242             throw new InvalidKeyException("Key type not supported");
243         }
244 
245         java.security.interfaces.RSAPublicKey rsaKey =
246             (java.security.interfaces.RSAPublicKey) key;
247 
248         if ((key instanceof sun.security.mscapi.RSAPublicKey) == false) {
249 
250             // convert key to MSCAPI format
251 
252             BigInteger modulus = rsaKey.getModulus();
253             BigInteger exponent =  rsaKey.getPublicExponent();
254 
255             // Check against the local and global values to make sure
256             // the sizes are ok.  Round up to the nearest byte.
257             RSAKeyFactory.checkKeyLengths(((modulus.bitLength() + 7) & ~7),
258                 exponent, -1, RSAKeyPairGenerator.KEY_SIZE_MAX);
259 
260             byte[] modulusBytes = modulus.toByteArray();
261             byte[] exponentBytes = exponent.toByteArray();
262 
263             // Adjust key length due to sign bit
264             int keyBitLength = (modulusBytes[0] == 0)
265                 ? (modulusBytes.length - 1) * 8
266                 : modulusBytes.length * 8;
267 
268             byte[] keyBlob = generatePublicKeyBlob(
269                 keyBitLength, modulusBytes, exponentBytes);
270 
271             try {
272                 publicKey = importPublicKey(keyBlob, keyBitLength);
273 
274             } catch (KeyStoreException e) {
275                 throw new InvalidKeyException(e);
276             }
277 
278         } else {
279             publicKey = (sun.security.mscapi.RSAPublicKey) key;
280         }
281 
282         this.privateKey = null;
283         resetDigest();
284     }

286     // initialize for signing. See JCA doc
287     protected void engineInitSign(PrivateKey key) throws InvalidKeyException
288     {
289         // This signature accepts only RSAPrivateKey
290         if ((key instanceof sun.security.mscapi.RSAPrivateKey) == false) {
291             throw new InvalidKeyException("Key type not supported");
292         }
293         privateKey = (sun.security.mscapi.RSAPrivateKey) key;
294 
295         // Check against the local and global values to make sure
296         // the sizes are ok.  Round up to nearest byte.
297         RSAKeyFactory.checkKeyLengths(((privateKey.bitLength() + 7) & ~7),
298             null, RSAKeyPairGenerator.KEY_SIZE_MIN,
299             RSAKeyPairGenerator.KEY_SIZE_MAX);
300 
301         this.publicKey = null;
302         resetDigest();
303     }

305     /**
306      * Resets the message digest if needed.
307      */
308     protected void resetDigest() {
309         if (needsReset) {
310             messageDigest.reset();
311             needsReset = false;
312         }
313     }
314 
315     protected byte[] getDigestValue() throws SignatureException {
316         needsReset = false;
317         return messageDigest.digest();
318     }

369     /**
370      * Returns the signature bytes of all the data
371      * updated so far.
372      * The format of the signature depends on the underlying
373      * signature scheme.
374      *
375      * @return the signature bytes of the signing operation's result.
376      *
377      * @exception SignatureException if the engine is not
378      * initialized properly or if this signature algorithm is unable to
379      * process the input data provided.
380      */
381     protected byte[] engineSign() throws SignatureException {
382 
383         byte[] hash = getDigestValue();
384 
385         // Omit the hash OID when generating a Raw signature
386         boolean noHashOID = this instanceof Raw;
387 
388         // Sign hash using MS Crypto APIs
389 
390         byte[] result = signHash(noHashOID, hash, hash.length,
391             messageDigestAlgorithm, privateKey.getHCryptProvider(),
392             privateKey.getHCryptKey());
393 
394         // Convert signature array from little endian to big endian
395         return convertEndianArray(result);
396     }
397 
398     /**
399      * Convert array from big endian to little endian, or vice versa.
400      */
401     private byte[] convertEndianArray(byte[] byteArray)
402     {
403         if (byteArray == null || byteArray.length == 0)
404             return byteArray;
405 
406         byte [] retval = new byte[byteArray.length];
407 
408         // make it big endian
409         for (int i=0;i < byteArray.length;i++)
410             retval[i] = byteArray[byteArray.length - i - 1];
411 
412         return retval;
413     }

430     /**
431      * Verifies the passed-in signature.
432      *
433      * @param sigBytes the signature bytes to be verified.
434      *
435      * @return true if the signature was verified, false if not.
436      *
437      * @exception SignatureException if the engine is not
438      * initialized properly, the passed-in signature is improperly
439      * encoded or of the wrong type, if this signature algorithm is unable to
440      * process the input data provided, etc.
441      */
442     protected boolean engineVerify(byte[] sigBytes)
443         throws SignatureException
444     {
445         byte[] hash = getDigestValue();
446 
447         return verifySignedHash(hash, hash.length,
448             messageDigestAlgorithm, convertEndianArray(sigBytes),
449             sigBytes.length, publicKey.getHCryptProvider(),
450             publicKey.getHCryptKey());
451     }
