24  /*
25   * @test
26   * @bug 6970930
27   * @summary verify that compare() throws NPE instead of IAE when an argument is null.
28   */
29  import java.text.*;
30  
31  public class Bug6970930 {
32  
33      private static boolean err = false;
34  
35      public static void main(String[] args) {
36          // Check if compare() throws NPE.
37          test1(null, null);
38          test1("\"foo\"", null);
39          test1(null, "\"bar\"");
40  
41          if (err) {
42              throw new RuntimeException("Failed.");
43          } else {
44              System.out.println("Passed.");
45          }
46      }
47  
48      private static void test1(String s1, String s2) {
49          RuleBasedCollator col = null;
50  
51          try {
52              col = new RuleBasedCollator("< a < b");
53          }
54          catch (ParseException e) {
55              err = true;
56              System.err.println(e);
57          }
58  
59          try {
60              col.compare("foo", "bar"); // This line is necessary to reproduce the bug.
61              col.compare(s1, s2);
62  
63              err = true;
64              System.err.println("No exception was thrown for compare(" +
65                                 s1 + ", " +  s2 + ").");
66          }
67          catch (NullPointerException e) {
68              System.out.println("NPE was thrown as expected for compare(" +
69                                 s1 + ", " + s2 + ").");
70          }
71          catch (Exception e) {
72              err = true;
73              System.err.println("Unexpected exception was thrown for compare(" +
74                                 s1 + ", " + s2 + "): " + e);
75          }
76      }
77  
78  }
