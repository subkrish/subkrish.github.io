140     private static void testMXBean(MBeanServer mbs, ObjectName on)
141             throws Exception {
142         MBeanInfo mbi = mbs.getMBeanInfo(on);
143         MBeanAttributeInfo[] attrs = mbi.getAttributes();
144         int nattrs = attrs.length;
145         if (mbi.getAttributes().length != 1)
146             failure("wrong number of attributes: " + attrs);
147         else {
148             MBeanAttributeInfo mbai = attrs[0];
149             if (mbai.getName().equals("Ints")
150                 && mbai.isReadable() && !mbai.isWritable()
151                 && mbai.getDescriptor().getFieldValue("openType")
152                     .equals(new ArrayType<int[]>(SimpleType.INTEGER, true))
153                 && attrs[0].getType().equals("[I"))
154                 success("MBeanAttributeInfo");
155             else
156                 failure("MBeanAttributeInfo: " + mbai);
157         }
158 
159         int[] ints = (int[]) mbs.getAttribute(on, "Ints");
160         if (equal(ints, new int[] {1, 2, 3}, null))
161             success("getAttribute");
162         else
163             failure("getAttribute: " + Arrays.toString(ints));
164 
165         ExplicitMXBean proxy =
166             JMX.newMXBeanProxy(mbs, on, ExplicitMXBean.class);
167         int[] pints = proxy.getInts();
168         if (equal(pints, new int[] {1, 2, 3}, null))
169             success("getAttribute through proxy");
170         else
171             failure("getAttribute through proxy: " + Arrays.toString(pints));
172     }

188     /* This is the core of the test.  Given the MXBean interface c, we
189        make an MXBean object that implements that interface by
190        constructing a dynamic proxy.  If the interface defines an
191        attribute Foo (with getFoo and setFoo methods), then it must
192        also contain a field (constant) Foo of the same type, and a
193        field (constant) FooType that is an OpenType.  The field Foo is
194        a reference value for this case.  We check that the attribute
195        does indeed have the given OpenType.  The dynamically-created
196        MXBean will return the reference value from the getFoo()
197        method, and we check that that value survives the mapping to
198        open values and back when the attribute is accessed through an
199        MXBean proxy.  The MXBean will also check in its setFoo method
200        that the value being set is equal to the reference value, which
201        tests that the mapping and unmapping also works in the other
202        direction.  The interface should define an operation opFoo with
203        two parameters and a return value all of the same type as the
204        attribute.  The MXBean will check that the two parameters are
205        equal to the reference value, and will return that value.  The
206        test checks that calling the operation through an MXBean proxy
207        returns the reference value, again after mapping to and back
208        from open values.
209 
210        If any field (constant) in the MXBean interface has a name that
211        ends with ObjectName, say FooObjectName, then its value must be
212        a String containing an ObjectName value.  There must be a field
213        (constant) called Foo that is a valid MXBean, and that MXBean
214        will be registered in the MBean Server with the given name before
215        the test starts.  This enables us to test that inter-MXBean
216        references are correctly converted to ObjectNames and back.
217      */
218     private static <T> void testInterface(Class<T> c, boolean nullTest)
219             throws Exception {
220 
221         System.out.println("Testing " + c.getName() +
222                            (nullTest ? " for null values" : "") + "...");
223 
224         MBeanServer mbs = MBeanServerFactory.newMBeanServer();
225 
226         JMXServiceURL url = new JMXServiceURL("rmi", null, 0);
227         JMXConnectorServer cs =
228             JMXConnectorServerFactory.newJMXConnectorServer(url, null, mbs);
229         cs.start();
230         JMXServiceURL addr = cs.getAddress();
231         JMXConnector cc = JMXConnectorFactory.connect(addr);
232         MBeanServerConnection mbsc = cc.getMBeanServerConnection();
233 
234         NamedMXBeans namedMXBeans = new NamedMXBeans(mbsc);
235         InvocationHandler ih =
236             nullTest ? new MXBeanNullImplInvocationHandler(c, namedMXBeans) :
237                        new MXBeanImplInvocationHandler(c, namedMXBeans);
238         T impl = c.cast(Proxy.newProxyInstance(c.getClassLoader(),
239                                                new Class[] {c},
240                                                ih));
241         ObjectName on = new ObjectName("test:type=" + c.getName());
242         mbs.registerMBean(impl, on);
243 
244         System.out.println("Register any MXBeans...");
245 
246         Field[] fields = c.getFields();
247         for (Field field : fields) {
248             String n = field.getName();
249             if (n.endsWith("ObjectName")) {
250                 String objectNameString = (String) field.get(null);
251                 String base = n.substring(0, n.length() - 10);
252                 Field f = c.getField(base);
253                 Object mxbean = f.get(null);
254                 ObjectName objectName =
255                     ObjectName.getInstance(objectNameString);
256                 mbs.registerMBean(mxbean, objectName);
257                 namedMXBeans.put(objectName, mxbean);
258             }
259         }
260 
261         try {
262             testInterface(c, mbsc, on, namedMXBeans, nullTest);
263         } finally {
264             try {
265                 cc.close();
266             } finally {
267                 cs.stop();
268             }
269         }
270     }

272     private static <T> void testInterface(Class<T> c,
273                                           MBeanServerConnection mbsc,
274                                           ObjectName on,
275                                           NamedMXBeans namedMXBeans,
276                                           boolean nullTest)
277             throws Exception {
278 
279         System.out.println("Type check...");
280 
281         MBeanInfo mbi = mbsc.getMBeanInfo(on);
282         MBeanAttributeInfo[] mbais = mbi.getAttributes();
283         for (int i = 0; i < mbais.length; i++) {
284             MBeanAttributeInfo mbai = mbais[i];
285             String name = mbai.getName();
286             Field typeField = c.getField(name + "Type");
287             OpenType typeValue = (OpenType) typeField.get(null);
288             OpenType openType =
289                 (OpenType) mbai.getDescriptor().getFieldValue("openType");
290             if (typeValue.equals(openType))
291                 success("attribute " + name);
292             else {
293                 final String msg =
294                     "Wrong type attribute " + name + ": " +
295                     openType + " should be " + typeValue;
296                 failure(msg);
297             }
298         }
299 
300         MBeanOperationInfo[] mbois = mbi.getOperations();
301         for (int i = 0; i < mbois.length; i++) {
302             MBeanOperationInfo mboi = mbois[i];
303             String oname = mboi.getName();
304             if (!oname.startsWith("op"))
305                 throw new Error();
306             OpenType retType =
307                 (OpenType) mboi.getDescriptor().getFieldValue("openType");
308             MBeanParameterInfo[] params = mboi.getSignature();
309             MBeanParameterInfo p1i = params[0];
310             MBeanParameterInfo p2i = params[1];
311             OpenType p1Type =
312                 (OpenType) p1i.getDescriptor().getFieldValue("openType");
313             OpenType p2Type =
314                 (OpenType) p2i.getDescriptor().getFieldValue("openType");
315             if (!retType.equals(p1Type) || !p1Type.equals(p2Type)) {
316                 final String msg =
317                     "Parameter and return open types should all be same " +
318                     "but are not: " + retType + " " + oname + "(" + p1Type +
319                     ", " + p2Type + ")";
320                 failure(msg);
321                 continue;
322             }
323             String name = oname.substring(2);
324             Field typeField = c.getField(name + "Type");
325             OpenType typeValue = (OpenType) typeField.get(null);
326             if (typeValue.equals(retType))
327                 success("operation " + oname);
328             else {
329                 final String msg =
330                     "Wrong type operation " + oname + ": " +
331                     retType + " should be " + typeValue;
332                 failure(msg);
333             }
334         }
335 
336 
337         System.out.println("Mapping check...");
338 
339         Object proxy =
340             JMX.newMXBeanProxy(mbsc, on, c);
341 
342         Method[] methods = c.getMethods();
343         for (int i = 0; i < methods.length; i++) {
344             final Method method = methods[i];
345             if (method.getDeclaringClass() != c)
346                 continue; // skip hashCode() etc inherited from Object
347             final String mname = method.getName();
348             final int what = getType(method);
349             final String name = getName(method);
350             final Field refField = c.getField(name);
351             if (nullTest && refField.getType().isPrimitive())
352                 continue;
353             final Field openTypeField = c.getField(name + "Type");
354             final OpenType openType = (OpenType) openTypeField.get(null);
355             final Object refValue = nullTest ? null : refField.get(null);
356             Object setValue = refValue;
357             try {
358                 Field onField = c.getField(name + "ObjectName");
359                 String refName = (String) onField.get(null);
360                 ObjectName refObjName = ObjectName.getInstance(refName);
361                 Class<?> mxbeanInterface = refField.getType();
362                 setValue = nullTest ? null :
363                     JMX.newMXBeanProxy(mbsc, refObjName, mxbeanInterface);
364             } catch (Exception e) {
365                 // no xObjectName field, setValue == refValue
366             }
367             boolean ok = true;
368             try {
369                 switch (what) {
370                 case GET:
371                     final Object gotOpen = mbsc.getAttribute(on, name);
372                     if (nullTest) {
373                         if (gotOpen != null) {
374                             failure(mname + " got non-null value " +
375                                     gotOpen);
376                             ok = false;
377                         }
378                     } else if (!openType.isValue(gotOpen)) {
379                         if (gotOpen instanceof TabularData) {
380                             // detail the mismatch
381                             TabularData gotTabular = (TabularData) gotOpen;
382                             compareTabularType((TabularType) openType,
383                                                gotTabular.getTabularType());
384                         }
385                         failure(mname + " got open data " + gotOpen +
386                                 " not valid for open type " + openType);
387                         ok = false;
388                     }
389                     final Object got = method.invoke(proxy, (Object[]) null);
390                     if (!equal(refValue, got, namedMXBeans)) {
391                         failure(mname + " got " + string(got) +
392                                 ", should be " + string(refValue));
393                         ok = false;
394                     }
395                     break;
396 
397                 case SET:
398                     method.invoke(proxy, new Object[] {setValue});
399                     break;
400 
401                 case OP:
402                     final Object opped =
403                         method.invoke(proxy, new Object[] {setValue, setValue});
404                     if (!equal(refValue, opped, namedMXBeans)) {
405                         failure(
406                                 mname + " got " + string(opped) +
407                                 ", should be " + string(refValue)
408                                 );
409                         ok = false;
410                     }
411                     break;
412 
413                 default:
414                     throw new Error();
415                 }
416 
417                 if (ok)
418                     success(mname);
419 
420             } catch (Exception e) {
421                 failure(mname, e);
422             }
423         }
424     }

442     private static class MXBeanImplInvocationHandler
443             implements InvocationHandler {
444         MXBeanImplInvocationHandler(Class intf, NamedMXBeans namedMXBeans) {
445             this.intf = intf;
446             this.namedMXBeans = namedMXBeans;
447         }
448 
449         public Object invoke(Object proxy, Method method, Object[] args)
450                 throws Throwable {
451             final String mname = method.getName();
452             final int what = getType(method);
453             final String name = getName(method);
454             final Field refField = intf.getField(name);
455             final Object refValue = getRefValue(refField);
456 
457             switch (what) {
458             case GET:
459                 assert args == null;
460                 return refValue;
461 
462             case SET:
463                 assert args.length == 1;
464                 Object setValue = args[0];
465                 if (!equal(refValue, setValue, namedMXBeans)) {
466                     final String msg =
467                         mname + "(" + string(setValue) +
468                         ") does not match ref: " + string(refValue);
469                     throw new IllegalArgumentException(msg);
470                 }
471                 return null;
472 
473             case OP:
474                 assert args.length == 2;
475                 Object arg1 = args[0];
476                 Object arg2 = args[1];
477                 if (!equal(arg1, arg2, namedMXBeans)) {
478                     final String msg =
479                         mname + "(" + string(arg1) + ", " + string(arg2) +
480                         "): args not equal";
481                     throw new IllegalArgumentException(msg);
482                 }
483                 if (!equal(refValue, arg1, namedMXBeans)) {
484                     final String msg =
485                         mname + "(" + string(arg1) + ", " + string(arg2) +
486                         "): args do not match ref: " + string(refValue);
487                     throw new IllegalArgumentException(msg);
488                 }
489                 return refValue;
490             default:
491                 throw new Error();
492             }
493         }
494 
495         Object getRefValue(Field refField) throws Exception {
496             return refField.get(null);
497         }
498 
499         private final Class intf;
500         private final NamedMXBeans namedMXBeans;
501     }

549     static boolean equal(Object o1, Object o2, NamedMXBeans namedMXBeans) {
550         if (o1 == o2)
551             return true;
552         if (o1 == null || o2 == null)
553             return false;
554         if (o1.getClass().isArray()) {
555             if (!o2.getClass().isArray())
556                 return false;
557             return deepEqual(o1, o2, namedMXBeans);
558         }
559         if (o1 instanceof Map) {
560             if (!(o2 instanceof Map))
561                 return false;
562             return equalMap((Map) o1, (Map) o2, namedMXBeans);
563         }
564         if (o1 instanceof CompositeData && o2 instanceof CompositeData) {
565             return compositeDataEqual((CompositeData) o1, (CompositeData) o2,
566                                       namedMXBeans);
567         }
568         if (Proxy.isProxyClass(o1.getClass())) {
569             if (Proxy.isProxyClass(o2.getClass()))
570                 return proxyEqual(o1, o2, namedMXBeans);
571             InvocationHandler ih = Proxy.getInvocationHandler(o1);
572 //            if (ih instanceof MXBeanInvocationHandler) {
573 //                return proxyEqualsObject((MXBeanInvocationHandler) ih,
574 //                                         o2, namedMXBeans);
575             if (ih instanceof MBeanServerInvocationHandler) {
576                 return true;
577             } else if (ih instanceof CompositeDataInvocationHandler) {
578                 return o2.equals(o1);
579                 // We assume the other object has a reasonable equals method
580             }
581         } else if (Proxy.isProxyClass(o2.getClass()))
582             return equal(o2, o1, namedMXBeans);
583         return o1.equals(o2);
584     }

586     // We'd use Arrays.deepEquals except we want the test to work on 1.4
587     // Note this code assumes no selfreferential arrays
588     // (as does Arrays.deepEquals)
589     private static boolean deepEqual(Object a1, Object a2,
590                                      NamedMXBeans namedMXBeans) {
591         int len = Array.getLength(a1);
592         if (len != Array.getLength(a2))
593             return false;
594         for (int i = 0; i < len; i++) {
595             Object e1 = Array.get(a1, i);
596             Object e2 = Array.get(a2, i);
597             if (!equal(e1, e2, namedMXBeans))
598                 return false;
599         }
600         return true;
601     }

603     private static boolean equalMap(Map<?,?> m1, Map<?,?> m2,
604                                     NamedMXBeans namedMXBeans) {
605         if (m1.size() != m2.size())
606             return false;
607         if ((m1 instanceof SortedMap) != (m2 instanceof SortedMap))
608             return false;
609         for (Object k1 : m1.keySet()) {
610             if (!m2.containsKey(k1))
611                 return false;
612             if (!equal(m1.get(k1), m2.get(k1), namedMXBeans))
613                 return false;
614         }
615         return true;
616     }

618     // This is needed to work around a bug (5095277)
619     // in CompositeDataSupport.equals
620     private static boolean compositeDataEqual(CompositeData cd1,
621                                               CompositeData cd2,
622                                               NamedMXBeans namedMXBeans) {
623         if (cd1 == cd2)
624             return true;
625         if (!cd1.getCompositeType().equals(cd2.getCompositeType()))
626             return false;
627         Collection v1 = cd1.values();
628         Collection v2 = cd2.values();
629         if (v1.size() != v2.size())
630             return false; // should not happen
631         for (Iterator i1 = v1.iterator(), i2 = v2.iterator();
632              i1.hasNext(); ) {
633             if (!equal(i1.next(), i2.next(), namedMXBeans))
634                 return false;
635         }
636         return true;
637     }

639     // Also needed for 5095277
640     private static boolean proxyEqual(Object proxy1, Object proxy2,
641                                       NamedMXBeans namedMXBeans) {
642         if (proxy1.getClass() != proxy2.getClass())
643             return proxy1.equals(proxy2);
644         InvocationHandler ih1 = Proxy.getInvocationHandler(proxy1);
645         InvocationHandler ih2 = Proxy.getInvocationHandler(proxy2);
646         if (!(ih1 instanceof CompositeDataInvocationHandler)
647             || !(ih2 instanceof CompositeDataInvocationHandler))
648             return proxy1.equals(proxy2);
649         CompositeData cd1 =
650             ((CompositeDataInvocationHandler) ih1).getCompositeData();
651         CompositeData cd2 =
652             ((CompositeDataInvocationHandler) ih2).getCompositeData();
653         return compositeDataEqual(cd1, cd2, namedMXBeans);
654     }

670     /* I wanted to call this method toString(Object), but oddly enough
671        this meant that I couldn't call it from the inner class
672        MXBeanImplInvocationHandler, because the inherited Object.toString()
673        prevented that.  */
674     static String string(Object o) {
675         if (o == null)
676             return "null";
677         if (o instanceof String)
678             return '"' + (String) o + '"';
679         if (o instanceof Collection)
680             return deepToString((Collection) o);
681         if (o.getClass().isArray())
682             return deepToString(o);
683         return o.toString();
684     }
685 
686     private static String deepToString(Object o) {
687         StringBuffer buf = new StringBuffer();
688         buf.append("[");
689         int len = Array.getLength(o);
690         for (int i = 0; i < len; i++) {
691             if (i > 0)
692                 buf.append(", ");
693             Object e = Array.get(o, i);
694             buf.append(string(e));
695         }
696         buf.append("]");
697         return buf.toString();
698     }
699 
700     private static String deepToString(Collection c) {
701         return deepToString(c.toArray());
702     }

704     private static void compareTabularType(TabularType t1, TabularType t2) {

712             System.out.println("same description");
713         else {
714             System.out.println("t1 description: " + t1.getDescription());
715             System.out.println("t2 description: " + t2.getDescription());
716         }
717         if (t1.getIndexNames().equals(t2.getIndexNames()))
718             System.out.println("same index names");
719         if (t1.getRowType().equals(t2.getRowType()))
720             System.out.println("same row type");
