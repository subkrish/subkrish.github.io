85      /*
86       * Define the server side of the test.
87       *
88       * If the server prematurely exits, serverReady will be set to true
89       * to avoid infinite hangs.
90       */
91      void doServerSide() throws Exception {
92          KeyStore ks = KeyStore.getInstance("JKS");
93          com.sun.net.ssl.SSLContext ctx =
94              com.sun.net.ssl.SSLContext.getInstance("TLS");
95          com.sun.net.ssl.KeyManagerFactory kmf =
96              com.sun.net.ssl.KeyManagerFactory.getInstance("SunX509");
97  
98          ks.load(new FileInputStream(keyFilename), cpasswd);
99          kmf.init(ks, cpasswd);
100 
101         com.sun.net.ssl.TrustManager [] tms =
102             new com.sun.net.ssl.TrustManager []
103             { new MyComX509TrustManager() };
104 
105         ctx.init(kmf.getKeyManagers(), tms, null);
106 
107         SSLServerSocketFactory sslssf =
108             (SSLServerSocketFactory) ctx.getServerSocketFactory();
109 
110         SSLServerSocket sslServerSocket =
111             (SSLServerSocket) sslssf.createServerSocket(serverPort);
112         serverPort = sslServerSocket.getLocalPort();
113 
114         sslServerSocket.setNeedClientAuth(true);
115 
116         /*
117          * Create using the other type.
118          */
119         SSLContext ctx1 =
120             SSLContext.getInstance("TLS");
121         KeyManagerFactory kmf1 =
122             KeyManagerFactory.getInstance("SunX509");
123 
124         TrustManager [] tms1 =
125             new TrustManager []
126             { new MyJavaxX509TrustManager() };
127 
128         kmf1.init(ks, cpasswd);
129 
130         ctx1.init(kmf1.getKeyManagers(), tms1, null);
131 
132         sslssf = (SSLServerSocketFactory) ctx1.getServerSocketFactory();
133 
134         SSLServerSocket sslServerSocket1 =
135             (SSLServerSocket) sslssf.createServerSocket(serverPort1);
136         serverPort1 = sslServerSocket1.getLocalPort();
137         sslServerSocket1.setNeedClientAuth(true);
138 
139         /*
140          * Signal Client, we're ready for his connect.
141          */
142         serverReady = true;
143 
144         SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
145         sslServerSocket.close();
146         serverReady = false;
147 
148         InputStream sslIS = sslSocket.getInputStream();
149         OutputStream sslOS = sslSocket.getOutputStream();
150 
151         sslIS.read();
152         sslOS.write(85);
153         sslOS.flush();
154         sslSocket.close();
155 
156         sslSocket = (SSLSocket) sslServerSocket1.accept();
157         sslIS = sslSocket.getInputStream();
158         sslOS = sslSocket.getOutputStream();
159 
160         sslIS.read();
161         sslOS.write(85);
162         sslOS.flush();
163         sslSocket.close();
164 
165         System.out.println("Server exiting!");
166         System.out.flush();
167     }

248     }

268     public static void main(String[] args) throws Exception {
269 
270         if (debug)
271             System.setProperty("javax.net.debug", "all");
272 
273         /*
274          * Start the tests.
275          */
276         new CheckMyTrustedKeystore();
277     }

279     Thread clientThread = null;
280     Thread serverThread = null;
281 
282     /*
283      * Primary constructor, used to drive remainder of the test.
284      *
285      * Fork off the other side, then do your work.
286      */
287     CheckMyTrustedKeystore() throws Exception {
288         if (separateServerThread) {
289             startServer(true);
290             startClient(false);
291         } else {
292             startClient(true);
293             startServer(false);
294         }
295 
296         /*
297          * Wait for other side to close down.
298          */
299         if (separateServerThread) {
300             serverThread.join();
301         } else {
302             clientThread.join();
303         }
304 
305         /*
306          * When we get here, the test is pretty much over.
307          *
308          * If the main thread excepted, that propagates back
309          * immediately.  If the other thread threw an exception, we
310          * should report back.
311          */
312         if (serverException != null) {
313             System.out.print("Server Exception:");
314             throw serverException;
315         }
316         if (clientException != null) {
317             System.out.print("Client Exception:");
318             throw clientException;
319         }
320     }
321 
322     void startServer(boolean newThread) throws Exception {
323         if (newThread) {
324             serverThread = new Thread() {
325                 public void run() {
326                     try {
327                         doServerSide();
328                     } catch (Exception e) {
329                         /*
330                          * Our server thread just died.
331                          *
332                          * Release the client, if not active already...
333                          */
334                         System.err.println("Server died...");
335                         serverReady = true;
336                         serverException = e;
337                     }
338                 }
339             };
340             serverThread.start();
341         } else {
342             doServerSide();
343         }
344     }
345 
346     void startClient(boolean newThread) throws Exception {
347         if (newThread) {
348             clientThread = new Thread() {
349                 public void run() {
350                     try {
351                         doClientSide();
352                     } catch (Exception e) {
353                         /*
354                          * Our client thread just died.
355                          */
356                         System.err.println("Client died...");
357                         clientException = e;
358                     }
359                 }
360             };
361             clientThread.start();
362         } else {
363             doClientSide();
364         }
365     }
