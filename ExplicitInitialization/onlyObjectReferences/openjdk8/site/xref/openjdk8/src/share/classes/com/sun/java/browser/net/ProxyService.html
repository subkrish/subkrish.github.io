26  package com.sun.java.browser.net;
27  
28  import java.net.URL;
29  import java.io.IOException;
30  
31  /**
32   *
33   * @author  Zhengyu Gu
34   */
35  public class ProxyService extends Object {
36      private static ProxyServiceProvider provider = null;
37  
38  
39      public static void setProvider(ProxyServiceProvider p)
40      throws IOException {
41          if(null == provider)
42              provider = p;
43          else
44              throw new IOException("Proxy service provider has already been set.");
45      }
46  
47  
48      /**
49       *  <p>The function returns proxy information of the specified URL.</p>
50       *  @param url URL
51       *  @return returns proxy information. If there is not proxy, returns null
52       *  @since 1.4
53       */
54      public static ProxyInfo[] getProxyInfo(URL url)
55      throws IOException {
56          if(null == provider)
57              throw new IOException("Proxy service provider is not yet set");
58  
59          return provider.getProxyInfo(url);
60      }
61  }
