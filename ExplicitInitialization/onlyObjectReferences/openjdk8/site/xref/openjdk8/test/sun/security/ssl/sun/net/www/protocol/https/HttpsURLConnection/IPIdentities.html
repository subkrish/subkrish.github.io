654         sslServerSocket =
655             (SSLServerSocket) sslssf.createServerSocket(serverPort);
656         serverPort = sslServerSocket.getLocalPort();
657 
658         /*
659          * Signal Client, we're ready for his connect.
660          */
661         serverReady = true;
662 
663         SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
664         sslSocket.setNeedClientAuth(true);
665 
666         PrintStream out =
667                 new PrintStream(sslSocket.getOutputStream());
668 
669         try {
670             // ignore request data
671

674             out.print("Content-Type: text/html; charset=iso-8859-1\r\n");
675             out.print("Content-Length: "+ 9 +"\r\n");
676             out.print("\r\n");
677             out.print("Testing\r\n");
678             out.flush();
679         } finally {
680              // close the socket
681              while (!closeReady) {
682                  Thread.sleep(50);
683              }
684 
685              System.out.println("Server closing socket");
686              sslSocket.close();

785          * should report back.
786          */
787         if (serverException != null)
788             throw serverException;
789         if (clientException != null)
790             throw clientException;

838     // get the ssl context
839     private static SSLContext getSSLContext(String trusedCertStr,
840             String keyCertStr, byte[] modulus,
841             byte[] privateExponent, char[] passphrase) throws Exception {
842 
843         // generate certificate from cert string
844         CertificateFactory cf = CertificateFactory.getInstance("X.509");
845 
846         ByteArrayInputStream is =
847                     new ByteArrayInputStream(trusedCertStr.getBytes());
848         Certificate trusedCert = cf.generateCertificate(is);
849         is.close();
850 
851         // create a key store
852         KeyStore ks = KeyStore.getInstance("JKS");
853         ks.load(null, null);
854 
855         // import the trused cert
856         ks.setCertificateEntry("RSA Export Signer", trusedCert);
857 
858         if (keyCertStr != null) {
859             // generate the private key.
860             RSAPrivateKeySpec priKeySpec = new RSAPrivateKeySpec(
861                                             new BigInteger(modulus),
862                                             new BigInteger(privateExponent));
863             KeyFactory kf = KeyFactory.getInstance("RSA");
864             RSAPrivateKey priKey =
865                     (RSAPrivateKey)kf.generatePrivate(priKeySpec);
866 
867             // generate certificate chain
868             is = new ByteArrayInputStream(keyCertStr.getBytes());
869             Certificate keyCert = cf.generateCertificate(is);
870             is.close();
871 
872             Certificate[] chain = new Certificate[2];
873             chain[0] = keyCert;
874             chain[1] = trusedCert;
875 
876             // import the key entry.
877             ks.setKeyEntry("Whatever", priKey, passphrase, chain);
878         }
879 
880         // create SSL context
881         TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
882         tmf.init(ks);
883 
884         SSLContext ctx = SSLContext.getInstance("TLS");
885 
886         if (keyCertStr != null) {
887             KeyManagerFactory kmf = KeyManagerFactory.getInstance("SunX509");
888             kmf.init(ks, passphrase);
889 
890             ctx.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
891         } else {
892             ctx.init(null, tmf.getTrustManagers(), null);
893         }
894 
895         return ctx;
896     }
