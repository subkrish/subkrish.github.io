1130     private boolean initialized = false;
1131 
1132     // If the sun.nio.cs.map property is defined on the command line we won't
1133     // see it in the system-properties table until after the charset subsystem
1134     // has been initialized.  We therefore delay the effect of this property
1135     // until after the JRE has completely booted.
1136     //
1137     // At the moment following values for this property are supported, property
1138     // value string is case insensitive.
1139     //
1140     // (1)"Windows-31J/Shift_JIS"
1141     // In 1.4.1 we added a correct implementation of the Shift_JIS charset
1142     // but in previous releases this charset name had been treated as an alias
1143     // for Windows-31J, aka MS932. Users who have existing code that depends
1144     // upon this alias can restore the previous behavior by defining this
1145     // property to have this value.
1146     //
1147     // (2)"x-windows-50221/ISO-2022-JP"
1148     //    "x-windows-50220/ISO-2022-JP"
1149     //    "x-windows-iso2022jp/ISO-2022-JP"
1150     // The charset ISO-2022-JP is a "standard based" implementation by default,
1151     // which supports ASCII, JIS_X_0201 and JIS_X_0208 mappings based encoding
1152     // and decoding only.
1153     // There are three Microsoft iso-2022-jp variants, namely x-windows-50220,
1154     // x-windows-50221 and x-windows-iso2022jp which behaves "slightly" differently
1155     // compared to the "standard based" implementation. See ISO2022_JP.java for
1156     // detailed description. Users who prefer the behavior of MS iso-2022-jp
1157     // variants should use these names explicitly instead of using "ISO-2022-JP"
1158     // and its aliases. However for those who need the ISO-2022-JP charset behaves
1159     // exactly the same as MS variants do, above properties can be defined to
1160     // switch.
1161     //
1162     // If we need to define other charset-alias mappings in the future then
1163     // this property could be further extended, the general idea being that its
1164     // value should be of the form
1165     //
1166     //     new-charset-1/old-charset-1,new-charset-2/old-charset-2,...
1167     //
1168     // where each charset named to the left of a slash is intended to replace
1169     // (most) uses of the charset named to the right of the slash.
1170     //
1171     protected void init() {
1172         if (initialized)
1173             return;
1174         if (!sun.misc.VM.isBooted())
1175             return;
1176 
1177         String map = AccessController.doPrivileged(
1178             new GetPropertyAction("sun.nio.cs.map"));
1179         boolean sjisIsMS932 = false;
1180         boolean iso2022jpIsMS50221 = false;
1181         boolean iso2022jpIsMS50220 = false;
1182         boolean iso2022jpIsMSISO2022JP = false;
1183         if (map != null) {
1184             String[] maps = map.split(",");
1185             for (int i = 0; i < maps.length; i++) {
1186                 if (maps[i].equalsIgnoreCase("Windows-31J/Shift_JIS")) {
1187                     sjisIsMS932 = true;
1188                 } else if (maps[i].equalsIgnoreCase("x-windows-50221/ISO-2022-JP")) {
1189                     iso2022jpIsMS50221 = true;
1190                 } else if (maps[i].equalsIgnoreCase("x-windows-50220/ISO-2022-JP")) {
1191                     iso2022jpIsMS50220 = true;
1192                 } else if (maps[i].equalsIgnoreCase("x-windows-iso2022jp/ISO-2022-JP")) {
1193                     iso2022jpIsMSISO2022JP = true;
1194                 }
1195             }
1196         }
1197         if (sjisIsMS932) {
1198             deleteCharset("Shift_JIS",
1199                           new String[] {
1200                               // IANA aliases
1201                               "sjis", // historical
1202                               "shift_jis",
1203                               "shift-jis",
1204                               "ms_kanji",
1205                               "x-sjis",
1206                               "csShiftJIS"
1207                           });
1208             deleteCharset("windows-31j",
1209                           new String[] {
1210                               "MS932", // JDK historical
1211                               "windows-932",
1212                               "csWindows31J"
1213                           });
1214             charset("Shift_JIS", "SJIS",
1215                     new String[] {
1216                         // IANA aliases
1217                         "sjis"          // JDK historical
1218                     });
1219             charset("windows-31j", "MS932",
1220                     new String[] {
1221                         "MS932",        // JDK historical
1222                         "windows-932",
1223                         "csWindows31J",
1224                         "shift-jis",
1225                         "ms_kanji",
1226                         "x-sjis",
1227                         "csShiftJIS",
1228                         // This alias takes precedence over the actual
1229                         // Shift_JIS charset itself since aliases are always
1230                         // resolved first, before looking up canonical names.
1231                         "shift_jis"
1232                     });
1233         }
1234         if (iso2022jpIsMS50221 ||
1235             iso2022jpIsMS50220 ||
1236             iso2022jpIsMSISO2022JP) {
1237             deleteCharset("ISO-2022-JP",
1238                           new String[] {
1239                               "iso2022jp",
1240                                 "jis",
1241                                 "csISO2022JP",
1242                                 "jis_encoding",
1243                                 "csjisencoding"
1244                           });
1245             if (iso2022jpIsMS50221) {
1246                 deleteCharset("x-windows-50221",
1247                               new String[] {
1248                                   "cp50221",
1249                                   "ms50221"
1250                               });
1251                 charset("x-windows-50221", "MS50221",
1252                         new String[] {
1253                             "cp50221",
1254                             "ms50221",
1255                             "iso-2022-jp",
1256                             "iso2022jp",
1257                             "jis",
1258                             "csISO2022JP",
1259                             "jis_encoding",
1260                             "csjisencoding"
1261                         });
1262             } else if (iso2022jpIsMS50220) {
1263                 deleteCharset("x-windows-50220",
1264                               new String[] {
1265                                   "cp50220",
1266                                   "ms50220"
1267                               });
1268                 charset("x-windows-50220", "MS50220",
1269                         new String[] {
1270                             "cp50220",
1271                             "ms50220",
1272                             "iso-2022-jp",
1273                             "iso2022jp",
1274                             "jis",
1275                             "csISO2022JP",
1276                             "jis_encoding",
1277                             "csjisencoding"
1278                         });
1279             } else {
1280                 deleteCharset("x-windows-iso2022jp",
1281                               new String[] {
1282                                   "windows-iso2022jp"
1283                               });
1284                 charset("x-windows-iso2022jp", "MSISO2022JP",
1285                         new String[] {
1286                             "windows-iso2022jp",
1287                             "iso-2022-jp",
1288                             "iso2022jp",
1289                             "jis",
1290                             "csISO2022JP",
1291                             "jis_encoding",
1292                             "csjisencoding"
1293                         });
1294 
1295 
1296             }
1297         }
1298         String osName = AccessController.doPrivileged(
1299             new GetPropertyAction("os.name"));
1300         if ("SunOS".equals(osName) || "Linux".equals(osName)
1301                || osName.contains("OS X")) {
1302             charset("x-COMPOUND_TEXT", "COMPOUND_TEXT",
1303                     new String[] {
1304                         "COMPOUND_TEXT",        // JDK historical
1305                         "x11-compound_text",
1306                         "x-compound-text"
1307                     });
1308         }
1309         initialized = true;
1310     }
