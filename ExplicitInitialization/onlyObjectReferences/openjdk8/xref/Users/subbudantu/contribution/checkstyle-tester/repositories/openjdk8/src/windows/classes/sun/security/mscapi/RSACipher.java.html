39  /**
40   * RSA cipher implementation using the Microsoft Crypto API.
41   * Supports RSA en/decryption and signing/verifying using PKCS#1 v1.5 padding.
42   *
43   * Objects should be instantiated by calling Cipher.getInstance() using the
44   * following algorithm name:
45   *
46   *  . "RSA/ECB/PKCS1Padding" (or "RSA") for PKCS#1 padding. The mode (blocktype)
47   *    is selected based on the en/decryption mode and public/private key used.
48   *
49   * We only do one RSA operation per doFinal() call. If the application passes
50   * more data via calls to update() or doFinal(), we throw an
51   * IllegalBlockSizeException when doFinal() is called (see JCE API spec).
52   * Bulk encryption using RSA does not make sense and is not standardized.
53   *
54   * Note: RSA keys should be at least 512 bits long
55   *
56   * @since   1.6
57   * @author  Andreas Sterbenz
58   * @author  Vincent Ryan
59   */
60  public final class RSACipher extends CipherSpi {

62      // constant for an empty byte array
63      private final static byte[] B0 = new byte[0];
64  
65      // mode constant for public key encryption
66      private final static int MODE_ENCRYPT = 1;
67      // mode constant for private key decryption
68      private final static int MODE_DECRYPT = 2;
69      // mode constant for private key encryption (signing)
70      private final static int MODE_SIGN    = 3;
71      // mode constant for public key decryption (verifying)
72      private final static int MODE_VERIFY  = 4;
73  
74      // constant for PKCS#1 v1.5 RSA
75      private final static String PAD_PKCS1 = "PKCS1Padding";
76      private final static int PAD_PKCS1_LENGTH = 11;
77  
78      // current mode, one of MODE_* above. Set when init() is called
79      private int mode;
80  
81      // active padding type, one of PAD_* above. Set by setPadding()
82      private String paddingType;
83      private int paddingLength = 0;
84  
85      // buffer for the data
86      private byte[] buffer;
87      // offset into the buffer (number of bytes buffered)
88      private int bufOfs;
89  
90      // size of the output (the length of the key).
91      private int outputSize;
92  
93      // the public key, if we were initialized using a public key
94      private sun.security.mscapi.Key publicKey;
95      // the private key, if we were initialized using a private key
96      private sun.security.mscapi.Key privateKey;

176     // initialize this cipher
177     private void init(int opmode, Key key) throws InvalidKeyException {
178 
179         boolean encrypt;
180 
181         switch (opmode) {
182         case Cipher.ENCRYPT_MODE:
183         case Cipher.WRAP_MODE:
184             paddingLength = PAD_PKCS1_LENGTH;
185             encrypt = true;
186             break;
187         case Cipher.DECRYPT_MODE:
188         case Cipher.UNWRAP_MODE:
189             paddingLength = 0; // reset
190             encrypt = false;
191             break;
192         default:
193             throw new InvalidKeyException("Unknown mode: " + opmode);
194         }
195 
196         if (!(key instanceof sun.security.mscapi.Key)) {
197             if (key instanceof java.security.interfaces.RSAPublicKey) {
198                 java.security.interfaces.RSAPublicKey rsaKey =
199                     (java.security.interfaces.RSAPublicKey) key;
200 
201                 // Convert key to MSCAPI format
202 
203                 BigInteger modulus = rsaKey.getModulus();
204                 BigInteger exponent =  rsaKey.getPublicExponent();
205 
206                 // Check against the local and global values to make sure
207                 // the sizes are ok.  Round up to the nearest byte.
208                 RSAKeyFactory.checkKeyLengths(((modulus.bitLength() + 7) & ~7),
209                     exponent, -1, RSAKeyPairGenerator.KEY_SIZE_MAX);
210 
211                 byte[] modulusBytes = modulus.toByteArray();
212                 byte[] exponentBytes = exponent.toByteArray();
213 
214                 // Adjust key length due to sign bit
215                 int keyBitLength = (modulusBytes[0] == 0)
216                     ? (modulusBytes.length - 1) * 8
217                     : modulusBytes.length * 8;
218 
219                 byte[] keyBlob = RSASignature.generatePublicKeyBlob(
220                     keyBitLength, modulusBytes, exponentBytes);
221 
222                 try {
223                     key = RSASignature.importPublicKey(keyBlob, keyBitLength);
224 
225                 } catch (KeyStoreException e) {
226                     throw new InvalidKeyException(e);
227                 }
228 
229             } else {
230                 throw new InvalidKeyException("Unsupported key type: " + key);
231             }
232         }
233 
234         if (key instanceof PublicKey) {
235             mode = encrypt ? MODE_ENCRYPT : MODE_VERIFY;
236             publicKey = (sun.security.mscapi.Key)key;
237             privateKey = null;
238             outputSize = publicKey.length() / 8;
239         } else if (key instanceof PrivateKey) {
240             mode = encrypt ? MODE_SIGN : MODE_DECRYPT;
241             privateKey = (sun.security.mscapi.Key)key;
242             publicKey = null;
243             outputSize = privateKey.length() / 8;
244         } else {
245             throw new InvalidKeyException("Unknown key type: " + key);
246         }
247 
248         bufOfs = 0;
249         buffer = new byte[outputSize];
250     }

252     // internal update method
253     private void update(byte[] in, int inOfs, int inLen) {
254         if ((inLen == 0) || (in == null)) {
255             return;
256         }
257         if (bufOfs + inLen > (buffer.length - paddingLength)) {
258             bufOfs = buffer.length + 1;
259             return;
260         }
261         System.arraycopy(in, inOfs, buffer, bufOfs, inLen);
262         bufOfs += inLen;
263     }
264 
265     // internal doFinal() method. Here we perform the actual RSA operation
266     private byte[] doFinal() throws BadPaddingException,
267             IllegalBlockSizeException {
268         if (bufOfs > buffer.length) {
269             throw new IllegalBlockSizeException("Data must not be longer "
270                 + "than " + (buffer.length - paddingLength)  + " bytes");
271         }
272 
273         try {
274             byte[] data = buffer;
275             switch (mode) {
276             case MODE_SIGN:
277                 return encryptDecrypt(data, bufOfs,
278                     privateKey.getHCryptKey(), true);
279 
280             case MODE_VERIFY:
281                 return encryptDecrypt(data, bufOfs,
282                     publicKey.getHCryptKey(), false);
283 
284             case MODE_ENCRYPT:
285                 return encryptDecrypt(data, bufOfs,
286                     publicKey.getHCryptKey(), true);
287 
288             case MODE_DECRYPT:
289                 return encryptDecrypt(data, bufOfs,
290                     privateKey.getHCryptKey(), false);
291 
292             default:
293                 throw new AssertionError("Internal error");
294             }
295 
296         } catch (KeyException e) {
297             throw new ProviderException(e);
298 
299         } finally {
300             bufOfs = 0;
301         }
302     }

306         update(in, inOfs, inLen);

317     // see JCE spec
318     protected byte[] engineDoFinal(byte[] in, int inOfs, int inLen)
319             throws BadPaddingException, IllegalBlockSizeException {
320         update(in, inOfs, inLen);
321         return doFinal();
322     }

324     // see JCE spec
325     protected int engineDoFinal(byte[] in, int inOfs, int inLen, byte[] out,
326             int outOfs) throws ShortBufferException, BadPaddingException,
327             IllegalBlockSizeException {
328         if (outputSize > out.length - outOfs) {
329             throw new ShortBufferException
330                 ("Need " + outputSize + " bytes for output");
331         }
332         update(in, inOfs, inLen);
333         byte[] result = doFinal();
334         int n = result.length;
335         System.arraycopy(result, 0, out, outOfs, n);
336         return n;
337     }

339     // see JCE spec
340     protected byte[] engineWrap(Key key) throws InvalidKeyException,
341             IllegalBlockSizeException {
342         byte[] encoded = key.getEncoded(); // TODO - unextractable key
343         if ((encoded == null) || (encoded.length == 0)) {
344             throw new InvalidKeyException("Could not obtain encoded key");
345         }
346         if (encoded.length > buffer.length) {
347             throw new InvalidKeyException("Key is too long for wrapping");
348         }
349         update(encoded, 0, encoded.length);
350         try {
351             return doFinal();
352         } catch (BadPaddingException e) {
353             // should not occur
354             throw new InvalidKeyException("Wrapping failed", e);
355         }
356     }

358     // see JCE spec
359     protected java.security.Key engineUnwrap(byte[] wrappedKey, String algorithm,
360             int type) throws InvalidKeyException, NoSuchAlgorithmException {
361 
362         if (wrappedKey.length > buffer.length) {
363             throw new InvalidKeyException("Key is too long for unwrapping");
364         }
365         update(wrappedKey, 0, wrappedKey.length);
366 
367         try {
368             byte[] encoding = doFinal();
369 
370             switch (type) {
371             case Cipher.PUBLIC_KEY:
372                 return constructPublicKey(encoding, algorithm);
373 
374             case Cipher.PRIVATE_KEY:
375                 return constructPrivateKey(encoding, algorithm);
376 
377             case Cipher.SECRET_KEY:
378                 return constructSecretKey(encoding, algorithm);
379 
380             default:
381                 throw new InvalidKeyException("Unknown key type " + type);
382             }
383 
384         } catch (BadPaddingException e) {
385             // should not occur
386             throw new InvalidKeyException("Unwrapping failed", e);
387 
388         } catch (IllegalBlockSizeException e) {
389             // should not occur, handled with length check above
390             throw new InvalidKeyException("Unwrapping failed", e);
391         }
392     }

394     // see JCE spec
395     protected int engineGetKeySize(Key key) throws InvalidKeyException {
396 
397         if (key instanceof sun.security.mscapi.Key) {
398             return ((sun.security.mscapi.Key) key).length();
399 
400         } else if (key instanceof RSAKey) {
401             return ((RSAKey) key).getModulus().bitLength();
402 
403         } else {
404             throw new InvalidKeyException("Unsupported key type: " + key);
405         }
406     }

408     // Construct an X.509 encoded public key.
409     private static PublicKey constructPublicKey(byte[] encodedKey,
410         String encodedKeyAlgorithm)
411             throws InvalidKeyException, NoSuchAlgorithmException {
412 
413         try {
414             KeyFactory keyFactory = KeyFactory.getInstance(encodedKeyAlgorithm);
415             X509EncodedKeySpec keySpec = new X509EncodedKeySpec(encodedKey);
416 
417             return keyFactory.generatePublic(keySpec);
418 
419         } catch (NoSuchAlgorithmException nsae) {
420             throw new NoSuchAlgorithmException("No installed provider " +
421                 "supports the " + encodedKeyAlgorithm + " algorithm", nsae);
422 
423         } catch (InvalidKeySpecException ike) {
424             throw new InvalidKeyException("Cannot construct public key", ike);
425         }
426     }
427 
428     // Construct a PKCS #8 encoded private key.
429     private static PrivateKey constructPrivateKey(byte[] encodedKey,
430         String encodedKeyAlgorithm)
431             throws InvalidKeyException, NoSuchAlgorithmException {
432 
433         try {
434             KeyFactory keyFactory = KeyFactory.getInstance(encodedKeyAlgorithm);
435             PKCS8EncodedKeySpec keySpec = new PKCS8EncodedKeySpec(encodedKey);
436 
437             return keyFactory.generatePrivate(keySpec);
438 
439         } catch (NoSuchAlgorithmException nsae) {
440             throw new NoSuchAlgorithmException("No installed provider " +
441                 "supports the " + encodedKeyAlgorithm + " algorithm", nsae);
442 
443         } catch (InvalidKeySpecException ike) {
444             throw new InvalidKeyException("Cannot construct private key", ike);
445         }
446     }

448     // Construct an encoded secret key.
