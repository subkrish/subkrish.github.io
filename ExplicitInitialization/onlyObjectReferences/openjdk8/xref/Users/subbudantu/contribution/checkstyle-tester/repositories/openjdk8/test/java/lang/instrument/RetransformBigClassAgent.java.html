24  import java.lang.instrument.*;
25  import java.util.*;
26  
27  public class RetransformBigClassAgent {
28      private static int N_RETRANSFORMS = 1000;
29      public static Class clz;
30      public static volatile boolean doneRetransforming = false;
31  
32      // just retransform the class in a loop via a Timer
33      public static void premain(String agentArgs, final Instrumentation inst) throws Exception {
34          String s = agentArgs.substring(0, agentArgs.indexOf(".class"));
35          clz = Class.forName(s.replace('/', '.'));
36  
37          ClassFileTransformer trans = new SimpleIdentityTransformer();
38          inst.addTransformer(trans, true /* canRetransform */);
39  
40          System.gc();  // throw away anything we can before we start testing
41  
42          new Timer(true).schedule(new TimerTask() {
43              public void run() {
44                  try {
45                      int i;
46                      System.out.println("Retransforming");
47                      for (i = 0; i < N_RETRANSFORMS; i++) {
48                          inst.retransformClasses(clz);
49                          System.gc();  // throw away anything we can
50                      }
51                      System.out.println("Retransformed " + i + " times.");
52                      RetransformBigClassAgent.doneRetransforming = true;
53                  }
54                  catch (Exception e) { e.printStackTrace(); }
55              }
56          }, 500);
57      }
58  }
