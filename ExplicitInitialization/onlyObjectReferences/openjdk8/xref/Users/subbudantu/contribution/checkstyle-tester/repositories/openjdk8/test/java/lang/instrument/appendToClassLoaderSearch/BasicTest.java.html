24  /*
25   *
26   *
27   * Basic unit test for Instrumentation appendToBootstrapClassLoaderSearch and
28   * appendToSystemClassLoaderSearch methods:
29   *
30   * 1. Adds a jar file to the bootstrap class loader search; Loads a class known
31   *    to be in the jar file and checks that it was loaded by the bootstrap class
32   *    loader.
33   *
34   * 2. Adds a jar file to the system class loader search; Loads a class known
35   *    to be in the jar file and checks that it was indeed loaded by the system
36   *    class loader.
37   */
38  import java.lang.instrument.Instrumentation;
39  import java.util.jar.JarFile;
40  import java.io.IOException;
41  
42  public class BasicTest {
43  
44      // count of failures
45      static int failures = 0;
46  
47      // check that the given class is loaded by the given loader
48      static void checkLoadedByLoader(String name, ClassLoader loader) {
49          try {
50              Class cl = Class.forName(name);
51              ClassLoader actual = cl.getClassLoader();
52              String loaderName = (actual == null) ? "boot class loader" : actual.toString();
53              if (actual != loader) {
54                  System.out.println("FAIL: " + name + " incorrectly loaded by: " + loaderName);
55                  failures++;
56              } else {
57                  System.out.println("PASS: " + name + " loaded by: " + loaderName);
58              }
59          } catch (Exception x) {
60              System.out.println("FAIL: Unable to load " + name + ":" + x);
61              failures++;
62          }
63      }
64  
65      public static void main(String args[]) throws IOException {
66          JarFile bootclasses = new JarFile("BootSupport.jar");
67          JarFile agentclasses = new JarFile("AgentSupport.jar");
68  
69          Instrumentation ins = Agent.getInstrumentation();
70  
71          // Test 1: Add BootSupport.jar to boot class path and check that
72          // BootSupport is loaded by the bootstrap class loader
73          ins.appendToBootstrapClassLoaderSearch(bootclasses);
74          checkLoadedByLoader("BootSupport", null);
75  
76          // Test 2: Add AgentSupport.jar to the system class path and check that
77          // AgentSupport is loaded by the system class loader.
78          try {
79              ins.appendToSystemClassLoaderSearch(agentclasses);
80              checkLoadedByLoader("AgentSupport", ClassLoader.getSystemClassLoader());
81          } catch (UnsupportedOperationException x) {
82              System.out.println("System class loader does not support adding to class path");
83          }
84  
85          // throw exception if a test failed
86          if (failures > 0) {
87              throw new RuntimeException(failures + " test(s) failed.");
88          }
89      }
90  }
