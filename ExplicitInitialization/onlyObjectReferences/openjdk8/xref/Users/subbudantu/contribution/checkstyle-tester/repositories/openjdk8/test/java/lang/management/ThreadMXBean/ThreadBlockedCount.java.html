34  import java.lang.management.*;

36  public class ThreadBlockedCount {
37      final static long EXPECTED_BLOCKED_COUNT = 3;
38      final static int  DEPTH = 10;
39      private static ThreadMXBean mbean
40          = ManagementFactory.getThreadMXBean();
41  
42      private static Object a = new Object();
43      private static Object b = new Object();
44      private static Object c = new Object();
45      private static boolean aNotified = false;
46      private static boolean bNotified = false;
47      private static boolean cNotified = false;
48      private static Object blockedObj1 = new Object();
49      private static Object blockedObj2 = new Object();
50      private static Object blockedObj3 = new Object();
51      private static volatile boolean testFailed = false;
52      private static BlockingThread blocking;
53      private static BlockedThread blocked;
54      private static ThreadExecutionSynchronizer thrsync;
55

58      public static void main(String args[]) throws Exception {
59          // Create the BlockingThread before BlockedThread
60          // to make sure BlockingThread enter the lock before BlockedThread
61          thrsync = new ThreadExecutionSynchronizer();
62  
63          blocking = new BlockingThread();
64          blocking.start();
65  
66          blocked = new BlockedThread();
67          blocked.start();
68  
69          try {
70              blocking.join();
71              blocked.join();
72          } catch (InterruptedException e) {
73              System.err.println("Unexpected exception.");
74              e.printStackTrace(System.err);
75              throw e;
76          }
77  
78          if (testFailed) {
79              throw new RuntimeException("TEST FAILED.");
80          }
81          System.out.println("Test passed.");
82      }

85      static class BlockedThread extends Thread {
86          // NOTE: We can't use a.wait() here because wait() call is counted
87          // as blockedCount.  Instead, we use a boolean flag and sleep.
88          //
89          public void run() {
90              // wait Blocking thread
91              thrsync.signal();
92  
93              // Enter lock a without blocking
94              synchronized (a) {
95                  // wait until BlockingThread holds blockedObj1
96                  while (!aNotified) {
97                      try {
98                          Thread.sleep(50);
99                      } catch (InterruptedException e) {
100                         System.err.println("Unexpected exception.");
101                         e.printStackTrace(System.err);
102                         testFailed = true;
103                         break;
104                     }
105                 }
106 
107                 // signal BlockingThread.
108                 thrsync.signal();
109 
110                 // Block to enter blockedObj1
111                 // blockedObj1 should be owned by BlockingThread
112                 synchronized (blockedObj1) {
113                     System.out.println("BlockedThread entered lock blockedObj1.");
114                 }
115             }
116 
117             // signal BlockingThread.
118             thrsync.signal();
119 
120             // Enter lock a without blocking
121             synchronized (b) {
122                 // wait until BlockingThread holds blockedObj2
123                 while (!bNotified) {
124                     try {
125                         Thread.sleep(50);
126                     } catch (InterruptedException e) {
127                         System.err.println("Unexpected exception.");
128                         e.printStackTrace(System.err);
129                         testFailed = true;
130                         break;
131                     }
132                 }
133 
134                 // signal BlockingThread.
135                 thrsync.signal();
136 
137                 // Block to enter blockedObj2
138                 // blockedObj2 should be owned by BlockingThread
139                 synchronized (blockedObj2) {
140                     System.out.println("BlockedThread entered lock blockedObj2.");
141                 }
142             }
143 
144             // signal BlockingThread.
145             thrsync.signal();
146 
147             // Enter lock a without blocking
148             synchronized (c) {
149                 // wait until BlockingThread holds blockedObj3
150                 while (!cNotified) {
151                     try {
152                         Thread.sleep(50);
153                     } catch (InterruptedException e) {
154                         System.err.println("Unexpected exception.");
155                         e.printStackTrace(System.err);
156                         testFailed = true;
157                         break;
158                     }
159                 }
160 
161                 // signal BlockingThread.
162                 thrsync.signal();
163 
164                 // Block to enter blockedObj3
165                 // blockedObj3 should be owned by BlockingThread
166                 synchronized (blockedObj3) {
167                     System.out.println("BlockedThread entered lock blockedObj3.");
168                 }
169             }
170 
171             // wait for the thread stats to be updated for 10 seconds
172             for (int i = 0; i < 100; i++) {
173                 if (getBlockedCount() == EXPECTED_BLOCKED_COUNT) {
174                     return;
175                 }
176                 try {
177                     Thread.sleep(100);
178                 } catch (InterruptedException e) {
179                     System.err.println("Unexpected exception.");
180                     e.printStackTrace(System.err);
181                     testFailed = true;
182                     return;
183                 }
184             }
185             long count = getBlockedCount();
186             if (count != EXPECTED_BLOCKED_COUNT) {
187                 System.err.println("TEST FAILED: Blocked thread has " + count +
188                                     " blocked counts. Expected " +
189                                     EXPECTED_BLOCKED_COUNT);
190                 testFailed = true;
191             }
192         } // run()
193     } // BlockingThread

195     static class BlockingThread extends Thread {
196         private void waitForSignalToRelease() {
197 
198             // wait for BlockedThread.
199             thrsync.waitForSignal();
200 
201             boolean threadBlocked = false;
202             while (!threadBlocked) {
203                 // give a chance for BlockedThread to really block
204                 try {
205                     Thread.sleep(50);
206                 } catch (InterruptedException e) {
207                     System.err.println("Unexpected exception.");
208                     e.printStackTrace(System.err);
209                     testFailed = true;
210                     break;
211                 }
212                 ThreadInfo info = mbean.getThreadInfo(blocked.getId());
213                 threadBlocked = (info.getThreadState() == Thread.State.BLOCKED);
214             }
215         }
216 
217         public void run() {
218             // wait for BlockedThread.
219             thrsync.waitForSignal();
220 
221             synchronized (blockedObj1) {
222                 System.out.println("BlockingThread attempts to notify a");
223                 aNotified = true;
224                 waitForSignalToRelease();
225             }
226 
227             // wait for BlockedThread.
228             thrsync.waitForSignal();
229 
230             // block until BlockedThread is ready
231             synchronized (blockedObj2) {
232                 System.out.println("BlockingThread attempts to notify b");
233                 bNotified = true;
234                 waitForSignalToRelease();
235             }
236 
237             // wait for BlockedThread.
238             thrsync.waitForSignal();
239 
240             // block until BlockedThread is ready
241             synchronized (blockedObj3) {
242                 System.out.println("BlockingThread attempts to notify c");
243                 cNotified = true;
244                 waitForSignalToRelease();
245             }
246 
247         } // run()
248     } // BlockedThread
249 
250     private static long getBlockedCount() {
251         long count;
252         // Check the mbean now
253         ThreadInfo ti = mbean.getThreadInfo(Thread.currentThread().
254                 getId());
255         count = ti.getBlockedCount();
256         return count;
257     }
