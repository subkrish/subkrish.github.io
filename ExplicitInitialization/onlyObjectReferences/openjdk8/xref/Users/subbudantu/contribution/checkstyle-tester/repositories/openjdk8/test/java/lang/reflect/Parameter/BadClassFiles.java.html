409     private static class InMemoryClassLoader extends ClassLoader {
410         public Class<?> defineClass(String name, byte[] b) {
411             return defineClass(name, b, 0, b.length);
412         }
413     };

442     public void assertBadParameters(Class<?> cls) throws NoSuchMethodException {
443         try {
444             System.err.println("Trying " + cls);
445             final Method method = cls.getMethod("m", int.class, int.class);
446             final Parameter[] params = method.getParameters();
447             System.err.println("Name " + params[0].getName());
448             System.err.println("Did not see expected exception");
449             errors++;
450         } catch(MalformedParametersException e) {
451             System.err.println("Expected exception seen");
452         }
453     }
454 
455     public void run() throws NoSuchMethodException {
456         for (Class<?> cls : classes)
457             assertBadParameters(cls);
458 
459         if (errors != 0)
460             throw new RuntimeException(errors + " errors in test");
461     }
