400     private static boolean areEqual(List<LanguageRange> expected,
401                                     List<LanguageRange> got) {
402         boolean error = false;
403 
404         int expectedSize = expected.size();
405         int actualSize = got.size();
406 
407         if (expectedSize != actualSize) {
408             error = true;
409 
410             System.err.println("  Expected size=" + expectedSize);
411             for (LanguageRange lr : expected) {
412                 System.err.println("    range=" + lr.getRange()
413                     + ", weight=" + lr.getWeight());
414             }
415 
416             System.out.println("  Actual size=" + actualSize);
417             for (LanguageRange lr : got) {
418                 System.err.println("    range=" + lr.getRange()
419                     + ", weight=" + lr.getWeight());
420             }
421         } else {
422             for (int i = 0; i < expectedSize; i++) {
423                 LanguageRange lr1 = expected.get(i);
424                 LanguageRange lr2 = got.get(i);
425 
426                 if (!lr1.getRange().equals(lr2.getRange())
427                     || lr1.getWeight() != lr2.getWeight()) {
428                     error = true;
429                     System.err.println("  " + i + ": Expected: range=" + lr1.getRange()
430                         + ", weight=" + lr1.getWeight());
431                     System.err.println("  " + i + ": Actual:   range=" + lr2.getRange()
432                         + ", weight=" + lr2.getWeight());
433                 }
434             }
435         }
436 
437         return !error;
438     }

917     private static List<Locale> generateLocales(String tags) {
918         if (tags == null) {
919             return null;
920         }
921 
922         List<Locale> localeList = new ArrayList<>();
923         if (tags.equals("")) {
924             return localeList;
925         }
926         String[] t = tags.split(", ");
927         for (String tag : t) {
928             localeList.add(Locale.forLanguageTag(tag));
929         }
930         return localeList;
931     }

933     private static List<String> generateLanguageTags(String tags) {
934         List<String> tagList = new ArrayList<>();
935         String[] t = tags.split(", ");
936         for (String tag : t) {
937             tagList.add(tag);
938         }
939         return tagList;
940     }

942     private static String showPriorityList(List<LanguageRange> priorityList) {
943         StringBuilder sb = new StringBuilder();
944 
945         Iterator<LanguageRange> itr = priorityList.iterator();
946         LanguageRange lr;
947         if (itr.hasNext()) {
948             lr = itr.next();
949             sb.append(lr.getRange());
950             sb.append(";q=");
951             sb.append(lr.getWeight());
952         }
953         while (itr.hasNext()) {
954             sb.append(", ");
955             lr = itr.next();
956             sb.append(lr.getRange());
957             sb.append(";q=");
958             sb.append(lr.getWeight());
959         }
960 
961         return sb.toString();
962     }

969             sb.append(itr.next());
970         }

972             sb.append(", ");
973             sb.append(itr.next());
974         }

985         }
986         while (itr.hasNext()) {
987             sb.append(", ");
