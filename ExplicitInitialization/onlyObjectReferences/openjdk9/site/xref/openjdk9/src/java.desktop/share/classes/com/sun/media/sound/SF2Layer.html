25  package com.sun.media.sound;
26  
27  import java.util.ArrayList;
28  import java.util.List;
29  
30  import javax.sound.midi.SoundbankResource;
31  
32  /**
33   * Soundfont layer.
34   *
35   * @author Karl Helgason
36   */
37  public final class SF2Layer extends SoundbankResource {
38  
39      String name = "";
40      SF2GlobalRegion globalregion = null;
41      List<SF2LayerRegion> regions = new ArrayList<>();
42  
43      public SF2Layer(SF2Soundbank soundBank) {
44          super(soundBank, null, null);
45      }
46  
47      public SF2Layer() {
48          super(null, null, null);
49      }
50  
51      @Override
52      public Object getData() {
53          return null;
54      }
55  
56      @Override
57      public String getName() {
58          return name;
59      }
60  
61      public void setName(String name) {
62          this.name = name;
63      }
64  
65      public List<SF2LayerRegion> getRegions() {
66          return regions;
67      }
68  
69      public SF2GlobalRegion getGlobalRegion() {
70          return globalregion;
71      }
72  
73      public void setGlobalZone(SF2GlobalRegion zone) {
74          globalregion = zone;
75      }
76  
77      @Override
78      public String toString() {
79          return "Layer: " + name;
80      }
81  }
