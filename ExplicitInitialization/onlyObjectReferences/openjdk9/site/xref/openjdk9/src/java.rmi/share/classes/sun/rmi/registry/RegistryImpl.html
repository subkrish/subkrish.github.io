67  /**
68   * A "registry" exists on every node that allows RMI connections to
69   * servers on that node.  The registry on a particular node contains a
70   * transient database that maps names to remote objects.  When the
71   * node boots, the registry database is empty.  The names stored in the
72   * registry are pure and are not parsed.  A service storing itself in
73   * the registry may want to prefix its name of the service by a package
74   * name (although not required), to reduce name collisions in the
75   * registry.
76   *
77   * The LocateRegistry class is used to obtain registry for different hosts.
78   *
79   * @see java.rmi.registry.LocateRegistry
80   */
81  public class RegistryImpl extends java.rmi.server.RemoteServer
82          implements Registry
83  {

117     /**
118      * Initialize the registryFilter from the security properties or system property; if any
119      * @return an ObjectInputFilter, or null
120      */
121     @SuppressWarnings("deprecation")
122     private static ObjectInputFilter initRegistryFilter() {
123         ObjectInputFilter filter = null;
124         String props = System.getProperty(REGISTRY_FILTER_PROPNAME);
125         if (props == null) {
126             props = Security.getProperty(REGISTRY_FILTER_PROPNAME);
127         }
128         if (props != null) {
129             filter = ObjectInputFilter.Config.createFilter(props);
130             Log regLog = Log.getLog("sun.rmi.registry", "registry", -1);
131             if (regLog.isLoggable(Log.BRIEF)) {
132                 regLog.log(Log.BRIEF, "registryFilter = " + filter);
133             }
134         }
135         return filter;
136     }

138     /**
139      * Construct a new RegistryImpl on the specified port with the
140      * given custom socket factory pair.
141      */
142     public RegistryImpl(int port,
143                         RMIClientSocketFactory csf,
144                         RMIServerSocketFactory ssf)
145         throws RemoteException
146     {
147         if (port == Registry.REGISTRY_PORT && System.getSecurityManager() != null) {
148             // grant permission for default port only.
149             try {
150                 AccessController.doPrivileged(new PrivilegedExceptionAction<Void>() {
151                     public Void run() throws RemoteException {
152                         LiveRef lref = new LiveRef(id, port, csf, ssf);
153                         setup(new UnicastServerRef2(lref, RegistryImpl::registryFilter));
154                         return null;
155                     }
156                 }, null, new SocketPermission("localhost:"+port, "listen,accept"));
157             } catch (PrivilegedActionException pae) {
158                 throw (RemoteException)pae.getException();
159             }
160         } else {
161             LiveRef lref = new LiveRef(id, port, csf, ssf);
162             setup(new UnicastServerRef2(lref, RegistryImpl::registryFilter));
163         }
164     }

205     /**
206      * Returns the remote object for specified name in the registry.
207      * @exception RemoteException If remote operation failed.
208      * @exception NotBoundException If name is not currently bound.
209      */
210     public Remote lookup(String name)
211         throws RemoteException, NotBoundException
212     {
213         synchronized (bindings) {
214             Remote obj = bindings.get(name);
215             if (obj == null)
216                 throw new NotBoundException(name);
217             return obj;
218         }
219     }
220 
221     /**
222      * Binds the name to the specified remote object.
223      * @exception RemoteException If remote operation failed.
224      * @exception AlreadyBoundException If name is already bound.
225      */
226     public void bind(String name, Remote obj)
227         throws RemoteException, AlreadyBoundException, AccessException
228     {
229         checkAccess("Registry.bind");
230         synchronized (bindings) {
231             Remote curr = bindings.get(name);
232             if (curr != null)
233                 throw new AlreadyBoundException(name);
234             bindings.put(name, obj);
235         }
236     }
237 
238     /**
239      * Unbind the name.
240      * @exception RemoteException If remote operation failed.
241      * @exception NotBoundException If name is not currently bound.
242      */
243     public void unbind(String name)
244         throws RemoteException, NotBoundException, AccessException
245     {
246         checkAccess("Registry.unbind");
247         synchronized (bindings) {
248             Remote obj = bindings.get(name);
249             if (obj == null)
250                 throw new NotBoundException(name);
251             bindings.remove(name);
252         }
253     }
254 
255     /**
256      * Rebind the name to a new object, replaces any existing binding.
257      * @exception RemoteException If remote operation failed.
258      */
259     public void rebind(String name, Remote obj)
260         throws RemoteException, AccessException
261     {
262         checkAccess("Registry.rebind");
263         bindings.put(name, obj);
264     }

266     /**
267      * Returns an enumeration of the names in the registry.
268      * @exception RemoteException If remote operation failed.
269      */
270     public String[] list()
271         throws RemoteException
272     {
273         String[] names;
274         synchronized (bindings) {
275             int i = bindings.size();
276             names = new String[i];
277             Enumeration<String> enum_ = bindings.keys();
278             while ((--i) >= 0)
279                 names[i] = enum_.nextElement();
280         }
281         return names;
282     }

284     /**
285      * Check that the caller has access to perform indicated operation.
286      * The client must be on same the same host as this server.
287      */
288     public static void checkAccess(String op) throws AccessException {
289 
290         try {
291             /*
292              * Get client host that this registry operation was made from.
293              */
294             final String clientHostName = getClientHost();
295             InetAddress clientHost;
296 
297             try {
298                 clientHost = java.security.AccessController.doPrivileged(
299                     new java.security.PrivilegedExceptionAction<InetAddress>() {
300                         public InetAddress run()
301                             throws java.net.UnknownHostException
302                         {
303                             return InetAddress.getByName(clientHostName);
304                         }
305                     });
306             } catch (PrivilegedActionException pae) {
307                 throw (java.net.UnknownHostException) pae.getException();
308             }
309 
310             // if client not yet seen, make sure client allowed access
311             if (allowedAccessCache.get(clientHost) == null) {
312 
313                 if (clientHost.isAnyLocalAddress()) {
314                     throw new AccessException(
315                         "Registry." + op + " disallowed; origin unknown");
316                 }
317 
318                 try {
319                     final InetAddress finalClientHost = clientHost;
320 
321                     java.security.AccessController.doPrivileged(
322                         new java.security.PrivilegedExceptionAction<Void>() {
323                             public Void run() throws java.io.IOException {
324                                 /*
325                                  * if a ServerSocket can be bound to the client's
326                                  * address then that address must be local
327                                  */
328                                 (new ServerSocket(0, 10, finalClientHost)).close();
329                                 allowedAccessCache.put(finalClientHost,
330                                                        finalClientHost);
331                                 return null;
332                             }
333                     });
334                 } catch (PrivilegedActionException pae) {
335                     // must have been an IOException
336 
337                     throw new AccessException(
338                         "Registry." + op + " disallowed; origin " +
339                         clientHost + " is non-local host");
340                 }
341             }
342         } catch (ServerNotActiveException ex) {
343             /*
344              * Local call from this VM: allow access.
345              */
346         } catch (java.net.UnknownHostException ex) {
347             throw new AccessException("Registry." + op +
348                                       " disallowed; origin is unknown host");
349         }
350     }

356     /**
357      * Retrieves text resources from the locale-specific properties file.
358      */
359     private static String getTextResource(String key) {
360         if (resources == null) {
361             try {
362                 resources = ResourceBundle.getBundle(
363                     "sun.rmi.registry.resources.rmiregistry");
364             } catch (MissingResourceException mre) {
365             }
366             if (resources == null) {
367                 // throwing an Error is a bit extreme, methinks
368                 return ("[missing resource file: " + key + "]");
369             }
370         }
371 
372         String val = null;
373         try {
374             val = resources.getString(key);
375         } catch (MissingResourceException mre) {
376         }
377 
378         if (val == null) {
379             return ("[missing resource: " + key + "]");
380         } else {
381             return (val);
382         }
383     }

385     /**
386      * Convert class path specification into an array of file URLs.
387      *
388      * The path of the file is converted to a URI then into URL
389      * form so that reserved characters can safely appear in the path.
390      */
391     private static URL[] pathToURLs(String path) {
392         List<URL> paths = new ArrayList<>();
393         for (String entry: path.split(File.pathSeparator)) {
394             Path p = Paths.get(entry);
395             try {
396                 p = p.toRealPath();
397             } catch (IOException x) {
398                 p = p.toAbsolutePath();
399             }
400             try {
401                 paths.add(p.toUri().toURL());
402             } catch (MalformedURLException e) {
403                 //ignore / skip entry
404             }
405         }
406         return paths.toArray(new URL[0]);
407     }

409     /**
410      * ObjectInputFilter to filter Registry input objects.
411      * The list of acceptable classes is limited to classes normally
412      * stored in a registry.
413      *
414      * @param filterInfo access to the class, array length, etc.
415      * @return  {@link ObjectInputFilter.Status#ALLOWED} if allowed,
416      *          {@link ObjectInputFilter.Status#REJECTED} if rejected,
417      *          otherwise {@link ObjectInputFilter.Status#UNDECIDED}
418      */
419     private static ObjectInputFilter.Status registryFilter(ObjectInputFilter.FilterInfo filterInfo) {
420         if (registryFilter != null) {
421             ObjectInputFilter.Status status = registryFilter.checkInput(filterInfo);
422             if (status != ObjectInputFilter.Status.UNDECIDED) {
423                 // The Registry filter can override the built-in white-list
424                 return status;
425             }
426         }
427 
428         if (filterInfo.depth() > REGISTRY_MAX_DEPTH) {
429             return ObjectInputFilter.Status.REJECTED;
430         }
431         Class<?> clazz = filterInfo.serialClass();
432         if (clazz != null) {
433             if (clazz.isArray()) {
434                 if (filterInfo.arrayLength() >= 0 && filterInfo.arrayLength() > REGISTRY_MAX_ARRAY_SIZE) {
435                     return ObjectInputFilter.Status.REJECTED;
436                 }
437                 do {
438                     // Arrays are allowed depending on the component type
439                     clazz = clazz.getComponentType();
440                 } while (clazz.isArray());
441             }
442             if (clazz.isPrimitive()) {
443                 // Arrays of primitives are allowed
444                 return ObjectInputFilter.Status.ALLOWED;
445             }
446             if (String.class == clazz
447                     || java.lang.Number.class.isAssignableFrom(clazz)
448                     || Remote.class.isAssignableFrom(clazz)
449                     || java.lang.reflect.Proxy.class.isAssignableFrom(clazz)
450                     || UnicastRef.class.isAssignableFrom(clazz)
451                     || RMIClientSocketFactory.class.isAssignableFrom(clazz)
452                     || RMIServerSocketFactory.class.isAssignableFrom(clazz)
453                     || java.rmi.activation.ActivationID.class.isAssignableFrom(clazz)
454                     || java.rmi.server.UID.class.isAssignableFrom(clazz)) {
455                 return ObjectInputFilter.Status.ALLOWED;
456             } else {
457                 return ObjectInputFilter.Status.REJECTED;
458             }
459         }
460         return ObjectInputFilter.Status.UNDECIDED;
461     }

463     /**
464      * Return a new RegistryImpl on the requested port and export it to serve
465      * registry requests. A classloader is initialized from the system property
466      * "env.class.path" and a security manager is set unless one is already set.
467      * <p>
468      * The returned Registry is fully functional within the current process and
469      * is usable for internal and testing purposes.
470      *
471      * @param regPort port on which the rmiregistry accepts requests;
472      *                if 0, an implementation specific port is assigned
473      * @return a RegistryImpl instance
474      * @exception RemoteException If remote operation failed.
475      * @since 9
476      */
477     public static RegistryImpl createRegistry(int regPort) throws RemoteException {
478         // Create and install the security manager if one is not installed
479         // already.
480         if (System.getSecurityManager() == null) {
481             System.setSecurityManager(new SecurityManager());
482         }
483 
484         /*
485          * Fix bugid 4147561: When JDK tools are executed, the value of
486          * the CLASSPATH environment variable for the shell in which they
487          * were invoked is no longer incorporated into the application
488          * class path; CLASSPATH's only effect is to be the value of the
489          * system property "env.class.path".  To preserve the previous
490          * (JDK1.1 and JDK1.2beta3) behavior of this tool, however, its
491          * CLASSPATH should still be considered when resolving classes
492          * being unmarshalled.  To effect this old behavior, a class
493          * loader that loads from the file path specified in the
494          * "env.class.path" property is created and set to be the context
495          * class loader before the remote object is exported.
496          */
497         String envcp = System.getProperty("env.class.path");
498         if (envcp == null) {
499             envcp = ".";            // preserve old default behavior
500         }
501         URL[] urls = pathToURLs(envcp);
502         ClassLoader cl = new URLClassLoader(urls);
503 
504         /*
505          * Fix bugid 4242317: Classes defined by this class loader should
506          * be annotated with the value of the "java.rmi.server.codebase"
507          * property, not the "file:" URLs for the CLASSPATH elements.
508          */
509         sun.rmi.server.LoaderHandler.registerCodebaseLoader(cl);
510 
511         Thread.currentThread().setContextClassLoader(cl);
512 
513         RegistryImpl registryImpl = null;
514         try {
515             registryImpl = AccessController.doPrivileged(
516                 new PrivilegedExceptionAction<RegistryImpl>() {
517                     public RegistryImpl run() throws RemoteException {
518                         return new RegistryImpl(regPort);
519                     }
520                 }, getAccessControlContext(regPort));
521         } catch (PrivilegedActionException ex) {
522             throw (RemoteException) ex.getException();
523         }
524 
525         return registryImpl;
526     }

528     /**
529      * Main program to start a registry. <br>
530      * The port number can be specified on the command line.
531      */
532     public static void main(String args[])
533     {
534         try {
535             final int regPort = (args.length >= 1) ? Integer.parseInt(args[0])
536                                                    : Registry.REGISTRY_PORT;
537 
538             registry = createRegistry(regPort);
539 
540             // prevent registry from exiting
541             while (true) {
542                 try {
543                     Thread.sleep(Long.MAX_VALUE);
544                 } catch (InterruptedException e) {
545                 }
546             }
547         } catch (NumberFormatException e) {
548             System.err.println(MessageFormat.format(
549                 getTextResource("rmiregistry.port.badnumber"),
550                 args[0] ));
551             System.err.println(MessageFormat.format(
552                 getTextResource("rmiregistry.usage"),
553                 "rmiregistry" ));
554         } catch (Exception e) {
555             e.printStackTrace();
556         }
557         System.exit(1);
558     }

560     /**
561      * Generates an AccessControlContext with minimal permissions.
562      * The approach used here is taken from the similar method
563      * getAccessControlContext() in the sun.applet.AppletPanel class.
564      */
565     private static AccessControlContext getAccessControlContext(int port) {
566         // begin with permissions granted to all code in current policy
567         PermissionCollection perms = AccessController.doPrivileged(
568             new java.security.PrivilegedAction<PermissionCollection>() {
569                 public PermissionCollection run() {
570                     CodeSource codesource = new CodeSource(null,
571                         (java.security.cert.Certificate[]) null);
572                     Policy p = java.security.Policy.getPolicy();
573                     if (p != null) {
574                         return p.getPermissions(codesource);
575                     } else {
576                         return new Permissions();
577                     }
578                 }
579             });
580 
581         /*
582          * Anyone can connect to the registry and the registry can connect
583          * to and possibly download stubs from anywhere. Downloaded stubs and
584          * related classes themselves are more tightly limited by RMI.
585          */
586         perms.add(new SocketPermission("*", "connect,accept"));
587         perms.add(new SocketPermission("localhost:"+port, "listen,accept"));
588 
589         perms.add(new RuntimePermission("accessClassInPackage.sun.jvmstat.*"));
590         perms.add(new RuntimePermission("accessClassInPackage.sun.jvm.hotspot.*"));
591 
592         perms.add(new FilePermission("<<ALL FILES>>", "read"));
593 
594         /*
595          * Create an AccessControlContext that consists of a single
596          * protection domain with only the permissions calculated above.
597          */
598         ProtectionDomain pd = new ProtectionDomain(
599             new CodeSource(null,
600                 (java.security.cert.Certificate[]) null), perms);
601         return new AccessControlContext(new ProtectionDomain[] { pd });
602     }
