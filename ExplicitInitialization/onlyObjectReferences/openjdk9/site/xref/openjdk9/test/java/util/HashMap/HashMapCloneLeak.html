24  /**
25   * @test
26   * @bug 7042126
27   * @summary Verify that we do not leak contents when we clone a HashMap
28   * @author david.buck@oracle.com
29   * @run main/othervm HashMapCloneLeak
30   * @run main/othervm -XX:+AggressiveOpts HashMapCloneLeak
31   */
32  
33  import java.util.HashMap;
34  import java.lang.ref.WeakReference;
35  
36  public class HashMapCloneLeak {
37  
38      static WeakReference<Object> wr = null;
39  
40      // helper method to keep testObject and map out of main method's scope
41      private static HashMap<Integer, Object> makeMap() {
42          HashMap<Integer, Object> map = new HashMap<Integer, Object>();
43          Object testObject = new Object();
44          wr = new WeakReference<Object>(testObject);
45          map.put(42, testObject);
46          return map;
47      }
48  
49      public static void main(String[] args) throws Exception {
50          HashMap<Integer, Object> hm = makeMap();
51          hm = (HashMap<Integer, Object>)hm.clone();
52          hm.clear();
53          // There should no longer be a strong reference to testObject
54          // the WeakReference should be nulled out by GC. If not,
55          // we will hang here until timed out by the test harness.
56          Object[] chain = null;
57          while (wr.get() != null) {
58              try {
59                  Object[] allocate = new Object[1000000];
60                  allocate[0] = chain;
61                  chain = allocate;
62              } catch (OutOfMemoryError oome) {
63                  chain = null;
64              }
65              System.gc();
66              Thread.sleep(100);
67          }
68      }
69  
70  }
