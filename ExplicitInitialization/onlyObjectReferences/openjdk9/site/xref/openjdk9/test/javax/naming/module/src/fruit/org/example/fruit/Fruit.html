24  package org.example.fruit;
25  
26  import javax.naming.*;
27  
28  /**
29    * This class is used by the StoreFruit test.
30    * It is a referenceable class that can be stored by service
31    * providers like the LDAP and file system providers.
32    */
33  public class Fruit implements Referenceable {
34      static String location = null;
35      String fruit;
36  
37      public Fruit(String f) {
38          fruit = f;
39      }
40  
41      public Reference getReference() throws NamingException {
42          return new Reference(
43              Fruit.class.getName(),
44              new StringRefAddr("fruit", fruit),
45              FruitFactory.class.getName(),
46              location);          // factory location
47      }
48  
49      public String toString() {
50          return fruit;
51      }
52  
53      public static void setLocation(String loc) {
54          location = loc;
55  System.out.println("setting location to : " + location);
56      }
57  }
