44  /**
45   * A build tool to extend the module-info.java in the source tree for
46   * platform-specific exports, opens, uses, and provides and write to
47   * the specified output file.
48   *
49   * GenModuleInfoSource will be invoked for each module that has
50   * module-info.java.extra in the source directory.
51   *
52   * The extra exports, opens, uses, provides can be specified
53   * in module-info.java.extra.
54   * Injecting platform-specific requires is not supported.
55   *
56   * @see build.tools.module.ModuleInfoExtraTest for basic testing
57   */
58  public class GenModuleInfoSource {
59      private final static String USAGE =
60          "Usage: GenModuleInfoSource -o <output file> \n" +
61          "  --source-file <module-info-java>\n" +
62          "  --modules <module-name>[,<module-name>...]\n" +
63          "  <module-info.java.extra> ...\n";
64  
65      static boolean verbose = false;
66      public static void main(String... args) throws Exception {
67          Path outfile = null;
68          Path moduleInfoJava = null;
69          Set<String> modules = Collections.emptySet();
70          List<Path> extras = new ArrayList<>();
71          // validate input arguments
72          for (int i = 0; i < args.length; i++){
73              String option = args[i];
74              String arg = i+1 < args.length ? args[i+1] : null;
75              switch (option) {
76                  case "-o":
77                      outfile = Paths.get(arg);
78                      i++;
79                      break;
80                  case "--source-file":
81                      moduleInfoJava = Paths.get(arg);
82                      if (Files.notExists(moduleInfoJava)) {
83                          throw new IllegalArgumentException(moduleInfoJava + " not exist");
84                      }
85                      i++;
86                      break;
87                  case "--modules":
88                      modules = Arrays.stream(arg.split(","))
89                                      .collect(toSet());
90                      i++;
91                      break;
92                  case "-v":
93                      verbose = true;
94                      break;
95                  default:
96                      Path file = Paths.get(option);
97                      if (Files.notExists(file)) {
98                          throw new IllegalArgumentException(file + " not exist");
99                      }
100                     extras.add(file);
101             }
102         }
103 
104         if (moduleInfoJava == null || outfile == null ||
105                 modules.isEmpty() || extras.isEmpty()) {
106             System.err.println(USAGE);
107             System.exit(-1);
108         }
109 
110         GenModuleInfoSource genModuleInfo =
111             new GenModuleInfoSource(moduleInfoJava, extras, modules);
112 
113         // generate new module-info.java
114         genModuleInfo.generate(outfile);
115     }
116 
117     final Path sourceFile;
118     final List<Path> extraFiles;
119     final ModuleInfo extras;
120     final Set<String> modules;
121     final ModuleInfo moduleInfo;
122     GenModuleInfoSource(Path sourceFile, List<Path> extraFiles, Set<String> modules)
123         throws IOException
124     {
125         this.sourceFile = sourceFile;
126         this.extraFiles = extraFiles;
127         this.modules = modules;
128         this.moduleInfo = new ModuleInfo();
129         this.moduleInfo.parse(sourceFile);
130 
131         // parse module-info.java.extra
132         this.extras = new ModuleInfo();
133         for (Path file : extraFiles) {
134             extras.parse(file);
135         }
136 
137         // merge with module-info.java.extra
138         moduleInfo.augmentModuleInfo(extras, modules);
139     }

141     void generate(Path output) throws IOException {
142         List<String> lines = Files.readAllLines(sourceFile);
143         try (BufferedWriter bw = Files.newBufferedWriter(output);
144              PrintWriter writer = new PrintWriter(bw)) {
145             // write the copyright header and lines up to module declaration
146             for (String l : lines) {
147                 writer.println(l);
148                 if (l.trim().startsWith("module ")) {
149                     // print URI rather than file path to avoid escape
150                     writer.format("    // source file: %s%n", sourceFile.toUri());
151                     for (Path file: extraFiles) {
152                         writer.format("    //              %s%n", file.toUri());
153                     }
154                     break;
155                 }
156             }
157 
158             // requires
159             for (String l : lines) {
160                 if (l.trim().startsWith("requires"))
161                     writer.println(l);
162             }
163 
164             // write exports, opens, uses, and provides
165             moduleInfo.print(writer);
166 
167             // close
168             writer.println("}");
169         }
170     }

173     class ModuleInfo {
174         final Map<String, Statement> exports = new HashMap<>();
175         final Map<String, Statement> opens = new HashMap<>();
176         final Map<String, Statement> uses = new HashMap<>();
177         final Map<String, Statement> provides = new HashMap<>();
178 
179         Statement getStatement(String directive, String name) {
180             switch (directive) {
181                 case "exports":
182                     if (moduleInfo.exports.containsKey(name) &&
183                         moduleInfo.exports.get(name).isUnqualified()) {
184                         throw new IllegalArgumentException(sourceFile +
185                             " already has " + directive + " " + name);
186                     }
187                     return exports.computeIfAbsent(name,
188                         _n -> new Statement("exports", "to", name));
189 
190                 case "opens":
191                     if (moduleInfo.opens.containsKey(name) &&
192                         moduleInfo.opens.get(name).isUnqualified()) {
193                         throw new IllegalArgumentException(sourceFile +
194                             " already has " + directive + " " + name);
195                     }
196 
197                     if (moduleInfo.opens.containsKey(name)) {
198                         throw new IllegalArgumentException(sourceFile +
199                             " already has " + directive + " " + name);
200                     }
201                     return opens.computeIfAbsent(name,
202                         _n -> new Statement("opens", "to", name));
203 
204                 case "uses":
205                     return uses.computeIfAbsent(name,
206                         _n -> new Statement("uses", "", name));
207 
208                 case "provides":
209                     return provides.computeIfAbsent(name,
210                         _n -> new Statement("provides", "with", name, true));
211 
212                 default:
213                     throw new IllegalArgumentException(directive);
214             }
215 
216         }

218         /*
219          * Augment this ModuleInfo with module-info.java.extra
220          */
221         void augmentModuleInfo(ModuleInfo extraFiles, Set<String> modules) {
222             // API package exported in the original module-info.java
223             extraFiles.exports.entrySet()
224                 .stream()
225                 .filter(e -> exports.containsKey(e.getKey()) &&
226                                 e.getValue().filter(modules))
227                 .forEach(e -> mergeExportsOrOpens(exports.get(e.getKey()),
228                                                   e.getValue(),
229                                                   modules));
230 
231             // add exports that are not defined in the original module-info.java
232             extraFiles.exports.entrySet()
233                 .stream()
234                 .filter(e -> !exports.containsKey(e.getKey()) &&
235                                 e.getValue().filter(modules))
236                 .forEach(e -> addTargets(getStatement("exports", e.getKey()),
237                                          e.getValue(),
238                                          modules));
239 
240             // API package opened in the original module-info.java
241             extraFiles.opens.entrySet()
242                 .stream()
243                 .filter(e -> opens.containsKey(e.getKey()) &&
244                                 e.getValue().filter(modules))
245                 .forEach(e -> mergeExportsOrOpens(opens.get(e.getKey()),
246                                                   e.getValue(),
247                                                   modules));
248 
249             // add opens that are not defined in the original module-info.java
250             extraFiles.opens.entrySet()
251                 .stream()
252                 .filter(e -> !opens.containsKey(e.getKey()) &&
253                                 e.getValue().filter(modules))
254                 .forEach(e -> addTargets(getStatement("opens", e.getKey()),
255                                          e.getValue(),
256                                          modules));
257 
258             // provides
259             extraFiles.provides.keySet()
260                 .stream()
261                 .filter(service -> provides.containsKey(service))
262                 .forEach(service -> mergeProvides(service,
263                                                   extraFiles.provides.get(service)));
264             extraFiles.provides.keySet()
265                 .stream()
266                 .filter(service -> !provides.containsKey(service))
267                 .forEach(service -> provides.put(service,
268                                                  extraFiles.provides.get(service)));
269 
270             // uses
271             extraFiles.uses.keySet()
272                 .stream()
273                 .filter(service -> !uses.containsKey(service))
274                 .forEach(service -> uses.put(service, extraFiles.uses.get(service)));
275         }

277         // add qualified exports or opens to known modules only
278         private void addTargets(Statement statement,
279                                 Statement extra,
280                                 Set<String> modules)
281         {
282             extra.targets.stream()
283                  .filter(mn -> modules.contains(mn))
284                  .forEach(mn -> statement.addTarget(mn));
285         }
286 
287         private void mergeExportsOrOpens(Statement statement,
288                                          Statement extra,
289                                          Set<String> modules)
290         {
291             String pn = statement.name;
292             if (statement.isUnqualified() && extra.isQualified()) {
293                 throw new RuntimeException("can't add qualified exports to " +
294                     "unqualified exports " + pn);
295             }
296 
297             Set<String> mods = extra.targets.stream()
298                 .filter(mn -> statement.targets.contains(mn))
299                 .collect(toSet());
300             if (mods.size() > 0) {
301                 throw new RuntimeException("qualified exports " + pn + " to " +
302                     mods.toString() + " already declared in " + sourceFile);
303             }
304 
305             // add qualified exports or opens to known modules only
306             addTargets(statement, extra, modules);
307         }
308 
309         private void mergeProvides(String service, Statement extra) {
310             Statement statement = provides.get(service);
311 
312             Set<String> mods = extra.targets.stream()
313                 .filter(mn -> statement.targets.contains(mn))
314                 .collect(toSet());
315 
316             if (mods.size() > 0) {
317                 throw new RuntimeException("qualified exports " + service + " to " +
318                     mods.toString() + " already declared in " + sourceFile);
319             }
320 
321             extra.targets.stream()
322                  .forEach(mn -> statement.addTarget(mn));
323         }

326         void print(PrintWriter writer) {
327             // print unqualified exports
328             exports.entrySet().stream()
329                 .filter(e -> e.getValue().targets.isEmpty())
330                 .sorted(Map.Entry.comparingByKey())
331                 .forEach(e -> writer.println(e.getValue()));
332 
333             // print qualified exports
334             exports.entrySet().stream()
335                 .filter(e -> !e.getValue().targets.isEmpty())
336                 .sorted(Map.Entry.comparingByKey())
337                 .forEach(e -> writer.println(e.getValue()));
338 
339             // print unqualified opens
340             opens.entrySet().stream()
341                 .filter(e -> e.getValue().targets.isEmpty())
342                 .sorted(Map.Entry.comparingByKey())
343                 .forEach(e -> writer.println(e.getValue()));
344 
345             // print qualified opens
346             opens.entrySet().stream()
347                 .filter(e -> !e.getValue().targets.isEmpty())
348                 .sorted(Map.Entry.comparingByKey())
349                 .forEach(e -> writer.println(e.getValue()));
350 
351             // uses and provides
352             writer.println();
353             uses.entrySet().stream()
354                 .sorted(Map.Entry.comparingByKey())
355                 .forEach(e -> writer.println(e.getValue()));
356             provides.entrySet().stream()
357                 .sorted(Map.Entry.comparingByKey())
358                 .forEach(e -> writer.println(e.getValue()));
359         }

361         private void parse(Path sourcefile) throws IOException {
362             List<String> lines = Files.readAllLines(sourcefile);
363             Statement statement = null;
364             boolean hasTargets = false;
365 
366             for (int lineNumber = 1; lineNumber <= lines.size(); ) {
367                 String l = lines.get(lineNumber-1).trim();
368                 int index = 0;
369 
370                 if (l.isEmpty()) {
371                     lineNumber++;
372                     continue;
373                 }
374 
375                 // comment block starts
376                 if (l.startsWith("/*")) {
377                     while (l.indexOf("*/") == -1) { // end comment block
378                         l = lines.get(lineNumber++).trim();
379                     }
380                     index = l.indexOf("*/") + 2;
381                     if (index >= l.length()) {
382                         lineNumber++;
383                         continue;
384                     } else {
385                         // rest of the line
386                         l = l.substring(index, l.length()).trim();
387                         index = 0;
388                     }
389                 }
390 
391                 // skip comment and annotations
392                 if (l.startsWith("//") || l.startsWith("@")) {
393                     lineNumber++;
394                     continue;
395                 }
396 
397                 int current = lineNumber;
398                 int count = 0;
399                 while (index < l.length()) {
400                     if (current == lineNumber && ++count > 20)
401                         throw new Error("Fail to parse line " + lineNumber + " " + sourcefile);
402 
403                     int end = l.indexOf(';');
404                     if (end == -1)
405                         end = l.length();
406                     String content = l.substring(0, end).trim();
407                     if (content.isEmpty()) {
408                         index = end+1;
409                         if (index < l.length()) {
410                             // rest of the line
411                             l = l.substring(index, l.length()).trim();
412                             index = 0;
413                         }
414                         continue;
415                     }
416 
417                     String[] s = content.split("\\s+");
418                     String keyword = s[0].trim();
419 
420                     String name = s.length > 1 ? s[1].trim() : null;
421                     trace("%d: %s index=%d len=%d%n", lineNumber, l, index, l.length());
422                     switch (keyword) {
423                         case "module":
424                         case "requires":
425                         case "}":
426                             index = l.length();  // skip to the end
427                             continue;
428 
429                         case "exports":
430                         case "opens":
431                         case "provides":
432                         case "uses":
433                             // assume name immediately after exports, opens, provides, uses
434                             statement = getStatement(keyword, name);
435                             hasTargets = false;
436 
437                             int i = l.indexOf(name, keyword.length()+1) + name.length() + 1;
438                             l = i < l.length() ? l.substring(i, l.length()).trim() : "";
439                             index = 0;
440 
441                             if (s.length >= 3) {
442                                 if (!s[2].trim().equals(statement.qualifier)) {
443                                     throw new RuntimeException(sourcefile + ", line " +
444                                         lineNumber + ", is malformed: " + s[2]);
445                                 }
446                             }
447 
448                             break;
449 
450                         case "to":
451                         case "with":
452                             if (statement == null) {
453                                 throw new RuntimeException(sourcefile + ", line " +
454                                     lineNumber + ", is malformed");
455                             }
456 
457                             hasTargets = true;
458                             String qualifier = statement.qualifier;
459                             i = l.indexOf(qualifier, index) + qualifier.length() + 1;
460                             l = i < l.length() ? l.substring(i, l.length()).trim() : "";
461                             index = 0;
462                             break;
463                     }
464 
465                     if (index >= l.length()) {
466                         // skip to next line
467                         continue;
468                     }
469 
470                         // comment block starts
471                     if (l.startsWith("/*")) {
472                         while (l.indexOf("*/") == -1) { // end comment block
473                             l = lines.get(lineNumber++).trim();
474                         }
475                         index = l.indexOf("*/") + 2;
476                         if (index >= l.length()) {
477                             continue;
478                         } else {
479                             // rest of the line
480                             l = l.substring(index, l.length()).trim();
481                             index = 0;
482                         }
483                     }
484 
485                     if (l.startsWith("//")) {
486                         index = l.length();
487                         continue;
488                     }
489 
490                     if (statement == null) {
491                         throw new RuntimeException(sourcefile + ", line " +
492                             lineNumber + ": missing keyword?");
493                     }
494 
495                     if (!hasTargets) {
496                         continue;
497                     }
498 
499                     if (index >= l.length()) {
500                         throw new RuntimeException(sourcefile + ", line " +
501                             lineNumber + ": " + l);
502                     }
503 
504                     // parse the target module of exports, opens, or provides
505                     Statement stmt = statement;
506 
507                     int terminal = l.indexOf(';', index);
508                     // determine up to which position to parse
509                     int pos = terminal != -1 ? terminal : l.length();
510                     // parse up to comments
511                     int pos1 = l.indexOf("//", index);
512                     if (pos1 != -1 && pos1 < pos) {
513                         pos = pos1;
514                     }
515                     int pos2 = l.indexOf("/*", index);
516                     if (pos2 != -1 && pos2 < pos) {
517                         pos = pos2;
518                     }
519                     // target module(s) for qualitifed exports or opens
520                     // or provider implementation class(es)
521                     String rhs = l.substring(index, pos).trim();
522                     index += rhs.length();
523                     trace("rhs: index=%d [%s] [line: %s]%n", index, rhs, l);
524 
525                     String[] targets = rhs.split(",");
526                     for (String t : targets) {
527                         String n = t.trim();
528                         if (n.length() > 0)
529                             stmt.addTarget(n);
530                     }
531 
532                     // start next statement
533                     if (pos == terminal) {
534                         statement = null;
535                         hasTargets = false;
536                         index = terminal + 1;
537                     }
538                     l = index < l.length() ? l.substring(index, l.length()).trim() : "";
539                     index = 0;
540                 }
541 
542                 lineNumber++;
543             }
544         }

547     static class Statement {
548         final String directive;
549         final String qualifier;
550         final String name;
551         final Set<String> targets = new LinkedHashSet<>();
552         final boolean ordered;
553 
554         Statement(String directive, String qualifier, String name) {
555             this(directive, qualifier, name, false);
556         }
557 
558         Statement(String directive, String qualifier, String name, boolean ordered) {
559             this.directive = directive;
560             this.qualifier = qualifier;
561             this.name = name;
562             this.ordered = ordered;
563         }
564 
565         Statement addTarget(String mn) {
566             if (mn.isEmpty())
567                 throw new IllegalArgumentException("empty module name");
568             targets.add(mn);
569             return this;
570         }
571 
572         boolean isQualified() {
573             return targets.size() > 0;
574         }
575 
576         boolean isUnqualified() {
577             return targets.isEmpty();
578         }
579 
580         /**
581          * Returns true if this statement is unqualified or it has
582          * at least one target in the given names.
583          */
584         boolean filter(Set<String> names) {
585             if (isUnqualified()) {
586                 return true;
587             } else {
588                 return targets.stream()
589                     .filter(mn -> names.contains(mn))
590                     .findAny().isPresent();
591             }
592         }
593 
594         @Override
595         public String toString() {
596             StringBuilder sb = new StringBuilder("    ");
597             sb.append(directive).append(" ").append(name);
598             if (targets.isEmpty()) {
599                 sb.append(";");
600             } else if (targets.size() == 1) {
601                 sb.append(" ").append(qualifier)
602                   .append(orderedTargets().collect(joining(",", " ", ";")));
603             } else {
604                 sb.append(" ").append(qualifier)
605                   .append(orderedTargets()
606                       .map(target -> String.format("        %s", target))
607                       .collect(joining(",\n", "\n", ";")));
608             }
609             return sb.toString();
610         }
611 
612         public Stream<String> orderedTargets() {
613             return ordered ? targets.stream()
614                            : targets.stream().sorted();
615         }
616     }

618     static void trace(String fmt, Object... params) {
619         if (verbose) {
620             System.out.format(fmt, params);
621         }
