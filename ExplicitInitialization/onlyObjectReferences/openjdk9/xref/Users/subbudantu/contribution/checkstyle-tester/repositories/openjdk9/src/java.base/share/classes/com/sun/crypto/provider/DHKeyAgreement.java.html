46  /**
47   * This class implements the Diffie-Hellman key agreement protocol between
48   * any number of parties.
49   *
50   * @author Jan Luehe
51   *
52   */
53  
54  public final class DHKeyAgreement
55  extends KeyAgreementSpi {
56  
57      private boolean generateSecret = false;
58      private BigInteger init_p = null;
59      private BigInteger init_g = null;
60      private BigInteger x = BigInteger.ZERO; // the private value
61      private BigInteger y = BigInteger.ZERO;
62  
63      /**
64       * Empty constructor
65       */
66      public DHKeyAgreement() {
67      }

69      /**
70       * Initializes this key agreement with the given key and source of
71       * randomness. The given key is required to contain all the algorithm
72       * parameters required for this key agreement.
73       *
74       * <p> If the key agreement algorithm requires random bytes, it gets them
75       * from the given source of randomness, <code>random</code>.
76       * However, if the underlying
77       * algorithm implementation does not require any random bytes,
78       * <code>random</code> is ignored.
79       *
80       * @param key the party's private information. For example, in the case
81       * of the Diffie-Hellman key agreement, this would be the party's own
82       * Diffie-Hellman private key.
83       * @param random the source of randomness
84       *
85       * @exception InvalidKeyException if the given key is
86       * inappropriate for this key agreement, e.g., is of the wrong type or
87       * has an incompatible algorithm type.
88       */
89      protected void engineInit(Key key, SecureRandom random)
90          throws InvalidKeyException
91      {
92          try {
93              engineInit(key, null, random);
94          } catch (InvalidAlgorithmParameterException e) {
95              // never happens, because we did not pass any parameters
96          }
97      }
98  
99      /**
100      * Initializes this key agreement with the given key, set of
101      * algorithm parameters, and source of randomness.
102      *
103      * @param key the party's private information. For example, in the case
104      * of the Diffie-Hellman key agreement, this would be the party's own
105      * Diffie-Hellman private key.
106      * @param params the key agreement parameters
107      * @param random the source of randomness
108      *
109      * @exception InvalidKeyException if the given key is
110      * inappropriate for this key agreement, e.g., is of the wrong type or
111      * has an incompatible algorithm type.
112      * @exception InvalidAlgorithmParameterException if the given parameters
113      * are inappropriate for this key agreement.
114      */
115     protected void engineInit(Key key, AlgorithmParameterSpec params,
116                               SecureRandom random)
117         throws InvalidKeyException, InvalidAlgorithmParameterException
118     {
119         // ignore "random" parameter, because our implementation does not
120         // require any source of randomness
121         generateSecret = false;
122         init_p = null;
123         init_g = null;
124 
125         if ((params != null) && !(params instanceof DHParameterSpec)) {
126             throw new InvalidAlgorithmParameterException
127                 ("Diffie-Hellman parameters expected");
128         }
129         if (!(key instanceof javax.crypto.interfaces.DHPrivateKey)) {
130             throw new InvalidKeyException("Diffie-Hellman private key "
131                                           + "expected");
132         }
133         javax.crypto.interfaces.DHPrivateKey dhPrivKey;
134         dhPrivKey = (javax.crypto.interfaces.DHPrivateKey)key;
135 
136         // check if private key parameters are compatible with
137         // initialized ones
138         if (params != null) {
139             init_p = ((DHParameterSpec)params).getP();
140             init_g = ((DHParameterSpec)params).getG();
141         }
142         BigInteger priv_p = dhPrivKey.getParams().getP();
143         BigInteger priv_g = dhPrivKey.getParams().getG();
144         if (init_p != null && priv_p != null && !(init_p.equals(priv_p))) {
145             throw new InvalidKeyException("Incompatible parameters");
146         }
147         if (init_g != null && priv_g != null && !(init_g.equals(priv_g))) {
148             throw new InvalidKeyException("Incompatible parameters");
149         }
150         if ((init_p == null && priv_p == null)
151             || (init_g == null && priv_g == null)) {
152             throw new InvalidKeyException("Missing parameters");
153         }
154         init_p = priv_p;
155         init_g = priv_g;
156 
157         // store the x value
158         this.x = dhPrivKey.getX();
159     }
160 
161     /**
162      * Executes the next phase of this key agreement with the given
163      * key that was received from one of the other parties involved in this key
164      * agreement.
165      *
166      * @param key the key for this phase. For example, in the case of
167      * Diffie-Hellman between 2 parties, this would be the other party's
168      * Diffie-Hellman public key.
169      * @param lastPhase flag which indicates whether or not this is the last
170      * phase of this key agreement.
171      *
172      * @return the (intermediate) key resulting from this phase, or null if
173      * this phase does not yield a key
174      *
175      * @exception InvalidKeyException if the given key is inappropriate for
176      * this phase.
177      * @exception IllegalStateException if this key agreement has not been
178      * initialized.
179      */
180     protected Key engineDoPhase(Key key, boolean lastPhase)
181         throws InvalidKeyException, IllegalStateException
182     {
183         if (!(key instanceof javax.crypto.interfaces.DHPublicKey)) {
184             throw new InvalidKeyException("Diffie-Hellman public key "
185                                           + "expected");
186         }
187         javax.crypto.interfaces.DHPublicKey dhPubKey;
188         dhPubKey = (javax.crypto.interfaces.DHPublicKey)key;
189 
190         if (init_p == null || init_g == null) {
191             throw new IllegalStateException("Not initialized");
192         }
193 
194         // check if public key parameters are compatible with
195         // initialized ones
196         BigInteger pub_p = dhPubKey.getParams().getP();
197         BigInteger pub_g = dhPubKey.getParams().getG();
198         if (pub_p != null && !(init_p.equals(pub_p))) {
199             throw new InvalidKeyException("Incompatible parameters");
200         }
201         if (pub_g != null && !(init_g.equals(pub_g))) {
202             throw new InvalidKeyException("Incompatible parameters");
203         }
204 
205         // validate the Diffie-Hellman public key
206         KeyUtil.validate(dhPubKey);
207 
208         // store the y value
209         this.y = dhPubKey.getY();
210 
211         // we've received a public key (from one of the other parties),
212         // so we are ready to create the secret, which may be an
213         // intermediate secret, in which case we wrap it into a
214         // Diffie-Hellman public key object and return it.
215         generateSecret = true;
216         if (lastPhase == false) {
217             byte[] intermediate = engineGenerateSecret();
218             return new DHPublicKey(new BigInteger(1, intermediate),
219                                    init_p, init_g);
220         } else {
221             return null;
222         }
223     }

225     /**
226      * Generates the shared secret and returns it in a new buffer.
227      *
228      * <p>This method resets this <code>KeyAgreementSpi</code> object,
229      * so that it
230      * can be reused for further key agreements. Unless this key agreement is
231      * reinitialized with one of the <code>engineInit</code> methods, the same
232      * private information and algorithm parameters will be used for
233      * subsequent key agreements.
234      *
235      * @return the new buffer with the shared secret
236      *
237      * @exception IllegalStateException if this key agreement has not been
238      * completed yet
239      */
240     protected byte[] engineGenerateSecret()
241         throws IllegalStateException
242     {
243         int expectedLen = (init_p.bitLength() + 7) >>> 3;
244         byte[] result = new byte[expectedLen];
245         try {
246             engineGenerateSecret(result, 0);
247         } catch (ShortBufferException sbe) {
248             // should never happen since length are identical
249         }
250         return result;
251     }
252 
253     /**
254      * Generates the shared secret, and places it into the buffer
255      * <code>sharedSecret</code>, beginning at <code>offset</code>.
256      *
257      * <p>If the <code>sharedSecret</code> buffer is too small to hold the
258      * result, a <code>ShortBufferException</code> is thrown.
259      * In this case, this call should be repeated with a larger output buffer.
260      *
261      * <p>This method resets this <code>KeyAgreementSpi</code> object,
262      * so that it
263      * can be reused for further key agreements. Unless this key agreement is
264      * reinitialized with one of the <code>engineInit</code> methods, the same
265      * private information and algorithm parameters will be used for
266      * subsequent key agreements.
267      *
268      * @param sharedSecret the buffer for the shared secret
269      * @param offset the offset in <code>sharedSecret</code> where the
270      * shared secret will be stored
271      *
272      * @return the number of bytes placed into <code>sharedSecret</code>
273      *
274      * @exception IllegalStateException if this key agreement has not been
275      * completed yet
276      * @exception ShortBufferException if the given output buffer is too small
277      * to hold the secret
278      */
279     protected int engineGenerateSecret(byte[] sharedSecret, int offset)
280         throws IllegalStateException, ShortBufferException
281     {
282         if (generateSecret == false) {
283             throw new IllegalStateException
284                 ("Key agreement has not been completed yet");
285         }
286 
287         if (sharedSecret == null) {
288             throw new ShortBufferException
289                 ("No buffer provided for shared secret");
290         }
291 
292         BigInteger modulus = init_p;
293         int expectedLen = (modulus.bitLength() + 7) >>> 3;
294         if ((sharedSecret.length - offset) < expectedLen) {
295             throw new ShortBufferException
296                     ("Buffer too short for shared secret");
297         }
298 
299         // Reset the key agreement after checking for ShortBufferException
300         // above, so user can recover w/o losing internal state
301         generateSecret = false;
302 
303         /*
304          * NOTE: BigInteger.toByteArray() returns a byte array containing
305          * the two's-complement representation of this BigInteger with
306          * the most significant byte is in the zeroth element. This
307          * contains the minimum number of bytes required to represent
308          * this BigInteger, including at least one sign bit whose value
309          * is always 0.
310          *
311          * Keys are always positive, and the above sign bit isn't
312          * actually used when representing keys.  (i.e. key = new
313          * BigInteger(1, byteArray))  To obtain an array containing
314          * exactly expectedLen bytes of magnitude, we strip any extra
315          * leading 0's, or pad with 0's in case of a "short" secret.
316          */
317         byte[] secret = this.y.modPow(this.x, modulus).toByteArray();
318         if (secret.length == expectedLen) {
319             System.arraycopy(secret, 0, sharedSecret, offset,
320                              secret.length);
321         } else {
322             // Array too short, pad it w/ leading 0s
323             if (secret.length < expectedLen) {
324                 System.arraycopy(secret, 0, sharedSecret,
325                     offset + (expectedLen - secret.length),
326                     secret.length);
327             } else {
328                 // Array too long, check and trim off the excess
329                 if ((secret.length == (expectedLen+1)) && secret[0] == 0) {
330                     // ignore the leading sign byte
331                     System.arraycopy(secret, 1, sharedSecret, offset, expectedLen);
332                 } else {
333                     throw new ProviderException("Generated secret is out-of-range");
334                 }
335             }
336         }
337         return expectedLen;
338     }

340     /**
341      * Creates the shared secret and returns it as a secret key object
342      * of the requested algorithm type.
343      *
344      * <p>This method resets this <code>KeyAgreementSpi</code> object,
345      * so that it
346      * can be reused for further key agreements. Unless this key agreement is
347      * reinitialized with one of the <code>engineInit</code> methods, the same
348      * private information and algorithm parameters will be used for
349      * subsequent key agreements.
350      *
351      * @param algorithm the requested secret key algorithm
352      *
353      * @return the shared secret key
354      *
355      * @exception IllegalStateException if this key agreement has not been
356      * completed yet
357      * @exception NoSuchAlgorithmException if the requested secret key
358      * algorithm is not available
359      * @exception InvalidKeyException if the shared secret key material cannot
360      * be used to generate a secret key of the requested algorithm type (e.g.,
361      * the key material is too short)
362      */
363     protected SecretKey engineGenerateSecret(String algorithm)
364         throws IllegalStateException, NoSuchAlgorithmException,
365             InvalidKeyException
366     {
367         if (algorithm == null) {
368             throw new NoSuchAlgorithmException("null algorithm");
369         }
370         byte[] secret = engineGenerateSecret();
371         if (algorithm.equalsIgnoreCase("DES")) {
372             // DES
373             return new DESKey(secret);
374         } else if (algorithm.equalsIgnoreCase("DESede")
375                    || algorithm.equalsIgnoreCase("TripleDES")) {
376             // Triple DES
377             return new DESedeKey(secret);
378         } else if (algorithm.equalsIgnoreCase("Blowfish")) {
379             // Blowfish
380             int keysize = secret.length;
381             if (keysize >= BlowfishConstants.BLOWFISH_MAX_KEYSIZE)
382                 keysize = BlowfishConstants.BLOWFISH_MAX_KEYSIZE;
383             SecretKeySpec skey = new SecretKeySpec(secret, 0, keysize,
384                                                    "Blowfish");
385             return skey;
386         } else if (algorithm.equalsIgnoreCase("AES")) {
387             // AES
388             int keysize = secret.length;
389             SecretKeySpec skey = null;
390             int idx = AESConstants.AES_KEYSIZES.length - 1;
391             while (skey == null && idx >= 0) {
392                 // Generate the strongest key using the shared secret
393                 // assuming the key sizes in AESConstants class are
394                 // in ascending order
395                 if (keysize >= AESConstants.AES_KEYSIZES[idx]) {
396                     keysize = AESConstants.AES_KEYSIZES[idx];
397                     skey = new SecretKeySpec(secret, 0, keysize, "AES");
398                 }
399                 idx--;
400             }
401             if (skey == null) {
402                 throw new InvalidKeyException("Key material is too short");
403             }
404             return skey;
405         } else if (algorithm.equals("TlsPremasterSecret")) {
406             // remove leading zero bytes per RFC 5246 Section 8.1.2
407             return new SecretKeySpec(
408                         KeyUtil.trimZeroes(secret), "TlsPremasterSecret");
409         } else {
410             throw new NoSuchAlgorithmException("Unsupported secret key "
411                                                + "algorithm: "+ algorithm);
412         }
413     }
