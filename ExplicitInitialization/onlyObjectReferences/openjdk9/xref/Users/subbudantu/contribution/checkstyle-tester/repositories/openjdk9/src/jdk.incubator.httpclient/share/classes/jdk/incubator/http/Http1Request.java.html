93      private void collectHeaders0() throws IOException {
94          if (Log.headers()) {
95              logHeaders();
96          }
97          StringBuilder sb = new StringBuilder(256);
98          collectHeaders1(sb, request, systemHeaders);
99          collectHeaders1(sb, request, userHeaders);
100         sb.append("\r\n");
101         String headers = sb.toString();
102         buffers[1] = ByteBuffer.wrap(headers.getBytes(StandardCharsets.US_ASCII));
103     }

105     private void collectHeaders1(StringBuilder sb,
106                                  HttpRequestImpl request,
107                                  HttpHeaders headers)
108         throws IOException
109     {
110         Map<String,List<String>> h = headers.map();
111         Set<Map.Entry<String,List<String>>> entries = h.entrySet();
112 
113         for (Map.Entry<String,List<String>> entry : entries) {
114             String key = entry.getKey();
115             List<String> values = entry.getValue();
116             for (String value : values) {
117                 sb.append(key)
118                   .append(": ")
119                   .append(value)
120                   .append("\r\n");
121             }
122         }
123     }

125     private String getPathAndQuery(URI uri) {
126         String path = uri.getPath();
127         String query = uri.getQuery();
128         if (path == null || path.equals("")) {
129             path = "/";
130         }
131         if (query == null) {
132             query = "";
133         }
134         if (query.equals("")) {
135             return path;
136         } else {
137             return path + "?" + query;
138         }
139     }

141     private String authorityString(InetSocketAddress addr) {
142         return addr.getHostString() + ":" + addr.getPort();
143     }
144 
145     private String hostString() {
146         URI uri = request.uri();
147         int port = uri.getPort();
148         String host = uri.getHost();
149 
150         boolean defaultPort;
151         if (port == -1) {
152             defaultPort = true;
153         } else if (request.secure()) {
154             defaultPort = port == 443;
155         } else {
156             defaultPort = port == 80;
157         }
158 
159         if (defaultPort) {
160             return host;
161         } else {
162             return host + ":" + Integer.toString(port);
163         }
164     }
165 
166     private String requestURI() {
167         URI uri = request.uri();
168         String method = request.method();
169 
170         if ((request.proxy(client) == null && !method.equals("CONNECT"))
171                 || request.isWebSocket()) {
172             return getPathAndQuery(uri);
173         }
174         if (request.secure()) {
175             if (request.method().equals("CONNECT")) {
176                 // use authority for connect itself
177                 return authorityString(request.authority());
178             } else {
179                 // requests over tunnel do not require full URL
180                 return getPathAndQuery(uri);
181             }
182         }
183         return uri == null? authorityString(request.authority()) : uri.toString();
184     }

186     void sendHeadersOnly() throws IOException {
187         collectHeaders();
188         chan.write(buffers, 0, 2);
189     }

191     void sendRequest() throws IOException {
192         collectHeaders();
193         chan.configureMode(Mode.BLOCKING);
194         if (contentLength == 0) {
195             chan.write(buffers, 0, 2);
196         } else if (contentLength > 0) {
197             writeFixedContent(true);
198         } else {
199             writeStreamedContent(true);
200         }
201         setFinished();
202     }

214     private void collectHeaders() throws IOException {
215         if (Log.requests() && request != null) {
216             Log.logRequest(request.toString());
217         }
218         String uriString = requestURI();
219         StringBuilder sb = new StringBuilder(64);
220         sb.append(request.method())
221           .append(' ')
222           .append(uriString)
223           .append(" HTTP/1.1\r\n");
224         String cmd = sb.toString();
225 
226         buffers[0] = ByteBuffer.wrap(cmd.getBytes(StandardCharsets.US_ASCII));
227         URI uri = request.uri();
228         if (uri != null) {
229             systemHeaders.setHeader("Host", hostString());
230         }
231         if (request == null) {
232             // this is not a user request. No content
233             contentLength = 0;
234         } else {
235             contentLength = requestProc.contentLength();
236         }
237 
238         if (contentLength == 0) {
239             systemHeaders.setHeader("Content-Length", "0");
240             collectHeaders0();
241         } else if (contentLength > 0) {
242             /* [0] request line [1] headers [2] body  */
243             systemHeaders.setHeader("Content-Length",
244                                     Integer.toString((int) contentLength));
245             streaming = false;
246             collectHeaders0();
247             buffers[2] = getBuffer();
248         } else {
249             /* Chunked:
250              *
251              * [0] request line [1] headers [2] chunk header [3] chunk data [4]
252              * final chunk header and trailing CRLF of previous chunks
253              *
254              * 2,3,4 used repeatedly */
255             streaming = true;
256             systemHeaders.setHeader("Transfer-encoding", "chunked");
257             collectHeaders0();
258             buffers[3] = getBuffer();
259         }
260     }

262     private ByteBuffer getBuffer() {
263         return ByteBuffer.allocate(Utils.BUFSIZE);
264     }

268     void continueRequest() throws IOException {
269         if (streaming) {
270             writeStreamedContent(false);
271         } else {
272             writeFixedContent(false);
273         }
274         setFinished();

277     class StreamSubscriber extends ProcessorBase
278         implements Flow.Subscriber<ByteBuffer>
279     {
280         volatile Flow.Subscription subscription;
281         volatile boolean includeHeaders;
282 
283         StreamSubscriber(boolean includeHeaders) {
284             this.includeHeaders = includeHeaders;
285         }
286 
287         @Override
288         public void onSubscribe(Flow.Subscription subscription) {
289             if (this.subscription != null) {
290                 throw new IllegalStateException("already subscribed");
291             }
292             this.subscription = subscription;
293             subscription.request(1);
294         }
295 
296         @Override
297         public void onNext(ByteBuffer item) {
298             int startbuf, nbufs;
299 
300             if (cf.isDone()) {
301                 throw new IllegalStateException("subscription already completed");
302             }
303 
304             if (includeHeaders) {
305                 startbuf = 0;
306                 nbufs = 5;
307             } else {
308                 startbuf = 2;
309                 nbufs = 3;
310             }
311             int chunklen = item.remaining();
312             buffers[3] = item;
313             buffers[2] = getHeader(chunklen);
314             buffers[4] = CRLF_BUFFER();
315             try {
316                 chan.write(buffers, startbuf, nbufs);
317             } catch (IOException e) {
318                 subscription.cancel();
319                 cf.completeExceptionally(e);
320             }
321             includeHeaders = false;
322             subscription.request(1);
323         }
324 
325         @Override
326         public void onError(Throwable throwable) {
327             if (cf.isDone()) {
328                 return;
329             }
330             subscription.cancel();
331             cf.completeExceptionally(throwable);
332         }
333 
334         @Override
335         public void onComplete() {
336             if (cf.isDone()) {
337                 throw new IllegalStateException("subscription already completed");
338             }
339             buffers[3] = EMPTY_CHUNK_HEADER();
340             buffers[4] = CRLF_BUFFER();
341             try {
342                 chan.write(buffers, 3, 2);
343             } catch (IOException ex) {
344                 cf.completeExceptionally(ex);
345                 return;
346             }
347             cf.complete(null);
348         }
349     }
350 
351     private void waitForCompletion() throws IOException {
352         try {
353             cf.join();
354         } catch (CompletionException e) {
355             throw Utils.getIOException(e);
356         }
357     }

359     /* Entire request is sent, or just body only  */
360     private void writeStreamedContent(boolean includeHeaders)
361         throws IOException
362     {
363         StreamSubscriber subscriber = new StreamSubscriber(includeHeaders);
364         subscriber.setClient(client);
365         requestProc.subscribe(subscriber);
366         waitForCompletion();
367     }

369     class FixedContentSubscriber extends ProcessorBase
370         implements Flow.Subscriber<ByteBuffer>
371     {
372         volatile Flow.Subscription subscription;
373         volatile boolean includeHeaders;
374         volatile long contentWritten = 0;
375 
376         FixedContentSubscriber(boolean includeHeaders) {
377             this.includeHeaders = includeHeaders;
378         }
379 
380         @Override
381         public void onSubscribe(Flow.Subscription subscription) {
382             if (this.subscription != null) {
383                 throw new IllegalStateException("already subscribed");
384             }
385             this.subscription = subscription;
386             subscription.request(1);
387         }
388 
389         @Override
390         public void onNext(ByteBuffer item) {
391             int startbuf, nbufs;
392             long headersLength;
393 
394             if (includeHeaders) {
395                 startbuf = 0;
396                 nbufs = 3;
397                 headersLength = buffers[0].remaining() + buffers[1].remaining();
398             } else {
399                 startbuf = 2;
400                 nbufs = 1;
401                 headersLength = 0;
402             }
403             buffers[2] = item;
404             try {
405                 long writing = buffers[2].remaining() + headersLength;
406                 contentWritten += buffers[2].remaining();
407                 chan.checkWrite(writing, buffers, startbuf, nbufs);
408 
409                 if (contentWritten > contentLength) {
410                     String msg = "Too many bytes in request body. Expected: " +
411                         Long.toString(contentLength) + " Sent: " +
412                         Long.toString(contentWritten);
413                     throw new IOException(msg);
414                 }
415                 subscription.request(1);
416             } catch (IOException e) {
417                 subscription.cancel();
418                 cf.completeExceptionally(e);
419             }
420         }
421 
422         @Override
423         public void onError(Throwable throwable) {
424             if (cf.isDone()) {
425                 return;
426             }
427             subscription.cancel();
428             cf.completeExceptionally(throwable);
429         }
430 
431         @Override
432         public void onComplete() {
433             if (cf.isDone()) {
434                 throw new IllegalStateException("subscription already completed");
435             }
436 
437             if (contentLength > contentWritten) {
438                 subscription.cancel();
439                 Exception e = new IOException("Too few bytes returned by the processor");
440                 cf.completeExceptionally(e);
441             } else {
442                 cf.complete(null);
443             }
444         }
445     }

447     /* Entire request is sent, or just body only */
448     private void writeFixedContent(boolean includeHeaders)
449             throws IOException {
450         if (contentLength == 0) {
451             return;
452         }
453         FixedContentSubscriber subscriber = new FixedContentSubscriber(includeHeaders);
454         subscriber.setClient(client);
455         requestProc.subscribe(subscriber);
456         waitForCompletion();
457     }

459     private static final byte[] CRLF = {'\r', '\n'};
460     private static final byte[] EMPTY_CHUNK_BYTES = {'0', '\r', '\n'};
461 
462     private ByteBuffer CRLF_BUFFER() {
463         return ByteBuffer.wrap(CRLF);
464     }
465 
466     private ByteBuffer EMPTY_CHUNK_HEADER() {
467         return ByteBuffer.wrap(EMPTY_CHUNK_BYTES);
468     }

470     /* Returns a header for a particular chunk size */
471     private static ByteBuffer getHeader(int size){
472         String hexStr =  Integer.toHexString(size);
473         byte[] hexBytes = hexStr.getBytes(US_ASCII);
474         byte[] header = new byte[hexStr.length()+2];
475         System.arraycopy(hexBytes, 0, header, 0, hexBytes.length);
476         header[hexBytes.length] = CRLF[0];
477         header[hexBytes.length+1] = CRLF[1];
478         return ByteBuffer.wrap(header);
479     }
