61      static final Random rnd = new Random();
62      private static void render(Graphics g, int w, int h, boolean useNonOpaque) {
63          if (useNonOpaque) {
64              Graphics2D g2d = (Graphics2D)g;
65              GradientPaint p =
66                  new GradientPaint(0.0f, 0.0f,
67                                    new Color(rnd.nextInt(0xffffff)),
68                                    w, h,
69                                    new Color(rnd.nextInt(0xff),
70                                              rnd.nextInt(0xff),
71                                              rnd.nextInt(0xff), 0),
72                                    true);
73              g2d.setPaint(p);
74              g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING,
75                                   RenderingHints.VALUE_ANTIALIAS_ON);
76              g2d.fillOval(0, 0, w, h);
77          } else {
78              g.setColor(new Color(rnd.nextInt(0xffffff)));
79              g.fillRect(0, 0, w, h);
80          }
81      }

167     public static Frame createGui(
168                                   final boolean useSwing,
169                                   final boolean useShape,
170                                   final boolean useTransl,
171                                   final boolean useNonOpaque,
172                                   final float factor)
173     {
174         Frame frame;
175         done = false;
176 
177         if (useNonOpaque) {
178             if (useSwing) {
179                 frame = new NonOpaqueJFrame();
180 //                frame = new NonOpaqueJAppletFrame(gc);
181             } else {
182                 frame = new NonOpaqueFrame();
183             }
184             animateComponent(frame);
185         } else if (useSwing) {
186             frame = new JFrame("Swing Frame");
187             JComponent p = new JButton("Swing!");
188             p.setPreferredSize(new Dimension(200, 100));
189             frame.add("North", p);
190             p = new MyJPanel();
191             animateComponent(p);
192             frame.add("Center", p);
193         } else {
194             frame = new Frame("AWT Frame") {
195                 public void paint(Graphics g) {
196                     g.setColor(Color.red);
197                     g.fillRect(0, 0, 100, 100);
198                 }
199             };
200             frame.setLayout(new BorderLayout());
201             Canvas c = new MyCanvas();
202             frame.add("North", c);
203             animateComponent(c);
204             c = new MyCanvas();
205             frame.add("Center", c);
206             animateComponent(c);
207             c = new MyCanvas();
208             frame.add("South", c);
209             animateComponent(c);
210         }
211         final Frame finalFrame = frame;
212         frame.addWindowListener(new WindowAdapter() {
213             @Override
214             public void windowClosing(WindowEvent e) {
215                 finalFrame.dispose();
216                 done = true;
217             }
218         });
219         frame.addMouseListener(new MouseAdapter() {
220             @Override
221             public void mouseClicked(MouseEvent e) {
222                 finalFrame.dispose();
223                 done = true;
224             }
225         });
226         frame.setPreferredSize(new Dimension(800, 600));
227 
228         if (useShape) {
229             frame.setUndecorated(true);
230         }
231 
232         frame.setLocation(450, 10);
233         frame.pack();
234 
235         GraphicsDevice gd = frame.getGraphicsConfiguration().getDevice();
236         if (useShape) {
237             if (gd.isWindowTranslucencySupported(WindowTranslucency.PERPIXEL_TRANSPARENT)) {
238                 System.out.println("applying PERPIXEL_TRANSPARENT");
239                 frame.setShape(new Ellipse2D.Double(0, 0, frame.getWidth(),
240                                                     frame.getHeight()/3));
241                 frame.setTitle("PERPIXEL_TRANSPARENT");
242             } else {
243                 System.out.println("Passed: PERPIXEL_TRANSPARENT unsupported");
244             }
245         }
246         if (useTransl) {
247             if (gd.isWindowTranslucencySupported(WindowTranslucency.TRANSLUCENT)) {
248                 System.out.println("applying TRANSLUCENT");
249                 frame.setOpacity(factor);
250                 frame.setTitle("TRANSLUCENT");
251             } else {
252                 System.out.println("Passed: TRANSLUCENT unsupported");
253             }
254         }
255         if (useNonOpaque) {
256             if (gd.isWindowTranslucencySupported(WindowTranslucency.PERPIXEL_TRANSLUCENT)) {
257                 System.out.println("applying PERPIXEL_TRANSLUCENT");
258                 frame.setBackground(new Color(0, 0, 0, 0));
259                 frame.setTitle("PERPIXEL_TRANSLUCENT");
260             } else {
261                 System.out.println("Passed: PERPIXEL_TRANSLUCENT unsupported");
262             }
263         }
264         frame.setVisible(true);
265         return frame;
266     }

268     public static void stopThreads() {
269         done = true;
270     }

272     private static void animateComponent(final Component comp) {
273         Thread t = new Thread(new Runnable() {
274             public void run() {
275                 do {
276                     try {
277                         Thread.sleep(50);
278                     } catch (InterruptedException ex) {}
279                     comp.repaint();
280                 } while (!done);
281             }
282         });
283         t.start();
284     }

295         });
