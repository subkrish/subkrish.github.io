24  import java.awt.AWTEvent;
25  import java.awt.Frame;
26  import java.awt.Robot;
27  import java.awt.Toolkit;
28  import java.awt.event.AWTEventListener;
29  import java.awt.event.KeyEvent;
30  import java.util.Locale;
31  
32  /*
33   * @test
34   * @key headful
35   * @bug 8022401 8160623
36   * @summary Wrong key char
37   * @author Alexandr Scherbatiy
38   * @run main KeyCharTest
39   */
40  public class KeyCharTest {
41  
42      private static volatile int eventsCount = 0;
43  
44      static {
45          Locale.setDefault(Locale.ENGLISH);
46  
47          Toolkit.getDefaultToolkit().addAWTEventListener(new AWTEventListener() {
48  
49              @Override
50              public void eventDispatched(AWTEvent event) {
51                  eventsCount++;
52                  char delete = ((KeyEvent) event).getKeyChar();
53                  if (delete != '\u007f') {
54                      throw new RuntimeException("Key char is not delete: '" + delete + "'");
55                  }
56              }
57          }, AWTEvent.KEY_EVENT_MASK);
58      }
59  
60      public static void main(String[] args) throws Exception {
61  
62  
63          Frame frame = new Frame();
64          frame.setSize(300, 300);
65          frame.setVisible(true);
66          Robot robot = new Robot();
67          robot.setAutoDelay(50);
68          robot.waitForIdle();
69  
70  
71          robot.keyPress(KeyEvent.VK_DELETE);
72          robot.keyRelease(KeyEvent.VK_DELETE);
73          robot.waitForIdle();
74  
75          frame.dispose();
76  
77          if (eventsCount != 3) {
78              throw new RuntimeException("Wrong number of key events: " + eventsCount);
79          }
80      }
81  }
