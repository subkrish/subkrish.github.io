37  import java.applet.Applet;
38  import java.awt.AWTEvent;
39  import java.awt.AWTException;
40  import java.awt.BorderLayout;
41  import java.awt.Point;
42  import java.awt.Robot;
43  import java.awt.Toolkit;
44  
45  import java.awt.event.ActionEvent;
46  import java.awt.event.ActionListener;
47  import java.awt.event.AWTEventListener;
48  import java.awt.event.InputEvent;
49  import java.awt.event.KeyEvent;
50  import java.awt.event.MouseEvent;
51  
52  import javax.swing.JFrame;
53  import javax.swing.JMenu;
54  import javax.swing.JMenuBar;
55  import javax.swing.JMenuItem;
56  import javax.swing.WindowConstants;

58  import javax.swing.event.MenuDragMouseEvent;
59  import javax.swing.event.MenuDragMouseListener;
60  
61  public class MenuDragEvents extends Applet
62  {
63      //Declare things used in the test, like buttons and labels here
64      boolean mouseDragged = false;
65      boolean mouseEntered = false;
66      boolean mouseReleased = false;
67      boolean actionReceived = false;
68  
69      public void init()
70      {
71          // Set up the environment -- set the layout manager, add
72          // buttons, etc.
73  
74          setLayout (new BorderLayout ());
75  
76      }//End  init()
77  
78      public void start ()
79      {
80          //Get things going.  Request focus, set size, et cetera
81          setSize (200,200);
82          setVisible(true);
83          validate();
84  
85          Toolkit.getDefaultToolkit().addAWTEventListener(new AWTEventListener() {
86                  public void eventDispatched(AWTEvent event) {
87                      int id = event.getID();
88                      if (id == MouseEvent.MOUSE_ENTERED || id == MouseEvent.MOUSE_EXITED) {
89                          System.err.println(event);
90                      }
91                  }
92              }, AWTEvent.MOUSE_EVENT_MASK | AWTEvent.MOUSE_MOTION_EVENT_MASK);
93          JMenuBar mb = new JMenuBar();
94  
95          JMenu m = new JMenu("A Menu");
96          mb.add(m);
97  
98          JMenuItem i = new JMenuItem("A menu item",KeyEvent.VK_A);
99          m.add(i);
100 
101         m = new JMenu("Another Menu");
102         mb.add(m);
103         i = new JMenuItem("Yet another menu item",KeyEvent.VK_Y);
104         m.add(i);
105         i.addMenuDragMouseListener(new MenuDragMouseListener() {
106                 public void menuDragMouseDragged(MenuDragMouseEvent e) {
107                     System.err.println(e);
108                     mouseDragged = true;
109                 }
110                 public void menuDragMouseEntered(MenuDragMouseEvent e) {
111                     System.err.println(e);
112                     mouseEntered = true;
113                 }
114                 public void menuDragMouseReleased(MenuDragMouseEvent e) {
115                     System.err.println(e);
116                     mouseReleased = true;
117                 }
118                 // perhaps we need to test mouse exited too
119                 // but this doesn't work even with tiger
120                 public void menuDragMouseExited(MenuDragMouseEvent e) {
121                     System.err.println(e);
122                 }
123             });
124 
125         i.addActionListener(new ActionListener() {
126                 public void actionPerformed(ActionEvent ae) {
127                     System.err.println(ae);
128                     actionReceived = true;
129                 }
130             });
131 
132         JFrame frame = new JFrame("Menu");
133         frame.setJMenuBar(mb);
134         frame.setSize(200, 200);
135         frame.setDefaultCloseOperation(WindowConstants.DISPOSE_ON_CLOSE);
136         frame.setVisible(true);
137 
138         Robot r = null;
139         try {
140             r = new Robot();
141             r.setAutoDelay(50);
142         }
143         catch (AWTException ae) {
144             throw new RuntimeException(ae);
145         }
146 
147         r.waitForIdle();
148 
149         Point loc = m.getLocationOnScreen();
150         loc.x += m.getWidth() / 2;
151         loc.y += m.getHeight() / 2;
152         r.mouseMove(loc.x, loc.y);
153         r.mousePress(InputEvent.BUTTON1_MASK);
154 
155         r.waitForIdle();
156         r.delay(1000);
157 
158         Point loc2 = i.getLocationOnScreen();
159         loc2.x += i.getWidth() / 2;
160         loc2.y += i.getHeight() / 2;
161 
162         // move from menu on menubar to menu item
163         dragMouse(r, loc, loc2);
164         r.mouseRelease(InputEvent.BUTTON1_MASK);
165         r.waitForIdle();
166         r.delay(1000);
167 
168         if (!mouseEntered || !mouseDragged || !mouseReleased || !actionReceived) {
169             throw new RuntimeException("we expected to receive both mouseEntered and MouseDragged ("
170                                        + mouseEntered + ", " + mouseDragged + ", " + mouseReleased
171                                        + ", " + actionReceived + ")");
172         }
173 
174         System.out.println("Test passed");
175     }// start()
176 
177     void dragMouse(Robot r, Point from, Point to) {
178         final int n_step = 10;
179         int step_x = (to.x - from.x) / n_step;
180         int step_y = (to.y - from.y) / n_step;
181         int x = from.x;
182         int y = from.y;
183         for (int idx = 0; idx < n_step; idx++) {
184             x += step_x;
185             y += step_y;
186             r.mouseMove(x, y);
187             r.delay(10);
188         }
189         if (x != to.x || y != to.y) {
190             r.mouseMove(to.x, to.y);
191             r.delay(10);
192         }
193     }
194 
195 }// class MenuDragEvents
