24  /*
25   * @test
26   * @bug 4881082 4916294 5002129 8158524
27   * @summary The test verifies whether the rendering operations on transparent
28   *          and translucent VolatileImage objects generate identical output
29   *          as generated with transparent and translucent BufferedImages.
30   * @key headful
31   * @run main/othervm -Dsun.java2d.uiScale=1 TransparentVImage
32   * @run main/othervm -Dsun.java2d.uiScale=1 -Dsun.java2d.opengl=True -Dsun.java2d.opengl.fbobject=false TransparentVImage
33   * @run main/othervm -Dsun.java2d.uiScale=1 -Dsun.java2d.opengl=True -Dsun.java2d.opengl.fbobject=true TransparentVImage
34   */
35  import java.awt.GraphicsConfiguration;
36  import java.awt.Graphics;
37  import java.awt.Image;
38  import java.awt.Transparency;
39  import java.awt.image.BufferedImage;
40  import java.awt.image.VolatileImage;
41  import java.awt.Color;
42  import javax.swing.JFrame;
43  import javax.swing.JComponent;
44  import javax.swing.ImageIcon;
45  import javax.swing.SwingUtilities;
46  
47  /*
48   * This test draws the same contents to 4 images: 2 BufferedImages that are
49   * transparent and translucent and 2 VolatileImages that are transparent and
50   * translucent. It compares the results pixel-by-pixel and fails if the
51   * results are not the same.
52   */
53  public class TransparentVImage
54      extends JComponent {
55  
56      BufferedImage cImgTransparent, cImgTranslucent;
57      VolatileImage vImgTransparent, vImgTranslucent;
58      Image sprite;
59      static final int IMAGE_SIZE = 250;
60      static final int WINDOW_SIZE = 600;
61      static boolean doneComparing = false;
62      static JFrame testFrame = null;
63  
64      @Override
65      public void paint(Graphics g) {
66          if (cImgTransparent == null) {
67              GraphicsConfiguration gc = getGraphicsConfiguration();
68              // doesn't exist yet: create it (and the other images)
69              cImgTransparent = (BufferedImage)
70                  gc.createCompatibleImage(IMAGE_SIZE, IMAGE_SIZE,
71                                           Transparency.BITMASK);
72              cImgTranslucent = (BufferedImage)
73                  gc.createCompatibleImage(IMAGE_SIZE, IMAGE_SIZE,
74                                           Transparency.TRANSLUCENT);
75              vImgTransparent = gc.createCompatibleVolatileImage(IMAGE_SIZE,
76                  IMAGE_SIZE, Transparency.BITMASK);
77              vImgTranslucent = gc.createCompatibleVolatileImage(IMAGE_SIZE,
78                  IMAGE_SIZE, Transparency.TRANSLUCENT);
79  
80              String fileName = "duke.gif";
81              String separator = System.getProperty("file.separator");
82              String dirPath = System.getProperty("test.src", ".");
83              String filePath = dirPath + separator + fileName;
84              sprite = new ImageIcon(filePath).getImage();
85  
86              // Now they exist; render to them
87              Graphics gImg[] = new Graphics[4];
88              gImg[0] = cImgTransparent.getGraphics();
89              gImg[1] = cImgTranslucent.getGraphics();
90              gImg[2] = vImgTransparent.getGraphics();
91              gImg[3] = vImgTranslucent.getGraphics();
92  
93              for (int i = 0; i < gImg.length; ++i) {
94                  /*
95                   * VolatileImage utilizes the power of accelerated rendering
96                   * using GPU. The GPU drivers for D3d and OpenGL are supplied
97                   * by OEM vendors and are external to graphics subsystem. Thus
98                   * one cannot guarentee that the drivers will render the
99                   * primitives using exactly the same algorithms as those used
100                  * by BufferedImage. This could result in minor differences in
101                  * pixel values between BufferedImage and VolatileImage.
102                  *
103                  * The pipelines for D3d and OpenGL adjust the rendering with
104                  * fudge factors to align output of GPU rendering with that of
105                  * CPU rendering. Some of the draw calls in this paint method
106                  * are commented indicating a need to fine tune the fudge
107                  * factors in the future. Once they are found to work on all
108                  * hardware, the draw calls will be enabled.
109                  */
110                 // rectangular fill
111                 gImg[i].setColor(Color.blue);
112                 gImg[i].fillRect(0, 0, IMAGE_SIZE, IMAGE_SIZE);
113 
114                 /*
115                  * Image copy. Copy it 3 times to get any image management
116                  * acceleration going
117                  */
118                 int spriteW = sprite.getWidth(null);
119                 gImg[i].drawImage(sprite, 0, 0, null);
120                 gImg[i].drawImage(sprite, spriteW, 0, null);
121                 gImg[i].drawImage(sprite, 2 * spriteW, 0, null);
122 
123                 // horizontal/vertical/diagonal lines
124                 gImg[i].setColor(Color.red);
125                 gImg[i].drawLine(0, 0,
126                                  IMAGE_SIZE - 1, IMAGE_SIZE - 1);
127                 gImg[i].drawLine(IMAGE_SIZE / 2, 0,
128                                  IMAGE_SIZE / 2, IMAGE_SIZE - 1);
129                 //gImg[i].drawLine(IMAGE_SIZE, 0,
130                 //                 0, IMAGE_SIZE - 1);
131                 gImg[i].drawLine(0, IMAGE_SIZE / 2,
132                                  IMAGE_SIZE - 1, IMAGE_SIZE / 2);
133 
134                 // filled circle
135                 //gImg[i].setColor(Color.yellow);
136                 //gImg[i].fillOval(IMAGE_SIZE / 2 - 20, IMAGE_SIZE / 2 - 20,
137                 //                 40, 40);
138             }
139 
140             /*
141              * Now everything is drawn: let's compare pixels
142              * First, grab the pixel arrays
143              */
144             int cRgbTransparent[] = new int[IMAGE_SIZE * IMAGE_SIZE];
145             int cRgbTranslucent[] = new int[IMAGE_SIZE * IMAGE_SIZE];
146             int vRgbTransparent[] = new int[IMAGE_SIZE * IMAGE_SIZE];
147             int vRgbTranslucent[] = new int[IMAGE_SIZE * IMAGE_SIZE];
148             cImgTransparent.getRGB(0, 0, IMAGE_SIZE, IMAGE_SIZE,
149                                    cRgbTransparent, 0, IMAGE_SIZE);
150             cImgTranslucent.getRGB(0, 0, IMAGE_SIZE, IMAGE_SIZE,
151                                    cRgbTranslucent, 0, IMAGE_SIZE);
152             BufferedImage bImgTransparent = vImgTransparent.getSnapshot();
153             bImgTransparent.getRGB(0, 0, IMAGE_SIZE, IMAGE_SIZE,
154                                    vRgbTransparent, 0, IMAGE_SIZE);
155             BufferedImage bImgTranslucent = vImgTranslucent.getSnapshot();
156             bImgTranslucent.getRGB(0, 0, IMAGE_SIZE, IMAGE_SIZE,
157                                    vRgbTranslucent, 0, IMAGE_SIZE);
158 
159             boolean failed = false;
160             for (int pixel = 0; pixel < cRgbTransparent.length; ++pixel) {
161                 if (cRgbTransparent[pixel] != vRgbTransparent[pixel]) {
162                     failed = true;
163                     System.out.println("Error in transparent image: " +
164                         "BI[" + pixel + "] = " +
165                         Integer.toHexString(cRgbTransparent[pixel]) +
166                         "VI[" + pixel + "] = " +
167                         Integer.toHexString(vRgbTransparent[pixel]));
168                     break;
169                 }
170                 if (cRgbTranslucent[pixel] != vRgbTranslucent[pixel]) {
171                     failed = true;
172                     System.out.println("Error in translucent image: " +
173                         "BI[" + pixel + "] = " +
174                         Integer.toHexString(cRgbTranslucent[pixel]) +
175                         "VI[" + pixel + "] = " +
176                         Integer.toHexString(vRgbTranslucent[pixel]));
177                     break;
178                 }
179             }
180             if (failed) {
181                 throw new RuntimeException("Failed: Pixel mis-match observed");
182             }
183             else {
184                 System.out.println("Passed");
185             }
186             doneComparing = true;
187         }
188 
189         g.drawImage(cImgTransparent, 0, 0, null);
190         g.drawImage(cImgTranslucent, getWidth() - IMAGE_SIZE, 0, null);
191         g.drawImage(vImgTransparent, 0, getHeight() - IMAGE_SIZE, null);
192         g.drawImage(vImgTranslucent, getWidth() - IMAGE_SIZE,
193                     getHeight() - IMAGE_SIZE, null);
194     }

196     private static void constructTestUI() {
197         testFrame = new JFrame();
198         testFrame.setSize(600, 600);
199         testFrame.setResizable(false);
200         testFrame.getContentPane().add(new TransparentVImage());
201 
202         testFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
203         testFrame.setLocationRelativeTo(null);
204         testFrame.setVisible(true);
205     }
206 
207     private static void destroyTestUI() {
208         testFrame.dispose();
209     }
210 
211     public static void main(String args[]) throws Exception {
212         SwingUtilities.invokeAndWait(new Runnable() {
213             @Override
214             public void run() {
215                 try {
216                     // Construct the test interface
217                     constructTestUI();
218                 } catch (Exception ex) {
219                     // Throw an exception indicating error while creating UI
220                     throw new RuntimeException("Test Failed: Error while" +
221                         " creating the test interface.");
222                 }
223             }
224         });
225 
226         /*
227          * Wait until the image comparison between VolatileImage and
228          * BufferedImage is complete.
229          */
230         while (!doneComparing) {
231             try {
232                 Thread.sleep(100);
233             }
234             catch (Exception e) {}
235         }
236 
237         /*
238          * Now sleep just a little longer to let the user see the resulting
239          * images in the frame
240          */
241         try {
242             Thread.sleep(5000);
243         }
244         catch (Exception e) {}
245 
246         SwingUtilities.invokeAndWait(new Runnable() {
247             @Override
248             public void run() {
249                 try {
250                     // Destroy the test interface
251                     destroyTestUI();
252                 } catch (Exception ex) {
253                     // Throw an exception indicating error while deleting UI
254                     throw new RuntimeException("Test Failed: Error while" +
255                         " deleting the test interface.");
256                 }
257             }
258         });
259     }
