348     public static class BaseLoggerFinder extends LoggerFinder implements TestLoggerFinder {
349         static final RuntimePermission LOGGERFINDER_PERMISSION =
350                 new RuntimePermission("loggerFinder");
351         @Override
352         public Logger getLogger(String name, Module caller) {
353             SecurityManager sm = System.getSecurityManager();
354             if (sm != null) {
355                 sm.checkPermission(LOGGERFINDER_PERMISSION);
356             }
357             PrivilegedAction<ClassLoader> pa = () -> caller.getClassLoader();
358             ClassLoader callerLoader = AccessController.doPrivileged(pa);
359             if (callerLoader == null) {
360                 return system.computeIfAbsent(name, (n) -> new LoggerImpl(n));
361             } else {
362                 return user.computeIfAbsent(name, (n) -> new LoggerImpl(n));
363             }
364         }
365     }

401     public static void main(String[] args) {
402         if (args.length == 0)
403             args = new String[] {
404                 "NOSECURITY",
405                 "NOPERMISSIONS",
406                 "WITHPERMISSIONS"
407             };
408 
409 
410         Stream.of(args).map(TestCases::valueOf).forEach((testCase) -> {
411             TestLoggerFinder provider;
412             switch (testCase) {
413                 case NOSECURITY:
414                     System.out.println("\n*** Without Security Manager\n");
415                     provider = TestLoggerFinder.class.cast(LoggerFinder.getLoggerFinder());
416                     test(provider, true);
417                     System.out.println("Tetscase count: " + sequencer.get());
418                     break;
419                 case NOPERMISSIONS:
420                     System.out.println("\n*** With Security Manager, without permissions\n");
421                     setSecurityManager();
422                     try {
423                         provider = TestLoggerFinder.class.cast(LoggerFinder.getLoggerFinder());
424                         throw new RuntimeException("Expected exception not raised");
425                     } catch (AccessControlException x) {
426                         if (!LOGGERFINDER_PERMISSION.equals(x.getPermission())) {
427                             throw new RuntimeException("Unexpected permission check", x);
428                         }
429                         final boolean control = allowControl.get().get();
430                         try {
431                             allowControl.get().set(true);
432                             provider = TestLoggerFinder.class.cast(LoggerFinder.getLoggerFinder());
433                         } finally {
434                             allowControl.get().set(control);
435                         }
436                     }
437                     test(provider, false);
438                     System.out.println("Tetscase count: " + sequencer.get());
439                     break;
440                 case WITHPERMISSIONS:
441                     System.out.println("\n*** With Security Manager, with control permission\n");
442                     setSecurityManager();
443                     final boolean control = allowControl.get().get();
444                     try {
445                         allowControl.get().set(true);
446                         provider = TestLoggerFinder.class.cast(LoggerFinder.getLoggerFinder());
447                         test(provider, true);
448                     } finally {
449                         allowControl.get().set(control);
450                     }
451                     break;
452                 default:
453                     throw new RuntimeException("Unknown test case: " + testCase);
454             }
455         });
456         System.out.println("\nPASSED: Tested " + sequencer.get() + " cases.");
457     }

633             TestLoggerFinder.LogEvent expected, boolean expectNotNull) {
634         TestLoggerFinder.LogEvent actual =  provider.eventQueue.poll();
635         if (actual == null && !expectNotNull) return;

642                     + "\n\texpected=" + expected
643                     + "\n\t  actual=" + actual);
644         } else {
645             verbose("Got expected results for "
646                     + desc + "\n\t" + expected);

652             if (BEST_EFFORT_FOR_LOGP && bundle == null
653                     && (className != null || methodName != null)) {
654                 final String cName = className == null ? "" :  className;
655                 final String mName = methodName == null ? "" : methodName;
656                 return () -> String.format("[%s %s] %s", cName, mName, msg.get());
657             } else {
658                 return msg;
659             }

974     final static class PermissionsBuilder {
975         final Permissions perms;
976         public PermissionsBuilder() {
977             this(new Permissions());
978         }
979         public PermissionsBuilder(Permissions perms) {
980             this.perms = perms;
981         }
982         public PermissionsBuilder add(Permission p) {
983             perms.add(p);
984             return this;
985         }
986         public PermissionsBuilder addAll(PermissionCollection col) {
987             if (col != null) {
988                 for (Enumeration<Permission> e = col.elements(); e.hasMoreElements(); ) {
989                     perms.add(e.nextElement());
990                 }
991             }
992             return this;
993         }
994         public Permissions toPermissions() {
995             final PermissionsBuilder builder = new PermissionsBuilder();
996             builder.addAll(perms);
997             return builder.perms;
998         }
999     }

1001     public static class SimplePolicy extends Policy {
1002         final static RuntimePermission CONTROL = LOGGERFINDER_PERMISSION;
1003         final static RuntimePermission ACCESS_LOGGER = new RuntimePermission("accessClassInPackage.jdk.internal.logger");
1004         final static RuntimePermission ACCESS_LOGGING = new RuntimePermission("accessClassInPackage.sun.util.logging");
1005 
1006         final Permissions permissions;
1007         final Permissions allPermissions;
1008         final ThreadLocal<AtomicBoolean> allowControl;
1009         final ThreadLocal<AtomicBoolean> allowAccess;
1010         final ThreadLocal<AtomicBoolean> allowAll;
1011         public SimplePolicy(ThreadLocal<AtomicBoolean> allowControl,
1012                 ThreadLocal<AtomicBoolean> allowAccess,
1013                 ThreadLocal<AtomicBoolean> allowAll) {
1014             this.allowControl = allowControl;
1015             this.allowAccess = allowAccess;
1016             this.allowAll = allowAll;
1017             permissions = new Permissions();
1018             allPermissions = new PermissionsBuilder()
1019                     .add(new java.security.AllPermission())
1020                     .toPermissions();
1021         }
1022 
1023         Permissions getPermissions() {
1024             if (allowControl.get().get() || allowAccess.get().get() || allowAll.get().get()) {
1025                 PermissionsBuilder builder =  new PermissionsBuilder()
1026                         .addAll(permissions);
1027                 if (allowControl.get().get()) {
1028                     builder.add(CONTROL);
1029                 }
1030                 if (allowAccess.get().get()) {
1031                     builder.add(ACCESS_LOGGER);
1032                     builder.add(ACCESS_LOGGING);
1033                 }
1034                 if (allowAll.get().get()) {
1035                     builder.addAll(allPermissions);
1036                 }
1037                 return builder.toPermissions();
1038             }
1039             return permissions;
1040         }
1041 
1042         @Override
1043         public boolean implies(ProtectionDomain domain, Permission permission) {
1044             return getPermissions().implies(permission);
1045         }
1046 
1047         @Override
1048         public PermissionCollection getPermissions(CodeSource codesource) {
1049             return new PermissionsBuilder().addAll(getPermissions()).toPermissions();
1050         }
1051 
1052         @Override
1053         public PermissionCollection getPermissions(ProtectionDomain domain) {
1054             return new PermissionsBuilder().addAll(getPermissions()).toPermissions();
1055         }
