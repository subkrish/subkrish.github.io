24  package jdk.test;
25  
26  import java.lang.reflect.Constructor;
27  import java.lang.reflect.InvocationHandler;
28  import java.lang.reflect.InvocationTargetException;
29  import java.lang.reflect.Module;
30  import java.lang.reflect.Proxy;
31  import java.util.Arrays;
32  
33  public class ProxyTest {
34      public static class Data {
35          private static int count = 0;
36          final int testcase;
37          final ClassLoader loader;
38          final Module module;
39          final Class<?>[] interfaces;
40          // Expected the proxy class in the specified module
41          public Data(Module m, ClassLoader loader, Class<?>... interfaces) {
42              this.module = m;
43              this.loader = loader;
44              this.interfaces = interfaces;
45              this.testcase = ++count;
46          }
47          // Expected the proxy class in a dynamic module
48          public Data(ClassLoader loader, Class<?>... interfaces) {
49              this(null, loader, interfaces);
50          }
51  
52          @Override
53          public String toString() {
54              String expected = module != null
55                      ? (module.isNamed() ? module.getName() : "unnamed")
56                      : "dynamic";
57              return String.format("%2d: Expected: %s %s loader: %s", testcase, expected,
58                      Arrays.toString(interfaces), loader);
59          }
60      }
61  
62      public void test(Data d) {
63          System.out.println(d);
64  
65          if (d.module != null) {
66              testProxyClass(d.module, d.loader, d.interfaces);
67          } else {
68              testDynamicModule(d);
69          }
70      }
71  
72      private void testDynamicModule(Data d) {
73          Class<?> proxyClass = Proxy.getProxyClass(d.loader, d.interfaces);
74          assertDynamicModule(proxyClass.getModule(), d.loader, proxyClass);
75  
76          Object proxy = Proxy.newProxyInstance(d.loader, d.interfaces, handler);
77          assertDynamicModule(proxy.getClass().getModule(), d.loader, proxy.getClass());
78      }
79  
80      private static void testProxyClass(Module module, ClassLoader ld, Class<?>... interfaces) {
81          Class<?> proxyClass = Proxy.getProxyClass(ld, interfaces);
82          assertEquals(proxyClass.getModule(), module);
83  
84          Object proxy = Proxy.newProxyInstance(ld, interfaces, handler);
85          assertEquals(proxy.getClass().getModule(), module);
86      }
87  
88      public static void assertDynamicModule(Module m, ClassLoader ld, Class<?> proxyClass) {
89          if (!m.isNamed() || !m.getName().startsWith("jdk.proxy")) {
90              throw new RuntimeException(m.getName() + " not dynamic module");
91          }
92  
93          if (ld != m.getClassLoader() || proxyClass.getClassLoader() != ld) {
94              throw new RuntimeException("unexpected class loader");
95          }
96  
97          try {
98              Constructor<?> cons = proxyClass.getConstructor(InvocationHandler.class);
99              cons.newInstance(handler);
100             throw new RuntimeException("Expected IllegalAccessException: " + proxyClass);
101         } catch (IllegalAccessException e) {
102             // expected
103         } catch (NoSuchMethodException|InstantiationException|InvocationTargetException e) {
104             throw new RuntimeException(e);
105         }
106     }
107 
108     public static void assertEquals(Object o1, Object o2) {
109         if (o1 != o2) {
110             throw new RuntimeException(o1 + " != " + o2);
111         }
112     }
113     private final static InvocationHandler handler =
114         (proxy, m, params) -> { throw new RuntimeException(m.toString()); };
115 }
