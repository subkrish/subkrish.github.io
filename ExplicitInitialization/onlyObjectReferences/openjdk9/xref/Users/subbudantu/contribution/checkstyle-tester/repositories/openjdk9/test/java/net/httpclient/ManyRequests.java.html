64      public static void main(String[] args) throws Exception {
65          Logger logger = Logger.getLogger("com.sun.net.httpserver");
66          logger.setLevel(Level.ALL);
67          logger.info("TEST");
68  
69          SSLContext ctx = new SimpleSSLContext().get();
70  
71          InetSocketAddress addr = new InetSocketAddress(0);
72          HttpsServer server = HttpsServer.create(addr, 0);
73          server.setHttpsConfigurator(new Configurator(ctx));
74  
75          HttpClient client = HttpClient.newBuilder()
76                                        .sslContext(ctx)
77                                        .build();
78          try {
79              test(server, client);
80              System.out.println("OK");
81          } finally {
82              server.stop(0);
83              ((ExecutorService)client.executor()).shutdownNow();
84          }
85      }

90      static void test(HttpsServer server, HttpClient client) throws Exception {
91          int port = server.getAddress().getPort();
92          URI uri = new URI("https://127.0.0.1:" + port + "/foo/x");
93          server.createContext("/foo", new EchoHandler());
94          server.start();
95  
96          RequestLimiter limiter = new RequestLimiter(40);
97          Random rand = new Random();
98          CompletableFuture<?>[] results = new CompletableFuture<?>[REQUESTS];
99          HashMap<HttpRequest,byte[]> bodies = new HashMap<>();
100 
101         for (int i=0; i<REQUESTS; i++) {
102             byte[] buf = new byte[i+1];  // different size bodies
103             rand.nextBytes(buf);
104             HttpRequest r = HttpRequest.newBuilder(uri)
105                                        .POST(fromByteArray(buf))
106                                        .build();
107             bodies.put(r, buf);
108 
109             results[i] =
110                 limiter.whenOkToSend()
111                        .thenCompose((v) -> client.sendAsync(r, asByteArray()))
112                        .thenCompose((resp) -> {
113                            limiter.requestComplete();
114                            if (resp.statusCode() != 200) {
115                                String s = "Expected 200, got: " + resp.statusCode();
116                                return completedWithIOException(s);
117                            } else {
118                                counter++;
119                                System.out.println("Result from " + counter);
120                            }
121                            return CompletableFuture.completedStage(resp.body())
122                                       .thenApply((b) -> new Pair<>(resp, b));
123                        })
124                       .thenAccept((pair) -> {
125                           HttpRequest request = pair.t.request();
126                           byte[] requestBody = bodies.get(request);
127                           check(Arrays.equals(requestBody, pair.u),
128                                 "bodies not equal:[" + bytesToHexString(requestBody)
129                                 + "] [" + bytesToHexString(pair.u) + "]");
130 
131                       });
132         }
133 
134         // wait for them all to complete and throw exception in case of error
135         CompletableFuture.allOf(results).join();
136     }

138     static <T> CompletableFuture<T> completedWithIOException(String message) {
139         return CompletableFuture.failedFuture(new IOException(message));
140     }

142     static String bytesToHexString(byte[] bytes) {
143         if (bytes == null)
144             return "null";
145 
146         StringBuilder sb = new StringBuilder(bytes.length * 2);
147 
148         Formatter formatter = new Formatter(sb);
149         for (byte b : bytes) {
150             formatter.format("%02x", b);
151         }
152 
153         return sb.toString();
154     }

156     static final class Pair<T,U> {

164     /**
165      * A simple limiter for controlling the number of requests to be run in
166      * parallel whenOkToSend() is called which returns a CF<Void> that allows
167      * each individual request to proceed, or block temporarily (blocking occurs
168      * on the waiters list here. As each request actually completes
169      * requestComplete() is called to notify this object, and allow some
170      * requests to continue.
171      */
172     static class RequestLimiter {
173 
174         static final CompletableFuture<Void> COMPLETED_FUTURE =
175                 CompletableFuture.completedFuture(null);
176 
177         final int maxnumber;
178         final LinkedList<CompletableFuture<Void>> waiters;
179         int number;
180         boolean blocked;
181 
182         RequestLimiter(int maximum) {
183             waiters = new LinkedList<>();
184             maxnumber = maximum;
185         }
186 
187         synchronized void requestComplete() {
188             number--;
189             // don't unblock until number of requests has halved.
190             if ((blocked && number <= maxnumber / 2) ||
191                         (!blocked && waiters.size() > 0)) {
192                 int toRelease = Math.min(maxnumber - number, waiters.size());
193                 for (int i=0; i<toRelease; i++) {
194                     CompletableFuture<Void> f = waiters.remove();
195                     number ++;
196                     f.complete(null);
197                 }
198                 blocked = number >= maxnumber;
199             }
200         }
201 
202         synchronized CompletableFuture<Void> whenOkToSend() {
203             if (blocked || number + 1 >= maxnumber) {
204                 blocked = true;
205                 CompletableFuture<Void> r = new CompletableFuture<>();
206                 waiters.add(r);
207                 return r;
208             } else {
209                 number++;
210                 return COMPLETED_FUTURE;
211             }
212         }
213     }
214 
215     static void check(boolean cond, Object... msg) {
216         if (cond)
217             return;
218         StringBuilder sb = new StringBuilder();
219         for (Object o : msg)
220             sb.append(o);
221         throw new RuntimeException(sb.toString());
222     }
