32  import static java.util.concurrent.Executors.privilegedCallable;
33  import static java.util.concurrent.Executors.privilegedCallableUsingCurrentClassLoader;
34  import static java.util.concurrent.Executors.privilegedThreadFactory;
35  
36  import java.security.AccessControlException;
37  import java.security.CodeSource;
38  import java.security.Permission;
39  import java.security.PermissionCollection;
40  import java.security.Permissions;
41  import java.security.ProtectionDomain;
42  import java.util.Random;
43  import java.util.concurrent.Callable;
44  
45  public class PrivilegedCallables {
46      Callable<Integer> real;
47  
48      final Callable<Integer> realCaller = new Callable<>() {
49          public Integer call() throws Exception {
50              return real.call(); }};
51  
52      final Random rnd = new Random();
53  
54      @SuppressWarnings("serial")
55      final Throwable[] throwables = {
56          new Exception() {},
57          new RuntimeException() {},
58          new Error() {}
59      };
60      Throwable randomThrowable() {
61          return throwables[rnd.nextInt(throwables.length)];
62      }
63      void throwThrowable(Throwable t) throws Exception {
64          if (t instanceof Error) throw (Error) t;
65          if (t instanceof RuntimeException) throw (RuntimeException) t;
66          throw (Exception) t;
67      }
68  
69      //----------------------------------------------------------------
70      // A Policy class designed to make permissions fiddling very easy.
71      //----------------------------------------------------------------
72      static class Policy extends java.security.Policy {
73          private Permissions perms;
74  
75          public void setPermissions(Permission...permissions) {
76              perms = new Permissions();
77              for (Permission permission : permissions)
78                  perms.add(permission);
79          }
80  
81          public Policy() { setPermissions(/* Nothing */); }
82  
83          public PermissionCollection getPermissions(CodeSource cs) {
84              return perms;
85          }
86  
87          public PermissionCollection getPermissions(ProtectionDomain pd) {
88              return perms;
89          }
90  
91          public boolean implies(ProtectionDomain pd, Permission p) {
92              return perms.implies(p);
93          }
94  
95          public void refresh() {}
96      }
97  
98      void test(String[] args) {
99          testPrivileged();
100 
101         final Policy policy = new Policy();
102         Policy.setPolicy(policy);
103         policy.setPermissions(new RuntimePermission("getClassLoader"),
104                               new RuntimePermission("setContextClassLoader"),
105                               new RuntimePermission("stopThread"));
106         System.setSecurityManager(new SecurityManager());
107 
108         testPrivileged();
109 
110         policy.setPermissions(/* Nothing */);
111 
112         THROWS(AccessControlException.class,
113                new F() {void f(){ privilegedCallableUsingCurrentClassLoader(realCaller); }},
114                new F() {void f(){ privilegedThreadFactory(); }});
115 
116         policy.setPermissions(new RuntimePermission("setSecurityManager"));
117         System.setSecurityManager(null);
118     }
119 
120     void testPrivileged() {
121         try { test(privilegedCallable(realCaller)); }
122         catch (Throwable t) { unexpected(t); }
123 
124         try { test(privilegedCallableUsingCurrentClassLoader(realCaller)); }
125         catch (Throwable t) { unexpected(t); }
126 
127         try { privilegedThreadFactory(); }
128         catch (Throwable t) { unexpected(t); }
129     }
130 
131     void test(final Callable<Integer> c) throws Throwable {
132         for (int i = 0; i < 20; i++)
133             if (rnd.nextBoolean()) {
134                 final Throwable t = randomThrowable();
135                 real = new Callable<>() {
136                     public Integer call() throws Exception {
137                         throwThrowable(t);
138                         return null; }};
139                 try {
140                     c.call();
141                     fail("Expected exception not thrown");
142                 } catch (Throwable tt) { check(t == tt); }
143             } else {
144                 final int n = rnd.nextInt();
145                 real = new Callable<>() {
146                     public Integer call() { return n; }};
147                 equal(c.call(), n);
148             }
149     }
150 
151     //--------------------- Infrastructure ---------------------------
152     volatile int passed = 0, failed = 0;
153     void pass() {passed++;}
154     void fail() {failed++; Thread.dumpStack();}
155     void fail(String msg) {System.err.println(msg); fail();}
156     void unexpected(Throwable t) {failed++; t.printStackTrace();}
157     void check(boolean cond) {if (cond) pass(); else fail();}
158     void equal(Object x, Object y) {
159         if (x == null ? y == null : x.equals(y)) pass();
160         else fail(x + " not equal to " + y);}
161     public static void main(String[] args) throws Throwable {
162         new PrivilegedCallables().instanceMain(args);}
163     void instanceMain(String[] args) throws Throwable {
164         try {test(args);} catch (Throwable t) {unexpected(t);}
165         System.out.printf("%nPassed = %d, failed = %d%n%n", passed, failed);
166         if (failed > 0) throw new AssertionError("Some tests failed");}
167     abstract class F {abstract void f() throws Throwable;}
168     void THROWS(Class<? extends Throwable> k, F... fs) {
169         for (F f : fs)
170             try {f.f(); fail("Expected " + k.getName() + " not thrown");}
171             catch (Throwable t) {
172                 if (k.isAssignableFrom(t.getClass())) pass();
173                 else unexpected(t);}}
