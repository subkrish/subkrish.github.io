79      static String outputOf(Reader r) throws IOException {
80          final StringBuilder sb = new StringBuilder();
81          final char[] buf = new char[1024];
82          int n;
83          while ((n = r.read(buf)) > 0)
84              sb.append(buf, 0, n);
85          return sb.toString();
86      }

89          return outputOf(new InputStreamReader(is, "UTF-8"));

92      static final ExecutorService drainers = Executors.newFixedThreadPool(12);
93      static Future<String> futureOutputOf(final InputStream is) {
94          return drainers.submit(
95              new Callable<String>() { public String call() throws IOException {
96                      return outputOf(is); }});}

98      static String outputOf(final Process p) {
99          try {
100             Future<String> outputFuture = futureOutputOf(p.getInputStream());
101             Future<String> errorFuture = futureOutputOf(p.getErrorStream());
102             final String output = outputFuture.get();
103             final String error = errorFuture.get();
104             // Check for successful process completion
105             equal(error, "");
106             equal(p.waitFor(), 0);
107             equal(p.exitValue(), 0);
108             return output;
109         } catch (Throwable t) { unexpected(t); throw new Error(t); }
110     }

117     // To be called exactly twice by the parent process
118     static <T> T rendezvousParent(Process p,
119                                   Callable<T> callable) throws Throwable {
120         p.getInputStream().read();
121         T result = callable.call();
122         sendByte(p.getOutputStream());
123         return result;
124     }

126     /** No guarantees, but effective in practice. */
127     private static void forceFullGc() {
128         CountDownLatch finalizeDone = new CountDownLatch(1);
129         WeakReference<?> ref = new WeakReference<Object>(new Object() {
130             protected void finalize() { finalizeDone.countDown(); }});
131         try {
132             for (int i = 0; i < 10; i++) {
133                 System.gc();
134                 if (finalizeDone.await(1L, SECONDS) && ref.get() == null) {
135                     System.runFinalization(); // try to pick up stragglers
136                     return;
137                 }
138             }
139         } catch (InterruptedException unexpected) {
140             throw new AssertionError("unexpected InterruptedException");
141         }
142         throw new AssertionError("failed to do a \"full\" gc");
143     }

145     // To be called exactly twice by the child process
146     public static void rendezvousChild() {
147         try {
148             forceFullGc();
149             sendByte(System.out);
150             System.in.read();
151         } catch (Throwable t) { throw new Error(t); }
152     }

154     static String match(String s, String regex, int group) {
155         Matcher matcher = Pattern.compile(regex).matcher(s);
156         matcher.find();
157         return matcher.group(group);
158     }

160     /** It's all about sending a message! */
161     static void sendByte(OutputStream s) throws IOException {
162         s.write('!');
163         s.flush();
164     }

166     static int objectsInUse(final Process child,
167                             final String childPid,
168                             final String className) {
169         final String regex =
170             "(?m)^ *[0-9]+: +([0-9]+) +[0-9]+ +\\Q"+className+"\\E(?:$| )";
171         final Callable<Integer> objectsInUse =
172             new Callable<Integer>() { public Integer call() {
173                 Integer i = Integer.parseInt(
174                     match(commandOutputOf(jmap, "-histo:live", childPid),
175                           regex, 1));
176                 if (i > 100)
177                     System.out.print(
178                         commandOutputOf(jmap,
179                                         "-dump:file=dump,format=b",
180                                         childPid));
181                 return i;
182             }};
183         try { return rendezvousParent(child, objectsInUse); }
184         catch (Throwable t) { unexpected(t); return -1; }
185     }

187     static void realMain(String[] args) throws Throwable {
188         // jmap doesn't work on Windows
189         if (System.getProperty("os.name").startsWith("Windows"))
190             return;
191 
192         final String childClassName = Job.class.getName();
193         final String classToCheckForLeaks = Job.classToCheckForLeaks();
194         final String uniqueID =
195             String.valueOf(ThreadLocalRandom.current().nextInt(Integer.MAX_VALUE));
196 
197         final String[] jobCmd = {
198             java, "-Xmx8m", "-XX:+UsePerfData",
199             "-classpath", System.getProperty("test.class.path"),
200             childClassName, uniqueID
201         };
202         final Process p = new ProcessBuilder(jobCmd).start();
203         // Ensure subprocess jvm has started, so that jps can find it
204         p.getInputStream().read();
205         sendByte(p.getOutputStream());
206 
207         final String childPid =
208             match(commandOutputOf(jps, "-m"),
209                   "(?m)^ *([0-9]+) +\\Q"+childClassName+"\\E *"+uniqueID+"$", 1);
210 
211         final int n0 = objectsInUse(p, childPid, classToCheckForLeaks);
212         final int n1 = objectsInUse(p, childPid, classToCheckForLeaks);
213         equal(p.waitFor(), 0);
214         equal(p.exitValue(), 0);
215         failed += p.exitValue();
216 
217         // Check that no objects were leaked.
218         //
219         // TODO: This test is very brittle, depending on current JDK
220         // implementation, and needing occasional adjustment.
221         System.out.printf("%d -> %d%n", n0, n1);
222         // Almost always n0 == n1
223         // Maximum jitter observed in practice is 10 -> 17
224         check(Math.abs(n1 - n0) < 10);
225         check(n1 < 25);
226         drainers.shutdown();
227         if (!drainers.awaitTermination(LONG_DELAY_MS, MILLISECONDS)) {
228             drainers.shutdownNow(); // last resort
229             throw new AssertionError("thread pool did not terminate");
230         }
231     }

233     //----------------------------------------------------------------
234     // The main class of the child process.
235     // Job's job is to:
236     // - provide the name of a class to check for leaks.
237     // - call rendezvousChild exactly twice, while quiescent.
238     // - in between calls to rendezvousChild, run code that may leak.
239     //----------------------------------------------------------------
240     public static class Job {
241         static String classToCheckForLeaks() {
242             return
243                 "java.util.concurrent.locks.AbstractQueuedSynchronizer$Node";
244         }
245 
246         public static void main(String[] args) throws Throwable {
247             // Synchronize with parent process, so that jps can find us
248             sendByte(System.out);
249             System.in.read();
250 
251             final ReentrantLock lock = new ReentrantLock();
252             lock.lock();
253 
254             final ReentrantReadWriteLock rwlock
255                 = new ReentrantReadWriteLock();
256             final ReentrantReadWriteLock.ReadLock readLock
257                 = rwlock.readLock();
258             final ReentrantReadWriteLock.WriteLock writeLock
259                 = rwlock.writeLock();
260             rwlock.writeLock().lock();
261 
262             final BlockingQueue<Object> q = new LinkedBlockingQueue<>();
263             final Semaphore fairSem = new Semaphore(0, true);
264             final Semaphore unfairSem = new Semaphore(0, false);
265             //final int threads =
266             //rnd.nextInt(Runtime.getRuntime().availableProcessors() + 1) + 1;
267             final int threads = 3;
268             // On Linux, this test runs very slowly for some reason,
269             // so use a smaller number of iterations.
270             // Solaris can handle 1 << 18.
271             // On the other hand, jmap is much slower on Solaris...
272             final int iterations = 1 << 8;
273             final CyclicBarrier cb = new CyclicBarrier(threads+1);
274 
275             for (int i = 0; i < threads; i++)
276                 new Thread() { public void run() {
277                     try {
278                         final ThreadLocalRandom rnd = ThreadLocalRandom.current();
279                         for (int j = 0; j < iterations; j++) {
280                             if (j == iterations/10 || j == iterations - 1) {
281                                 cb.await(); // Quiesce
282                                 cb.await(); // Resume
283                             }
284                             //int t = rnd.nextInt(2000);
285                             int t = rnd.nextInt(900);
286                             check(! lock.tryLock(t, NANOSECONDS));
287                             check(! readLock.tryLock(t, NANOSECONDS));
288                             check(! writeLock.tryLock(t, NANOSECONDS));
289                             equal(null, q.poll(t, NANOSECONDS));
290                             check(! fairSem.tryAcquire(t, NANOSECONDS));
291                             check(! unfairSem.tryAcquire(t, NANOSECONDS));
292                         }
293                     } catch (Throwable t) { unexpected(t); }
294                 }}.start();
295 
296             cb.await();         // Quiesce
297             rendezvousChild();  // Measure
298             cb.await();         // Resume
299 
300             cb.await();         // Quiesce
301             rendezvousChild();  // Measure
302             cb.await();         // Resume
303 
304             System.exit(failed);
305         }
306 
307         // If something goes wrong, we might never see it, since IO
308         // streams are connected to the parent.  So we need a special
309         // purpose print method to debug Jobs.
310         static void debugPrintf(String format, Object... args) {
311             try {
312                 new PrintStream(new FileOutputStream("/dev/tty"))
313                     .printf(format, args);
314             } catch (Throwable t) { throw new Error(t); }
315         }
316     }

318     //--------------------- Infrastructure ---------------------------
319     static volatile int passed = 0, failed = 0;
320     static void pass() {passed++;}
321     static void fail() {failed++; Thread.dumpStack();}
322     static void fail(String msg) {System.out.println(msg); fail();}
323     static void unexpected(Throwable t) {failed++; t.printStackTrace();}
324     static void check(boolean cond) {if (cond) pass(); else fail();}
325     static void check(boolean cond, String m) {if (cond) pass(); else fail(m);}
326     static void equal(Object x, Object y) {
327         if (x == null ? y == null : x.equals(y)) pass();
328         else fail(x + " not equal to " + y);}
329     public static void main(String[] args) throws Throwable {
330         try {realMain(args);} catch (Throwable t) {unexpected(t);}
331         System.out.printf("%nPassed = %d, failed = %d%n%n", passed, failed);
332         if (failed > 0) throw new AssertionError("Some tests failed");}
