130 /**
131  * Base class for JSR166 Junit TCK tests.  Defines some constants,
132  * utility methods and classes, as well as a simple framework for
133  * helping to make sure that assertions failing in generated threads
134  * cause the associated test that generated them to itself fail (which
135  * JUnit does not otherwise arrange).  The rules for creating such
136  * tests are:
137  *
138  * <ol>
139  *
140  * <li>All assertions in code running in generated threads must use
141  * the forms {@link #threadFail}, {@link #threadAssertTrue}, {@link
142  * #threadAssertEquals}, or {@link #threadAssertNull}, (not
143  * {@code fail}, {@code assertTrue}, etc.) It is OK (but not
144  * particularly recommended) for other code to use these forms too.
145  * Only the most typically used JUnit assertion methods are defined
146  * this way, but enough to live with.
147  *
148  * <li>If you override {@link #setUp} or {@link #tearDown}, make sure
149  * to invoke {@code super.setUp} and {@code super.tearDown} within
150  * them. These methods are used to clear and check for thread
151  * assertion failures.
152  *
153  * <li>All delays and timeouts must use one of the constants {@code
154  * SHORT_DELAY_MS}, {@code SMALL_DELAY_MS}, {@code MEDIUM_DELAY_MS},
155  * {@code LONG_DELAY_MS}. The idea here is that a SHORT is always
156  * discriminable from zero time, and always allows enough time for the
157  * small amounts of computation (creating a thread, calling a few
158  * methods, etc) needed to reach a timeout point. Similarly, a SMALL
159  * is always discriminable as larger than SHORT and smaller than
160  * MEDIUM.  And so on. These constants are set to conservative values,
161  * but even so, if there is ever any doubt, they can all be increased
162  * in one spot to rerun tests on slower platforms.
163  *
164  * <li>All threads generated must be joined inside each test case
165  * method (or {@code fail} to do so) before returning from the
166  * method. The {@code joinPool} method can be used to do this when
167  * using Executors.
168  *
169  * </ol>
170  *
171  * <p><b>Other notes</b>
172  * <ul>
173  *
174  * <li>Usually, there is one testcase method per JSR166 method
175  * covering "normal" operation, and then as many exception-testing
176  * methods as there are exceptions the method can throw. Sometimes
177  * there are multiple tests per JSR166 method when the different
178  * "normal" behaviors differ significantly. And sometimes testcases
179  * cover multiple methods when they cannot be tested in isolation.
180  *
181  * <li>The documentation style for testcases is to provide as javadoc
182  * a simple sentence or two describing the property that the testcase
183  * method purports to test. The javadocs do not say anything about how
184  * the property is tested. To find out, read the code.
185  *
186  * <li>These tests are "conformance tests", and do not attempt to
187  * test throughput, latency, scalability or other performance factors
188  * (see the separate "jtreg" tests for a set intended to check these
189  * for the most central aspects of functionality.) So, most tests use
190  * the smallest sensible numbers of threads, collection sizes, etc
191  * needed to check basic conformance.
192  *
193  * <li>The test classes currently do not declare inclusion in
194  * any particular package to simplify things for people integrating
195  * them in TCK test suites.
196  *
197  * <li>As a convenience, the {@code main} of this class (JSR166TestCase)
198  * runs all JSR166 unit tests.
199  *
200  * </ul>
201  */

268     private static float delayFactor() {
269         float x;
270         if (!Float.isNaN(x = systemPropertyValue("jsr166.delay.factor")))
271             return x;
272         if (!Float.isNaN(x = systemPropertyValue("test.timeout.factor")))
273             return x;
274         String prop = System.getProperty("java.vm.version");
275         if (prop != null && prop.matches(".*debug.*"))
276             return 4.0f; // How much slower is fastdebug than product?!
277         return 1.0f;
278     }

346     protected void runTest() throws Throwable {
347         for (int i = 0; i < runsPerTest; i++) {
348             // currentRun = i;
349             if (profileTests)
350                 runTestProfiled();
351             else
352                 super.runTest();
353         }
354     }
355 
356     protected void runTestProfiled() throws Throwable {
357         for (int i = 0; i < 2; i++) {
358             long startTime = System.nanoTime();
359             super.runTest();
360             long elapsedMillis = millisElapsedSince(startTime);
361             if (elapsedMillis < profileThreshold)
362                 break;
363             // Never report first run of any test; treat it as a
364             // warmup run, notably to trigger all needed classloading,
365             if (i > 0)
366                 System.out.printf("%n%s: %d%n", toString(), elapsedMillis);
367         }
368     }

418                 System.exit(1);
419             System.gc();
420             System.runFinalization();
421         }

424     public static TestSuite newTestSuite(Object... suiteOrClasses) {
425         TestSuite suite = new TestSuite();
426         for (Object suiteOrClass : suiteOrClasses) {
427             if (suiteOrClass instanceof TestSuite)
428                 suite.addTest((TestSuite) suiteOrClass);
429             else if (suiteOrClass instanceof Class)

431             else

433         }
434         return suite;

599     /** Returns list of junit-style test method names in given class. */
600     public static ArrayList<String> testMethodNames(Class<?> testClass) {
601         Method[] methods = testClass.getDeclaredMethods();
602         ArrayList<String> names = new ArrayList<>(methods.length);
603         for (Method method : methods) {
604             if (method.getName().startsWith("test")
605                 && Modifier.isPublic(method.getModifiers())
606                 // method.getParameterCount() requires jdk8+
607                 && method.getParameterTypes().length == 0) {
608                 names.add(method.getName());
609             }
610         }
611         return names;
612     }

651                     .invoke(null, data);
652             } catch (Exception e) {
653                 throw new Error(e);
654             }
655         } else {
656             return new TestSuite();
657         }

743         Throwable t = threadFailure.getAndSet(null);
744         if (t != null) {
745             if (t instanceof Error)
746                 throw (Error) t;
747             else if (t instanceof RuntimeException)
748                 throw (RuntimeException) t;
749             else if (t instanceof Exception)
750                 throw (Exception) t;
751             else {
752                 AssertionFailedError afe =
753                     new AssertionFailedError(t.toString());
754                 afe.initCause(t);
755                 throw afe;
756             }
757         }
758 
759         if (Thread.interrupted())
760             tearDownFail("interrupt status set in main thread");
761

763     }

922     /**
923      * Delays, via Thread.sleep, for the given millisecond delay, but
924      * if the sleep is shorter than specified, may re-sleep or yield
925      * until time elapses.  Ensures that the given time, as measured
926      * by System.nanoTime(), has elapsed.
927      */
928     static void delay(long millis) throws InterruptedException {
929         long nanos = millis * (1000 * 1000);
930         final long wakeupTime = System.nanoTime() + nanos;
931         do {
932             if (millis > 0L)
933                 Thread.sleep(millis);
934             else // too short to sleep
935                 Thread.yield();
936             nanos = wakeupTime - System.nanoTime();
937             millis = nanos / (1000 * 1000);
938         } while (nanos >= 0L);
939     }

999         try {

1001             if (!pool.awaitTermination(2 * LONG_DELAY_MS, MILLISECONDS)) {

1313      * thread to enter a wait state: BLOCKED, WAITING, or TIMED_WAITING.
1314      */
1315     void waitForThreadToEnterWaitState(Thread thread, long timeoutMillis) {
1316         long startTime = 0L;
1317         for (;;) {
1318             Thread.State s = thread.getState();
1319             if (s == Thread.State.BLOCKED ||
1320                 s == Thread.State.WAITING ||
1321                 s == Thread.State.TIMED_WAITING)
1322                 return;
1323             else if (s == Thread.State.TERMINATED)
1324                 fail("Unexpected thread termination");
1325             else if (startTime == 0L)
1326                 startTime = System.nanoTime();
1327             else if (millisElapsedSince(startTime) > timeoutMillis) {
1328                 threadAssertTrue(thread.isAlive());
1329                 fail("timed out waiting for thread to enter wait state");
1330             }
1331             Thread.yield();
1332         }
