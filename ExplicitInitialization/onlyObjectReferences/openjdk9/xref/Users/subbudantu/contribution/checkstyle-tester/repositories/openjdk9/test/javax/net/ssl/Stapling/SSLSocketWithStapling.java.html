386     /**
387      * This test initiates stapling from the client, but the server does not
388      * support OCSP stapling for this connection.  In this case it happens
389      * because the latency of the OCSP responders is longer than the server
390      * is willing to wait.  To keep the test streamlined, we will set the server
391      * latency to a 1 second wait, and set the responder latency to 3 seconds.
392      *
393      * @param fallback if we allow client-side OCSP fallback, which
394      * will change the result from the client failing with CPVE (no fallback)
395      * to a pass (fallback active).
396      */

398         ClientParameters cliParams = new ClientParameters();
399         ServerParameters servParams = new ServerParameters();
400         serverReady = false;
401 
402         // Give a 1 second delay before running the test.
403         intOcsp.setDelay(3000);
404         rootOcsp.setDelay(3000);
405         Thread.sleep(1000);
406 
407         // Wait 5 seconds for server ready
408         for (int i = 0; (i < 100 && (!intOcsp.isServerReady() || !rootOcsp.isServerReady())); i++) {
409             Thread.sleep(50);
410         }
411         if (!intOcsp.isServerReady() || !rootOcsp.isServerReady()) {
412             throw new RuntimeException("Server not ready yet");
413         }
414 
415         System.out.println("========================================");
416         System.out.println("Stapling enbled in client.  Server does");
417         System.out.println("not support stapling due to OCSP latency.");
418         System.out.println("PKIXParameters with Revocation checking");
419         System.out.println("enabled, client-side OCSP checking is.");
420         System.out.println(fallback ? "enabled" : "disabled");
421         System.out.println("========================================");
422 
423         Security.setProperty("ocsp.enable", fallback.toString());
424         cliParams.pkixParams = new PKIXBuilderParameters(trustStore,
425                 new X509CertSelector());
426         cliParams.pkixParams.setRevocationEnabled(true);

446                         "Expected client failure, but the client succeeded");
447                 }
448             }
449         }
450         System.out.println("                 PASS");
451         System.out.println("========================================\n");
452 
453         // Remove the OCSP responder latency
454         intOcsp.setDelay(0);
455         rootOcsp.setDelay(0);
456         Thread.sleep(1000);

458         // Wait 5 seconds for server ready
459         for (int i = 0; (i < 100 && (!intOcsp.isServerReady() || !rootOcsp.isServerReady())); i++) {
460             Thread.sleep(50);
461         }
462         if (!intOcsp.isServerReady() || !rootOcsp.isServerReady()) {
463             throw new RuntimeException("Server not ready yet");
464         }

467     /*
468      * Define the server side of the test.
469      *
470      * If the server prematurely exits, serverReady will be set to true
471      * to avoid infinite hangs.
472      */
473     void doServerSide(ServerParameters servParams) throws Exception {
474 
475         // Selectively enable or disable the feature
476         System.setProperty("jdk.tls.server.enableStatusRequestExtension",
477                 Boolean.toString(servParams.enabled));
478 
479         // Set all the other operating parameters
480         System.setProperty("jdk.tls.stapling.cacheSize",
481                 Integer.toString(servParams.cacheSize));
482         System.setProperty("jdk.tls.stapling.cacheLifetime",
483                 Integer.toString(servParams.cacheLifetime));
484         System.setProperty("jdk.tls.stapling.responseTimeout",
485                 Integer.toString(servParams.respTimeout));
486         System.setProperty("jdk.tls.stapling.responderURI", servParams.respUri);
487         System.setProperty("jdk.tls.stapling.responderOverride",
488                 Boolean.toString(servParams.respOverride));
489         System.setProperty("jdk.tls.stapling.ignoreExtensions",
490                 Boolean.toString(servParams.ignoreExts));
491 
492         // Set keystores and trust stores for the server
493         KeyManagerFactory kmf = KeyManagerFactory.getInstance("SunX509");
494         kmf.init(serverKeystore, passwd.toCharArray());
495         TrustManagerFactory tmf = TrustManagerFactory.getInstance("SunX509");
496         tmf.init(trustStore);
497 
498         SSLContext sslc = SSLContext.getInstance("TLS");
499         sslc.init(kmf.getKeyManagers(), tmf.getTrustManagers(), null);
500 
501         SSLServerSocketFactory sslssf = sslc.getServerSocketFactory();
502 
503         try (SSLServerSocket sslServerSocket =
504                 (SSLServerSocket) sslssf.createServerSocket(serverPort)) {
505 
506             serverPort = sslServerSocket.getLocalPort();
507 
508             /*
509              * Signal Client, we're ready for his connect.
510              */
511             serverReady = true;
512 
513             try (SSLSocket sslSocket = (SSLSocket) sslServerSocket.accept();
514                     InputStream sslIS = sslSocket.getInputStream();
515                     OutputStream sslOS = sslSocket.getOutputStream()) {
516                 int numberIn = sslIS.read();
517                 int numberSent = 85;
518                 log("Server received number: " + numberIn);
519                 sslOS.write(numberSent);
520                 sslOS.flush();
521                 log("Server sent number: " + numberSent);
522             }
523         }
524     }

527      * Define the client side of the test.
528      *
529      * If the server prematurely exits, serverReady will be set to true
530      * to avoid infinite hangs.
531      */
532     void doClientSide(ClientParameters cliParams) throws Exception {
533 
534         // Wait 5 seconds for server ready
535         for (int i = 0; (i < 100 && !serverReady); i++) {
536             Thread.sleep(50);
537         }
538         if (!serverReady) {
539             throw new RuntimeException("Server not ready yet");
540         }
541 
542         // Selectively enable or disable the feature
543         System.setProperty("jdk.tls.client.enableStatusRequestExtension",
544                 Boolean.toString(cliParams.enabled));
545 
546         // Create the Trust Manager Factory using the PKIX variant
547         TrustManagerFactory tmf = TrustManagerFactory.getInstance("PKIX");
548 
549         // If we have a customized pkixParameters then use it
550         if (cliParams.pkixParams != null) {
551             // LIf we have a customized PKIXRevocationChecker, add
552             // it to the PKIXBuilderParameters.
553             if (cliParams.revChecker != null) {
554                 cliParams.pkixParams.addCertPathChecker(cliParams.revChecker);
555             }
556 
557             ManagerFactoryParameters trustParams =
558                     new CertPathTrustManagerParameters(cliParams.pkixParams);
559             tmf.init(trustParams);
560         } else {
561             tmf.init(trustStore);
562         }
563 
564         SSLContext sslc = SSLContext.getInstance("TLS");
565         sslc.init(null, tmf.getTrustManagers(), null);
566 
567         SSLSocketFactory sslsf = sslc.getSocketFactory();
568         try (SSLSocket sslSocket = (SSLSocket)sslsf.createSocket("localhost",
569                 serverPort);
570                 InputStream sslIS = sslSocket.getInputStream();
571                 OutputStream sslOS = sslSocket.getOutputStream()) {
572             int numberSent = 80;
573             sslOS.write(numberSent);
574             sslOS.flush();
575             log("Client sent number: " + numberSent);
576             int numberIn = sslIS.read();
577             log("Client received number:" + numberIn);
578         }
579     }

581     /*
582      * Primary constructor, used to drive remainder of the test.
583      *
584      * Fork off the other side, then do your work.
585      */
586     SSLSocketWithStapling(ClientParameters cliParams,
587             ServerParameters servParams) throws Exception {
588         Exception startException = null;
589         try {
590             if (separateServerThread) {
591                 startServer(servParams, true);
592                 startClient(cliParams, false);
593             } else {
594                 startClient(cliParams, true);
595                 startServer(servParams, false);
596             }
597         } catch (Exception e) {
598             startException = e;
599         }
600 
601         /*
602          * Wait for other side to close down.
603          */
604         if (separateServerThread) {
605             if (serverThread != null) {
606                 serverThread.join();
607             }
608         } else {
609             if (clientThread != null) {
610                 clientThread.join();
611             }
612         }
613     }

615     /**
616      * Checks a validation failure to see if it failed for the reason we think
617      * it should.  This comes in as an SSLException of some sort, but it
618      * encapsulates a ValidatorException which in turn encapsulates the
619      * CertPathValidatorException we are interested in.
620      *
621      * @param e the exception thrown at the top level
622      * @param reason the underlying CertPathValidatorException BasicReason
623      * we are expecting it to have.
624      *
625      * @return true if the reason matches up, false otherwise.
626      */
627     static boolean checkClientValidationFailure(Exception e,
628             BasicReason reason) {
629         boolean result = false;
630 
631         if (e instanceof SSLException) {
632             Throwable valExc = e.getCause();
633             if (valExc instanceof sun.security.validator.ValidatorException) {
634                 Throwable cause = valExc.getCause();
635                 if (cause instanceof CertPathValidatorException) {
636                     CertPathValidatorException cpve =
637                             (CertPathValidatorException)cause;
638                     if (cpve.getReason() == reason) {
639                         result = true;
640                     }
641                 }
642             }
643         }
644         return result;
645     }

654     void startServer(ServerParameters servParams, boolean newThread)
655             throws Exception {
656         if (newThread) {
657             serverThread = new Thread() {
658                 public void run() {
659                     try {
660                         doServerSide(servParams);
661                     } catch (Exception e) {
662                         /*
663                          * Our server thread just died.
664                          *
665                          * Release the client, if not active already...
666                          */
667                         System.err.println("Server died...");
668                         e.printStackTrace(System.err);
669                         serverReady = true;
670                         serverException = e;
671                     }
672                 }
673             };
674             serverThread.start();
675         } else {
676             try {
677                 doServerSide(servParams);
678             } catch (Exception e) {
679                 serverException = e;
680             } finally {
681                 serverReady = true;
682             }

712     /**
713      * Creates the PKI components necessary for this test, including
714      * Root CA, Intermediate CA and SSL server certificates, the keystores
715      * for each entity, a client trust store, and starts the OCSP responders.
716      */
717     private static void createPKI() throws Exception {
718         CertificateBuilder cbld = new CertificateBuilder();
719         KeyPairGenerator keyGen = KeyPairGenerator.getInstance("RSA");
720         keyGen.initialize(2048);
721         KeyStore.Builder keyStoreBuilder =
722                 KeyStore.Builder.newInstance("PKCS12", null,
723                         new KeyStore.PasswordProtection(passwd.toCharArray()));
724 
725         // Generate Root, IntCA, EE keys
726         KeyPair rootCaKP = keyGen.genKeyPair();
727         log("Generated Root CA KeyPair");
728         KeyPair intCaKP = keyGen.genKeyPair();
729         log("Generated Intermediate CA KeyPair");
730         KeyPair sslKP = keyGen.genKeyPair();
731         log("Generated SSL Cert KeyPair");
732 
733         // Set up the Root CA Cert
734         cbld.setSubjectName("CN=Root CA Cert, O=SomeCompany");
735         cbld.setPublicKey(rootCaKP.getPublic());
736         cbld.setSerialNumber(new BigInteger("1"));
737         // Make a 3 year validity starting from 60 days ago
738         long start = System.currentTimeMillis() - TimeUnit.DAYS.toMillis(60);
739         long end = start + TimeUnit.DAYS.toMillis(1085);
740         cbld.setValidity(new Date(start), new Date(end));
741         addCommonExts(cbld, rootCaKP.getPublic(), rootCaKP.getPublic());
742         addCommonCAExts(cbld);
743         // Make our Root CA Cert!
744         X509Certificate rootCert = cbld.build(null, rootCaKP.getPrivate(),
745                 "SHA256withRSA");
746         log("Root CA Created:\n" + certInfo(rootCert));
747 
748         // Now build a keystore and add the keys and cert
749         rootKeystore = keyStoreBuilder.getKeyStore();
750         Certificate[] rootChain = {rootCert};
751         rootKeystore.setKeyEntry(ROOT_ALIAS, rootCaKP.getPrivate(),
752                 passwd.toCharArray(), rootChain);
753 
754         // Now fire up the OCSP responder
755         rootOcsp = new SimpleOCSPServer(rootKeystore, passwd, ROOT_ALIAS, null);
756         rootOcsp.enableLog(debug);
757         rootOcsp.setNextUpdateInterval(3600);
758         rootOcsp.start();
759 
760         // Wait 5 seconds for server ready
761         for (int i = 0; (i < 100 && !rootOcsp.isServerReady()); i++) {
762             Thread.sleep(50);
763         }
764         if (!rootOcsp.isServerReady()) {
765             throw new RuntimeException("Server not ready yet");
766         }
767 
768         rootOcspPort = rootOcsp.getPort();
769         String rootRespURI = "http://localhost:" + rootOcspPort;
770         log("Root OCSP Responder URI is " + rootRespURI);
771 
772         // Now that we have the root keystore and OCSP responder we can
773         // create our intermediate CA.
774         cbld.reset();
775         cbld.setSubjectName("CN=Intermediate CA Cert, O=SomeCompany");
776         cbld.setPublicKey(intCaKP.getPublic());
777         cbld.setSerialNumber(new BigInteger("100"));
778         // Make a 2 year validity starting from 30 days ago
779         start = System.currentTimeMillis() - TimeUnit.DAYS.toMillis(30);
780         end = start + TimeUnit.DAYS.toMillis(730);
781         cbld.setValidity(new Date(start), new Date(end));
782         addCommonExts(cbld, intCaKP.getPublic(), rootCaKP.getPublic());
783         addCommonCAExts(cbld);
784         cbld.addAIAExt(Collections.singletonList(rootRespURI));
785         // Make our Intermediate CA Cert!
786         X509Certificate intCaCert = cbld.build(rootCert, rootCaKP.getPrivate(),
787                 "SHA256withRSA");
788         log("Intermediate CA Created:\n" + certInfo(intCaCert));
789 
790         // Provide intermediate CA cert revocation info to the Root CA
791         // OCSP responder.
792         Map<BigInteger, SimpleOCSPServer.CertStatusInfo> revInfo =
793             new HashMap<>();
794         revInfo.put(intCaCert.getSerialNumber(),
795                 new SimpleOCSPServer.CertStatusInfo(
796                         SimpleOCSPServer.CertStatus.CERT_STATUS_GOOD));
797         rootOcsp.updateStatusDb(revInfo);
798 
799         // Now build a keystore and add the keys, chain and root cert as a TA
800         intKeystore = keyStoreBuilder.getKeyStore();
801         Certificate[] intChain = {intCaCert, rootCert};
802         intKeystore.setKeyEntry(INT_ALIAS, intCaKP.getPrivate(),
803                 passwd.toCharArray(), intChain);
804         intKeystore.setCertificateEntry(ROOT_ALIAS, rootCert);
805 
806         // Now fire up the Intermediate CA OCSP responder
807         intOcsp = new SimpleOCSPServer(intKeystore, passwd,
808                 INT_ALIAS, null);
809         intOcsp.enableLog(debug);
810         intOcsp.setNextUpdateInterval(3600);
811         intOcsp.start();
812 
813         // Wait 5 seconds for server ready
814         for (int i = 0; (i < 100 && !intOcsp.isServerReady()); i++) {
815             Thread.sleep(50);
816         }
817         if (!intOcsp.isServerReady()) {
818             throw new RuntimeException("Server not ready yet");
819         }
820 
821         intOcspPort = intOcsp.getPort();
822         String intCaRespURI = "http://localhost:" + intOcspPort;
823         log("Intermediate CA OCSP Responder URI is " + intCaRespURI);
824 
825         // Last but not least, let's make our SSLCert and add it to its own
826         // Keystore
827         cbld.reset();
828         cbld.setSubjectName("CN=SSLCertificate, O=SomeCompany");
829         cbld.setPublicKey(sslKP.getPublic());
830         cbld.setSerialNumber(new BigInteger("4096"));
831         // Make a 1 year validity starting from 7 days ago
832         start = System.currentTimeMillis() - TimeUnit.DAYS.toMillis(7);
833         end = start + TimeUnit.DAYS.toMillis(365);
834         cbld.setValidity(new Date(start), new Date(end));
835 
836         // Add extensions
837         addCommonExts(cbld, sslKP.getPublic(), intCaKP.getPublic());
838         boolean[] kuBits = {true, false, true, false, false, false,
839             false, false, false};
840         cbld.addKeyUsageExt(kuBits);
841         List<String> ekuOids = new ArrayList<>();
842         ekuOids.add("1.3.6.1.5.5.7.3.1");
843         ekuOids.add("1.3.6.1.5.5.7.3.2");
844         cbld.addExtendedKeyUsageExt(ekuOids);
845         cbld.addSubjectAltNameDNSExt(Collections.singletonList("localhost"));
846         cbld.addAIAExt(Collections.singletonList(intCaRespURI));
847         // Make our SSL Server Cert!
848         X509Certificate sslCert = cbld.build(intCaCert, intCaKP.getPrivate(),
849                 "SHA256withRSA");
850         log("SSL Certificate Created:\n" + certInfo(sslCert));
851 
852         // Provide SSL server cert revocation info to the Intermeidate CA
853         // OCSP responder.
854         revInfo = new HashMap<>();
855         revInfo.put(sslCert.getSerialNumber(),
856                 new SimpleOCSPServer.CertStatusInfo(
857                         SimpleOCSPServer.CertStatus.CERT_STATUS_GOOD));
858         intOcsp.updateStatusDb(revInfo);
859 
860         // Now build a keystore and add the keys, chain and root cert as a TA
861         serverKeystore = keyStoreBuilder.getKeyStore();
862         Certificate[] sslChain = {sslCert, intCaCert, rootCert};
863         serverKeystore.setKeyEntry(SSL_ALIAS, sslKP.getPrivate(),
864                 passwd.toCharArray(), sslChain);
865         serverKeystore.setCertificateEntry(ROOT_ALIAS, rootCert);
866 
867         // And finally a Trust Store for the client
868         trustStore = keyStoreBuilder.getKeyStore();
869         trustStore.setCertificateEntry(ROOT_ALIAS, rootCert);
870     }

872     private static void addCommonExts(CertificateBuilder cbld,
873             PublicKey subjKey, PublicKey authKey) throws IOException {
874         cbld.addSubjectKeyIdExt(subjKey);
875         cbld.addAuthorityKeyIdExt(authKey);
876     }

878     private static void addCommonCAExts(CertificateBuilder cbld)
879             throws IOException {
880         cbld.addBasicConstraintsExt(true, true, -1);
881         // Set key usage bits for digitalSignature, keyCertSign and cRLSign
882         boolean[] kuBitSettings = {true, false, false, false, false, true,
883             true, false, false};
884         cbld.addKeyUsageExt(kuBitSettings);
885     }

887     /**
888      * Helper routine that dumps only a few cert fields rather than
889      * the whole toString() output.
890      *
891      * @param cert an X509Certificate to be displayed
892      *
893      * @return the String output of the issuer, subject and
894      * serial number
895      */
896     private static String certInfo(X509Certificate cert) {
897         StringBuilder sb = new StringBuilder();
898         sb.append("Issuer: ").append(cert.getIssuerX500Principal()).
899                 append("\n");
900         sb.append("Subject: ").append(cert.getSubjectX500Principal()).
901                 append("\n");
902         sb.append("Serial: ").append(cert.getSerialNumber()).append("\n");
903         return sb.toString();
904     }
