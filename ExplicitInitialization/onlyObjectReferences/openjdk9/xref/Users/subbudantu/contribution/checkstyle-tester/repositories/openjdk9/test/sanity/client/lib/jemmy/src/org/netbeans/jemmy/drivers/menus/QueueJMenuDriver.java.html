49  /**
50   *
51   * 100% stable menu driver. Tries to do next steps during one action executed
52   * through EventQueue:<br>
53   * find showing window containing popup<br>
54   * find showing popup<br>
55   * find showing menuitem<br>
56   * enter mouse into it<br>
57   *
58   * Repeats this action as many times as "JMenuOperator.WaitPopupTimeout" timeout
59   * allows.
60   *
61   * @author Alexandre Iline (alexandre.iline@oracle.com)
62   *
63   */
64  public class QueueJMenuDriver extends LightSupportiveDriver implements MenuDriver {
65  
66      QueueTool queueTool;
67  
68      public QueueJMenuDriver() {
69          super(new String[]{"org.netbeans.jemmy.operators.JMenuOperator",
70              "org.netbeans.jemmy.operators.JMenuBarOperator",
71              "org.netbeans.jemmy.operators.JPopupMenuOperator"});
72          queueTool = new QueueTool();
73      }
74  
75      @Override
76      public Object pushMenu(final ComponentOperator oper, PathChooser chooser) {
77          queueTool.setOutput(oper.getOutput().createErrorOutput());
78          checkSupported(oper);
79          JMenuItem result;
80          OneReleaseAction action;
81          if (oper instanceof JMenuBarOperator) {
82              action = new OneReleaseAction(chooser, 0, oper, false) {
83                  @Override
84                  protected void pushAlone(JMenuItemOperator subMenuOper) {
85                      if (subMenuOper.getSource() instanceof JMenu
86                              && isMenuBarSelected((JMenuBar) oper.getSource())) {
87                          DriverManager.getMouseDriver(subMenuOper).enterMouse(subMenuOper);
88                      } else {
89                          DriverManager.getButtonDriver(subMenuOper).push(subMenuOper);
90                      }
91                  }
92  
93                  @Override
94                  protected boolean inTheMiddle(JMenuOperator subMenuOper, boolean mousePressed) {
95                      if (isMenuBarSelected((JMenuBar) oper.getSource())) {
96                          DriverManager.getMouseDriver(subMenuOper).enterMouse(subMenuOper);
97                          return false;
98                      } else {
99                          return super.inTheMiddle(subMenuOper, mousePressed);
100                     }
101                 }
102 
103                 @Override
104                 protected void process(MenuElement element) {
105                     super.process(element);
106                 }
107 
108                 @Override
109                 public MenuElement getMenuElement() {
110                     return (MenuElement) oper.getSource();
111                 }
112             };
113         } else if (oper instanceof JPopupMenuOperator) {
114             action = new OneReleaseAction(chooser, 0, oper, false) {
115                 @Override
116                 public MenuElement getMenuElement() {
117                     return (MenuElement) oper.getSource();
118                 }
119             };
120         } else {
121             DriverManager.getButtonDriver(oper).press(oper);
122             action = new OneReleaseAction(chooser, 0, oper, false) {
123                 @Override
124                 public MenuElement launch() {
125                     process((MenuElement) oper.getSource());
126                     return (MenuElement) oper.getSource();
127                 }
128 
129                 @Override
130                 public MenuElement getMenuElement() {
131                     return null;
132                 }
133             };
134         }
135         queueTool.waitEmpty(10);
136         queueTool.waitEmpty(10);
137         queueTool.waitEmpty(10);
138         result = runAction(action, oper,
139                 oper.getTimeouts().getTimeout("ComponentOperator.WaitComponentTimeout"),
140                 (chooser instanceof DescriptablePathChooser)
141                         ? ((DescriptablePathChooser) chooser).getDescription()
142                         : "Menu pushing");
143         if (result instanceof JMenu) {
144             for (int i = 1; i < chooser.getDepth(); i++) {
145                 final JMenu menu = (JMenu) result;
146                 final ComponentChooser popupChooser = new PopupMenuChooser(menu);
147                 action = new OneReleaseAction(chooser, i, oper, action.mousePressed) {
148                     @Override
149                     public MenuElement getMenuElement() {
150                         Window win = JPopupMenuOperator.findJPopupWindow(popupChooser);
151                         if (win != null && win.isShowing()) {
152                             return JPopupMenuOperator.findJPopupMenu(win, popupChooser);
153                         } else {
154                             return null;
155                         }
156                     }
157                 };
158                 result = runAction(action, oper,
159                         oper.getTimeouts().getTimeout("JMenuOperator.WaitPopupTimeout"),
160                         (chooser instanceof DescriptablePathChooser)
161                                 ? ((DescriptablePathChooser) chooser).getDescription()
162                                 : "Menu pushing");
163             }
164         }
165         return result;
166     }
167 
168     private JMenuItem runAction(final OneReleaseAction action, ComponentOperator env, long waitingTime, final String description) {
169         Waiter<MenuElement, Void> waiter = new Waiter<>(new Waitable<MenuElement, Void>() {
170             @Override
171             public MenuElement actionProduced(Void param) {
172                 return queueTool.invokeSmoothly(action);
173             }
174 
175             @Override
176             public String getDescription() {
177                 return description;
178             }
179 
180             @Override
181             public String toString() {
182                 return "runAction.Waiter{description = " + getDescription() + '}';
183             }
184         });
185         waiter.setOutput(env.getOutput().createErrorOutput());
186         waiter.setTimeouts(env.getTimeouts().cloneThis());
187         waiter.getTimeouts().setTimeout("Waiter.WaitingTime",
188                 waitingTime);
189         waiter.getTimeouts().setTimeout("Waiter.TimeDelta", 100);
190         //1.5 workaround
191         if (System.getProperty("java.specification.version").compareTo("1.4") > 0) {
192             queueTool.setOutput(env.getOutput().createErrorOutput());
193             queueTool.waitEmpty(10);
194             queueTool.waitEmpty(10);
195             queueTool.waitEmpty(10);
196         }
197         //end of 1.5 workaround
198         try {
199             return (JMenuItem) waiter.waitAction(null);
200         } catch (InterruptedException e) {
201             action.stop();
202             throw (new JemmyException("Waiting has been interrupted", e));
203         }
204     }

206     private boolean isMenuBarSelected(JMenuBar bar) {
207         MenuElement[] subElements = bar.getSubElements();
208         for (MenuElement subElement : subElements) {
209             if (subElement instanceof JMenu
210                     && ((JMenu) subElement).isPopupMenuVisible()) {
211                 return true;
212             }
213         }
214         return false;
215     }
216 
217     private abstract class OneReleaseAction extends QueueTool.QueueAction<MenuElement> {
218 
219         PathChooser chooser;
220         int depth;
221         ComponentOperator env;
222         boolean mousePressed = false;
223         private boolean stopped = false;
224 
225         public OneReleaseAction(PathChooser chooser, int depth, ComponentOperator env, boolean mousePressed) {
226             super("Menu pushing");
227             this.chooser = chooser;
228             this.depth = depth;
229             this.env = env;
230             this.mousePressed = mousePressed;
231         }
232 
233         protected void pushAlone(JMenuItemOperator subMenuOper) {
234             DriverManager.getButtonDriver(subMenuOper).push(subMenuOper);
235         }
236 
237         protected void pushLast(JMenuItemOperator subMenuOper, boolean mousePressed) {
238             DriverManager.getMouseDriver(subMenuOper).enterMouse(subMenuOper);
239             DriverManager.getButtonDriver(subMenuOper).release(subMenuOper);
240         }
241 
242         protected boolean inTheMiddle(JMenuOperator subMenuOper, boolean mousePressed) {
243             if (!subMenuOper.isPopupMenuVisible()) {
244                 if (!mousePressed) {
245                     DriverManager.getMouseDriver(subMenuOper).enterMouse(subMenuOper);
246                     DriverManager.getButtonDriver(subMenuOper).press(subMenuOper);
247                 } else {
248                     DriverManager.getMouseDriver(subMenuOper).enterMouse(subMenuOper);
249                 }
250                 return true;
251             }
252             return mousePressed;
253         }
254 
255         protected void process(MenuElement element) {
256             if (depth == chooser.getDepth() - 1) {
257                 JMenuItemOperator subMenuOper = new JMenuItemOperator((JMenuItem) element);
258                 subMenuOper.copyEnvironment(env);
259                 if (depth == 0) {
260                     pushAlone(subMenuOper);
261                 } else {
262                     pushLast(subMenuOper, mousePressed);
263                 }
264             } else if (element instanceof JMenu) {
265                 JMenuOperator subMenuOper = new JMenuOperator((JMenu) element);
266                 subMenuOper.copyEnvironment(env);
267                 mousePressed = inTheMiddle(subMenuOper, mousePressed);
268             } else {
269                 throw (new JemmyException("Menu path too long"));
270             }
271         }
272 
273         @Override
274         public MenuElement launch() {
275             MenuElement element = getMenuElement();
276             if (element != null) {
277                 MenuElement[] subElements = element.getSubElements();
278                 for (MenuElement subElement : subElements) {
279                     if (((Component) subElement).isShowing()
280                             && ((Component) subElement).isEnabled()
281                             && chooser.checkPathComponent(depth, subElement)) {
282                         process(subElement);
283                         return subElement;
284                     }
285                     if (stopped) {
286                         return null;
287                     }
288                 }
289             }
290             return null;
291         }
292 
293         public abstract MenuElement getMenuElement();
294 
295         private void stop() {
296             stopped = true;
297         }
298     }
299 
300     private static class PopupMenuChooser implements ComponentChooser {
301 
302         JMenu menu;
303 
304         public PopupMenuChooser(JMenu menu) {
305             this.menu = menu;
306         }
307 
308         @Override
309         public boolean checkComponent(Component comp) {
310             return (comp == menu.getPopupMenu()
311                     && comp.isShowing() && comp.isEnabled());
312         }
313 
314         @Override
315         public String getDescription() {
316             return menu.getText() + "'s popup";
317         }
318 
319         @Override
320         public String toString() {
321             return "PopupMenuChooser{" + "menu=" + menu + '}';
322         }
323     }
