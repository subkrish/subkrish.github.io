78      // this has to be a public static class, so that JCommander can use it!
79      public static class PropertyConverter implements IStringConverter<Properties> {
80  
81          private static final char SEPARATOR = '=';
82  
83          public Properties convert(String value) {
84              int indexOfSeparator = value.indexOf(SEPARATOR);
85              if (indexOfSeparator < 0) {
86                  throw new ParameterException(
87                          "Property name must be separated with an = sign from it value: name=value.");
88              }
89              String propertyName = value.substring(0, indexOfSeparator);
90              String propertyValue = value.substring(indexOfSeparator + 1);
91              Properties properties = new Properties();
92              properties.put(propertyName, propertyValue);
93              return properties;
94          }
95      }

97      // this has to be a public static class, so that JCommander can use it!
98      public static class RulePriorityConverter implements IStringConverter<RulePriority> {
99  
100         public int validate(String value) throws ParameterException {
101             int minPriorityValue = Integer.parseInt(value);
102             if (minPriorityValue < 1 || minPriorityValue > 5) {
103                 throw new ParameterException("Priority values can only be integer value, between 1 and 5," + value
104                         + " is not valid");
105             }
106             return minPriorityValue;
107         }
108 
109         public RulePriority convert(String value) {
110             return RulePriority.valueOf(validate(value));
111         }
112     }

114     public static PMDConfiguration transformParametersIntoConfiguration(PMDParameters params) {
115         if (null == params.getSourceDir() && null == params.getUri()) {
116             throw new IllegalArgumentException(
117                     "Please provide either source root directory (-dir or -d) or database URI (-uri or -u) parameter");
118         }
119         PMDConfiguration configuration = new PMDConfiguration();
120         configuration.setInputPaths(params.getSourceDir());
121         configuration.setInputUri(params.getUri());
122         configuration.setReportFormat(params.getFormat());
123         configuration.setBenchmark(params.isBenchmark());
124         configuration.setDebug(params.isDebug());
125         configuration.setMinimumPriority(params.getMinimumPriority());
126         configuration.setReportFile(params.getReportfile());
127         configuration.setReportProperties(params.getProperties());
128         configuration.setReportShortNames(params.isShortnames());
129         configuration.setRuleSets(params.getRulesets());
130         configuration.setShowSuppressedViolations(params.isShowsuppressed());
131         configuration.setSourceEncoding(params.getEncoding());
132         configuration.setStressTest(params.isStress());
133         configuration.setSuppressMarker(params.getSuppressmarker());
134         configuration.setThreads(params.getThreads());
135 
136         LanguageVersion languageVersion = LanguageRegistry.findLanguageVersionByTerseName(params.getLanguage() + " " + params.getVersion());
137         if(languageVersion != null) {
138             configuration.getLanguageVersionDiscoverer().setDefaultLanguageVersion(languageVersion);
139         }
140         try {
141             configuration.prependClasspath(params.getAuxclasspath());
142         } catch (IOException e) {
143             throw new IllegalArgumentException("Invalid auxiliary classpath: " + e.getMessage(), e);
144         }
145         return configuration;
146     }
