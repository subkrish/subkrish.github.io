56  public abstract class ResultSetMappingBinder {
57  	/**
58  	 * Build a ResultSetMappingDefinition given a containing element for the "return-XXX" elements
59  	 *
60  	 * @param resultSetElem The element containing the return definitions.
61  	 * @param path No clue...
62  	 * @param mappings The current processing state.
63  	 * @return The description of the mappings...
64  	 */
65  	protected static ResultSetMappingDefinition buildResultSetMappingDefinition(Element resultSetElem, String path, Mappings mappings) {
66  		String resultSetName = resultSetElem.attribute( "name" ).getValue();
67  		if ( path != null ) {
68  			resultSetName = path + '.' + resultSetName;
69  		}
70  		ResultSetMappingDefinition definition = new ResultSetMappingDefinition( resultSetName );
71  
72  		int cnt = 0;
73  		Iterator returns = resultSetElem.elementIterator();
74  		while ( returns.hasNext() ) {
75  			cnt++;
76  			Element returnElem = (Element) returns.next();
77  			String name = returnElem.getName();
78  			if ( "return-scalar".equals( name ) ) {
79  				String column = returnElem.attributeValue( "column" );
80  				String typeFromXML = HbmBinder.getTypeFromXML( returnElem );
81  				Type type = null;
82  				if(typeFromXML!=null) {
83  					type = mappings.getTypeResolver().heuristicType( typeFromXML );
84  					if ( type == null ) {
85  						throw new MappingException( "could not determine type " + type );
86  					}
87  				}
88  				definition.addQueryReturn( new NativeSQLQueryScalarReturn( column, type ) );
89  			}
90  			else if ( "return".equals( name ) ) {
91  				definition.addQueryReturn( bindReturn( returnElem, mappings, cnt ) );
92  			}
93  			else if ( "return-join".equals( name ) ) {
94  				definition.addQueryReturn( bindReturnJoin( returnElem, mappings ) );
95  			}
96  			else if ( "load-collection".equals( name ) ) {
97  				definition.addQueryReturn( bindLoadCollection( returnElem, mappings ) );
98  			}
99  		}
100 		return definition;
101 	}

103 	private static NativeSQLQueryRootReturn bindReturn(Element returnElem, Mappings mappings, int elementCount) {
104 		String alias = returnElem.attributeValue( "alias" );
105 		if( StringHelper.isEmpty( alias )) {
106 			alias = "alias_" + elementCount; // hack/workaround as sqlquery impl depend on having a key.
107 		}
108 
109 		String entityName = HbmBinder.getEntityName(returnElem, mappings);
110 		if(entityName==null) {
111 			throw new MappingException( "<return alias='" + alias + "'> must specify either a class or entity-name");
112 		}
113 		LockMode lockMode = getLockMode( returnElem.attributeValue( "lock-mode" ) );
114 
115 		PersistentClass pc = mappings.getClass( entityName );
116 		java.util.Map propertyResults = bindPropertyResults(alias, returnElem, pc, mappings );
117 
118 		return new NativeSQLQueryRootReturn(
119 				alias,
120 				entityName,
121 				propertyResults,
122 				lockMode
123 			);
124 	}

126 	private static NativeSQLQueryJoinReturn bindReturnJoin(Element returnElem, Mappings mappings) {
127 		String alias = returnElem.attributeValue( "alias" );
128 		String roleAttribute = returnElem.attributeValue( "property" );
129 		LockMode lockMode = getLockMode( returnElem.attributeValue( "lock-mode" ) );
130 		int dot = roleAttribute.lastIndexOf( '.' );
131 		if ( dot == -1 ) {
132 			throw new MappingException(
133 					"Role attribute for sql query return [alias=" + alias +
134 					"] not formatted correctly {owningAlias.propertyName}"
135 				);
136 		}
137 		String roleOwnerAlias = roleAttribute.substring( 0, dot );
138 		String roleProperty = roleAttribute.substring( dot + 1 );
139 
140 		//FIXME: get the PersistentClass
141 		java.util.Map propertyResults = bindPropertyResults(alias, returnElem, null, mappings );
142 
143 		return new NativeSQLQueryJoinReturn(
144 				alias,
145 				roleOwnerAlias,
146 				roleProperty,
147 				propertyResults, // TODO: bindpropertyresults(alias, returnElem)
148 				lockMode
149 			);
150 	}

152 	private static NativeSQLQueryCollectionReturn bindLoadCollection(Element returnElem, Mappings mappings) {
153 		String alias = returnElem.attributeValue( "alias" );
154 		String collectionAttribute = returnElem.attributeValue( "role" );
155 		LockMode lockMode = getLockMode( returnElem.attributeValue( "lock-mode" ) );
156 		int dot = collectionAttribute.lastIndexOf( '.' );
157 		if ( dot == -1 ) {
158 			throw new MappingException(
159 					"Collection attribute for sql query return [alias=" + alias +
160 					"] not formatted correctly {OwnerClassName.propertyName}"
161 				);
162 		}
163 		String ownerClassName = HbmBinder.getClassName( collectionAttribute.substring( 0, dot ), mappings );
164 		String ownerPropertyName = collectionAttribute.substring( dot + 1 );
165 
166 		//FIXME: get the PersistentClass
167 		java.util.Map propertyResults = bindPropertyResults(alias, returnElem, null, mappings );
168 
169 		return new NativeSQLQueryCollectionReturn(
170 				alias,
171 				ownerClassName,
172 				ownerPropertyName,
173 				propertyResults,
174 				lockMode
175 			);
176 	}

178 	private static java.util.Map bindPropertyResults(
179 			String alias, Element returnElement, PersistentClass pc, Mappings mappings
180 	) {
181 
182 		HashMap propertyresults = new HashMap(); // maybe a concrete SQLpropertyresult type, but Map is exactly what is required at the moment
183 
184 		Element discriminatorResult = returnElement.element("return-discriminator");
185 		if(discriminatorResult!=null) {
186 			ArrayList resultColumns = getResultColumns(discriminatorResult);
187 			propertyresults.put("class", ArrayHelper.toStringArray( resultColumns ) );
188 		}
189 		Iterator iterator = returnElement.elementIterator("return-property");
190 		List properties = new ArrayList();
191 		List propertyNames = new ArrayList();
192 		while ( iterator.hasNext() ) {
193 			Element propertyresult = (Element) iterator.next();
194 			String name = propertyresult.attributeValue("name");
195 			if ( pc == null || name.indexOf( '.') == -1) { //if dotted and not load-collection nor return-join
196 				//regular property
197 				properties.add(propertyresult);
198 				propertyNames.add(name);
199 			}
200 			else {
201 				/**
202 				 * Reorder properties
203 				 * 1. get the parent property
204 				 * 2. list all the properties following the expected one in the parent property
205 				 * 3. calculate the lowest index and insert the property
206 				 */
207 				if (pc == null)
208 					throw new MappingException("dotted notation in <return-join> or <load_collection> not yet supported");
209 				int dotIndex = name.lastIndexOf( '.' );
210 				String reducedName = name.substring( 0, dotIndex );
211 				Value value = pc.getRecursiveProperty( reducedName ).getValue();
212 				Iterator parentPropIter;
213 				if ( value instanceof Component ) {
214 					Component comp = (Component) value;
215 					parentPropIter = comp.getPropertyIterator();
216 				}
217 				else if ( value instanceof ToOne ) {
218 					ToOne toOne = (ToOne) value;
219 					PersistentClass referencedPc = mappings.getClass( toOne.getReferencedEntityName() );
220 					if ( toOne.getReferencedPropertyName() != null ) {
221 						try {
222 							parentPropIter = ( (Component) referencedPc.getRecursiveProperty( toOne.getReferencedPropertyName() ).getValue() ).getPropertyIterator();
223 						} catch (ClassCastException e) {
224 							throw new MappingException("dotted notation reference neither a component nor a many/one to one", e);
225 						}
226 					}
227 					else {
228 						try {
229 							if ( referencedPc.getIdentifierMapper() == null ) {
230 								parentPropIter = ( (Component) referencedPc.getIdentifierProperty().getValue() ).getPropertyIterator();
231 							}
232 							else {
233 								parentPropIter = referencedPc.getIdentifierMapper().getPropertyIterator();
234 							}
235 						}
236 						catch (ClassCastException e) {
237 							throw new MappingException("dotted notation reference neither a component nor a many/one to one", e);
238 						}
239 					}
240 				}
241 				else {
242 					throw new MappingException("dotted notation reference neither a component nor a many/one to one");
243 				}
244 				boolean hasFollowers = false;
245 				List followers = new ArrayList();
246 				while ( parentPropIter.hasNext() ) {
247 					String currentPropertyName = ( (Property) parentPropIter.next() ).getName();
248 					String currentName = reducedName + '.' + currentPropertyName;
249 					if (hasFollowers) {
250 						followers.add( currentName );
251 					}
252 					if ( name.equals( currentName ) ) hasFollowers = true;
253 				}
254 
255 				int index = propertyNames.size();
256 				int followersSize = followers.size();
257 				for (int loop = 0 ; loop < followersSize ; loop++) {
258 					String follower = (String) followers.get(loop);
259 					int currentIndex = getIndexOfFirstMatchingProperty(propertyNames, follower);
260 					index = currentIndex != -1 && currentIndex < index ? currentIndex : index;
261 				}
262 				propertyNames.add(index, name);
263 				properties.add(index, propertyresult);
264 			}
265 		}
266 
267 		Set uniqueReturnProperty = new HashSet();
268 		iterator = properties.iterator();
269 		while ( iterator.hasNext() ) {
270 			Element propertyresult = (Element) iterator.next();
271 			String name = propertyresult.attributeValue("name");
272 			if ( "class".equals(name) ) {
273 				throw new MappingException(
274 						"class is not a valid property name to use in a <return-property>, use <return-discriminator> instead"
275 					);
276 			}
277 			//TODO: validate existing of property with the chosen name. (secondpass )
278 			ArrayList allResultColumns = getResultColumns(propertyresult);
279 
280 			if ( allResultColumns.isEmpty() ) {
281 				throw new MappingException(
282 						"return-property for alias " + alias +
283 						" must specify at least one column or return-column name"
284 					);
285 			}
286 			if ( uniqueReturnProperty.contains( name ) ) {
287 				throw new MappingException(
288 						"duplicate return-property for property " + name +
289 						" on alias " + alias
290 					);
291 			}
292 			uniqueReturnProperty.add(name);
293 
294 			// the issue here is that for <return-join/> representing an entity collection,
295 			// the collection element values (the property values of the associated entity)
296 			// are represented as 'element.{propertyname}'.  Thus the StringHelper.root()
297 			// here puts everything under 'element' (which additionally has significant
298 			// meaning).  Probably what we need to do is to something like this instead:
299 			//      String root = StringHelper.root( name );
300 			//      String key = root; // by default
301 			//      if ( !root.equals( name ) ) {
302 			//	        // we had a dot
303 			//          if ( !root.equals( alias ) {
304 			//              // the root does not apply to the specific alias
305 			//              if ( "elements".equals( root ) {
306 			//                  // we specifically have a <return-join/> representing an entity collection
307 			//                  // and this <return-property/> is one of that entity's properties
308 			//                  key = name;
309 			//              }
310 			//          }
311 			//      }
312 			// but I am not clear enough on the intended purpose of this code block, especially
313 			// in relation to the "Reorder properties" code block above...
314 //			String key = StringHelper.root( name );
315 			String key = name;
316 			ArrayList intermediateResults = (ArrayList) propertyresults.get( key );
317 			if (intermediateResults == null) {
318 				propertyresults.put( key, allResultColumns );
319 			}
320 			else {
321 				intermediateResults.addAll( allResultColumns );
322 			}
323 		}
324 
325 		Iterator entries = propertyresults.entrySet().iterator();
326 		while ( entries.hasNext() ) {
327 			Map.Entry entry = (Map.Entry) entries.next();
328 			if (entry.getValue() instanceof ArrayList) {
329 				ArrayList list = (ArrayList) entry.getValue();
330 				entry.setValue( list.toArray( new String[ list.size() ] ) );
331 			}
332 		}
333 		return propertyresults.isEmpty() ? Collections.EMPTY_MAP : propertyresults;
334 	}

336 	private static int getIndexOfFirstMatchingProperty(List propertyNames, String follower) {
337 		int propertySize = propertyNames.size();
338 		for (int propIndex = 0 ; propIndex < propertySize ; propIndex++) {
339 			if ( ( (String) propertyNames.get(propIndex) ).startsWith( follower ) ) {
340 				return propIndex;
341 			}
342 		}
343 		return -1;
344 	}

346 	private static ArrayList getResultColumns(Element propertyresult) {
347 		String column = unquote(propertyresult.attributeValue("column"));
348 		ArrayList allResultColumns = new ArrayList();
349 		if(column!=null) allResultColumns.add(column);
350 		Iterator resultColumns = propertyresult.elementIterator("return-column");
351 		while ( resultColumns.hasNext() ) {
352 			Element element = (Element) resultColumns.next();
353 			allResultColumns.add( unquote(element.attributeValue("name")) );
354 		}
355 		return allResultColumns;
356 	}

358 	private static String unquote(String name) {
359 		if (name!=null && name.charAt(0)=='`') {
360 			name=name.substring( 1, name.length()-1 );
361 		}
362 		return name;
363 	}

365 	private static LockMode getLockMode(String lockMode) {
366 		if ( lockMode == null || "read".equals( lockMode ) ) {
367 			return LockMode.READ;
368 		}
369 		else if ( "none".equals( lockMode ) ) {
370 			return LockMode.NONE;
371 		}
372 		else if ( "upgrade".equals( lockMode ) ) {
373 			return LockMode.UPGRADE;
374 		}
375 		else if ( "upgrade-nowait".equals( lockMode ) ) {
376 			return LockMode.UPGRADE_NOWAIT;
377 		}
378 		else if ( "write".equals( lockMode ) ) {
379 			return LockMode.WRITE;
380 		}
381 		else if ( "force".equals( lockMode ) ) {
382 			return LockMode.FORCE;
383 		}
384 		else if ( "optimistic".equals( lockMode ) ) {
385 			return LockMode.OPTIMISTIC;
386 		}
387 		else if ( "optimistic_force_increment".equals( lockMode ) ) {
388 			return LockMode.OPTIMISTIC_FORCE_INCREMENT;
389 		}
390 		else if ( "pessimistic_read".equals( lockMode ) ) {
391 			return LockMode.PESSIMISTIC_READ;
392 		}
393 		else if ( "pessimistic_write".equals( lockMode ) ) {
394 			return LockMode.PESSIMISTIC_WRITE;
395 		}
396 		else if ( "pessimistic_force_increment".equals( lockMode ) ) {
397 			return LockMode.PESSIMISTIC_FORCE_INCREMENT;
398 		}
399 		else {
400 			throw new MappingException( "unknown lockmode" );
401 		}
