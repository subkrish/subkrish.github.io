48  /**
49   * Defines the default create event listener used by hibernate for creating
50   * transient entities in response to generated create events.
51   *
52   * @author Gavin King
53   */
54  public class DefaultPersistEventListener extends AbstractSaveEventListener implements PersistEventListener {
55  
56      private static final CoreMessageLogger LOG = Logger.getMessageLogger(
57  			CoreMessageLogger.class,
58  			DefaultPersistEventListener.class.getName()
59  	);
60  
61  	@Override
62      protected CascadingAction getCascadeAction() {
63  		return CascadingAction.PERSIST;
64  	}
65  
66  	@Override
67      protected Boolean getAssumedUnsaved() {
68  		return Boolean.TRUE;
69  	}
70  
71  	/**
72  	 * Handle the given create event.
73  	 *
74  	 * @param event The create event to be handled.
75  	 * @throws HibernateException
76  	 */
77  	public void onPersist(PersistEvent event) throws HibernateException {
78  		onPersist( event, new IdentityHashMap(10) );
79  	}
80  
81  	/**
82  	 * Handle the given create event.
83  	 *
84  	 * @param event The create event to be handled.
85  	 * @throws HibernateException
86  	 */
87  	public void onPersist(PersistEvent event, Map createCache) throws HibernateException {
88  		final SessionImplementor source = event.getSession();
89  		final Object object = event.getObject();
90  
91  		final Object entity;
92  		if ( object instanceof HibernateProxy ) {
93  			LazyInitializer li = ( (HibernateProxy) object ).getHibernateLazyInitializer();
94  			if ( li.isUninitialized() ) {
95  				if ( li.getSession() == source ) {
96  					return; //NOTE EARLY EXIT!
97  				}
98  				else {
99  					throw new PersistentObjectException( "uninitialized proxy passed to persist()" );
100 				}
101 			}
102 			entity = li.getImplementation();
103 		}
104 		else {
105 			entity = object;
106 		}
107 
108 		final String entityName;
109 		if ( event.getEntityName() != null ) {
110 			entityName = event.getEntityName();
111 		}
112 		else {
113 			entityName = source.bestGuessEntityName( entity );
114 			event.setEntityName( entityName );
115 		}
116 
117 		final EntityEntry entityEntry = source.getPersistenceContext().getEntry( entity );
118 		EntityState entityState = getEntityState( entity, entityName, entityEntry, source );
119 		if ( entityState == EntityState.DETACHED ) {
120 			// JPA 2, in its version of a "foreign generated", allows the id attribute value
121 			// to be manually set by the user, even though this manual value is irrelevant.
122 			// The issue is that this causes problems with the Hibernate unsaved-value strategy
123 			// which comes into play here in determining detached/transient state.
124 			//
125 			// Detect if we have this situation and if so null out the id value and calculate the
126 			// entity state again.
127 
128 			// NOTE: entityEntry must be null to get here, so we cannot use any of its values
129 			EntityPersister persister = source.getFactory().getEntityPersister( entityName );
130 			if ( ForeignGenerator.class.isInstance( persister.getIdentifierGenerator() ) ) {
131 				if ( LOG.isDebugEnabled() && persister.getIdentifier( entity, source ) != null ) {
132 					LOG.debug( "Resetting entity id attribute to null for foreign generator" );
133 				}
134 				persister.setIdentifier( entity, null, source );
135 				entityState = getEntityState( entity, entityName, entityEntry, source );
136 			}
137 		}
138 
139 		switch ( entityState ) {
140 			case DETACHED: {
141 				throw new PersistentObjectException(
142 						"detached entity passed to persist: " +
143 								getLoggableName( event.getEntityName(), entity )
144 				);
145 			}
146 			case PERSISTENT: {
147 				entityIsPersistent( event, createCache );
148 				break;
149 			}
150 			case TRANSIENT: {
151 				entityIsTransient( event, createCache );
152 				break;
153 			}
154 			case DELETED: {
155 				entityEntry.setStatus( Status.MANAGED );
156 				entityEntry.setDeletedState( null );
157 				event.getSession().getActionQueue().unScheduleDeletion( entityEntry, event.getObject() );
158 				entityIsDeleted( event, createCache );
159 				break;
160 			}
161 			default: {
162 				throw new ObjectDeletedException(
163 						"deleted entity passed to persist",
164 						null,
165 						getLoggableName( event.getEntityName(), entity )
166 				);
167 			}
168 		}
169 
170 	}
171 
172 	@SuppressWarnings( {"unchecked"})
173 	protected void entityIsPersistent(PersistEvent event, Map createCache) {
174 		LOG.trace( "Ignoring persistent instance" );
175 		final EventSource source = event.getSession();
176 
177 		//TODO: check that entry.getIdentifier().equals(requestedId)
178 
179 		final Object entity = source.getPersistenceContext().unproxy( event.getObject() );
180 		final EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
181 
182 		if ( createCache.put(entity, entity)==null ) {
183 			justCascade( createCache, source, entity, persister );
184 
185 		}
186 	}
187 
188 	private void justCascade(Map createCache, EventSource source, Object entity, EntityPersister persister) {
189 		//TODO: merge into one method!
190 		cascadeBeforeSave(source, persister, entity, createCache);
191 		cascadeAfterSave(source, persister, entity, createCache);
192 	}
193 
194 	/**
195 	 * Handle the given create event.
196 	 *
197 	 * @param event The save event to be handled.
198 	 * @param createCache The copy cache of entity instance to merge/copy instance.
199 	 */
200 	@SuppressWarnings( {"unchecked"})
201 	protected void entityIsTransient(PersistEvent event, Map createCache) {
202 		LOG.trace( "Saving transient instance" );
203 
204 		final EventSource source = event.getSession();
205 		final Object entity = source.getPersistenceContext().unproxy( event.getObject() );
206 
207 		if ( createCache.put( entity, entity ) == null ) {
208 			saveWithGeneratedId( entity, event.getEntityName(), createCache, source, false );
209 		}
210 	}
211 
212 	@SuppressWarnings( {"unchecked"})
213 	private void entityIsDeleted(PersistEvent event, Map createCache) {
214 		final EventSource source = event.getSession();
215 
216 		final Object entity = source.getPersistenceContext().unproxy( event.getObject() );
217 		final EntityPersister persister = source.getEntityPersister( event.getEntityName(), entity );
218 
219 		LOG.tracef(
220 				"un-scheduling entity deletion [%s]",
221 				MessageHelper.infoString(
222 						persister,
223 						persister.getIdentifier( entity, source ),
224 						source.getFactory()
225 				)
226 		);
227 
228 		if ( createCache.put( entity, entity ) == null ) {
229 			justCascade( createCache, source, entity, persister );
230 		}
231 	}
