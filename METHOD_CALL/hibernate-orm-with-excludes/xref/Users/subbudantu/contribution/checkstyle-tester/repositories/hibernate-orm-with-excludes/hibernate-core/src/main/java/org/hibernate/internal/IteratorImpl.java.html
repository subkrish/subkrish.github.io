25  package org.hibernate.internal;
26  import java.sql.PreparedStatement;
27  import java.sql.ResultSet;
28  import java.sql.SQLException;
29  import java.util.NoSuchElementException;
30  
31  import org.jboss.logging.Logger;
32  
33  import org.hibernate.HibernateException;
34  import org.hibernate.JDBCException;
35  import org.hibernate.engine.HibernateIterator;
36  import org.hibernate.event.spi.EventSource;
37  import org.hibernate.hql.internal.HolderInstantiator;
38  import org.hibernate.type.EntityType;
39  import org.hibernate.type.Type;
40  
41  /**
42   * An implementation of <tt>java.util.Iterator</tt> that is
43   * returned by <tt>iterate()</tt> query execution methods.
44   * @author Gavin King
45   */
46  public final class IteratorImpl implements HibernateIterator {
47  
48      private static final CoreMessageLogger LOG = Logger.getMessageLogger(CoreMessageLogger.class, IteratorImpl.class.getName());
49  
50  	private ResultSet rs;
51  	private final EventSource session;
52  	private boolean readOnly;
53  	private final Type[] types;
54  	private final boolean single;
55  	private Object currentResult;
56  	private boolean hasNext;
57  	private final String[][] names;
58  	private PreparedStatement ps;
59  	private HolderInstantiator holderInstantiator;
60  
61  	public IteratorImpl(
62  	        ResultSet rs,
63  	        PreparedStatement ps,
64  	        EventSource sess,
65  	        boolean readOnly,
66  	        Type[] types,
67  	        String[][] columnNames,
68  	        HolderInstantiator holderInstantiator)
69  	throws HibernateException, SQLException {
70  
71  		this.rs=rs;
72  		this.ps=ps;
73  		this.session = sess;
74  		this.readOnly = readOnly;
75  		this.types = types;
76  		this.names = columnNames;
77  		this.holderInstantiator = holderInstantiator;
78  
79  		single = types.length==1;
80  
81  		postNext();
82  	}
83  
84  	public void close() throws JDBCException {
85  		if (ps!=null) {
86  			LOG.debug("Closing iterator");
87  			session.getTransactionCoordinator().getJdbcCoordinator().release( ps );
88  			ps = null;
89  			rs = null;
90  			hasNext = false;
91  			try {
92  				session.getPersistenceContext().getLoadContexts().cleanup( rs );
93  			}
94  			catch( Throwable ignore ) {
95  				// ignore this error for now
96                  LOG.debugf("Exception trying to cleanup load context : %s", ignore.getMessage());
97  			}
98  		}
99  	}
100 
101 	private void postNext() throws SQLException {
102 		LOG.debug("Attempting to retrieve next results");
103 		this.hasNext = rs.next();
104 		if (!hasNext) {
105 			LOG.debug("Exhausted results");
106 			close();
107 		} else LOG.debug("Retrieved next results");
108 	}
109 
110 	public boolean hasNext() {
111 		return hasNext;
112 	}
113 
114 	public Object next() throws HibernateException {
115 		if ( !hasNext ) throw new NoSuchElementException("No more results");
116 		boolean sessionDefaultReadOnlyOrig = session.isDefaultReadOnly();
117 		session.setDefaultReadOnly( readOnly );
118 		try {
119 			boolean isHolder = holderInstantiator.isRequired();
120 
121 			LOG.debugf( "Assembling results" );
122 			if ( single && !isHolder ) {
123 				currentResult = types[0].nullSafeGet( rs, names[0], session, null );
124 			}
125 			else {
126 				Object[] currentResults = new Object[types.length];
127 				for (int i=0; i<types.length; i++) {
128 					currentResults[i] = types[i].nullSafeGet( rs, names[i], session, null );
129 				}
130 
131 				if (isHolder) {
132 					currentResult = holderInstantiator.instantiate(currentResults);
133 				}
134 				else {
135 					currentResult = currentResults;
136 				}
137 			}
138 
139 			postNext();
140 			LOG.debugf( "Returning current results" );
141 			return currentResult;
142 		}
143 		catch (SQLException sqle) {
144 			throw session.getFactory().getSQLExceptionHelper().convert(
145 					sqle,
146 					"could not get next iterator result"
147 				);
148 		}
149 		finally {
150 			session.setDefaultReadOnly( sessionDefaultReadOnlyOrig );
151 		}
152 	}
153 
154 	public void remove() {
155 		if (!single) {
156 			throw new UnsupportedOperationException("Not a single column hibernate query result set");
157 		}
158 		if (currentResult==null) {
159 			throw new IllegalStateException("Called Iterator.remove() before next()");
160 		}
161 		if ( !( types[0] instanceof EntityType ) ) {
162 			throw new UnsupportedOperationException("Not an entity");
163 		}
164 
165 		session.delete(
166 				( (EntityType) types[0] ).getAssociatedEntityName(),
167 				currentResult,
168 				false,
169 		        null
170 			);
171 	}
172 }
