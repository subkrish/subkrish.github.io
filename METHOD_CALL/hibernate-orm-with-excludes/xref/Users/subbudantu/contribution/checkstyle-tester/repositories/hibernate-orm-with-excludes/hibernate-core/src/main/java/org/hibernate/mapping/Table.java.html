109 	public String getQualifiedName(Dialect dialect, String defaultCatalog, String defaultSchema) {
110 		if ( subselect != null ) {
111 			return "( " + subselect + " )";
112 		}
113 		String quotedName = getQuotedName( dialect );
114 		String usedSchema = schema == null ?
115 				defaultSchema :
116 				getQuotedSchema( dialect );
117 		String usedCatalog = catalog == null ?
118 				defaultCatalog :
119 				getQuotedCatalog( dialect );
120 		return qualify( usedCatalog, usedSchema, quotedName );
121 	}

254 	private void cleanseUniqueKeyMapIfNeeded() {
255 		if ( uniqueKeys.size() == sizeOfUniqueKeyMapOnLastCleanse ) {
256 			// nothing to do
257 			return;
258 		}
259 		cleanseUniqueKeyMap();
260 		sizeOfUniqueKeyMapOnLastCleanse = uniqueKeys.size();
261 	}

263 	private void cleanseUniqueKeyMap() {
264 		// We need to account for a few conditions here...
265 		// 	1) If there are multiple unique keys contained in the uniqueKeys Map, we need to deduplicate
266 		// 		any sharing the same columns as other defined unique keys; this is needed for the annotation
267 		// 		processor since it creates unique constraints automagically for the user
268 		//	2) Remove any unique keys that share the same columns as the primary key; again, this is
269 		//		needed for the annotation processor to handle @Id @OneToOne cases.  In such cases the
270 		//		unique key is unnecessary because a primary key is already unique by definition.  We handle
271 		//		this case specifically because some databases fail if you try to apply a unique key to
272 		//		the primary key columns which causes schema export to fail in these cases.
273 		if ( uniqueKeys.isEmpty() ) {
274 			// nothing to do
275 			return;
276 		}
277 		else if ( uniqueKeys.size() == 1 ) {
278 			// we have to worry about condition 2 above, but not condition 1
279 			final Map.Entry<String,UniqueKey> uniqueKeyEntry = uniqueKeys.entrySet().iterator().next();
280 			if ( isSameAsPrimaryKeyColumns( uniqueKeyEntry.getValue() ) ) {
281 				uniqueKeys.remove( uniqueKeyEntry.getKey() );
282 			}
283 		}
284 		else {
285 			// we have to check both conditions 1 and 2
286 			final Iterator<Map.Entry<String,UniqueKey>> uniqueKeyEntries = uniqueKeys.entrySet().iterator();
287 			while ( uniqueKeyEntries.hasNext() ) {
288 				final Map.Entry<String,UniqueKey> uniqueKeyEntry = uniqueKeyEntries.next();
289 				final UniqueKey uniqueKey = uniqueKeyEntry.getValue();
290 				boolean removeIt = false;
291 
292 				// condition 1 : check against other unique keys
293 				for ( UniqueKey otherUniqueKey : uniqueKeys.values() ) {
294 					// make sure its not the same unique key
295 					if ( uniqueKeyEntry.getValue() == otherUniqueKey ) {
296 						continue;
297 					}
298 					if ( otherUniqueKey.getColumns().containsAll( uniqueKey.getColumns() )
299 							&& uniqueKey.getColumns().containsAll( otherUniqueKey.getColumns() ) ) {
300 						removeIt = true;
301 						break;
302 					}
303 				}
304 
305 				// condition 2 : check against pk
306 				if ( isSameAsPrimaryKeyColumns( uniqueKeyEntry.getValue() ) ) {
307 					removeIt = true;
308 				}
309 
310 				if ( removeIt ) {
311 					//uniqueKeys.remove( uniqueKeyEntry.getKey() );
312 					uniqueKeyEntries.remove();
313 				}
314 			}
315 
316 		}
317 	}

319 	private boolean isSameAsPrimaryKeyColumns(UniqueKey uniqueKey) {
320 		if ( primaryKey == null || ! primaryKey.columnIterator().hasNext() ) {
321 			// happens for many-to-many tables
322 			return false;
323 		}
324 		return primaryKey.getColumns().containsAll( uniqueKey.getColumns() )
325 				&& uniqueKey.getColumns().containsAll( primaryKey.getColumns() );
326 	}

329 	public int hashCode() {
330 		final int prime = 31;
331 		int result = 1;
332 		result = prime * result
333 			+ ((catalog == null) ? 0 : isCatalogQuoted() ? catalog.hashCode() : catalog.toLowerCase().hashCode());
334 		result = prime * result + ((name == null) ? 0 : isQuoted() ? name.hashCode() : name.toLowerCase().hashCode());
335 		result = prime * result
336 			+ ((schema == null) ? 0 : isSchemaQuoted() ? schema.hashCode() : schema.toLowerCase().hashCode());
337 		return result;
338 	}

345 	public boolean equals(Table table) {
346 		if (null == table) {
347 			return false;
348 		}
349 		if (this == table) {
350 			return true;
351 		}
352 
353 		return isQuoted() ? name.equals(table.getName()) : name.equalsIgnoreCase(table.getName())
354 			&& ((schema == null && table.getSchema() != null) ? false : (schema == null) ? true : isSchemaQuoted() ? schema.equals(table.getSchema()) : schema.equalsIgnoreCase(table.getSchema()))
355 			&& ((catalog == null && table.getCatalog() != null) ? false : (catalog == null) ? true : isCatalogQuoted() ? catalog.equals(table.getCatalog()) : catalog.equalsIgnoreCase(table.getCatalog()));
356 	}

358 	public void validateColumns(Dialect dialect, Mapping mapping, TableMetadata tableInfo) {
359 		Iterator iter = getColumnIterator();
360 		while ( iter.hasNext() ) {
361 			Column col = (Column) iter.next();
362 
363 			ColumnMetadata columnInfo = tableInfo.getColumnMetadata( col.getName() );
364 
365 			if ( columnInfo == null ) {
366 				throw new HibernateException( "Missing column: " + col.getName() + " in " + Table.qualify( tableInfo.getCatalog(), tableInfo.getSchema(), tableInfo.getName()));
367 			}
368 			else {
369 				final boolean typesMatch = col.getSqlType( dialect, mapping ).toLowerCase()
370 						.startsWith( columnInfo.getTypeName().toLowerCase() )
371 						|| columnInfo.getTypeCode() == col.getSqlTypeCode( mapping );
372 				if ( !typesMatch ) {
373 					throw new HibernateException(
374 							"Wrong column type in " +
375 							Table.qualify( tableInfo.getCatalog(), tableInfo.getSchema(), tableInfo.getName()) +
376 							" for column " + col.getName() +
377 							". Found: " + columnInfo.getTypeName().toLowerCase() +
378 							", expected: " + col.getSqlType( dialect, mapping )
379 					);
380 				}
381 			}
382 		}

386 	public Iterator sqlAlterStrings(Dialect dialect, Mapping p, TableMetadata tableInfo, String defaultCatalog,
387 									String defaultSchema)
388 			throws HibernateException {
389 
390 		StringBuilder root = new StringBuilder( "alter table " )
391 				.append( getQualifiedName( dialect, defaultCatalog, defaultSchema ) )
392 				.append( ' ' )
393 				.append( dialect.getAddColumnString() );
394 
395 		Iterator iter = getColumnIterator();
396 		List results = new ArrayList();
397 		
398 		while ( iter.hasNext() ) {
399 			Column column = (Column) iter.next();
400 
401 			ColumnMetadata columnInfo = tableInfo.getColumnMetadata( column.getName() );
402 
403 			if ( columnInfo == null ) {
404 				// the column doesnt exist at all.
405 				StringBuilder alter = new StringBuilder( root.toString() )
406 						.append( ' ' )
407 						.append( column.getQuotedName( dialect ) )
408 						.append( ' ' )
409 						.append( column.getSqlType( dialect, p ) );
410 
411 				String defaultValue = column.getDefaultValue();
412 				if ( defaultValue != null ) {
413 					alter.append( " default " ).append( defaultValue );
414 				}
415 
416 				if ( column.isNullable() ) {
417 					alter.append( dialect.getNullColumnString() );
418 				}
419 				else {
420 					alter.append( " not null" );
421 				}
422 
423 				if ( column.isUnique() ) {
424 					String keyName = Constraint.generateName( "UK_", this, column );
425 					UniqueKey uk = getOrCreateUniqueKey( keyName );
426 					uk.addColumn( column );
427 					alter.append( dialect.getUniqueDelegate()
428 							.applyUniqueToColumn( column ) );
429 				}
430 
431 				if ( column.hasCheckConstraint() && dialect.supportsColumnCheck() ) {
432 					alter.append( " check(" )
433 							.append( column.getCheckConstraint() )
434 							.append( ")" );
435 				}
436 
437 				String columnComment = column.getComment();
438 				if ( columnComment != null ) {
439 					alter.append( dialect.getColumnComment( columnComment ) );
440 				}
441 
442 				results.add( alter.toString() );
443 			}
444 
445 		}
446 
447 		return results.iterator();
448 	}

455 		StringBuilder buffer = new StringBuilder( dialect.getCreateTemporaryTableString() )
456 				.append( ' ' )
457 				.append( name )

465 				buffer.append( dialect.getNullColumnString() );
466 			}

468 				buffer.append( " not null" );

472 			}
473 		}
474 		buffer.append( ") " );
475 		buffer.append( dialect.getCreateTemporaryTablePostfix() );

479 	public String sqlCreateString(Dialect dialect, Mapping p, String defaultCatalog, String defaultSchema) {
480 		StringBuilder buf = new StringBuilder( hasPrimaryKey() ? dialect.getCreateTableString() : dialect.getCreateMultisetTableString() )
481 				.append( ' ' )
482 				.append( getQualifiedName( dialect, defaultCatalog, defaultSchema ) )
483 				.append( " (" );
484 
485 		boolean identityColumn = idValue != null && idValue.isIdentityColumn( p.getIdentifierGeneratorFactory(), dialect );
486 
487 		// Try to find out the name of the primary key to create it as identity if the IdentityGenerator is used
488 		String pkname = null;
489 		if ( hasPrimaryKey() && identityColumn ) {
490 			pkname = ( (Column) getPrimaryKey().getColumnIterator().next() ).getQuotedName( dialect );
491 		}
492 
493 		Iterator iter = getColumnIterator();
494 		while ( iter.hasNext() ) {
495 			Column col = (Column) iter.next();
496 
497 			buf.append( col.getQuotedName( dialect ) )
498 					.append( ' ' );
499 
500 			if ( identityColumn && col.getQuotedName( dialect ).equals( pkname ) ) {
501 				// to support dialects that have their own identity data type
502 				if ( dialect.hasDataTypeInIdentityColumn() ) {
503 					buf.append( col.getSqlType( dialect, p ) );
504 				}
505 				buf.append( ' ' )
506 						.append( dialect.getIdentityColumnString( col.getSqlTypeCode( p ) ) );
507 			}
508 			else {
509 
510 				buf.append( col.getSqlType( dialect, p ) );
511 
512 				String defaultValue = col.getDefaultValue();
513 				if ( defaultValue != null ) {
514 					buf.append( " default " ).append( defaultValue );
515 				}
516 
517 				if ( col.isNullable() ) {
518 					buf.append( dialect.getNullColumnString() );
519 				}
520 				else {
521 					buf.append( " not null" );
522 				}
523 
524 			}
525 			
526 			if ( col.isUnique() ) {
527 				String keyName = Constraint.generateName( "UK_", this, col );
528 				UniqueKey uk = getOrCreateUniqueKey( keyName );
529 				uk.addColumn( col );
530 				buf.append( dialect.getUniqueDelegate()
531 						.applyUniqueToColumn( col ) );
532 			}
533 				
534 			if ( col.hasCheckConstraint() && dialect.supportsColumnCheck() ) {
535 				buf.append( " check (" )
536 						.append( col.getCheckConstraint() )
537 						.append( ")" );
538 			}
539 
540 			String columnComment = col.getComment();
541 			if ( columnComment != null ) {
542 				buf.append( dialect.getColumnComment( columnComment ) );
543 			}
544 
545 			if ( iter.hasNext() ) {
546 				buf.append( ", " );
547 			}
548 
549 		}
550 		if ( hasPrimaryKey() ) {
551 			buf.append( ", " )
552 					.append( getPrimaryKey().sqlConstraintString( dialect ) );
553 		}
554 
555 		buf.append( dialect.getUniqueDelegate().applyUniquesToTable( this ) );
556 
557 		if ( dialect.supportsTableCheck() ) {
558 			Iterator chiter = checkConstraints.iterator();
559 			while ( chiter.hasNext() ) {
560 				buf.append( ", check (" )
561 						.append( chiter.next() )
562 						.append( ')' );
563 			}
564 		}
565 
566 		buf.append( ')' );
567 
568 		if ( comment != null ) {
569 			buf.append( dialect.getTableComment( comment ) );
570 		}
571 
572 		return buf.append( dialect.getTableTypeString() ).toString();
573 	}

653 	public ForeignKey createForeignKey(String keyName, List keyColumns, String referencedEntityName,
654 									   List referencedColumns) {
655 		Object key = new ForeignKeyKey( keyColumns, referencedEntityName, referencedColumns );
656 
657 		ForeignKey fk = (ForeignKey) foreignKeys.get( key );
658 		if ( fk == null ) {
659 			fk = new ForeignKey();
660 			fk.setTable( this );
661 			fk.setReferencedEntityName( referencedEntityName );
662 			fk.addColumns( keyColumns.iterator() );
663 			if ( referencedColumns != null ) {
664 				fk.addReferencedColumns( referencedColumns.iterator() );
665 			}
666 			
667 			if ( keyName != null ) {
668 				fk.setName( keyName );
669 			}
670 			else {
671 				fk.setName( Constraint.generateName( fk.generatedConstraintNamePrefix(),
672 						this, keyColumns ) );
673 			}
674 			
675 			foreignKeys.put( key, fk );
676 		}
677 
678 		if ( keyName != null ) {
679 			fk.setName( keyName );
680 		}
681 
682 		return fk;
683 	}

823 	public Iterator sqlCommentStrings(Dialect dialect, String defaultCatalog, String defaultSchema) {
824 		List comments = new ArrayList();
825 		if ( dialect.supportsCommentOn() ) {
826 			String tableName = getQualifiedName( dialect, defaultCatalog, defaultSchema );
827 			if ( comment != null ) {
828 				StringBuilder buf = new StringBuilder()
829 						.append( "comment on table " )
830 						.append( tableName )
831 						.append( " is '" )
832 						.append( comment )
833 						.append( "'" );
834 				comments.add( buf.toString() );
835 			}
836 			Iterator iter = getColumnIterator();
837 			while ( iter.hasNext() ) {
838 				Column column = (Column) iter.next();
839 				String columnComment = column.getComment();
840 				if ( columnComment != null ) {
841 					StringBuilder buf = new StringBuilder()
842 							.append( "comment on column " )
843 							.append( tableName )
844 							.append( '.' )
845 							.append( column.getQuotedName( dialect ) )
846 							.append( " is '" )
847 							.append( columnComment )
848 							.append( "'" );
849 					comments.add( buf.toString() );
850 				}
851 			}
852 		}
853 		return comments.iterator();
854 	}
