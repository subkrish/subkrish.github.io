142 	public ConfiguredClass(
143 			ClassInfo classInfo,
144 			AccessType defaultAccessType,
145 			ConfiguredClass parent,
146 			AnnotationBindingContext context) {
147 		this.parent = parent;
148 		this.classInfo = classInfo;
149 		this.clazz = context.locateClassByName( classInfo.toString() );
150 		this.configuredClassType = determineType();
151 		this.classAccessType = determineClassAccessType( defaultAccessType );
152 		this.customTuplizer = determineCustomTuplizer();
153 
154 		this.simpleAttributeMap = new TreeMap<String, BasicAttribute>();
155 		this.idAttributeMap = new TreeMap<String, BasicAttribute>();
156 		this.associationAttributeMap = new TreeMap<String, AssociationAttribute>();
157 
158 		this.localBindingContext = new EntityBindingContext( context, this );
159 
160 		collectAttributes();
161 		attributeOverrideMap = Collections.unmodifiableMap( findAttributeOverrides() );
162 	}

225 	private ConfiguredClassType determineType() {
226 		if ( classInfo.annotations().containsKey( JPADotNames.ENTITY ) ) {
227 			return ConfiguredClassType.ENTITY;
228 		}
229 		else if ( classInfo.annotations().containsKey( JPADotNames.MAPPED_SUPERCLASS ) ) {
230 			return ConfiguredClassType.MAPPED_SUPERCLASS;
231 		}
232 		else if ( classInfo.annotations().containsKey( JPADotNames.EMBEDDABLE ) ) {
233 			return ConfiguredClassType.EMBEDDABLE;
234 		}
235 		else {
236 			return ConfiguredClassType.NON_ENTITY;
237 		}
238 	}

240 	private AccessType determineClassAccessType(AccessType defaultAccessType) {
241 		// default to the hierarchy access type to start with
242 		AccessType accessType = defaultAccessType;
243 
244 		AnnotationInstance accessAnnotation = JandexHelper.getSingleAnnotation( classInfo, JPADotNames.ACCESS );
245 		if ( accessAnnotation != null && accessAnnotation.target().getClass().equals( ClassInfo.class ) ) {
246 			accessType = JandexHelper.getEnumValue( accessAnnotation, "value", AccessType.class );
247 		}
248 
249 		return accessType;
250 	}

252 	/**
253 	 * Find all attributes for this configured class and add them to the corresponding map
254 	 */
255 	private void collectAttributes() {
256 		// find transient field and method names
257 		findTransientFieldAndMethodNames();
258 
259 		// use the class mate library to generic types
260 		ResolvedTypeWithMembers resolvedType = localBindingContext.resolveMemberTypes(
261 				localBindingContext.getResolvedType(
262 						clazz
263 				)
264 		);
265 		for ( HierarchicType hierarchicType : resolvedType.allTypesAndOverrides() ) {
266 			if ( hierarchicType.getType().getErasedType().equals( clazz ) ) {
267 				resolvedType = localBindingContext.resolveMemberTypes( hierarchicType.getType() );
268 				break;
269 			}
270 		}
271 
272 		if ( resolvedType == null ) {
273 			throw new AssertionFailure( "Unable to resolve types for " + clazz.getName() );
274 		}
275 
276 		Set<String> explicitlyConfiguredMemberNames = createExplicitlyConfiguredAccessProperties( resolvedType );
277 
278 		if ( AccessType.FIELD.equals( classAccessType ) ) {
279 			Field fields[] = clazz.getDeclaredFields();
280 			Field.setAccessible( fields, true );
281 			for ( Field field : fields ) {
282 				if ( isPersistentMember( transientFieldNames, explicitlyConfiguredMemberNames, field ) ) {
283 					createMappedAttribute( field, resolvedType, AccessType.FIELD );
284 				}
285 			}
286 		}
287 		else {
288 			Method[] methods = clazz.getDeclaredMethods();
289 			Method.setAccessible( methods, true );
290 			for ( Method method : methods ) {
291 				if ( isPersistentMember( transientMethodNames, explicitlyConfiguredMemberNames, method ) ) {
292 					createMappedAttribute( method, resolvedType, AccessType.PROPERTY );
293 				}
294 			}
295 		}
296 	}

298 	private boolean isPersistentMember(Set<String> transientNames, Set<String> explicitlyConfiguredMemberNames, Member member) {
299 		if ( !ReflectionHelper.isProperty( member ) ) {
300 			return false;
301 		}
302 
303 		if ( transientNames.contains( member.getName() ) ) {
304 			return false;
305 		}
306 
307 		if ( explicitlyConfiguredMemberNames.contains( ReflectionHelper.getPropertyName( member ) ) ) {
308 			return false;
309 		}
310 
311 		return true;
312 	}

314 	/**
315 	 * Creates {@code MappedProperty} instances for the explicitly configured persistent properties
316 	 *
317 	 * @param resolvedMembers the resolved type parameters for this class
318 	 *
319 	 * @return the property names of the explicitly configured attribute names in a set
320 	 */
321 	private Set<String> createExplicitlyConfiguredAccessProperties(ResolvedTypeWithMembers resolvedMembers) {
322 		Set<String> explicitAccessPropertyNames = new HashSet<String>();
323 
324 		List<AnnotationInstance> accessAnnotations = classInfo.annotations().get( JPADotNames.ACCESS );
325 		if ( accessAnnotations == null ) {
326 			return explicitAccessPropertyNames;
327 		}
328 
329 		// iterate over all @Access annotations defined on the current class
330 		for ( AnnotationInstance accessAnnotation : accessAnnotations ) {
331 			// we are only interested at annotations defined on fields and methods
332 			AnnotationTarget annotationTarget = accessAnnotation.target();
333 			if ( !( annotationTarget.getClass().equals( MethodInfo.class ) || annotationTarget.getClass()
334 					.equals( FieldInfo.class ) ) ) {
335 				continue;
336 			}
337 
338 			AccessType accessType = JandexHelper.getEnumValue( accessAnnotation, "value", AccessType.class );
339 
340 			if ( !isExplicitAttributeAccessAnnotationPlacedCorrectly( annotationTarget, accessType ) ) {
341 				continue;
342 			}
343 
344 			// the placement is correct, get the member
345 			Member member;
346 			if ( annotationTarget instanceof MethodInfo ) {
347 				Method m;
348 				try {
349 					m = clazz.getMethod( ( (MethodInfo) annotationTarget ).name() );
350 				}
351 				catch ( NoSuchMethodException e ) {
352 					throw new HibernateException(
353 							"Unable to load method "
354 									+ ( (MethodInfo) annotationTarget ).name()
355 									+ " of class " + clazz.getName()
356 					);
357 				}
358 				member = m;
359 				accessType = AccessType.PROPERTY;
360 			}
361 			else {
362 				Field f;
363 				try {
364 					f = clazz.getField( ( (FieldInfo) annotationTarget ).name() );
365 				}
366 				catch ( NoSuchFieldException e ) {
367 					throw new HibernateException(
368 							"Unable to load field "
369 									+ ( (FieldInfo) annotationTarget ).name()
370 									+ " of class " + clazz.getName()
371 					);
372 				}
373 				member = f;
374 				accessType = AccessType.FIELD;
375 			}
376 			if ( ReflectionHelper.isProperty( member ) ) {
377 				createMappedAttribute( member, resolvedMembers, accessType );
378 				explicitAccessPropertyNames.add( ReflectionHelper.getPropertyName( member ) );
379 			}
380 		}
381 		return explicitAccessPropertyNames;
382 	}

384 	private boolean isExplicitAttributeAccessAnnotationPlacedCorrectly(AnnotationTarget annotationTarget, AccessType accessType) {
385 		// when the access type of the class is FIELD
386 		// overriding access annotations must be placed on properties AND have the access type PROPERTY
387 		if ( AccessType.FIELD.equals( classAccessType ) ) {
388 			if ( !( annotationTarget instanceof MethodInfo ) ) {
389 				LOG.tracef(
390 						"The access type of class %s is AccessType.FIELD. To override the access for an attribute " +
391 								"@Access has to be placed on the property (getter)", classInfo.name().toString()
392 				);
393 				return false;
394 			}
395 
396 			if ( !AccessType.PROPERTY.equals( accessType ) ) {
397 				LOG.tracef(
398 						"The access type of class %s is AccessType.FIELD. To override the access for an attribute " +
399 								"@Access has to be placed on the property (getter) with an access type of AccessType.PROPERTY. " +
400 								"Using AccessType.FIELD on the property has no effect",
401 						classInfo.name().toString()
402 				);
403 				return false;
404 			}
405 		}
406 
407 		// when the access type of the class is PROPERTY
408 		// overriding access annotations must be placed on fields and have the access type FIELD
409 		if ( AccessType.PROPERTY.equals( classAccessType ) ) {
410 			if ( !( annotationTarget instanceof FieldInfo ) ) {
411 				LOG.tracef(
412 						"The access type of class %s is AccessType.PROPERTY. To override the access for a field " +
413 								"@Access has to be placed on the field ", classInfo.name().toString()
414 				);
415 				return false;
416 			}
417 
418 			if ( !AccessType.FIELD.equals( accessType ) ) {
419 				LOG.tracef(
420 						"The access type of class %s is AccessType.PROPERTY. To override the access for a field " +
421 								"@Access has to be placed on the field with an access type of AccessType.FIELD. " +
422 								"Using AccessType.PROPERTY on the field has no effect",
423 						classInfo.name().toString()
424 				);
425 				return false;
426 			}
427 		}
428 		return true;
429 	}

431 	private void createMappedAttribute(Member member, ResolvedTypeWithMembers resolvedType, AccessType accessType) {
432 		final String attributeName = ReflectionHelper.getPropertyName( member );
433 		ResolvedMember[] resolvedMembers;
434 		if ( member instanceof Field ) {
435 			resolvedMembers = resolvedType.getMemberFields();
436 		}
437 		else {
438 			resolvedMembers = resolvedType.getMemberMethods();
439 		}
440 		Class<?> attributeType = (Class<?>) findResolvedType( member.getName(), resolvedMembers );
441 		final Map<DotName, List<AnnotationInstance>> annotations = JandexHelper.getMemberAnnotations(
442 				classInfo, member.getName()
443 		);
444 
445 		AttributeNature attributeNature = determineAttributeNature( annotations );
446 		String accessTypeString = accessType.toString().toLowerCase();
447 		switch ( attributeNature ) {
448 			case BASIC: {
449 				BasicAttribute attribute = BasicAttribute.createSimpleAttribute(
450 						attributeName, attributeType, annotations, accessTypeString, getLocalBindingContext()
451 				);
452 				if ( attribute.isId() ) {
453 					idAttributeMap.put( attributeName, attribute );
454 				}
455 				else if ( attribute.isVersioned() ) {
456 					if ( versionAttribute == null ) {
457 						versionAttribute = attribute;
458 					}
459 					else {
460 						throw new MappingException( "Multiple version attributes", localBindingContext.getOrigin() );
461 					}
462 				}
463 				else {
464 					simpleAttributeMap.put( attributeName, attribute );
465 				}
466 				break;
467 			}
468 			case ELEMENT_COLLECTION: {
469 				throw new NotYetImplementedException( "Element collections must still be implemented." );
470 			}
471 			case EMBEDDED_ID: {
472 				throw new NotYetImplementedException( "Embedded ids must still be implemented." );
473 			}
474 			case EMBEDDED: {
475 				AnnotationInstance targetAnnotation = JandexHelper.getSingleAnnotation(
476 						getClassInfo(),
477 						HibernateDotNames.TARGET
478 				);
479 				if ( targetAnnotation != null ) {
480 					attributeType = localBindingContext.locateClassByName(
481 							JandexHelper.getValue( targetAnnotation, "value", String.class )
482 					);
483 				}
484 				resolveEmbeddable( attributeName, attributeType );
485 				break;
486 			}
487 			// OneToOne, OneToMany, ManyToOne, ManyToMany
488 			default: {
489 				AssociationAttribute attribute = AssociationAttribute.createAssociationAttribute(
490 						attributeName,
491 						attributeType,
492 						attributeNature,
493 						accessTypeString,
494 						annotations,
495 						getLocalBindingContext()
496 				);
497 				associationAttributeMap.put( attributeName, attribute );
498 			}
499 		}
500 	}

502 	private void resolveEmbeddable(String attributeName, Class<?> type) {
503 		ClassInfo embeddableClassInfo = localBindingContext.getClassInfo( type.getName() );
504 		if ( embeddableClassInfo == null ) {
505 			String msg = String.format(
506 					"Attribute '%s#%s' is annotated with @Embedded, but '%s' does not seem to be annotated " +
507 							"with @Embeddable. Are all annotated classes added to the configuration?",
508 					getConfiguredClass().getSimpleName(),
509 					attributeName,
510 					type.getSimpleName()
511 			);
512 			throw new AnnotationException( msg );
513 		}
514 
515 		localBindingContext.resolveAllTypes( type.getName() );
516 		EmbeddableHierarchy hierarchy = EmbeddableHierarchy.createEmbeddableHierarchy(
517 				localBindingContext.<Object>locateClassByName( embeddableClassInfo.toString() ),
518 				attributeName,
519 				classAccessType,
520 				localBindingContext
521 		);
522 		embeddedClasses.put( attributeName, hierarchy.getLeaf() );
523 	}

525 	/**
526 	 * Given the annotations defined on a persistent attribute this methods determines the attribute type.
527 	 *
528 	 * @param annotations the annotations defined on the persistent attribute
529 	 *
530 	 * @return an instance of the {@code AttributeType} enum
531 	 */
532 	private AttributeNature determineAttributeNature(Map<DotName, List<AnnotationInstance>> annotations) {
533 		EnumMap<AttributeNature, AnnotationInstance> discoveredAttributeTypes =
534 				new EnumMap<AttributeNature, AnnotationInstance>( AttributeNature.class );
535 
536 		AnnotationInstance oneToOne = JandexHelper.getSingleAnnotation( annotations, JPADotNames.ONE_TO_ONE );
537 		if ( oneToOne != null ) {
538 			discoveredAttributeTypes.put( AttributeNature.ONE_TO_ONE, oneToOne );
539 		}
540 
541 		AnnotationInstance oneToMany = JandexHelper.getSingleAnnotation( annotations, JPADotNames.ONE_TO_MANY );
542 		if ( oneToMany != null ) {
543 			discoveredAttributeTypes.put( AttributeNature.ONE_TO_MANY, oneToMany );
544 		}
545 
546 		AnnotationInstance manyToOne = JandexHelper.getSingleAnnotation( annotations, JPADotNames.MANY_TO_ONE );
547 		if ( manyToOne != null ) {
548 			discoveredAttributeTypes.put( AttributeNature.MANY_TO_ONE, manyToOne );
549 		}
550 
551 		AnnotationInstance manyToMany = JandexHelper.getSingleAnnotation( annotations, JPADotNames.MANY_TO_MANY );
552 		if ( manyToMany != null ) {
553 			discoveredAttributeTypes.put( AttributeNature.MANY_TO_MANY, manyToMany );
554 		}
555 
556 		AnnotationInstance embedded = JandexHelper.getSingleAnnotation( annotations, JPADotNames.EMBEDDED );
557 		if ( embedded != null ) {
558 			discoveredAttributeTypes.put( AttributeNature.EMBEDDED, embedded );
559 		}
560 
561 		AnnotationInstance embeddedId = JandexHelper.getSingleAnnotation( annotations, JPADotNames.EMBEDDED_ID );
562 		if ( embeddedId != null ) {
563 			discoveredAttributeTypes.put( AttributeNature.EMBEDDED_ID, embeddedId );
564 		}
565 
566 		AnnotationInstance elementCollection = JandexHelper.getSingleAnnotation(
567 				annotations,
568 				JPADotNames.ELEMENT_COLLECTION
569 		);
570 		if ( elementCollection != null ) {
571 			discoveredAttributeTypes.put( AttributeNature.ELEMENT_COLLECTION, elementCollection );
572 		}
573 
574 		if ( discoveredAttributeTypes.size() == 0 ) {
575 			return AttributeNature.BASIC;
576 		}
577 		else if ( discoveredAttributeTypes.size() == 1 ) {
578 			return discoveredAttributeTypes.keySet().iterator().next();
579 		}
580 		else {
581 			throw new AnnotationException( "More than one association type configured for property  " + getName() + " of class " + getName() );
582 		}
583 	}

585 	private Type findResolvedType(String name, ResolvedMember[] resolvedMembers) {
586 		for ( ResolvedMember resolvedMember : resolvedMembers ) {
587 			if ( resolvedMember.getName().equals( name ) ) {
588 				return resolvedMember.getType().getErasedType();
589 			}
590 		}
591 		throw new AssertionFailure(
592 				String.format(
593 						"Unable to resolve type of attribute %s of class %s",
594 						name,
595 						classInfo.name().toString()
596 				)
597 		);
598 	}

600 	/**
601 	 * Populates the sets of transient field and method names.
602 	 */
603 	private void findTransientFieldAndMethodNames() {
604 		List<AnnotationInstance> transientMembers = classInfo.annotations().get( JPADotNames.TRANSIENT );
605 		if ( transientMembers == null ) {
606 			return;
607 		}
608 
609 		for ( AnnotationInstance transientMember : transientMembers ) {
610 			AnnotationTarget target = transientMember.target();
611 			if ( target instanceof FieldInfo ) {
612 				transientFieldNames.add( ( (FieldInfo) target ).name() );
613 			}
614 			else {
615 				transientMethodNames.add( ( (MethodInfo) target ).name() );
616 			}
617 		}
618 	}

620 	private Map<String, AttributeOverride> findAttributeOverrides() {
621 		Map<String, AttributeOverride> attributeOverrideList = new HashMap<String, AttributeOverride>();
622 
623 		AnnotationInstance attributeOverrideAnnotation = JandexHelper.getSingleAnnotation(
624 				classInfo,
625 				JPADotNames.ATTRIBUTE_OVERRIDE
626 		);
627 		if ( attributeOverrideAnnotation != null ) {
628 			String prefix = createPathPrefix( attributeOverrideAnnotation.target() );
629 			AttributeOverride override = new AttributeOverride( prefix, attributeOverrideAnnotation );
630 			attributeOverrideList.put( override.getAttributePath(), override );
631 		}
632 
633 		AnnotationInstance attributeOverridesAnnotation = JandexHelper.getSingleAnnotation(
634 				classInfo,
635 				JPADotNames.ATTRIBUTE_OVERRIDES
636 		);
637 		if ( attributeOverridesAnnotation != null ) {
638 			AnnotationInstance[] annotationInstances = attributeOverridesAnnotation.value().asNestedArray();
639 			for ( AnnotationInstance annotationInstance : annotationInstances ) {
640 				String prefix = createPathPrefix( attributeOverridesAnnotation.target() );
641 				AttributeOverride override = new AttributeOverride( prefix, annotationInstance );
642 				attributeOverrideList.put( override.getAttributePath(), override );
643 			}
644 		}
645 		return attributeOverrideList;
646 	}

656 	private List<AnnotationInstance> findAssociationOverrides() {
657 		List<AnnotationInstance> associationOverrideList = new ArrayList<AnnotationInstance>();
658 
659 		AnnotationInstance associationOverrideAnnotation = JandexHelper.getSingleAnnotation(
660 				classInfo,
661 				JPADotNames.ASSOCIATION_OVERRIDE
662 		);
663 		if ( associationOverrideAnnotation != null ) {
664 			associationOverrideList.add( associationOverrideAnnotation );
665 		}
666 
667 		AnnotationInstance associationOverridesAnnotation = JandexHelper.getSingleAnnotation(
668 				classInfo,
669 				JPADotNames.ASSOCIATION_OVERRIDES
670 		);
671 		if ( associationOverrideAnnotation != null ) {
672 			AnnotationInstance[] attributeOverride = associationOverridesAnnotation.value().asNestedArray();
673 			Collections.addAll( associationOverrideList, attributeOverride );
674 		}
675 
676 		return associationOverrideList;
677 	}

679 	private String determineCustomTuplizer() {
680 		final AnnotationInstance tuplizersAnnotation = JandexHelper.getSingleAnnotation(
681 				classInfo, HibernateDotNames.TUPLIZERS
682 		);
683 		if ( tuplizersAnnotation == null ) {
684 			return null;
685 		}
686 
687 		AnnotationInstance[] annotations = JandexHelper.getValue(
688 				tuplizersAnnotation,
689 				"value",
690 				AnnotationInstance[].class
691 		);
692 
693 		AnnotationInstance pojoTuplizerAnnotation = null;
694 		for ( AnnotationInstance tuplizerAnnotation : annotations ) {
695 			if ( EntityMode.valueOf( tuplizerAnnotation.value( "entityModeType" ).asEnum() ) == EntityMode.POJO ) {
696 				pojoTuplizerAnnotation = tuplizerAnnotation;
697 				break;
698 			}
699 		}
700 
701 		String customTuplizer = null;
702 		if ( pojoTuplizerAnnotation != null ) {
703 			customTuplizer = pojoTuplizerAnnotation.value( "impl" ).asString();
704 		}
705 		return customTuplizer;
706 	}
