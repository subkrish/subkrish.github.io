39  /**
40   * This is how to do batch processing in Hibernate. Remember to enable JDBC batch updates, or this test will take a
41   * VeryLongTime!
42   *
43   * @author Gavin King
44   */
45  public class BatchTest extends BaseCoreFunctionalTestCase {
46  	@Override
47  	public String[] getMappings() {
48  		return new String[] { "batch/DataPoint.hbm.xml" };
49  	}
50  
51  	@Override
52  	public String getCacheConcurrencyStrategy() {
53  		return null;
54  	}
55  
56  	@Override
57  	public void configure(Configuration cfg) {
58  		cfg.setProperty( Environment.STATEMENT_BATCH_SIZE, "20" );
59  	}
60  
61  	@Test
62  	public void testBatchInsertUpdate() {
63  		long start = System.currentTimeMillis();
64  		final int N = 5000; //26 secs with batch flush, 26 without
65  		//final int N = 100000; //53 secs with batch flush, OOME without
66  		//final int N = 250000; //137 secs with batch flush, OOME without
67  		int batchSize = ( ( SessionFactoryImplementor ) sessionFactory() ).getSettings().getJdbcBatchSize();
68  		doBatchInsertUpdate( N, batchSize );
69  		System.out.println( System.currentTimeMillis() - start );
70  	}
71  
72  	@Test
73  	public void testBatchInsertUpdateSizeEqJdbcBatchSize() {
74  		int batchSize = ( ( SessionFactoryImplementor ) sessionFactory() ).getSettings().getJdbcBatchSize();
75  		doBatchInsertUpdate( 50, batchSize );
76  	}
77  
78  	@Test
79  	public void testBatchInsertUpdateSizeLtJdbcBatchSize() {
80  		int batchSize = ( ( SessionFactoryImplementor ) sessionFactory() ).getSettings().getJdbcBatchSize();
81  		doBatchInsertUpdate( 50, batchSize - 1 );
82  	}
83  
84  	@Test
85  	public void testBatchInsertUpdateSizeGtJdbcBatchSize() {
86  		long start = System.currentTimeMillis();
87  		int batchSize = ( ( SessionFactoryImplementor ) sessionFactory() ).getSettings().getJdbcBatchSize();
88  		doBatchInsertUpdate( 50, batchSize + 1 );
89  	}
90  
91  	public void doBatchInsertUpdate(int nEntities, int nBeforeFlush) {
92  		Session s = openSession();
93  		s.setCacheMode( CacheMode.IGNORE );
94  		Transaction t = s.beginTransaction();
95  		for ( int i = 0; i < nEntities; i++ ) {
96  			DataPoint dp = new DataPoint();
97  			dp.setX( new BigDecimal( i * 0.1d ).setScale( 19, BigDecimal.ROUND_DOWN ) );
98  			dp.setY( new BigDecimal( Math.cos( dp.getX().doubleValue() ) ).setScale( 19, BigDecimal.ROUND_DOWN ) );
99  			s.save( dp );
100 			if ( i + 1 % nBeforeFlush == 0 ) {
101 				s.flush();
102 				s.clear();
103 			}
104 		}
105 		t.commit();
106 		s.close();
107 
108 		s = openSession();
109 		s.setCacheMode( CacheMode.IGNORE );
110 		t = s.beginTransaction();
111 		int i = 0;
112 		ScrollableResults sr = s.createQuery( "from DataPoint dp order by dp.x asc" )
113 				.scroll( ScrollMode.FORWARD_ONLY );
114 		while ( sr.next() ) {
115 			DataPoint dp = ( DataPoint ) sr.get( 0 );
116 			dp.setDescription( "done!" );
117 			if ( ++i % nBeforeFlush == 0 ) {
118 				s.flush();
119 				s.clear();
120 			}
121 		}
122 		t.commit();
123 		s.close();
124 
125 		s = openSession();
126 		s.setCacheMode( CacheMode.IGNORE );
127 		t = s.beginTransaction();
128 		i = 0;
129 		sr = s.createQuery( "from DataPoint dp order by dp.x asc" )
130 				.scroll( ScrollMode.FORWARD_ONLY );
131 		while ( sr.next() ) {
132 			DataPoint dp = ( DataPoint ) sr.get( 0 );
133 			s.delete( dp );
134 			if ( ++i % nBeforeFlush == 0 ) {
135 				s.flush();
136 				s.clear();
137 			}
138 		}
139 		t.commit();
140 		s.close();
141 	}
142 }
