36  /**
37   * Tes for parameterized user collection types.
38   *
39   * @author Holger Brands
40   * @author Steve Ebersole
41   */
42  public abstract class ParameterizedUserCollectionTypeTest extends BaseCoreFunctionalTestCase {
43  	@SuppressWarnings( {"unchecked"})
44  	@Test
45  	public void testBasicOperation() {
46  		Session s = openSession();
47  		Transaction t = s.beginTransaction();
48  		Entity entity = new Entity( "tester" );
49  		entity.getValues().add( "value-1" );
50  		s.persist( entity );
51  		t.commit();
52  		s.close();
53  
54  		s = openSession();
55  		t = s.beginTransaction();
56  		entity = ( Entity ) s.get( Entity.class, "tester" );
57  		assertTrue( Hibernate.isInitialized( entity.getValues() ) );
58  		assertEquals( 1, entity.getValues().size() );
59          assertEquals( "Hello", ( ( DefaultableList ) entity.getValues() ).getDefaultValue() );
60  		s.delete( entity );
61  		t.commit();
62  		s.close();
63  	}
64  }
