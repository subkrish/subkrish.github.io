68  	@Override
69  	public void afterConfigurationBuilt(Mappings mappings, Dialect dialect) {
70  		super.afterConfigurationBuilt( mappings, dialect );
71  		// Oracle and Postgres do not have year() functions, so we need to
72  		// redefine the 'User.person.yob' formula
73  		//
74  		// consider temporary until we add the capability to define
75  		// mapping formulas which can use dialect-registered functions...
76  		PersistentClass user = mappings.getClass( User.class.getName() );
77  		org.hibernate.mapping.Property personProperty = user.getProperty( "person" );
78  		Component component = ( Component ) personProperty.getValue();
79  		Formula f = ( Formula ) component.getProperty( "yob" ).getValue().getColumnIterator().next();
80  
81  		SQLFunction yearFunction = ( SQLFunction ) dialect.getFunctions().get( "year" );
82  		if ( yearFunction == null ) {
83  			// the dialect not know to support a year() function, so rely on the
84  			// ANSI SQL extract function
85  			f.setFormula( "extract( year from dob )");
86  		}
87  		else {
88  			List args = new ArrayList();
89  			args.add( "dob" );
90  			f.setFormula( yearFunction.render( StandardBasicTypes.INTEGER, args, null ) );
91  		}
92  	}

238 	@Test
239 	public void testCustomColumnReadAndWrite() {
240 		Session s = openSession();
241 		Transaction t = s.beginTransaction();
242 		User u = new User( "steve", "hibernater", new Person( "Steve Ebersole", new Date(), "Main St") );
243 		final double HEIGHT_INCHES = 73;
244 		final double HEIGHT_CENTIMETERS = HEIGHT_INCHES * 2.54d;
245 		u.getPerson().setHeightInches(HEIGHT_INCHES);
246 		s.persist( u );
247 		s.flush();
248 		
249 		// Test value conversion during insert
250 		// Value returned by Oracle native query is a Types.NUMERIC, which is mapped to a BigDecimalType;
251 		// Cast returned value to Number then call Number.doubleValue() so it works on all dialects.
252 		Double heightViaSql =
253 				( (Number)s.createSQLQuery("select height_centimeters from T_USER where T_USER.username='steve'").uniqueResult())
254 						.doubleValue();
255 		assertEquals(HEIGHT_CENTIMETERS, heightViaSql, 0.01d);
256 
257 		// Test projection
258 		Double heightViaHql = (Double)s.createQuery("select u.person.heightInches from User u where u.id = 'steve'").uniqueResult();
259 		assertEquals(HEIGHT_INCHES, heightViaHql, 0.01d);
260 		
261 		// Test restriction and entity load via criteria
262 		u = (User)s.createCriteria(User.class)
263 			.add(Restrictions.between("person.heightInches", HEIGHT_INCHES - 0.01d, HEIGHT_INCHES + 0.01d))
264 			.uniqueResult();
265 		assertEquals(HEIGHT_INCHES, u.getPerson().getHeightInches(), 0.01d);
266 		
267 		// Test predicate and entity load via HQL
268 		u = (User)s.createQuery("from User u where u.person.heightInches between ? and ?")
269 			.setDouble(0, HEIGHT_INCHES - 0.01d)
270 			.setDouble(1, HEIGHT_INCHES + 0.01d)
271 			.uniqueResult();
272 		assertEquals(HEIGHT_INCHES, u.getPerson().getHeightInches(), 0.01d);
273 		
274 		// Test update
275 		u.getPerson().setHeightInches(1);
276 		s.flush();
277 		heightViaSql =
278 				( (Number)s.createSQLQuery("select height_centimeters from T_USER where T_USER.username='steve'").uniqueResult() )
279 						.doubleValue();
280 		assertEquals(2.54d, heightViaSql, 0.01d);
281 		s.delete(u);
282 		t.commit();
283 		s.close();
284 	}

299 		Session s = openSession();
300 		Transaction t = s.beginTransaction();
301 		Employee emp = new Employee();
302 		emp.setHireDate( new Date() );
303 		emp.setPerson( new Person() );
304 		emp.getPerson().setName( "steve" );
305 		emp.getPerson().setDob( new Date() );
306 		s.persist( emp );
307 		t.commit();
308 		s.close();
309 
310 		s = openSession();
311 		t = s.beginTransaction();
312 		emp = (Employee)s.get( Employee.class, emp.getId() );
313 		t.commit();
314 		s.close();
315 
316 		assertNull(emp.getOptionalComponent());
317 		emp.setOptionalComponent( new OptionalComponent() );
318 		emp.getOptionalComponent().setValue1( "emp-value1" );
319 		emp.getOptionalComponent().setValue2( "emp-value2" );
320 
321 		s = openSession();
322 		t = s.beginTransaction();
323 		emp = (Employee)s.merge( emp );
324 		t.commit();
325 		s.close();
326 
327 		s = openSession();
328 		t = s.beginTransaction();

332

334 		assertEquals("emp-value2", emp.getOptionalComponent().getValue2());

336 		emp.getOptionalComponent().setValue2( null );

343
