250 	public static CollectionBinder getCollectionBinder(
251 			String entityName,
252 			XProperty property,
253 			boolean isIndexed,
254 			boolean isHibernateExtensionMapping,
255 			Mappings mappings) {
256 		CollectionBinder result;
257 		if ( property.isArray() ) {
258 			if ( property.getElementClass().isPrimitive() ) {
259 				result = new PrimitiveArrayBinder();
260 			}
261 			else {
262 				result = new ArrayBinder();
263 			}
264 		}
265 		else if ( property.isCollection() ) {
266 			//TODO consider using an XClass
267 			Class returnedClass = property.getCollectionClass();
268 			if ( java.util.Set.class.equals( returnedClass ) ) {
269 				if ( property.isAnnotationPresent( CollectionId.class ) ) {
270 					throw new AnnotationException( "Set do not support @CollectionId: "
271 							+ StringHelper.qualify( entityName, property.getName() ) );
272 				}
273 				result = new SetBinder();
274 			}
275 			else if ( java.util.SortedSet.class.equals( returnedClass ) ) {
276 				if ( property.isAnnotationPresent( CollectionId.class ) ) {
277 					throw new AnnotationException( "Set do not support @CollectionId: "
278 							+ StringHelper.qualify( entityName, property.getName() ) );
279 				}
280 				result = new SetBinder( true );
281 			}
282 			else if ( java.util.Map.class.equals( returnedClass ) ) {
283 				if ( property.isAnnotationPresent( CollectionId.class ) ) {
284 					throw new AnnotationException( "Map do not support @CollectionId: "
285 							+ StringHelper.qualify( entityName, property.getName() ) );
286 				}
287 				result = new MapBinder();
288 			}
289 			else if ( java.util.SortedMap.class.equals( returnedClass ) ) {
290 				if ( property.isAnnotationPresent( CollectionId.class ) ) {
291 					throw new AnnotationException( "Map do not support @CollectionId: "
292 							+ StringHelper.qualify( entityName, property.getName() ) );
293 				}
294 				result = new MapBinder( true );
295 			}
296 			else if ( java.util.Collection.class.equals( returnedClass ) ) {
297 				if ( property.isAnnotationPresent( CollectionId.class ) ) {
298 					result = new IdBagBinder();
299 				}
300 				else {
301 					result = new BagBinder();
302 				}
303 			}
304 			else if ( java.util.List.class.equals( returnedClass ) ) {
305 				if ( isIndexed ) {
306 					if ( property.isAnnotationPresent( CollectionId.class ) ) {
307 						throw new AnnotationException(
308 								"List do not support @CollectionId and @OrderColumn (or @IndexColumn) at the same time: "
309 								+ StringHelper.qualify( entityName, property.getName() ) );
310 					}
311 					result = new ListBinder();
312 				}
313 				else if ( property.isAnnotationPresent( CollectionId.class ) ) {
314 					result = new IdBagBinder();
315 				}
316 				else {
317 					result = new BagBinder();
318 				}
319 			}
320 			else {
321 				throw new AnnotationException(
322 						returnedClass.getName() + " collection not yet supported: "
323 								+ StringHelper.qualify( entityName, property.getName() )
324 				);
325 			}
326 		}
327 		else {
328 			throw new AnnotationException(
329 					"Illegal attempt to map a non collection as a @OneToMany, @ManyToMany or @CollectionOfElements: "
330 							+ StringHelper.qualify( entityName, property.getName() )
331 			);
332 		}
333 		result.setIsHibernateExtensionMapping( isHibernateExtensionMapping );
334 
335 		final CollectionType typeAnnotation = property.getAnnotation( CollectionType.class );
336 		if ( typeAnnotation != null ) {
337 			final String typeName = typeAnnotation.type();
338 			// see if it names a type-def
339 			final TypeDef typeDef = mappings.getTypeDef( typeName );
340 			if ( typeDef != null ) {
341 				result.explicitType = typeDef.getTypeClass();
342 				result.explicitTypeParameters.putAll( typeDef.getParameters() );
343 			}
344 			else {
345 				result.explicitType = typeName;
346 				for ( Parameter param : typeAnnotation.parameters() ) {
347 					result.explicitTypeParameters.setProperty( param.name(), param.value() );
348 				}
349 			}
350 		}
351 
352 		return result;
353 	}

398 	public void bind() {
399 		this.collection = createCollection( propertyHolder.getPersistentClass() );
400 		String role = StringHelper.qualify( propertyHolder.getPath(), propertyName );
401 		LOG.debugf( "Collection role: %s", role );
402 		collection.setRole( role );
403 		collection.setNodeName( propertyName );
404 
405 		if ( property.isAnnotationPresent( MapKeyColumn.class )
406 			&& mapKeyPropertyName != null ) {
407 			throw new AnnotationException(
408 					"Cannot mix @javax.persistence.MapKey and @MapKeyColumn or @org.hibernate.annotations.MapKey "
409 							+ "on the same collection: " + StringHelper.qualify(
410 							propertyHolder.getPath(), propertyName
411 					)
412 			);
413 		}
414 
415 		// set explicit type information
416 		if ( explicitType != null ) {
417 			final TypeDef typeDef = mappings.getTypeDef( explicitType );
418 			if ( typeDef == null ) {
419 				collection.setTypeName( explicitType );
420 				collection.setTypeParameters( explicitTypeParameters );
421 			}
422 			else {
423 				collection.setTypeName( typeDef.getTypeClass() );
424 				collection.setTypeParameters( typeDef.getParameters() );
425 			}
426 		}
427 
428 		//set laziness
429 		defineFetchingStrategy();
430 		collection.setBatchSize( batchSize );
431 		if ( orderBy != null && hqlOrderBy != null ) {
432 			throw new AnnotationException(
433 					"Cannot use sql order by clause in conjunction of EJB3 order by clause: " + safeCollectionRole()
434 			);
435 		}
436 
437 		collection.setMutable( !property.isAnnotationPresent( Immutable.class ) );
438 
439 		//work on association
440 		boolean isMappedBy = !BinderHelper.isEmptyAnnotationValue( mappedBy );
441 
442 		final OptimisticLock lockAnn = property.getAnnotation( OptimisticLock.class );
443 		final boolean includeInOptimisticLockChecks = ( lockAnn != null )
444 				? ! lockAnn.excluded()
445 				: ! isMappedBy;
446 		collection.setOptimisticLocked( includeInOptimisticLockChecks );
447 
448 		Persister persisterAnn = property.getAnnotation( Persister.class );
449 		if ( persisterAnn != null ) {
450 			collection.setCollectionPersisterClass( persisterAnn.impl() );
451 		}
452 
453 		// set ordering
454 		if ( orderBy != null ) collection.setOrderBy( orderBy );
455 		if ( isSorted ) {
456 			collection.setSorted( true );
457 			if ( comparator != null ) {
458 				try {
459 					collection.setComparator( (Comparator) comparator.newInstance() );
460 				}
461 				catch (ClassCastException e) {
462 					throw new AnnotationException(
463 							"Comparator not implementing java.util.Comparator class: "
464 									+ comparator.getName() + "(" + safeCollectionRole() + ")"
465 					);
466 				}
467 				catch (Exception e) {
468 					throw new AnnotationException(
469 							"Could not instantiate comparator class: "
470 									+ comparator.getName() + "(" + safeCollectionRole() + ")"
471 					);
472 				}
473 			}
474 		}
475 		else {
476 			if ( hasToBeSorted ) {
477 				throw new AnnotationException(
478 						"A sorted collection has to define @Sort: "
479 								+ safeCollectionRole()
480 				);
481 			}
482 		}
483 
484 		//set cache
485 		if ( StringHelper.isNotEmpty( cacheConcurrencyStrategy ) ) {
486 			collection.setCacheConcurrencyStrategy( cacheConcurrencyStrategy );
487 			collection.setCacheRegionName( cacheRegionName );
488 		}
489 
490 		//SQL overriding
491 		SQLInsert sqlInsert = property.getAnnotation( SQLInsert.class );
492 		SQLUpdate sqlUpdate = property.getAnnotation( SQLUpdate.class );
493 		SQLDelete sqlDelete = property.getAnnotation( SQLDelete.class );
494 		SQLDeleteAll sqlDeleteAll = property.getAnnotation( SQLDeleteAll.class );
495 		Loader loader = property.getAnnotation( Loader.class );
496 		if ( sqlInsert != null ) {
497 			collection.setCustomSQLInsert( sqlInsert.sql().trim(), sqlInsert.callable(),
498 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlInsert.check().toString().toLowerCase() )
499 			);
500 
501 		}
502 		if ( sqlUpdate != null ) {
503 			collection.setCustomSQLUpdate( sqlUpdate.sql(), sqlUpdate.callable(),
504 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlUpdate.check().toString().toLowerCase() )
505 			);
506 		}
507 		if ( sqlDelete != null ) {
508 			collection.setCustomSQLDelete( sqlDelete.sql(), sqlDelete.callable(),
509 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlDelete.check().toString().toLowerCase() )
510 			);
511 		}
512 		if ( sqlDeleteAll != null ) {
513 			collection.setCustomSQLDeleteAll( sqlDeleteAll.sql(), sqlDeleteAll.callable(),
514 					ExecuteUpdateResultCheckStyle.fromExternalName( sqlDeleteAll.check().toString().toLowerCase() )
515 			);
516 		}
517 		if ( loader != null ) {
518 			collection.setLoaderName( loader.namedQuery() );
519 		}
520 
521 		if (isMappedBy
522 				&& (property.isAnnotationPresent( JoinColumn.class )
523 					|| property.isAnnotationPresent( JoinColumns.class )
524 					|| propertyHolder.getJoinTable( property ) != null ) ) {
525 			String message = "Associations marked as mappedBy must not define database mappings like @JoinTable or @JoinColumn: ";
526 			message += StringHelper.qualify( propertyHolder.getPath(), propertyName );
527 			throw new AnnotationException( message );
528 		}
529 
530 		collection.setInverse( isMappedBy );
531 
532 		//many to many may need some second pass informations
533 		if ( !oneToMany && isMappedBy ) {
534 			mappings.addMappedBy( getCollectionType().getName(), mappedBy, propertyName );
535 		}
536 		//TODO reducce tableBinder != null and oneToMany
537 		XClass collectionType = getCollectionType();
538 		if ( inheritanceStatePerClass == null) throw new AssertionFailure( "inheritanceStatePerClass not set" );
539 		SecondPass sp = getSecondPass(
540 				fkJoinColumns,
541 				joinColumns,
542 				inverseJoinColumns,
543 				elementColumns,
544 				mapKeyColumns, mapKeyManyToManyColumns, isEmbedded,
545 				property, collectionType,
546 				ignoreNotFound, oneToMany,
547 				tableBinder, mappings
548 		);
549 		if ( collectionType.isAnnotationPresent( Embeddable.class )
550 				|| property.isAnnotationPresent( ElementCollection.class ) //JPA 2
551 				) {
552 			// do it right away, otherwise @ManyToOne on composite element call addSecondPass
553 			// and raise a ConcurrentModificationException
554 			//sp.doSecondPass( CollectionHelper.EMPTY_MAP );
555 			mappings.addSecondPass( sp, !isMappedBy );
556 		}
557 		else {
558 			mappings.addSecondPass( sp, !isMappedBy );
559 		}
560 
561 		mappings.addCollection( collection );
562 
563 		//property building
564 		PropertyBinder binder = new PropertyBinder();
565 		binder.setName( propertyName );
566 		binder.setValue( collection );
567 		binder.setCascade( cascadeStrategy );
568 		if ( cascadeStrategy != null && cascadeStrategy.indexOf( "delete-orphan" ) >= 0 ) {
569 			collection.setOrphanDelete( true );
570 		}
571 		binder.setAccessType( accessType );
572 		binder.setProperty( property );
573 		binder.setInsertable( insertable );
574 		binder.setUpdatable( updatable );
575 		Property prop = binder.makeProperty();
576 		//we don't care about the join stuffs because the column is on the association table.
577 		if (! declaringClassSet) throw new AssertionFailure( "DeclaringClass is not set in CollectionBinder while binding" );
578 		propertyHolder.addProperty( prop, declaringClass );
579 	}

581 	private void defineFetchingStrategy() {
582 		LazyCollection lazy = property.getAnnotation( LazyCollection.class );
583 		Fetch fetch = property.getAnnotation( Fetch.class );
584 		OneToMany oneToMany = property.getAnnotation( OneToMany.class );
585 		ManyToMany manyToMany = property.getAnnotation( ManyToMany.class );
586 		ElementCollection elementCollection = property.getAnnotation( ElementCollection.class ); //jpa 2
587 		ManyToAny manyToAny = property.getAnnotation( ManyToAny.class );
588 		FetchType fetchType;
589 		if ( oneToMany != null ) {
590 			fetchType = oneToMany.fetch();
591 		}
592 		else if ( manyToMany != null ) {
593 			fetchType = manyToMany.fetch();
594 		}
595 		else if ( elementCollection != null ) {
596 			fetchType = elementCollection.fetch();
597 		}
598 		else if ( manyToAny != null ) {
599 			fetchType = FetchType.LAZY;
600 		}
601 		else {
602 			throw new AssertionFailure(
603 					"Define fetch strategy on a property not annotated with @ManyToOne nor @OneToMany nor @CollectionOfElements"
604 			);
605 		}
606 		if ( lazy != null ) {
607 			collection.setLazy( !( lazy.value() == LazyCollectionOption.FALSE ) );
608 			collection.setExtraLazy( lazy.value() == LazyCollectionOption.EXTRA );
609 		}
610 		else {
611 			collection.setLazy( fetchType == FetchType.LAZY );
612 			collection.setExtraLazy( false );
613 		}
614 		if ( fetch != null ) {
615 			if ( fetch.value() == org.hibernate.annotations.FetchMode.JOIN ) {
616 				collection.setFetchMode( FetchMode.JOIN );
617 				collection.setLazy( false );
618 			}
619 			else if ( fetch.value() == org.hibernate.annotations.FetchMode.SELECT ) {
620 				collection.setFetchMode( FetchMode.SELECT );
621 			}
622 			else if ( fetch.value() == org.hibernate.annotations.FetchMode.SUBSELECT ) {
623 				collection.setFetchMode( FetchMode.SELECT );
624 				collection.setSubselectLoadable( true );
625 				collection.getOwner().setSubselectLoadableCollections( true );
626 			}
627 			else {

678 	 * return true if it's a Fk, false if it's an association table
679 	 */
680 	protected boolean bindStarToManySecondPass(
681 			Map persistentClasses,
682 			XClass collType,
683 			Ejb3JoinColumn[] fkJoinColumns,
684 			Ejb3JoinColumn[] keyColumns,
685 			Ejb3JoinColumn[] inverseColumns,
686 			Ejb3Column[] elementColumns,
687 			boolean isEmbedded,
688 			XProperty property,
689 			boolean unique,
690 			TableBinder associationTableBinder,
691 			boolean ignoreNotFound,
692 			Mappings mappings) {
693 		PersistentClass persistentClass = (PersistentClass) persistentClasses.get( collType.getName() );
694 		boolean reversePropertyInJoin = false;
695 		if ( persistentClass != null && StringHelper.isNotEmpty( this.mappedBy ) ) {
696 			try {
697 				reversePropertyInJoin = 0 != persistentClass.getJoinNumber(
698 						persistentClass.getRecursiveProperty( this.mappedBy )
699 				);
700 			}
701 			catch (MappingException e) {
702 				StringBuilder error = new StringBuilder( 80 );
703 				error.append( "mappedBy reference an unknown target entity property: " )
704 						.append( collType ).append( "." ).append( this.mappedBy )
705 						.append( " in " )
706 						.append( collection.getOwnerEntityName() )
707 						.append( "." )
708 						.append( property.getName() );
709 				throw new AnnotationException( error.toString() );
710 			}
711 		}
712 		if ( persistentClass != null
713 				&& !reversePropertyInJoin
714 				&& oneToMany
715 				&& !this.isExplicitAssociationTable
716 				&& ( joinColumns[0].isImplicit() && !BinderHelper.isEmptyAnnotationValue( this.mappedBy ) //implicit @JoinColumn
717 				|| !fkJoinColumns[0].isImplicit() ) //this is an explicit @JoinColumn
718 				) {
719 			//this is a Foreign key
720 			bindOneToManySecondPass(
721 					getCollection(),
722 					persistentClasses,
723 					fkJoinColumns,
724 					collType,
725 					cascadeDeleteEnabled,
726 					ignoreNotFound, hqlOrderBy,
727 					mappings,
728 					inheritanceStatePerClass
729 			);
730 			return true;
731 		}
732 		else {
733 			//this is an association table
734 			bindManyToManySecondPass(
735 					this.collection,
736 					persistentClasses,
737 					keyColumns,
738 					inverseColumns,
739 					elementColumns,
740 					isEmbedded, collType,
741 					ignoreNotFound, unique,
742 					cascadeDeleteEnabled,
743 					associationTableBinder, property, propertyHolder, hqlOrderBy, mappings
744 			);
745 			return false;
746 		}
747 	}

749 	protected void bindOneToManySecondPass(
750 			Collection collection,
751 			Map persistentClasses,
752 			Ejb3JoinColumn[] fkJoinColumns,
753 			XClass collectionType,
754 			boolean cascadeDeleteEnabled,
755 			boolean ignoreNotFound,
756 			String hqlOrderBy,
757 			Mappings mappings,
758 			Map<XClass, InheritanceState> inheritanceStatePerClass) {
759 
760 		final boolean debugEnabled = LOG.isDebugEnabled();
761 		if ( debugEnabled ) {
762 			LOG.debugf( "Binding a OneToMany: %s.%s through a foreign key", propertyHolder.getEntityName(), propertyName );
763 		}
764 		org.hibernate.mapping.OneToMany oneToMany = new org.hibernate.mapping.OneToMany( mappings, collection.getOwner() );
765 		collection.setElement( oneToMany );
766 		oneToMany.setReferencedEntityName( collectionType.getName() );
767 		oneToMany.setIgnoreNotFound( ignoreNotFound );
768 
769 		String assocClass = oneToMany.getReferencedEntityName();
770 		PersistentClass associatedClass = (PersistentClass) persistentClasses.get( assocClass );
771 		String orderBy = buildOrderByClauseFromHql( hqlOrderBy, associatedClass, collection.getRole() );
772 		if ( orderBy != null ) collection.setOrderBy( orderBy );
773 		if ( mappings == null ) {
774 			throw new AssertionFailure(
775 					"CollectionSecondPass for oneToMany should not be called with null mappings"
776 			);
777 		}
778 		Map<String, Join> joins = mappings.getJoins( assocClass );
779 		if ( associatedClass == null ) {
780 			throw new MappingException(
781 					"Association references unmapped class: " + assocClass
782 			);
783 		}
784 		oneToMany.setAssociatedClass( associatedClass );
785 		for (Ejb3JoinColumn column : fkJoinColumns) {
786 			column.setPersistentClass( associatedClass, joins, inheritanceStatePerClass );
787 			column.setJoins( joins );
788 			collection.setCollectionTable( column.getTable() );
789 		}
790 		if ( debugEnabled ) {
791 			LOG.debugf( "Mapping collection: %s -> %s", collection.getRole(), collection.getCollectionTable().getName() );
792 		}
793 		bindFilters( false );
794 		bindCollectionSecondPass( collection, null, fkJoinColumns, cascadeDeleteEnabled, property, mappings );
795 		if ( !collection.isInverse()
796 				&& !collection.getKey().isNullable() ) {
797 			// for non-inverse one-to-many, with a not-null fk, add a backref!
798 			String entityName = oneToMany.getReferencedEntityName();
799 			PersistentClass referenced = mappings.getClass( entityName );
800 			Backref prop = new Backref();
801 			prop.setName( '_' + fkJoinColumns[0].getPropertyName() + '_' + fkJoinColumns[0].getLogicalColumnName() + "Backref" );
802 			prop.setUpdateable( false );
803 			prop.setSelectable( false );
804 			prop.setCollectionRole( collection.getRole() );
805 			prop.setEntityName( collection.getOwner().getEntityName() );
806 			prop.setValue( collection.getKey() );
807 			referenced.addProperty( prop );
808 		}
809 	}

1030 	protected void bindManyToManySecondPass(
1031 			Collection collValue,
1032 			Map persistentClasses,
1033 			Ejb3JoinColumn[] joinColumns,
1034 			Ejb3JoinColumn[] inverseJoinColumns,
1035 			Ejb3Column[] elementColumns,
1036 			boolean isEmbedded,
1037 			XClass collType,
1038 			boolean ignoreNotFound, boolean unique,
1039 			boolean cascadeDeleteEnabled,
1040 			TableBinder associationTableBinder,
1041 			XProperty property,
1042 			PropertyHolder parentPropertyHolder,
1043 			String hqlOrderBy,
1044 			Mappings mappings) throws MappingException {
1045 
1046 		PersistentClass collectionEntity = (PersistentClass) persistentClasses.get( collType.getName() );
1047 		boolean isCollectionOfEntities = collectionEntity != null;
1048 		ManyToAny anyAnn = property.getAnnotation( ManyToAny.class );
1049         if (LOG.isDebugEnabled()) {
1050 			String path = collValue.getOwnerEntityName() + "." + joinColumns[0].getPropertyName();
1051             if (isCollectionOfEntities && unique) LOG.debugf("Binding a OneToMany: %s through an association table", path);
1052             else if (isCollectionOfEntities) LOG.debugf("Binding as ManyToMany: %s", path);
1053             else if (anyAnn != null) LOG.debugf("Binding a ManyToAny: %s", path);
1054             else LOG.debugf("Binding a collection of element: %s", path);
1055 		}
1056 		//check for user error
1057 		if ( !isCollectionOfEntities ) {
1058 			if ( property.isAnnotationPresent( ManyToMany.class ) || property.isAnnotationPresent( OneToMany.class ) ) {
1059 				String path = collValue.getOwnerEntityName() + "." + joinColumns[0].getPropertyName();
1060 				throw new AnnotationException(
1061 						"Use of @OneToMany or @ManyToMany targeting an unmapped class: " + path + "[" + collType + "]"
1062 				);
1063 			}
1064 			else if ( anyAnn != null ) {
1065 				if ( parentPropertyHolder.getJoinTable( property ) == null ) {
1066 					String path = collValue.getOwnerEntityName() + "." + joinColumns[0].getPropertyName();
1067 					throw new AnnotationException(
1068 							"@JoinTable is mandatory when @ManyToAny is used: " + path
1069 					);
1070 				}
1071 			}
1072 			else {
1073 				JoinTable joinTableAnn = parentPropertyHolder.getJoinTable( property );
1074 				if ( joinTableAnn != null && joinTableAnn.inverseJoinColumns().length > 0 ) {
1075 					String path = collValue.getOwnerEntityName() + "." + joinColumns[0].getPropertyName();
1076 					throw new AnnotationException(
1077 							"Use of @JoinTable.inverseJoinColumns targeting an unmapped class: " + path + "[" + collType + "]"
1078 					);
1079 				}
1080 			}
1081 		}
1082 
1083 		boolean mappedBy = !BinderHelper.isEmptyAnnotationValue( joinColumns[0].getMappedBy() );
1084 		if ( mappedBy ) {
1085 			if ( !isCollectionOfEntities ) {
1086 				StringBuilder error = new StringBuilder( 80 )
1087 						.append(
1088 								"Collection of elements must not have mappedBy or association reference an unmapped entity: "
1089 						)
1090 						.append( collValue.getOwnerEntityName() )
1091 						.append( "." )
1092 						.append( joinColumns[0].getPropertyName() );
1093 				throw new AnnotationException( error.toString() );
1094 			}
1095 			Property otherSideProperty;
1096 			try {
1097 				otherSideProperty = collectionEntity.getRecursiveProperty( joinColumns[0].getMappedBy() );
1098 			}
1099 			catch (MappingException e) {
1100 				StringBuilder error = new StringBuilder( 80 );
1101 				error.append( "mappedBy reference an unknown target entity property: " )
1102 						.append( collType ).append( "." ).append( joinColumns[0].getMappedBy() )
1103 						.append( " in " )
1104 						.append( collValue.getOwnerEntityName() )
1105 						.append( "." )
1106 						.append( joinColumns[0].getPropertyName() );
1107 				throw new AnnotationException( error.toString() );
1108 			}
1109 			Table table;
1110 			if ( otherSideProperty.getValue() instanceof Collection ) {
1111 				//this is a collection on the other side
1112 				table = ( (Collection) otherSideProperty.getValue() ).getCollectionTable();
1113 			}
1114 			else {
1115 				//This is a ToOne with a @JoinTable or a regular property
1116 				table = otherSideProperty.getValue().getTable();
1117 			}
1118 			collValue.setCollectionTable( table );
1119 			String entityName = collectionEntity.getEntityName();
1120 			for (Ejb3JoinColumn column : joinColumns) {
1121 				//column.setDefaultColumnHeader( joinColumns[0].getMappedBy() ); //seems not to be used, make sense
1122 				column.setManyToManyOwnerSideEntityName( entityName );
1123 			}
1124 		}
1125 		else {
1126 			//TODO: only for implicit columns?
1127 			//FIXME NamingStrategy
1128 			for (Ejb3JoinColumn column : joinColumns) {
1129 				String mappedByProperty = mappings.getFromMappedBy(
1130 						collValue.getOwnerEntityName(), column.getPropertyName()
1131 				);
1132 				Table ownerTable = collValue.getOwner().getTable();
1133 				column.setMappedBy(
1134 						collValue.getOwner().getEntityName(),
1135 						collValue.getOwner().getJpaEntityName(),
1136 						mappings.getLogicalTableName( ownerTable ),
1137 						mappedByProperty
1138 				);
1139 //				String header = ( mappedByProperty == null ) ? mappings.getLogicalTableName( ownerTable ) : mappedByProperty;
1140 //				column.setDefaultColumnHeader( header );
1141 			}
1142 			if ( StringHelper.isEmpty( associationTableBinder.getName() ) ) {
1143 				//default value
1144 				associationTableBinder.setDefaultName(
1145 						collValue.getOwner().getEntityName(),
1146 						collValue.getOwner().getJpaEntityName(),
1147 						mappings.getLogicalTableName( collValue.getOwner().getTable() ),
1148 						collectionEntity != null ? collectionEntity.getEntityName() : null,
1149 						collectionEntity != null ? collectionEntity.getJpaEntityName() : null,
1150 						collectionEntity != null ? mappings.getLogicalTableName( collectionEntity.getTable() ) : null,
1151 						joinColumns[0].getPropertyName()
1152 				);
1153 			}
1154 			associationTableBinder.setJPA2ElementCollection( !isCollectionOfEntities && property.isAnnotationPresent( ElementCollection.class ));
1155 			collValue.setCollectionTable( associationTableBinder.bind() );
1156 		}
1157 		bindFilters( isCollectionOfEntities );
1158 		bindCollectionSecondPass( collValue, collectionEntity, joinColumns, cascadeDeleteEnabled, property, mappings );
1159 
1160 		ManyToOne element = null;
1161 		if ( isCollectionOfEntities ) {
1162 			element =
1163 					new ManyToOne( mappings,  collValue.getCollectionTable() );
1164 			collValue.setElement( element );
1165 			element.setReferencedEntityName( collType.getName() );
1166 			//element.setFetchMode( fetchMode );
1167 			//element.setLazy( fetchMode != FetchMode.JOIN );
1168 			//make the second join non lazy
1169 			element.setFetchMode( FetchMode.JOIN );
1170 			element.setLazy( false );
1171 			element.setIgnoreNotFound( ignoreNotFound );
1172 			// as per 11.1.38 of JPA-2 spec, default to primary key if no column is specified by @OrderBy.
1173 			if ( hqlOrderBy != null ) {
1174 				collValue.setManyToManyOrdering(
1175 						buildOrderByClauseFromHql( hqlOrderBy, collectionEntity, collValue.getRole() )
1176 				);
1177 			}
1178 			ForeignKey fk = property != null ? property.getAnnotation( ForeignKey.class ) : null;
1179 			String fkName = fk != null ? fk.inverseName() : "";
1180 			if ( !BinderHelper.isEmptyAnnotationValue( fkName ) ) element.setForeignKeyName( fkName );
1181 		}
1182 		else if ( anyAnn != null ) {
1183 			//@ManyToAny
1184 			//Make sure that collTyp is never used during the @ManyToAny branch: it will be set to void.class
1185 			PropertyData inferredData = new PropertyInferredData(null, property, "unsupported", mappings.getReflectionManager() );
1186 			//override the table
1187 			for (Ejb3Column column : inverseJoinColumns) {
1188 				column.setTable( collValue.getCollectionTable() );
1189 			}
1190 			Any any = BinderHelper.buildAnyValue( anyAnn.metaDef(), inverseJoinColumns, anyAnn.metaColumn(),
1191 					inferredData, cascadeDeleteEnabled, Nullability.NO_CONSTRAINT,
1192 					propertyHolder, new EntityBinder(), true, mappings );
1193 			collValue.setElement( any );
1194 		}
1195 		else {
1196 			XClass elementClass;
1197 			AnnotatedClassType classType;
1198 
1199 			PropertyHolder holder = null;
1200 			if ( BinderHelper.PRIMITIVE_NAMES.contains( collType.getName() ) ) {
1201 				classType = AnnotatedClassType.NONE;
1202 				elementClass = null;
1203 			}
1204 			else {
1205 				elementClass = collType;
1206 				classType = mappings.getClassType( elementClass );
1207 
1208 				holder = PropertyHolderBuilder.buildPropertyHolder(
1209 						collValue,
1210 						collValue.getRole(),
1211 						elementClass,
1212 						property, parentPropertyHolder, mappings
1213 				);
1214 				//force in case of attribute override
1215 				boolean attributeOverride = property.isAnnotationPresent( AttributeOverride.class )
1216 						|| property.isAnnotationPresent( AttributeOverrides.class );
1217 				if ( isEmbedded || attributeOverride ) {
1218 					classType = AnnotatedClassType.EMBEDDABLE;
1219 				}
1220 			}
1221 
1222 			if ( AnnotatedClassType.EMBEDDABLE.equals( classType ) ) {
1223 				EntityBinder entityBinder = new EntityBinder();
1224 				PersistentClass owner = collValue.getOwner();
1225 				boolean isPropertyAnnotated;
1226 				//FIXME support @Access for collection of elements
1227 				//String accessType = access != null ? access.value() : null;
1228 				if ( owner.getIdentifierProperty() != null ) {
1229 					isPropertyAnnotated = owner.getIdentifierProperty().getPropertyAccessorName().equals( "property" );
1230 				}
1231 				else if ( owner.getIdentifierMapper() != null && owner.getIdentifierMapper().getPropertySpan() > 0 ) {
1232 					Property prop = (Property) owner.getIdentifierMapper().getPropertyIterator().next();
1233 					isPropertyAnnotated = prop.getPropertyAccessorName().equals( "property" );
1234 				}
1235 				else {
1236 					throw new AssertionFailure( "Unable to guess collection property accessor name" );
1237 				}
1238 
1239 				PropertyData inferredData;
1240 				if ( isMap() ) {
1241 					//"value" is the JPA 2 prefix for map values (used to be "element")
1242 					if ( isHibernateExtensionMapping() ) {
1243 						inferredData = new PropertyPreloadedData( AccessType.PROPERTY, "element", elementClass );
1244 					}
1245 					else {
1246 						inferredData = new PropertyPreloadedData( AccessType.PROPERTY, "value", elementClass );
1247 					}
1248 				}
1249 				else {
1250 					if ( isHibernateExtensionMapping() ) {
1251 						inferredData = new PropertyPreloadedData( AccessType.PROPERTY, "element", elementClass );
1252 					}
1253 					else {
1254 						//"collection&&element" is not a valid property name => placeholder
1255 						inferredData = new PropertyPreloadedData( AccessType.PROPERTY, "collection&&element", elementClass );
1256 					}
1257 				}
1258 				//TODO be smart with isNullable
1259 				Component component = AnnotationBinder.fillComponent(
1260 						holder, inferredData, isPropertyAnnotated ? AccessType.PROPERTY : AccessType.FIELD, true,
1261 						entityBinder, false, false,
1262 						true, mappings, inheritanceStatePerClass
1263 				);
1264 
1265 				collValue.setElement( component );
1266 
1267 				if ( StringHelper.isNotEmpty( hqlOrderBy ) ) {
1268 					String path = collValue.getOwnerEntityName() + "." + joinColumns[0].getPropertyName();
1269 					String orderBy = adjustUserSuppliedValueCollectionOrderingFragment( hqlOrderBy );
1270 					if ( orderBy != null ) {
1271 						collValue.setOrderBy( orderBy );
1272 					}
1273 				}
1274 			}
1275 			else {
1276 				SimpleValueBinder elementBinder = new SimpleValueBinder();
1277 				elementBinder.setMappings( mappings );
1278 				elementBinder.setReturnedClassName( collType.getName() );
1279 				if ( elementColumns == null || elementColumns.length == 0 ) {
1280 					elementColumns = new Ejb3Column[1];
1281 					Ejb3Column column = new Ejb3Column();
1282 					column.setImplicit( false );
1283 					//not following the spec but more clean
1284 					column.setNullable( true );
1285 					column.setLength( Ejb3Column.DEFAULT_COLUMN_LENGTH );
1286 					column.setLogicalColumnName( Collection.DEFAULT_ELEMENT_COLUMN_NAME );
1287 					//TODO create an EMPTY_JOINS collection
1288 					column.setJoins( new HashMap<String, Join>() );
1289 					column.setMappings( mappings );
1290 					column.bind();
1291 					elementColumns[0] = column;
1292 				}
1293 				//override the table
1294 				for (Ejb3Column column : elementColumns) {
1295 					column.setTable( collValue.getCollectionTable() );
1296 				}
1297 				elementBinder.setColumns( elementColumns );
1298 				elementBinder.setType( property, elementClass, collValue.getOwnerEntityName() );
1299 				elementBinder.setPersistentClassName( propertyHolder.getEntityName() );
1300 				elementBinder.setAccessType( accessType );
1301 				collValue.setElement( elementBinder.make() );
1302 				String orderBy = adjustUserSuppliedValueCollectionOrderingFragment( hqlOrderBy );
1303 				if ( orderBy != null ) {
1304 					collValue.setOrderBy( orderBy );
1305 				}
1306 			}
1307 		}
1308 
1309 		checkFilterConditions( collValue );
1310 
1311 		//FIXME: do optional = false
1312 		if ( isCollectionOfEntities ) {
1313 			bindManytoManyInverseFk( collectionEntity, inverseJoinColumns, element, unique, mappings );
1314 		}
1315 
1316 	}

1318 	private static void checkFilterConditions(Collection collValue) {
1319 		//for now it can't happen, but sometime soon...
1320 		if ( ( collValue.getFilters().size() != 0 || StringHelper.isNotEmpty( collValue.getWhere() ) ) &&
1321 				collValue.getFetchMode() == FetchMode.JOIN &&
1322 				!( collValue.getElement() instanceof SimpleValue ) && //SimpleValue (CollectionOfElements) are always SELECT but it does not matter
1323 				collValue.getElement().getFetchMode() != FetchMode.JOIN ) {
1324 			throw new MappingException(
1325 					"@ManyToMany or @CollectionOfElements defining filter or where without join fetching "
1326 							+ "not valid within collection using join fetching[" + collValue.getRole() + "]"
1327 			);
1328 		}

1367 	/**
1368 	 * bind the inverse FK of a ManyToMany
1369 	 * If we are in a mappedBy case, read the columns from the associated
1370 	 * collection element
1371 	 * Otherwise delegates to the usual algorithm
1372 	 */
1373 	public static void bindManytoManyInverseFk(
1374 			PersistentClass referencedEntity,
1375 			Ejb3JoinColumn[] columns,
1376 			SimpleValue value,
1377 			boolean unique,
1378 			Mappings mappings) {
1379 		final String mappedBy = columns[0].getMappedBy();
1380 		if ( StringHelper.isNotEmpty( mappedBy ) ) {
1381 			final Property property = referencedEntity.getRecursiveProperty( mappedBy );
1382 			Iterator mappedByColumns;
1383 			if ( property.getValue() instanceof Collection ) {
1384 				mappedByColumns = ( (Collection) property.getValue() ).getKey().getColumnIterator();
1385 			}
1386 			else {
1387 				//find the appropriate reference key, can be in a join
1388 				Iterator joinsIt = referencedEntity.getJoinIterator();
1389 				KeyValue key = null;
1390 				while ( joinsIt.hasNext() ) {
1391 					Join join = (Join) joinsIt.next();
1392 					if ( join.containsProperty( property ) ) {
1393 						key = join.getKey();
1394 						break;
1395 					}
1396 				}
1397 				if ( key == null ) key = property.getPersistentClass().getIdentifier();
1398 				mappedByColumns = key.getColumnIterator();
1399 			}
1400 			while ( mappedByColumns.hasNext() ) {
1401 				Column column = (Column) mappedByColumns.next();
1402 				columns[0].linkValueUsingAColumnCopy( column, value );
1403 			}
1404 			String referencedPropertyName =
1405 					mappings.getPropertyReferencedAssociation(
1406 							"inverse__" + referencedEntity.getEntityName(), mappedBy
1407 					);
1408 			if ( referencedPropertyName != null ) {
1409 				//TODO always a many to one?
1410 				( (ManyToOne) value ).setReferencedPropertyName( referencedPropertyName );
1411 				mappings.addUniquePropertyReference( referencedEntity.getEntityName(), referencedPropertyName );
1412 			}
1413 			( (ManyToOne) value ).setReferenceToPrimaryKey( referencedPropertyName == null );
1414 			value.createForeignKey();
1415 		}
1416 		else {
1417 			BinderHelper.createSyntheticPropertyReference( columns, referencedEntity, null, value, true, mappings );
1418 			TableBinder.bindFk( referencedEntity, null, columns, value, unique, mappings );
1419 		}
1420 	}
