23  package org.hibernate.test.annotations.idclass;
24  import java.io.Serializable;
25  
26  /**
27   * A DomainAdminId.
28   *
29   * @author <a href="mailto:stale.pedersen@jboss.org">Stale W. Pedersen</a>
30   */
31  @SuppressWarnings("serial")
32  public class DomainAdminId implements Serializable {
33  
34  	private String domainName;
35  
36  	private String adminUser;
37  
38  	public DomainAdminId() {
39  	}
40  
41  	public DomainAdminId(String domainName, String adminUser) {
42  		this.domainName = domainName;
43  		this.adminUser = adminUser;
44  	}
45  
46  	public String getDomainName() {
47  		return domainName;
48  	}
49  
50  	public void setDomainName(String domainName) {
51  		this.domainName = domainName;
52  	}
53  
54  	public String getAdminUser() {
55  		return adminUser;
56  	}
57  
58  	public void setAdminUser(String adminUser) {
59  		this.adminUser = adminUser;
60  	}
61  
62  	@Override
63  	public boolean equals(Object o) {
64  		return ( ( o instanceof DomainAdminId ) && domainName.equals( ( ( DomainAdminId ) o ).getDomainName() ) &&
65  				adminUser.equals( ( ( DomainAdminId ) o ).getAdminUser() ) );
66  	}
67  
68  	@Override
69  	public int hashCode() {
70  		return ( domainName + adminUser ).hashCode();
71  	}
72  }
