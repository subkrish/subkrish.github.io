45   * @author Gavin King

47  public class CompositePropertyRefTest extends BaseCoreFunctionalTestCase {
48  	public String[] getMappings() {
49  		return new String[] { "cuk/Person.hbm.xml" };
50  	}
51  
52  	public void configure(Configuration cfg) {
53  		cfg.setProperty(Environment.DEFAULT_BATCH_FETCH_SIZE, "1");
54  	}
55  
56  	@Test
57  	@SuppressWarnings( {"unchecked", "UnusedAssignment"})
58  	public void testOneToOnePropertyRef() {
59  		Session s = openSession();
60  		Transaction t = s.beginTransaction();
61  		Person p = new Person();
62  		p.setName("Steve");
63  		p.setUserId("steve");
64  		Address a = new Address();
65  		a.setAddress("Texas");
66  		a.setCountry("USA");
67  		p.setAddress(a);
68  		a.setPerson(p);
69  		s.save(p);
70  		Person p2 = new Person();
71  		p2.setName("Max");
72  		p2.setUserId("max");
73  		s.save(p2);
74  		Account act = new Account();
75  		act.setType('c');
76  		act.setUser(p2);
77  		p2.getAccounts().add(act);
78  		s.save(act);
79  		s.flush();
80  		s.clear();
81  		
82  		p = (Person) s.get( Person.class, p.getId() ); //get address reference by outer join
83  		p2 = (Person) s.get( Person.class, p2.getId() ); //get null address reference by outer join
84  		assertNull( p2.getAddress() );
85  		assertNotNull( p.getAddress() );
86  		List l = s.createQuery("from Person").list(); //pull address references for cache
87  		assertEquals( l.size(), 2 );
88  		assertTrue( l.contains(p) && l.contains(p2) );
89  		s.clear();
90  		
91  		l = s.createQuery("from Person p order by p.name").list(); //get address references by sequential selects
92  		assertEquals( l.size(), 2 );
93  		assertNull( ( (Person) l.get(0) ).getAddress() );
94  		assertNotNull( ( (Person) l.get(1) ).getAddress() );
95  		s.clear();
96  		
97  		l = s.createQuery("from Person p left join fetch p.address a order by a.country").list(); //get em by outer join
98  		assertEquals( l.size(), 2 );
99  		if ( ( (Person) l.get(0) ).getName().equals("Max") ) {
100 			assertNull( ( (Person) l.get(0) ).getAddress() );
101 			assertNotNull( ( (Person) l.get(1) ).getAddress() );
102 		}
103 		else {
104 			assertNull( ( (Person) l.get(1) ).getAddress() );
105 			assertNotNull( ( (Person) l.get(0) ).getAddress() );
106 		}
107 		s.clear();
108 		
109 		l = s.createQuery("from Person p left join p.accounts").list();
110 		for ( int i=0; i<2; i++ ) {
111 			Object[] row = (Object[]) l.get(i);
112 			Person px = (Person) row[0];
113 			Set accounts = px.getAccounts();
114 			assertFalse( Hibernate.isInitialized(accounts) );
115 			assertTrue( px.getAccounts().size()>0 || row[1]==null );
116 		}
117 		s.clear();
118 
119 		l = s.createQuery("from Person p left join fetch p.accounts a order by p.name").list();
120 		Person p0 = (Person) l.get(0);
121 		assertTrue( Hibernate.isInitialized( p0.getAccounts() ) );
122 		assertEquals( p0.getAccounts().size(), 1 );
123 		assertSame( ( (Account) p0.getAccounts().iterator().next() ).getUser(), p0 );
124 		Person p1 = (Person) l.get(1);
125 		assertTrue( Hibernate.isInitialized( p1.getAccounts() ) );
126 		assertEquals( p1.getAccounts().size(), 0 );
127 		s.clear();
128 		
129 		l = s.createQuery("from Account a join fetch a.user").list();
130 		
131 		s.clear();
132 		
133 		l = s.createQuery("from Person p left join fetch p.address").list();
134 		
135 		s.clear();
136 		s.createQuery( "delete Address" ).executeUpdate();
137 		s.createQuery( "delete Account" ).executeUpdate();
138 		s.createQuery( "delete Person" ).executeUpdate();
139 		t.commit();
140 		s.close();
141 	}
