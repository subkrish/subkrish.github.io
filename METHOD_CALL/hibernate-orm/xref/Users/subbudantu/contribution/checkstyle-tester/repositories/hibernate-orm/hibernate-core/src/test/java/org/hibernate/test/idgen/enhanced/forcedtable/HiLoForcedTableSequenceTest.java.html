37  import static org.junit.Assert.assertTrue;

39  /**
40   * {@inheritDoc}
41   *
42   * @author Steve Ebersole
43   */
44  public class HiLoForcedTableSequenceTest extends BaseCoreFunctionalTestCase {
45  	public String[] getMappings() {
46  		return new String[] { "idgen/enhanced/forcedtable/HiLo.hbm.xml" };
47  	}
48  
49  	@Test
50  	public void testNormalBoundary() {
51  		EntityPersister persister = sessionFactory().getEntityPersister( Entity.class.getName() );
52  		assertTrue(
53  				"sequence style generator was not used",
54  				SequenceStyleGenerator.class.isInstance( persister.getIdentifierGenerator() )
55  		);
56  		SequenceStyleGenerator generator = ( SequenceStyleGenerator ) persister.getIdentifierGenerator();
57  		assertTrue(
58  				"table structure was not used",
59  				TableStructure.class.isInstance( generator.getDatabaseStructure() )
60  		);
61  		assertTrue(
62  				"hilo optimizer was not used",
63  				OptimizerFactory.HiLoOptimizer.class.isInstance( generator.getOptimizer() )
64  		);
65  		OptimizerFactory.HiLoOptimizer optimizer = ( OptimizerFactory.HiLoOptimizer ) generator.getOptimizer();
66  
67  		int increment = optimizer.getIncrementSize();
68  		Entity[] entities = new Entity[ increment + 1 ];
69  		Session s = openSession();
70  		s.beginTransaction();
71  		for ( int i = 0; i < increment; i++ ) {
72  			entities[i] = new Entity( "" + ( i + 1 ) );
73  			s.save( entities[i] );
74  			long expectedId = i + 1;
75  			assertEquals( expectedId, entities[i].getId().longValue() );
76  			assertEquals( 1, ( (BasicHolder) optimizer.getLastSourceValue() ).getActualLongValue() );
77  			assertEquals( i + 1, ( (BasicHolder) optimizer.getLastValue() ).getActualLongValue() );
78  			assertEquals( increment + 1, ( (BasicHolder) optimizer.getHiValue() ).getActualLongValue() );
79  		}
80  		// now force a "clock over"
81  		entities[ increment ] = new Entity( "" + increment );
82  		s.save( entities[ increment ] );
83  		long expectedId = optimizer.getIncrementSize() + 1;
84  		assertEquals( expectedId, entities[ optimizer.getIncrementSize() ].getId().longValue() );
85  		assertEquals( 2, ( (BasicHolder) optimizer.getLastSourceValue() ).getActualLongValue() ); // initialization + clock-over
86  		assertEquals( increment + 1, ( (BasicHolder) optimizer.getLastValue() ).getActualLongValue() );
87  		assertEquals( ( increment * 2 ) + 1, ( (BasicHolder) optimizer.getHiValue() ).getActualLongValue() );
88  
89  		s.getTransaction().commit();
90  
91  		s.beginTransaction();
92  		for ( int i = 0; i < entities.length; i++ ) {
93  			assertEquals( i + 1, entities[i].getId().intValue() );
94  			s.delete( entities[i] );
95  		}
96  		s.getTransaction().commit();
97  		s.close();
98  	}
99  }
