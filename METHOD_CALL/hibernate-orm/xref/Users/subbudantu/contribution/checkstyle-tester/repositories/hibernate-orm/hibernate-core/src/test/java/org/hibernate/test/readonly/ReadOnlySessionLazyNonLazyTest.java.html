48  /**
49   * Model:
50   *
51   * Container
52   * |-- N : 1 -- noProxyOwner (property-ref="name" lazy="no-proxy" cascade="all")
53   * |-- N : 1 -- proxyOwner (property-ref="name" lazy="proxy" cascade="all")
54   * |-- N : 1 -- nonLazyOwner (property-ref="name" lazy="false" cascade="all")
55   * |-- N : 1 -- noProxyInfo" (lazy="no-proxy" cascade="all")
56   * |-- N : 1 -- proxyInfo (lazy="proxy" cascade="all"
57   * |-- N : 1 -- nonLazyInfo" (lazy="false" cascade="all")
58   * |
59   * |-- 1 : N -- lazyDataPoints" (lazy="true" inverse="false" cascade="all")
60   * |-- 1 : N -- nonLazySelectDataPoints" (lazy="false" inverse="false" cascade="all" fetch="select")
61   * |-- 1 : N -- nonLazyJoinDataPoints" (lazy="false" inverse="false" cascade="all" fetch="join")
62   *
63   * Note: the following many-to-one properties use a property-ref so they are
64   * initialized, regardless of how the lazy attribute is mapped:
65   * noProxyOwner, proxyOwner, nonLazyOwner
66   *
67   * @author Gail Badner
68   */

76  	@SuppressWarnings( {"unchecked"})
77  	public void testExistingModifiableAfterSetSessionReadOnly() {
78  		Container cOrig = createContainer();
79  		Set expectedInitializedObjects = new HashSet(
80  				Arrays.asList(
81  						cOrig,
82  						cOrig.getNoProxyInfo(),
83  						cOrig.getProxyInfo(),
84  						cOrig.getNonLazyInfo(),
85  						cOrig.getNoProxyOwner(),
86  						cOrig.getProxyOwner(),
87  						cOrig.getNonLazyOwner(),
88  						cOrig.getLazyDataPoints().iterator().next(),
89  						cOrig.getNonLazyJoinDataPoints().iterator().next(),
90  						cOrig.getNonLazySelectDataPoints().iterator().next()
91  				)
92  		);
93  		Set expectedReadOnlyObjects = new HashSet();
94  
95  		Session s = openSession();
96  		assertFalse( s.isDefaultReadOnly() );
97  		Transaction t = s.beginTransaction();
98  		s.save( cOrig );
99  		checkContainer( cOrig, expectedInitializedObjects, expectedReadOnlyObjects, s );
100 		s.setDefaultReadOnly( true );
101 		assertTrue( s.isDefaultReadOnly() );
102 		checkContainer( cOrig, expectedInitializedObjects, expectedReadOnlyObjects, s );
103 		t.commit();
104 
105 		t = s.beginTransaction();
106 		assertTrue( s.isDefaultReadOnly() );
107 		checkContainer( cOrig, expectedInitializedObjects, expectedReadOnlyObjects, s );
108 		Container c = ( Container ) s.load( Container.class, cOrig.getId() );
109 		assertSame( cOrig, c );
110 		checkContainer( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
111 		c = ( Container ) s.get( Container.class, cOrig.getId() );
112 		assertSame( cOrig, c );
113 		checkContainer( cOrig, expectedInitializedObjects, expectedReadOnlyObjects, s );
114 		s.refresh( cOrig );
115 		assertSame( cOrig, c );
116 		checkContainer( cOrig, expectedInitializedObjects, expectedReadOnlyObjects, s );
117 		s.evict( cOrig );
118 		c = ( Container ) s.get( Container.class, cOrig.getId()  );
119 		assertNotSame( cOrig, c );
120 		expectedInitializedObjects = new HashSet(
121 				Arrays.asList(
122 						c,
123 						c.getNonLazyInfo(),
124 						c.getNoProxyOwner(),
125 						c.getProxyOwner(),
126 						c.getNonLazyOwner(),
127 						c.getNonLazyJoinDataPoints().iterator().next(),
128 						c.getNonLazySelectDataPoints().iterator().next()
129 				)
130 		);
131 		expectedReadOnlyObjects = new HashSet(
132 				Arrays.asList(
133 						c,
134 						c.getNoProxyInfo(),
135 						c.getProxyInfo(),
136 						c.getNonLazyInfo(),
137 						c.getNoProxyOwner(),
138 						c.getProxyOwner(),
139 						c.getNonLazyOwner(),
140 						//c.getLazyDataPoints(),
141 						c.getNonLazyJoinDataPoints().iterator().next(),
142 						c.getNonLazySelectDataPoints().iterator().next()
143 				)
144 		);
145 		checkContainer( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
146 		assertFalse( Hibernate.isInitialized( c.getNoProxyInfo() ) );
147 		Hibernate.initialize( c.getNoProxyInfo() );
148 		expectedInitializedObjects.add( c.getNoProxyInfo() );
149 		checkContainer( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
150 		assertFalse( Hibernate.isInitialized( c.getProxyInfo() ) );
151 		Hibernate.initialize( c.getProxyInfo() );
152 		expectedInitializedObjects.add( c.getProxyInfo() );
153 		checkContainer( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
154 		assertFalse( Hibernate.isInitialized( c.getLazyDataPoints() ) );
155 		Hibernate.initialize( c.getLazyDataPoints() );
156 		expectedInitializedObjects.add( c.getLazyDataPoints().iterator().next() );
157 		expectedReadOnlyObjects.add(c.getLazyDataPoints().iterator().next() );
158 		checkContainer( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
159 		t.commit();
160 		s.close();
161 		s = openSession();
162 		t = s.beginTransaction();
163 		s.createQuery("delete from DataPoint").executeUpdate();
164 		s.createQuery("delete from Container").executeUpdate();
165 		s.createQuery("delete from Info").executeUpdate();
166 		s.createQuery("delete from Owner").executeUpdate();
167 		t.commit();
168 		s.close();
169

1352 	private void checkContainer(Container c, Set expectedInitializedObjects, Set expectedReadOnlyObjects, Session s) {
1353 		checkObject( c, expectedInitializedObjects, expectedReadOnlyObjects, s );
1354 		if ( ! expectedInitializedObjects.contains( c ) ) {
1355 			return;
1356 		}
1357 		checkObject( c.getNoProxyInfo(), expectedInitializedObjects, expectedReadOnlyObjects, s);
1358 		checkObject( c.getProxyInfo(), expectedInitializedObjects, expectedReadOnlyObjects, s);
1359 		checkObject( c.getNonLazyInfo(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1360 		checkObject( c.getNoProxyOwner(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1361 		checkObject( c.getProxyOwner(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1362 		checkObject( c.getNonLazyOwner(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1363 		if ( Hibernate.isInitialized( c.getLazyDataPoints() ) ) {
1364 			for ( Iterator it=c.getLazyDataPoints().iterator(); it.hasNext(); ) {
1365 				checkObject( it.next(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1366 			}
1367 		}
1368 		for ( Iterator it=c.getNonLazyJoinDataPoints().iterator(); it.hasNext(); ) {
1369 			checkObject( it.next(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1370 		}
1371 		for ( Iterator it=c.getNonLazySelectDataPoints().iterator(); it.hasNext(); ) {
1372 			checkObject( it.next(), expectedInitializedObjects, expectedReadOnlyObjects, s );
1373 		}
1374 	}

1376 	private void checkObject(Object entityOrProxy, Set expectedInitializedObjects, Set expectedReadOnlyObjects, Session s) {
1377 		boolean isExpectedToBeInitialized = expectedInitializedObjects.contains( entityOrProxy );
1378 		boolean isExpectedToBeReadOnly = expectedReadOnlyObjects.contains( entityOrProxy );
1379 		SessionImplementor si = (SessionImplementor) s;
1380 		assertEquals( isExpectedToBeInitialized, Hibernate.isInitialized( entityOrProxy ) );
1381 		assertEquals( isExpectedToBeReadOnly, s.isReadOnly( entityOrProxy ) );
1382 		if ( Hibernate.isInitialized( entityOrProxy ) ) {
1383 			Object entity = ( entityOrProxy instanceof HibernateProxy ?
1384 					( ( HibernateProxy ) entityOrProxy ).getHibernateLazyInitializer().getImplementation( si ) :
1385 					entityOrProxy
1386 			);
1387 			assertNotNull( entity );
1388 			assertEquals( isExpectedToBeReadOnly, s.isReadOnly( entity ));
1389 		}
