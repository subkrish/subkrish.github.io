47  	@Test
48  	@SuppressWarnings( {"unchecked"})
49  	public void testEntitySubselect() {
50  		Session s = openSession();
51  		Transaction t = s.beginTransaction();
52  		Human gavin = new Human();
53  		gavin.setName( "gavin" );
54  		gavin.setSex( 'M' );
55  		gavin.setAddress( "Melbourne, Australia" );
56  		Alien x23y4 = new Alien();
57  		x23y4.setIdentity( "x23y4$$hu%3" );
58  		x23y4.setPlanet( "Mars" );
59  		x23y4.setSpecies( "martian" );
60  		s.save(gavin);
61  		s.save(x23y4);
62  		s.flush();
63  		List<Being> beings = ( List<Being>) s.createQuery("from Being").list();
64  		for ( Being being : beings ) {
65  			assertNotNull( being.getLocation() );
66  			assertNotNull( being.getIdentity() );
67  			assertNotNull( being.getSpecies() );
68  		}
69  		s.clear();
70  		sessionFactory().getCache().evictEntityRegion( Being.class );
71  		Being gav = (Being) s.get(Being.class, gavin.getId());
72  		assertEquals( gav.getLocation(), gavin.getAddress() );
73  		assertEquals( gav.getSpecies(), "human" );
74  		assertEquals( gav.getIdentity(), gavin.getName() );
75  		s.clear();
76  		//test the <synchronized> tag:
77  		gavin = (Human) s.get(Human.class, gavin.getId());
78  		gavin.setAddress( "Atlanta, GA" );
79  		gav = (Being) s.createQuery("from Being b where b.location like '%GA%'").uniqueResult();
80  		assertEquals( gav.getLocation(), gavin.getAddress() );
81  		s.delete(gavin);
82  		s.delete(x23y4);
83  		assertTrue( s.createQuery("from Being").list().isEmpty() );
84  		t.commit();
85  		s.close();
86  	}

88  	@Test
89  	public void testCustomColumnReadAndWrite() {
90  		Session s = openSession();
91  		Transaction t = s.beginTransaction();
92  		final double HUMAN_INCHES = 73;
93  		final double ALIEN_INCHES = 931;
94  		final double HUMAN_CENTIMETERS = HUMAN_INCHES * 2.54d;		
95  		final double ALIEN_CENTIMETERS = ALIEN_INCHES * 2.54d;		
96  		Human gavin = new Human();
97  		gavin.setName( "gavin" );
98  		gavin.setSex( 'M' );
99  		gavin.setAddress( "Melbourne, Australia" );
100 		gavin.setHeightInches( HUMAN_INCHES );
101 		Alien x23y4 = new Alien();
102 		x23y4.setIdentity( "x23y4$$hu%3" );
103 		x23y4.setPlanet( "Mars" );
104 		x23y4.setSpecies( "martian" );
105 		x23y4.setHeightInches( ALIEN_INCHES );
106 		s.save(gavin);
107 		s.save(x23y4);
108 		s.flush();
109 		
110 		// Test value conversion during insert
111 		// Value returned by Oracle native query is a Types.NUMERIC, which is mapped to a BigDecimalType;
112 		// Cast returned value to Number then call Number.doubleValue() so it works on all dialects.
113 		Double humanHeightViaSql =
114 				( (Number)s.createSQLQuery("select height_centimeters from humans").uniqueResult() ).doubleValue();
115 		assertEquals(HUMAN_CENTIMETERS, humanHeightViaSql, 0.01d);
116 		Double alienHeightViaSql =
117 				( (Number)s.createSQLQuery("select height_centimeters from aliens").uniqueResult() ).doubleValue();
118 		assertEquals(ALIEN_CENTIMETERS, alienHeightViaSql, 0.01d);
119 		s.clear();
120 		
121 		// Test projection
122 		Double heightViaHql = (Double)s.createQuery("select heightInches from Being b where b.identity = 'gavin'").uniqueResult();
123 		assertEquals(HUMAN_INCHES, heightViaHql, 0.01d);
124 		
125 		// Test restriction and entity load via criteria
126 		Being b = (Being)s.createCriteria(Being.class)
127 			.add(Restrictions.between("heightInches", HUMAN_INCHES - 0.01d, HUMAN_INCHES + 0.01d))
128 			.uniqueResult();
129 		assertEquals(HUMAN_INCHES, b.getHeightInches(), 0.01d);
130 		
131 		// Test predicate and entity load via HQL
132 		b = (Being)s.createQuery("from Being b where b.heightInches between ? and ?")
133 			.setDouble(0, ALIEN_INCHES - 0.01d)
134 			.setDouble(1, ALIEN_INCHES + 0.01d)
135 			.uniqueResult();
136 		assertEquals(ALIEN_INCHES, b.getHeightInches(), 0.01d);
137                 s.delete(gavin);
138                 s.delete(x23y4);		
139 		t.commit();
140 		s.close();
141 		
142 	}
