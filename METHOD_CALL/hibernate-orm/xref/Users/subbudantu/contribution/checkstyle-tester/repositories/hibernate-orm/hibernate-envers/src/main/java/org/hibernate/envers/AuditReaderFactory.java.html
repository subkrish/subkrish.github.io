24  package org.hibernate.envers;
25  
26  import javax.persistence.EntityManager;
27  
28  import org.hibernate.Session;
29  import org.hibernate.engine.spi.SessionImplementor;
30  import org.hibernate.envers.event.EnversListener;
31  import org.hibernate.envers.exception.AuditException;
32  import org.hibernate.envers.reader.AuditReaderImpl;
33  import org.hibernate.event.service.spi.EventListenerRegistry;
34  import org.hibernate.event.spi.EventType;
35  import org.hibernate.event.spi.PostInsertEventListener;
36  
37  /**
38   * @author Adam Warski (adam at warski dot org)
39   */
40  public class AuditReaderFactory {
41      private AuditReaderFactory() { }
42  
43      /**
44       * Create an audit reader associated with an open session.
45       * @param session An open session.
46       * @return An audit reader associated with the given sesison. It shouldn't be used
47       * after the session is closed.
48       * @throws AuditException When the given required listeners aren't installed.
49       */
50      public static AuditReader get(Session session) throws AuditException {
51          SessionImplementor sessionImpl;
52  		if (!(session instanceof SessionImplementor)) {
53  			sessionImpl = (SessionImplementor) session.getSessionFactory().getCurrentSession();
54  		} else {
55  			sessionImpl = (SessionImplementor) session;
56  		}
57  
58  		// todo : I wonder if there is a better means to do this via "named lookup" based on the session factory name/uuid
59  		final EventListenerRegistry listenerRegistry = sessionImpl
60  				.getFactory()
61  				.getServiceRegistry()
62  				.getService( EventListenerRegistry.class );
63  
64  		for ( PostInsertEventListener listener : listenerRegistry.getEventListenerGroup( EventType.POST_INSERT ).listeners() ) {
65  			if ( listener instanceof EnversListener ) {
66  				// todo : slightly different from original code in that I am not checking the other listener groups...
67  				return new AuditReaderImpl(
68  						( (EnversListener) listener ).getAuditConfiguration(),
69  						session,
70  						sessionImpl
71  				);
72  			}
73  		}
74  
75          throw new AuditException( "Envers listeners were not properly registered" );
76      }
77  
78      /**
79       * Create an audit reader associated with an open entity manager.
80       * @param entityManager An open entity manager.
81       * @return An audit reader associated with the given entity manager. It shouldn't be used
82       * after the entity manager is closed.
83       * @throws AuditException When the given entity manager is not based on Hibernate, or if the required
84       * listeners aren't installed.
85       */
86      public static AuditReader get(EntityManager entityManager) throws AuditException {
87          if (entityManager.getDelegate() instanceof Session) {
88              return get((Session) entityManager.getDelegate());
89          }
90  
91          if (entityManager.getDelegate() instanceof EntityManager) {
92              return get((EntityManager) entityManager.getDelegate());
93          }
94  
95          throw new AuditException("Hibernate EntityManager not present!");
96      }
97  }
