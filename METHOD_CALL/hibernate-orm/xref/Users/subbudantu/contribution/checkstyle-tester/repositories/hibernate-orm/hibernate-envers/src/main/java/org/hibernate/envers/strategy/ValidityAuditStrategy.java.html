46  /**
47   *  Audit strategy which persists and retrieves audit information using a validity algorithm, based on the 
48   *  start-revision and end-revision of a row in the audit tables. 
49   *  <p>This algorithm works as follows:
50   *  <ul>
51   *  <li>For a <strong>new row</strong> that is persisted in an audit table, only the <strong>start-revision</strong> column of that row is set</li>
52   *  <li>At the same time the <strong>end-revision</strong> field of the <strong>previous</strong> audit row is set to this revision</li>
53   *  <li>Queries are retrieved using 'between start and end revision', instead of a subquery.</li>
54   *  </ul>
55   *  </p>
56   *  
57   *  <p>
58   *  This has a few important consequences that need to be judged against against each other:
59   *  <ul>
60   *  <li>Persisting audit information is a bit slower, because an extra row is updated</li>
61   *  <li>Retrieving audit information is a lot faster</li>
62   *  </ul>
63   *  </p>
64   * 
65   * @author Stephanie Pau
66   * @author Adam Warski (adam at warski dot org)
67   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
68   */
69  public class ValidityAuditStrategy implements AuditStrategy {

81      public void perform(
82  			final Session session,
83  			String entityName,
84  			final AuditConfiguration auditCfg,
85  			final Serializable id,
86  			Object data,
87  			final Object revision) {
88          final AuditEntitiesConfiguration audEntitiesCfg = auditCfg.getAuditEntCfg();
89          final String auditedEntityName = audEntitiesCfg.getAuditEntityName( entityName );
90  		final String revisionInfoEntityName = auditCfg.getAuditEntCfg().getRevisionInfoEntityName();
91  		final SessionImplementor sessionImplementor = (SessionImplementor) session;
92  		final Dialect dialect = sessionImplementor.getFactory().getDialect();
93  
94          // Save the audit data
95          session.save(auditedEntityName, data);
96          sessionCacheCleaner.scheduleAuditDataRemoval(session, data);
97  
98          // Update the end date of the previous row if this operation is expected to have a previous row
99          if (getRevisionType(auditCfg, data) != RevisionType.ADD) {
100 			final Queryable productionEntityQueryable = getQueryable( entityName, sessionImplementor );
101 			final Queryable rootProductionEntityQueryable = getQueryable( productionEntityQueryable.getRootEntityName(), sessionImplementor );
102 			final Queryable auditedEntityQueryable = getQueryable( auditedEntityName, sessionImplementor );
103 			final Queryable rootAuditedEntityQueryable = getQueryable( auditedEntityQueryable.getRootEntityName(), sessionImplementor );
104 			final Queryable revisionInfoEntityQueryable = getQueryable( revisionInfoEntityName, sessionImplementor );
105 
106 			final String updateTableName;
107 			if ( UnionSubclassEntityPersister.class.isInstance( rootProductionEntityQueryable ) ) {
108 				// this is the condition causing all the problems in terms of the generated SQL UPDATE
109 				// the problem being that we currently try to update the in-line view made up of the union query
110 				//
111 				// this is extremely hacky means to get the root table name for the union subclass style entities.
112 				// hacky because it relies on internal behavior of UnionSubclassEntityPersister
113 				// !!!!!! NOTICE - using subclass persister, not root !!!!!!
114 				updateTableName = auditedEntityQueryable.getSubclassTableName( 0 );
115 			}
116 			else {
117 				updateTableName = rootAuditedEntityQueryable.getTableName();
118 			}
119 
120 
121 			// first we need to flush the session in order to have the new audit data inserted
122 			// todo: expose org.hibernate.internal.SessionImpl.autoFlushIfRequired via SessionImplementor
123 			// for now, we duplicate some of that logic here
124 			autoFlushIfRequired( sessionImplementor, rootAuditedEntityQueryable, revisionInfoEntityQueryable );
125 
126 			final Type revisionInfoIdType = sessionImplementor.getFactory()
127 					.getEntityPersister( revisionInfoEntityName )
128 					.getIdentifierType();
129 			final String revEndColumnName = rootAuditedEntityQueryable.toColumns( auditCfg.getAuditEntCfg().getRevisionEndFieldName() )[0];
130 
131 			final boolean isRevisionEndTimestampEnabled = auditCfg.getAuditEntCfg().isRevisionEndTimestampEnabled();
132 
133 			// update audit_ent set REVEND = ? [, REVEND_TSTMP = ?] where (prod_ent_id) = ? and REV <> ? and REVEND is null
134 			final Update update = new Update( dialect ).setTableName( updateTableName );
135 			// set REVEND = ?
136 			update.addColumn( revEndColumnName );
137 			// set [, REVEND_TSTMP = ?]
138 			if ( isRevisionEndTimestampEnabled ) {
139 				update.addColumn(
140 						rootAuditedEntityQueryable.toColumns(
141 								auditCfg.getAuditEntCfg().getRevisionEndTimestampFieldName()
142 						)[0]
143 				);
144 			}
145 
146 			// where (prod_ent_id) = ?
147 			update.addPrimaryKeyColumns( rootProductionEntityQueryable.getIdentifierColumnNames() );
148 			// where REV <> ?
149 			update.addWhereColumn(
150 					rootAuditedEntityQueryable.toColumns(
151 							auditCfg.getAuditEntCfg().getRevisionNumberPath()
152 					)[0],
153 					"<> ?"
154 			);
155 			// where REVEND is null
156 			update.addWhereColumn( revEndColumnName, " is null" );
157 
158 			// Now lets execute the sql...
159 			final String updateSql = update.toStatementString();
160 
161 			int rowCount = session.doReturningWork(
162 					new ReturningWork<Integer>() {
163 						@Override
164 						public Integer execute(Connection connection) throws SQLException {
165 							PreparedStatement preparedStatement = sessionImplementor.getTransactionCoordinator().getJdbcCoordinator().getStatementPreparer().prepareStatement( updateSql );
166 
167 							try {
168 								int index = 1;
169 
170 								// set REVEND = ?
171 								final Number revisionNumber = auditCfg.getRevisionInfoNumberReader().getRevisionNumber( revision );
172 								revisionInfoIdType.nullSafeSet( preparedStatement, revisionNumber, index, sessionImplementor );
173 								index += revisionInfoIdType.getColumnSpan( sessionImplementor.getFactory() );
174 
175 								// set [, REVEND_TSTMP = ?]
176 								if ( isRevisionEndTimestampEnabled ) {
177 									final Object revEndTimestampObj = revisionTimestampGetter.get( revision );
178 									final Date revisionEndTimestamp = convertRevEndTimestampToDate( revEndTimestampObj );
179 									final Type revEndTsType = rootAuditedEntityQueryable.getPropertyType(
180 											auditCfg.getAuditEntCfg().getRevisionEndTimestampFieldName()
181 									);
182 									revEndTsType.nullSafeSet( preparedStatement, revisionEndTimestamp, index, sessionImplementor );
183 									index += revEndTsType.getColumnSpan( sessionImplementor.getFactory() );
184 								}
185 
186 								// where (prod_ent_id) = ?
187 								final Type idType = rootProductionEntityQueryable.getIdentifierType();
188 								idType.nullSafeSet( preparedStatement, id, index, sessionImplementor );
189 								index += idType.getColumnSpan( sessionImplementor.getFactory() );
190 
191 								// where REV <> ?
192 								final Type revType = rootAuditedEntityQueryable.getPropertyType(
193 										auditCfg.getAuditEntCfg().getRevisionNumberPath()
194 								);
195 								revType.nullSafeSet( preparedStatement, revisionNumber, index, sessionImplementor );
196 
197 								// where REVEND is null
198 								// 		nothing to bind....
199 
200 								return sessionImplementor.getTransactionCoordinator().getJdbcCoordinator().getResultSetReturn().executeUpdate( preparedStatement );
201 							}
202 							finally {
203 								sessionImplementor.getTransactionCoordinator().getJdbcCoordinator().release( preparedStatement );
204 							}
205 						}
206 					}
207 			);
208 
209 			if ( rowCount != 1 ) {
210 				throw new RuntimeException(
211 						"Cannot update previous revision for entity " + auditedEntityName + " and id " + id
212 				);
213 			}
214 		}
215     }

221 	private void autoFlushIfRequired(
222 			SessionImplementor sessionImplementor,
223 			Queryable auditedEntityQueryable,
224 			Queryable revisionInfoEntityQueryable) {
225 		final Set<String> querySpaces = new HashSet<String>();
226 		querySpaces.add( auditedEntityQueryable.getTableName() );
227 		querySpaces.add( revisionInfoEntityQueryable.getTableName() );
228 		final AutoFlushEvent event = new AutoFlushEvent( querySpaces, (EventSource) sessionImplementor );
229 		final Iterable<AutoFlushEventListener> listeners = sessionImplementor.getFactory().getServiceRegistry()
230 				.getService( EventListenerRegistry.class )
231 				.getEventListenerGroup( EventType.AUTO_FLUSH )
232 				.listeners();
233 		for ( AutoFlushEventListener listener : listeners ) {
234 			listener.onAutoFlush( event );
235 		}
236 	}

238     @SuppressWarnings({"unchecked"})
239     public void performCollectionChange(Session session, String entityName, String propertyName, AuditConfiguration auditCfg,
240                                         PersistentCollectionChangeData persistentCollectionChangeData, Object revision) {
241         final QueryBuilder qb = new QueryBuilder(persistentCollectionChangeData.getEntityName(), MIDDLE_ENTITY_ALIAS);
242 
243         final String originalIdPropName = auditCfg.getAuditEntCfg().getOriginalIdPropName();
244         final Map<String, Object> originalId = (Map<String, Object>) persistentCollectionChangeData.getData().get(originalIdPropName);
245 		final String revisionFieldName = auditCfg.getAuditEntCfg().getRevisionFieldName();
246 		final String revisionTypePropName = auditCfg.getAuditEntCfg().getRevisionTypePropName();
247 
248 		// Adding a parameter for each id component, except the rev number and type.
249         for (Map.Entry<String, Object> originalIdEntry : originalId.entrySet()) {
250             if (!revisionFieldName.equals(originalIdEntry.getKey()) && !revisionTypePropName.equals(originalIdEntry.getKey())) {
251                 qb.getRootParameters().addWhereWithParam(originalIdPropName + "." + originalIdEntry.getKey(),
252                         true, "=", originalIdEntry.getValue());
253             }
254         }
255 
256 		final SessionFactoryImplementor sessionFactory = ( (SessionImplementor) session ).getFactory();
257 		final Type propertyType = sessionFactory.getEntityPersister( entityName ).getPropertyType( propertyName );
258 		if ( propertyType.isCollectionType() ) {
259 			CollectionType collectionPropertyType = (CollectionType) propertyType;
260 			// Handling collection of components.
261 			if ( collectionPropertyType.getElementType( sessionFactory ) instanceof ComponentType ) {
262 				// Adding restrictions to compare data outside of primary key.
263 				for ( Map.Entry<String, Object> dataEntry : persistentCollectionChangeData.getData().entrySet() ) {
264 					if ( !originalIdPropName.equals( dataEntry.getKey() ) ) {
265 						qb.getRootParameters().addWhereWithParam( dataEntry.getKey(), true, "=", dataEntry.getValue() );
266 					}
267 				}
268 			}
269 		}
270 
271         addEndRevisionNullRestriction(auditCfg, qb.getRootParameters());
272 
273         final List<Object> l = qb.toQuery(session).setLockOptions(LockOptions.UPGRADE).list();
274 
275         // Update the last revision if one exists.
276         // HHH-5967: with collections, the same element can be added and removed multiple times. So even if it's an
277         // ADD, we may need to update the last revision.
278         if (l.size() > 0) {
279             updateLastRevision(session, auditCfg, l, originalId, persistentCollectionChangeData.getEntityName(), revision);
280         }
281 
282         // Save the audit data
283         session.save(persistentCollectionChangeData.getEntityName(), persistentCollectionChangeData.getData());
284         sessionCacheCleaner.scheduleAuditDataRemoval(session, persistentCollectionChangeData.getData());
285     }

287     private void addEndRevisionNullRestriction(AuditConfiguration auditCfg, Parameters rootParameters) {
288         rootParameters.addWhere(auditCfg.getAuditEntCfg().getRevisionEndFieldName(), true, "is", "null", false);

309     private void addRevisionRestriction(Parameters rootParameters, String revisionProperty, String revisionEndProperty,
310 										boolean addAlias, boolean inclusive) {
311 		// e.revision <= _revision and (e.endRevision > _revision or e.endRevision is null)
312 		Parameters subParm = rootParameters.addSubParameters("or");
313 		rootParameters.addWhereWithNamedParam(revisionProperty, addAlias, inclusive ? "<=" : "<", REVISION_PARAMETER);
314 		subParm.addWhereWithNamedParam(revisionEndProperty + ".id", addAlias, inclusive ? ">" : ">=", REVISION_PARAMETER);
315 		subParm.addWhere(revisionEndProperty, addAlias, "is", "null", false);
316 	}

323     @SuppressWarnings({"unchecked"})
324     private void updateLastRevision(Session session, AuditConfiguration auditCfg, List<Object> l,
325                                     Object id, String auditedEntityName, Object revision) {
326         // There should be one entry
327         if (l.size() == 1) {
328             // Setting the end revision to be the current rev
329             Object previousData = l.get(0);
330             String revisionEndFieldName = auditCfg.getAuditEntCfg().getRevisionEndFieldName();
331             ((Map<String, Object>) previousData).put(revisionEndFieldName, revision);
332 
333             if (auditCfg.getAuditEntCfg().isRevisionEndTimestampEnabled()) {
334                 // Determine the value of the revision property annotated with @RevisionTimestamp
335             	String revEndTimestampFieldName = auditCfg.getAuditEntCfg().getRevisionEndTimestampFieldName();
336             	Object revEndTimestampObj = this.revisionTimestampGetter.get(revision);
337                Date revisionEndTimestamp = convertRevEndTimestampToDate(revEndTimestampObj);
338 
339             	// Setting the end revision timestamp
340             	((Map<String, Object>) previousData).put(revEndTimestampFieldName, revisionEndTimestamp);
341             }
342             
343             // Saving the previous version
344             session.save(auditedEntityName, previousData);
345             sessionCacheCleaner.scheduleAuditDataRemoval(session, previousData);
346         } else {
347             throw new RuntimeException("Cannot find previous revision for entity " + auditedEntityName + " and id " + id);
348         }
349     }
350 
351     private Date convertRevEndTimestampToDate(Object revEndTimestampObj) {
352         // convert to a java.util.Date
353         if (revEndTimestampObj instanceof Date) {
354             return (Date) revEndTimestampObj;
355         }
356         return new Date((Long) revEndTimestampObj);
357     }
