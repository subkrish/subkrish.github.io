19  /**
20   * Copied & modified from {@link org.junit.runners.Parameterized}.
21   *
22   * The modification is that the generated runners extend {@link CustomRunner} instead of the default
23   * {@code TestClassRunnerForParameters}.
24   *
25   * The runner itself sets the data using a setter instead of a constructor, and creates only one test instance. Moreover
26   * it doesn't override {@code classBlock} which causes the custom {@code @BeforeClassOnce} and {@code @AfterClassOnce}
27   * annotations to work.
28   * @author Adam Warski (adam at warski dot org)
29   */
30  public class EnversRunner extends Suite {
31  	private class TestClassCustomRunnerForParameters extends CustomRunner {
32  		private final int fParameterSetNumber;
33  
34  		private final List<Object[]> fParameterList;
35  
36  		TestClassCustomRunnerForParameters(Class<?> type, List<Object[]> parameterList, int i)
37                  throws InitializationError, NoTestsRemainException {
38  			super(type);
39  			fParameterList= parameterList;
40  			fParameterSetNumber= i;
41  		}
42  
43          @Override
44          protected Object getTestInstance() throws Exception {
45              Object testInstance = super.getTestInstance();
46              if ( AbstractEnversTest.class.isInstance( testInstance ) ) {
47                  ( (AbstractEnversTest) testInstance ).setTestData( computeParams() );
48              }
49              else if ( BaseEnversFunctionalTestCase.class.isInstance( testInstance ) ) {
50                  ( (BaseEnversFunctionalTestCase) testInstance ).setTestData( computeParams() );
51              }
52              return testInstance;
53          }
54  
55  		private Object[] computeParams() throws Exception {
56  			try {
57  				return fParameterList.get(fParameterSetNumber);
58  			} catch (ClassCastException e) {
59  				throw new Exception(String.format(
60  						"%s.%s() must return a Collection of arrays.",
61  						getTestClass().getName(), getParametersMethod(
62  								getTestClass()).getName()));
63  			}
64  		}
65  
66  		@Override
67  		protected String getName() {
68  			return String.format("[%s]", fParameterSetNumber);
69  		}
70  
71  		@Override
72  		protected String testName(final FrameworkMethod method) {
73  			return String.format("%s[%s]", method.getName(),
74                      fParameterSetNumber);
75  		}
76  
77  		@Override
78  		protected void sortMethods(List<FrameworkMethod> computedTestMethods) {
79  			super.sortMethods( computedTestMethods );
80  			Collections.sort(computedTestMethods, new Comparator<FrameworkMethod>() {
81  				private int getPriority(FrameworkMethod fm) {
82  					Priority p = fm.getAnnotation(Priority.class);
83  					return p == null ? 0 : p.value();
84  				}
85  
86  				@Override
87  				public int compare(FrameworkMethod fm1, FrameworkMethod fm2) {
88  					return getPriority(fm2) - getPriority(fm1);
89  				}
90  			});
91  		}
92      }
93  
94  	private final ArrayList<Runner> runners= new ArrayList<Runner>();
95  
96  	/**
97  	 * Only called reflectively. Do not use programmatically.
98  	 */
99  	public EnversRunner(Class<?> klass) throws Throwable {
100 		super(klass, Collections.<Runner>emptyList());
101 		List<Object[]> parametersList= getParametersList(getTestClass());
102 		for (int i= 0; i < parametersList.size(); i++)
103 			runners.add(new TestClassCustomRunnerForParameters(getTestClass().getJavaClass(),
104 					parametersList, i));
105 	}
106 
107 	@Override
108 	protected List<Runner> getChildren() {
109 		return runners;
110 	}
111 
112 	@SuppressWarnings("unchecked")
113 	private List<Object[]> getParametersList(TestClass klass)
114 			throws Throwable {
115 		return (List<Object[]>) getParametersMethod(klass).invokeExplosively(
116 				null);
117 	}
118 
119 	private FrameworkMethod getParametersMethod(TestClass testClass)
120 			throws Exception {
121 		List<FrameworkMethod> methods= testClass.getAnnotatedMethods(Parameterized.Parameters.class);
122 
123 		for (FrameworkMethod each : methods) {
124 			int modifiers= each.getMethod().getModifiers();
125 			if (Modifier.isStatic(modifiers) && Modifier.isPublic(modifiers))
126 				return each;
127 		}
128 
129 		throw new Exception("No public static parameters method on class "
130 				+ testClass.getName());
131 	}
