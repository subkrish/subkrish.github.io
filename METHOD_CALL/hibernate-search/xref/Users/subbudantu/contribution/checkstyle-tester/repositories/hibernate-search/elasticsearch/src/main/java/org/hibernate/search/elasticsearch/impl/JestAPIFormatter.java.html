11  import org.hibernate.search.engine.service.spi.Service;

17  import com.google.gson.Gson;
18  import com.google.gson.JsonElement;
19  import com.google.gson.JsonParseException;
20  
21  import io.searchbox.action.Action;
22  import io.searchbox.action.DocumentTargetedAction;
23  import io.searchbox.client.JestResult;
24  import io.searchbox.core.BulkResult;
25  import io.searchbox.core.BulkResult.BulkResultItem;

27  /**
28   * @author Yoann Rodiere
29   */
30  public class JestAPIFormatter implements Service, Startable, Stoppable {
31  
32  	private ServiceManager serviceManager;
33  	private GsonService gsonService;
34  
35  	@Override
36  	public void start(Properties properties, BuildContext context) {
37  		serviceManager = context.getServiceManager();
38  		gsonService = serviceManager.requestService( GsonService.class );
39  	}
40  
41  	@Override
42  	public void stop() {
43  		gsonService = null;
44  		serviceManager.releaseService( GsonService.class );
45  		serviceManager = null;
46  	}
47  
48  	public String formatRequest(Action<?> action) {
49  		StringBuilder sb = new StringBuilder();
50  
51  		sb.append( "Operation: " ).append( action.getClass().getSimpleName() ).append( "\n" );
52  		sb.append( "URI:" ).append( action.getURI() ).append( "\n" );
53  
54  		if ( action instanceof DocumentTargetedAction ) {
55  			sb.append( "Index: " ).append( ( (DocumentTargetedAction<?>) action ).getIndex() ).append( "\n" );
56  			sb.append( "Type: " ).append( ( (DocumentTargetedAction<?>) action ).getType() ).append( "\n" );
57  			sb.append( "Id: " ).append( ( (DocumentTargetedAction<?>) action ).getId() ).append( "\n" );
58  		}
59  
60  		sb.append( "Data:\n" );
61  		sb.append( formatRequestData( action ) );
62  		sb.append( "\n" );
63  		return sb.toString();
64  	}
65  
66  	public String formatRequestData(Action<?> search) {
67  		Gson gson = gsonService.getGson();
68  		/*
69  		 * The Gson we use as an argument is not always used, which means we have to reformat
70  		 * the result ourselves to be sure it's pretty-printed.
71  		 */
72  		String data = search.getData( gson );
73  
74  		// Make sure the JSON is pretty-printed
75  		try {
76  			gson = gsonService.getGsonPrettyPrinting();
77  			JsonElement tree = gson.fromJson( data, JsonElement.class );
78  			return gson.toJson( tree );
79  		}
80  		catch (JsonParseException e) {
81  			/*
82  			 * Sometimes, for example for bulk requests, the data isn't really JSON,
83  			 * but rather multiple "\n"-separated JSON documents.
84  			 * In that case we just give up and return the raw, potentially not
85  			 * pretty-printed data.
86  			 */
87  			return data;
88  		}
89  	}
90  
91  	public String formatResult(JestResult result) {
92  		StringBuilder sb = new StringBuilder();
93  		sb.append( "Status: " ).append( result.getResponseCode() ).append( "\n" );
94  		sb.append( "Error message: " ).append( result.getErrorMessage() ).append( "\n" );
95  		sb.append( "Cluster name: " ).append( property( result, "cluster_name" ) ).append( "\n" );
96  		sb.append( "Cluster status: " ).append( property( result, "status" ) ).append( "\n" );
97  		sb.append( "\n" );
98  
99  		if ( result instanceof BulkResult ) {
100 			for ( BulkResultItem item : ( (BulkResult) result ).getItems() ) {
101 				sb.append( "Operation: " ).append( item.operation ).append( "\n" );
102 				sb.append( "  Index: " ).append( item.index ).append( "\n" );
103 				sb.append( "  Type: " ).append( item.type ).append( "\n" );
104 				sb.append( "  Id: " ).append( item.id ).append( "\n" );
105 				sb.append( "  Status: " ).append( item.status ).append( "\n" );
106 				sb.append( "  Error: " ).append( item.error ).append( "\n" );
107 			}
108 		}
109 
110 		return sb.toString();
111 	}
112 
113 	private String property(JestResult result, String name) {
114 		if ( result.getJsonObject() == null ) {
115 			return null;
116 		}
117 		JsonElement propretyValue = result.getJsonObject().get( name );
118 		if ( propretyValue == null ) {
119 			return null;
120 		}
121 		return propretyValue.getAsString();
122 	}
