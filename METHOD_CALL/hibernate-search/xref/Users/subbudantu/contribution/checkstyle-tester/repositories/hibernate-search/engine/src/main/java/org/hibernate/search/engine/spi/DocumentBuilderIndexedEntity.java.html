151 	/**
152 	 * Creates a document builder for entities annotated with <code>@Indexed</code>.
153 	 *
154 	 * @param clazz The class for which to build a <code>DocumentBuilderContainedEntity</code>
155 	 * @param typeMetadata all the metadata for the entity type
156 	 * @param context Handle to default configuration settings
157 	 * @param reflectionManager Reflection manager to use for processing the annotations
158 	 * @param optimizationBlackList mutable register, keeps track of types on which we need to disable collection events optimizations
159 	 * @param instanceInitializer helper class for class object graph navigation
160 	 */
161 	public DocumentBuilderIndexedEntity(XClass clazz, TypeMetadata typeMetadata, ConfigContext context,
162 			ReflectionManager reflectionManager, Set<XClass> optimizationBlackList, InstanceInitializer instanceInitializer) {
163 		super( clazz, typeMetadata, reflectionManager, optimizationBlackList, instanceInitializer );
164 
165 		ProvidedId providedIdAnnotation = findProvidedId( clazz, reflectionManager );
166 		if ( providedIdAnnotation != null || context.isProvidedIdImplicit() ) {
167 			idProvided = true;
168 		}
169 
170 		if ( idPropertyMetadata == null ) {
171 			idPropertyMetadata = getTypeMetadata().getIdPropertyMetadata();
172 		}
173 
174 		if ( idPropertyMetadata == null ) {
175 			throw log.noDocumentIdFoundException( clazz.getName() );
176 		}
177 
178 		idFieldName = idPropertyMetadata.getFieldMetadataSet().iterator().next().getAbsoluteName();
179 
180 		checkAllowFieldSelection();
181 		if ( log.isDebugEnabled() ) {
182 			log.debugf(
183 					"Field selection in projections is set to %b for entity %s.",
184 					allowFieldSelectionInProjection,
185 					clazz
186 			);
187 		}
188 		this.entityState = EntityState.INDEXED;
189 		this.idPropertyName = idProvided ? null : idPropertyMetadata.getPropertyAccessor().getName();
190 
191 		this.nestingContextFactory = getNestingContextFactory( context );
192 	}

194 	private static NestingContextFactory getNestingContextFactory(ConfigContext context) {
195 		try {
196 			NestingContextFactoryProvider nestingContextFactoryProvider = context.getServiceManager().requestService( NestingContextFactoryProvider.class );
197 			return nestingContextFactoryProvider.getNestingContextFactory();
198 		}
199 		catch (Exception e) {
200 			return DefaultNestingContextFactory.INSTANCE;
201 		}
202 		finally {
203 			context.getServiceManager().releaseService( NestingContextFactoryProvider.class );
204 		}
205 	}

211 	private ProvidedId findProvidedId(XClass clazz, ReflectionManager reflectionManager) {
212 		ProvidedId id = null;
213 		XClass currentClass = clazz;
214 		while ( id == null && ( !reflectionManager.equals( currentClass, Object.class ) ) ) {
215 			id = currentClass.getAnnotation( ProvidedId.class );
216 			currentClass = currentClass.getSuperclass();
217 		}
218 		return id;
219 	}

221 	@Override
222 	public void addWorkToQueue(String tenantId, Class<?> entityClass, Object entity, Serializable id, boolean delete, boolean add, List<LuceneWork> queue, ConversionContext contextualBridge) {
223 		DocumentFieldMetadata idFieldMetadata = idPropertyMetadata.getFieldMetadata( idFieldName );
224 		String idInString = objectToString( getIdBridge(), idFieldMetadata.getAbsoluteName(), id, contextualBridge );
225 		if ( delete && !add ) {
226 			queue.add( new DeleteLuceneWork( tenantId, id, idInString, entityClass ) );
227 		}
228 		else if ( add && !delete ) {
229 			queue.add(
230 					createAddWork(
231 							tenantId,
232 							entityClass,
233 							entity,
234 							id,
235 							idInString,
236 							getInstanceInitializer(),
237 							contextualBridge
238 					)
239 			);
240 		}
241 		else if ( add && delete ) {
242 			queue.add(
243 					createUpdateWork(
244 							tenantId,
245 							entityClass,
246 							entity,
247 							id,
248 							idInString,
249 							getInstanceInitializer(),
250 							contextualBridge
251 					)
252 			);
253 		}
254 	}

256 	private String objectToString(TwoWayFieldBridge bridge, String fieldName, Object value, ConversionContext conversionContext) {
257 		conversionContext.pushProperty( fieldName );
258 		String stringValue;
259 		try {
260 			stringValue = conversionContext
261 					.setClass( getBeanClass() )
262 					.twoWayConversionContext( bridge )
263 					.objectToString( value );
264 		}
265 		finally {
266 			conversionContext.popProperty();
267 		}
268 		return stringValue;
269 	}

287 		Map<String, String> fieldToAnalyzerMap = new HashMap<String, String>();

290 		if ( fieldToAnalyzerMap.isEmpty() ) {

296 		return addWork;

312 	/**
313 	 * Builds the Lucene {@code Document} for a given entity instance and its id.
314 	 *
315 	 * @param tenantId the identifier of the tenant or null if there isn't one
316 	 * @param instance The entity for which to build the matching Lucene {@code Document}
317 	 * @param id the entity id.
318 	 * @param fieldToAnalyzerMap this maps gets populated while generating the {@code Document}.
319 	 * It allows to specify for any document field a named analyzer to use. This parameter cannot be {@code null}.
320 	 * @param objectInitializer used to ensure that all objects are initialized
321 	 * @param conversionContext a {@link org.hibernate.search.bridge.spi.ConversionContext} object.
322 	 * @param includedFieldNames list of field names to consider. Others can be excluded. Null if all fields are considered.
323 	 *
324 	 * @return The Lucene {@code Document} for the specified entity.
325 	 */
326 	public Document getDocument(
327 			String tenantId,
328 			Object instance,
329 			Serializable id,
330 			Map<String, String> fieldToAnalyzerMap,
331 			InstanceInitializer objectInitializer,
332 			ConversionContext conversionContext,
333 			String[] includedFieldNames) {
334 		// TODO as it is, includedFieldNames is not generally useful as we don't know if a field bridge creates specific fields or not
335 		// TODO only used at the moment to filter the id field and the class field
336 
337 		if ( fieldToAnalyzerMap == null ) {
338 			throw new IllegalArgumentException( "fieldToAnalyzerMap cannot be null" );
339 		}
340 
341 		//sensible default for outside callers
342 		if ( objectInitializer == null ) {
343 			objectInitializer = getInstanceInitializer();
344 		}
345 
346 		Document doc = new Document();
347 		FacetHandling faceting = new FacetHandling();
348 		Class<?> entityType = objectInitializer.getClass( instance );
349 		float documentLevelBoost = getMetadata().getClassBoost( instance );
350 
351 		// add the class name of the entity to the document
352 		if ( containsFieldName( ProjectionConstants.OBJECT_CLASS, includedFieldNames ) ) {
353 			@SuppressWarnings( "deprecation" )
354 			Field classField =
355 					new Field(
356 							ProjectionConstants.OBJECT_CLASS,
357 							entityType.getName(),
358 							Field.Store.YES,
359 							Field.Index.NOT_ANALYZED_NO_NORMS,
360 							Field.TermVector.NO
361 					);
362 			doc.add( classField );
363 		}
364 
365 		addTenantIdIfRequired( tenantId, doc );
366 
367 		// now add the entity id to the document
368 		if ( containsFieldName( idFieldName, includedFieldNames ) ) {
369 			DocumentFieldMetadata idFieldMetaData = idPropertyMetadata.getFieldMetadata( idFieldName );
370 			LuceneOptions luceneOptions = new LuceneOptionsImpl( idFieldMetaData, idFieldMetaData.getBoost(), documentLevelBoost );
371 			final FieldBridge contextualizedBridge = conversionContext.oneWayConversionContext( getIdBridge() );
372 			conversionContext.setClass( entityType );
373 
374 			if ( idPropertyName != null ) {
375 				conversionContext.pushProperty( idPropertyName );
376 			}
377 
378 			try {
379 				contextualizedBridge.set( idFieldMetaData.getAbsoluteName(), id, doc, luceneOptions );
380 				addSortFieldDocValues( doc, idPropertyMetadata, documentLevelBoost, id );
381 			}
382 			finally {
383 				if ( idPropertyName != null ) {
384 					conversionContext.popProperty();
385 				}
386 			}
387 		}
388 
389 		// finally add all other document fields
390 		Set<String> processedFieldNames = new HashSet<>();
391 		buildDocumentFields(
392 				instance,
393 				doc,
394 				faceting,
395 				getMetadata(),
396 				fieldToAnalyzerMap,
397 				processedFieldNames,
398 				conversionContext,
399 				objectInitializer,
400 				documentLevelBoost,
401 				false,
402 				nestingContextFactory.createNestingContext( getTypeMetadata().getType() )
403 		);
404 
405 
406 		doc = faceting.build( doc );
407 		return doc;
408 	}

435 	 * @param inheritedBoost Boost inherited from the parent structure of the given instance: the document-level boost
436 	 * in case of a top-level field, the product of the document-level boost and the boost(s) of the parent
437 	 * embeddable(s) in case of an embedded field
438 	 */
439 	private void buildDocumentFields(Object instance,
440 			Document doc,
441 			FacetHandling faceting,
442 			TypeMetadata typeMetadata,
443 			Map<String, String> fieldToAnalyzerMap,
444 			Set<String> processedFieldNames,
445 			ConversionContext conversionContext,
446 			InstanceInitializer objectInitializer,
447 			final float inheritedBoost,
448 			boolean isParentPropertyMultiValued,
449 			NestingContext nestingContext) {
450 
451 		// needed for field access: I cannot work in the proxied version
452 		Object unproxiedInstance = unproxy( instance, objectInitializer );
453 
454 		buildDocumentFieldForClassBridges( doc, typeMetadata, conversionContext, inheritedBoost, unproxiedInstance );
455 		buildDocumentFieldsForProperties(
456 				doc,
457 				faceting,
458 				typeMetadata,
459 				conversionContext,
460 				objectInitializer,
461 				inheritedBoost,
462 				unproxiedInstance,
463 				isParentPropertyMultiValued
464 		);
465 
466 		// allow analyzer override for the fields added by the class and field bridges
467 		allowAnalyzerDiscriminatorOverride(
468 				doc, typeMetadata, fieldToAnalyzerMap, processedFieldNames, unproxiedInstance
469 		);
470 
471 		buildDocumentFieldsForEmbeddedObjects(
472 				doc,
473 				faceting,
474 				typeMetadata,
475 				fieldToAnalyzerMap,
476 				processedFieldNames,
477 				conversionContext,
478 				objectInitializer,
479 				inheritedBoost,
480 				unproxiedInstance,
481 				isParentPropertyMultiValued,
482 				nestingContext
483 		);
484 	}

486 	private void buildDocumentFieldsForEmbeddedObjects(Document doc,
487 			FacetHandling faceting,
488 			TypeMetadata typeMetadata,
489 			Map<String, String> fieldToAnalyzerMap,
490 			Set<String> processedFieldNames,
491 			ConversionContext conversionContext,
492 			InstanceInitializer objectInitializer,
493 			float inheritedBoost,
494 			Object unproxiedInstance,
495 			boolean isParentPropertyMultiValued,
496 			NestingContext nestingContext) {
497 		for ( EmbeddedTypeMetadata embeddedTypeMetadata : typeMetadata.getEmbeddedTypeMetadata() ) {
498 			XMember member = embeddedTypeMetadata.getEmbeddedGetter();
499 			float embeddedBoost = inheritedBoost * embeddedTypeMetadata.getStaticBoost();
500 
501 			conversionContext.pushProperty( embeddedTypeMetadata.getEmbeddedPropertyName() );
502 			nestingContext.push( embeddedTypeMetadata );
503 			try {
504 				Object value = ReflectionHelper.getMemberValue( unproxiedInstance, member );
505 				if ( value == null ) {
506 					/*
507 					 * This must be executed after the "nestingContext.push" so as to
508 					 * include this embedded's metadata in the nesting context (which
509 					 * will allow backends to detect a embedded null value)
510 					 */
511 					nestingContext.mark( doc );
512 					processEmbeddedNullValue( doc, embeddedTypeMetadata, conversionContext );
513 					continue;
514 				}
515 
516 				switch ( embeddedTypeMetadata.getEmbeddedContainer() ) {
517 					case ARRAY:
518 						if ( value instanceof Object[] ) {
519 							objectInitializer.initializeArray( (Object[]) value );
520 						}
521 						// Use CollectionHelper.iterableFromArray to also support arrays of primitive values
522 						for ( Object arrayValue : CollectionHelper.iterableFromArray( value ) ) {
523 							nestingContext.mark( doc );
524 							buildDocumentFields(
525 									arrayValue,
526 									doc,
527 									faceting,
528 									embeddedTypeMetadata,
529 									fieldToAnalyzerMap,
530 									processedFieldNames,
531 									conversionContext,
532 									objectInitializer,
533 									embeddedBoost,
534 									true,
535 									nestingContext
536 							);
537 							nestingContext.incrementCollectionIndex();
538 						}
539 						break;
540 					case COLLECTION:
541 						Collection<?> collection = objectInitializer.initializeCollection( (Collection<?>) value );
542 						for ( Object collectionValue : collection ) {
543 							nestingContext.mark( doc );
544 							buildDocumentFields(
545 									collectionValue,
546 									doc,
547 									faceting,
548 									embeddedTypeMetadata,
549 									fieldToAnalyzerMap,
550 									processedFieldNames,
551 									conversionContext,
552 									objectInitializer,
553 									embeddedBoost,
554 									true,
555 									nestingContext
556 							);
557 							nestingContext.incrementCollectionIndex();
558 						}
559 						break;
560 					case MAP:
561 						Map<?, ?> map = objectInitializer.initializeMap( (Map<?, ?>) value );
562 						for ( Object collectionValue : map.values() ) {
563 							nestingContext.mark( doc );
564 							buildDocumentFields(
565 									collectionValue,
566 									doc,
567 									faceting,
568 									embeddedTypeMetadata,
569 									fieldToAnalyzerMap,
570 									processedFieldNames,
571 									conversionContext,
572 									objectInitializer,
573 									embeddedBoost,
574 									true,
575 									nestingContext
576 							);
577 							nestingContext.incrementCollectionIndex();
578 						}
579 						break;
580 					case OBJECT:
581 						nestingContext.mark( doc );
582 						buildDocumentFields(
583 								value,
584 								doc,
585 								faceting,
586 								embeddedTypeMetadata,
587 								fieldToAnalyzerMap,
588 								processedFieldNames,
589 								conversionContext,
590 								objectInitializer,
591 								embeddedBoost,
592 								isParentPropertyMultiValued,
593 								nestingContext
594 						);
595 						break;
596 					default:
597 						throw new AssertionFailure(
598 								"Unknown embedded container: "
599 										+ embeddedTypeMetadata.getEmbeddedContainer()
600 						);
601 				}
602 			}
603 			finally {
604 				conversionContext.popProperty();
605 				nestingContext.pop();
606 			}
607 		}
608 	}

619 			InstanceInitializer objectInitializer,
620 			float documentBoost,
621 			Object unproxiedInstance,
622 			boolean isParentPropertyMultiValued) {
623 		XMember previousMember = null;
624 		Object currentFieldValue = null;
625 
626 		for ( PropertyMetadata propertyMetadata : typeMetadata.getAllPropertyMetadata() ) {
627 			XMember member = propertyMetadata.getPropertyAccessor();
628 			if ( previousMember != member ) {
629 				currentFieldValue = unproxy(
630 						ReflectionHelper.getMemberValue( unproxiedInstance, member ),
631 						objectInitializer
632 				);
633 				previousMember = member;
634 				if ( member.isCollection() ) {
635 					if ( currentFieldValue instanceof Collection ) {
636 						objectInitializer.initializeCollection( (Collection) currentFieldValue );
637 					}
638 					else if ( currentFieldValue instanceof Map ) {
639 						objectInitializer.initializeMap( (Map) currentFieldValue );
640 					}
641 				}
642 				else if ( member.isArray() && currentFieldValue instanceof Object[] ) {
643 					objectInitializer.initializeArray( (Object[]) currentFieldValue );
644 				}
645 			}
646 
647 			try {
648 				conversionContext.pushProperty( propertyMetadata.getPropertyAccessorName() );
649 
650 				for ( DocumentFieldMetadata fieldMetadata : propertyMetadata.getFieldMetadataSet() ) {
651 					final FieldBridge fieldBridge = fieldMetadata.getFieldBridge();
652 					final String fieldName = fieldMetadata.getAbsoluteName();
653 					final FieldBridge oneWayConversionContext = conversionContext.oneWayConversionContext(
654 							fieldBridge
655 					);
656 
657 					// handle the default field creation via the bridge
658 					oneWayConversionContext.set(
659 							fieldName,
660 							currentFieldValue,
661 							document,
662 							typeMetadata.getFieldLuceneOptions(
663 									propertyMetadata, fieldMetadata, currentFieldValue, documentBoost
664 							)
665 					);
666 
667 					// handle faceting fields
668 					if ( fieldMetadata.hasFacets() ) {
669 						faceting.enableFacetProcessing();
670 						for ( FacetMetadata facetMetadata : fieldMetadata.getFacetMetadata() ) {
671 							boolean multiValued = isParentPropertyMultiValued;
672 
673 							if ( member.isCollection() && currentFieldValue instanceof Collection ) {
674 								multiValued = true;
675 								for ( Object element : (Collection<?>) currentFieldValue ) {
676 									addFacetDocValues( document, fieldMetadata, facetMetadata, element );
677 								}
678 							}
679 							else if ( member.isCollection() && currentFieldValue instanceof Map ) {
680 								multiValued = true;
681 								for ( Object element : ((Map<?,?>) currentFieldValue).values() ) {
682 									addFacetDocValues( document, fieldMetadata, facetMetadata, element );
683 								}
684 							}
685 							else if ( member.isArray() ) {
686 								multiValued = true;
687 								// Use CollectionHelper.iterableFromArray to also support arrays of primitive values
688 								for ( Object element : CollectionHelper.iterableFromArray( currentFieldValue ) ) {
689 									addFacetDocValues( document, fieldMetadata, facetMetadata, element );
690 								}
691 							}
692 							else {
693 								addFacetDocValues( document, fieldMetadata, facetMetadata, currentFieldValue );
694 							}
695 
696 							if ( multiValued ) {
697 								faceting.setMultiValued( facetMetadata.getAbsoluteName() );
698 							}
699 						}
700 					}
701 				}
702 
703 				// add the doc value fields required for sorting, but only if this property is not part of an embedded
704 				// to-many assoc, in which case sorting on these fields would not make sense
705 				if ( !isParentPropertyMultiValued ) {
706 					addSortFieldDocValues( document, propertyMetadata, documentBoost, currentFieldValue );
707 				}
708 			}
709 			finally {
710 				conversionContext.popProperty();
711 			}
712 		}
713 	}

715 	private void addFacetDocValues(Document document,
716 			DocumentFieldMetadata fieldMetadata,
717 			FacetMetadata facetMetadata,
718 			Object value) {
719 		// we don't add null values to the facet field
720 		if ( value == null ) {
721 			return;
722 		}
723 
724 		Field facetField;
725 		switch ( facetMetadata.getEncoding() ) {
726 			case STRING:
727 				String stringValue = value.toString();
728 				// we don't add empty strings to the facet field
729 				if ( stringValue.isEmpty() ) {
730 					return;
731 				}
732 				facetField = new SortedSetDocValuesFacetField( facetMetadata.getAbsoluteName(), stringValue );
733 				break;
734 			case LONG:
735 				if ( value instanceof Number ) {
736 					facetField = new NumericDocValuesField(
737 							facetMetadata.getAbsoluteName(),
738 							( (Number) value ).longValue()
739 					);
740 				}
741 				else if ( Date.class.isAssignableFrom( value.getClass() ) ) {
742 					// XXX GSM: this is a workaround for now, we need to find something cleaner.
743 					Date date = (Date) value;
744 					FieldBridge fieldBridge = fieldMetadata.getFieldBridge();
745 					if ( fieldBridge instanceof NumericEncodingDateBridge ) {
746 						NumericEncodingDateBridge dateBridge = (NumericEncodingDateBridge) fieldBridge;
747 						long numericDateValue = DateTools.round( date.getTime(), dateBridge.getResolution() );
748 
749 						facetField = new NumericDocValuesField( facetMetadata.getAbsoluteName(), numericDateValue );
750 					}
751 					else if ( fieldBridge instanceof TwoWayFieldBridge ) {
752 						// Date might be stored in String for specific backends (Elasticsearch for instance).
753 						facetField = new SortedSetDocValuesFacetField( facetMetadata.getAbsoluteName(),
754 								( (TwoWayFieldBridge) fieldBridge ).objectToString( value ) );
755 					}
756 					else {
757 						throw log.numericDateFacetForNonNumericField(
758 								facetMetadata.getAbsoluteName(),
759 								fieldMetadata.getAbsoluteName() );
760 					}
761 				}
762 				else if ( Calendar.class.isAssignableFrom( value.getClass() ) ) {
763 					Calendar calendar = (Calendar) value;
764 					facetField = new NumericDocValuesField(
765 							facetMetadata.getAbsoluteName(),
766 							calendar.getTime().getTime()
767 					);
768 				}
769 				else {
770 					throw new AssertionFailure( "Unexpected value type for faceting: " + value.getClass().getName() );
771 				}
772 				break;
773 			case DOUBLE:
774 				if ( value instanceof Number ) {
775 					facetField = new DoubleDocValuesField(
776 							facetMetadata.getAbsoluteName(),
777 							( (Number) value ).doubleValue()
778 					);
779 				}
780 				else {
781 					throw new AssertionFailure( "Unexpected value type for faceting: " + value.getClass().getName() );
782 				}
783 				break;
784 			case AUTO:
785 				throw new AssertionFailure( "The facet type should have been resolved during bootstrapping" );
786 			default:
787 				throw new AssertionFailure(
788 						"Unexpected facet encoding type '"
789 								+ facetMetadata.getEncoding()
790 								+ "' Has the enum been modified?"
791 				);
792 		}
793 
794 		document.add( facetField );
795 	}

797 	/**
798 	 * Adds the doc field values to the document required to map the configured sort fields. The value from the
799 	 * underlying field will be obtained from the document (it has been written at this point already) and an equivalent
800 	 * doc field value will be added.
801 	 * For non-numeric fields, if the field value is supposed to be analyzed, the analysis will be performed by
802 	 * this method, because Lucene does not analyze SortedDocValuesFields values automatically.
803 	 */
804 	private void addSortFieldDocValues(Document document, PropertyMetadata propertyMetadata, float documentBoost, Object propertyValue) {
805 		for ( SortableFieldMetadata sortField : propertyMetadata.getSortableFieldMetadata() ) {
806 			// field marked as sortable by custom bridge to allow sort field validation pass, but that bridge itself is
807 			// in charge of adding the required field
808 			if ( propertyMetadata.getDeclaringType().getBridgeDefinedFieldMetadataFor( sortField.getAbsoluteName() ) != null ) {
809 				continue;
810 			}
811 
812 			DocumentFieldMetadata fieldMetaData = propertyMetadata.getFieldMetadata( sortField.getAbsoluteName() );
813 			if ( fieldMetaData == null ) {
814 				throw new AssertionFailure( "A sortable field did not match neither an @Field nor a bridge-defined field" );
815 			}
816 
817 			IndexableField field;
818 
819 			// A non-stored, non-indexed field will not be added to the actual document; in that case retrieve
820 			// its value via a dummy document, adjusting the options to index the field
821 			if ( fieldMetaData.getIndex() == Index.NO && fieldMetaData.getStore() == Store.NO ) {
822 				FieldBridge fieldBridge = fieldMetaData.getFieldBridge();
823 
824 				NullMarker nullMarker = fieldMetaData.getNullMarkerCodec().getNullMarker();
825 				LuceneOptionsImpl luceneOptions = new LuceneOptionsImpl(
826 						Index.NOT_ANALYZED,
827 						fieldMetaData.getTermVector(),
828 						fieldMetaData.getStore(),
829 						nullMarker == null ? null : nullMarker.nullRepresentedAsString(),
830 						fieldMetaData.getBoost() * propertyMetadata.getDynamicBoostStrategy().defineBoost( propertyValue ),
831 						documentBoost
832 				);
833 
834 				Document dummy = new Document();
835 				fieldBridge.set(
836 						"dummy",
837 						propertyValue,
838 						dummy,
839 						luceneOptions
840 				);
841 
842 				field = dummy.getField( "dummy" );
843 			}
844 			else {
845 				field = document.getField( sortField.getAbsoluteName() );
846 			}
847 
848 			if ( field != null ) {
849 				Number numericValue = field.numericValue();
850 
851 				if ( numericValue != null ) {
852 					if ( numericValue instanceof Double ) {
853 						document.add( new DoubleDocValuesField( sortField.getAbsoluteName(), (double) numericValue ) );
854 					}
855 					else if ( numericValue instanceof Float ) {
856 						document.add( new FloatDocValuesField( sortField.getAbsoluteName(), (float) numericValue ) );
857 					}
858 					else {
859 						document.add( new NumericDocValuesField( sortField.getAbsoluteName(), numericValue.longValue() ) );
860 					}
861 				}
862 				else {
863 					String value = field.stringValue();
864 					AnalyzerReference analyzerReference = fieldMetaData.getAnalyzerReference();
865 					if ( fieldMetaData.getIndex().isAnalyzed() && analyzerReference.is( LuceneAnalyzerReference.class ) ) {
866 						/*
867 						 * Necessary because Lucene doesn't automatically analyze SortedDocValuesFields.
868 						 * See https://hibernate.atlassian.net/browse/HSEARCH-2376
869 						 *
870 						 * Analysis is skipped altogether when the analyzer is remote. It's up to the backend to handle it.
871 						 */
872 						Analyzer analyzer = analyzerReference.unwrap( LuceneAnalyzerReference.class ).getAnalyzer();
873 						value = InternalAnalyzerUtils.analyzeSortableValue( analyzer, sortField.getAbsoluteName(), value );
874 					}
875 					if ( value != null ) {
876 						document.add( new SortedDocValuesField( sortField.getAbsoluteName(), new BytesRef( value ) ) );
877 					}
878 				}
879 			}
880 		}
881 	}

900 	 * Check if a given value is present in an array.
901 	 *
902 	 * A array {@code null} contains all possible values.
903 	 * Otherwise, a value {@code null} is always considered non present.
904 	 * The last behavior is not currently used, it is merely an optimization.
905 	 */
906 	private boolean containsFieldName(String value, String[] array) {
907 		//null array means contains all possible values
908 		if ( array == null ) {
909 			return true;
910 		}
911 		//null value is meaningless
912 		if ( value == null ) {
913 			return false;
914 		}
915 		for ( String containedValue : array ) {
916 			if ( value.equals( containedValue ) ) {
917 				return true;
918 			}
919 		}
920 		return false;
921 	}

947 	 * Allows a analyzer discriminator to override the analyzer used for any field in the Lucene document.
948 	 *
949 	 * @param doc The Lucene <code>Document</code> which shall be indexed.
950 	 * @param typeMetadata The metadata for the entity we currently add to the document.
951 	 * @param fieldToAnalyzerMap This map contains the actual override data. It is a map between document fields names and
952 	 * analyzer definition names. This map will be added to the <code>Work</code> instance and processed at actual indexing time.
953 	 * @param processedFieldNames A list of field names we have already processed.
954 	 * @param unproxiedInstance The entity we currently "add" to the document.
955 	 */
956 	private void allowAnalyzerDiscriminatorOverride(Document doc,
957 			TypeMetadata typeMetadata,
958 			Map<String, String> fieldToAnalyzerMap,
959 			Set<String> processedFieldNames,
960 			Object unproxiedInstance) {
961 
962 		Discriminator discriminator = typeMetadata.getDiscriminator();
963 		if ( discriminator == null ) {
964 			return;
965 		}
966 
967 		Object value = null;
968 		if ( typeMetadata.getDiscriminatorGetter() != null ) {
969 			value = ReflectionHelper.getMemberValue( unproxiedInstance, typeMetadata.getDiscriminatorGetter() );
970 		}
971 
972 		// now we give the discriminator the opportunity to specify a analyzer per field level
973 		for ( IndexableField field : doc.getFields() ) {
974 			if ( !processedFieldNames.contains( field.name() ) ) {
975 				String analyzerName = discriminator.getAnalyzerDefinitionName( value, unproxiedInstance, field.name() );
976 				if ( analyzerName != null ) {
977 					fieldToAnalyzerMap.put( field.name(), analyzerName );
978 				}
979 				processedFieldNames.add( field.name() );
980 			}
981 		}
982 	}

1078 	private FieldBridge getBridge(TypeMetadata typeMetadata, String fieldName) {
1079 		// process base fields
1080 		DocumentFieldMetadata documentFieldMetadata = typeMetadata.getDocumentFieldMetadataFor( fieldName );
1081 		if ( documentFieldMetadata != null && documentFieldMetadata.getFieldBridge() != null ) {
1082 			return documentFieldMetadata.getFieldBridge();
1083 		}
1084 
1085 		// process embedded fields
1086 		FieldBridge fieldBridge;
1087 
1088 		for ( EmbeddedTypeMetadata embeddedTypeMetadata : typeMetadata.getEmbeddedTypeMetadata() ) {
1089 			fieldBridge = getBridge( embeddedTypeMetadata, fieldName );
1090 			if ( fieldBridge != null ) {
1091 				return fieldBridge;
1092 			}
1093 		}
1094 
1095 		// process null embedded fields
1096 		for ( EmbeddedTypeMetadata embeddedTypeMetadata : typeMetadata.getEmbeddedTypeMetadata() ) {
1097 			fieldBridge = getNullBridge( embeddedTypeMetadata, fieldName );
1098 			if ( fieldBridge != null ) {
1099 				return fieldBridge;
1100 			}
1101 		}
1102 
1103 		//process class bridges
1104 		DocumentFieldMetadata fieldMetadata = typeMetadata.getFieldMetadataForClassBridgeField( fieldName );
1105 		fieldBridge = fieldMetadata == null ? null : fieldMetadata.getFieldBridge();
1106 		return fieldBridge;
1107 	}

1109 	/**
1110 	 * Checks whether all involved bridges allow to optimize document retrieval by using
1111 	 * {@code FieldSelector} (see HSEARCH-213).
1112 	 */
1113 	private void checkAllowFieldSelection() {
1114 		allowFieldSelectionInProjection = true;
1115 		if ( fieldBridgeProhibitsFieldSelectionInProjection( getIdBridge() ) ) {
1116 			allowFieldSelectionInProjection = false;
1117 			return;
1118 		}
1119 		for ( PropertyMetadata propertyMetadata : getMetadata().getAllPropertyMetadata() ) {
1120 			for ( DocumentFieldMetadata documentFieldMetadata : propertyMetadata.getFieldMetadataSet() ) {
1121 				FieldBridge bridge = documentFieldMetadata.getFieldBridge();
1122 				if ( fieldBridgeProhibitsFieldSelectionInProjection( bridge ) ) {
1123 					allowFieldSelectionInProjection = false;
1124 					return;
1125 				}
1126 			}
1127 		}
1128 		for ( DocumentFieldMetadata fieldMetadata : getMetadata().getClassBridgeMetadata() ) {
1129 			FieldBridge bridge = fieldMetadata.getFieldBridge();
1130 			if ( fieldBridgeProhibitsFieldSelectionInProjection( bridge ) ) {
1131 				allowFieldSelectionInProjection = false;
1132 				return;
1133 			}
1134 		}
1135 	}

1137 	private boolean fieldBridgeProhibitsFieldSelectionInProjection(FieldBridge bridge) {
1138 		return !( bridge instanceof TwoWayStringBridge
1139 				|| bridge instanceof TwoWayString2FieldBridgeAdaptor
1140 				|| bridge instanceof NumericFieldBridge
1141 				|| bridge instanceof NumericEncodingCalendarBridge
1142 				|| bridge instanceof NumericEncodingDateBridge
1143 				|| bridge instanceof StringEncodingDateBridge );
1144 	}
