53  /**
54   * Hibernate ORM event listener called by various ORM life cycle events. This listener must be registered in order
55   * to enable automatic index updates.
56   *
57   * @author Gavin King
58   * @author Emmanuel Bernard
59   * @author Mattias Arbin
60   * @author Sanne Grinovero
61   * @author Hardy Ferentschik
62   */
63  @SuppressWarnings("serial")
64  public final class FullTextIndexEventListener implements PostDeleteEventListener,
65  		PostInsertEventListener, PostUpdateEventListener,
66  		PostCollectionRecreateEventListener, PostCollectionRemoveEventListener,
67  		PostCollectionUpdateEventListener, FlushEventListener,
68  		Serializable {
69  
70  	private static final Log log = LoggerFactory.make();
71  
72  	private boolean disabled;
73  	private boolean skipDirtyChecks = true;
74  	private ExtendedSearchIntegrator extendedIntegrator;
75  
76  	//only used by the FullTextIndexEventListener instance playing in the FlushEventListener role.
77  	// transient because it's not serializable (and state doesn't need to live longer than a flush).
78  	// final because its initialization should be published to other threads.
79  	// ! update the readObject() method in case of name changes !
80  	// make sure the Synchronization doesn't contain references to Session, otherwise we'll leak memory.
81  	private final transient Map<Session, Synchronization> flushSynch = Maps.createIdentityWeakKeyConcurrentMap( 64, 32 );
82  
83  	@Override
84  	public void onPostDelete(PostDeleteEvent event) {
85  		if ( disabled ) {
86  			return;
87  		}
88  
89  		final Object entity = event.getEntity();
90  		if ( getDocumentBuilder( entity ) != null ) {
91  			// FIXME The engine currently needs to know about details such as identifierRollbackEnabled
92  			// but we should not move the responsibility to figure out the proper id to the engine
93  			boolean identifierRollbackEnabled = event.getSession()
94  					.getFactory()
95  					.getSettings()
96  					.isIdentifierRollbackEnabled();
97  			processWork( tenantIdentifier( event ), entity, event.getId(), WorkType.DELETE, event, identifierRollbackEnabled );
98  		}
99  	}
100 
101 	@Override
102 	public void onPostInsert(PostInsertEvent event) {
103 		if ( disabled ) {
104 			return;
105 		}
106 
107 		final Object entity = event.getEntity();
108 		if ( getDocumentBuilder( entity ) != null ) {
109 			Serializable id = event.getId();
110 			processWork( tenantIdentifier( event ), entity, id, WorkType.ADD, event, false );
111 		}
112 	}
113 
114 	private String tenantIdentifier(AbstractEvent event) {
115 		EventSource session = event.getSession();
116 		if ( session instanceof SessionImplementor ) {
117 			return ( (SessionImplementor) session ).getTenantIdentifier();
118 		}
119 		return null;
120 	}
121 
122 	@Override
123 	public void onPostUpdate(PostUpdateEvent event) {
124 		if ( disabled ) {
125 			return;
126 		}
127 
128 		final Object entity = event.getEntity();
129 		final AbstractDocumentBuilder docBuilder = getDocumentBuilder( entity );
130 		if ( docBuilder != null && ( skipDirtyChecks || docBuilder.isDirty(
131 				getDirtyPropertyNames(
132 						event
133 				)
134 		) ) ) {
135 			Serializable id = event.getId();
136 			processWork( tenantIdentifier( event ), entity, id, WorkType.UPDATE, event, false );
137 		}
138 	}
139 
140 	@Override
141 	public void onPostRecreateCollection(PostCollectionRecreateEvent event) {
142 		processCollectionEvent( event );
143 	}
144 
145 	@Override
146 	public void onPostRemoveCollection(PostCollectionRemoveEvent event) {
147 		processCollectionEvent( event );
148 	}
149 
150 	@Override
151 	public void onPostUpdateCollection(PostCollectionUpdateEvent event) {
152 		processCollectionEvent( event );
153 	}

155 	/**
156 	 * Make sure the indexes are updated right after the hibernate flush,
157 	 * avoiding object loading during a flush. Not needed during transactions.
158 	 */
159 	@Override
160 	public void onFlush(FlushEvent event) {
161 		if ( disabled ) {
162 			return;
163 		}
164 
165 		Session session = event.getSession();
166 		Synchronization synchronization = flushSynch.get( session );
167 		if ( synchronization != null ) {
168 			//first cleanup
169 			flushSynch.remove( session );
170 			log.debug( "flush event causing index update out of transaction" );
171 			synchronization.beforeCompletion();
172 			synchronization.afterCompletion( Status.STATUS_COMMITTED );
173 		}
174 	}
175 
176 	public ExtendedSearchIntegrator getExtendedSearchFactoryIntegrator() {
177 		return extendedIntegrator;
178 	}
179 
180 	public String[] getDirtyPropertyNames(PostUpdateEvent event) {
181 		EntityPersister persister = event.getPersister();
182 		final int[] dirtyProperties = event.getDirtyProperties();
183 		if ( dirtyProperties != null && dirtyProperties.length > 0 ) {
184 			String[] propertyNames = persister.getPropertyNames();
185 			int length = dirtyProperties.length;
186 			String[] dirtyPropertyNames = new String[length];
187 			for ( int i = 0; i < length; i++ ) {
188 				dirtyPropertyNames[i] = propertyNames[dirtyProperties[i]];
189 			}
190 			return dirtyPropertyNames;
191 		}
192 		else {
193 			return null;
194 		}
195 	}

197 	/**
198 	 * Initialize method called by Hibernate Core when the SessionFactory starts.
199 	 * @param extendedIntegrator the {@link ExtendedSearchIntegrator}
200 	 */
201 	public void initialize(ExtendedSearchIntegrator extendedIntegrator) {
202 		this.extendedIntegrator = extendedIntegrator;
203 
204 		if ( extendedIntegrator.getIndexingMode() == IndexingMode.EVENT ) {
205 			disabled = extendedIntegrator.getIndexBindings().size() == 0;
206 		}
207 		else if ( extendedIntegrator.getIndexingMode() == IndexingMode.MANUAL ) {
208 			disabled = true;
209 		}
210 
211 		log.debug( "Hibernate Search event listeners " + ( disabled ? "deactivated" : "activated" ) );
212 
213 		if ( ! disabled ) {
214 			skipDirtyChecks = !extendedIntegrator.isDirtyChecksEnabled();
215 			log.debug( "Hibernate Search dirty checks " + ( skipDirtyChecks ? "disabled" : "enabled" ) );
216 		}
217 	}
218 
219 	/**
220 	 * Adds a synchronization to be performed in the onFlush method;
221 	 * should only be used as workaround for the case a flush is happening
222 	 * out of transaction.
223 	 * Warning: if the synchronization contains a hard reference
224 	 * to the Session proper cleanup is not guaranteed and memory leaks
225 	 * will happen.
226 	 *
227 	 * @param eventSource should be the Session doing the flush
228 	 * @param synchronization the synchronisation instance
229 	 */
230 	public void addSynchronization(EventSource eventSource, Synchronization synchronization) {
231 		this.flushSynch.put( eventSource, synchronization );
232 	}

234 	protected void processWork(String tenantIdentifier, Object entity, Serializable id, WorkType workType, AbstractEvent event, boolean identifierRollbackEnabled) {
235 		Work work = new Work( tenantIdentifier, entity, id, workType, identifierRollbackEnabled );
236 		final EventSourceTransactionContext transactionContext = new EventSourceTransactionContext( event.getSession() );
237 		extendedIntegrator.getWorker().performWork( work, transactionContext );
238 	}

240 	protected void processCollectionEvent(AbstractCollectionEvent event) {
241 		if ( disabled ) {
242 			return;
243 		}
244 
245 		Object entity = event.getAffectedOwnerOrNull();
246 		if ( entity == null ) {
247 			//Hibernate cannot determine every single time the owner especially in case detached objects are involved
248 			// or property-ref is used
249 			//Should log really but we don't know if we're interested in this collection for indexing
250 			return;
251 		}
252 		PersistentCollection persistentCollection = event.getCollection();
253 		final String collectionRole;
254 		if ( persistentCollection != null ) {
255 			if ( !persistentCollection.wasInitialized() ) {
256 				// non-initialized collections will still trigger events, but we want to skip them
257 				// as they won't contain new values affecting the index state
258 				return;
259 			}
260 			collectionRole = persistentCollection.getRole();
261 		}
262 		else {
263 			collectionRole = null;
264 		}
265 		AbstractDocumentBuilder documentBuilder = getDocumentBuilder( entity );
266 
267 		if ( documentBuilder != null && documentBuilder.collectionChangeRequiresIndexUpdate( collectionRole ) ) {
268 			Serializable id = getId( entity, event );
269 			if ( id == null ) {
270 				log.idCannotBeExtracted( event.getAffectedOwnerEntityName() );
271 				return;
272 			}
273 			processWork( tenantIdentifier( event ), entity, id, WorkType.COLLECTION, event, false );
274 		}
275 	}
276 
277 	private Serializable getId(Object entity, AbstractCollectionEvent event) {
278 		Serializable id = event.getAffectedOwnerIdOrNull();
279 		if ( id == null ) {
280 			// most likely this recovery is unnecessary since Hibernate Core probably try that
281 			EntityEntry entityEntry = event.getSession().getPersistenceContext().getEntry( entity );
282 			id = entityEntry == null ? null : entityEntry.getId();
283 		}
284 		return id;
285 	}

287 	private void writeObject(ObjectOutputStream os) throws IOException {
288 		os.defaultWriteObject();
289 	}

291 	//needs to implement custom readObject to restore the transient fields
292 
293 	private void readObject(ObjectInputStream is)
294 			throws IOException, ClassNotFoundException, NoSuchFieldException, IllegalAccessException {
295 		is.defaultReadObject();
296 		Class<FullTextIndexEventListener> cl = FullTextIndexEventListener.class;
297 		Field f = cl.getDeclaredField( "flushSynch" );
298 		ReflectionHelper.setAccessible( f );
299 		Map<Session, Synchronization> flushSynch = Maps.createIdentityWeakKeyConcurrentMap( 64, 32 );
300 		// setting a final field by reflection during a readObject is considered as safe as in a constructor:
301 		f.set( this, flushSynch );
302 	}
303 
304 	/**
305 	 * It is not suggested to extend FullTextIndexEventListener, but when needed to implement special
306 	 * use cases implementors might need this method. If you have to extent this, please report
307 	 * your use case so that better long term solutions can be discussed.
308 	 *
309 	 * @param instance the object instance for which to retrieve the document builder
310 	 *
311 	 * @return the {@code DocumentBuilder} for the specified object
312 	 */
313 	protected AbstractDocumentBuilder getDocumentBuilder(final Object instance) {
314 		Class<?> clazz = instance.getClass();
315 		EntityIndexBinding entityIndexBinding = extendedIntegrator.getIndexBinding( clazz );
316 		if ( entityIndexBinding != null ) {
317 			return entityIndexBinding.getDocumentBuilder();
318 		}
319 		else {
320 			return extendedIntegrator.getDocumentBuilderContainedEntity( clazz );
321 		}
322 	}
