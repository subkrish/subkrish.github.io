9   import org.apache.lucene.search.NumericRangeQuery;
10  import org.apache.lucene.search.Query;
11  import org.junit.Test;

13  import org.hibernate.search.jpa.FullTextEntityManager;
14  import org.hibernate.search.jpa.Search;

16  import org.hibernate.search.testsupport.serialization.SerializationTestHelper;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  /**
21   * Serialization test for entity manager.
22   *
23   * @author Hardy Ferentschik
24   */
25  @TestForIssue(jiraKey = "HSEARCH-117")
26  public class EntityManagerSerializationTest extends JPATestCase {
27  
28  	/**
29  	 * Test that a entity manager can successfully be serialized and deserialized.
30  	 *
31  	 * @throws Exception in case the test fails.
32  	 */
33  	@Test
34  	public void testSerialization() throws Exception {
35  		FullTextEntityManager em = Search.getFullTextEntityManager( factory.createEntityManager() );
36  
37  		indexSearchAssert( em );
38  
39  		FullTextEntityManager clone = SerializationTestHelper.duplicateBySerialization( em );
40  
41  		indexSearchAssert( clone );
42  
43  		clone.close();
44  		em.close();
45  	}
46  
47  	@Override
48  	public Class[] getAnnotatedClasses() {
49  		return new Class[] { Bretzel.class };
50  	}
51  
52  	/**
53  	 * Helper method for testing the entity manager before and after
54  	 * serialization.
55  	 *
56  	 * @param em Entity manager used for indexing and searching
57  	 *
58  	 * @throws Exception
59  	 */
60  	private static void indexSearchAssert(FullTextEntityManager em) throws Exception {
61  		// index a Bretzel
62  		em.getTransaction().begin();
63  		Bretzel bretzel = new Bretzel( 23, 34 );
64  		em.persist( bretzel );
65  		em.getTransaction().commit();
66  		em.clear();
67  		em.getTransaction().begin();
68  
69  		// execute a non matching query
70  		Query query = NumericRangeQuery.newIntRange( "saltQty", 0, 0, true, true );
71  		assertEquals( 0, em.createFullTextQuery( query ).getResultList().size() );
72  
73  		// execute a matching query
74  		query = NumericRangeQuery.newIntRange( "saltQty", 23, 23, true, true );
75  		assertEquals(
76  				"getResultList", 1, em.createFullTextQuery( query )
77  				.getResultList().size()
78  		);
79  		assertEquals(
80  				"getSingleResult and object retrieval", 23, ( (Bretzel) em
81  				.createFullTextQuery( query )
82  				.getSingleResult() )
83  				.getSaltQty()
84  		);
85  		assertEquals( 1, em.createFullTextQuery( query ).getResultSize() );
86  		em.getTransaction().commit();
87  
88  		em.clear();
89  
90  		em.getTransaction().begin();
91  		em.remove( em.find( Bretzel.class, bretzel.getId() ) );
92  		em.getTransaction().commit();
93  	}
94  }
