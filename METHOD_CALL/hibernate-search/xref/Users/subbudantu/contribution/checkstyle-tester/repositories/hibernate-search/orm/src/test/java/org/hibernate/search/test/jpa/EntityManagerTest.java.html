7   package org.hibernate.search.test.jpa;
8   
9   import org.apache.lucene.search.NumericRangeQuery;
10  import org.apache.lucene.search.Query;
11  import org.junit.Before;
12  import org.junit.Test;
13  
14  import org.hibernate.search.jpa.FullTextEntityManager;
15  import org.hibernate.search.jpa.FullTextQuery;
16  import org.hibernate.search.jpa.Search;
17  import org.hibernate.search.query.dsl.QueryBuilder;
18  
19  import static org.junit.Assert.assertEquals;
20  
21  /**
22   * @author Emmanuel Bernard
23   */
24  public class EntityManagerTest extends JPATestCase {
25  	private FullTextEntityManager em;
26  	private Bretzel bretzel;
27  
28  	@Before
29  	@Override
30  	public void setUp() {
31  		super.setUp();
32  
33  		em = Search.getFullTextEntityManager( factory.createEntityManager() );
34  		em.getTransaction().begin();
35  		bretzel = new Bretzel( 23, 34 );
36  		em.persist( bretzel );
37  		em.getTransaction().commit();
38  		em.clear();
39  	}
40  
41  	@Test
42  	public void testMassIndexer() throws Exception {
43  		// verify against index
44  		assertEquals( "At the beginning of the test there should be an indexed Bretzel", 1, countBretzelsViaIndex( em ) );
45  
46  		// clear index
47  		em.purgeAll( Bretzel.class );
48  		em.flushToIndexes();
49  
50  		// verify Bretzel removed from index
51  		assertEquals( "The index should be empty after an purge and flush", 0, countBretzelsViaIndex( em ) );
52  
53  		// re-index
54  		em.createIndexer( Bretzel.class ).startAndWait();
55  
56  		assertEquals( "After re-indexing the bretzel should be indexed again", 1, countBretzelsViaIndex( em ) );
57  	}
58  
59  	@Test
60  	public void testNonMatchingQueryDoesReturnEmptyResults() throws Exception {
61  		em.getTransaction().begin();
62  
63  		Query query = NumericRangeQuery.newIntRange( "saltQty", 0, 0, true, true );
64  		assertEquals( 0, em.createFullTextQuery( query ).getResultList().size() );
65  
66  		em.getTransaction().commit();
67  	}
68  
69  	@Test
70  	public void testGetResultList() throws Exception {
71  		em.getTransaction().begin();
72  
73  		Query query = NumericRangeQuery.newIntRange( "saltQty", 23, 23, true, true );
74  		assertEquals( "getResultList should return a result", 1, em.createFullTextQuery( query ).getResultList().size() );
75  
76  		em.getTransaction().commit();
77  	}
78  
79  	@Test
80  	public void testGetSingleResult() throws Exception {
81  		em.getTransaction().begin();
82  
83  		Query query = NumericRangeQuery.newIntRange( "saltQty", 23, 23, true, true );
84  		assertEquals(
85  				"getSingleResult should return a result", 23,
86  				( (Bretzel) em.createFullTextQuery( query ).getSingleResult() ).getSaltQty()
87  		);
88  		em.getTransaction().commit();
89  	}
90  
91  	@Test
92  	public void testGetResultSize() throws Exception {
93  		em.getTransaction().begin();
94  
95  		Query query = NumericRangeQuery.newIntRange( "saltQty", 23, 23, true, true );
96  		assertEquals( "Wrong result size", 1, em.createFullTextQuery( query ).getResultSize() );
97  
98  		em.getTransaction().commit();
99  	}
100 
101 	@Test
102 	public void testIndex() {
103 		em.getTransaction().begin();
104 
105 		// verify against index
106 		assertEquals( "At the beginning of the test there should be an indexed Bretzel", 1, countBretzelsViaIndex( em ) );
107 
108 		// clear index
109 		em.purgeAll( Bretzel.class );
110 		em.flushToIndexes();
111 
112 		// verify Bretzel removed from index
113 		assertEquals( "The index should be empty after an purge and flush", 0, countBretzelsViaIndex( em ) );
114 
115 		// re-index manually
116 		em.index( em.find( Bretzel.class, bretzel.getId() ) );
117 		em.getTransaction().commit();
118 
119 		assertEquals( "After re-indexing the bretzel should be indexed again", 1, countBretzelsViaIndex( em ) );
120 	}
121 
122 	@Override
123 	public Class[] getAnnotatedClasses() {
124 		return new Class[] {
125 				Bretzel.class
126 		};
127 	}
128 
129 	private int countBretzelsViaIndex(FullTextEntityManager em) {
130 		QueryBuilder queryBuilder = em.getSearchFactory().buildQueryBuilder().forEntity( Bretzel.class ).get();
131 		Query allQuery = queryBuilder.all().createQuery();
132 		FullTextQuery fullTextQuery = em.createFullTextQuery( allQuery, Bretzel.class );
133 		return fullTextQuery.getResultSize();
134 	}
135 }
