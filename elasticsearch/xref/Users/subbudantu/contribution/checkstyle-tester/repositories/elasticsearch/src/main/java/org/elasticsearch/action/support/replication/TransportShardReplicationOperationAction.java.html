160     /**
161      * Should an exception be ignored when the operation is performed on the replica.
162      */
163     protected boolean ignoreReplicaException(Throwable e) {
164         if (TransportActions.isShardNotAvailableException(e)) {
165             return true;
166         }
167         // on version conflict or document missing, it means
168         // that a new change has crept into the replica, and it's fine
169         if (isConflictException(e)) {
170             return true;
171         }
172         return false;
173     }
174 
175     protected boolean isConflictException(Throwable e) {
176         Throwable cause = ExceptionsHelper.unwrapCause(e);
177         // on version conflict or document missing, it means
178         // that a new change has crept into the replica, and it's fine
179         if (cause instanceof VersionConflictEngineException) {
180             return true;
181         }
182         if (cause instanceof DocumentAlreadyExistsException) {
183             return true;
184         }
185         return false;
186     }

188     class OperationTransportHandler extends BaseTransportRequestHandler<Request> {
189 
190         @Override
191         public Request newInstance() {
192             return newRequestInstance();
193         }
194 
195         @Override
196         public String executor() {
197             return ThreadPool.Names.SAME;
198         }
199 
200         @Override
201         public void messageReceived(final Request request, final TransportChannel channel) throws Exception {
202             // no need to have a threaded listener since we just send back a response
203             request.listenerThreaded(false);
204             // if we have a local operation, execute it on a thread since we don't spawn
205             request.operationThreaded(true);
206             execute(request, new ActionListener<Response>() {
207                 @Override
208                 public void onResponse(Response result) {
209                     try {
210                         channel.sendResponse(result);
211                     } catch (Throwable e) {
212                         onFailure(e);
213                     }
214                 }
215 
216                 @Override
217                 public void onFailure(Throwable e) {
218                     try {
219                         channel.sendResponse(e);
220                     } catch (Throwable e1) {
221                         logger.warn("Failed to send response for " + actionName, e1);
222                     }
223                 }
224             });
225         }

268     protected class ReplicaOperationRequest extends TransportRequest implements IndicesRequest {
269 
270         public ShardId shardId;
271         public ReplicaRequest request;
272 
273         ReplicaOperationRequest() {
274         }
275 
276         ReplicaOperationRequest(ShardId shardId, ReplicaRequest request) {
277             super(request);
278             this.shardId = shardId;
279             this.request = request;
280         }
281 
282         @Override
283         public String[] indices() {
284             return request.indices();
285         }
286 
287         @Override
288         public IndicesOptions indicesOptions() {
289             return request.indicesOptions();
290         }
291 
292         @Override
293         public void readFrom(StreamInput in) throws IOException {
294             super.readFrom(in);
295             int shard = -1;
296             if (in.getVersion().onOrAfter(Version.V_1_4_0_Beta1)) {
297                 shardId = ShardId.readShardId(in);
298             } else {
299                 shard = in.readVInt();
300             }
301             request = newReplicaRequestInstance();
302             request.readFrom(in);
303             if (in.getVersion().before(Version.V_1_4_0_Beta1)) {
304                 assert shard >= 0;
305                 //older nodes will send the concrete index as part of the request
306                 shardId = new ShardId(request.index(), shard);
307             }
308         }
309 
310         @Override
311         public void writeTo(StreamOutput out) throws IOException {
312             super.writeTo(out);
313             if (out.getVersion().onOrAfter(Version.V_1_4_0_Beta1)) {
314                 shardId.writeTo(out);
315             } else {
316                 out.writeVInt(shardId.id());
317                 //older nodes expect the concrete index as part of the request
318                 request.index(shardId.getIndex());
319             }
320             request.writeTo(out);
321         }

324     protected class AsyncShardOperationAction {
325         private final ActionListener<Response> listener;
326         private final InternalRequest internalRequest;
327         private volatile ShardIterator shardIt;
328         private final AtomicBoolean primaryOperationStarted = new AtomicBoolean();
329         private final ReplicationType replicationType;
330         private volatile ClusterStateObserver observer;
331 
332         AsyncShardOperationAction(Request request, ActionListener<Response> listener) {
333             this.internalRequest = new InternalRequest(request);
334             this.listener = listener;
335 
336             if (request.replicationType() != ReplicationType.DEFAULT) {
337                 replicationType = request.replicationType();
338             } else {
339                 replicationType = defaultReplicationType;
340             }
341         }
342 
343         public void start() {
344             this.observer = new ClusterStateObserver(clusterService, internalRequest.request().timeout(), logger);
345             doStart();
346         }
347 
348         /**
349          * Returns <tt>true</tt> if the action starting to be performed on the primary (or is done).
350          */
351         protected void doStart() throws ElasticsearchException {
352             try {
353                 ClusterBlockException blockException = checkGlobalBlock(observer.observedState());
354                 if (blockException != null) {
355                     if (blockException.retryable()) {
356                         logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
357                         retry(blockException);
358                         return;
359                     } else {
360                         throw blockException;
361                     }
362                 }
363                 if (resolveIndex()) {
364                     internalRequest.concreteIndex(observer.observedState().metaData().concreteSingleIndex(internalRequest.request().index(), internalRequest.request().indicesOptions()));
365                 } else {
366                     internalRequest.concreteIndex(internalRequest.request().index());
367                 }
368                 // check if we need to execute, and if not, return
369                 if (!resolveRequest(observer.observedState(), internalRequest, listener)) {
370                     return;
371                 }
372                 blockException = checkRequestBlock(observer.observedState(), internalRequest);
373                 if (blockException != null) {
374                     if (blockException.retryable()) {
375                         logger.trace("cluster is blocked ({}), scheduling a retry", blockException.getMessage());
376                         retry(blockException);
377                         return;
378                     } else {
379                         throw blockException;
380                     }
381                 }
382                 shardIt = shards(observer.observedState(), internalRequest);
383             } catch (Throwable e) {
384                 listener.onFailure(e);
385                 return;
386             }
387 
388             // no shardIt, might be in the case between index gateway recovery and shardIt initialization
389             if (shardIt.size() == 0) {
390                 logger.trace("no shard instances known for shard [{}], scheduling a retry", shardIt.shardId());
391                 retryBecauseUnavailable(shardIt.shardId(), "No active shards.");
392                 return;
393             }
394 
395             boolean foundPrimary = false;
396             ShardRouting shardX;
397             while ((shardX = shardIt.nextOrNull()) != null) {
398                 final ShardRouting shard = shardX;
399                 // we only deal with primary shardIt here...
400                 if (!shard.primary()) {
401                     continue;
402                 }
403                 if (!shard.active() || !observer.observedState().nodes().nodeExists(shard.currentNodeId())) {
404                     logger.trace("primary shard [{}] is not yet active or we do not know the node it is assigned to [{}], scheduling a retry.", shard.shardId(), shard.currentNodeId());
405                     retryBecauseUnavailable(shardIt.shardId(), "Primary shard is not active or isn't assigned to a known node.");
406                     return;
407                 }
408 
409                 if (!primaryOperationStarted.compareAndSet(false, true)) {
410                     return;
411                 }
412 
413                 foundPrimary = true;
414                 if (shard.currentNodeId().equals(observer.observedState().nodes().localNodeId())) {
415                     try {
416                         if (internalRequest.request().operationThreaded()) {
417                             internalRequest.request().beforeLocalFork();
418                             threadPool.executor(executor).execute(new Runnable() {
419                                 @Override
420                                 public void run() {
421                                     try {
422                                         performOnPrimary(shard.id(), shard);
423                                     } catch (Throwable t) {
424                                         listener.onFailure(t);
425                                     }
426                                 }
427                             });
428                         } else {
429                             performOnPrimary(shard.id(), shard);
430                         }
431                     } catch (Throwable t) {
432                         listener.onFailure(t);
433                     }
434                 } else {
435                     DiscoveryNode node = observer.observedState().nodes().get(shard.currentNodeId());
436                     transportService.sendRequest(node, actionName, internalRequest.request(), transportOptions, new BaseTransportResponseHandler<Response>() {
437 
438                         @Override
439                         public Response newInstance() {
440                             return newResponseInstance();
441                         }
442 
443                         @Override
444                         public String executor() {
445                             return ThreadPool.Names.SAME;
446                         }
447 
448                         @Override
449                         public void handleResponse(Response response) {
450                             listener.onResponse(response);
451                         }
452 
453                         @Override
454                         public void handleException(TransportException exp) {
455                             // if we got disconnected from the node, or the node / shard is not in the right state (being closed)
456                             if (exp.unwrapCause() instanceof ConnectTransportException || exp.unwrapCause() instanceof NodeClosedException ||
457                                     retryPrimaryException(exp)) {
458                                 primaryOperationStarted.set(false);
459                                 internalRequest.request().setCanHaveDuplicates();
460                                 // we already marked it as started when we executed it (removed the listener) so pass false
461                                 // to re-add to the cluster listener
462                                 logger.trace("received an error from node the primary was assigned to ({}), scheduling a retry", exp.getMessage());
463                                 retry(exp);
464                             } else {
465                                 listener.onFailure(exp);
466                             }
467                         }
468                     });
469                 }
470                 break;
471             }
472             // we won't find a primary if there are no shards in the shard iterator, retry...
473             if (!foundPrimary) {
474                 logger.trace("couldn't find a eligible primary shard, scheduling for retry.");
475                 retryBecauseUnavailable(shardIt.shardId(), "No active shards.");
476             }
477         }

479         void retry(Throwable failure) {
480             assert failure != null;
481             if (observer.isTimedOut()) {
482                 // we running as a last attempt after a timeout has happened. don't retry
483                 listener.onFailure(failure);
484                 return;
485             }
486             // make it threaded operation so we fork on the discovery listener thread
487             internalRequest.request().beforeLocalFork();
488             internalRequest.request().operationThreaded(true);
489 
490             observer.waitForNextChange(new ClusterStateObserver.Listener() {
491                 @Override
492                 public void onNewClusterState(ClusterState state) {
493                     doStart();
494                 }
495 
496                 @Override
497                 public void onClusterServiceClose() {
498                     listener.onFailure(new NodeClosedException(clusterService.localNode()));
499                 }
500 
501                 @Override
502                 public void onTimeout(TimeValue timeout) {
503                     // Try one more time...
504                     doStart();
505                 }
506             });
507         }

509         void performOnPrimary(int primaryShardId, final ShardRouting shard) {
510             ClusterState clusterState = observer.observedState();
511             if (raiseFailureIfHaveNotEnoughActiveShardCopies(shard, clusterState)) {
512                 return;
513             }
514             try {
515                 PrimaryResponse<Response, ReplicaRequest> response = shardOperationOnPrimary(clusterState, new PrimaryOperationRequest(primaryShardId, internalRequest.concreteIndex(), internalRequest.request()));
516                 performReplicas(response);
517             } catch (Throwable e) {
518                 internalRequest.request.setCanHaveDuplicates();
519                 // shard has not been allocated yet, retry it here
520                 if (retryPrimaryException(e)) {
521                     primaryOperationStarted.set(false);
522                     logger.trace("had an error while performing operation on primary ({}), scheduling a retry.", e.getMessage());
523                     retry(e);
524                     return;
525                 }
526                 if (e instanceof ElasticsearchException && ((ElasticsearchException) e).status() == RestStatus.CONFLICT) {
527                     if (logger.isTraceEnabled()) {
528                         logger.trace(shard.shortSummary() + ": Failed to execute [" + internalRequest.request() + "]", e);
529                     }
530                 } else {
531                     if (logger.isDebugEnabled()) {
532                         logger.debug(shard.shortSummary() + ": Failed to execute [" + internalRequest.request() + "]", e);
533                     }
534                 }
535                 listener.onFailure(e);
536             }
537         }

539         void performReplicas(final PrimaryResponse<Response, ReplicaRequest> response) {
540             if (ignoreReplicas()) {
541                 postPrimaryOperation(internalRequest, response);
542                 listener.onResponse(response.response());
543                 return;
544             }
545             ShardRouting shard;
546 
547             // we double check on the state, if it got changed we need to make sure we take the latest one cause
548             // maybe a replica shard started its recovery process and we need to apply it there...
549 
550             // we also need to make sure if the new state has a new primary shard (that we indexed to before) started
551             // and assigned to another node (while the indexing happened). In that case, we want to apply it on the
552             // new primary shard as well...
553             ClusterState newState = clusterService.state();
554             ShardRouting newPrimaryShard = null;
555             if (observer.observedState() != newState) {
556                 shardIt.reset();
557                 ShardRouting originalPrimaryShard = null;
558                 while ((shard = shardIt.nextOrNull()) != null) {
559                     if (shard.primary()) {
560                         originalPrimaryShard = shard;
561                         break;
562                     }
563                 }
564                 if (originalPrimaryShard == null || !originalPrimaryShard.active()) {
565                     throw new ElasticsearchIllegalStateException("unexpected state, failed to find primary shard on an index operation that succeeded");
566                 }
567 
568                 observer.reset(newState);
569                 shardIt = shards(newState, internalRequest);
570                 while ((shard = shardIt.nextOrNull()) != null) {
571                     if (shard.primary()) {
572                         if (originalPrimaryShard.currentNodeId().equals(shard.currentNodeId())) {
573                             newPrimaryShard = null;
574                         } else {
575                             newPrimaryShard = shard;
576                         }
577                         break;
578                     }
579                 }
580                 shardIt.reset();
581                 internalRequest.request().setCanHaveDuplicates(); // safe side, cluster state changed, we might have dups
582             } else {
583                 shardIt.reset();
584                 while ((shard = shardIt.nextOrNull()) != null) {
585                     if (shard.state() != ShardRoutingState.STARTED) {
586                         internalRequest.request().setCanHaveDuplicates();
587                     }
588                 }
589                 shardIt.reset();
590             }
591 
592             // initialize the counter
593             int replicaCounter = shardIt.assignedReplicasIncludingRelocating();
594 
595             if (newPrimaryShard != null) {
596                 replicaCounter++;
597             }
598 
599             if (replicaCounter == 0) {
600                 postPrimaryOperation(internalRequest, response);
601                 listener.onResponse(response.response());
602                 return;
603             }
604 
605             if (replicationType == ReplicationType.ASYNC) {
606                 postPrimaryOperation(internalRequest, response);
607                 // async replication, notify the listener
608                 listener.onResponse(response.response());
609                 // now, trick the counter so it won't decrease to 0 and notify the listeners
610                 replicaCounter = Integer.MIN_VALUE;
611             }
612 
613             // we add one to the replica count to do the postPrimaryOperation
614             replicaCounter++;
615             AtomicInteger counter = new AtomicInteger(replicaCounter);
616 
617 
618             IndexMetaData indexMetaData = observer.observedState().metaData().index(internalRequest.concreteIndex());
619 
620             if (newPrimaryShard != null) {
621                 performOnReplica(response, counter, newPrimaryShard, newPrimaryShard.currentNodeId(), indexMetaData);
622             }
623 
624             shardIt.reset(); // reset the iterator
625             while ((shard = shardIt.nextOrNull()) != null) {
626                 // if its unassigned, nothing to do here...
627                 if (shard.unassigned()) {
628                     continue;
629                 }
630 
631                 // if the shard is primary and relocating, add one to the counter since we perform it on the replica as well
632                 // (and we already did it on the primary)
633                 boolean doOnlyOnRelocating = false;
634                 if (shard.primary()) {
635                     if (shard.relocating()) {
636                         doOnlyOnRelocating = true;
637                     } else {
638                         continue;
639                     }
640                 }
641                 // we index on a replica that is initializing as well since we might not have got the event
642                 // yet that it was started. We will get an exception IllegalShardState exception if its not started
643                 // and that's fine, we will ignore it
644                 if (!doOnlyOnRelocating) {
645                     performOnReplica(response, counter, shard, shard.currentNodeId(), indexMetaData);
646                 }
647                 if (shard.relocating()) {
648                     performOnReplica(response, counter, shard, shard.relocatingNodeId(), indexMetaData);
649                 }
650             }
651 
652             // now do the postPrimary operation, and check if the listener needs to be invoked
653             postPrimaryOperation(internalRequest, response);
654             // we also invoke here in case replicas finish before postPrimaryAction does
655             if (counter.decrementAndGet() == 0) {
656                 listener.onResponse(response.response());
657             }
658         }

660         void performOnReplica(final PrimaryResponse<Response, ReplicaRequest> response, final AtomicInteger counter, final ShardRouting shard, String nodeId, final IndexMetaData indexMetaData) {
661             // if we don't have that node, it means that it might have failed and will be created again, in
662             // this case, we don't have to do the operation, and just let it failover
663             if (!observer.observedState().nodes().nodeExists(nodeId)) {
664                 if (counter.decrementAndGet() == 0) {
665                     listener.onResponse(response.response());
666                 }
667                 return;
668             }
669 
670             final ReplicaOperationRequest shardRequest = new ReplicaOperationRequest(shardIt.shardId(), response.replicaRequest());
671 
672             // If the replicas use shadow replicas, there is no reason to
673             // perform the action on the replica, so skip it and
674             // immediately return
675             if (IndexMetaData.isIndexUsingShadowReplicas(indexMetaData.settings())) {
676                 // this delays mapping updates on replicas because they have
677                 // to wait until they get the new mapping through the cluster
678                 // state, which is why we recommend pre-defined mappings for
679                 // indices using shadow replicas
680                 if (counter.decrementAndGet() == 0) {
681                     listener.onResponse(response.response());
682                 }
683                 return;
684             }
685 
686             if (!nodeId.equals(observer.observedState().nodes().localNodeId())) {
687                 final DiscoveryNode node = observer.observedState().nodes().get(nodeId);
688                 transportService.sendRequest(node, transportReplicaAction, shardRequest,
689                         transportOptions, new EmptyTransportResponseHandler(ThreadPool.Names.SAME) {
690                     @Override
691                     public void handleResponse(TransportResponse.Empty vResponse) {
692                         finishIfPossible();
693                     }
694 
695                     @Override
696                     public void handleException(TransportException exp) {
697                         logger.trace("[{}] Transport failure during replica request [{}] ", exp, node, internalRequest.request());
698                         if (!ignoreReplicaException(exp)) {
699                             logger.warn("Failed to perform " + actionName + " on remote replica " + node + shardIt.shardId(), exp);
700                             shardStateAction.shardFailed(shard, indexMetaData.getUUID(),
701                                     "Failed to perform [" + actionName + "] on replica, message [" + ExceptionsHelper.detailedMessage(exp) + "]");
702                         }
703                         finishIfPossible();
704                     }
705 
706                     private void finishIfPossible() {
707                         if (counter.decrementAndGet() == 0) {
708                             listener.onResponse(response.response());
709                         }
710                     }
711                 });
712             } else {
713                 if (internalRequest.request().operationThreaded()) {
714                     internalRequest.request().beforeLocalFork();
715                     try {
716                         threadPool.executor(executor).execute(new AbstractRunnable() {
717                             @Override
718                             protected void doRun() {
719                                 try {
720                                     shardOperationOnReplica(shardRequest);
721                                 } catch (Throwable e) {
722                                     failReplicaIfNeeded(shard.index(), shard.id(), e);
723                                 }
724                                 if (counter.decrementAndGet() == 0) {
725                                     listener.onResponse(response.response());
726                                 }
727                             }
728 
729                             // we must never reject on because of thread pool capacity on replicas
730                             @Override
731                             public boolean isForceExecution() {
732                                 return true;
733                             }
734 
735                             @Override
736                             public void onFailure(Throwable t) {
737                             }
738                         });
739                     } catch (Throwable e) {
740                         failReplicaIfNeeded(shard.index(), shard.id(), e);
741                         // we want to decrement the counter here, in teh failure handling, cause we got rejected
742                         // from executing on the thread pool
743                         if (counter.decrementAndGet() == 0) {
744                             listener.onResponse(response.response());
745                         }
746                     }
747                 } else {
748                     try {
749                         shardOperationOnReplica(shardRequest);
750                     } catch (Throwable e) {
751                         failReplicaIfNeeded(shard.index(), shard.id(), e);
752                     }
753                     if (counter.decrementAndGet() == 0) {
754                         listener.onResponse(response.response());
755                     }
756                 }
757             }
758         }
759 
760         boolean raiseFailureIfHaveNotEnoughActiveShardCopies(ShardRouting shard, ClusterState state) {
761             if (!checkWriteConsistency) {
762                 return false;
763             }
764 
765             final WriteConsistencyLevel consistencyLevel;
766             if (internalRequest.request().consistencyLevel() != WriteConsistencyLevel.DEFAULT) {
767                 consistencyLevel = internalRequest.request().consistencyLevel();
768             } else {
769                 consistencyLevel = defaultWriteConsistencyLevel;
770             }
771             final int sizeActive;
772             final int requiredNumber;
773             IndexRoutingTable indexRoutingTable =  state.getRoutingTable().index(shard.index());
774             if (indexRoutingTable != null) {
775                 IndexShardRoutingTable shardRoutingTable = indexRoutingTable.shard(shard.getId());
776                 if (shardRoutingTable != null) {
777                     sizeActive =  shardRoutingTable.activeShards().size();
778                     if (consistencyLevel == WriteConsistencyLevel.QUORUM && shardRoutingTable.getSize() > 2) {
779                         // only for more than 2 in the number of shardIt it makes sense, otherwise its 1 shard with 1 replica, quorum is 1 (which is what it is initialized to)
780                         requiredNumber = (shardRoutingTable.getSize() / 2) + 1;
781                     } else if (consistencyLevel == WriteConsistencyLevel.ALL) {
782                         requiredNumber = shardRoutingTable.getSize();
783                     } else {
784                         requiredNumber = 1;
785                     }
786                 } else {
787                     sizeActive = 0;
788                     requiredNumber = 1;
789                 }
790             } else {
791                 sizeActive = 0;
792                 requiredNumber = 1;
793             }
794 
795             if (sizeActive < requiredNumber) {
796                 logger.trace("not enough active copies of shard [{}] to meet write consistency of [{}] (have {}, needed {}), scheduling a retry.",
797                         shard.shardId(), consistencyLevel, sizeActive, requiredNumber);
798                 primaryOperationStarted.set(false);
799                 // A dedicated exception would be nice...
800                 retryBecauseUnavailable(shard.shardId(), "Not enough active copies to meet write consistency of [" + consistencyLevel + "] (have " + sizeActive + ", needed " + requiredNumber + ").");
801                 return true;
802             } else {
803                 return false;
804             }
805         }
806 
807         void retryBecauseUnavailable(ShardId shardId, String message) {
808             retry(new UnavailableShardsException(shardId, message + " Timeout: [" + internalRequest.request().timeout() +"], request: " +  internalRequest.request().toString()));
809         }
810 
811     }

813     private void failReplicaIfNeeded(String index, int shardId, Throwable t) {
814         logger.trace("failure on replica [{}][{}]", t, index, shardId);
815         if (!ignoreReplicaException(t)) {
816             IndexService indexService = indicesService.indexService(index);
817             if (indexService == null) {
818                 logger.debug("ignoring failed replica [{}][{}] because index was already removed.", index, shardId);
819                 return;
820             }
821             IndexShard indexShard = indexService.shard(shardId);
822             if (indexShard == null) {
823                 logger.debug("ignoring failed replica [{}][{}] because index was already removed.", index, shardId);
824                 return;
825             }
826             indexShard.failShard(actionName + " failed on replica", t);
827         }
