32  /**
33   * A query that applies a filter to the results of another query.
34   * <p/>
35   * <p>Note: the bits are retrieved from the filter each time this
36   * query is used in a search - use a CachingWrapperFilter to avoid
37   * regenerating the bits every time.
38   *
39   * @see CachingWrapperFilter
40   * @since 1.4
41   */
42  // Changes are marked with //CHANGE:
43  // Delegate to FilteredQuery - this version fixes the bug in LUCENE-4705 and uses ApplyAcceptedDocsFilter internally
44  public final class XFilteredQuery extends Query {
45      private final Filter rawFilter;
46      private final FilteredQuery delegate;
47      private final FilterStrategy strategy;
48  
49      /**
50       * Constructs a new query which applies a filter to the results of the original query.
51       * {@link Filter#getDocIdSet} will be called every time this query is used in a search.
52       *
53       * @param query  Query to be filtered, cannot be <code>null</code>.
54       * @param filter Filter to apply to query results, cannot be <code>null</code>.
55       */
56      public XFilteredQuery(Query query, Filter filter) {
57          this(query, filter, FilteredQuery.RANDOM_ACCESS_FILTER_STRATEGY);
58      }
59  
60      /**
61       * Expert: Constructs a new query which applies a filter to the results of the original query.
62       * {@link Filter#getDocIdSet} will be called every time this query is used in a search.
63       *
64       * @param query    Query to be filtered, cannot be <code>null</code>.
65       * @param filter   Filter to apply to query results, cannot be <code>null</code>.
66       * @param strategy a filter strategy used to create a filtered scorer.
67       * @see FilterStrategy
68       */
69      public XFilteredQuery(Query query, Filter filter, FilterStrategy strategy) {
70          this(new FilteredQuery(query, new ApplyAcceptedDocsFilter(filter), strategy), filter, strategy);
71      }
72  
73      private XFilteredQuery(FilteredQuery delegate, Filter filter, FilterStrategy strategy) {
74          this.delegate = delegate;
75          // CHANGE: we need to wrap it in post application of accepted docs
76          this.rawFilter = filter;
77          this.strategy = strategy;
78      }
79  
80      /**
81       * Returns a Weight that applies the filter to the enclosed query's Weight.
82       * This is accomplished by overriding the Scorer returned by the Weight.
83       */
84      @Override
85      public Weight createWeight(final IndexSearcher searcher) throws IOException {
86          return delegate.createWeight(searcher);
87      }
88  
89      /**
90       * Rewrites the query. If the wrapped is an instance of
91       * {@link MatchAllDocsQuery} it returns a {@link ConstantScoreQuery}. Otherwise
92       * it returns a new {@code FilteredQuery} wrapping the rewritten query.
93       */
94      @Override
95      public Query rewrite(IndexReader reader) throws IOException {
96          Query query = delegate.getQuery();
97          final Query queryRewritten = query.rewrite(reader);
98  
99          // CHANGE: if we push back to Lucene, would love to have an extension for "isMatchAllQuery"
100         if (queryRewritten instanceof MatchAllDocsQuery || Queries.isConstantMatchAllQuery(queryRewritten)) {
101             // Special case: If the query is a MatchAllDocsQuery, we only
102             // return a CSQ(filter).
103             final Query rewritten = new ConstantScoreQuery(delegate.getFilter());
104             // Combine boost of MatchAllDocsQuery and the wrapped rewritten query:
105             rewritten.setBoost(delegate.getBoost() * queryRewritten.getBoost());
106             return rewritten;
107         }
108 
109         if (queryRewritten != query) {
110             // rewrite to a new FilteredQuery wrapping the rewritten query
111             final Query rewritten = new XFilteredQuery(queryRewritten, rawFilter, strategy);
112             rewritten.setBoost(delegate.getBoost());
113             return rewritten;
114         } else {
115             // nothing to rewrite, we are done!
116             return this;
117         }
118     }
119 
120     @Override
121     public void setBoost(float b) {
122         delegate.setBoost(b);
123     }
124 
125     @Override
126     public float getBoost() {
127         return delegate.getBoost();
128     }

137     /**
138      * Returns this FilteredQuery's filter
139      */
140     public final Filter getFilter() {
141         // CHANGE: unwrap the accepted docs filter
142         if (rawFilter instanceof ApplyAcceptedDocsFilter) {
143             return ((ApplyAcceptedDocsFilter) rawFilter).filter();
144         }
145         return rawFilter;
146     }

188     /**
189      * Extends {@link org.apache.lucene.search.FilteredQuery.RandomAccessFilterStrategy}.
190      * <p/>
191      * Adds a threshold value, which defaults to -1. When set to -1, it will check if the filter docSet is
192      * *not*  a fast docSet, and if not, it will use {@link FilteredQuery#QUERY_FIRST_FILTER_STRATEGY} (since
193      * the assumption is that its a "slow" filter and better computed only on whatever matched the query).
194      * <p/>
195      * If the threshold value is 0, it always tries to pass "down" the filter as acceptDocs, and it the filter
196      * can't be represented as Bits (never really), then it uses {@link FilteredQuery#LEAP_FROG_QUERY_FIRST_STRATEGY}.
197      * <p/>
198      * If the above conditions are not met, then it reverts to the {@link FilteredQuery.RandomAccessFilterStrategy} logic,
199      * with the threshold used to control {@link #useRandomAccess(org.apache.lucene.util.Bits, int)}.
200      */
201     public static class CustomRandomAccessFilterStrategy extends FilteredQuery.RandomAccessFilterStrategy {
202 
203         private final int threshold;
204 
205         public CustomRandomAccessFilterStrategy() {
206             this.threshold = -1;
207         }
208 
209         public CustomRandomAccessFilterStrategy(int threshold) {
210             this.threshold = threshold;
211         }
212 
213         @Override
214         public Scorer filteredScorer(AtomicReaderContext context, Weight weight, DocIdSet docIdSet) throws IOException {
215             // CHANGE: If threshold is 0, always pass down the accept docs, don't pay the price of calling nextDoc even...
216             if (threshold == 0) {
217                 final Bits filterAcceptDocs = docIdSet.bits();
218                 if (filterAcceptDocs != null) {
219                     return weight.scorer(context, filterAcceptDocs);
220                 } else {
221                     return FilteredQuery.LEAP_FROG_QUERY_FIRST_STRATEGY.filteredScorer(context, weight, docIdSet);
222                 }
223             }
224 
225             // CHANGE: handle "default" value
226             if (threshold == -1) {
227                 // default  value, don't iterate on only apply filter after query if its not a "fast" docIdSet
228                 if (!DocIdSets.isFastIterator(ApplyAcceptedDocsFilter.unwrap(docIdSet))) {
229                     return FilteredQuery.QUERY_FIRST_FILTER_STRATEGY.filteredScorer(context, weight, docIdSet);
230                 }
231             }
232 
233             return super.filteredScorer(context, weight, docIdSet);
234         }
235 
236         /**
237          * Expert: decides if a filter should be executed as "random-access" or not.
238          * random-access means the filter "filters" in a similar way as deleted docs are filtered
239          * in Lucene. This is faster when the filter accepts many documents.
240          * However, when the filter is very sparse, it can be faster to execute the query+filter
241          * as a conjunction in some cases.
242          * <p/>
243          * The default implementation returns <code>true</code> if the first document accepted by the
244          * filter is < threshold, if threshold is -1 (the default), then it checks for < 100.
245          */
246         protected boolean useRandomAccess(Bits bits, int firstFilterDoc) {
247             // "default"
248             if (threshold == -1) {
249                 return firstFilterDoc < 100;
250             }
251             //TODO once we have a cost API on filters and scorers we should rethink this heuristic
252             return firstFilterDoc < threshold;
253         }
254     }
255 
256     @Override
257     public Query clone() {
258         return new XFilteredQuery((FilteredQuery) delegate.clone(), rawFilter, strategy);
259     }
