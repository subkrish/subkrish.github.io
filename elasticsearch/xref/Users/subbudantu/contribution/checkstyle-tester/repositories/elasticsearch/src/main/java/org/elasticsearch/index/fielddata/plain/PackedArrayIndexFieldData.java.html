70      public PackedArrayIndexFieldData(Index index, @IndexSettings Settings indexSettings, FieldMapper.Names fieldNames,
71                                       FieldDataType fieldDataType, IndexFieldDataCache cache, NumericType numericType,
72                                       CircuitBreakerService breakerService) {
73          super(index, indexSettings, fieldNames, fieldDataType, cache);
74          Preconditions.checkNotNull(numericType);
75          Preconditions.checkArgument(EnumSet.of(NumericType.BYTE, NumericType.SHORT, NumericType.INT, NumericType.LONG).contains(numericType), getClass().getSimpleName() + " only supports integer types, not " + numericType);
76          this.numericType = numericType;
77          this.breakerService = breakerService;
78      }

85      @Override
86      public AtomicNumericFieldData loadDirect(AtomicReaderContext context) throws Exception {
87          final AtomicReader reader = context.reader();
88          Terms terms = reader.terms(getFieldNames().indexName());
89          AtomicNumericFieldData data = null;
90          PackedArrayEstimator estimator = new PackedArrayEstimator(breakerService.getBreaker(CircuitBreaker.Name.FIELDDATA), getNumericType(), getFieldNames().fullName());
91          if (terms == null) {
92              data = AtomicLongFieldData.empty(reader.maxDoc());
93              estimator.adjustForNoTerms(data.ramBytesUsed());
94              return data;
95          }
96          // TODO: how can we guess the number of terms? numerics end up creating more terms per value...
97          // Lucene encodes numeric data so that the lexicographical (encoded) order matches the integer order so we know the sequence of
98          // longs is going to be monotonically increasing
99          final PackedLongValues.Builder valuesBuilder = PackedLongValues.monotonicBuilder(PackedInts.COMPACT);
100 
101         final float acceptableTransientOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_transient_overhead_ratio", OrdinalsBuilder.DEFAULT_ACCEPTABLE_OVERHEAD_RATIO);
102         TermsEnum termsEnum = estimator.beforeLoad(terms);
103         boolean success = false;
104         try (OrdinalsBuilder builder = new OrdinalsBuilder(-1, reader.maxDoc(), acceptableTransientOverheadRatio)) {
105             BytesRefIterator iter = builder.buildFromTerms(termsEnum);
106             BytesRef term;
107             assert !getNumericType().isFloatingPoint();
108             final boolean indexedAsLong = getNumericType().requiredBits() > 32;
109             while ((term = iter.next()) != null) {
110                 final long value = indexedAsLong
111                         ? NumericUtils.prefixCodedToLong(term)
112                         : NumericUtils.prefixCodedToInt(term);
113                 valuesBuilder.add(value);
114             }
115             final PackedLongValues values = valuesBuilder.build();
116             final Ordinals build = builder.build(fieldDataType.getSettings());
117             CommonSettings.MemoryStorageFormat formatHint = CommonSettings.getMemoryStorageHint(fieldDataType);
118 
119             RandomAccessOrds ordinals = build.ordinals();
120             if (FieldData.isMultiValued(ordinals) || formatHint == CommonSettings.MemoryStorageFormat.ORDINALS) {
121                 final long ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
122                 data = new AtomicLongFieldData(ramBytesUsed) {
123 
124                     @Override
125                     public SortedNumericDocValues getLongValues() {
126                         return withOrdinals(build, values, reader.maxDoc());
127                     }
128 
129                 };
130             } else {
131                 final FixedBitSet docsWithValues = builder.buildDocsWithValuesSet();
132 
133                 long minV, maxV;
134                 minV = maxV = 0;
135                 if (values.size() > 0) {
136                     minV = values.get(0);
137                     maxV = values.get(values.size() - 1);
138                 }
139 
140 
141                 final float acceptableOverheadRatio = fieldDataType.getSettings().getAsFloat("acceptable_overhead_ratio", PackedInts.DEFAULT);
142                 final int pageSize = fieldDataType.getSettings().getAsInt("single_value_page_size", 1024);
143 
144                 if (formatHint == null) {
145                     formatHint = chooseStorageFormat(reader, values, build, ordinals, minV, maxV, acceptableOverheadRatio, pageSize);
146                 }
147 
148                 logger.trace("single value format for field [{}] set to [{}]", getFieldNames().fullName(), formatHint);
149 
150                 switch (formatHint) {
151                     case PACKED:
152                         // Encode document without a value with a special value
153                         long missingV = 0;
154                         if (docsWithValues != null) {
155                             if ((maxV - minV + 1) == values.size()) {
156                                 // values are dense
157                                 if (minV > Long.MIN_VALUE) {
158                                     missingV = --minV;
159                                 } else {
160                                     assert maxV != Long.MAX_VALUE;
161                                     missingV = ++maxV;
162                                 }
163                             } else {
164                                 for (long i = 1; i < values.size(); ++i) {
165                                     if (values.get(i) > values.get(i - 1) + 1) {
166                                         missingV = values.get(i - 1) + 1;
167                                         break;
168                                     }
169                                 }
170                             }
171                             missingV -= minV;
172                         }
173                         final long missingValue = missingV;
174                         final long minValue = minV;
175                         final long maxValue = maxV;
176 
177                         final long valuesDelta = maxValue - minValue;
178                         int bitsRequired = valuesDelta < 0 ? 64 : PackedInts.bitsRequired(valuesDelta);
179                         final PackedInts.Mutable sValues = PackedInts.getMutable(reader.maxDoc(), bitsRequired, acceptableOverheadRatio);
180 
181                         if (docsWithValues != null) {
182                             sValues.fill(0, sValues.size(), missingV);
183                         }
184 
185                         for (int i = 0; i < reader.maxDoc(); i++) {
186                             ordinals.setDocument(i);
187                             if (ordinals.cardinality() > 0) {
188                                 final long ord = ordinals.ordAt(0);
189                                 long value = values.get(ord);
190                                 sValues.set(i, value - minValue);
191                             }
192                         }
193                         long ramBytesUsed = values.ramBytesUsed() + (docsWithValues == null ? 0 : docsWithValues.ramBytesUsed());
194                         data = new AtomicLongFieldData(ramBytesUsed) {
195 
196                             @Override
197                             public SortedNumericDocValues getLongValues() {
198                                 if (docsWithValues == null) {
199                                     return singles(sValues, minValue);
200                                 } else {
201                                     return sparseSingles(sValues, minValue, missingValue, reader.maxDoc());
202                                 }
203                             }
204 
205                         };
206                         break;
207                     case PAGED:
208                         final PackedLongValues.Builder dpValues = PackedLongValues.deltaPackedBuilder(pageSize, acceptableOverheadRatio);
209 
210                         long lastValue = 0;
211                         for (int i = 0; i < reader.maxDoc(); i++) {
212                             ordinals.setDocument(i);
213                             if (ordinals.cardinality() > 0) {
214                                 final long ord = ordinals.ordAt(i);
215                                 lastValue = values.get(ord);
216                             }
217                             dpValues.add(lastValue);
218                         }
219                         ramBytesUsed = dpValues.ramBytesUsed();
220                         final PackedLongValues pagedValues = dpValues.build();
221                         data = new AtomicLongFieldData(ramBytesUsed) {
222 
223                             @Override
224                             public SortedNumericDocValues getLongValues() {
225                                 return pagedSingles(pagedValues, docsWithValues);
226                             }
227 
228                         };
229                         break;
230                     case ORDINALS:
231                         ramBytesUsed = build.ramBytesUsed() + values.ramBytesUsed();
232                         data = new AtomicLongFieldData(ramBytesUsed) {
233 
234                             @Override
235                             public SortedNumericDocValues getLongValues() {
236                                 return withOrdinals(build, values, reader.maxDoc());
237                             }
238 
239                         };
240                         break;
241                     default:
242                         throw new ElasticsearchException("unknown memory format: " + formatHint);
243                 }
244 
245             }
246 
247             success = true;
248             return data;
249         } finally {
250             if (!success) {
251                 // If something went wrong, unwind any current estimations we've made
252                 estimator.afterLoad(termsEnum, 0);
253             } else {
254                 // Adjust as usual, based on the actual size of the field data
255                 estimator.afterLoad(termsEnum, data.ramBytesUsed());
256             }
257 
258         }
259 
260     }

262     protected CommonSettings.MemoryStorageFormat chooseStorageFormat(AtomicReader reader, PackedLongValues values, Ordinals build, RandomAccessOrds ordinals,
263                                                                      long minValue, long maxValue, float acceptableOverheadRatio, int pageSize) {
264 
265         CommonSettings.MemoryStorageFormat format;
266 
267         // estimate memory usage for a single packed array
268         long packedDelta = maxValue - minValue + 1; // allow for a missing value
269         // valuesDelta can be negative if the difference between max and min values overflows the positive side of longs.
270         int bitsRequired = packedDelta < 0 ? 64 : PackedInts.bitsRequired(packedDelta);
271         PackedInts.FormatAndBits formatAndBits = PackedInts.fastestFormatAndBits(reader.maxDoc(), bitsRequired, acceptableOverheadRatio);
272         final long singleValuesSize = formatAndBits.format.longCount(PackedInts.VERSION_CURRENT, reader.maxDoc(), formatAndBits.bitsPerValue) * 8L;
273 
274         // ordinal memory usage
275         final long ordinalsSize = build.ramBytesUsed() + values.ramBytesUsed();
276 
277         // estimate the memory signature of paged packing
278         long pagedSingleValuesSize = (reader.maxDoc() / pageSize + 1) * RamUsageEstimator.NUM_BYTES_OBJECT_REF; // array of pages
279         int pageIndex = 0;
280         long pageMinOrdinal = Long.MAX_VALUE;
281         long pageMaxOrdinal = Long.MIN_VALUE;
282         for (int i = 1; i < reader.maxDoc(); ++i, pageIndex = (pageIndex + 1) % pageSize) {
283             ordinals.setDocument(i);
284             if (ordinals.cardinality() > 0) {
285                 long ordinal = ordinals.ordAt(0);
286                 pageMaxOrdinal = Math.max(ordinal, pageMaxOrdinal);
287                 pageMinOrdinal = Math.min(ordinal, pageMinOrdinal);
288             }
289             if (pageIndex == pageSize - 1) {
290                 // end of page, we now know enough to estimate memory usage
291                 pagedSingleValuesSize += getPageMemoryUsage(values, acceptableOverheadRatio, pageSize, pageMinOrdinal, pageMaxOrdinal);
292 
293                 pageMinOrdinal = Long.MAX_VALUE;
294                 pageMaxOrdinal = Long.MIN_VALUE;
295             }
296         }
297 
298         if (pageIndex > 0) {
299             // last page estimation
300             pageIndex++;
301             pagedSingleValuesSize += getPageMemoryUsage(values, acceptableOverheadRatio, pageSize, pageMinOrdinal, pageMaxOrdinal);
302         }
303 
304         if (ordinalsSize < singleValuesSize) {
305             if (ordinalsSize < pagedSingleValuesSize) {
306                 format = CommonSettings.MemoryStorageFormat.ORDINALS;
307             } else {
308                 format = CommonSettings.MemoryStorageFormat.PAGED;
309             }
310         } else {
311             if (pagedSingleValuesSize < singleValuesSize) {
312                 format = CommonSettings.MemoryStorageFormat.PAGED;
313             } else {
314                 format = CommonSettings.MemoryStorageFormat.PACKED;
315             }
316         }
317         return format;
318     }

320     private long getPageMemoryUsage(PackedLongValues values, float acceptableOverheadRatio, int pageSize, long pageMinOrdinal, long pageMaxOrdinal) {
321         int bitsRequired;
322         long pageMemorySize = 0;
323         PackedInts.FormatAndBits formatAndBits;
324         if (pageMaxOrdinal == Long.MIN_VALUE) {
325             // empty page - will use the null reader which just stores size
326             pageMemorySize += RamUsageEstimator.alignObjectSize(RamUsageEstimator.NUM_BYTES_OBJECT_HEADER + RamUsageEstimator.NUM_BYTES_INT);
327 
328         } else {
329             long pageMinValue = values.get(pageMinOrdinal);
330             long pageMaxValue = values.get(pageMaxOrdinal);
331             long pageDelta = pageMaxValue - pageMinValue;
332             if (pageDelta != 0) {
333                 bitsRequired = pageDelta < 0 ? 64 : PackedInts.bitsRequired(pageDelta);
334                 formatAndBits = PackedInts.fastestFormatAndBits(pageSize, bitsRequired, acceptableOverheadRatio);
335                 pageMemorySize += formatAndBits.format.longCount(PackedInts.VERSION_CURRENT, pageSize, formatAndBits.bitsPerValue) * RamUsageEstimator.NUM_BYTES_LONG;
336                 pageMemorySize += RamUsageEstimator.NUM_BYTES_LONG; // min value per page storage
337             } else {
338                 // empty page
339                 pageMemorySize += RamUsageEstimator.alignObjectSize(RamUsageEstimator.NUM_BYTES_OBJECT_HEADER + RamUsageEstimator.NUM_BYTES_INT);
340             }
341         }
342         return pageMemorySize;
343     }

350     /**
351      * Estimator that wraps numeric field data loading in a
352      * RamAccountingTermsEnum, adjusting the breaker after data has been
353      * loaded
354      */
355     public class PackedArrayEstimator implements PerValueEstimator {
356 
357         private final CircuitBreaker breaker;
358         private final NumericType type;
359         private final String fieldName;
360 
361         public PackedArrayEstimator(CircuitBreaker breaker, NumericType type, String fieldName) {
362             this.breaker = breaker;
363             this.type = type;
364             this.fieldName = fieldName;
365         }
366 
367         /**
368          * @return number of bytes per term, based on the NumericValue.requiredBits()
369          */
370         @Override
371         public long bytesPerValue(BytesRef term) {
372             // Estimate about  about 0.8 (8 / 10) compression ratio for
373             // numbers, but at least 4 bytes
374             return Math.max(type.requiredBits() / 10, 4);
375         }
376 
377         /**
378          * @return A TermsEnum wrapped in a RamAccountingTermsEnum
379          * @throws IOException
380          */
381         @Override
382         public TermsEnum beforeLoad(Terms terms) throws IOException {
383             return new RamAccountingTermsEnum(type.wrapTermsEnum(terms.iterator(null)), breaker, this, this.fieldName);
384         }
385 
386         /**
387          * Adjusts the breaker based on the aggregated value from the RamAccountingTermsEnum
388          *
389          * @param termsEnum  terms that were wrapped and loaded
390          * @param actualUsed actual field data memory usage
391          */
392         @Override
393         public void afterLoad(TermsEnum termsEnum, long actualUsed) {
394             assert termsEnum instanceof RamAccountingTermsEnum;
395             long estimatedBytes = ((RamAccountingTermsEnum) termsEnum).getTotalBytes();
396             breaker.addWithoutBreaking(-(estimatedBytes - actualUsed));
397         }
398 
399         /**
400          * Adjust the breaker when no terms were actually loaded, but the field
401          * data takes up space regardless. For instance, when ordinals are
402          * used.
403          *
404          * @param actualUsed bytes actually used
405          */
406         public void adjustForNoTerms(long actualUsed) {
407             breaker.addWithoutBreaking(actualUsed);
408         }
409     }

411     private static SortedNumericDocValues withOrdinals(Ordinals ordinals, final LongValues values, int maxDoc) {
412         final RandomAccessOrds ords = ordinals.ordinals();
413         final SortedDocValues singleOrds = DocValues.unwrapSingleton(ords);
414         if (singleOrds != null) {
415             final NumericDocValues singleValues = new NumericDocValues() {
416                 @Override
417                 public long get(int docID) {
418                     final int ord = singleOrds.getOrd(docID);
419                     if (ord >= 0) {
420                         return values.get(singleOrds.getOrd(docID));
421                     } else {
422                         return 0;
423                     }
424                 }
425             };
426             return DocValues.singleton(singleValues, DocValues.docsWithValue(ords, maxDoc));
427         } else {
428             return new SortedNumericDocValues() {
429                 @Override
430                 public long valueAt(int index) {
431                     return values.get(ords.ordAt(index));
432                 }
433 
434                 @Override
435                 public void setDocument(int doc) {
436                     ords.setDocument(doc);
437                 }
438 
439                 @Override
440                 public int count() {
441                     return ords.cardinality();
442                 }
443             };
444         }
445     }

447     private static SortedNumericDocValues singles(final NumericDocValues deltas, final long minValue) {
448         final NumericDocValues values;
449         if (minValue == 0) {
450             values = deltas;
451         } else {
452             values = new NumericDocValues() {
453                 @Override
454                 public long get(int docID) {
455                     return minValue + deltas.get(docID);
456                 }
457             };
458         }
459         return DocValues.singleton(values, null);
460     }
461 
462     private static SortedNumericDocValues sparseSingles(final NumericDocValues deltas, final long minValue,  final long missingValue, final int maxDoc) {
463         final NumericDocValues values = new NumericDocValues() {
464             @Override
465             public long get(int docID) {
466                 final long delta = deltas.get(docID);
467                 if (delta == missingValue) {
468                     return 0;
469                 }
470                 return minValue + delta;
471             }
472         };
473         final Bits docsWithFields = new Bits() {
474             @Override
475             public boolean get(int index) {
476                 return deltas.get(index) != missingValue;
477             }
478             @Override
479             public int length() {
480                 return maxDoc;
481             }
482         };
483         return DocValues.singleton(values, docsWithFields);
484     }
485 
486     private static SortedNumericDocValues pagedSingles(final PackedLongValues values, final FixedBitSet docsWithValue) {
487         return DocValues.singleton(new NumericDocValues() {
488             // we need to wrap since NumericDocValues must return 0 when a doc has no value
489             @Override
490             public long get(int docID) {
491                 if (docsWithValue == null || docsWithValue.get(docID)) {
492                     return values.get(docID);
493                 } else {
494                     return 0;
495                 }
496             }
497         }, docsWithValue);
498     }
