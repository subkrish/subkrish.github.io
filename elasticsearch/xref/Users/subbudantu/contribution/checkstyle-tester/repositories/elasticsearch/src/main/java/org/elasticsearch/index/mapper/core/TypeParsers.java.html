50      public static final String MULTI_FIELD_CONTENT_TYPE = "multi_field";
51      public static final Mapper.TypeParser multiFieldConverterTypeParser = new Mapper.TypeParser() {
52  
53          @Override
54          public Mapper.Builder<?, ?> parse(String name, Map<String, Object> node, ParserContext parserContext) throws MapperParsingException {
55              ContentPath.Type pathType = null;
56              AbstractFieldMapper.Builder mainFieldBuilder = null;
57              List<AbstractFieldMapper.Builder> fields = null;
58              String firstType = null;
59  
60              for (Map.Entry<String, Object> entry : node.entrySet()) {
61                  String fieldName = Strings.toUnderscoreCase(entry.getKey());
62                  Object fieldNode = entry.getValue();
63                  if (fieldName.equals("path")) {
64                      pathType = parsePathType(name, fieldNode.toString());
65                  } else if (fieldName.equals("fields")) {
66                      Map<String, Object> fieldsNode = (Map<String, Object>) fieldNode;
67                      for (Map.Entry<String, Object> entry1 : fieldsNode.entrySet()) {
68                          String propName = entry1.getKey();
69                          Map<String, Object> propNode = (Map<String, Object>) entry1.getValue();
70  
71                          String type;
72                          Object typeNode = propNode.get("type");
73                          if (typeNode != null) {
74                              type = typeNode.toString();
75                              if (firstType == null) {
76                                  firstType = type;
77                              }
78                          } else {
79                              throw new MapperParsingException("No type specified for property [" + propName + "]");
80                          }
81  
82                          Mapper.TypeParser typeParser = parserContext.typeParser(type);
83                          if (typeParser == null) {
84                              throw new MapperParsingException("No handler for type [" + type + "] declared on field [" + fieldName + "]");
85                          }
86                          if (propName.equals(name)) {
87                              mainFieldBuilder = (AbstractFieldMapper.Builder) typeParser.parse(propName, propNode, parserContext);
88                          } else {
89                              if (fields == null) {
90                                  fields = new ArrayList<>(2);
91                              }
92                              fields.add((AbstractFieldMapper.Builder) typeParser.parse(propName, propNode, parserContext));
93                          }
94                      }
95                  }
96              }
97  
98              if (mainFieldBuilder == null) {
99                  if (fields == null) {
100                     // No fields at all were specified in multi_field, so lets return a non indexed string field.
101                     return new StringFieldMapper.Builder(name).index(false);
102                 }
103                 Mapper.TypeParser typeParser = parserContext.typeParser(firstType);
104                 if (typeParser == null) {
105                     // The first multi field's type is unknown
106                     mainFieldBuilder = new StringFieldMapper.Builder(name).index(false);
107                 } else {
108                     Mapper.Builder substitute = typeParser.parse(name, Collections.<String, Object>emptyMap(), parserContext);
109                     if (substitute instanceof AbstractFieldMapper.Builder) {
110                         mainFieldBuilder = ((AbstractFieldMapper.Builder) substitute).index(false);
111                     } else {
112                         // The first multi isn't a core field type
113                         mainFieldBuilder =  new StringFieldMapper.Builder(name).index(false);
114                     }
115                 }
116             }
117 
118             if (fields != null && pathType != null) {
119                 for (Mapper.Builder field : fields) {
120                     mainFieldBuilder.addMultiField(field);
121                 }
122                 mainFieldBuilder.multiFieldPathType(pathType);
123             } else if (fields != null) {
124                 for (Mapper.Builder field : fields) {
125                     mainFieldBuilder.addMultiField(field);
126                 }
127             } else if (pathType != null) {
128                 mainFieldBuilder.multiFieldPathType(pathType);
129             }
130             return mainFieldBuilder;
131         }
132

141     public static void parseNumberField(NumberFieldMapper.Builder builder, String name, Map<String, Object> numberNode, Mapper.TypeParser.ParserContext parserContext) {
142         parseField(builder, name, numberNode, parserContext);
143         for (Map.Entry<String, Object> entry : numberNode.entrySet()) {
144             String propName = Strings.toUnderscoreCase(entry.getKey());
145             Object propNode = entry.getValue();
146             if (propName.equals("precision_step")) {
147                 builder.precisionStep(nodeIntegerValue(propNode));
148             } else if (propName.equals("ignore_malformed")) {
149                 builder.ignoreMalformed(nodeBooleanValue(propNode));
150             } else if (propName.equals("coerce")) {
151                 builder.coerce(nodeBooleanValue(propNode));
152             } else if (propName.equals("omit_norms")) {
153                 builder.omitNorms(nodeBooleanValue(propNode));
154             } else if (propName.equals("similarity")) {
155                 builder.similarity(parserContext.similarityLookupService().similarity(propNode.toString()));
156             } else {
157                 parseMultiField(builder, name, parserContext, propName, propNode);
158             }
159         }
160     }

162     public static void parseField(AbstractFieldMapper.Builder builder, String name, Map<String, Object> fieldNode, Mapper.TypeParser.ParserContext parserContext) {
163         for (Map.Entry<String, Object> entry : fieldNode.entrySet()) {
164             final String propName = Strings.toUnderscoreCase(entry.getKey());
165             final Object propNode = entry.getValue();
166             if (propName.equals("index_name")) {
167                 builder.indexName(propNode.toString());
168             } else if (propName.equals("store")) {
169                 builder.store(parseStore(name, propNode.toString()));
170             } else if (propName.equals("index")) {
171                 parseIndex(name, propNode.toString(), builder);
172             } else if (propName.equals("tokenized")) {
173                 builder.tokenized(nodeBooleanValue(propNode));
174             } else if (propName.equals(DOC_VALUES)) {
175                 builder.docValues(nodeBooleanValue(propNode));
176             } else if (propName.equals("term_vector")) {
177                 parseTermVector(name, propNode.toString(), builder);
178             } else if (propName.equals("boost")) {
179                 builder.boost(nodeFloatValue(propNode));
180             } else if (propName.equals("store_term_vectors")) {
181                 builder.storeTermVectors(nodeBooleanValue(propNode));
182             } else if (propName.equals("store_term_vector_offsets")) {
183                 builder.storeTermVectorOffsets(nodeBooleanValue(propNode));
184             } else if (propName.equals("store_term_vector_positions")) {
185                 builder.storeTermVectorPositions(nodeBooleanValue(propNode));
186             } else if (propName.equals("store_term_vector_payloads")) {
187                 builder.storeTermVectorPayloads(nodeBooleanValue(propNode));
188             } else if (propName.equals("omit_norms")) {
189                 builder.omitNorms(nodeBooleanValue(propNode));
190             } else if (propName.equals("norms")) {
191                 final Map<String, Object> properties = nodeMapValue(propNode, "norms");
192                 for (Map.Entry<String, Object> entry2 : properties.entrySet()) {
193                     final String propName2 = Strings.toUnderscoreCase(entry2.getKey());
194                     final Object propNode2 = entry2.getValue();
195                     if (propName2.equals("enabled")) {
196                         builder.omitNorms(!nodeBooleanValue(propNode2));
197                     } else if (propName2.equals(Loading.KEY)) {
198                         builder.normsLoading(Loading.parse(nodeStringValue(propNode2, null), null));
199                     }
200                 }
201             } else if (propName.equals("omit_term_freq_and_positions")) {
202                 final IndexOptions op = nodeBooleanValue(propNode) ? IndexOptions.DOCS_ONLY : IndexOptions.DOCS_AND_FREQS_AND_POSITIONS;
203                 if (parserContext.indexVersionCreated().onOrAfter(Version.V_1_0_0_RC2)) {
204                     throw new ElasticsearchParseException("'omit_term_freq_and_positions' is not supported anymore - use ['index_options' : '" + op.name() + "']  instead");
205                 }
206                 // deprecated option for BW compat
207                 builder.indexOptions(op);
208             } else if (propName.equals("index_options")) {
209                 builder.indexOptions(nodeIndexOptionValue(propNode));
210             } else if (propName.equals("analyzer")) {
211                 NamedAnalyzer analyzer = parserContext.analysisService().analyzer(propNode.toString());
212                 if (analyzer == null) {
213                     throw new MapperParsingException("Analyzer [" + propNode.toString() + "] not found for field [" + name + "]");
214                 }
215                 builder.indexAnalyzer(analyzer);
216                 builder.searchAnalyzer(analyzer);
217             } else if (propName.equals("index_analyzer")) {
218                 NamedAnalyzer analyzer = parserContext.analysisService().analyzer(propNode.toString());
219                 if (analyzer == null) {
220                     throw new MapperParsingException("Analyzer [" + propNode.toString() + "] not found for field [" + name + "]");
221                 }
222                 builder.indexAnalyzer(analyzer);
223             } else if (propName.equals("search_analyzer")) {
224                 NamedAnalyzer analyzer = parserContext.analysisService().analyzer(propNode.toString());
225                 if (analyzer == null) {
226                     throw new MapperParsingException("Analyzer [" + propNode.toString() + "] not found for field [" + name + "]");
227                 }
228                 builder.searchAnalyzer(analyzer);
229             } else if (propName.equals("include_in_all")) {
230                 builder.includeInAll(nodeBooleanValue(propNode));
231             } else if (propName.equals("postings_format")) {
232                 String postingFormatName = propNode.toString();
233                 builder.postingsFormat(parserContext.postingFormatService().get(postingFormatName));
234             } else if (propName.equals(DOC_VALUES_FORMAT)) {
235                 String docValuesFormatName = propNode.toString();
236                 builder.docValuesFormat(parserContext.docValuesFormatService().get(docValuesFormatName));
237             } else if (propName.equals("similarity")) {
238                 builder.similarity(parserContext.similarityLookupService().similarity(propNode.toString()));
239             } else if (propName.equals("fielddata")) {
240                 final Settings settings = ImmutableSettings.builder().put(SettingsLoader.Helper.loadNestedFromMap(nodeMapValue(propNode, "fielddata"))).build();
241                 builder.fieldDataSettings(settings);
242             } else if (propName.equals("copy_to")) {
243                 parseCopyFields(propNode, builder);
244             }
245         }
246     }

248     public static void parseMultiField(AbstractFieldMapper.Builder builder, String name, Mapper.TypeParser.ParserContext parserContext, String propName, Object propNode) {
249         if (propName.equals("path")) {
250             builder.multiFieldPathType(parsePathType(name, propNode.toString()));
251         } else if (propName.equals("fields")) {
252 
253             final Map<String, Object> multiFieldsPropNodes;
254 
255             if (propNode instanceof List && ((List<?>) propNode).isEmpty()) {
256                 multiFieldsPropNodes = Collections.emptyMap();
257             } else if (propNode instanceof Map) {
258                 multiFieldsPropNodes = (Map<String, Object>) propNode;
259             } else {
260                 throw new MapperParsingException("Expected map for property [fields] on field [" + propNode + "] or " +
261                         "[" + propName + "] but got a " + propNode.getClass());
262             }
263 
264             for (Map.Entry<String, Object> multiFieldEntry : multiFieldsPropNodes.entrySet()) {
265                 String multiFieldName = multiFieldEntry.getKey();
266                 if (!(multiFieldEntry.getValue() instanceof Map)) {
267                     throw new MapperParsingException("Illegal field [" + multiFieldName + "], only fields can be specified inside fields");
268                 }
269                 @SuppressWarnings("unchecked")
270                 Map<String, Object> multiFieldNodes = (Map<String, Object>) multiFieldEntry.getValue();
271 
272                 String type;
273                 Object typeNode = multiFieldNodes.get("type");
274                 if (typeNode != null) {
275                     type = typeNode.toString();
276                 } else {
277                     throw new MapperParsingException("No type specified for property [" + multiFieldName + "]");
278                 }
279 
280                 Mapper.TypeParser typeParser = parserContext.typeParser(type);
281                 if (typeParser == null) {
282                     throw new MapperParsingException("No handler for type [" + type + "] declared on field [" + multiFieldName + "]");
283                 }
284                 builder.addMultiField(typeParser.parse(multiFieldName, multiFieldNodes, parserContext));
285             }
286         }

289     private static IndexOptions nodeIndexOptionValue(final Object propNode) {
290         final String value = propNode.toString();
291         if (INDEX_OPTIONS_OFFSETS.equalsIgnoreCase(value)) {
292             return IndexOptions.DOCS_AND_FREQS_AND_POSITIONS_AND_OFFSETS;
293         } else if (INDEX_OPTIONS_POSITIONS.equalsIgnoreCase(value)) {
294             return IndexOptions.DOCS_AND_FREQS_AND_POSITIONS;
295         } else if (INDEX_OPTIONS_FREQS.equalsIgnoreCase(value)) {
296             return IndexOptions.DOCS_AND_FREQS;
297         } else if (INDEX_OPTIONS_DOCS.equalsIgnoreCase(value)) {
298             return IndexOptions.DOCS_ONLY;
299         } else {
300             throw new ElasticsearchParseException("Failed to parse index option [" + value + "]");
301         }
302     }

308     public static void parseTermVector(String fieldName, String termVector, AbstractFieldMapper.Builder builder) throws MapperParsingException {
309         termVector = Strings.toUnderscoreCase(termVector);
310         if ("no".equals(termVector)) {
311             builder.storeTermVectors(false);
312         } else if ("yes".equals(termVector)) {
313             builder.storeTermVectors(true);
314         } else if ("with_offsets".equals(termVector)) {
315             builder.storeTermVectorOffsets(true);
316         } else if ("with_positions".equals(termVector)) {
317             builder.storeTermVectorPositions(true);
318         } else if ("with_positions_offsets".equals(termVector)) {
319             builder.storeTermVectorPositions(true);
320             builder.storeTermVectorOffsets(true);
321         } else if ("with_positions_payloads".equals(termVector)) {
322             builder.storeTermVectorPositions(true);
323             builder.storeTermVectorPayloads(true);
324         } else if ("with_positions_offsets_payloads".equals(termVector)) {
325             builder.storeTermVectorPositions(true);
326             builder.storeTermVectorOffsets(true);
327             builder.storeTermVectorPayloads(true);
328         } else {
329             throw new MapperParsingException("Wrong value for termVector [" + termVector + "] for field [" + fieldName + "]");
330         }
331     }

333     public static void parseIndex(String fieldName, String index, AbstractFieldMapper.Builder builder) throws MapperParsingException {
334         index = Strings.toUnderscoreCase(index);
335         if ("no".equals(index)) {
336             builder.index(false);
337         } else if ("not_analyzed".equals(index)) {
338             builder.index(true);
339             builder.tokenized(false);
340         } else if ("analyzed".equals(index)) {
341             builder.index(true);
342             builder.tokenized(true);
343         } else {
344             throw new MapperParsingException("Wrong value for index [" + index + "] for field [" + fieldName + "]");
345         }
346     }

348     public static boolean parseStore(String fieldName, String store) throws MapperParsingException {
349         if ("no".equals(store)) {
350             return false;
351         } else if ("yes".equals(store)) {
352             return true;
353         } else {
354             return nodeBooleanValue(store);
355         }
356     }

358     public static ContentPath.Type parsePathType(String name, String path) throws MapperParsingException {
359         path = Strings.toUnderscoreCase(path);
360         if ("just_name".equals(path)) {
361             return ContentPath.Type.JUST_NAME;
362         } else if ("full".equals(path)) {
363             return ContentPath.Type.FULL;
364         } else {
365             throw new MapperParsingException("Wrong value for pathType [" + path + "] for object [" + name + "]");
366         }
367     }

369     @SuppressWarnings("unchecked")
370     public static void parseCopyFields(Object propNode, AbstractFieldMapper.Builder builder) {
371         AbstractFieldMapper.CopyTo.Builder copyToBuilder = new AbstractFieldMapper.CopyTo.Builder();
372         if (isArray(propNode)) {
373             for(Object node : (List<Object>) propNode) {
374                 copyToBuilder.add(nodeStringValue(node, null));
375             }
376         } else {
377             copyToBuilder.add(nodeStringValue(propNode, null));
378         }
379         builder.copyTo(copyToBuilder.build());
380     }
