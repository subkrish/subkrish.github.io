89      /**
90       * Runner to test our cache cases when using date range filter
91       * @param lte could be null
92       * @param gte could be null
93       * @param forcedCache true if we want to force the cache, false if we want to force no cache, null either
94       * @param expectedCache true if we expect a cached filter
95       */
96      private void testDateRangeFilterCache(IndexQueryParserService queryParser, Object gte, Object lte, Boolean forcedCache, boolean expectedCache) {
97          RangeFilterBuilder filterBuilder = FilterBuilders.rangeFilter("born")
98                  .gte(gte)
99                  .lte(lte);
100         if (forcedCache != null) {
101             filterBuilder.cache(forcedCache);
102         }
103 
104         Query parsedQuery = queryParser.parse(QueryBuilders.constantScoreQuery(filterBuilder)).query();
105         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
106 
107 
108         if (expectedCache) {
109             if (((ConstantScoreQuery)parsedQuery).getFilter() instanceof CachedFilter) {
110                 logger.info("gte [{}], lte [{}], _cache [{}] is cached", gte, lte, forcedCache);
111             } else {
112                 logger.warn("gte [{}], lte [{}], _cache [{}] should be cached", gte, lte, forcedCache);
113             }
114         } else {
115             if (((ConstantScoreQuery)parsedQuery).getFilter() instanceof NoCacheFilter) {
116                 logger.info("gte [{}], lte [{}], _cache [{}] is not cached", gte, lte, forcedCache);
117             } else {
118                 logger.warn("gte [{}], lte [{}], _cache [{}] should not be cached", gte, lte, forcedCache);
119             }
120         }
121 
122        if (expectedCache) {
123             assertThat(((ConstantScoreQuery)parsedQuery).getFilter(), instanceOf(CachedFilter.class));
124         } else {
125             assertThat(((ConstantScoreQuery)parsedQuery).getFilter(), instanceOf(NoCacheFilter.class));
126         }
127     }

201     @Test
202     public void testNoFilterParsing() throws IOException {
203         IndexQueryParserService queryParser = queryParser();
204         String query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean.json");
205         Query parsedQuery = queryParser.parse(query).query();
206         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
207         assertThat(((ConstantScoreQuery)parsedQuery).getFilter(), instanceOf(XBooleanFilter.class));
208         assertThat(((XBooleanFilter)((ConstantScoreQuery)parsedQuery).getFilter()).clauses().get(1).getFilter(), instanceOf(NoCacheFilter.class));
209         assertThat(((XBooleanFilter)((ConstantScoreQuery)parsedQuery).getFilter()).clauses().size(), is(2));
210 
211         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_with_long_value.json");
212         parsedQuery = queryParser.parse(query).query();
213         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
214         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(XBooleanFilter.class));
215         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().get(1).getFilter(), instanceOf(CachedFilter.class));
216         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().size(), is(2));
217 
218         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_with_long_value_not_cached.json");
219         parsedQuery = queryParser.parse(query).query();
220         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
221         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(XBooleanFilter.class));
222         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().get(1).getFilter(), instanceOf(NoCacheFilter.class));
223         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().size(), is(2));
224 
225         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_cached_now.json");
226         parsedQuery = queryParser.parse(query).query();
227         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
228         assertThat(((ConstantScoreQuery)parsedQuery).getFilter(), instanceOf(XBooleanFilter.class));
229         assertThat(((XBooleanFilter)((ConstantScoreQuery)parsedQuery).getFilter()).clauses().get(1).getFilter(), instanceOf(NoCacheFilter.class));
230         assertThat(((XBooleanFilter)((ConstantScoreQuery)parsedQuery).getFilter()).clauses().size(), is(2));
231 
232         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_cached_complex_now.json");
233         parsedQuery = queryParser.parse(query).query();
234         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
235         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(XBooleanFilter.class));
236         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().get(1).getFilter(), instanceOf(NoCacheFilter.class));
237         assertThat(((XBooleanFilter) ((ConstantScoreQuery) parsedQuery).getFilter()).clauses().size(), is(2));
238 
239         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_cached.json");
240         parsedQuery = queryParser.parse(query).query();
241         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
242         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(CachedFilter.class));
243 
244         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_cached_now_with_rounding.json");
245         parsedQuery = queryParser.parse(query).query();
246         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
247         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(CachedFilter.class));
248 
249         query = copyToStringFromClasspath("/org/elasticsearch/index/query/date_range_in_boolean_cached_complex_now_with_rounding.json");
250         parsedQuery = queryParser.parse(query).query();
251         assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
252         assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(CachedFilter.class));
253 
254         try {
255             SearchContext.setCurrent(new TestSearchContext());
256             query = copyToStringFromClasspath("/org/elasticsearch/index/query/has-child.json");
257             parsedQuery = queryParser.parse(query).query();
258             assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
259             assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(NoCacheFilter.class));
260

268             assertThat(parsedQuery, instanceOf(ConstantScoreQuery.class));
269             assertThat(((ConstantScoreQuery) parsedQuery).getFilter(), instanceOf(AndFilter.class));
270         } finally {
271             SearchContext.removeCurrent();
