24  package org.hibernate.event.internal;
25  
26  import org.hibernate.AssertionFailure;
27  import org.hibernate.LockMode;
28  import org.hibernate.action.internal.EntityIncrementVersionProcess;
29  import org.hibernate.action.internal.EntityVerifyVersionProcess;
30  import org.hibernate.classic.Lifecycle;
31  import org.hibernate.engine.spi.EntityEntry;
32  import org.hibernate.event.spi.PostLoadEvent;
33  import org.hibernate.event.spi.PostLoadEventListener;
34  import org.hibernate.persister.entity.EntityPersister;
35  
36  /**
37   * We do 2 things here:<ul>
38   * <li>Call {@link Lifecycle} interface if necessary</li>
39   * <li>Perform needed {@link EntityEntry#getLockMode()} related processing</li>
40   * </ul>
41   *
42   * @author Gavin King
43   * @author Steve Ebersole
44   */
45  public class DefaultPostLoadEventListener implements PostLoadEventListener {
46  	
47  	public void onPostLoad(PostLoadEvent event) {
48  		final Object entity = event.getEntity();
49  		final EntityEntry entry = event.getSession().getPersistenceContext().getEntry( entity );
50  		if ( entry == null ) {
51  			throw new AssertionFailure( "possible non-threadsafe access to the session" );
52  		}
53  
54  		final LockMode lockMode = entry.getLockMode();
55  		if ( LockMode.PESSIMISTIC_FORCE_INCREMENT.equals( lockMode ) ) {
56  			final EntityPersister persister = entry.getPersister();
57  			Object nextVersion = persister.forceVersionIncrement(
58  					entry.getId(), entry.getVersion(), event.getSession()
59  			);
60  			entry.forceLocked( entity, nextVersion );
61  		}
62  		else if ( LockMode.OPTIMISTIC_FORCE_INCREMENT.equals( lockMode ) ) {
63  			EntityIncrementVersionProcess incrementVersion = new EntityIncrementVersionProcess( entity, entry );
64  			event.getSession().getActionQueue().registerProcess( incrementVersion );
65  		}
66  		else if ( LockMode.OPTIMISTIC.equals( lockMode ) ) {
67  			EntityVerifyVersionProcess verifyVersion = new EntityVerifyVersionProcess( entity, entry );
68  			event.getSession().getActionQueue().registerProcess( verifyVersion );
69  		}
70  
71  		if ( event.getPersister().implementsLifecycle() ) {
72  			//log.debug( "calling onLoad()" );
73  			( ( Lifecycle ) event.getEntity() ).onLoad( event.getSession(), event.getId() );
74  		}
75  
76  	}
77  }
