302 	private IdType determineIdType() {
303 		List<AnnotationInstance> idAnnotations = findIdAnnotations( JPADotNames.ID );
304 		List<AnnotationInstance> embeddedIdAnnotations = findIdAnnotations( JPADotNames.EMBEDDED_ID );
305 
306 		if ( !idAnnotations.isEmpty() && !embeddedIdAnnotations.isEmpty() ) {
307 			throw new MappingException(
308 					"@EmbeddedId and @Id cannot be used together. Check the configuration for " + getName() + "."
309 			);
310 		}
311 
312 		if ( !embeddedIdAnnotations.isEmpty() ) {
313 			if ( embeddedIdAnnotations.size() == 1 ) {
314 				return IdType.EMBEDDED;
315 			}
316 			else {
317 				throw new AnnotationException( "Multiple @EmbeddedId annotations are not allowed" );
318 			}
319 		}
320 
321 		if ( !idAnnotations.isEmpty() ) {
322 			return idAnnotations.size() == 1 ? IdType.SIMPLE : IdType.COMPOSED;
323 		}
324 		return IdType.NONE;
325 	}

327 	private List<AnnotationInstance> findIdAnnotations(DotName idAnnotationType) {
328 		List<AnnotationInstance> idAnnotationList = new ArrayList<AnnotationInstance>();
329 		if ( getClassInfo().annotations().containsKey( idAnnotationType ) ) {
330 			idAnnotationList.addAll( getClassInfo().annotations().get( idAnnotationType ) );
331 		}
332 		ConfiguredClass parent = getParent();
333 		while ( parent != null ) {
334 			if ( parent.getClassInfo().annotations().containsKey( idAnnotationType ) ) {
335 				idAnnotationList.addAll( parent.getClassInfo().annotations().get( idAnnotationType ) );
336 			}
337 			parent = parent.getParent();
338 		}
339 		return idAnnotationList;
340 	}

342 	private void processDiscriminator() {
343 		if ( !InheritanceType.SINGLE_TABLE.equals( inheritanceType ) ) {
344 			return;
345 		}
346 
347 		final AnnotationInstance discriminatorValueAnnotation = JandexHelper.getSingleAnnotation(
348 				getClassInfo(), JPADotNames.DISCRIMINATOR_VALUE
349 		);
350 		if ( discriminatorValueAnnotation != null ) {
351 			this.discriminatorMatchValue = discriminatorValueAnnotation.value().asString();
352 		}
353 
354 		final AnnotationInstance discriminatorColumnAnnotation = JandexHelper.getSingleAnnotation(
355 				getClassInfo(), JPADotNames.DISCRIMINATOR_COLUMN
356 		);
357 
358 		final AnnotationInstance discriminatorFormulaAnnotation = JandexHelper.getSingleAnnotation(
359 				getClassInfo(),
360 				HibernateDotNames.DISCRIMINATOR_FORMULA
361 		);
362 
363 
364 		Class<?> type = String.class; // string is the discriminator default
365 		if ( discriminatorFormulaAnnotation != null ) {
366 			String expression = JandexHelper.getValue( discriminatorFormulaAnnotation, "value", String.class );
367 			discriminatorFormula = new FormulaValue( getPrimaryTableSource().getExplicitTableName(), expression );
368 		}
369 		discriminatorColumnValues = new ColumnValues( null ); //(stliu) give null here, will populate values below
370 		discriminatorColumnValues.setNullable( false ); // discriminator column cannot be null
371 		if ( discriminatorColumnAnnotation != null ) {
372 
373 			DiscriminatorType discriminatorType = Enum.valueOf(
374 					DiscriminatorType.class, discriminatorColumnAnnotation.value( "discriminatorType" ).asEnum()
375 			);
376 			switch ( discriminatorType ) {
377 				case STRING: {
378 					type = String.class;
379 					break;
380 				}
381 				case CHAR: {
382 					type = Character.class;
383 					break;
384 				}
385 				case INTEGER: {
386 					type = Integer.class;
387 					break;
388 				}
389 				default: {
390 					throw new AnnotationException( "Unsupported discriminator type: " + discriminatorType );
391 				}
392 			}
393 
394 			discriminatorColumnValues.setName(
395 					JandexHelper.getValue(
396 							discriminatorColumnAnnotation,
397 							"name",
398 							String.class
399 					)
400 			);
401 			discriminatorColumnValues.setLength(
402 					JandexHelper.getValue(
403 							discriminatorColumnAnnotation,
404 							"length",
405 							Integer.class
406 					)
407 			);
408 			discriminatorColumnValues.setColumnDefinition(
409 					JandexHelper.getValue(
410 							discriminatorColumnAnnotation,
411 							"columnDefinition",
412 							String.class
413 					)
414 			);
415 		}
416 		discriminatorType = type;
417 
418 		AnnotationInstance discriminatorOptionsAnnotation = JandexHelper.getSingleAnnotation(
419 				getClassInfo(), HibernateDotNames.DISCRIMINATOR_OPTIONS
420 		);
421 		if ( discriminatorOptionsAnnotation != null ) {
422 			isDiscriminatorForced = discriminatorOptionsAnnotation.value( "force" ).asBoolean();
423 			isDiscriminatorIncludedInSql = discriminatorOptionsAnnotation.value( "insert" ).asBoolean();
424 		}
425 		else {
426 			isDiscriminatorForced = false;
427 			isDiscriminatorIncludedInSql = true;
428 		}
429 	}

431 	private void processHibernateEntitySpecificAnnotations() {
432 		final AnnotationInstance hibernateEntityAnnotation = JandexHelper.getSingleAnnotation(
433 				getClassInfo(), HibernateDotNames.ENTITY
434 		);
435 
436 		// see HHH-6400
437 		PolymorphismType polymorphism = PolymorphismType.IMPLICIT;
438 		if ( hibernateEntityAnnotation != null && hibernateEntityAnnotation.value( "polymorphism" ) != null ) {
439 			polymorphism = PolymorphismType.valueOf( hibernateEntityAnnotation.value( "polymorphism" ).asEnum() );
440 		}
441 		isExplicitPolymorphism = polymorphism == PolymorphismType.EXPLICIT;
442 
443 		// see HHH-6401
444 		OptimisticLockType optimisticLockType = OptimisticLockType.VERSION;
445 		if ( hibernateEntityAnnotation != null && hibernateEntityAnnotation.value( "optimisticLock" ) != null ) {
446 			optimisticLockType = OptimisticLockType.valueOf(
447 					hibernateEntityAnnotation.value( "optimisticLock" )
448 							.asEnum()
449 			);
450 		}
451 		optimisticLockStyle = OptimisticLockStyle.valueOf( optimisticLockType.name() );
452 
453 		final AnnotationInstance hibernateImmutableAnnotation = JandexHelper.getSingleAnnotation(
454 				getClassInfo(), HibernateDotNames.IMMUTABLE
455 		);
456 		isMutable = hibernateImmutableAnnotation == null
457 				&& hibernateEntityAnnotation != null
458 				&& hibernateEntityAnnotation.value( "mutable" ) != null
459 				&& hibernateEntityAnnotation.value( "mutable" ).asBoolean();
460 
461 
462 		final AnnotationInstance whereAnnotation = JandexHelper.getSingleAnnotation(
463 				getClassInfo(), HibernateDotNames.WHERE
464 		);
465 		whereClause = whereAnnotation != null && whereAnnotation.value( "clause" ) != null ?
466 				whereAnnotation.value( "clause" ).asString() : null;
467 
468 		final AnnotationInstance rowIdAnnotation = JandexHelper.getSingleAnnotation(
469 				getClassInfo(), HibernateDotNames.ROW_ID
470 		);
471 		rowId = rowIdAnnotation != null && rowIdAnnotation.value() != null
472 				? rowIdAnnotation.value().asString() : null;
473 
474 		caching = determineCachingSettings();
475 
476 		// see HHH-6397
477 		isDynamicInsert =
478 				hibernateEntityAnnotation != null
479 						&& hibernateEntityAnnotation.value( "dynamicInsert" ) != null
480 						&& hibernateEntityAnnotation.value( "dynamicInsert" ).asBoolean();
481 
482 		// see HHH-6398
483 		isDynamicUpdate =
484 				hibernateEntityAnnotation != null
485 						&& hibernateEntityAnnotation.value( "dynamicUpdate" ) != null
486 						&& hibernateEntityAnnotation.value( "dynamicUpdate" ).asBoolean();
487 
488 
489 		// see HHH-6399
490 		isSelectBeforeUpdate =
491 				hibernateEntityAnnotation != null
492 						&& hibernateEntityAnnotation.value( "selectBeforeUpdate" ) != null
493 						&& hibernateEntityAnnotation.value( "selectBeforeUpdate" ).asBoolean();
494 
495 		// Custom persister
496 		final String entityPersisterClass;
497 		final AnnotationInstance persisterAnnotation = JandexHelper.getSingleAnnotation(
498 				getClassInfo(), HibernateDotNames.PERSISTER
499 		);
500 		if ( persisterAnnotation == null || persisterAnnotation.value( "impl" ) == null ) {
501 			if ( hibernateEntityAnnotation != null && hibernateEntityAnnotation.value( "persister" ) != null ) {
502 				entityPersisterClass = hibernateEntityAnnotation.value( "persister" ).asString();
503 			}
504 			else {
505 				entityPersisterClass = null;
506 			}
507 		}
508 		else {
509 			if ( hibernateEntityAnnotation != null && hibernateEntityAnnotation.value( "persister" ) != null ) {
510 				// todo : error?
511 			}
512 			entityPersisterClass = persisterAnnotation.value( "impl" ).asString();
513 		}
514 		this.customPersister = entityPersisterClass;
515 	}

517 	private Caching determineCachingSettings() {
518 		final AnnotationInstance hibernateCacheAnnotation = JandexHelper.getSingleAnnotation(
519 				getClassInfo(), HibernateDotNames.CACHE
520 		);
521 		if ( hibernateCacheAnnotation != null ) {
522 			final org.hibernate.cache.spi.access.AccessType accessType = hibernateCacheAnnotation.value( "usage" ) == null
523 					? getLocalBindingContext().getMappingDefaults().getCacheAccessType()
524 					: CacheConcurrencyStrategy.parse( hibernateCacheAnnotation.value( "usage" ).asEnum() )
525 					.toAccessType();
526 			return new Caching(
527 					hibernateCacheAnnotation.value( "region" ) == null
528 							? getName()
529 							: hibernateCacheAnnotation.value( "region" ).asString(),
530 					accessType,
531 					hibernateCacheAnnotation.value( "include" ) != null
532 							&& "all".equals( hibernateCacheAnnotation.value( "include" ).asString() )
533 			);
534 		}
535 
536 		final AnnotationInstance jpaCacheableAnnotation = JandexHelper.getSingleAnnotation(
537 				getClassInfo(), JPADotNames.CACHEABLE
538 		);
539 
540 		boolean cacheable = true; // true is the default
541 		if ( jpaCacheableAnnotation != null && jpaCacheableAnnotation.value() != null ) {
542 			cacheable = jpaCacheableAnnotation.value().asBoolean();
543 		}
544 
545 		final boolean doCaching;
546 		switch ( getLocalBindingContext().getMetadataImplementor().getOptions().getSharedCacheMode() ) {
547 			case ALL: {
548 				doCaching = true;
549 				break;
550 			}
551 			case ENABLE_SELECTIVE: {
552 				doCaching = cacheable;
553 				break;
554 			}
555 			case DISABLE_SELECTIVE: {
556 				doCaching = jpaCacheableAnnotation == null || cacheable;
557 				break;
558 			}
559 			default: {
560 				// treat both NONE and UNSPECIFIED the same
561 				doCaching = false;
562 				break;
563 			}
564 		}
565 
566 		if ( !doCaching ) {
567 			return null;
568 		}
569 
570 		return new Caching(
571 				getName(),
572 				getLocalBindingContext().getMappingDefaults().getCacheAccessType(),
573 				true
574 		);
575 	}

577 	/**
578 	 * todo see {@code Binder#createTable}
579 	 *
580 	 * @param tableAnnotation a annotation instance, either {@link javax.persistence.Table} or {@link javax.persistence.SecondaryTable}
581 	 *
582 	 * @return A table source for the specified annotation instance
583 	 */
584 	private TableSource createTableSource(AnnotationInstance tableAnnotation) {
585 		String schema = null;
586 		String catalog = null;
587 		if ( tableAnnotation != null ) {
588 			schema = JandexHelper.getValue( tableAnnotation, "schema", String.class );
589 			catalog = JandexHelper.getValue( tableAnnotation, "catalog", String.class );
590 		}
591 		// process the table name
592 		String tableName = null;
593 		String logicalTableName = null;
594 
595 		if ( tableAnnotation != null ) {
596 			logicalTableName = JandexHelper.getValue( tableAnnotation, "name", String.class );
597 			if ( StringHelper.isNotEmpty( logicalTableName ) ) {
598 				tableName = logicalTableName;
599 			}
600 			createUniqueConstraints( tableAnnotation, tableName );
601 		}
602 
603 		TableSourceImpl tableSourceImpl;
604 		if ( tableAnnotation == null || JPADotNames.TABLE.equals( tableAnnotation.name() ) ) {
605 			// for the main table @Table we use 'null' as logical name
606 			tableSourceImpl = new TableSourceImpl( schema, catalog, tableName, null );
607 		}
608 		else {
609 			// for secondary tables a name must be specified which is used as logical table name
610 			tableSourceImpl = new TableSourceImpl( schema, catalog, tableName, logicalTableName );
611 		}
612 		return tableSourceImpl;
613 	}

615 	private Set<TableSource> createSecondaryTableSources() {
616 		Set<TableSource> secondaryTableSources = new HashSet<TableSource>();
617 		AnnotationInstance secondaryTables = JandexHelper.getSingleAnnotation(
618 				getClassInfo(),
619 				JPADotNames.SECONDARY_TABLES
620 		);
621 		AnnotationInstance secondaryTable = JandexHelper.getSingleAnnotation(
622 				getClassInfo(),
623 				JPADotNames.SECONDARY_TABLE
624 		);
625 		// collect all @secondaryTable annotations
626 		List<AnnotationInstance> secondaryTableAnnotations = new ArrayList<AnnotationInstance>();
627 		if ( secondaryTable != null ) {
628 			secondaryTableAnnotations.add(
629 					secondaryTable
630 			);
631 		}
632 
633 		if ( secondaryTables != null ) {
634 			secondaryTableAnnotations.addAll(
635 					Arrays.asList(
636 							JandexHelper.getValue( secondaryTables, "value", AnnotationInstance[].class )
637 					)
638 			);
639 		}
640 
641 		// create table sources
642 		for ( AnnotationInstance annotationInstance : secondaryTableAnnotations ) {
643 			secondaryTableSources.add( createTableSource( annotationInstance ) );
644 		}
645 
646 		return secondaryTableSources;
647 	}

650 	private void createUniqueConstraints(AnnotationInstance tableAnnotation, String tableName) {
651 		AnnotationValue value = tableAnnotation.value( "uniqueConstraints" );
652 		if ( value == null ) {
653 			return;
654 		}
655 
656 		AnnotationInstance[] uniqueConstraints = value.asNestedArray();
657 		for ( AnnotationInstance unique : uniqueConstraints ) {
658 			String name = unique.value( "name" ) == null ? null : unique.value( "name" ).asString();
659 			String[] columnNames = unique.value( "columnNames" ).asStringArray();
660 			UniqueConstraintSourceImpl uniqueConstraintSource =
661 					new UniqueConstraintSourceImpl(
662 							name, tableName, Arrays.asList( columnNames )
663 					);
664 			constraintSources.add( uniqueConstraintSource );
665 		}
666 	}

668 	private String determineCustomLoader() {
669 		String customLoader = null;
670 		// Custom sql loader
671 		final AnnotationInstance sqlLoaderAnnotation = JandexHelper.getSingleAnnotation(
672 				getClassInfo(), HibernateDotNames.LOADER
673 		);
674 		if ( sqlLoaderAnnotation != null ) {
675 			customLoader = sqlLoaderAnnotation.value( "namedQuery" ).asString();
676 		}
677 		return customLoader;

680 	private CustomSQL createCustomSQL(AnnotationInstance customSqlAnnotation) {
681 		if ( customSqlAnnotation == null ) {
682 			return null;
683 		}
684 
685 		final String sql = customSqlAnnotation.value( "sql" ).asString();
686 		final boolean isCallable = customSqlAnnotation.value( "callable" ) != null
687 				&& customSqlAnnotation.value( "callable" ).asBoolean();
688 
689 		final ExecuteUpdateResultCheckStyle checkStyle = customSqlAnnotation.value( "check" ) == null
690 				? isCallable
691 				? ExecuteUpdateResultCheckStyle.NONE
692 				: ExecuteUpdateResultCheckStyle.COUNT
693 				: ExecuteUpdateResultCheckStyle.valueOf( customSqlAnnotation.value( "check" ).asEnum() );
694 
695 		return new CustomSQL( sql, isCallable, checkStyle );
696 	}

732 		// Proxy generation
733 		final AnnotationInstance hibernateProxyAnnotation = JandexHelper.getSingleAnnotation(
734 				getClassInfo(), HibernateDotNames.PROXY
735 		);

739 			if ( isLazy ) {

744 				else {

747 			}
748 			else {

750 			}
751 		}
752 		else {
753 			isLazy = true;
754 			proxy = getName();
755 		}

765 	private List<JpaCallbackClass> determineEntityListeners() {
766 		List<JpaCallbackClass> callbackClassList = new ArrayList<JpaCallbackClass>();
767 
768 		// Bind default JPA entity listener callbacks (unless excluded), using superclasses first (unless excluded)
769 		if ( JandexHelper.getSingleAnnotation( getClassInfo(), JPADotNames.EXCLUDE_DEFAULT_LISTENERS ) == null ) {
770 			List<AnnotationInstance> defaultEntityListenerAnnotations = getLocalBindingContext().getIndex()
771 					.getAnnotations( PseudoJpaDotNames.DEFAULT_ENTITY_LISTENERS );
772 			for ( AnnotationInstance annotation : defaultEntityListenerAnnotations ) {
773 				for ( Type callbackClass : annotation.value().asClassArray() ) {
774 					String callbackClassName = callbackClass.name().toString();
775 					try {
776 						processDefaultJpaCallbacks( callbackClassName, callbackClassList );
777 					}
778 					catch ( PersistenceException error ) {
779 						throw new PersistenceException( error.getMessage() + "default entity listener " + callbackClassName );
780 					}
781 				}
782 			}
783 		}
784 
785 		// Bind JPA entity listener callbacks, using superclasses first (unless excluded)
786 		List<AnnotationInstance> annotationList = getClassInfo().annotations().get( JPADotNames.ENTITY_LISTENERS );
787 		if ( annotationList != null ) {
788 			for ( AnnotationInstance annotation : annotationList ) {
789 				for ( Type callbackClass : annotation.value().asClassArray() ) {
790 					String callbackClassName = callbackClass.name().toString();
791 					try {
792 						processJpaCallbacks( callbackClassName, true, callbackClassList );
793 					}
794 					catch ( PersistenceException error ) {
795 						throw new PersistenceException( error.getMessage() + "entity listener " + callbackClassName );
796 					}
797 				}
798 			}
799 		}
800 
801 		// Bind JPA entity.mapped superclass callbacks, using superclasses first (unless excluded)
802 		try {
803 			processJpaCallbacks( getName(), false, callbackClassList );
804 		}
805 		catch ( PersistenceException error ) {
806 			throw new PersistenceException(
807 					error.getMessage() + "entity/mapped superclass " + getClassInfo().name().toString()
808 			);
809 		}
810 
811 		return callbackClassList;
812 	}

814 	private void processDefaultJpaCallbacks(String instanceCallbackClassName, List<JpaCallbackClass> jpaCallbackClassList) {
815 		ClassInfo callbackClassInfo = getLocalBindingContext().getClassInfo( instanceCallbackClassName );
816 
817 		// Process superclass first if available and not excluded
818 		if ( JandexHelper.getSingleAnnotation( callbackClassInfo, JPADotNames.EXCLUDE_SUPERCLASS_LISTENERS ) != null ) {
819 			DotName superName = callbackClassInfo.superName();
820 			if ( superName != null ) {
821 				processDefaultJpaCallbacks( instanceCallbackClassName, jpaCallbackClassList );
822 			}
823 		}
824 
825 		String callbackClassName = callbackClassInfo.name().toString();
826 		Map<Class<?>, String> callbacksByType = new HashMap<Class<?>, String>();
827 		createDefaultCallback(
828 				PrePersist.class, PseudoJpaDotNames.DEFAULT_PRE_PERSIST, callbackClassName, callbacksByType
829 		);
830 		createDefaultCallback(
831 				PreRemove.class, PseudoJpaDotNames.DEFAULT_PRE_REMOVE, callbackClassName, callbacksByType
832 		);
833 		createDefaultCallback(
834 				PreUpdate.class, PseudoJpaDotNames.DEFAULT_PRE_UPDATE, callbackClassName, callbacksByType
835 		);
836 		createDefaultCallback(
837 				PostLoad.class, PseudoJpaDotNames.DEFAULT_POST_LOAD, callbackClassName, callbacksByType
838 		);
839 		createDefaultCallback(
840 				PostPersist.class, PseudoJpaDotNames.DEFAULT_POST_PERSIST, callbackClassName, callbacksByType
841 		);
842 		createDefaultCallback(
843 				PostRemove.class, PseudoJpaDotNames.DEFAULT_POST_REMOVE, callbackClassName, callbacksByType
844 		);
845 		createDefaultCallback(
846 				PostUpdate.class, PseudoJpaDotNames.DEFAULT_POST_UPDATE, callbackClassName, callbacksByType
847 		);
848 		if ( !callbacksByType.isEmpty() ) {
849 			jpaCallbackClassList.add( new JpaCallbackClassImpl( instanceCallbackClassName, callbacksByType, true ) );
850 		}
851 	}

853 	private void processJpaCallbacks(String instanceCallbackClassName, boolean isListener, List<JpaCallbackClass> callbackClassList) {
854 
855 		ClassInfo callbackClassInfo = getLocalBindingContext().getClassInfo( instanceCallbackClassName );
856 
857 		// Process superclass first if available and not excluded
858 		if ( JandexHelper.getSingleAnnotation( callbackClassInfo, JPADotNames.EXCLUDE_SUPERCLASS_LISTENERS ) != null ) {
859 			DotName superName = callbackClassInfo.superName();
860 			if ( superName != null ) {
861 				processJpaCallbacks(
862 						instanceCallbackClassName,
863 						isListener,
864 						callbackClassList
865 				);
866 			}
867 		}
868 
869 		Map<Class<?>, String> callbacksByType = new HashMap<Class<?>, String>();
870 		createCallback( PrePersist.class, JPADotNames.PRE_PERSIST, callbacksByType, callbackClassInfo, isListener );
871 		createCallback( PreRemove.class, JPADotNames.PRE_REMOVE, callbacksByType, callbackClassInfo, isListener );
872 		createCallback( PreUpdate.class, JPADotNames.PRE_UPDATE, callbacksByType, callbackClassInfo, isListener );
873 		createCallback( PostLoad.class, JPADotNames.POST_LOAD, callbacksByType, callbackClassInfo, isListener );
874 		createCallback( PostPersist.class, JPADotNames.POST_PERSIST, callbacksByType, callbackClassInfo, isListener );
875 		createCallback( PostRemove.class, JPADotNames.POST_REMOVE, callbacksByType, callbackClassInfo, isListener );
876 		createCallback( PostUpdate.class, JPADotNames.POST_UPDATE, callbacksByType, callbackClassInfo, isListener );
877 		if ( !callbacksByType.isEmpty() ) {
878 			callbackClassList.add( new JpaCallbackClassImpl( instanceCallbackClassName, callbacksByType, isListener ) );
879 		}
880 	}

882 	private void createDefaultCallback(Class callbackTypeClass,
883 									   DotName callbackTypeName,
884 									   String callbackClassName,
885 									   Map<Class<?>, String> callbacksByClass) {
886 		for ( AnnotationInstance callback : getLocalBindingContext().getIndex().getAnnotations( callbackTypeName ) ) {
887 			MethodInfo methodInfo = (MethodInfo) callback.target();
888 			validateMethod( methodInfo, callbackTypeClass, callbacksByClass, true );
889 			if ( methodInfo.declaringClass().name().toString().equals( callbackClassName ) ) {
890 				if ( methodInfo.args().length != 1 ) {
891 					throw new PersistenceException(
892 							String.format(
893 									"Callback method %s must have exactly one argument defined as either Object or %s in ",
894 									methodInfo.name(),
895 									getEntityName()
896 							)
897 					);
898 				}
899 				callbacksByClass.put( callbackTypeClass, methodInfo.name() );
900 			}
901 		}
902 	}
903 
904 	private void createCallback(Class callbackTypeClass,
905 								DotName callbackTypeName,
906 								Map<Class<?>, String> callbacksByClass,
907 								ClassInfo callbackClassInfo,
908 								boolean isListener) {
909 		Map<DotName, List<AnnotationInstance>> annotations = callbackClassInfo.annotations();
910 		List<AnnotationInstance> annotationInstances = annotations.get( callbackTypeName );
911 		if ( annotationInstances == null ) {
912 			return;
913 		}
914 		for ( AnnotationInstance callbackAnnotation : annotationInstances ) {
915 			MethodInfo methodInfo = (MethodInfo) callbackAnnotation.target();
916 			validateMethod( methodInfo, callbackTypeClass, callbacksByClass, isListener );
917 			callbacksByClass.put( callbackTypeClass, methodInfo.name() );
918 		}
919 	}
920 
921 	private void validateMethod(MethodInfo methodInfo,
922 								Class callbackTypeClass,
923 								Map<Class<?>, String> callbacksByClass,
924 								boolean isListener) {
925 		if ( methodInfo.returnType().kind() != Kind.VOID ) {
926 			throw new PersistenceException( "Callback method " + methodInfo.name() + " must have a void return type in " );
927 		}
928 		if ( Modifier.isStatic( methodInfo.flags() ) || Modifier.isFinal( methodInfo.flags() ) ) {
929 			throw new PersistenceException( "Callback method " + methodInfo.name() + " must not be static or final in " );
930 		}
931 		Type[] argTypes = methodInfo.args();
932 		if ( isListener ) {
933 			if ( argTypes.length != 1 ) {
934 				throw new PersistenceException( "Callback method " + methodInfo.name() + " must have exactly one argument in " );
935 			}
936 			String argTypeName = argTypes[0].name().toString();
937 			if ( !argTypeName.equals( Object.class.getName() ) && !argTypeName.equals( getName() ) ) {
938 				throw new PersistenceException(
939 						"The argument for callback method " + methodInfo.name() +
940 								" must be defined as either Object or " + getEntityName() + " in "
941 				);
942 			}
943 		}
944 		else if ( argTypes.length != 0 ) {
945 			throw new PersistenceException( "Callback method " + methodInfo.name() + " must have no arguments in " );
946 		}
947 		if ( callbacksByClass.containsKey( callbackTypeClass ) ) {
948 			throw new PersistenceException(
949 					"Only one method may be annotated as a " + callbackTypeClass.getSimpleName() +
950 							" callback method in "
951 			);
952 		}
953 	}

955 	// Process JPA callbacks, in superclass-first order (unless superclasses are excluded), using default listeners first
956 	// (unless default listeners are excluded), then entity listeners, and finally the entity/mapped superclass itself
957 	private class JpaCallbackClassImpl implements JpaCallbackClass {
958 
959 		private final Map<Class<?>, String> callbacksByType;
960 		private final String name;
961 		private final boolean isListener;
962 
963 		private JpaCallbackClassImpl(String name,
964 									 Map<Class<?>, String> callbacksByType,
965 									 boolean isListener) {
966 			this.name = name;
967 			this.callbacksByType = callbacksByType;
968 			this.isListener = isListener;
969 		}
970 
971 		@Override
972 		public String getCallbackMethod(Class<?> callbackType) {
973 			return callbacksByType.get( callbackType );
974 		}
975 
976 		@Override
977 		public String getName() {
978 			return name;
979 		}
980 
981 		@Override
982 		public boolean isListener() {
983 			return isListener;
984 		}
985 	}
