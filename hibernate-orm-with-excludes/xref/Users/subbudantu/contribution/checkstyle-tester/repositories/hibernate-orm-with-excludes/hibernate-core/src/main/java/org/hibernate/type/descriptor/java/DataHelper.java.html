42  /**
43   * A help for dealing with BLOB and CLOB data
44   *
45   * @author Steve Ebersole
46   */
47  public class DataHelper {
48  
49  	/** The size of the buffer we will use to deserialize larger streams */
50  	private static final int BUFFER_SIZE = 1024 * 4;
51  
52  	private static final CoreMessageLogger LOG = Logger.getMessageLogger( CoreMessageLogger.class, DataHelper.class.getName() );
53  
54  	public static boolean isNClob(final Class type) {
55  		return java.sql.NClob.class.isAssignableFrom( type );
56  	}
57  
58  	/**
59  	 * Extract the contents of the given reader/stream as a string.
60  	 * The reader will be closed.
61  	 *
62  	 * @param reader The reader for the content
63  	 *
64  	 * @return The content as string
65  	 */
66  	public static String extractString(Reader reader) {
67  		return extractString( reader, BUFFER_SIZE );
68  	}
69  
70  	/**
71  	 * Extract the contents of the given reader/stream as a string.
72  	 * The reader will be closed.
73  	 *
74  	 * @param reader The reader for the content
75  	 * @param lengthHint if the length is known in advance the implementation can be slightly more efficient
76  	 *
77  	 * @return The content as string
78  	 */
79  	public static String extractString(Reader reader, int lengthHint) {
80  		// read the Reader contents into a buffer and return the complete string
81  		final int bufferSize = getSuggestedBufferSize( lengthHint );
82  		final StringBuilder stringBuilder = new StringBuilder( bufferSize );
83  		try {
84  			char[] buffer = new char[bufferSize];
85  			while (true) {
86  				int amountRead = reader.read( buffer, 0, bufferSize );
87  				if ( amountRead == -1 ) {
88  					break;
89  				}
90  				stringBuilder.append( buffer, 0, amountRead );
91  			}
92  		}
93  		catch ( IOException ioe ) {
94  			throw new HibernateException( "IOException occurred reading text", ioe );
95  		}
96  		finally {
97  			try {
98  				reader.close();
99  			}
100 			catch (IOException e) {
101 				LOG.unableToCloseStream( e );
102 			}
103 		}
104 		return stringBuilder.toString();
105 	}
106 
107 	/**
108 	 * Extracts a portion of the contents of the given reader/stream as a string.
109 	 *
110 	 * @param characterStream The reader for the content
111 	 * @param start The start position/offset (0-based, per general stream/reader contracts).
112 	 * @param length The amount to extract
113 	 *
114 	 * @return The content as string
115 	 */
116 	private static String extractString(Reader characterStream, long start, int length) {
117 		if ( length == 0 ) {
118 			return "";
119 		}
120 		StringBuilder stringBuilder = new StringBuilder( length );
121 		try {
122 			long skipped = characterStream.skip( start );
123 			if ( skipped != start ) {
124 				throw new HibernateException( "Unable to skip needed bytes" );
125 			}
126 			final int bufferSize = getSuggestedBufferSize( length );
127 			char[] buffer = new char[bufferSize];
128 			int charsRead = 0;
129 			while ( true ) {
130 				int amountRead = characterStream.read( buffer, 0, bufferSize );
131 				if ( amountRead == -1 ) {
132 					break;
133 				}
134 				stringBuilder.append( buffer, 0, amountRead );
135 				if ( amountRead < bufferSize ) {
136 					// we have read up to the end of stream
137 					break;
138 				}
139 				charsRead += amountRead;
140 				if ( charsRead >= length ) {
141 					break;
142 				}
143 			}
144 		}
145 		catch ( IOException ioe ) {
146 			throw new HibernateException( "IOException occurred reading a binary value", ioe );
147 		}
148 		return stringBuilder.toString();
149 	}
150 
151 	/**
152 	 * Extract a portion of a reader, wrapping the portion in a new reader.
153 	 *
154 	 * @param characterStream The reader for the content
155 	 * @param start The start position/offset (0-based, per general stream/reader contracts).
156 	 * @param length The amount to extract
157 	 *
158 	 * @return The content portion as a reader
159 	 */
160 	public static Object subStream(Reader characterStream, long start, int length) {
161 		return new StringReader( extractString( characterStream, start, length ) );
162 	}

164 	/**
165 	 * Extract by bytes from the given stream.
166 	 *
167 	 * @param inputStream The stream of bytes.
168 	 *
169 	 * @return The contents as a {@code byte[]}
170 	 */
171 	public static byte[] extractBytes(InputStream inputStream) {
172 		if ( BinaryStream.class.isInstance( inputStream ) ) {
173 			return ( (BinaryStream ) inputStream ).getBytes();
174 		}
175 
176 		// read the stream contents into a buffer and return the complete byte[]
177 		ByteArrayOutputStream outputStream = new ByteArrayOutputStream(BUFFER_SIZE);
178 		try {
179 			byte[] buffer = new byte[BUFFER_SIZE];
180 			while (true) {
181 				int amountRead = inputStream.read( buffer );
182 				if ( amountRead == -1 ) {
183 					break;
184 				}
185 				outputStream.write( buffer, 0, amountRead );
186 			}
187 		}
188 		catch ( IOException ioe ) {
189 			throw new HibernateException( "IOException occurred reading a binary value", ioe );
190 		}
191 		finally {
192 			try {
193 				inputStream.close();
194 			}
195 			catch ( IOException e ) {
196 				LOG.unableToCloseInputStream( e );
197 			}
198 			try {
199 				outputStream.close();
200 			}
201 			catch ( IOException e ) {
202 				LOG.unableToCloseOutputStream( e );
203 			}
204 		}
205 		return outputStream.toByteArray();
206 	}
207 
208 	/**
209 	 * Extract a portion of the bytes from the given stream.
210 	 *
211 	 * @param inputStream The stream of bytes.
212 	 * @param start The start position/offset (0-based, per general stream/reader contracts).
213 	 * @param length The amount to extract
214 	 *
215 	 * @return The extracted bytes
216 	 */
217 	public static byte[] extractBytes(InputStream inputStream, long start, int length) {
218 		if ( BinaryStream.class.isInstance( inputStream ) && Integer.MAX_VALUE > start ) {
219 			byte[] data = ( (BinaryStream ) inputStream ).getBytes();
220 			int size = Math.min( length, data.length );
221 			byte[] result = new byte[size];
222 			System.arraycopy( data, (int) start, result, 0, size );
223 			return result;
224 		}
225 
226 		ByteArrayOutputStream outputStream = new ByteArrayOutputStream( length );
227 		try {
228 			long skipped = inputStream.skip( start );
229 			if ( skipped != start ) {
230 				throw new HibernateException( "Unable to skip needed bytes" );
231 			}
232 			byte[] buffer = new byte[BUFFER_SIZE];
233 			int bytesRead = 0;
234 			while ( true ) {
235 				int amountRead = inputStream.read( buffer );
236 				if ( amountRead == -1 ) {
237 					break;
238 				}
239 				outputStream.write( buffer, 0, amountRead );
240 				if ( amountRead < buffer.length ) {
241 					// we have read up to the end of stream
242 					break;
243 				}
244 				bytesRead += amountRead;
245 				if ( bytesRead >= length ) {
246 					break;
247 				}
248 			}
249 		}
250 		catch ( IOException ioe ) {
251 			throw new HibernateException( "IOException occurred reading a binary value", ioe );
252 		}
253 		return outputStream.toByteArray();
254 	}
255 
256 	/**
257 	 * Extract a portion of the bytes from the given stream., wrapping them in a new stream.
258 	 *
259 	 * @param inputStream The stream of bytes.
260 	 * @param start The start position/offset (0-based, per general stream/reader contracts).
261 	 * @param length The amount to extract
262 	 *
263 	 * @return The extracted bytes as a stream
264 	 */
265 	public static InputStream subStream(InputStream inputStream, long start, int length) {
266 		return new BinaryStreamImpl( extractBytes( inputStream, start, length ) );
267 	}

269 	/**
270 	 * Extract the contents of the given Clob as a string.
271 	 *
272 	 * @param value The clob to to be extracted from
273 	 *
274 	 * @return The content as string
275 	 */
276 	public static String extractString(final Clob value) {
277 		try {
278 			final Reader characterStream = value.getCharacterStream();
279 			final long length = determineLengthForBufferSizing( value );
280 			return length > Integer.MAX_VALUE
281 					? extractString( characterStream, Integer.MAX_VALUE )
282 					: extractString( characterStream, (int) length );
283 		}
284 		catch ( SQLException e ) {
285 			throw new HibernateException( "Unable to access lob stream", e );
286 		}
287 	}
288 
289 	/**
290 	 * Determine a buffer size for reading the underlying character stream.
291 	 *
292 	 * @param value The Clob value
293 	 *
294 	 * @return The appropriate buffer size ({@link java.sql.Clob#length()} by default.
295 	 *
296 	 * @throws SQLException
297 	 */
298 	private static long determineLengthForBufferSizing(Clob value) throws SQLException {
299 		try {
300 			return value.length();
301 		}
302 		catch ( SQLFeatureNotSupportedException e ) {
303 			return BUFFER_SIZE;
304 		}
305 	}
306 
307 	/**
308 	 * Make sure we allocate a buffer sized not bigger than 2048,
309 	 * not higher than what is actually needed, and at least one.
310 	 * 
311 	 * @param lengthHint the expected size of the full value
312 	 * @return the buffer size
313 	 */
314 	private static int getSuggestedBufferSize(final int lengthHint) {
315 		return Math.max( 1, Math.min( lengthHint , BUFFER_SIZE ) );
316 	}
