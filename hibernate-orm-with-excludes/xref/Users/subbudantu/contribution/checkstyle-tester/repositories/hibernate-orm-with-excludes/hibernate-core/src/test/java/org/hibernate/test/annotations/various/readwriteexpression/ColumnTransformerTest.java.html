24  package org.hibernate.test.annotations.various.readwriteexpression;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.Transaction;
30  import org.hibernate.criterion.Restrictions;
31  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
32  
33  import static org.junit.Assert.assertEquals;
34  
35  /**
36   * @author Emmanuel Bernard
37   */
38  public class ColumnTransformerTest extends BaseCoreFunctionalTestCase {
39  	@Test
40  	public void testCustomColumnReadAndWrite() throws Exception{
41  		Session s = openSession();
42  		Transaction t = s.beginTransaction();
43  		final double HEIGHT_INCHES = 73;
44  		final double HEIGHT_CENTIMETERS = HEIGHT_INCHES * 2.54d;
45  
46  		Staff staff = new Staff(HEIGHT_INCHES, HEIGHT_INCHES, HEIGHT_INCHES*2, 1);
47  		s.persist( staff );
48  		s.flush();
49  
50  		// Test value conversion during insert
51  		// Value returned by Oracle native query is a Types.NUMERIC, which is mapped to a BigDecimalType;
52  		// Cast returned value to Number then call Number.doubleValue() so it works on all dialects.
53  		double heightViaSql =
54  				( (Number)s.createSQLQuery("select size_in_cm from t_staff where t_staff.id=1").uniqueResult() )
55  						.doubleValue();
56  		assertEquals(HEIGHT_CENTIMETERS, heightViaSql, 0.01d);
57  
58  		heightViaSql =
59  				( (Number)s.createSQLQuery("select radiusS from t_staff where t_staff.id=1").uniqueResult() )
60  						.doubleValue();
61  		assertEquals(HEIGHT_CENTIMETERS, heightViaSql, 0.01d);
62  
63  		heightViaSql =
64  				( (Number)s.createSQLQuery("select diamet from t_staff where t_staff.id=1").uniqueResult() )
65  						.doubleValue();
66  		assertEquals(HEIGHT_CENTIMETERS*2, heightViaSql, 0.01d);
67  
68  		// Test projection
69  		Double heightViaHql = (Double)s.createQuery("select s.sizeInInches from Staff s where s.id = 1").uniqueResult();
70  		assertEquals(HEIGHT_INCHES, heightViaHql, 0.01d);
71  
72  		// Test restriction and entity load via criteria
73  		staff = (Staff)s.createCriteria(Staff.class)
74  			.add( Restrictions.between("sizeInInches", HEIGHT_INCHES - 0.01d, HEIGHT_INCHES + 0.01d))
75  			.uniqueResult();
76  		assertEquals(HEIGHT_INCHES, staff.getSizeInInches(), 0.01d);
77  		
78  		// Test predicate and entity load via HQL
79  		staff = (Staff)s.createQuery("from Staff s where s.sizeInInches between ? and ?")
80  			.setDouble(0, HEIGHT_INCHES - 0.01d)
81  			.setDouble(1, HEIGHT_INCHES + 0.01d)
82  			.uniqueResult();
83  		assertEquals(HEIGHT_INCHES, staff.getSizeInInches(), 0.01d);
84  
85  		// Test update
86  		staff.setSizeInInches(1);
87  		s.flush();
88  		heightViaSql =
89  				( (Number)s.createSQLQuery("select size_in_cm from t_staff where t_staff.id=1").uniqueResult() )
90  						.doubleValue();
91  		assertEquals(2.54d, heightViaSql, 0.01d);
92  		s.delete(staff);
93  		t.commit();
94  		s.close();
95  	}
96  
97  	@Override
98  	protected Class<?>[] getAnnotatedClasses() {
99  		return new Class<?>[] {
100 				Staff.class
101 		};
102 	}
