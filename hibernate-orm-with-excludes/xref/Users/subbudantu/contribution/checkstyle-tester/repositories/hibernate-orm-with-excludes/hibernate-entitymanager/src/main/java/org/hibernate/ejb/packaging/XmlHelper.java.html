24  import java.util.ArrayList;

27  import org.w3c.dom.Element;

31  /**
32   * A utility class to cover up the rough bits of xml parsing
33   *
34   * @author <a href="mailto:chris@kimptoc.net">Chris Kimpton</a>
35   */
36  public final class XmlHelper {
37  	private XmlHelper() {
38  	}
39  
40  	/**
41  	 * Returns an iterator over the children of the given element with
42  	 * the given tag name.
43  	 *
44  	 * @param element The parent element
45  	 * @param tagName The name of the desired child
46  	 * @return An interator of children or null if element is null.
47  	 */
48  	public static Iterator getChildrenByTagName(
49  			Element element,
50  			String tagName) {
51  		if ( element == null ) return null;
52  		// getElementsByTagName gives the corresponding elements in the whole
53  		// descendance. We want only children
54  
55  		NodeList children = element.getChildNodes();
56  		ArrayList goodChildren = new ArrayList();
57  		for ( int i = 0; i < children.getLength() ; i++ ) {
58  			Node currentChild = children.item( i );
59  			if ( currentChild.getNodeType() == Node.ELEMENT_NODE &&
60  					( (Element) currentChild ).getTagName().equals( tagName ) ) {
61  				goodChildren.add( (Element) currentChild );
62  			}
63  		}
64  		return goodChildren.iterator();
65  	}
66  
67  	/**
68  	 * Gets the child of the specified element having the specified unique
69  	 * name.  If there are more than one children elements with the same name
70  	 * and exception is thrown.
71  	 *
72  	 * @param element The parent element
73  	 * @param tagName The name of the desired child
74  	 * @return The named child.
75  	 * @throws Exception Child was not found or was not unique.
76  	 */
77  	public static Element getUniqueChild(Element element, String tagName)
78  			throws Exception {
79  		Iterator goodChildren = getChildrenByTagName( element, tagName );
80  
81  		if ( goodChildren != null && goodChildren.hasNext() ) {
82  			Element child = (Element) goodChildren.next();
83  			if ( goodChildren.hasNext() ) {
84  				throw new Exception
85  						( "expected only one " + tagName + " tag" );
86  			}
87  			return child;
88  		}
89  		else {
90  			throw new Exception
91  					( "expected one " + tagName + " tag" );
92  		}
93  	}
94  
95  	/**
96  	 * Gets the child of the specified element having the
97  	 * specified name. If the child with this name doesn't exist
98  	 * then null is returned instead.
99  	 *
100 	 * @param element the parent element
101 	 * @param tagName the name of the desired child
102 	 * @return either the named child or null
103 	 */
104 	public static Element getOptionalChild(Element element, String tagName)
105 			throws Exception {
106 		return getOptionalChild( element, tagName, null );
107 	}
108 
109 	/**
110 	 * Gets the child of the specified element having the
111 	 * specified name. If the child with this name doesn't exist
112 	 * then the supplied default element is returned instead.
113 	 *
114 	 * @param element		the parent element
115 	 * @param tagName		the name of the desired child
116 	 * @param defaultElement the element to return if the child
117 	 *                       doesn't exist
118 	 * @return either the named child or the supplied default
119 	 */
120 	public static Element getOptionalChild(
121 			Element element,
122 			String tagName,
123 			Element defaultElement) throws Exception {
124 		Iterator goodChildren = getChildrenByTagName( element, tagName );
125 
126 		if ( goodChildren != null && goodChildren.hasNext() ) {
127 			Element child = (Element) goodChildren.next();
128 			if ( goodChildren.hasNext() ) {
129 				throw new Exception
130 						( "expected only one " + tagName + " tag" );
131 			}
132 			return child;
133 		}
134 		else {
135 			return defaultElement;
136 		}
137 	}
138 
139 	/**
140 	 * Get the content of the given element.
141 	 *
142 	 * @param element The element to get the content for.
143 	 * @return The content of the element or null.
144 	 */
145 	public static String getElementContent(final Element element)
146 			throws Exception {
147 		return getElementContent( element, null );
148 	}
149 
150 	/**
151 	 * Get the content of the given element.
152 	 *
153 	 * @param element	The element to get the content for.
154 	 * @param defaultStr The default to return when there is no content.
155 	 * @return The content of the element or the default.
156 	 */
157 	public static String getElementContent(Element element, String defaultStr)
158 			throws Exception {
159 		if ( element == null ) {
160 			return defaultStr;
161 		}
162 
163 		NodeList children = element.getChildNodes();
164 		StringBuilder result = new StringBuilder("");
165 		for ( int i = 0; i < children.getLength() ; i++ ) {
166 			if ( children.item( i ).getNodeType() == Node.TEXT_NODE ||
167 					children.item( i ).getNodeType() == Node.CDATA_SECTION_NODE ) {
168 				result.append( children.item( i ).getNodeValue() );
169 			}
170 //			else if ( children.item( i ).getNodeType() == Node.COMMENT_NODE ) {
171 //				// Ignore comment nodes
172 //			}
173 		}
174 		return result.toString().trim();
175 	}
176 
177 	/**
178 	 * Macro to get the content of a unique child element.
179 	 *
180 	 * @param element The parent element.
181 	 * @param tagName The name of the desired child.
182 	 * @return The element content or null.
183 	 */
184 	public static String getUniqueChildContent(
185 			Element element,
186 			String tagName) throws Exception {
187 		return getElementContent( getUniqueChild( element, tagName ) );
188 	}
189 
190 	/**
191 	 * Macro to get the content of an optional child element.
192 	 *
193 	 * @param element The parent element.
194 	 * @param tagName The name of the desired child.
195 	 * @return The element content or null.
196 	 */
197 	public static String getOptionalChildContent(
198 			Element element,
199 			String tagName) throws Exception {
200 		return getElementContent( getOptionalChild( element, tagName ) );
201 	}
202 
203 	public static boolean getOptionalChildBooleanContent(Element element, String name) throws Exception {
204 		Element child = getOptionalChild( element, name );
205 		if ( child != null ) {
206 			String value = getElementContent( child ).toLowerCase();
207 			return value.equals( "true" ) || value.equals( "yes" );
208 		}
209 
210 		return false;
211 	}
