42  /**
43   * @author Steve Ebersole

45  public class AggregationResultTest extends AbstractMetamodelSpecificTest {
46  	private CriteriaBuilder builder;
47  
48  	@Before
49  	public void createTestData() {
50  		builder = entityManagerFactory().getCriteriaBuilder();
51  
52  		EntityManager em = getOrCreateEntityManager();
53  		em.getTransaction().begin();
54  		Product product = new Product();
55  		product.setId( "product1" );
56  		product.setPrice( 1.23d );
57  		product.setQuantity( 1000 );
58  		product.setPartNumber( ( (long) Integer.MAX_VALUE ) + 1 );
59  		product.setRating( 1.999f );
60  		product.setSomeBigInteger( BigInteger.valueOf( 987654321 ) );
61  		product.setSomeBigDecimal( BigDecimal.valueOf( 987654.32 ) );
62  		em.persist( product );
63  		em.getTransaction().commit();
64  		em.close();
65  	}
66  
67  	@After
68  	public void cleanUpTestData() throws Exception {
69  		EntityManager em = getOrCreateEntityManager();
70  		em.getTransaction().begin();
71  		em.createQuery( "delete Product" ).executeUpdate();
72  		em.getTransaction().commit();
73  		em.close();
74  	}
75  
76  	/**
77  	 * Sum of Longs should return a Long
78  	 */
79  	@Test
80  	public void testSumOfLongs() {
81  		EntityManager em = getOrCreateEntityManager();
82  		em.getTransaction().begin();
83  		CriteriaQuery<Long> criteria = builder.createQuery( Long.class );
84  		Root<Product> productRoot = criteria.from( Product.class );
85  		criteria.select( builder.sum( productRoot.get( Product_.partNumber ) ) );
86  		Object sumResult = em.createQuery( criteria ).getSingleResult();
87  		assertReturnType( Long.class, sumResult );
88  		em.getTransaction().commit();
89  		em.close();
90  	}
91

172 	private void assertReturnType(Class expectedType, Object value) {
173 		if ( value != null && ! expectedType.isInstance( value ) ) {
174 			throw new AssertionFailedError(
175 					"Result value was not of expected type: expected [" + expectedType.getName()
176 							+ "] but found [" + value.getClass().getName() + "]"
177 			);
178 		}
