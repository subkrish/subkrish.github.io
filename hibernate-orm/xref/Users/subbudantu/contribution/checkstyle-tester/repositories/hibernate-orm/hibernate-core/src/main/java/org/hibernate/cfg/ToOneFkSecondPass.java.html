24  package org.hibernate.cfg;
25  
26  import java.util.Iterator;
27  
28  import org.hibernate.AnnotationException;
29  import org.hibernate.AssertionFailure;
30  import org.hibernate.MappingException;
31  import org.hibernate.cfg.annotations.TableBinder;
32  import org.hibernate.internal.util.StringHelper;
33  import org.hibernate.mapping.Component;
34  import org.hibernate.mapping.KeyValue;
35  import org.hibernate.mapping.ManyToOne;
36  import org.hibernate.mapping.OneToOne;
37  import org.hibernate.mapping.PersistentClass;
38  import org.hibernate.mapping.Property;
39  import org.hibernate.mapping.ToOne;
40  
41  /**
42   * Enable a proper set of the FK columns in respect with the id column order
43   * Allow the correct implementation of the default EJB3 values which needs both
44   * sides of the association to be resolved
45   *
46   * @author Emmanuel Bernard
47   */
48  public class ToOneFkSecondPass extends FkSecondPass {
49  	private boolean unique;
50  	private Mappings mappings;
51  	private String path;
52  	private String entityClassName;
53  
54  	public ToOneFkSecondPass(
55  			ToOne value,
56  			Ejb3JoinColumn[] columns,
57  			boolean unique,
58  			String entityClassName,
59  			String path,
60  			Mappings mappings) {
61  		super( value, columns );
62  		this.mappings = mappings;
63  		this.unique = unique;
64  		this.entityClassName = entityClassName;
65  		this.path = entityClassName != null ? path.substring( entityClassName.length() + 1 ) : path;
66  	}
67  
68  	@Override
69      public String getReferencedEntityName() {
70  		return ( (ToOne) value ).getReferencedEntityName();
71  	}
72  
73  	@Override
74      public boolean isInPrimaryKey() {
75  		if ( entityClassName == null ) return false;
76  		final PersistentClass persistentClass = mappings.getClass( entityClassName );
77  		Property property = persistentClass.getIdentifierProperty();
78  		if ( path == null ) {
79  			return false;
80  		}
81  		else if ( property != null) {
82  			//try explicit identifier property
83  			return path.startsWith( property.getName() + "." );
84  		}
85  		else {
86  			//try the embedded property
87  			//embedded property starts their path with 'id.' See PropertyPreloadedData( ) use when idClass != null in AnnotationSourceProcessor
88  			if ( path.startsWith( "id." ) ) {
89  				KeyValue valueIdentifier = persistentClass.getIdentifier();
90  				String localPath = path.substring( 3 );
91  				if ( valueIdentifier instanceof Component ) {
92  					Iterator it = ( (Component) valueIdentifier ).getPropertyIterator();
93  					while ( it.hasNext() ) {
94  						Property idProperty = (Property) it.next();
95  						if ( localPath.startsWith( idProperty.getName() ) ) return true;
96  					}
97  
98  				}
99  			}
100 		}
101 		return false;
102 	}
103 
104 	public void doSecondPass(java.util.Map persistentClasses) throws MappingException {
105 		if ( value instanceof ManyToOne ) {
106 			ManyToOne manyToOne = (ManyToOne) value;
107 			PersistentClass ref = (PersistentClass) persistentClasses.get( manyToOne.getReferencedEntityName() );
108 			if ( ref == null ) {
109 				throw new AnnotationException(
110 						"@OneToOne or @ManyToOne on "
111 								+ StringHelper.qualify( entityClassName, path )
112 								+ " references an unknown entity: "
113 								+ manyToOne.getReferencedEntityName()
114 				);
115 			}
116 			BinderHelper.createSyntheticPropertyReference( columns, ref, null, manyToOne, false, mappings );
117 			TableBinder.bindFk( ref, null, columns, manyToOne, unique, mappings );
118 			/*
119 			 * HbmMetadataSourceProcessorImpl does this only when property-ref != null, but IMO, it makes sense event if it is null
120 			 */
121 			if ( !manyToOne.isIgnoreNotFound() ) manyToOne.createPropertyRefConstraints( persistentClasses );
122 		}
123 		else if ( value instanceof OneToOne ) {
124 			( (OneToOne) value ).createForeignKey();
125 		}
126 		else {
127 			throw new AssertionFailure( "FkSecondPass for a wrong value type: " + value.getClass().getName() );
128 		}
129 	}
130 }
