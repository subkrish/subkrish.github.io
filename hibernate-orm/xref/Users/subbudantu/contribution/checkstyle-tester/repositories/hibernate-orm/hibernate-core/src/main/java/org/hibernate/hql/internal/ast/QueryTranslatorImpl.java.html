158 	/**
159 	 * Performs both filter and non-filter compiling.
160 	 *
161 	 * @param replacements   Defined query substitutions.
162 	 * @param shallow        Does this represent a shallow (scalar or entity-id) select?
163 	 * @param collectionRole the role name of the collection used as the basis for the filter, NULL if this
164 	 *                       is not a filter.
165 	 */
166 	private synchronized void doCompile(Map replacements, boolean shallow, String collectionRole) {
167 		// If the query is already compiled, skip the compilation.
168 		if ( compiled ) {
169 			LOG.debug( "compile() : The query is already compiled, skipping..." );
170 			return;
171 		}
172 
173 		// Remember the parameters for the compilation.
174 		this.tokenReplacements = replacements;
175 		if ( tokenReplacements == null ) {
176 			tokenReplacements = new HashMap();
177 		}
178 		this.shallowQuery = shallow;
179 
180 		try {
181 			// PHASE 1 : Parse the HQL into an AST.
182 			HqlParser parser = parse( true );
183 
184 			// PHASE 2 : Analyze the HQL AST, and produce an SQL AST.
185 			HqlSqlWalker w = analyze( parser, collectionRole );
186 
187 			sqlAst = ( Statement ) w.getAST();
188 
189 			// at some point the generate phase needs to be moved out of here,
190 			// because a single object-level DML might spawn multiple SQL DML
191 			// command executions.
192 			//
193 			// Possible to just move the sql generation for dml stuff, but for
194 			// consistency-sake probably best to just move responsiblity for
195 			// the generation phase completely into the delegates
196 			// (QueryLoader/StatementExecutor) themselves.  Also, not sure why
197 			// QueryLoader currently even has a dependency on this at all; does
198 			// it need it?  Ideally like to see the walker itself given to the delegates directly...
199 
200 			if ( sqlAst.needsExecutor() ) {
201 				statementExecutor = buildAppropriateStatementExecutor( w );
202 			}
203 			else {
204 				// PHASE 3 : Generate the SQL.
205 				generate( ( QueryNode ) sqlAst );
206 				queryLoader = new QueryLoader( this, factory, w.getSelectClause() );
207 			}
208 
209 			compiled = true;
210 		}
211 		catch ( QueryException qe ) {
212 			qe.setQueryString( hql );
213 			throw qe;
214 		}
215 		catch ( RecognitionException e ) {
216 			// we do not actually propagate ANTLRExceptions as a cause, so
217 			// log it here for diagnostic purposes
218 			LOG.trace( "Converted antlr.RecognitionException", e );
219 			throw QuerySyntaxException.convert( e, hql );
220 		}
221 		catch ( ANTLRException e ) {
222 			// we do not actually propagate ANTLRExceptions as a cause, so
223 			// log it here for diagnostic purposes
224 			LOG.trace( "Converted antlr.ANTLRException", e );
225 			throw new QueryException( e.getMessage(), hql );
226 		}
227 
228 		this.enabledFilters = null; //only needed during compilation phase...
229 	}
230 
231 	private void generate(AST sqlAst) throws QueryException, RecognitionException {
232 		if ( sql == null ) {
233 			SqlGenerator gen = new SqlGenerator(factory);
234 			gen.statement( sqlAst );
235 			sql = gen.getSQL();
236 			if ( LOG.isDebugEnabled() ) {
237 				LOG.debugf( "HQL: %s", hql );
238 				LOG.debugf( "SQL: %s", sql );
239 			}
240 			gen.getParseErrorHandler().throwQueryException();
241 			collectedParameterSpecifications = gen.getCollectedParameters();
242 		}
243 	}

245 	private HqlSqlWalker analyze(HqlParser parser, String collectionRole) throws QueryException, RecognitionException {
246 		HqlSqlWalker w = new HqlSqlWalker( this, factory, parser, tokenReplacements, collectionRole );
247 		AST hqlAst = parser.getAST();
248 
249 		// Transform the tree.
250 		w.statement( hqlAst );
251 
252 		if ( LOG.isDebugEnabled() ) {
253 			ASTPrinter printer = new ASTPrinter( SqlTokenTypes.class );
254 			LOG.debug( printer.showAsString( w.getAST(), "--- SQL AST ---" ) );
255 		}
256 
257 		w.getParseErrorHandler().throwQueryException();
258 
259 		return w;
260 	}

262 	private HqlParser parse(boolean filter) throws TokenStreamException, RecognitionException {
263 		// Parse the query string into an HQL AST.
264 		HqlParser parser = HqlParser.getInstance( hql );
265 		parser.setFilter( filter );
266 
267 		LOG.debugf( "parse() - HQL: %s", hql );
268 		parser.statement();
269 
270 		AST hqlAst = parser.getAST();
271 
272 		JavaConstantConverter converter = new JavaConstantConverter();
273 		NodeTraverser walker = new NodeTraverser( converter );
274 		walker.traverseDepthFirst( hqlAst );
275 
276 		showHqlAst( hqlAst );
277 
278 		parser.getParseErrorHandler().throwQueryException();
279 		return parser;
280 	}

337 	public List list(SessionImplementor session, QueryParameters queryParameters)
338 			throws HibernateException {
339 		// Delegate to the QueryLoader...
340 		errorIfDML();
341 		QueryNode query = ( QueryNode ) sqlAst;
342 		boolean hasLimit = queryParameters.getRowSelection() != null && queryParameters.getRowSelection().definesLimits();
343 		boolean needsDistincting = ( query.getSelectClause().isDistinct() || hasLimit ) && containsCollectionFetches();
344 
345 		QueryParameters queryParametersToUse;
346 		if ( hasLimit && containsCollectionFetches() ) {
347 			LOG.firstOrMaxResultsSpecifiedWithCollectionFetch();
348 			RowSelection selection = new RowSelection();
349 			selection.setFetchSize( queryParameters.getRowSelection().getFetchSize() );
350 			selection.setTimeout( queryParameters.getRowSelection().getTimeout() );
351 			queryParametersToUse = queryParameters.createCopyUsing( selection );
352 		}
353 		else {
354 			queryParametersToUse = queryParameters;
355 		}
356 
357 		List results = queryLoader.list( session, queryParametersToUse );
358 
359 		if ( needsDistincting ) {
360 			int includedCount = -1;
361 			// NOTE : firstRow is zero-based
362 			int first = !hasLimit || queryParameters.getRowSelection().getFirstRow() == null
363 						? 0
364 						: queryParameters.getRowSelection().getFirstRow().intValue();
365 			int max = !hasLimit || queryParameters.getRowSelection().getMaxRows() == null
366 						? -1
367 						: queryParameters.getRowSelection().getMaxRows().intValue();
368 			int size = results.size();
369 			List tmp = new ArrayList();
370 			IdentitySet distinction = new IdentitySet();
371 			for ( int i = 0; i < size; i++ ) {
372 				final Object result = results.get( i );
373 				if ( !distinction.add( result ) ) {
374 					continue;
375 				}
376 				includedCount++;
377 				if ( includedCount < first ) {
378 					continue;
379 				}
380 				tmp.add( result );
381 				// NOTE : ( max - 1 ) because first is zero-based while max is not...
382 				if ( max >= 0 && ( includedCount - first ) >= ( max - 1 ) ) {
383 					break;
384 				}
385 			}
386 			results = tmp;
387 		}
388 
389 		return results;
390 	}

467 	public void validateScrollability() throws HibernateException {
468 		// Impl Note: allows multiple collection fetches as long as the
469 		// entire fecthed graph still "points back" to a single
470 		// root entity for return
471 
472 		errorIfDML();
473 
474 		QueryNode query = ( QueryNode ) sqlAst;
475 
476 		// If there are no collection fetches, then no further checks are needed
477 		List collectionFetches = query.getFromClause().getCollectionFetches();
478 		if ( collectionFetches.isEmpty() ) {
479 			return;
480 		}
481 
482 		// A shallow query is ok (although technically there should be no fetching here...)
483 		if ( isShallowQuery() ) {
484 			return;
485 		}
486 
487 		// Otherwise, we have a non-scalar select with defined collection fetch(es).
488 		// Make sure that there is only a single root entity in the return (no tuples)
489 		if ( getReturnTypes().length > 1 ) {
490 			throw new HibernateException( "cannot scroll with collection fetches and returned tuples" );
491 		}
492 
493 		FromElement owner = null;
494 		Iterator itr = query.getSelectClause().getFromElementsForLoad().iterator();
495 		while ( itr.hasNext() ) {
496 			// should be the first, but just to be safe...
497 			final FromElement fromElement = ( FromElement ) itr.next();
498 			if ( fromElement.getOrigin() == null ) {
499 				owner = fromElement;
500 				break;
501 			}
502 		}
503 
504 		if ( owner == null ) {
505 			throw new HibernateException( "unable to locate collection fetch(es) owner for scrollability checks" );
506 		}
507 
508 		// This is not strictly true.  We actually just need to make sure that
509 		// it is ordered by root-entity PK and that that order-by comes before
510 		// any non-root-entity ordering...
511 
512 		AST primaryOrdering = query.getOrderByClause().getFirstChild();
513 		if ( primaryOrdering != null ) {
514 			// TODO : this is a bit dodgy, come up with a better way to check this (plus see above comment)
515 			String [] idColNames = owner.getQueryable().getIdentifierColumnNames();
516 			String expectedPrimaryOrderSeq = StringHelper.join(
517 			        ", ",
518 			        StringHelper.qualify( owner.getTableAlias(), idColNames )
519 			);
520 			if (  !primaryOrdering.getText().startsWith( expectedPrimaryOrderSeq ) ) {
521 				throw new HibernateException( "cannot scroll results with collection fetches which are not ordered primarily by the root entity's PK" );
522 			}
523 		}
524 	}

526 	private StatementExecutor buildAppropriateStatementExecutor(HqlSqlWalker walker) {
527 		Statement statement = ( Statement ) walker.getAST();
528 		if ( walker.getStatementType() == HqlSqlTokenTypes.DELETE ) {
529 			FromElement fromElement = walker.getFinalFromClause().getFromElement();
530 			Queryable persister = fromElement.getQueryable();
531 			if ( persister.isMultiTable() ) {
532 				return new MultiTableDeleteExecutor( walker );
533 			}
534 			else {
535 				return new DeleteExecutor( walker, persister );
536 			}
537 		}
538 		else if ( walker.getStatementType() == HqlSqlTokenTypes.UPDATE ) {
539 			FromElement fromElement = walker.getFinalFromClause().getFromElement();
540 			Queryable persister = fromElement.getQueryable();
541 			if ( persister.isMultiTable() ) {
542 				// even here, if only properties mapped to the "base table" are referenced
543 				// in the set and where clauses, this could be handled by the BasicDelegate.
544 				// TODO : decide if it is better performance-wise to doAfterTransactionCompletion that check, or to simply use the MultiTableUpdateDelegate
545 				return new MultiTableUpdateExecutor( walker );
546 			}
547 			else {
548 				return new BasicExecutor( walker, persister );
549 			}
550 		}
551 		else if ( walker.getStatementType() == HqlSqlTokenTypes.INSERT ) {
552 			return new BasicExecutor( walker, ( ( InsertStatement ) statement ).getIntoClause().getQueryable() );
553 		}
554 		else {
555 			throw new QueryException( "Unexpected statement type" );
556 		}
557 	}

576 	public static class JavaConstantConverter implements NodeTraverser.VisitationStrategy {
577 		private AST dotRoot;
578 		public void visit(AST node) {
579 			if ( dotRoot != null ) {
580 				// we are already processing a dot-structure
581                 if (ASTUtil.isSubtreeChild(dotRoot, node)) return;
582                 // we are now at a new tree level
583                 dotRoot = null;
584 			}
585 
586 			if ( dotRoot == null && node.getType() == HqlTokenTypes.DOT ) {
587 				dotRoot = node;
588 				handleDotStructure( dotRoot );
589 			}
590 		}
591 		private void handleDotStructure(AST dotStructureRoot) {
592 			String expression = ASTUtil.getPathText( dotStructureRoot );
593 			Object constant = ReflectHelper.getConstantValue( expression );
594 			if ( constant != null ) {
595 				dotStructureRoot.setFirstChild( null );
596 				dotStructureRoot.setType( HqlTokenTypes.JAVA_CONSTANT );
597 				dotStructureRoot.setText( expression );
598 			}
599 		}
600 	}
