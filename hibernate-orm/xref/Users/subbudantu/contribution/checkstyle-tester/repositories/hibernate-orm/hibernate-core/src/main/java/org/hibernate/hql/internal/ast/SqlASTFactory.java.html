76  /**
77   * Custom AST factory the intermediate tree that causes ANTLR to create specialized
78   * AST nodes, given the AST node type (from HqlSqlTokenTypes).   HqlSqlWalker registers
79   * this factory with itself when it is initialized.
80   *
81   * @author Joshua
82   */
83  public class SqlASTFactory extends ASTFactory implements HqlSqlTokenTypes {
84  	private HqlSqlWalker walker;
85  
86  	/**
87  	 * Create factory with a specific mapping from token type
88  	 * to Java AST node type.  Your subclasses of ASTFactory
89  	 * can override and reuse the map stuff.
90  	 */
91  	public SqlASTFactory(HqlSqlWalker walker) {
92  		super();
93  		this.walker = walker;
94  	}
95  
96  	/**
97  	 * Returns the class for a given token type (a.k.a. AST node type).
98  	 *
99  	 * @param tokenType The token type.
100 	 * @return Class - The AST node class to instantiate.
101 	 */
102 	public Class getASTNodeType(int tokenType) {
103 		switch ( tokenType ) {
104 			case SELECT:
105 			case QUERY:
106 				return QueryNode.class;
107 			case UPDATE:
108 				return UpdateStatement.class;
109 			case DELETE:
110 				return DeleteStatement.class;
111 			case INSERT:
112 				return InsertStatement.class;
113 			case INTO:
114 				return IntoClause.class;
115 			case FROM:
116 				return FromClause.class;
117 			case FROM_FRAGMENT:
118 				return FromElement.class;
119 			case IMPLIED_FROM:
120 				return ImpliedFromElement.class;
121 			case DOT:
122 				return DotNode.class;
123 			case INDEX_OP:
124 				return IndexNode.class;
125 				// Alias references and identifiers use the same node class.
126 			case ALIAS_REF:
127 			case IDENT:
128 				return IdentNode.class;
129 			case RESULT_VARIABLE_REF:
130 				return ResultVariableRefNode.class;
131 			case SQL_TOKEN:
132 				return SqlFragment.class;
133 			case METHOD_CALL:
134 				return MethodNode.class;
135 			case CAST:
136 				return CastFunctionNode.class;
137 			case ELEMENTS:
138 			case INDICES:
139 				return CollectionFunction.class;
140 			case SELECT_CLAUSE:
141 				return SelectClause.class;
142 			case SELECT_EXPR:
143 				return SelectExpressionImpl.class;
144 			case AGGREGATE:
145 				return AggregateNode.class;
146 			case COUNT:
147 				return CountNode.class;
148 			case CONSTRUCTOR:
149 				return ConstructorNode.class;
150 			case NUM_INT:
151 			case NUM_FLOAT:
152 			case NUM_LONG:
153 			case NUM_DOUBLE:
154 			case NUM_BIG_INTEGER:
155 			case NUM_BIG_DECIMAL:
156 			case QUOTED_STRING:
157 				return LiteralNode.class;
158 			case TRUE:
159 			case FALSE:
160 				return BooleanLiteralNode.class;
161 			case JAVA_CONSTANT:
162 				return JavaConstantNode.class;
163 			case ORDER:
164 				return OrderByClause.class;
165 			case PLUS:
166 			case MINUS:
167 			case STAR:
168 			case DIV:
169 			case MOD:
170 				return BinaryArithmeticOperatorNode.class;
171 			case UNARY_MINUS:
172 			case UNARY_PLUS:
173 				return UnaryArithmeticNode.class;
174 			case CASE2:
175 				return SimpleCaseNode.class;
176 			case CASE:
177 				return SearchedCaseNode.class;
178 			case PARAM:
179 			case NAMED_PARAM:
180 				return ParameterNode.class;
181 			case EQ:
182 			case NE:
183 			case LT:
184 			case GT:
185 			case LE:
186 			case GE:
187 			case LIKE:
188 			case NOT_LIKE:
189 				return BinaryLogicOperatorNode.class;
190 			case IN:
191 			case NOT_IN:
192 				return InLogicOperatorNode.class;
193 			case BETWEEN:
194 			case NOT_BETWEEN:
195 				return BetweenOperatorNode.class;
196 			case IS_NULL:
197 				return IsNullLogicOperatorNode.class;
198 			case IS_NOT_NULL:
199 				return IsNotNullLogicOperatorNode.class;
200 			case EXISTS:
201 				return UnaryLogicOperatorNode.class;
202 			case KEY: {
203 				return MapKeyNode.class;
204 			}
205 			case VALUE: {
206 				return MapValueNode.class;
207 			}
208 			case ENTRY: {
209 				return MapEntryNode.class;
210 			}
211 			default:
212 				return SqlNode.class;
213 		} // switch
214 	}

216 	protected AST createUsingCtor(Token token, String className) {
217 		Class c;
218 		AST t;
219 		try {
220 			c = Class.forName( className );
221 			Class[] tokenArgType = new Class[]{antlr.Token.class};
222 			Constructor ctor = c.getConstructor( tokenArgType );
223 			if ( ctor != null ) {
224 				t = ( AST ) ctor.newInstance( new Object[]{token} ); // make a new one
225 				initializeSqlNode( t );
226 			}
227 			else {
228 				// just do the regular thing if you can't find the ctor
229 				// Your AST must have default ctor to use this.
230 				t = create( c );
231 			}
232 		}
233 		catch ( Exception e ) {
234 			throw new IllegalArgumentException( "Invalid class or can't make instance, " + className );
235 		}
236 		return t;
237 	}
238 
239 	private void initializeSqlNode(AST t) {
240 		// Initialize SQL nodes here.
241 		if ( t instanceof InitializeableNode ) {
242 			InitializeableNode initializeableNode = ( InitializeableNode ) t;
243 			initializeableNode.initialize( walker );
244 		}
245 		if ( t instanceof SessionFactoryAwareNode ) {
246 			( ( SessionFactoryAwareNode ) t ).setSessionFactory( walker.getSessionFactoryHelper().getFactory() );
247 		}
248 	}
249 
250 	/**
251 	 * Actually instantiate the AST node.
252 	 *
253 	 * @param c The class to instantiate.
254 	 * @return The instantiated and initialized node.
255 	 */
256 	protected AST create(Class c) {
257 		AST t;
258 		try {
259 			t = ( AST ) c.newInstance(); // make a new one
260 			initializeSqlNode( t );
261 		}
262 		catch ( Exception e ) {
263 			error( "Can't create AST Node " + c.getName() );
264 			return null;
265 		}
266 		return t;
267 	}
