99  	FromElement addFromElement() throws SemanticException {
100 		FromClause parentFromClause = fromClause.getParentFromClause();
101 		if ( parentFromClause != null ) {
102 			// Look up class name using the first identifier in the path.
103 			String pathAlias = PathHelper.getAlias( path );
104 			FromElement parentFromElement = parentFromClause.getFromElement( pathAlias );
105 			if ( parentFromElement != null ) {
106 				return createFromElementInSubselect( path, pathAlias, parentFromElement, classAlias );
107 			}
108 		}
109 
110 		EntityPersister entityPersister = fromClause.getSessionFactoryHelper().requireClassPersister( path );
111 
112 		FromElement elem = createAndAddFromElement( path,
113 				classAlias,
114 				entityPersister,
115 				( EntityType ) ( ( Queryable ) entityPersister ).getType(),
116 				null );
117 
118 		// Add to the query spaces.
119 		fromClause.getWalker().addQuerySpaces( entityPersister.getQuerySpaces() );
120 
121 		return elem;
122 	}

124 	private FromElement createFromElementInSubselect(
125 	        String path,
126 	        String pathAlias,
127 	        FromElement parentFromElement,
128 	        String classAlias) throws SemanticException {
129 		LOG.debugf( "createFromElementInSubselect() : path = %s", path );
130 		// Create an DotNode AST for the path and resolve it.
131 		FromElement fromElement = evaluateFromElementPath( path, classAlias );
132 		EntityPersister entityPersister = fromElement.getEntityPersister();
133 
134 		// If the first identifier in the path refers to the class alias (not the class name), then this
135 		// is a correlated subselect.  If it's a correlated sub-select, use the existing table alias.  Otherwise
136 		// generate a new one.
137 		String tableAlias = null;
138 		boolean correlatedSubselect = pathAlias.equals( parentFromElement.getClassAlias() );
139 		if ( correlatedSubselect ) {
140 			tableAlias = fromElement.getTableAlias();
141 		}
142 		else {
143 			tableAlias = null;
144 		}
145 
146 		// If the from element isn't in the same clause, create a new from element.
147 		if ( fromElement.getFromClause() != fromClause ) {
148 			LOG.debug( "createFromElementInSubselect() : creating a new FROM element..." );
149 			fromElement = createFromElement( entityPersister );
150 			initializeAndAddFromElement( fromElement,
151 					path,
152 					classAlias,
153 					entityPersister,
154 					( EntityType ) ( ( Queryable ) entityPersister ).getType(),
155 					tableAlias
156 			);
157 		}
158 		LOG.debugf( "createFromElementInSubselect() : %s -> %s", path, fromElement );
159 		return fromElement;
160 	}

162 	private FromElement evaluateFromElementPath(String path, String classAlias) throws SemanticException {
163 		ASTFactory factory = fromClause.getASTFactory();
164 		FromReferenceNode pathNode = ( FromReferenceNode ) PathHelper.parsePath( path, factory );
165 		pathNode.recursiveResolve( FromReferenceNode.ROOT_LEVEL, // This is the root level node.
166 				false, // Generate an explicit from clause at the root.
167 				classAlias,
168 		        null
169 		);
170         if (pathNode.getImpliedJoin() != null) return pathNode.getImpliedJoin();
171         return pathNode.getFromElement();
172 	}

174 	FromElement createCollectionElementsJoin(
175 	        QueryableCollection queryableCollection,
176 	        String collectionName) throws SemanticException {
177 		JoinSequence collectionJoinSequence = fromClause.getSessionFactoryHelper()
178 		        .createCollectionJoinSequence( queryableCollection, collectionName );
179 		this.queryableCollection = queryableCollection;
180 		return createCollectionJoin( collectionJoinSequence, null );
181 	}

183 	FromElement createCollection(
184 	        QueryableCollection queryableCollection,
185 	        String role,
186 	        JoinType joinType,
187 	        boolean fetchFlag,
188 	        boolean indexed)
189 			throws SemanticException {
190 		if ( !collection ) {
191 			throw new IllegalStateException( "FromElementFactory not initialized for collections!" );
192 		}
193 		this.inElementsFunction = indexed;
194 		FromElement elem;
195 		this.queryableCollection = queryableCollection;
196 		collectionType = queryableCollection.getCollectionType();
197 		String roleAlias = fromClause.getAliasGenerator().createName( role );
198 
199 		// Correlated subqueries create 'special' implied from nodes
200 		// because correlated subselects can't use an ANSI-style join
201 		boolean explicitSubqueryFromElement = fromClause.isSubQuery() && !implied;
202 		if ( explicitSubqueryFromElement ) {
203 			String pathRoot = StringHelper.root( path );
204 			FromElement origin = fromClause.getFromElement( pathRoot );
205 			if ( origin == null || origin.getFromClause() != fromClause ) {
206 				implied = true;
207 			}
208 		}
209 
210 		// super-duper-classic-parser-regression-testing-mojo-magic...
211 		if ( explicitSubqueryFromElement && DotNode.useThetaStyleImplicitJoins ) {
212 			implied = true;
213 		}
214 
215 		Type elementType = queryableCollection.getElementType();
216 		if ( elementType.isEntityType() ) { 			// A collection of entities...
217 			elem = createEntityAssociation( role, roleAlias, joinType );
218 		}
219 		else if ( elementType.isComponentType() ) {		// A collection of components...
220 			JoinSequence joinSequence = createJoinSequence( roleAlias, joinType );
221 			elem = createCollectionJoin( joinSequence, roleAlias );
222 		}
223 		else {											// A collection of scalar elements...
224 			JoinSequence joinSequence = createJoinSequence( roleAlias, joinType );
225 			elem = createCollectionJoin( joinSequence, roleAlias );
226 		}
227 
228 		elem.setRole( role );
229 		elem.setQueryableCollection( queryableCollection );
230 		// Don't include sub-classes for implied collection joins or subquery joins.
231 		if ( implied ) {
232 			elem.setIncludeSubclasses( false );
233 		}
234 
235 		if ( explicitSubqueryFromElement ) {
236 			elem.setInProjectionList( true );	// Treat explict from elements in sub-queries properly.
237 		}
238 
239 		if ( fetchFlag ) {
240 			elem.setFetch( true );
241 		}
242 		return elem;
243 	}

245 	FromElement createEntityJoin(
246 	        String entityClass,
247 	        String tableAlias,
248 	        JoinSequence joinSequence,
249 	        boolean fetchFlag,
250 	        boolean inFrom,
251 	        EntityType type) throws SemanticException {
252 		FromElement elem = createJoin( entityClass, tableAlias, joinSequence, type, false );
253 		elem.setFetch( fetchFlag );
254 		EntityPersister entityPersister = elem.getEntityPersister();
255 		int numberOfTables = entityPersister.getQuerySpaces().length;
256 		if ( numberOfTables > 1 && implied && !elem.useFromFragment() ) {
257 			LOG.debug( "createEntityJoin() : Implied multi-table entity join" );
258 			elem.setUseFromFragment( true );
259 		}
260 
261 		// If this is an implied join in a FROM clause, then use ANSI-style joining, and set the
262 		// flag on the FromElement that indicates that it was implied in the FROM clause itself.
263 		if ( implied && inFrom ) {
264 			joinSequence.setUseThetaStyle( false );
265 			elem.setUseFromFragment( true );
266 			elem.setImpliedInFromClause( true );
267 		}
268 		if ( elem.getWalker().isSubQuery() ) {
269 			// two conditions where we need to transform this to a theta-join syntax:
270 			//      1) 'elem' is the "root from-element" in correlated subqueries
271 			//      2) The DotNode.useThetaStyleImplicitJoins has been set to true
272 			//          and 'elem' represents an implicit join
273 			if ( elem.getFromClause() != elem.getOrigin().getFromClause() ||
274 //			        ( implied && DotNode.useThetaStyleImplicitJoins ) ) {
275 			        DotNode.useThetaStyleImplicitJoins ) {
276 				// the "root from-element" in correlated subqueries do need this piece
277 				elem.setType( FROM_FRAGMENT );
278 				joinSequence.setUseThetaStyle( true );
279 				elem.setUseFromFragment( false );
280 			}
281 		}
282 
283 		return elem;
284 	}

292 	FromElement createElementJoin(QueryableCollection queryableCollection) throws SemanticException {
293 		FromElement elem;
294 
295 		implied = true; //TODO: always true for now, but not if we later decide to support elements() in the from clause
296 		inElementsFunction = true;
297 		Type elementType = queryableCollection.getElementType();
298 		if ( !elementType.isEntityType() ) {
299 			throw new IllegalArgumentException( "Cannot create element join for a collection of non-entities!" );
300 		}
301 		this.queryableCollection = queryableCollection;
302 		SessionFactoryHelper sfh = fromClause.getSessionFactoryHelper();
303 		FromElement destination = null;
304 		String tableAlias = null;
305 		EntityPersister entityPersister = queryableCollection.getElementPersister();
306 		tableAlias = fromClause.getAliasGenerator().createName( entityPersister.getEntityName() );
307 		String associatedEntityName = entityPersister.getEntityName();
308 		EntityPersister targetEntityPersister = sfh.requireClassPersister( associatedEntityName );
309 		// Create the FROM element for the target (the elements of the collection).
310 		destination = createAndAddFromElement(
311 				associatedEntityName,
312 				classAlias,
313 				targetEntityPersister,
314 				( EntityType ) queryableCollection.getElementType(),
315 				tableAlias
316 			);
317 		// If the join is implied, then don't include sub-classes on the element.
318 		if ( implied ) {
319 			destination.setIncludeSubclasses( false );
320 		}
321 		fromClause.addCollectionJoinFromElementByPath( path, destination );
322 //		origin.addDestination(destination);
323 		// Add the query spaces.
324 		fromClause.getWalker().addQuerySpaces( entityPersister.getQuerySpaces() );
325 
326 		CollectionType type = queryableCollection.getCollectionType();
327 		String role = type.getRole();
328 		String roleAlias = origin.getTableAlias();
329 
330 		String[] targetColumns = sfh.getCollectionElementColumns( role, roleAlias );
331 		AssociationType elementAssociationType = sfh.getElementAssociationType( type );
332 
333 		// Create the join element under the from element.
334 		JoinType joinType = JoinType.INNER_JOIN;
335 		JoinSequence joinSequence = sfh.createJoinSequence( implied, elementAssociationType, tableAlias, joinType, targetColumns );
336 		elem = initializeJoin( path, destination, joinSequence, targetColumns, origin, false );
337 		elem.setUseFromFragment( true );	// The associated entity is implied, but it must be included in the FROM.
338 		elem.setCollectionTableAlias( roleAlias );	// The collection alias is the role.
339 		return elem;
340 	}

342 	private FromElement createCollectionJoin(JoinSequence collectionJoinSequence, String tableAlias) throws SemanticException {
343 		String text = queryableCollection.getTableName();
344 		AST ast = createFromElement( text );
345 		FromElement destination = ( FromElement ) ast;
346 		Type elementType = queryableCollection.getElementType();
347 		if ( elementType.isCollectionType() ) {
348 			throw new SemanticException( "Collections of collections are not supported!" );
349 		}
350 		destination.initializeCollection( fromClause, classAlias, tableAlias );
351 		destination.setType( JOIN_FRAGMENT );		// Tag this node as a JOIN.
352 		destination.setIncludeSubclasses( false );	// Don't include subclasses in the join.
353 		destination.setCollectionJoin( true );		// This is a clollection join.
354 		destination.setJoinSequence( collectionJoinSequence );
355 		destination.setOrigin( origin, false );
356 		destination.setCollectionTableAlias(tableAlias);
357 //		origin.addDestination( destination );
358 // This was the cause of HHH-242
359 //		origin.setType( FROM_FRAGMENT );			// Set the parent node type so that the AST is properly formed.
360 		origin.setText( "" );						// The destination node will have all the FROM text.
361 		origin.setCollectionJoin( true );			// The parent node is a collection join too (voodoo - see JoinProcessor)
362 		fromClause.addCollectionJoinFromElementByPath( path, destination );
363 		fromClause.getWalker().addQuerySpaces( queryableCollection.getCollectionSpaces() );
364 		return destination;
365 	}

367 	private FromElement createEntityAssociation(
368 	        String role,
369 	        String roleAlias,
370 	        JoinType joinType) throws SemanticException {
371 		FromElement elem;
372 		Queryable entityPersister = ( Queryable ) queryableCollection.getElementPersister();
373 		String associatedEntityName = entityPersister.getEntityName();
374 		// Get the class name of the associated entity.
375 		if ( queryableCollection.isOneToMany() ) {
376 			LOG.debugf( "createEntityAssociation() : One to many - path = %s role = %s associatedEntityName = %s",
377 					path,
378 					role,
379 					associatedEntityName );
380 			JoinSequence joinSequence = createJoinSequence( roleAlias, joinType );
381 
382 			elem = createJoin( associatedEntityName, roleAlias, joinSequence, ( EntityType ) queryableCollection.getElementType(), false );
383 		}
384 		else {
385 			LOG.debugf( "createManyToMany() : path = %s role = %s associatedEntityName = %s", path, role, associatedEntityName );
386 			elem = createManyToMany( role, associatedEntityName,
387 					roleAlias, entityPersister, ( EntityType ) queryableCollection.getElementType(), joinType );
388 			fromClause.getWalker().addQuerySpaces( queryableCollection.getCollectionSpaces() );
389 		}
390 		elem.setCollectionTableAlias( roleAlias );
391 		return elem;
392 	}

394 	private FromElement createJoin(
395 	        String entityClass,
396 	        String tableAlias,
397 	        JoinSequence joinSequence,
398 	        EntityType type,
399 	        boolean manyToMany) throws SemanticException {
400 		//  origin, path, implied, columns, classAlias,
401 		EntityPersister entityPersister = fromClause.getSessionFactoryHelper().requireClassPersister( entityClass );
402 		FromElement destination = createAndAddFromElement( entityClass,
403 				classAlias,
404 				entityPersister,
405 				type,
406 				tableAlias );
407 		return initializeJoin( path, destination, joinSequence, getColumns(), origin, manyToMany );
408 	}

410 	private FromElement createManyToMany(
411 	        String role,
412 	        String associatedEntityName,
413 	        String roleAlias,
414 	        Queryable entityPersister,
415 	        EntityType type,
416 	        JoinType joinType) throws SemanticException {
417 		FromElement elem;
418 		SessionFactoryHelper sfh = fromClause.getSessionFactoryHelper();
419 		if ( inElementsFunction /*implied*/ ) {
420 			// For implied many-to-many, just add the end join.
421 			JoinSequence joinSequence = createJoinSequence( roleAlias, joinType );
422 			elem = createJoin( associatedEntityName, roleAlias, joinSequence, type, true );
423 		}
424 		else {
425 			// For an explicit many-to-many relationship, add a second join from the intermediate
426 			// (many-to-many) table to the destination table.  Also, make sure that the from element's
427 			// idea of the destination is the destination table.
428 			String tableAlias = fromClause.getAliasGenerator().createName( entityPersister.getEntityName() );
429 			String[] secondJoinColumns = sfh.getCollectionElementColumns( role, roleAlias );
430 			// Add the second join, the one that ends in the destination table.
431 			JoinSequence joinSequence = createJoinSequence( roleAlias, joinType );
432 			joinSequence.addJoin( sfh.getElementAssociationType( collectionType ), tableAlias, joinType, secondJoinColumns );
433 			elem = createJoin( associatedEntityName, tableAlias, joinSequence, type, false );
434 			elem.setUseFromFragment( true );
435 		}
436 		return elem;
437 	}

439 	private JoinSequence createJoinSequence(String roleAlias, JoinType joinType) {
440 		SessionFactoryHelper sessionFactoryHelper = fromClause.getSessionFactoryHelper();
441 		String[] joinColumns = getColumns();
442 		if ( collectionType == null ) {
443 			throw new IllegalStateException( "collectionType is null!" );
444 		}
445 		return sessionFactoryHelper.createJoinSequence( implied, collectionType, roleAlias, joinType, joinColumns );
446 	}

448 	private FromElement createAndAddFromElement(
449 	        String className,
450 	        String classAlias,
451 	        EntityPersister entityPersister,
452 	        EntityType type,
453 	        String tableAlias) {
454 		if ( !( entityPersister instanceof Joinable ) ) {
455 			throw new IllegalArgumentException( "EntityPersister " + entityPersister + " does not implement Joinable!" );
456 		}
457 		FromElement element = createFromElement( entityPersister );
458 		initializeAndAddFromElement( element, className, classAlias, entityPersister, type, tableAlias );
459 		return element;
460 	}

462 	private void initializeAndAddFromElement(
463 	        FromElement element,
464 	        String className,
465 	        String classAlias,
466 	        EntityPersister entityPersister,
467 	        EntityType type,
468 	        String tableAlias) {
469 		if ( tableAlias == null ) {
470 			AliasGenerator aliasGenerator = fromClause.getAliasGenerator();
471 			tableAlias = aliasGenerator.createName( entityPersister.getEntityName() );
472 		}
473 		element.initializeEntity( fromClause, className, entityPersister, type, classAlias, tableAlias );
474 	}

476 	private FromElement createFromElement(EntityPersister entityPersister) {
477 		Joinable joinable = ( Joinable ) entityPersister;
478 		String text = joinable.getTableName();
479 		AST ast = createFromElement( text );
480 		FromElement element = ( FromElement ) ast;
481 		return element;
482 	}
483 
484 	private AST createFromElement(String text) {
485 		AST ast = ASTUtil.create( fromClause.getASTFactory(),
486 				implied ? IMPLIED_FROM : FROM_FRAGMENT, // This causes the factory to instantiate the desired class.
487 				text );
488 		// Reset the node type, because the rest of the system is expecting FROM_FRAGMENT, all we wanted was
489 		// for the factory to create the right sub-class.  This might get reset again later on anyway to make the
490 		// SQL generation simpler.
491 		ast.setType( FROM_FRAGMENT );
492 		return ast;
493 	}

495 	private FromElement initializeJoin(
496 	        String path,
497 	        FromElement destination,
498 	        JoinSequence joinSequence,
499 	        String[] columns,
500 	        FromElement origin,
501 	        boolean manyToMany) {
502 		destination.setType( JOIN_FRAGMENT );
503 		destination.setJoinSequence( joinSequence );
504 		destination.setColumns( columns );
505 		destination.setOrigin( origin, manyToMany );
506 		fromClause.addJoinByPathMap( path, destination );
507 		return destination;
508 	}

510 	private String[] getColumns() {
511 		if ( columns == null ) {
512 			throw new IllegalStateException( "No foriegn key columns were supplied!" );
513 		}
514 		return columns;
