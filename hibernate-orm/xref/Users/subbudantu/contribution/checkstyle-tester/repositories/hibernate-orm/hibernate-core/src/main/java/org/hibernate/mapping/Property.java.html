109 	public CascadeStyle getCascadeStyle() throws MappingException {
110 		Type type = value.getType();
111 		if ( type.isComponentType() ) {
112 			return getCompositeCascadeStyle( (CompositeType) type, cascade );
113 		}
114 		else if ( type.isCollectionType() ) {
115 			return getCollectionCascadeStyle( ( (Collection) value ).getElement().getType(), cascade );
116 		}
117 		else {
118 			return getCascadeStyle( cascade );			
119 		}
120 	}
121 
122 	private static CascadeStyle getCompositeCascadeStyle(CompositeType compositeType, String cascade) {
123 		if ( compositeType.isAnyType() ) {
124 			return getCascadeStyle( cascade );
125 		}
126 		int length = compositeType.getSubtypes().length;
127 		for ( int i=0; i<length; i++ ) {
128 			if ( compositeType.getCascadeStyle(i) != CascadeStyle.NONE ) {
129 				return CascadeStyle.ALL;
130 			}
131 		}
132 		return getCascadeStyle( cascade );
133 	}
134 
135 	private static CascadeStyle getCollectionCascadeStyle(Type elementType, String cascade) {
136 		if ( elementType.isComponentType() ) {
137 			return getCompositeCascadeStyle( (CompositeType) elementType, cascade );
138 		}
139 		else {
140 			return getCascadeStyle( cascade );
141 		}
142 	}
143 	
144 	private static CascadeStyle getCascadeStyle(String cascade) {
145 		if ( cascade==null || cascade.equals("none") ) {
146 			return CascadeStyle.NONE;
147 		}
148 		else {
149 			StringTokenizer tokens = new StringTokenizer(cascade, ", ");
150 			CascadeStyle[] styles = new CascadeStyle[ tokens.countTokens() ] ;
151 			int i=0;
152 			while ( tokens.hasMoreTokens() ) {
153 				styles[i++] = CascadeStyle.getCascadeStyle( tokens.nextToken() );
154 			}
155 			return new CascadeStyle.MultipleCascadeStyle(styles);
156 		}		
157 	}

175 	public boolean isUpdateable() {
176 		// if the property mapping consists of all formulas,
177 		// make it non-updateable
178 		return updateable && !ArrayHelper.isAllFalse( value.getColumnUpdateability() );
179 	}
180 
181 	public boolean isInsertable() {
182 		// if the property mapping consists of all formulas, 
183 		// make it non-insertable
184 		final boolean[] columnInsertability = value.getColumnInsertability();
185 		return insertable && (
186 				columnInsertability.length==0 ||
187 				!ArrayHelper.isAllFalse( columnInsertability )
188 			);
189 	}

250 	public boolean isLazy() {
251 		if ( value instanceof ToOne ) {
252 			// both many-to-one and one-to-one are represented as a
253 			// Property.  EntityPersister is relying on this value to
254 			// determine "lazy fetch groups" in terms of field-level
255 			// interception.  So we need to make sure that we return
256 			// true here for the case of many-to-one and one-to-one
257 			// with lazy="no-proxy"
258 			//
259 			// * impl note - lazy="no-proxy" currently forces both
260 			// lazy and unwrap to be set to true.  The other case we
261 			// are extremely interested in here is that of lazy="proxy"
262 			// where lazy is set to true, but unwrap is set to false.
263 			// thus we use both here under the assumption that this
264 			// return is really only ever used during persister
265 			// construction to determine the lazy property/field fetch
266 			// groupings.  If that assertion changes then this check
267 			// needs to change as well.  Partially, this is an issue with
268 			// the overloading of the term "lazy" here...
269 			ToOne toOneValue = ( ToOne ) value;
270 			return toOneValue.isLazy() && toOneValue.isUnwrapProxy();
271 		}
272 		return lazy;
273 	}
