81  	 */
82  	@Override
83      protected String generateDeleteString() {
84  		
85  		Delete delete = new Delete()
86  				.setTableName( qualifiedTableName )
87  				.addPrimaryKeyColumns( keyColumnNames );
88  		
89  		if ( hasWhere ) delete.setWhere( sqlWhereString );
90  		
91  		if ( getFactory().getSettings().isCommentsEnabled() ) {
92  			delete.setComment( "delete collection " + getRole() );
93  		}
94  		
95  		return delete.toStatementString();
96  	}

101 	@Override
102     protected String generateInsertRowString() {
103 		
104 		Insert insert = new Insert( getDialect() )
105 				.setTableName( qualifiedTableName )
106 				.addColumns( keyColumnNames );
107 		
108 		if ( hasIdentifier) insert.addColumn( identifierColumnName );
109 		
110 		if ( hasIndex /*&& !indexIsFormula*/ ) {
111 			insert.addColumns( indexColumnNames, indexColumnIsSettable );
112 		}
113 		
114 		if ( getFactory().getSettings().isCommentsEnabled() ) {
115 			insert.setComment( "insert collection row " + getRole() );
116 		}
117 		
118 		//if ( !elementIsFormula ) {
119 			insert.addColumns( elementColumnNames, elementColumnIsSettable, elementColumnWriters );
120 		//}
121 		
122 		return insert.toStatementString();
123 	}

126 	 * Generate the SQL UPDATE that updates a row
127 	 */
128 	@Override
129     protected String generateUpdateRowString() {
130 		
131 		Update update = new Update( getDialect() )
132 			.setTableName( qualifiedTableName );
133 		
134 		//if ( !elementIsFormula ) {
135 			update.addColumns( elementColumnNames, elementColumnIsSettable, elementColumnWriters );
136 		//}
137 		
138 		if ( hasIdentifier ) {
139 			update.addPrimaryKeyColumns( new String[]{ identifierColumnName } );
140 		}
141 		else if ( hasIndex && !indexContainsFormula ) {
142 			update.addPrimaryKeyColumns( ArrayHelper.join( keyColumnNames, indexColumnNames ) );
143 		}
144 		else {
145 			update.addPrimaryKeyColumns( keyColumnNames );
146 			update.addPrimaryKeyColumns( elementColumnNames, elementColumnIsInPrimaryKey, elementColumnWriters );
147 		}
148 		
149 		if ( getFactory().getSettings().isCommentsEnabled() ) {
150 			update.setComment( "update collection row " + getRole() );
151 		}
152 		
153 		return update.toStatementString();
154 	}

207 	private BasicBatchKey updateBatchKey;
208 
209 	@Override
210     protected int doUpdateRows(Serializable id, PersistentCollection collection, SessionImplementor session)
211 			throws HibernateException {
212 		
213 		if ( ArrayHelper.isAllFalse(elementColumnIsSettable) ) return 0;
214 
215 		try {
216 			PreparedStatement st = null;
217 			Expectation expectation = Expectations.appropriateExpectation( getUpdateCheckStyle() );
218 			boolean callable = isUpdateCallable();
219 			boolean useBatch = expectation.canBeBatched();
220 			Iterator entries = collection.entries( this );
221 			String sql = getSQLUpdateRowString();
222 			int i = 0;
223 			int count = 0;
224 			while ( entries.hasNext() ) {
225 				Object entry = entries.next();
226 				if ( collection.needsUpdating( entry, i, elementType ) ) {
227 					int offset = 1;
228 
229 					if ( useBatch ) {
230 						if ( updateBatchKey == null ) {
231 							updateBatchKey = new BasicBatchKey(
232 									getRole() + "#UPDATE",
233 									expectation
234 							);
235 						}
236 						st = session.getTransactionCoordinator()
237 								.getJdbcCoordinator()
238 								.getBatch( updateBatchKey )
239 								.getBatchStatement( sql, callable );
240 					}
241 					else {
242 						st = session.getTransactionCoordinator()
243 								.getJdbcCoordinator()
244 								.getStatementPreparer()
245 								.prepareStatement( sql, callable );
246 					}
247 
248 					try {
249 						offset+= expectation.prepare( st );
250 						int loc = writeElement( st, collection.getElement( entry ), offset, session );
251 						if ( hasIdentifier ) {
252 							writeIdentifier( st, collection.getIdentifier( entry, i ), loc, session );
253 						}
254 						else {
255 							loc = writeKey( st, id, loc, session );
256 							if ( hasIndex && !indexContainsFormula ) {
257 								writeIndexToWhere( st, collection.getIndex( entry, i, this ), loc, session );
258 							}
259 							else {
260 								writeElementToWhere( st, collection.getSnapshotElement( entry, i ), loc, session );
261 							}
262 						}
263 
264 						if ( useBatch ) {
265 							session.getTransactionCoordinator()
266 									.getJdbcCoordinator()
267 									.getBatch( updateBatchKey )
268 									.addToBatch();
269 						}
270 						else {
271 							expectation.verifyOutcome( session.getTransactionCoordinator().getJdbcCoordinator().getResultSetReturn().executeUpdate( st ), st, -1 );
272 						}
273 					}
274 					catch ( SQLException sqle ) {
275 						if ( useBatch ) {
276 							session.getTransactionCoordinator().getJdbcCoordinator().abortBatch();
277 						}
278 						throw sqle;
279 					}
280 					finally {
281 						if ( !useBatch ) {
282 							session.getTransactionCoordinator().getJdbcCoordinator().release( st );
283 						}
284 					}
285 					count++;
286 				}
287 				i++;
288 			}
289 			return count;
290 		}
291 		catch ( SQLException sqle ) {
292 			throw getSQLExceptionHelper().convert(
293 					sqle,
294 					"could not update collection rows: " + MessageHelper.collectionInfoString( this, collection, id, session ),
295 					getSQLUpdateRowString()
296 				);
297 		}
298 	}

300 	public String selectFragment(
301 	        Joinable rhs,
302 	        String rhsAlias,
303 	        String lhsAlias,
304 	        String entitySuffix,
305 	        String collectionSuffix,
306 	        boolean includeCollectionColumns) {
307 		// we need to determine the best way to know that two joinables
308 		// represent a single many-to-many...
309 		if ( rhs != null && isManyToMany() && !rhs.isCollection() ) {
310 			AssociationType elementType = ( ( AssociationType ) getElementType() );
311 			if ( rhs.equals( elementType.getAssociatedJoinable( getFactory() ) ) ) {
312 				return manyToManySelectFragment( rhs, rhsAlias, lhsAlias, collectionSuffix );
313 			}
314 		}
315 		return includeCollectionColumns ? selectFragment( lhsAlias, collectionSuffix ) : "";
316 	}
317 
318 	private String manyToManySelectFragment(
319 	        Joinable rhs,
320 	        String rhsAlias,
321 	        String lhsAlias,
322 	        String collectionSuffix) {
323 		SelectFragment frag = generateSelectFragment( lhsAlias, collectionSuffix );
324 
325 		String[] elementColumnNames = rhs.getKeyColumnNames();
326 		frag.addColumns( rhsAlias, elementColumnNames, elementColumnAliases );
327 		appendIndexColumns( frag, lhsAlias );
328 		appendIdentifierColumns( frag, lhsAlias );
329 
330 		return frag.toFragmentString()
331 				.substring( 2 ); //strip leading ','
332 	}
