82  		super( entityMetamodel, mappedEntity );
83  		this.mappedClass = mappedEntity.getMappedClass();
84  		this.proxyInterface = mappedEntity.getProxyInterface();
85  		this.lifecycleImplementor = Lifecycle.class.isAssignableFrom( mappedClass );
86  		this.isInstrumented = entityMetamodel.isInstrumented();
87  
88  		Iterator iter = mappedEntity.getPropertyClosureIterator();
89  		while ( iter.hasNext() ) {
90  			Property property = (Property) iter.next();
91  			if ( property.isLazy() ) {
92  				lazyPropertyNames.add( property.getName() );
93  			}
94  		}
95  
96  		String[] getterNames = new String[propertySpan];
97  		String[] setterNames = new String[propertySpan];
98  		Class[] propTypes = new Class[propertySpan];
99  		for ( int i = 0; i < propertySpan; i++ ) {
100 			getterNames[i] = getters[i].getMethodName();
101 			setterNames[i] = setters[i].getMethodName();
102 			propTypes[i] = getters[i].getReturnType();
103 		}
104 
105 		if ( hasCustomAccessors || !Environment.useReflectionOptimizer() ) {
106 			optimizer = null;
107 		}
108 		else {
109 			// todo : YUCK!!!
110 			optimizer = Environment.getBytecodeProvider().getReflectionOptimizer( mappedClass, getterNames, setterNames, propTypes );
111 //			optimizer = getFactory().getSettings().getBytecodeProvider().getReflectionOptimizer(
112 //					mappedClass, getterNames, setterNames, propTypes

126 			if ( property.isLazy() ) {

155 	 * {@inheritDoc}
156 	 */
157 	@Override
158     protected ProxyFactory buildProxyFactory(PersistentClass persistentClass, Getter idGetter, Setter idSetter) {
159 		// determine the id getter and setter methods from the proxy interface (if any)
160         // determine all interfaces needed by the resulting proxy
161 		
162 		/*
163 		 * We need to preserve the order of the interfaces they were put into the set, since javassist will choose the
164 		 * first one's class-loader to construct the proxy class with. This is also the reason why HibernateProxy.class
165 		 * should be the last one in the order (on JBossAS7 its class-loader will be org.hibernate module's class-
166 		 * loader, which will not see the classes inside deployed apps.  See HHH-3078
167 		 */
168 		Set<Class> proxyInterfaces = new java.util.LinkedHashSet<Class>();
169 
170 		Class mappedClass = persistentClass.getMappedClass();
171 		Class proxyInterface = persistentClass.getProxyInterface();
172 
173 		if ( proxyInterface!=null && !mappedClass.equals( proxyInterface ) ) {
174 			if ( !proxyInterface.isInterface() ) {
175 				throw new MappingException(
176 						"proxy must be either an interface, or the class itself: " + getEntityName()
177 				);
178 			}
179 			proxyInterfaces.add( proxyInterface );
180 		}
181 
182 		if ( mappedClass.isInterface() ) {
183 			proxyInterfaces.add( mappedClass );
184 		}
185 
186 		Iterator subclasses = persistentClass.getSubclassIterator();
187 		while ( subclasses.hasNext() ) {
188 			final Subclass subclass = ( Subclass ) subclasses.next();
189 			final Class subclassProxy = subclass.getProxyInterface();
190 			final Class subclassClass = subclass.getMappedClass();
191 			if ( subclassProxy!=null && !subclassClass.equals( subclassProxy ) ) {
192 				if ( !subclassProxy.isInterface() ) {
193 					throw new MappingException(
194 							"proxy must be either an interface, or the class itself: " + subclass.getEntityName()
195 					);
196 				}
197 				proxyInterfaces.add( subclassProxy );
198 			}
199 		}
200 
201 		proxyInterfaces.add( HibernateProxy.class );
202 
203 		Iterator properties = persistentClass.getPropertyIterator();
204 		Class clazz = persistentClass.getMappedClass();
205 		while ( properties.hasNext() ) {
206 			Property property = (Property) properties.next();
207 			Method method = property.getGetter(clazz).getMethod();
208 			if ( method != null && Modifier.isFinal( method.getModifiers() ) ) {
209                 LOG.gettersOfLazyClassesCannotBeFinal(persistentClass.getEntityName(), property.getName());
210 			}
211 			method = property.getSetter(clazz).getMethod();
212             if ( method != null && Modifier.isFinal( method.getModifiers() ) ) {
213                 LOG.settersOfLazyClassesCannotBeFinal(persistentClass.getEntityName(), property.getName());
214 			}
215 		}
216 
217 		Method idGetterMethod = idGetter==null ? null : idGetter.getMethod();
218 		Method idSetterMethod = idSetter==null ? null : idSetter.getMethod();
219 
220 		Method proxyGetIdentifierMethod = idGetterMethod==null || proxyInterface==null ?
221 				null :
222 		        ReflectHelper.getMethod(proxyInterface, idGetterMethod);
223 		Method proxySetIdentifierMethod = idSetterMethod==null || proxyInterface==null  ?
224 				null :
225 		        ReflectHelper.getMethod(proxyInterface, idSetterMethod);
226 
227 		ProxyFactory pf = buildProxyFactoryInternal( persistentClass, idGetter, idSetter );
228 		try {
229 			pf.postInstantiate(
230 					getEntityName(),
231 					mappedClass,
232 					proxyInterfaces,
233 					proxyGetIdentifierMethod,
234 					proxySetIdentifierMethod,
235 					persistentClass.hasEmbeddedIdentifier() ?
236 			                (CompositeType) persistentClass.getIdentifier().getType() :
237 			                null
238 			);
239 		}
240 		catch ( HibernateException he ) {
241             LOG.unableToCreateProxyFactory(getEntityName(), he);
242 			pf = null;
243 		}
244 		return pf;
245 	}

269 	@Override
270 	protected ProxyFactory buildProxyFactory(EntityBinding entityBinding, Getter idGetter, Setter idSetter) {
271 		// determine the id getter and setter methods from the proxy interface (if any)
272 		// determine all interfaces needed by the resulting proxy
273 		HashSet<Class> proxyInterfaces = new HashSet<Class>();
274 		proxyInterfaces.add( HibernateProxy.class );
275 
276 		Class mappedClass = entityBinding.getEntity().getClassReference();
277 		Class proxyInterface = entityBinding.getProxyInterfaceType().getValue();
278 
279 		if ( proxyInterface!=null && !mappedClass.equals( proxyInterface ) ) {
280 			if ( ! proxyInterface.isInterface() ) {
281 				throw new MappingException(
282 						"proxy must be either an interface, or the class itself: " + getEntityName()
283 				);
284 			}
285 			proxyInterfaces.add( proxyInterface );
286 		}
287 
288 		if ( mappedClass.isInterface() ) {
289 			proxyInterfaces.add( mappedClass );
290 		}
291 
292 		for ( EntityBinding subEntityBinding : entityBinding.getPostOrderSubEntityBindingClosure() ) {
293 			final Class subclassProxy = subEntityBinding.getProxyInterfaceType().getValue();
294 			final Class subclassClass = subEntityBinding.getClassReference();
295 			if ( subclassProxy!=null && !subclassClass.equals( subclassProxy ) ) {
296 				if ( ! subclassProxy.isInterface() ) {
297 					throw new MappingException(
298 							"proxy must be either an interface, or the class itself: " + subEntityBinding.getEntity().getName()
299 					);
300 				}
301 				proxyInterfaces.add( subclassProxy );
302 			}
303 		}
304 
305 		for ( AttributeBinding property : entityBinding.attributeBindings() ) {
306 			Method method = getGetter( property ).getMethod();
307 			if ( method != null && Modifier.isFinal( method.getModifiers() ) ) {
308 				LOG.gettersOfLazyClassesCannotBeFinal(entityBinding.getEntity().getName(), property.getAttribute().getName());
309 			}
310 			method = getSetter( property ).getMethod();
311 			if ( method != null && Modifier.isFinal( method.getModifiers() ) ) {
312 				LOG.settersOfLazyClassesCannotBeFinal(entityBinding.getEntity().getName(), property.getAttribute().getName());
313 			}
314 		}
315 
316 		Method idGetterMethod = idGetter==null ? null : idGetter.getMethod();
317 		Method idSetterMethod = idSetter==null ? null : idSetter.getMethod();
318 
319 		Method proxyGetIdentifierMethod = idGetterMethod==null || proxyInterface==null ?
320 				null :
321 		        ReflectHelper.getMethod(proxyInterface, idGetterMethod);
322 		Method proxySetIdentifierMethod = idSetterMethod==null || proxyInterface==null  ?
323 				null :
324 		        ReflectHelper.getMethod(proxyInterface, idSetterMethod);
325 
326 		ProxyFactory pf = buildProxyFactoryInternal( entityBinding, idGetter, idSetter );
327 		try {
328 			pf.postInstantiate(
329 					getEntityName(),
330 					mappedClass,
331 					proxyInterfaces,
332 					proxyGetIdentifierMethod,
333 					proxySetIdentifierMethod,
334 					entityBinding.getHierarchyDetails().getEntityIdentifier().isEmbedded()
335 							? ( CompositeType ) entityBinding
336 									.getHierarchyDetails()
337 									.getEntityIdentifier()
338 									.getValueBinding()
339 									.getHibernateTypeDescriptor()
340 									.getResolvedTypeMapping()
341 							: null
342 			);
343 		}
344 		catch ( HibernateException he ) {
345 			LOG.unableToCreateProxyFactory(getEntityName(), he);
346 			pf = null;
347 		}
348 		return pf;
349 	}

505 	@Override
506     public void afterInitialize(Object entity, boolean lazyPropertiesAreUnfetched, SessionImplementor session) {
507 		if ( isInstrumented() ) {
508 			Set lazyProps = lazyPropertiesAreUnfetched && getEntityMetamodel().hasLazyProperties() ?
509 					lazyPropertyNames : null;
510 			//TODO: if we support multiple fetch groups, we would need
511 			//      to clone the set of lazy properties!
512 			FieldInterceptionHelper.injectFieldInterceptor( entity, getEntityName(), lazyProps, session );
513 
514             //also clear the fields that are marked as dirty in the dirtyness tracker
515             if(entity instanceof org.hibernate.engine.spi.SelfDirtinessTracker) {
516                 ((org.hibernate.engine.spi.SelfDirtinessTracker) entity).$$_hibernate_clearDirtyAttributes();
517             }
518 		}
519 	}

542 	/**
543 	 * {@inheritDoc}
544 	 */
545 	public String determineConcreteSubclassEntityName(Object entityInstance, SessionFactoryImplementor factory) {
546 		final Class concreteEntityClass = entityInstance.getClass();
547 		if ( concreteEntityClass == getMappedClass() ) {
548 			return getEntityName();
549 		}
550 		else {
551 			String entityName = getEntityMetamodel().findEntityNameByEntityClass( concreteEntityClass );
552 			if ( entityName == null ) {
553 				throw new HibernateException(
554 						"Unable to resolve entity name from Class [" + concreteEntityClass.getName() + "]"
555 								+ " expected instance/subclass of [" + getEntityName() + "]"
556 				);
557 			}
558 			return entityName;
559 		}
560 	}
