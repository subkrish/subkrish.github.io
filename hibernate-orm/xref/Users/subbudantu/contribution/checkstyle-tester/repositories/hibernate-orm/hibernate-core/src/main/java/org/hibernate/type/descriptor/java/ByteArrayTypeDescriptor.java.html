41  public class ByteArrayTypeDescriptor extends AbstractTypeDescriptor<Byte[]> {
42  	public static final ByteArrayTypeDescriptor INSTANCE = new ByteArrayTypeDescriptor();
43  
44  	@SuppressWarnings({ "unchecked" })
45  	public ByteArrayTypeDescriptor() {
46  		super( Byte[].class, ArrayMutabilityPlan.INSTANCE );
47  	}
48  
49  	@SuppressWarnings({ "UnnecessaryUnboxing" })
50  	public String toString(Byte[] bytes) {
51  		final StringBuilder buf = new StringBuilder();
52  		for ( Byte aByte : bytes ) {
53  			final String hexStr = Integer.toHexString( aByte.byteValue() - Byte.MIN_VALUE );
54  			if ( hexStr.length() == 1 ) {
55  				buf.append( '0' );
56  			}
57  			buf.append( hexStr );
58  		}
59  		return buf.toString();
60  	}
61  
62  	@SuppressWarnings({ "UnnecessaryBoxing" })
63  	public Byte[] fromString(String string) {
64  		if ( string == null ) {
65  			return null;
66  		}
67  		if ( string.length() % 2 != 0 ) {
68  			throw new IllegalArgumentException( "The string is not a valid string representation of a binary content." );
69  		}
70  		Byte[] bytes = new Byte[string.length() / 2];
71  		for ( int i = 0; i < bytes.length; i++ ) {
72  			final String hexStr = string.substring( i * 2, (i + 1) * 2 );
73  			bytes[i] = Byte.valueOf( (byte) (Integer.parseInt(hexStr, 16) + Byte.MIN_VALUE) );
74  		}
75  		return bytes;
76  	}
77  
78  	@SuppressWarnings({ "unchecked" })
79  	public <X> X unwrap(Byte[] value, Class<X> type, WrapperOptions options) {
80  		if ( value == null ) {
81  			return null;
82  		}
83  		if ( Byte[].class.isAssignableFrom( type ) ) {
84  			return (X) value;
85  		}
86  		if ( byte[].class.isAssignableFrom( type ) ) {
87  			return (X) unwrapBytes( value );
88  		}
89  		if ( InputStream.class.isAssignableFrom( type ) ) {
90  			return (X) new ByteArrayInputStream( unwrapBytes( value ) );
91  		}
92  		if ( BinaryStream.class.isAssignableFrom( type ) ) {
93  			return (X) new BinaryStreamImpl( unwrapBytes( value ) );
94  		}
95  		if ( Blob.class.isAssignableFrom( type ) ) {
96  			return (X) options.getLobCreator().createBlob( unwrapBytes( value ) );
97  		}
98  
99  		throw unknownUnwrap( type );
100 	}
101 
102 	public <X> Byte[] wrap(X value, WrapperOptions options) {
103 		if ( value == null ) {
104 			return null;
105 		}
106 		if ( Byte[].class.isInstance( value ) ) {
107 			return (Byte[]) value;
108 		}
109 		if ( byte[].class.isInstance( value ) ) {
110 			return wrapBytes( (byte[]) value );
111 		}
112 		if ( InputStream.class.isInstance( value ) ) {
113 			return wrapBytes( DataHelper.extractBytes( (InputStream) value ) );
114 		}
115 		if ( Blob.class.isInstance( value ) || DataHelper.isNClob( value.getClass() ) ) {
116 			try {
117 				return wrapBytes( DataHelper.extractBytes( ( (Blob) value ).getBinaryStream() ) );
118 			}
119 			catch ( SQLException e ) {
120 				throw new HibernateException( "Unable to access lob stream", e );
121 			}
122 		}
123 
124 		throw unknownWrap( value.getClass() );
125 	}
126 
127 	@SuppressWarnings({ "UnnecessaryBoxing" })
128 	private Byte[] wrapBytes(byte[] bytes) {
129 		if ( bytes == null ) {
130 			return null;
131 		}
132 		final Byte[] result = new Byte[bytes.length];
133 		for ( int i = 0; i < bytes.length; i++ ) {
134 			result[i] = Byte.valueOf( bytes[i] );
135 		}
136 		return result;
137 	}
138 
139 	@SuppressWarnings({ "UnnecessaryUnboxing" })
140 	private byte[] unwrapBytes(Byte[] bytes) {
141 		if ( bytes == null ) {
142 			return null;
143 		}
144 		final byte[] result = new byte[bytes.length];
145 		for ( int i = 0; i < bytes.length; i++ ) {
146 			result[i] = bytes[i].byteValue();
147 		}
148 		return result;
149 	}
