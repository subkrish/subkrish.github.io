36  	@Test
37  	public void testTypeDefinition() {
38  		Configuration cfg = configuration();
39  		PersistentClass pc = cfg.getClassMapping( EntityEnum.class.getName() );
40  
41  		// ordinal default of EnumType
42  		Type ordinalEnum = pc.getProperty( "ordinal" ).getType();
43  		assertEquals( Common.class, ordinalEnum.getReturnedClass() );
44  		assertEquals( EnumType.class.getName(), ordinalEnum.getName() );
45  
46  		// string defined by Enumerated(STRING)
47  		Type stringEnum = pc.getProperty( "string" ).getType();
48  		assertEquals( Common.class, stringEnum.getReturnedClass() );
49  		assertEquals( EnumType.class.getName(), stringEnum.getName() );
50  
51  		// explicit defined by @Type
52  		Type first = pc.getProperty( "firstLetter" ).getType();
53  		assertEquals( FirstLetter.class, first.getReturnedClass() );
54  		assertEquals( FirstLetterType.class.getName(), first.getName() );
55  
56  		// implicit defined by @TypeDef in somewhere
57  		Type last = pc.getProperty( "lastNumber" ).getType();
58  		assertEquals( LastNumber.class, last.getReturnedClass() );
59  		assertEquals( LastNumberType.class.getName(), last.getName() );
60  
61  		// implicit defined by @TypeDef in anywhere, but overrided by Enumerated(STRING)
62  		Type implicitOverrideExplicit = pc.getProperty( "explicitOverridingImplicit" ).getType();
63  		assertEquals( LastNumber.class, implicitOverrideExplicit.getReturnedClass() );
64  		assertEquals( EnumType.class.getName(), implicitOverrideExplicit.getName() );
65  	}

345 	@Test
346 	@TestForIssue(jiraKey = "HHH-4699")
347 	@SkipForDialect(value = Oracle8iDialect.class, jiraKey = "HHH-8516",
348 			comment = "HHH-4699 was specifically for using a CHAR, but Oracle does not handle the 2nd query correctly without VARCHAR. ")
349 	public void testTrimmedEnumChar() throws SQLException {
350 		// use native SQL to insert, forcing whitespace to occur
351 		final Session s = openSession();
352         final Connection connection = ((SessionImplementor)s).connection();
353         final Statement statement = connection.createStatement();
354         statement.execute("insert into EntityEnum (id, trimmed) values(1, '" + Trimmed.A.name() + "')");
355         statement.execute("insert into EntityEnum (id, trimmed) values(2, '" + Trimmed.B.name() + "')");
356 
357         s.getTransaction().begin();
358 
359         // ensure EnumType can do #fromName with the trimming
360         List<EntityEnum> resultList = s.createQuery("select e from EntityEnum e").list();
361         assertEquals( resultList.size(), 2 );
362         assertEquals( resultList.get(0).getTrimmed(), Trimmed.A );
363         assertEquals( resultList.get(1).getTrimmed(), Trimmed.B );
364 
365         // ensure querying works
366         final Query query = s.createQuery("select e from EntityEnum e where e.trimmed=?");
367         query.setParameter( 0, Trimmed.A );
368         resultList = query.list();
369         assertEquals( resultList.size(), 1 );
370         assertEquals( resultList.get(0).getTrimmed(), Trimmed.A );
371 
372         s.getTransaction().rollback();
373         s.close();
374 	}
