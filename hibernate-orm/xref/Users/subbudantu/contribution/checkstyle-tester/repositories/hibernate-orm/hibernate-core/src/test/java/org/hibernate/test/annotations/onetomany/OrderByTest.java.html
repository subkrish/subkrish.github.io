321 	@Test
322 	@TestForIssue(jiraKey = "HHH-5732")
323 	public void testInverseIndex() {
324 		final CollectionPersister transactionsPersister = sessionFactory().getCollectionPersister(
325 				BankAccount.class.getName() + ".transactions" );
326 		assertTrue( transactionsPersister.isInverse() );
327 
328 		Session s = openSession();
329 		s.getTransaction().begin();
330 
331 		BankAccount account = new BankAccount();
332 		account.addTransaction( "zzzzz" );
333 		account.addTransaction( "aaaaa" );
334 		account.addTransaction( "mmmmm" );
335 		s.save( account );
336 		s.getTransaction().commit();
337 
338 		s.close();
339 
340 		s = openSession();
341 		s.getTransaction().begin();
342 		
343 		try {
344 			final QueryableCollection queryableCollection = (QueryableCollection) transactionsPersister;
345 			SimpleSelect select = new SimpleSelect( getDialect() )
346 					.setTableName( queryableCollection.getTableName() )
347 					.addColumn( "code" )
348 					.addColumn( "transactions_index" );
349 			PreparedStatement preparedStatement = ((SessionImplementor)s).getTransactionCoordinator().getJdbcCoordinator().getStatementPreparer().prepareStatement( select.toStatementString() );
350 			ResultSet resultSet = ((SessionImplementor)s).getTransactionCoordinator().getJdbcCoordinator().getResultSetReturn().extract( preparedStatement );
351 			Map<Integer, String> valueMap = new HashMap<Integer, String>();
352 			while ( resultSet.next() ) {
353 				final String code = resultSet.getString( 1 );
354 				assertFalse( "code column was null", resultSet.wasNull() );
355 				final int indx = resultSet.getInt( 2 );
356 				assertFalse( "List index column was null", resultSet.wasNull() );
357 				valueMap.put( indx, code );
358 			}
359 			assertEquals( 3, valueMap.size() );
360 			assertEquals( "zzzzz", valueMap.get( 0 ) );
361 			assertEquals( "aaaaa", valueMap.get( 1 ) );
362 			assertEquals( "mmmmm", valueMap.get( 2 ) );
363 		}
364 		catch ( SQLException e ) {
365 			fail(e.getMessage());
366 		}
367 		finally {
368 			s.getTransaction().rollback();
369 			s.close();
370 		}
371 	}

394 		final User user = new User();
395 		user.setName( "john" );
396 		user.setForum( forum );
397 		forum.getUsers().add( user );
398 
399 		forum = (Forum) s.merge( forum );
400 
401 		s.flush();
402 		s.clear();
403 		sessionFactory().getCache().evictEntityRegions();
404 
405 		forum = (Forum) s.get( Forum.class, forum.getId() );
406 
407 		final Post post2 = new Post();
408 		post2.setName( "post2" );
409 		post2.setForum( forum );
410 		forum.getPosts().add( post2 );

413
