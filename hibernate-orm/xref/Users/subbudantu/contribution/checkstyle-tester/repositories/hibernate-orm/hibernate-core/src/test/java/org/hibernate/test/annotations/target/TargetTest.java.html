24  package org.hibernate.test.annotations.target;
25  
26  import org.junit.Test;
27  
28  import org.hibernate.Session;
29  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
30  
31  import static org.junit.Assert.assertEquals;
32  
33  /**
34   * @author Emmanuel Bernard
35   */
36  public class TargetTest extends BaseCoreFunctionalTestCase {
37  	@Test
38  	public void testTargetOnEmbedded() throws Exception {
39  		Session s = openSession();
40  		s.getTransaction().begin();
41  		Luggage l = new LuggageImpl();
42  		l.setHeight( 12 );
43  		l.setWidth( 12 );
44  		Owner o = new OwnerImpl();
45  		o.setName( "Emmanuel" );
46  		l.setOwner( o );
47  		s.persist( l );
48  		s.flush();
49  		s.clear();
50  		l = (Luggage) s.get(LuggageImpl.class, ( (LuggageImpl) l).getId() );
51  		assertEquals( "Emmanuel", l.getOwner().getName() );
52  		s.getTransaction().rollback();
53  		s.close();
54  	}
55  
56  	@Test
57  	public void testTargetOnMapKey() throws Exception {
58  		Session s = openSession();
59  		s.getTransaction().begin();
60  		Luggage l = new LuggageImpl();
61  		l.setHeight( 12 );
62  		l.setWidth( 12 );
63  		Size size = new SizeImpl();
64  		size.setName( "S" );
65  		Owner o = new OwnerImpl();
66  		o.setName( "Emmanuel" );
67  		l.setOwner( o );
68  		s.persist( l );
69  		Brand b = new Brand();
70  		s.persist( b );
71  		b.getLuggagesBySize().put( size, l );
72  		s.flush();
73  		s.clear();
74  		b = (Brand) s.get(Brand.class, b.getId() );
75  		assertEquals( "S", b.getLuggagesBySize().keySet().iterator().next().getName() );
76  		s.getTransaction().rollback();
77  		s.close();
78  	}
79  
80  	@Test
81  	public void testTargetOnMapKeyManyToMany() throws Exception {
82  		Session s = openSession();
83  		s.getTransaction().begin();
84  		Luggage l = new LuggageImpl();
85  		l.setHeight( 12 );
86  		l.setWidth( 12 );
87  		Size size = new SizeImpl();
88  		size.setName( "S" );
89  		Owner o = new OwnerImpl();
90  		o.setName( "Emmanuel" );
91  		l.setOwner( o );
92  		s.persist( l );
93  		Brand b = new Brand();
94  		s.persist( b );
95  		b.getSizePerLuggage().put( l, size );
96  		s.flush();
97  		s.clear();
98  		b = (Brand) s.get(Brand.class, b.getId() );
99  		assertEquals( 12d, b.getSizePerLuggage().keySet().iterator().next().getWidth(), 0.01 );
100 		s.getTransaction().rollback();
101 		s.close();
102 	}
103 
104 	@Override
105 	protected Class[] getAnnotatedClasses() {
106 		return new Class[] { LuggageImpl.class, Brand.class };
107 	}
