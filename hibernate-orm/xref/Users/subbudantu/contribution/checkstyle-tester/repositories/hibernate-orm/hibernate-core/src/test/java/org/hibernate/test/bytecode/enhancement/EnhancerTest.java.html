128 	private void testFor(Class entityClassToEnhance) throws Exception {
129 		Enhancer enhancer = new Enhancer( enhancementContext );
130 		CtClass entityCtClass = generateCtClassForAnEntity( entityClassToEnhance );
131 		byte[] original = entityCtClass.toBytecode();
132 		//byte[] enhanced = enhancer.enhance( entityCtClass.getName(), original );
133         byte[] enhanced = enhancer.enhance( entityCtClass.getName(), original );
134 		assertFalse( "entity was not enhanced", Arrays.equals( original, enhanced ) );
135 
136 		ClassLoader cl = new ClassLoader() { };
137 		ClassPool cp = new ClassPool( false );
138 		cp.appendClassPath( new LoaderClassPath( cl ) );
139 		CtClass enhancedCtClass = cp.makeClass( new ByteArrayInputStream( enhanced ) );
140         enhancedCtClass.debugWriteFile("/tmp");
141         //just for debugging
142         Class addressClass = null;
143         Class countryClass = null;
144         if(entityClassToEnhance.getName().endsWith("SimpleEntity")) {
145             CtClass addressCtClass = generateCtClassForAnEntity( Address.class );
146             byte[] enhancedAddress = enhancer.enhanceComposite(Address.class.getName(), addressCtClass.toBytecode());
147             CtClass enhancedCtClassAddress = cp.makeClass( new ByteArrayInputStream( enhancedAddress ) );
148             enhancedCtClassAddress.debugWriteFile("/tmp");
149             addressClass = enhancedCtClassAddress.toClass( cl, this.getClass().getProtectionDomain() );
150 
151             CtClass countryCtClass = generateCtClassForAnEntity( Country.class );
152             byte[] enhancedCountry = enhancer.enhanceComposite(Country.class.getName(), countryCtClass.toBytecode());
153             CtClass enhancedCtClassCountry = cp.makeClass( new ByteArrayInputStream( enhancedCountry ) );
154             enhancedCtClassCountry.debugWriteFile("/tmp");
155             countryClass = enhancedCtClassCountry.toClass( cl, this.getClass().getProtectionDomain() );
156 
157         }
158 		Class entityClass = enhancedCtClass.toClass( cl, this.getClass().getProtectionDomain() );
159 		Object entityInstance = entityClass.newInstance();
160 
161 		assertTyping( ManagedEntity.class, entityInstance );
162 
163 		// call the new methods
164 		//
165 		Method setter = entityClass.getMethod( EnhancerConstants.ENTITY_ENTRY_SETTER_NAME, EntityEntry.class );
166 		Method getter = entityClass.getMethod( EnhancerConstants.ENTITY_ENTRY_GETTER_NAME );
167 		assertNull( getter.invoke( entityInstance ) );
168 		setter.invoke( entityInstance, makeEntityEntry() );
169 		assertNotNull( getter.invoke( entityInstance ) );
170 		setter.invoke( entityInstance, new Object[] {null} );
171 		assertNull( getter.invoke( entityInstance ) );
172 
173 		Method entityInstanceGetter = entityClass.getMethod( EnhancerConstants.ENTITY_INSTANCE_GETTER_NAME );
174 		assertSame( entityInstance, entityInstanceGetter.invoke( entityInstance ) );
175 
176 		Method previousGetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_GETTER_NAME );
177 		Method previousSetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_SETTER_NAME, ManagedEntity.class );
178 		previousSetter.invoke( entityInstance, entityInstance );
179 		assertSame( entityInstance, previousGetter.invoke( entityInstance ) );
180 
181 		Method nextGetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_GETTER_NAME );
182 		Method nextSetter = entityClass.getMethod( EnhancerConstants.PREVIOUS_SETTER_NAME, ManagedEntity.class );
183 		nextSetter.invoke( entityInstance, entityInstance );
184 		assertSame( entityInstance, nextGetter.invoke( entityInstance ) );
185 
186 		// add an attribute interceptor...
187 		Method interceptorGetter = entityClass.getMethod( EnhancerConstants.INTERCEPTOR_GETTER_NAME );
188 		Method interceptorSetter = entityClass.getMethod( EnhancerConstants.INTERCEPTOR_SETTER_NAME, PersistentAttributeInterceptor.class );
189 
190 		assertNull( interceptorGetter.invoke( entityInstance ) );
191 		entityClass.getMethod( "getId" ).invoke( entityInstance );
192 
193 		interceptorSetter.invoke( entityInstance, new LocalPersistentAttributeInterceptor() );
194 		assertNotNull(interceptorGetter.invoke(entityInstance));
195 
196 		// dirty checking is unfortunately just printlns for now... just verify the test output
197 		entityClass.getMethod( "getId" ).invoke( entityInstance );
198 		entityClass.getMethod( "setId", Long.class ).invoke( entityInstance, entityClass.getMethod( "getId" ).invoke( entityInstance ) );
199 		entityClass.getMethod( "setId", Long.class ).invoke( entityInstance, 1L );
200         assertTrue((Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance));
201 
202 		entityClass.getMethod( "isActive" ).invoke( entityInstance );
203 		entityClass.getMethod( "setActive", boolean.class ).invoke( entityInstance, entityClass.getMethod( "isActive" ).invoke( entityInstance ) );
204 		entityClass.getMethod( "setActive", boolean.class ).invoke(entityInstance, true);
205 
206 		entityClass.getMethod( "getSomeNumber" ).invoke( entityInstance );
207 		entityClass.getMethod( "setSomeNumber", long.class ).invoke( entityInstance, entityClass.getMethod( "getSomeNumber" ).invoke( entityInstance ) );
208 		entityClass.getMethod( "setSomeNumber", long.class ).invoke(entityInstance, 1L);
209         assertEquals(3, ((Set) entityClass.getMethod("$$_hibernate_getDirtyAttributes").invoke(entityInstance)).size());
210 
211         entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
212         //setting the same value should not make it dirty
213         entityClass.getMethod( "setSomeNumber", long.class ).invoke(entityInstance, 1L);
214         //assertEquals(0, ((Set) entityClass.getMethod("$$_hibernate_getDirtyAttributes").invoke(entityInstance)).size());
215         //assertFalse((Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance));
216 
217         if(entityClass.getName().endsWith("SimpleEntity")) {
218 
219             List<String> strings = new ArrayList<String>();
220             strings.add("FooBar");
221 
222             entityClass.getMethod( "setSomeStrings", java.util.List.class ).invoke(entityInstance, strings);
223 
224             assertTrue((Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance));
225 
226             Set tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
227             assertEquals(1, tracked.size());
228             assertEquals("someStrings", tracked.iterator().next());
229 
230             entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
231 
232             ((List) entityClass.getMethod( "getSomeStrings").invoke(entityInstance)).add("JADA!");
233             Boolean isDirty = (Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance);
234             assertTrue(isDirty);
235             assertEquals("someStrings",
236                     ((Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance)).iterator().next());
237             entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
238 
239             //this should not set the entity to dirty
240             Set<Integer> ints = new HashSet<Integer>();
241             ints.add(42);
242             entityClass.getMethod( "setSomeInts", java.util.Set.class ).invoke(entityInstance, ints);
243             isDirty = (Boolean) entityClass.getMethod("$$_hibernate_hasDirtyAttributes").invoke(entityInstance);
244             assertFalse(isDirty);
245 
246             //testing composite object
247             assert addressClass != null;
248             Object address =  addressClass.newInstance();
249 
250             assert countryClass != null;
251             Object country =  countryClass.newInstance();
252 
253             //Method adrInterceptorGetter = addressClass.getMethod( EnhancerConstants.INTERCEPTOR_GETTER_NAME );
254             //Method adrInterceptorSetter = addressClass.getMethod( EnhancerConstants.INTERCEPTOR_SETTER_NAME, PersistentAttributeInterceptor.class );
255             //adrInterceptorSetter.invoke( address, new LocalPersistentAttributeInterceptor() );
256 
257             entityClass.getMethod("setAddress", addressClass).invoke(entityInstance, address);
258 
259             addressClass.getMethod("setCity", String.class).invoke(address, "Arendal");
260 
261             tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
262             assertEquals(2, tracked.size());
263             Iterator iter = tracked.iterator();
264             assertEquals("address", iter.next());
265             assertEquals("address.city", iter.next());
266 
267             entityClass.getMethod("$$_hibernate_clearDirtyAttributes").invoke(entityInstance);
268 
269             //make sure that new composite instances are cleared
270             Object address2 =  addressClass.newInstance();
271 
272             entityClass.getMethod("setAddress", addressClass).invoke(entityInstance, address2);
273             addressClass.getMethod("setStreet1", String.class).invoke(address, "Heggedalveien");
274 
275             tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
276             assertEquals(1, tracked.size());
277 
278             addressClass.getMethod("setCountry", countryClass).invoke(address2, country);
279             countryClass.getMethod("setName", String.class).invoke(country, "Norway");
280 
281             tracked = (Set) entityClass.getMethod(EnhancerConstants.TRACKER_GET_NAME).invoke(entityInstance);
282             assertEquals(3, tracked.size());
283         }
284 	}

286 	private CtClass generateCtClassForAnEntity(Class entityClassToEnhance) throws Exception {
287 		ClassPool cp = new ClassPool( false );
288 		return cp.makeClass(
289 				getClass().getClassLoader().getResourceAsStream(
290 						entityClassToEnhance.getName().replace( '.', '/' ) + ".class"
291 				)
292 		);
293 	}
