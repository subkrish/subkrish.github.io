52  	@Override
53  	public void afterConfigurationBuilt(Mappings mappings, Dialect dialect) {
54  		super.afterConfigurationBuilt( mappings, dialect );
55  		Collection children = mappings.getCollection( Parent.class.getName() + ".children" );
56  		Component childComponents = ( Component ) children.getElement();
57  		Formula f = ( Formula ) childComponents.getProperty( "bioLength" ).getValue().getColumnIterator().next();
58  
59  		SQLFunction lengthFunction = ( SQLFunction ) dialect.getFunctions().get( "length" );
60  		if ( lengthFunction != null ) {
61  			ArrayList args = new ArrayList();
62  			args.add( "bio" );
63  			f.setFormula( lengthFunction.render( StandardBasicTypes.INTEGER, args, null ) );
64  		}
65  	}

67  	@Test
68  	public void testHandSQL() {
69  		Session s = openSession();
70  		Transaction t = s.beginTransaction();
71  		Child c = new Child( "Child One" );
72  		Parent p = new Parent( "Parent" );
73  		p.getChildren().add( c );
74  		c.setParent( p );
75  		s.save( p );
76  		s.flush();
77  
78  		p.getChildren().remove( c );
79  		c.setParent( null );
80  		s.flush();
81  
82  		p.getChildren().add( c );
83  		c.setParent( p );
84  		t.commit();
85  		s.close();
86  
87  		s = openSession();
88  		t = s.beginTransaction();
89  		s.createQuery( "select distinct p from Parent p join p.children c where c.name like 'Child%'" ).uniqueResult();
90  		s.clear();
91  		s.createQuery( "select new Child(c.name) from Parent p left outer join p.children c where c.name like 'Child%'" )

107 	@Test
108 	public void testCustomColumnReadAndWrite() {
109 		Session s = openSession();
110 		Transaction t = s.beginTransaction();
111 		Child c = new Child( "Child One" );
112 		c.setPosition( 1 );
113 		Parent p = new Parent( "Parent" );
114 		p.getChildren().add( c );
115 		c.setParent( p );
116 		s.save( p );
117 		s.flush();
118 
119 		// Oracle returns BigDecimaal while other dialects return Integer;
120 		// casting to Number so it works on all dialects
121 		Number sqlValue = ((Number) s.createSQLQuery("select child_position from ParentChild c where c.name='Child One'")
122 				.uniqueResult());
123 		assertEquals( 0, sqlValue.intValue() );
124 
125 		Integer hqlValue = (Integer)s.createQuery("select c.position from Parent p join p.children c where p.name='Parent'")
126 				.uniqueResult();
127 		assertEquals( 1, hqlValue.intValue() );
128 
129 		p = (Parent)s.createCriteria(Parent.class).add(Restrictions.eq("name", "Parent")).uniqueResult();
130 		c = (Child)p.getChildren().iterator().next();
131 		assertEquals( 1, c.getPosition() );
132 
133 		p = (Parent)s.createQuery("from Parent p join p.children c where c.position = 1").uniqueResult();
134 		c = (Child)p.getChildren().iterator().next();
135 		assertEquals( 1, c.getPosition() );
136 
137 		c.setPosition( 2 );
138 		s.flush();
139 		sqlValue = ( (Number) s.createSQLQuery("select child_position from ParentChild c where c.name='Child One'")
140 				.uniqueResult() );
141 		assertEquals( 1, sqlValue.intValue() );
142 		s.delete( p );
143 		t.commit();
144 		s.close();
145 	}
