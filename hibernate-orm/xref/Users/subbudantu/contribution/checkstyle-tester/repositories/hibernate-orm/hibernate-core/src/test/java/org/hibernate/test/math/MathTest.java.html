21  package org.hibernate.test.math;
22  
23  import static org.junit.Assert.assertEquals;
24  
25  import org.hibernate.Session;
26  import org.hibernate.dialect.H2Dialect;
27  import org.hibernate.dialect.Oracle8iDialect;
28  import org.hibernate.testing.RequiresDialect;
29  import org.hibernate.testing.junit4.BaseCoreFunctionalTestCase;
30  import org.junit.Test;
31  
32  /**
33   * @author Brett Meyer
34   */
35  @RequiresDialect( value = { Oracle8iDialect.class, H2Dialect.class } )
36  public class MathTest extends BaseCoreFunctionalTestCase {
37  
38  	@Override
39  	public String[] getMappings() {
40  		return new String[]{"math/Math.hbm.xml"};
41  	}
42  	
43  	@Test
44  	public void testBitAnd() {
45  		MathEntity me = new MathEntity();
46  		me.setValue( 5 );
47  		
48  		Session s = openSession();
49  		s.beginTransaction();
50  		Long id = (Long) s.save( me );
51  		s.getTransaction().commit();
52  		s.close();
53  		
54  		s = openSession();
55  		s.beginTransaction();
56  		int value1 = ((Integer) s.createQuery( "select bitand(m.value,0) from MathEntity m where m.id=" + id ).uniqueResult()).intValue();
57  		int value2 = ((Integer) s.createQuery( "select bitand(m.value,2) from MathEntity m where m.id=" + id ).uniqueResult()).intValue();
58  		int value3 = ((Integer )s.createQuery( "select bitand(m.value,3) from MathEntity m where m.id=" + id ).uniqueResult()).intValue();
59  		s.getTransaction().commit();
60  		s.close();
61  
62  		assertEquals(value1, 0);
63  		assertEquals(value2, 0);
64  		assertEquals(value3, 1);
65  	}
66  
67  }
