53  	@Test
54  	public void testSerialization() throws Exception {
55  		ByteArrayOutputStream stream = new ByteArrayOutputStream();
56  		ObjectOutput out = new ObjectOutputStream( stream );
57  		out.writeObject( entityManagerFactory() );
58  		out.close();
59  		byte[] serialized = stream.toByteArray();
60  		stream.close();
61  		ByteArrayInputStream byteIn = new ByteArrayInputStream( serialized );
62  		ObjectInputStream in = new ObjectInputStream( byteIn );
63  		EntityManagerFactory serializedFactory = (EntityManagerFactory) in.readObject();
64  		in.close();
65  		byteIn.close();
66  		EntityManager em = serializedFactory.createEntityManager();
67  		//em.getTransaction().begin();
68  		//em.setFlushMode( FlushModeType.NEVER );
69  		Cat cat = new Cat();
70  		cat.setAge( 3 );
71  		cat.setDateOfBirth( new Date() );
72  		cat.setLength( 22 );
73  		cat.setName( "Kitty" );
74  		em.persist( cat );
75  		Item item = new Item();
76  		item.setName( "Train Ticket" );
77  		item.setDescr( "Paris-London" );
78  		em.persist( item );
79  		//em.getTransaction().commit();
80  		//em.getTransaction().begin();
81  		item.setDescr( "Paris-Bruxelles" );
82  		//em.getTransaction().commit();
83  
84  		//fake the in container work
85  		( (HibernateEntityManager) em ).getSession().disconnect();
86  		stream = new ByteArrayOutputStream();
87  		out = new ObjectOutputStream( stream );
88  		out.writeObject( em );
89  		out.close();
90  		serialized = stream.toByteArray();
91  		stream.close();
92  		byteIn = new ByteArrayInputStream( serialized );
93  		in = new ObjectInputStream( byteIn );
94  		em = (EntityManager) in.readObject();
95  		in.close();
96  		byteIn.close();
97  		//fake the in container work
98  		em.getTransaction().begin();
99  		item = em.find( Item.class, item.getName() );
100 		item.setDescr( item.getDescr() + "-Amsterdam" );
101 		cat = (Cat) em.createQuery( "select c from " + Cat.class.getName() + " c" ).getSingleResult();
102 		cat.setLength( 34 );
103 		em.flush();
104 		em.remove( item );
105 		em.remove( cat );
106 		em.flush();
107 		em.getTransaction().commit();
108 
109 		em.close();
110 	}

112 	@Test
113 	public void testEntityManagerFactorySerialization() throws Exception {
114 		EntityManagerFactory entityManagerFactory = entityManagerFactory();
115 
116 		ByteArrayOutputStream stream = new ByteArrayOutputStream();
117 		ObjectOutput out = new ObjectOutputStream( stream );
118 		out.writeObject( entityManagerFactory );
119 		out.close();
120 		byte[] serialized = stream.toByteArray();
121 		stream.close();
122 		ByteArrayInputStream byteIn = new ByteArrayInputStream( serialized );
123 		ObjectInputStream in = new ObjectInputStream( byteIn );
124 		EntityManagerFactory entityManagerFactory2 = (EntityManagerFactory) in.readObject();
125 		in.close();
126 		byteIn.close();
127 
128 		assertTrue("deserialized EntityManagerFactory should be the same original EntityManagerFactory instance",
129 				entityManagerFactory2 == entityManagerFactory);
130 	}

132 	@Test
133 	public void testEntityManagerFactoryProperties() {
134 		EntityManagerFactory entityManagerFactory = entityManagerFactory();
135 		assertTrue( entityManagerFactory.getProperties().containsKey( AvailableSettings.USER ) );
136 		if ( entityManagerFactory.getProperties().containsKey( AvailableSettings.PASS ) ) {
137 			assertEquals( "****",  entityManagerFactory.getProperties().get( AvailableSettings.PASS ) );
138 		}
139 	}
