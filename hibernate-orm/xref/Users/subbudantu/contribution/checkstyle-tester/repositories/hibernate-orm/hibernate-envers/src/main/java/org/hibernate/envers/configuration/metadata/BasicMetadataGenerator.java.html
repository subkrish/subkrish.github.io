32  import org.hibernate.mapping.SimpleValue;
33  import org.hibernate.mapping.Value;
34  import org.hibernate.type.BasicType;
35  import org.hibernate.type.CustomType;
36  import org.hibernate.type.EnumType;
37  import org.hibernate.type.SerializableToBlobType;
38  import org.hibernate.type.Type;
39  import org.hibernate.usertype.DynamicParameterizedType;

41  /**
42   * Generates metadata for basic properties: immutable types (including enums).
43   * @author Adam Warski (adam at warski dot org)
44   */
45  public final class BasicMetadataGenerator {
46  	@SuppressWarnings({ "unchecked" })
47  	boolean addBasic(Element parent, PropertyAuditingData propertyAuditingData,
48  					 Value value, SimpleMapperBuilder mapper, boolean insertable, boolean key) {
49  		Type type = value.getType();
50  
51  		if ( type instanceof BasicType || type instanceof SerializableToBlobType ||
52  				"org.hibernate.type.PrimitiveByteArrayBlobType".equals( type.getClass().getName() ) ) {
53  			if ( parent != null ) {
54  				boolean addNestedType = ( value instanceof SimpleValue ) && ( (SimpleValue) value ).getTypeParameters() != null;
55  
56  				String typeName = type.getName();
57  				if ( typeName == null ) {
58  					typeName = type.getClass().getName();
59  				}
60  
61  				Element prop_mapping = MetadataTools.addProperty(
62  						parent, propertyAuditingData.getName(),
63  						addNestedType ? null : typeName, propertyAuditingData.isForceInsertable() || insertable, key
64  				);
65  				MetadataTools.addColumns( prop_mapping, value.getColumnIterator() );
66  
67  				if ( addNestedType ) {
68  					Properties typeParameters = ( (SimpleValue) value ).getTypeParameters();
69  					Element type_mapping = prop_mapping.addElement( "type" );
70  					type_mapping.addAttribute( "name", typeName );
71  
72  					if ( "org.hibernate.type.EnumType".equals( typeName ) ) {
73  						// Proper handling of enumeration type
74  						mapEnumerationType( type_mapping, type, typeParameters );
75  					}
76  					else {
77  						// By default copying all Hibernate properties
78  						for ( Object object : typeParameters.keySet() ) {
79  							String keyType = (String) object;
80  							String property = typeParameters.getProperty( keyType );
81  
82  							if ( property != null ) {
83  								type_mapping.addElement( "param" ).addAttribute( "name", keyType ).setText( property );
84  							}
85  						}
86  					}
87  				}
88  			}
89  
90  			// A null mapper means that we only want to add xml mappings
91  			if ( mapper != null ) {
92  				mapper.add( propertyAuditingData.getPropertyData() );
93  			}
94  		}
95  		else {
96  			return false;
97  		}
98  
99  		return true;
100 	}
101 
102 	private void mapEnumerationType(Element parent, Type type, Properties parameters) {
103 		if ( parameters.getProperty( EnumType.ENUM ) != null ) {
104 			parent.addElement( "param" ).addAttribute( "name", EnumType.ENUM ).setText( parameters.getProperty( EnumType.ENUM ) );
105 		}
106 		else {
107 			parent.addElement( "param" ).addAttribute( "name", EnumType.ENUM ).setText( type.getReturnedClass().getName() );
108 		}
109 		if ( parameters.getProperty( EnumType.NAMED ) != null ) {
110 			parent.addElement( "param" ).addAttribute( "name", EnumType.NAMED ).setText( parameters.getProperty( EnumType.NAMED ) );
111 		}
112 		else if ( parameters.get( DynamicParameterizedType.XPROPERTY ) != null ) {
113 			// Case of annotations.
114 			parent.addElement( "param" ).addAttribute( "name", EnumType.NAMED )
115 										.setText( "" + !( (EnumType) ( (CustomType) type ).getUserType() ).isOrdinal() );
116 		}
117 		else {
118 			// Otherwise we assume that the choice between ordinal and named representation has been omitted.
119 			// Call to EnumType#isOrdinal() would always return the default Types.INTEGER. We let Hibernate
120 			// to choose the proper strategy during runtime.
121 		}
122 	}
123 
124 	@SuppressWarnings({ "unchecked" })
125 	boolean addManyToOne(Element parent, PropertyAuditingData propertyAuditingData, Value value, SimpleMapperBuilder mapper) {
126 		Type type = value.getType();
127 
128 		// A null mapper occurs when adding to composite-id element
129 		Element manyToOneElement = parent.addElement( mapper != null ? "many-to-one" : "key-many-to-one" );
130 		manyToOneElement.addAttribute( "name", propertyAuditingData.getName() );
131 		manyToOneElement.addAttribute( "class", type.getName() );
132 		MetadataTools.addColumns( manyToOneElement, value.getColumnIterator() );
133 
134 		// A null mapper means that we only want to add xml mappings
135 		if ( mapper != null ) {
136 			mapper.add( propertyAuditingData.getPropertyData() );
137 		}
138 
139 		return true;
140 	}
