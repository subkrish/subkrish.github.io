125 	/**
126      * @param mainGenerator Main generator, giving access to configuration and the basic mapper.
127      * @param propertyValue Value of the collection, as mapped by Hibernate.
128      * @param currentMapper Mapper, to which the appropriate {@link org.hibernate.envers.entities.mapper.PropertyMapper}
129      * will be added.

131      * @param xmlMappingData In case this collection requires a middle table, additional mapping documents will
132      * be created using this object.

137     public CollectionMetadataGenerator(AuditMetadataGenerator mainGenerator,
138                                        Collection propertyValue, CompositeMapperBuilder currentMapper,
139                                        String referencingEntityName, EntityXmlMappingData xmlMappingData,
140                                        PropertyAuditingData propertyAuditingData) {
141         this.mainGenerator = mainGenerator;
142         this.propertyValue = propertyValue;
143         this.currentMapper = currentMapper;
144         this.referencingEntityName = referencingEntityName;
145         this.xmlMappingData = xmlMappingData;
146         this.propertyAuditingData = propertyAuditingData;
147 
148         this.propertyName = propertyAuditingData.getName();
149 
150         referencingEntityConfiguration = mainGenerator.getEntitiesConfigurations().get(referencingEntityName);
151         if (referencingEntityConfiguration == null) {
152             throw new MappingException("Unable to read auditing configuration for " + referencingEntityName + "!");
153         }
154 
155         referencedEntityName = MappingTools.getReferencedEntityName(propertyValue.getElement());
156     }

158     void addCollection() {
159         Type type = propertyValue.getType();
160         Value value = propertyValue.getElement();
161 
162         boolean oneToManyAttachedType = type instanceof BagType || type instanceof SetType || type instanceof MapType || type instanceof ListType;
163         boolean inverseOneToMany = (value instanceof OneToMany) && (propertyValue.isInverse());
164         boolean owningManyToOneWithJoinTableBidirectional = (value instanceof ManyToOne) && (propertyAuditingData.getRelationMappedBy() != null);
165         boolean fakeOneToManyBidirectional = (value instanceof OneToMany) && (propertyAuditingData.getAuditMappedBy() != null);
166 
167         if (oneToManyAttachedType && (inverseOneToMany || fakeOneToManyBidirectional || owningManyToOneWithJoinTableBidirectional)) {
168             // A one-to-many relation mapped using @ManyToOne and @OneToMany(mappedBy="...")
169             addOneToManyAttached(fakeOneToManyBidirectional);
170         } else {
171             // All other kinds of relations require a middle (join) table.
172             addWithMiddleTable();
173         }
174     }

178                 mainGenerator.getEntitiesConfigurations().containsKey(entityName));

181     @SuppressWarnings({"unchecked"})
182     private void addOneToManyAttached(boolean fakeOneToManyBidirectional) {
183         LOG.debugf("Adding audit mapping for property %s.%s: one-to-many collection, using a join column on the referenced entity",
184                    referencingEntityName,
185                    propertyName);
186 
187         String mappedBy = getMappedBy(propertyValue);
188 
189         IdMappingData referencedIdMapping = mainGenerator.getReferencedIdMappingData(referencingEntityName,
190                     referencedEntityName, propertyAuditingData, false);
191         IdMappingData referencingIdMapping = referencingEntityConfiguration.getIdMappingData();
192 
193         // Generating the id mappers data for the referencing side of the relation.
194         MiddleIdData referencingIdData = createMiddleIdData(referencingIdMapping,
195                 mappedBy + "_", referencingEntityName);
196 
197         // And for the referenced side. The prefixed mapper won't be used (as this collection isn't persisted
198         // in a join table, so the prefix value is arbitrary).
199         MiddleIdData referencedIdData = createMiddleIdData(referencedIdMapping,
200                 null, referencedEntityName);
201 
202         // Generating the element mapping.
203         MiddleComponentData elementComponentData = new MiddleComponentData(
204                 new MiddleRelatedComponentMapper(referencedIdData), 0);
205 
206         // Generating the index mapping, if an index exists. It can only exists in case a javax.persistence.MapKey
207         // annotation is present on the entity. So the middleEntityXml will be not be used. The queryGeneratorBuilder
208         // will only be checked for nullnes.
209         MiddleComponentData indexComponentData = addIndex(null, null);
210 
211         // Generating the query generator - it should read directly from the related entity.
212         RelationQueryGenerator queryGenerator = new OneAuditEntityQueryGenerator(mainGenerator.getGlobalCfg(),
213                 mainGenerator.getVerEntCfg(), mainGenerator.getAuditStrategy(),
214                 referencingIdData, referencedEntityName, referencedIdData, isEmbeddableElementType());
215 
216         // Creating common mapper data.
217         CommonCollectionMapperData commonCollectionMapperData = new CommonCollectionMapperData(
218                 mainGenerator.getVerEntCfg(), referencedEntityName,
219                 propertyAuditingData.getPropertyData(),
220                 referencingIdData, queryGenerator);
221 
222         PropertyMapper fakeBidirectionalRelationMapper;
223         PropertyMapper fakeBidirectionalRelationIndexMapper;
224         if (fakeOneToManyBidirectional) {
225             // In case of a fake many-to-one bidirectional relation, we have to generate a mapper which maps
226             // the mapped-by property name to the id of the related entity (which is the owner of the collection).
227             String auditMappedBy = propertyAuditingData.getAuditMappedBy();
228 
229             // Creating a prefixed relation mapper.
230             IdMapper relMapper = referencingIdMapping.getIdMapper().prefixMappedProperties(
231                     MappingTools.createToOneRelationPrefix(auditMappedBy));
232 
233             fakeBidirectionalRelationMapper = new ToOneIdMapper(
234                     relMapper,
235                     // The mapper will only be used to map from entity to map, so no need to provide other details
236                     // when constructing the PropertyData.
237                     new PropertyData(auditMappedBy, null, null, null),
238                     referencingEntityName, false);
239 
240             // Checking if there's an index defined. If so, adding a mapper for it.
241             if (propertyAuditingData.getPositionMappedBy() != null) {
242                 String positionMappedBy = propertyAuditingData.getPositionMappedBy();
243                 fakeBidirectionalRelationIndexMapper = new SinglePropertyMapper(new PropertyData(positionMappedBy, null, null, null));
244 
245                 // Also, overwriting the index component data to properly read the index.
246                 indexComponentData = new MiddleComponentData(new MiddleStraightComponentMapper(positionMappedBy), 0);
247             } else {
248                 fakeBidirectionalRelationIndexMapper = null;
249             }
250         } else {
251             fakeBidirectionalRelationMapper = null;
252             fakeBidirectionalRelationIndexMapper = null;
253         }
254 
255         // Checking the type of the collection and adding an appropriate mapper.
256         addMapper(commonCollectionMapperData, elementComponentData, indexComponentData);
257 
258         // Storing information about this relation.
259         referencingEntityConfiguration.addToManyNotOwningRelation(propertyName, mappedBy,
260                 referencedEntityName, referencingIdData.getPrefixedMapper(), fakeBidirectionalRelationMapper,
261                 fakeBidirectionalRelationIndexMapper);
262     }

264     /**
265      * Adds mapping of the id of a related entity to the given xml mapping, prefixing the id with the given prefix.
266      * @param xmlMapping Mapping, to which to add the xml.
267      * @param prefix Prefix for the names of properties which will be prepended to properties that form the id.
268      * @param columnNameIterator Iterator over the column names that will be used for properties that form the id.
269      * @param relatedIdMapping Id mapping data of the related entity.
270      */
271     @SuppressWarnings({"unchecked"})
272     private void addRelatedToXmlMapping(Element xmlMapping, String prefix,
273                                         MetadataTools.ColumnNameIterator columnNameIterator,
274                                         IdMappingData relatedIdMapping) {
275         Element properties = (Element) relatedIdMapping.getXmlRelationMapping().clone();
276         MetadataTools.prefixNamesInPropertyElement(properties, prefix, columnNameIterator, true, true);
277         for (Element idProperty : (java.util.List<Element>) properties.elements()) {
278             xmlMapping.add((Element) idProperty.clone());
279         }
280     }

282     private String getMiddleTableName(Collection value, String entityName) {
283         // We check how Hibernate maps the collection.
284         if (value.getElement() instanceof OneToMany && !value.isInverse()) {
285             // This must be a @JoinColumn+@OneToMany mapping. Generating the table name, as Hibernate doesn't use a
286             // middle table for mapping this relation.
287             return StringTools.getLastComponent(entityName) + "_" + StringTools.getLastComponent(MappingTools.getReferencedEntityName(value.getElement()));
288         }
289         // Hibernate uses a middle table for mapping this relation, so we get it's name directly.
290         return value.getCollectionTable().getName();
291     }

293     @SuppressWarnings({"unchecked"})
294     private void addWithMiddleTable() {
295 
296         LOG.debugf("Adding audit mapping for property %s.%s: collection with a join table", referencingEntityName, propertyName);
297 
298         // Generating the name of the middle table
299         String auditMiddleTableName;
300         String auditMiddleEntityName;
301         if (!StringTools.isEmpty(propertyAuditingData.getJoinTable().name())) {
302             auditMiddleTableName = propertyAuditingData.getJoinTable().name();
303             auditMiddleEntityName = propertyAuditingData.getJoinTable().name();
304         } else {
305             String middleTableName = getMiddleTableName(propertyValue, referencingEntityName);
306             auditMiddleTableName = mainGenerator.getVerEntCfg().getAuditTableName(null, middleTableName);
307             auditMiddleEntityName = mainGenerator.getVerEntCfg().getAuditEntityName(middleTableName);
308         }
309 
310         LOG.debugf("Using join table name: %s", auditMiddleTableName);
311 
312         // Generating the XML mapping for the middle entity, only if the relation isn't inverse.
313         // If the relation is inverse, will be later checked by comparing middleEntityXml with null.
314         Element middleEntityXml;
315         if (!propertyValue.isInverse()) {
316             // Generating a unique middle entity name
317             auditMiddleEntityName = mainGenerator.getAuditEntityNameRegister().createUnique(auditMiddleEntityName);
318 
319             // Registering the generated name
320             mainGenerator.getAuditEntityNameRegister().register(auditMiddleEntityName);
321 
322             middleEntityXml = createMiddleEntityXml(auditMiddleTableName, auditMiddleEntityName, propertyValue.getWhere());
323         } else {
324             middleEntityXml = null;
325         }
326 
327         // ******
328         // Generating the mapping for the referencing entity (it must be an entity).
329         // ******
330         // Getting the id-mapping data of the referencing entity (the entity that "owns" this collection).
331         IdMappingData referencingIdMapping = referencingEntityConfiguration.getIdMappingData();
332 
333         // Only valid for an inverse relation; null otherwise.
334         String mappedBy;
335 
336         // The referencing prefix is always for a related entity. So it has always the "_" at the end added.
337         String referencingPrefixRelated;
338         String referencedPrefix;
339 
340         if (propertyValue.isInverse()) {
341             // If the relation is inverse, then referencedEntityName is not null.
342             mappedBy = getMappedBy(propertyValue.getCollectionTable(), mainGenerator.getCfg().getClassMapping(referencedEntityName));
343 
344             referencingPrefixRelated = mappedBy + "_";
345             referencedPrefix = StringTools.getLastComponent(referencedEntityName);
346         } else {
347             mappedBy = null;
348 
349             referencingPrefixRelated = StringTools.getLastComponent(referencingEntityName) + "_";
350             referencedPrefix = referencedEntityName == null ? "element" : propertyName;
351         }
352 
353         // Storing the id data of the referencing entity: original mapper, prefixed mapper and entity name.
354         MiddleIdData referencingIdData = createMiddleIdData(referencingIdMapping,
355                 referencingPrefixRelated, referencingEntityName);
356 
357         // Creating a query generator builder, to which additional id data will be added, in case this collection
358         // references some entities (either from the element or index). At the end, this will be used to build
359         // a query generator to read the raw data collection from the middle table.
360 		QueryGeneratorBuilder queryGeneratorBuilder = new QueryGeneratorBuilder(mainGenerator.getGlobalCfg(),
361                 mainGenerator.getVerEntCfg(), mainGenerator.getAuditStrategy(), referencingIdData,
362 				auditMiddleEntityName, isEmbeddableElementType());
363 
364         // Adding the XML mapping for the referencing entity, if the relation isn't inverse.
365         if (middleEntityXml != null) {
366             // Adding related-entity (in this case: the referencing's entity id) id mapping to the xml.
367             addRelatedToXmlMapping(middleEntityXml, referencingPrefixRelated,
368                     MetadataTools.getColumnNameIterator(propertyValue.getKey().getColumnIterator()),
369                     referencingIdMapping);
370         }
371 
372         // ******
373         // Generating the element mapping.
374         // ******
375         MiddleComponentData elementComponentData = addValueToMiddleTable(propertyValue.getElement(), middleEntityXml,
376                 queryGeneratorBuilder, referencedPrefix, propertyAuditingData.getJoinTable().inverseJoinColumns());
377 
378         // ******
379         // Generating the index mapping, if an index exists.
380         // ******
381         MiddleComponentData indexComponentData = addIndex(middleEntityXml, queryGeneratorBuilder);
382 
383         // ******
384         // Generating the property mapper.
385         // ******
386         // Building the query generator.
387         RelationQueryGenerator queryGenerator = queryGeneratorBuilder.build(elementComponentData, indexComponentData);
388 
389         // Creating common data
390         CommonCollectionMapperData commonCollectionMapperData = new CommonCollectionMapperData(
391                 mainGenerator.getVerEntCfg(), auditMiddleEntityName,
392                 propertyAuditingData.getPropertyData(),
393                 referencingIdData, queryGenerator);
394 
395         // Checking the type of the collection and adding an appropriate mapper.
396         addMapper(commonCollectionMapperData, elementComponentData, indexComponentData);
397 
398         // ******
399         // Storing information about this relation.
400         // ******
401         storeMiddleEntityRelationInformation(mappedBy);
402     }

404     private MiddleComponentData addIndex(Element middleEntityXml, QueryGeneratorBuilder queryGeneratorBuilder) {
405         if (propertyValue instanceof IndexedCollection) {
406             IndexedCollection indexedValue = (IndexedCollection) propertyValue;
407             String mapKey = propertyAuditingData.getMapKey();
408             if (mapKey == null) {
409                 // This entity doesn't specify a javax.persistence.MapKey. Mapping it to the middle entity.
410                 return addValueToMiddleTable(indexedValue.getIndex(), middleEntityXml,
411                         queryGeneratorBuilder, "mapkey", null);
412             } else {
413                 IdMappingData referencedIdMapping = mainGenerator.getEntitiesConfigurations()
414                         .get(referencedEntityName).getIdMappingData();
415                 int currentIndex = queryGeneratorBuilder == null ? 0 : queryGeneratorBuilder.getCurrentIndex();
416                 if ("".equals(mapKey)) {
417                     // The key of the map is the id of the entity.
418                     return new MiddleComponentData(new MiddleMapKeyIdComponentMapper(mainGenerator.getVerEntCfg(),
419                             referencedIdMapping.getIdMapper()), currentIndex);
420                 } else {
421                     // The key of the map is a property of the entity.
422                     return new MiddleComponentData(new MiddleMapKeyPropertyComponentMapper(mapKey,
423                             propertyAuditingData.getAccessType()), currentIndex);
424                 }
425             }
426         } else {
427             // No index - creating a dummy mapper.
428             return new MiddleComponentData(new MiddleDummyComponentMapper(), 0);
429         }
430     }

432     /**
433      *
434      * @param value Value, which should be mapped to the middle-table, either as a relation to another entity,
435      * or as a simple value.
436      * @param xmlMapping If not <code>null</code>, xml mapping for this value is added to this element.
437      * @param queryGeneratorBuilder In case <code>value</code> is a relation to another entity, information about it
438      * should be added to the given.
439      * @param prefix Prefix for proeprty names of related entities identifiers.
440      * @param joinColumns Names of columns to use in the xml mapping, if this array isn't null and has any elements.
441      * @return Data for mapping this component.
442      */
443     @SuppressWarnings({"unchecked"})
444     private MiddleComponentData addValueToMiddleTable(Value value, Element xmlMapping,
445                                                       QueryGeneratorBuilder queryGeneratorBuilder,
446                                                       String prefix, JoinColumn[] joinColumns) {
447         Type type = value.getType();
448         if (type instanceof ManyToOneType) {
449             String prefixRelated = prefix + "_";
450 
451             String referencedEntityName = MappingTools.getReferencedEntityName(value);
452 
453             IdMappingData referencedIdMapping = mainGenerator.getReferencedIdMappingData(referencingEntityName,
454                     referencedEntityName, propertyAuditingData, true);
455 
456             // Adding related-entity (in this case: the referenced entities id) id mapping to the xml only if the
457             // relation isn't inverse (so when <code>xmlMapping</code> is not null).
458             if (xmlMapping != null) {
459                 addRelatedToXmlMapping(xmlMapping, prefixRelated,
460                         joinColumns != null && joinColumns.length > 0
461                                 ? MetadataTools.getColumnNameIterator(joinColumns)
462                                 : MetadataTools.getColumnNameIterator(value.getColumnIterator()),
463                         referencedIdMapping);
464             }
465 
466             // Storing the id data of the referenced entity: original mapper, prefixed mapper and entity name.
467             MiddleIdData referencedIdData = createMiddleIdData(referencedIdMapping,
468                     prefixRelated, referencedEntityName);
469             // And adding it to the generator builder.
470             queryGeneratorBuilder.addRelation(referencedIdData);
471 
472             return new MiddleComponentData(new MiddleRelatedComponentMapper(referencedIdData),
473                     queryGeneratorBuilder.getCurrentIndex());
474 		} else if ( type instanceof ComponentType ) {
475 			// Collection of embeddable elements.
476 			final Component component = (Component) value;
477 			final MiddleEmbeddableComponentMapper componentMapper = new MiddleEmbeddableComponentMapper( new MultiPropertyMapper(), component.getComponentClassName() );
478 
479 			final Element parentXmlMapping = xmlMapping.getParent();
480 			final ComponentAuditingData auditData = new ComponentAuditingData();
481 			final ReflectionManager reflectionManager = mainGenerator.getCfg().getReflectionManager();
482 
483 			new ComponentAuditedPropertiesReader(
484 					ModificationStore.FULL,
485 					new AuditedPropertiesReader.ComponentPropertiesSource( reflectionManager, component ),
486 					auditData, mainGenerator.getGlobalCfg(), reflectionManager, ""
487 			).read();
488 
489 			// Emulating first pass.
490 			for ( String auditedPropertyName : auditData.getPropertyNames() ) {
491 				PropertyAuditingData nestedAuditingData = auditData.getPropertyAuditingData( auditedPropertyName );
492 				mainGenerator.addValue(
493 						parentXmlMapping, component.getProperty( auditedPropertyName ).getValue(), componentMapper,
494 						prefix, xmlMappingData, nestedAuditingData, true, true, true
495 				);
496 			}
497 
498 			// Emulating second pass so that the relations can be mapped too.
499 			for ( String auditedPropertyName : auditData.getPropertyNames() ) {
500 				PropertyAuditingData nestedAuditingData = auditData.getPropertyAuditingData( auditedPropertyName );
501 				mainGenerator.addValue(
502 						parentXmlMapping, component.getProperty( auditedPropertyName ).getValue(),
503 						componentMapper, referencingEntityName, xmlMappingData, nestedAuditingData,
504 						true, false, true
505 				);
506 			}
507 
508 			// Add an additional column holding a number to make each entry unique within the set.
509 			// Embeddable properties may contain null values, so cannot be stored within composite primary key.
510 			if ( propertyValue.isSet() ) {
511 				final String setOrdinalPropertyName = mainGenerator.getVerEntCfg().getEmbeddableSetOrdinalPropertyName();
512 				final Element ordinalProperty = MetadataTools.addProperty(
513 						xmlMapping, setOrdinalPropertyName, "integer", true, true
514 				);
515 				MetadataTools.addColumn(
516 						ordinalProperty, setOrdinalPropertyName, null, null, null, null, null, null, false
517 				);
518 			}
519 
520 			return new MiddleComponentData( componentMapper, 0 );
521         } else {
522             // Last but one parameter: collection components are always insertable
523             boolean mapped = mainGenerator.getBasicMetadataGenerator().addBasic(xmlMapping,
524                     new PropertyAuditingData(prefix, "field", ModificationStore.FULL, RelationTargetAuditMode.AUDITED, null, null, false),
525                     value, null, true, true);
526 
527             if (mapped) {
528                 // Simple values are always stored in the first item of the array returned by the query generator.
529                 return new MiddleComponentData(new MiddleSimpleComponentMapper(mainGenerator.getVerEntCfg(), prefix), 0);
530             } else {
531                 mainGenerator.throwUnsupportedTypeException(type, referencingEntityName, propertyName);
532                 // Impossible to get here.
533                 throw new AssertionError();
534             }
535         }
536     }

538     private void addMapper(CommonCollectionMapperData commonCollectionMapperData, MiddleComponentData elementComponentData,
539                            MiddleComponentData indexComponentData) {
540         Type type = propertyValue.getType();
541 		boolean embeddableElementType = isEmbeddableElementType();
542         if (type instanceof SortedSetType) {
543 			currentMapper.addComposite( propertyAuditingData.getPropertyData(), new SortedSetCollectionMapper(
544 					commonCollectionMapperData, TreeSet.class, SortedSetProxy.class, elementComponentData,
545 					propertyValue.getComparator(), embeddableElementType, embeddableElementType ) );
546 		} else if (type instanceof SetType) {
547 			currentMapper.addComposite( propertyAuditingData.getPropertyData(), new BasicCollectionMapper<Set>(
548 					commonCollectionMapperData, HashSet.class, SetProxy.class, elementComponentData,
549 					embeddableElementType, embeddableElementType ) );
550         } else if (type instanceof SortedMapType) {
551             // Indexed collection, so <code>indexComponentData</code> is not null.
552 			currentMapper.addComposite(propertyAuditingData.getPropertyData(),
553 					new SortedMapCollectionMapper(commonCollectionMapperData,
554 							TreeMap.class, SortedMapProxy.class, elementComponentData, indexComponentData, propertyValue.getComparator(),
555 							embeddableElementType));
556         } else if (type instanceof MapType) {
557             // Indexed collection, so <code>indexComponentData</code> is not null.
558             currentMapper.addComposite(propertyAuditingData.getPropertyData(),
559                     new MapCollectionMapper<Map>(commonCollectionMapperData,
560                     HashMap.class, MapProxy.class, elementComponentData, indexComponentData, embeddableElementType));
561         } else if (type instanceof BagType) {
562 			currentMapper.addComposite( propertyAuditingData.getPropertyData(), new BasicCollectionMapper<List>(
563 					commonCollectionMapperData, ArrayList.class, ListProxy.class, elementComponentData,
564 					embeddableElementType, embeddableElementType ) );
565         } else if (type instanceof ListType) {
566             // Indexed collection, so <code>indexComponentData</code> is not null.
567             currentMapper.addComposite(propertyAuditingData.getPropertyData(),
568                     new ListCollectionMapper(commonCollectionMapperData,
569                     elementComponentData, indexComponentData, embeddableElementType));
570         } else {
571             mainGenerator.throwUnsupportedTypeException(type, referencingEntityName, propertyName);
572         }
573     }

586     private Element createMiddleEntityXml(String auditMiddleTableName, String auditMiddleEntityName, String where) {
587         String schema = mainGenerator.getSchema(propertyAuditingData.getJoinTable().schema(), propertyValue.getCollectionTable());
588         String catalog = mainGenerator.getCatalog(propertyAuditingData.getJoinTable().catalog(), propertyValue.getCollectionTable());
589 
590         Element middleEntityXml = MetadataTools.createEntity(xmlMappingData.newAdditionalMapping(),
591                 new AuditTableData(auditMiddleEntityName, auditMiddleTableName, schema, catalog), null, null);
592         Element middleEntityXmlId = middleEntityXml.addElement("composite-id");
593 
594         // If there is a where clause on the relation, adding it to the middle entity.
595         if (where != null) {
596             middleEntityXml.addAttribute("where", where);
597         }
598 
599         middleEntityXmlId.addAttribute("name", mainGenerator.getVerEntCfg().getOriginalIdPropName());
600 
601         // Adding the revision number as a foreign key to the revision info entity to the composite id of the
602         // middle table.
603         mainGenerator.addRevisionInfoRelation(middleEntityXmlId);
604 
605         // Adding the revision type property to the entity xml.
606         mainGenerator.addRevisionType(isEmbeddableElementType() ? middleEntityXmlId : middleEntityXml, middleEntityXml);
607 
608         // All other properties should also be part of the primary key of the middle entity.
609         return middleEntityXmlId;
610     }

619     private String getMappedBy(Collection collectionValue) {
620         PersistentClass referencedClass = null;
621         if (collectionValue.getElement() instanceof OneToMany) {
622             OneToMany oneToManyValue = (OneToMany) collectionValue.getElement();
623             referencedClass = oneToManyValue.getAssociatedClass();
624         } else if (collectionValue.getElement() instanceof ManyToOne) {
625             // Case for bi-directional relation with @JoinTable on the owning @ManyToOne side.
626             ManyToOne manyToOneValue = (ManyToOne) collectionValue.getElement();
627             referencedClass = manyToOneValue.getMappings().getClass(manyToOneValue.getReferencedEntityName());
628         }
629 
630         // If there's an @AuditMappedBy specified, returning it directly.
631         String auditMappedBy = propertyAuditingData.getAuditMappedBy();
632         if (auditMappedBy != null) {
633             return auditMappedBy;
634         }
635 
636         // searching in referenced class
637         String mappedBy = this.searchMappedBy(referencedClass, collectionValue);
638 
639         if(mappedBy == null) {
640             LOG.debugf("Going to search the mapped by attribute for %s in superclasses of entity: %s",
641                        propertyName,
642                        referencedClass.getClassName());
643 
644             PersistentClass tempClass = referencedClass;
645 			while ((mappedBy == null) && (tempClass.getSuperclass() != null)) {
646                 LOG.debugf("Searching in superclass: %s", tempClass.getSuperclass().getClassName());
647 				mappedBy = this.searchMappedBy(tempClass.getSuperclass(), collectionValue);
648 				tempClass = tempClass.getSuperclass();
649 			}
650         }
651 
652         if(mappedBy == null) {
653 	        throw new MappingException("Unable to read the mapped by attribute for " + propertyName + " in "
654 	                + referencedClass.getClassName() + "!");
655         }
656 
657         return mappedBy;
658     }

660     @SuppressWarnings({"unchecked"})
661     private String searchMappedBy(PersistentClass referencedClass, Collection collectionValue) {
662         Iterator<Property> assocClassProps = referencedClass.getPropertyIterator();
663         while (assocClassProps.hasNext()) {
664             Property property = assocClassProps.next();
665 
666             if (Tools.iteratorsContentEqual(property.getValue().getColumnIterator(),
667                     collectionValue.getKey().getColumnIterator())) {
668                 return property.getName();
669             }
670         }
671         return null;
672     }

674     private String getMappedBy(Table collectionTable, PersistentClass referencedClass) {
675         // If there's an @AuditMappedBy specified, returning it directly.
676         String auditMappedBy = propertyAuditingData.getAuditMappedBy();
677         if (auditMappedBy != null) {
678             return auditMappedBy;
679         }
680 
681         // searching in referenced class
682         String mappedBy = this.searchMappedBy(referencedClass, collectionTable);
683 
684         // not found on referenced class, searching on superclasses
685         if(mappedBy == null) {
686             LOG.debugf("Going to search the mapped by attribute for %s in superclasses of entity: %s",
687                        propertyName,
688                        referencedClass.getClassName());
689 
690             PersistentClass tempClass = referencedClass;
691             while ((mappedBy == null) && (tempClass.getSuperclass() != null)) {
692                 LOG.debugf("Searching in superclass: %s", tempClass.getSuperclass().getClassName());
693                 mappedBy = this.searchMappedBy(tempClass.getSuperclass(), collectionTable);
694                 tempClass = tempClass.getSuperclass();
695 			}
696         }
697 
698         if(mappedBy == null) {
699 	        throw new MappingException("Unable to read the mapped by attribute for " + propertyName + " in "
700 	                + referencedClass.getClassName() + "!");
701         }
702 
703         return mappedBy;
704     }

706     @SuppressWarnings({"unchecked"})
707     private String searchMappedBy(PersistentClass referencedClass, Table collectionTable) {
708         Iterator<Property> properties = referencedClass.getPropertyIterator();
709         while (properties.hasNext()) {
710             Property property = properties.next();
711             if (property.getValue() instanceof Collection) {
712                 // The equality is intentional. We want to find a collection property with the same collection table.
713                 //noinspection ObjectEquality
714                 if (((Collection) property.getValue()).getCollectionTable() == collectionTable) {
715                     return property.getName();
716                 }
717             }
718         }
719         return null;
720     }
