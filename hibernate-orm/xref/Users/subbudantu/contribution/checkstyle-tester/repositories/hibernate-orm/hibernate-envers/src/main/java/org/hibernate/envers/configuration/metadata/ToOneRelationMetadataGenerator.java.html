42  /**
43   * Generates metadata for to-one relations (reference-valued properties).
44   * @author Adam Warski (adam at warski dot org)
45   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
46   */
47  public final class ToOneRelationMetadataGenerator {
48      private final AuditMetadataGenerator mainGenerator;
49  
50      ToOneRelationMetadataGenerator(AuditMetadataGenerator auditMetadataGenerator) {
51          mainGenerator = auditMetadataGenerator;
52      }
53  
54      @SuppressWarnings({"unchecked"})
55      void addToOne(Element parent, PropertyAuditingData propertyAuditingData, Value value,
56                    CompositeMapperBuilder mapper, String entityName, boolean insertable) {
57          String referencedEntityName = ((ToOne) value).getReferencedEntityName();
58  
59          IdMappingData idMapping = mainGenerator.getReferencedIdMappingData(entityName, referencedEntityName,
60                  propertyAuditingData, true);
61  
62          String lastPropertyPrefix = MappingTools.createToOneRelationPrefix(propertyAuditingData.getName());
63  
64          // Generating the id mapper for the relation
65          IdMapper relMapper = idMapping.getIdMapper().prefixMappedProperties(lastPropertyPrefix);
66  
67          // Storing information about this relation
68          mainGenerator.getEntitiesConfigurations().get(entityName).addToOneRelation(
69                  propertyAuditingData.getName(), referencedEntityName, relMapper, insertable);
70  
71          // If the property isn't insertable, checking if this is not a "fake" bidirectional many-to-one relationship,
72          // that is, when the one side owns the relation (and is a collection), and the many side is non insertable.
73          // When that's the case and the user specified to store this relation without a middle table (using
74          // @AuditMappedBy), we have to make the property insertable for the purposes of Envers. In case of changes to
75          // the entity that didn't involve the relation, it's value will then be stored properly. In case of changes
76          // to the entity that did involve the relation, it's the responsibility of the collection side to store the
77          // proper data.
78          boolean nonInsertableFake;
79          if (!insertable && propertyAuditingData.isForceInsertable()) {
80              nonInsertableFake = true;
81              insertable = true;
82          } else {
83              nonInsertableFake = false;
84          }
85  
86          // Adding an element to the mapping corresponding to the references entity id's
87          Element properties = (Element) idMapping.getXmlRelationMapping().clone();
88          properties.addAttribute("name", propertyAuditingData.getName());
89  
90          MetadataTools.prefixNamesInPropertyElement(properties, lastPropertyPrefix,
91                  MetadataTools.getColumnNameIterator(value.getColumnIterator()), false, insertable);
92  
93  		// Extracting related id properties from properties tag
94  		for (Object o : properties.content()) {
95  			Element element = (Element) o;
96  			element.setParent(null);
97  			parent.add(element);
98  		}
99  
100 		// Adding mapper for the id
101         PropertyData propertyData = propertyAuditingData.getPropertyData();
102         mapper.addComposite(propertyData, new ToOneIdMapper(relMapper, propertyData, referencedEntityName, nonInsertableFake));
103     }
104 
105     @SuppressWarnings({"unchecked"})
106     void addOneToOneNotOwning(PropertyAuditingData propertyAuditingData, Value value,
107                               CompositeMapperBuilder mapper, String entityName) {
108         OneToOne propertyValue = (OneToOne) value;
109 
110         String owningReferencePropertyName = propertyValue.getReferencedPropertyName(); // mappedBy
111 
112         EntityConfiguration configuration = mainGenerator.getEntitiesConfigurations().get(entityName);
113         if (configuration == null) {
114             throw new MappingException("An audited relation to a non-audited entity " + entityName + "!");
115         }
116 
117         IdMappingData ownedIdMapping = configuration.getIdMappingData();
118 
119         if (ownedIdMapping == null) {
120             throw new MappingException("An audited relation to a non-audited entity " + entityName + "!");
121         }
122 
123         String lastPropertyPrefix = MappingTools.createToOneRelationPrefix(owningReferencePropertyName);
124         String referencedEntityName = propertyValue.getReferencedEntityName();
125 
126         // Generating the id mapper for the relation
127         IdMapper ownedIdMapper = ownedIdMapping.getIdMapper().prefixMappedProperties(lastPropertyPrefix);
128 
129         // Storing information about this relation
130         mainGenerator.getEntitiesConfigurations().get(entityName).addToOneNotOwningRelation(
131                 propertyAuditingData.getName(), owningReferencePropertyName,
132                 referencedEntityName, ownedIdMapper);
133 
134         // Adding mapper for the id
135         PropertyData propertyData = propertyAuditingData.getPropertyData();
136         mapper.addComposite(propertyData, new OneToOneNotOwningMapper(entityName, referencedEntityName,
137                 owningReferencePropertyName, propertyData));
138     }
139 
140     @SuppressWarnings({"unchecked"})
141     void addOneToOnePrimaryKeyJoinColumn(PropertyAuditingData propertyAuditingData, Value value,
142                                          CompositeMapperBuilder mapper, String entityName, boolean insertable) {
143         String referencedEntityName = ((ToOne) value).getReferencedEntityName();
144 
145         IdMappingData idMapping = mainGenerator.getReferencedIdMappingData(entityName, referencedEntityName,
146                                                                            propertyAuditingData, true);
147 
148         String lastPropertyPrefix = MappingTools.createToOneRelationPrefix(propertyAuditingData.getName());
149 
150         // Generating the id mapper for the relation
151         IdMapper relMapper = idMapping.getIdMapper().prefixMappedProperties(lastPropertyPrefix);
152 
153         // Storing information about this relation
154         mainGenerator.getEntitiesConfigurations().get(entityName).addToOneRelation(propertyAuditingData.getName(),
155                                                                                    referencedEntityName, relMapper, insertable);
156 
157         // Adding mapper for the id
158         PropertyData propertyData = propertyAuditingData.getPropertyData();
159         mapper.addComposite(propertyData, new OneToOnePrimaryKeyJoinColumnMapper(entityName, referencedEntityName, propertyData));
160     }
