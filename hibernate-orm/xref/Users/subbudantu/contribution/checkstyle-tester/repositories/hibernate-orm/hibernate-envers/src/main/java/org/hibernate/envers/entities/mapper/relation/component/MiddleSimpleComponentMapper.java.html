24  package org.hibernate.envers.entities.mapper.relation.component;
25  import java.util.Map;
26  
27  import org.hibernate.engine.spi.SessionImplementor;
28  import org.hibernate.envers.configuration.AuditEntitiesConfiguration;
29  import org.hibernate.envers.entities.EntityInstantiator;
30  import org.hibernate.envers.tools.query.Parameters;
31  
32  /**
33   * @author Adam Warski (adam at warski dot org)
34   */
35  public final class MiddleSimpleComponentMapper implements MiddleComponentMapper {
36      private final String propertyName;
37      private final AuditEntitiesConfiguration verEntCfg;
38  
39      public MiddleSimpleComponentMapper(AuditEntitiesConfiguration verEntCfg, String propertyName) {
40          this.propertyName = propertyName;
41          this.verEntCfg = verEntCfg;
42      }
43  
44      @SuppressWarnings({"unchecked"})
45      public Object mapToObjectFromFullMap(EntityInstantiator entityInstantiator, Map<String, Object> data,
46                                           Object dataObject, Number revision) {
47          return ((Map<String, Object>) data.get(verEntCfg.getOriginalIdPropName())).get(propertyName);
48      }
49  
50      public void mapToMapFromObject(SessionImplementor session, Map<String, Object> idData, Map<String, Object> data, Object obj) {
51  		idData.put(propertyName, obj);
52      }
53  
54      public void addMiddleEqualToQuery(Parameters parameters, String idPrefix1, String prefix1, String idPrefix2, String prefix2) {
55          parameters.addWhere(idPrefix1 + "." + propertyName, false, "=", idPrefix2 + "." + propertyName, false);
56      }
57  }
