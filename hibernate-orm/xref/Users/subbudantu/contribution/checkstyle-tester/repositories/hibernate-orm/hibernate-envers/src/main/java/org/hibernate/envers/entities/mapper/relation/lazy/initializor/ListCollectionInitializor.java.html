24  package org.hibernate.envers.entities.mapper.relation.lazy.initializor;
25  import java.util.ArrayList;
26  import java.util.List;
27  import java.util.Map;
28  
29  import org.hibernate.envers.configuration.AuditConfiguration;
30  import org.hibernate.envers.entities.mapper.relation.MiddleComponentData;
31  import org.hibernate.envers.entities.mapper.relation.query.RelationQueryGenerator;
32  import org.hibernate.envers.reader.AuditReaderImplementor;
33  
34  /**
35   * Initializes a map.
36   * @author Adam Warski (adam at warski dot org)
37   */
38  public class ListCollectionInitializor extends AbstractCollectionInitializor<List> {
39      private final MiddleComponentData elementComponentData;
40      private final MiddleComponentData indexComponentData;
41  
42      public ListCollectionInitializor(AuditConfiguration verCfg,
43                                      AuditReaderImplementor versionsReader,
44                                      RelationQueryGenerator queryGenerator,
45                                      Object primaryKey, Number revision, boolean removed,
46                                      MiddleComponentData elementComponentData,
47                                      MiddleComponentData indexComponentData) {
48          super(verCfg, versionsReader, queryGenerator, primaryKey, revision, removed);
49  
50          this.elementComponentData = elementComponentData;
51          this.indexComponentData = indexComponentData;
52      }
53  
54      @SuppressWarnings({"unchecked"})
55      protected List initializeCollection(int size) {
56          // Creating a list of the given capacity with all elements null initially. This ensures that we can then
57          // fill the elements safely using the <code>List.set</code> method.
58          List list = new ArrayList(size);
59          for (int i=0; i<size; i++) { list.add(null); }
60          return list;
61      }
62  
63      @SuppressWarnings({"unchecked"})
64      protected void addToCollection(List collection, Object collectionRow) {
65          // collectionRow will be the actual object if retrieved from audit relation or middle table
66          // otherwise it will be a List
67          Object elementData = collectionRow;
68  	Object indexData = collectionRow;
69  	if (collectionRow instanceof java.util.List) {
70              elementData = ((List) collectionRow).get(elementComponentData.getComponentIndex());
71              indexData = ((List) collectionRow).get(indexComponentData.getComponentIndex());
72  	}
73          Object element = elementData instanceof Map ?
74                  elementComponentData.getComponentMapper().mapToObjectFromFullMap(entityInstantiator,
75                  (Map<String, Object>) elementData, null, revision)
76                  : elementData ;
77  
78          Object indexObj = indexComponentData.getComponentMapper().mapToObjectFromFullMap(entityInstantiator,
79                  (Map<String, Object>) indexData, element, revision);
80          int index = ((Number) indexObj).intValue();
81  
82          collection.set(index, element);
83      }
84  }
