46  /**
47   * Base class for Envers' collection event related listeners
48   *
49   * @author Adam Warski (adam at warski dot org)
50   * @author HernпїЅn Chanfreau
51   * @author Steve Ebersole
52   * @author Michal Skowronek (mskowr at o2 dot pl)
53   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
54   */
55  public abstract class BaseEnversCollectionEventListener extends BaseEnversEventListener {
56  	protected BaseEnversCollectionEventListener(AuditConfiguration enversConfiguration) {
57  		super( enversConfiguration );
58  	}
59  
60      protected final CollectionEntry getCollectionEntry(AbstractCollectionEvent event) {
61          return event.getSession().getPersistenceContext().getCollectionEntry(event.getCollection());
62      }
63  
64      protected final void onCollectionAction(
65  			AbstractCollectionEvent event,
66  			PersistentCollection newColl,
67  			Serializable oldColl,
68  			CollectionEntry collectionEntry) {
69          if ( shouldGenerateRevision( event ) ) {
70              checkIfTransactionInProgress(event.getSession());
71              
72              AuditProcess auditProcess = getAuditConfiguration().getSyncManager().get(event.getSession());
73  
74  			String entityName = event.getAffectedOwnerEntityName();
75              String ownerEntityName = ((AbstractCollectionPersister) collectionEntry.getLoadedPersister()).getOwnerEntityName();
76              String referencingPropertyName = collectionEntry.getRole().substring(ownerEntityName.length() + 1);
77  
78              // Checking if this is not a "fake" many-to-one bidirectional relation. The relation description may be
79              // null in case of collections of non-entities.
80              RelationDescription rd = searchForRelationDescription( entityName, referencingPropertyName );
81              if ( rd != null && rd.getMappedByPropertyName() != null ) {
82                  generateFakeBidirecationalRelationWorkUnits(
83  						auditProcess,
84  						newColl,
85  						oldColl,
86  						entityName,
87                          referencingPropertyName,
88  						event,
89  						rd
90  				);
91              }
92  			else {
93                  PersistentCollectionChangeWorkUnit workUnit = new PersistentCollectionChangeWorkUnit(
94  						event.getSession(),
95  						entityName,
96  						getAuditConfiguration(),
97  						newColl,
98  						collectionEntry,
99  						oldColl,
100 						event.getAffectedOwnerIdOrNull(),
101 						referencingPropertyName
102 				);
103 				auditProcess.addWorkUnit( workUnit );
104 
105                 if (workUnit.containsWork()) {
106                     // There are some changes: a revision needs also be generated for the collection owner
107                     auditProcess.addWorkUnit(
108 							new CollectionChangeWorkUnit(
109 									event.getSession(),
110 									event.getAffectedOwnerEntityName(),
111 									referencingPropertyName,
112 									getAuditConfiguration(),
113 									event.getAffectedOwnerIdOrNull(),
114 									event.getAffectedOwnerOrNull()
115 							)
116 					);
117 
118                     generateBidirectionalCollectionChangeWorkUnits( auditProcess, event, workUnit, rd );
119                 }
120             }
121         }
122     }
123 
124 	/**
125 	 * Forces persistent collection initialization.
126 	 * @param event Collection event.
127 	 * @return Stored snapshot.
128 	 */
129 	protected Serializable initializeCollection(AbstractCollectionEvent event) {
130 		event.getCollection().forceInitialization();
131 		return event.getCollection().getStoredSnapshot();
132 	}
133 
134 	/**
135 	 * Checks whether modification of not-owned relation field triggers new revision and owner entity is versioned.
136 	 * @param event Collection event.
137 	 * @return {@code true} if revision based on given event should be generated, {@code false} otherwise.
138 	 */
139 	protected boolean shouldGenerateRevision(AbstractCollectionEvent event) {
140 		final String entityName = event.getAffectedOwnerEntityName();
141 		return getAuditConfiguration().getGlobalCfg().isGenerateRevisionsForCollections()
142 				&& getAuditConfiguration().getEntCfg().isVersioned( entityName );
143 	}
144 
145     /**
146      * Looks up a relation description corresponding to the given property in the given entity. If no description is
147      * found in the given entity, the parent entity is checked (so that inherited relations work).
148 	 *
149      * @param entityName Name of the entity, in which to start looking.
150      * @param referencingPropertyName The name of the property.
151 	 * 
152      * @return A found relation description corresponding to the given entity or {@code null}, if no description can
153      * be found.
154      */
155     private RelationDescription searchForRelationDescription(String entityName, String referencingPropertyName) {
156         EntityConfiguration configuration = getAuditConfiguration().getEntCfg().get( entityName );
157         RelationDescription rd = configuration.getRelationDescription(referencingPropertyName);
158         if ( rd == null && configuration.getParentEntityName() != null ) {
159             return searchForRelationDescription( configuration.getParentEntityName(), referencingPropertyName );
160         }
161 
162         return rd;
163     }

165     private void generateFakeBidirecationalRelationWorkUnits(
166 			AuditProcess auditProcess,
167 			PersistentCollection newColl,
168 			Serializable oldColl,
169 			String collectionEntityName,
170 			String referencingPropertyName,
171 			AbstractCollectionEvent event,
172 			RelationDescription rd) {
173         // First computing the relation changes
174         List<PersistentCollectionChangeData> collectionChanges = getAuditConfiguration()
175 				.getEntCfg()
176 				.get( collectionEntityName )
177 				.getPropertyMapper()
178                 .mapCollectionChanges( event.getSession(), referencingPropertyName, newColl, oldColl, event.getAffectedOwnerIdOrNull() );
179 
180         // Getting the id mapper for the related entity, as the work units generated will corrspond to the related
181         // entities.
182         String relatedEntityName = rd.getToEntityName();
183         IdMapper relatedIdMapper = getAuditConfiguration().getEntCfg().get(relatedEntityName).getIdMapper();
184 
185         // For each collection change, generating the bidirectional work unit.
186         for ( PersistentCollectionChangeData changeData : collectionChanges ) {
187             Object relatedObj = changeData.getChangedElement();
188             Serializable relatedId = (Serializable) relatedIdMapper.mapToIdFromEntity(relatedObj);
189             RevisionType revType = (RevisionType) changeData.getData().get(
190 					getAuditConfiguration().getAuditEntCfg().getRevisionTypePropName()
191 			);
192 
193             // This can be different from relatedEntityName, in case of inheritance (the real entity may be a subclass
194             // of relatedEntityName).
195             String realRelatedEntityName = event.getSession().bestGuessEntityName(relatedObj);
196 
197             // By default, the nested work unit is a collection change work unit.
198             AuditWorkUnit nestedWorkUnit = new CollectionChangeWorkUnit(
199 					event.getSession(),
200 					realRelatedEntityName,
201 					rd.getMappedByPropertyName(),
202 					getAuditConfiguration(),
203                     relatedId,
204 					relatedObj
205 			);
206 
207             auditProcess.addWorkUnit(
208 					new FakeBidirectionalRelationWorkUnit(
209 							event.getSession(),
210 							realRelatedEntityName,
211 							getAuditConfiguration(),
212 							relatedId,
213 							referencingPropertyName,
214 							event.getAffectedOwnerOrNull(),
215 							rd,
216 							revType,
217 							changeData.getChangedElementIndex(),
218 							nestedWorkUnit
219 					)
220 			);
221         }
222 
223         // We also have to generate a collection change work unit for the owning entity.
224         auditProcess.addWorkUnit(
225 				new CollectionChangeWorkUnit(
226 						event.getSession(),
227 						collectionEntityName,
228 						referencingPropertyName,
229 						getAuditConfiguration(),
230 						event.getAffectedOwnerIdOrNull(),
231 						event.getAffectedOwnerOrNull()
232 				)
233 		);
234     }
235 
236     private void generateBidirectionalCollectionChangeWorkUnits(
237 			AuditProcess auditProcess,
238 			AbstractCollectionEvent event,
239 			PersistentCollectionChangeWorkUnit workUnit,
240 			RelationDescription rd) {
241         // Checking if this is enabled in configuration ...
242         if ( ! getAuditConfiguration().getGlobalCfg().isGenerateRevisionsForCollections() ) {
243             return;
244         }
245 
246         // Checking if this is not a bidirectional relation - then, a revision needs also be generated for
247         // the other side of the relation.
248         // relDesc can be null if this is a collection of simple values (not a relation).
249         if ( rd != null && rd.isBidirectional() ) {
250             String relatedEntityName = rd.getToEntityName();
251             IdMapper relatedIdMapper = getAuditConfiguration().getEntCfg().get( relatedEntityName ).getIdMapper();
252 
253 			Set<String> toPropertyNames = getAuditConfiguration().getEntCfg()
254 					.getToPropertyNames(event.getAffectedOwnerEntityName(), rd.getFromPropertyName(), relatedEntityName);
255 			String toPropertyName = toPropertyNames.iterator().next();
256 
257             for ( PersistentCollectionChangeData changeData : workUnit.getCollectionChanges() ) {
258                 Object relatedObj = changeData.getChangedElement();
259                 Serializable relatedId = (Serializable) relatedIdMapper.mapToIdFromEntity( relatedObj );
260 
261                 auditProcess.addWorkUnit(
262 						new CollectionChangeWorkUnit(
263 								event.getSession(),
264 								event.getSession().bestGuessEntityName(relatedObj),
265 								toPropertyName,
266 								getAuditConfiguration(),
267 								relatedId,
268 								relatedObj
269 						)
270 				);
271             }
272         }
273     }
