1   package org.hibernate.envers.synchronization;
2   
3   import org.hibernate.Session;
4   import org.hibernate.action.spi.AfterTransactionCompletionProcess;
5   import org.hibernate.engine.spi.SessionImplementor;
6   import org.hibernate.event.spi.EventSource;
7   
8   /**
9    * Class responsible for evicting audit data entries that have been stored in the session level cache.
10   * This operation increases Envers performance in case of massive entity updates without clearing persistence context.
11   * @author Lukasz Antoniak (lukasz dot antoniak at gmail dot com)
12   */
13  public class SessionCacheCleaner {
14      /**
15       * Schedules audit data removal from session level cache after transaction completion. The operation is performed
16       * regardless of commit success.
17       * @param session Active Hibernate session.
18       * @param data Audit data that shall be evicted (e.g. revision data or entity snapshot)
19       */
20      public void scheduleAuditDataRemoval(final Session session, final Object data) {
21          ((EventSource) session).getActionQueue().registerProcess(new AfterTransactionCompletionProcess() {
22              public void doAfterTransactionCompletion(boolean success, SessionImplementor session) {
23                  if (!session.isClosed()) {
24                      ((Session) session).evict(data);
25                  }
26              }
27          });
28      }
29  }
