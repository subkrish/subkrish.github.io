24  package org.hibernate.envers.test.integration.modifiedflags;
25  
26  import java.util.List;
27  import javax.persistence.EntityManager;
28  
29  import org.junit.Test;
30  
31  import org.hibernate.ejb.Ejb3Configuration;
32  import org.hibernate.envers.query.AuditEntity;
33  import org.hibernate.envers.test.Priority;
34  import org.hibernate.envers.test.integration.auditReader.AuditedTestEntity;
35  import org.hibernate.envers.test.integration.auditReader.NotAuditedTestEntity;
36  
37  import static junit.framework.Assert.assertEquals;
38  import static junit.framework.Assert.assertTrue;
39  
40  /**
41   * A test which checks the correct behavior of AuditReader.isEntityClassAudited(Class entityClass).
42   * 
43   * @author Hernan Chanfreau
44   * @author Michal Skowronek (mskowr at o2 dot pl)
45   */
46  public class HasChangedAPITest extends AbstractModifiedFlagsEntityTest {
47      public void configure(Ejb3Configuration cfg) {
48          cfg.addAnnotatedClass(AuditedTestEntity.class);
49          cfg.addAnnotatedClass(NotAuditedTestEntity.class);
50      }
51  
52      @Test
53      @Priority(10)
54      public void initData() {
55          EntityManager em = getEntityManager();
56          em.getTransaction().begin();
57          AuditedTestEntity ent1 = new AuditedTestEntity(1, "str1");
58          NotAuditedTestEntity ent2 = new NotAuditedTestEntity(1, "str1");
59  
60          em.persist(ent1);
61          em.persist(ent2);
62          em.getTransaction().commit();
63  
64          em.getTransaction().begin();
65  
66          ent1 = em.find(AuditedTestEntity.class, 1);
67          ent2 = em.find(NotAuditedTestEntity.class, 1);
68          ent1.setStr1("str2");
69          ent2.setStr1("str2");
70          em.getTransaction().commit();
71      }
72  
73      @Test
74  	public void testHasChangedHasNotChangedCriteria() throws Exception {
75  		List list = getAuditReader().createQuery().forRevisionsOfEntity(AuditedTestEntity.class, true, true)
76                  .add(AuditEntity.property("str1").hasChanged()).getResultList();
77  		assertEquals(2, list.size());
78  		assertEquals("str1", ((AuditedTestEntity) list.get(0)).getStr1());
79  		assertEquals("str2", ((AuditedTestEntity) list.get(1)).getStr1());
80  
81  		list = getAuditReader().createQuery().forRevisionsOfEntity(AuditedTestEntity.class, true, true)
82                  .add(AuditEntity.property("str1").hasNotChanged()).getResultList();
83  		assertTrue(list.isEmpty());
84  	}
85  
86  }
