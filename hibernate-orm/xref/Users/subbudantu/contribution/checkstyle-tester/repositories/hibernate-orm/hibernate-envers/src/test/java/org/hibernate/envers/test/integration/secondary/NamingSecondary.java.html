24  package org.hibernate.envers.test.integration.secondary;
25  
26  import java.util.Arrays;
27  import java.util.Iterator;
28  import javax.persistence.EntityManager;
29  
30  import org.junit.Test;
31  
32  import org.hibernate.ejb.Ejb3Configuration;
33  import org.hibernate.envers.test.BaseEnversJPAFunctionalTestCase;
34  import org.hibernate.envers.test.Priority;
35  import org.hibernate.mapping.Join;
36  
37  /**
38   * @author Adam Warski (adam at warski dot org)
39   */
40  public class NamingSecondary extends BaseEnversJPAFunctionalTestCase {
41      private Integer id;
42  
43      public void configure(Ejb3Configuration cfg) {
44          cfg.addAnnotatedClass(SecondaryNamingTestEntity.class);
45      }
46  
47      @Test
48      @Priority(10)
49      public void initData() {
50          SecondaryNamingTestEntity ste = new SecondaryNamingTestEntity("a", "1");
51  
52          // Revision 1
53          EntityManager em = getEntityManager();
54          em.getTransaction().begin();
55  
56          em.persist(ste);
57  
58          em.getTransaction().commit();
59  
60          // Revision 2
61          em.getTransaction().begin();
62  
63          ste = em.find(SecondaryNamingTestEntity.class, ste.getId());
64          ste.setS1("b");
65          ste.setS2("2");
66  
67          em.getTransaction().commit();
68  
69          //
70  
71          id = ste.getId();
72      }
73  
74      @Test
75      public void testRevisionsCounts() {
76          assert Arrays.asList(1, 2).equals(getAuditReader().getRevisions(SecondaryNamingTestEntity.class, id));
77      }
78  
79      @Test
80      public void testHistoryOfId() {
81          SecondaryNamingTestEntity ver1 = new SecondaryNamingTestEntity(id, "a", "1");
82          SecondaryNamingTestEntity ver2 = new SecondaryNamingTestEntity(id, "b", "2");
83  
84          assert getAuditReader().find(SecondaryNamingTestEntity.class, id, 1).equals(ver1);
85          assert getAuditReader().find(SecondaryNamingTestEntity.class, id, 2).equals(ver2);
86      }
87  
88      @SuppressWarnings({"unchecked"})
89      @Test
90      public void testTableNames() {
91          assert "sec_versions".equals(((Iterator<Join>)
92                  getCfg().getClassMapping("org.hibernate.envers.test.integration.secondary.SecondaryNamingTestEntity_AUD")
93                          .getJoinIterator())
94                  .next().getTable().getName());
95      }
96  }
