26  public class EvictAuditDataAfterCommitTest extends BaseEnversFunctionalTestCase {
27      @Override
28      protected Class<?>[] getAnnotatedClasses() {
29          return new Class[]{StrTestEntity.class, SetRefEdEntity.class, SetRefIngEntity.class};
30      }

32      @Test
33      @TestForIssue(jiraKey = "HHH-6614")
34      public void testSessionCacheClear() {
35          Session session = openSession();
36          session.getTransaction().begin();
37          StrTestEntity ste = new StrTestEntity("data");
38          session.persist(ste);
39          session.getTransaction().commit();
40          checkEmptyAuditSessionCache(session, "org.hibernate.envers.test.entities.StrTestEntity_AUD");
41      }
42  
43      @Test
44      @TestForIssue(jiraKey = "HHH-6614")
45      public void testSessionCacheCollectionClear() {
46          final String[] auditEntityNames = new String[] {"org.hibernate.envers.test.entities.onetomany.SetRefEdEntity_AUD",
47                                                          "org.hibernate.envers.test.entities.onetomany.SetRefIngEntity_AUD"};
48  
49          SetRefEdEntity ed1 = new SetRefEdEntity(1, "data_ed_1");
50          SetRefEdEntity ed2 = new SetRefEdEntity(2, "data_ed_2");
51          SetRefIngEntity ing1 = new SetRefIngEntity(3, "data_ing_1");
52          SetRefIngEntity ing2 = new SetRefIngEntity(4, "data_ing_2");
53  
54          Session session = openSession();
55          session.getTransaction().begin();
56          session.persist(ed1);
57          session.persist(ed2);
58          session.persist(ing1);
59          session.persist(ing2);
60          session.getTransaction().commit();
61          checkEmptyAuditSessionCache(session, auditEntityNames);
62  
63          session.getTransaction().begin();
64          ed1 = (SetRefEdEntity) session.load(SetRefEdEntity.class, ed1.getId());
65          ing1.setReference(ed1);
66          ing2.setReference(ed1);
67          session.getTransaction().commit();
68          checkEmptyAuditSessionCache(session, auditEntityNames);
69  
70          session.getTransaction().begin();
71          ed2 = (SetRefEdEntity) session.load(SetRefEdEntity.class, ed2.getId());
72          Set<SetRefIngEntity> reffering = new HashSet<SetRefIngEntity>();
73          reffering.add(ing1);
74          reffering.add(ing2);
75          ed2.setReffering(reffering);
76          session.getTransaction().commit();
77          checkEmptyAuditSessionCache(session, auditEntityNames);
78  
79          session.getTransaction().begin();
80          ed2 = (SetRefEdEntity) session.load(SetRefEdEntity.class, ed2.getId());
81          ed2.getReffering().remove(ing1);
82          session.getTransaction().commit();
83          checkEmptyAuditSessionCache(session, auditEntityNames);
84  
85          session.getTransaction().begin();
86          ed2 = (SetRefEdEntity) session.load(SetRefEdEntity.class, ed2.getId());
87          ed2.getReffering().iterator().next().setData("mod_data_ing_2");
88          session.getTransaction().commit();
89          checkEmptyAuditSessionCache(session, auditEntityNames);
90  
91          session.close();
92      }
93  
94      private void checkEmptyAuditSessionCache(Session session, String ... auditEntityNames) {
95          List<String> entityNames = Arrays.asList(auditEntityNames);
96          PersistenceContext persistenceContext = ((SessionImplementor) session).getPersistenceContext();
97          for ( Map.Entry<Object,EntityEntry> entrySet : persistenceContext.reentrantSafeEntityEntries() ) {
98              final EntityEntry entityEntry = entrySet.getValue();
99              if (entityNames.contains(entityEntry.getEntityName())) {
100                 assert false : "Audit data shall not be stored in the session level cache. This causes performance issues.";
101             }
102             Assert.assertFalse("Revision entity shall not be stored in the session level cache. This causes performance issues.",
103                                SequenceIdRevisionEntity.class.getName().equals(entityEntry.getEntityName()));
104         }
105     }
