178 	protected Configuration constructConfiguration() {
179 		Configuration configuration = new Configuration()
180 				.setProperty(Environment.CACHE_REGION_FACTORY, CachingRegionFactory.class.getName()  );
181 		configuration.setProperty( AvailableSettings.USE_NEW_ID_GENERATOR_MAPPINGS, "true" );
182 		if ( createSchema() ) {
183 			configuration.setProperty( Environment.HBM2DDL_AUTO, "create-drop" );
184 			final String secondSchemaName = createSecondSchema();
185 			if ( StringHelper.isNotEmpty( secondSchemaName ) ) {
186 				if ( !( getDialect() instanceof H2Dialect ) ) {
187 					throw new UnsupportedOperationException( "Only H2 dialect supports creation of second schema." );
188 				}
189 				Helper.createH2Schema( secondSchemaName, configuration );
190 			}
191 		}
192 		configuration.setProperty( Environment.DIALECT, getDialect().getClass().getName() );
193 		return configuration;
194 	}

199 	protected void addMappings(Configuration configuration) {
200 		String[] mappings = getMappings();
201 		if ( mappings != null ) {
202 			for ( String mapping : mappings ) {
203 				configuration.addResource(
204 						getBaseForMappings() + mapping,
205 						getClass().getClassLoader()
206 				);
207 			}
208 		}
209 		Class<?>[] annotatedClasses = getAnnotatedClasses();
210 		if ( annotatedClasses != null ) {
211 			for ( Class<?> annotatedClass : annotatedClasses ) {
212 				configuration.addAnnotatedClass( annotatedClass );
213 			}
214 		}
215 		String[] annotatedPackages = getAnnotatedPackages();
216 		if ( annotatedPackages != null ) {
217 			for ( String annotatedPackage : annotatedPackages ) {
218 				configuration.addPackage( annotatedPackage );
219 			}
220 		}
221 		String[] xmlFiles = getXmlFiles();
222 		if ( xmlFiles != null ) {
223 			for ( String xmlFile : xmlFiles ) {
224 				InputStream is = Thread.currentThread().getContextClassLoader().getResourceAsStream( xmlFile );
225 				configuration.addInputStream( is );
226 			}
227 		}
228 	}

285 	protected void applyCacheSettings(Configuration configuration) {
286 		if ( getCacheConcurrencyStrategy() != null ) {
287 			Iterator itr = configuration.getClassMappings();
288 			while ( itr.hasNext() ) {
289 				PersistentClass clazz = (PersistentClass) itr.next();
290 				Iterator props = clazz.getPropertyClosureIterator();
291 				boolean hasLob = false;
292 				while ( props.hasNext() ) {
293 					Property prop = (Property) props.next();
294 					if ( prop.getValue().isSimpleValue() ) {
295 						String type = ( (SimpleValue) prop.getValue() ).getTypeName();
296 						if ( "blob".equals(type) || "clob".equals(type) ) {
297 							hasLob = true;
298 						}
299 						if ( Blob.class.getName().equals(type) || Clob.class.getName().equals(type) ) {
300 							hasLob = true;
301 						}
302 					}
303 				}
304 				if ( !hasLob && !clazz.isInherited() && overrideCacheStrategy() ) {
305 					configuration.setCacheConcurrencyStrategy( clazz.getEntityName(), getCacheConcurrencyStrategy() );
306 				}
307 			}
308 			itr = configuration.getCollectionMappings();
309 			while ( itr.hasNext() ) {
310 				Collection coll = (Collection) itr.next();
311 				configuration.setCollectionCacheConcurrencyStrategy( coll.getRole(), getCacheConcurrencyStrategy() );
312 			}
313 		}
314 	}

375 	@AfterClassOnce
376 	@SuppressWarnings( {"UnusedDeclaration"})
377 	private void releaseSessionFactory() {
378 		if ( sessionFactory == null ) {
379 			return;
380 		}
381 		sessionFactory.close();
382 		sessionFactory = null;
383         if ( serviceRegistry != null ) {
384 			if ( serviceRegistry.isActive() ) {
385 				try {
386 					serviceRegistry.destroy();
387 				}
388 				catch (Exception ignore) {
389 				}
390 				fail( "StandardServiceRegistry was not closed down as expected" );
391 			}
392 		}
393 		serviceRegistry = null;
394 	}

396 	@OnFailure
397 	@OnExpectedFailure
398 	@SuppressWarnings( {"UnusedDeclaration"})
399 	public void onFailure() {
400 		if ( rebuildSessionFactoryOnError() ) {
401 			releaseSessionFactory();
402 			rebuildSessionFactory();
403 		}
404 	}

406 	protected void rebuildSessionFactory() {
407 		serviceRegistry = buildServiceRegistry( configuration );
408 		if ( isMetadataUsed ) {
409 			// need to rebuild metadata because serviceRegistry was recreated
410 			sessionFactory = ( SessionFactoryImplementor ) buildMetadata( serviceRegistry ).buildSessionFactory();
411 		}
412 		else {
413 			sessionFactory = ( SessionFactoryImplementor ) configuration.buildSessionFactory( serviceRegistry );
414 		}
415 		afterSessionFactoryBuilt();
416 	}

463 	private void cleanupSession() {
464 		if ( session != null && ! ( (SessionImplementor) session ).isClosed() ) {
465 			if ( session.isConnected() ) {
466 				session.doWork( new RollbackWork() );
467 			}
468 			session.close();
469 		}
470 		session = null;
471 	}

474 		public void execute(Connection connection) throws SQLException {
475 			connection.rollback();
476 		}
477 	}

482 	@SuppressWarnings( {"UnnecessaryBoxing", "UnnecessaryUnboxing"})
483 	protected void assertAllDataRemoved() {
484 		if ( !createSchema() ) {
485 			return; // no tables were created...
486 		}
487 		if ( !Boolean.getBoolean( VALIDATE_DATA_CLEANUP ) ) {
488 			return;
489 		}
490 
491 		Session tmpSession = sessionFactory.openSession();
492 		try {
493 			List list = tmpSession.createQuery( "select o from java.lang.Object o" ).list();
494 
495 			Map<String,Integer> items = new HashMap<String,Integer>();
496 			if ( !list.isEmpty() ) {
497 				for ( Object element : list ) {
498 					Integer l = items.get( tmpSession.getEntityName( element ) );
499 					if ( l == null ) {
500 						l = 0;
501 					}
502 					l = l + 1 ;
503 					items.put( tmpSession.getEntityName( element ), l );
504 					System.out.println( "Data left: " + element );
505 				}
506 				fail( "Data is left in the database: " + items.toString() );
507 			}
508 		}
509 		finally {
510 			try {
511 				tmpSession.close();
512 			}
513 			catch( Throwable t ) {
514 				// intentionally empty
515 			}
516 		}
517 	}

519 	protected boolean readCommittedIsolationMaintained(String scenario) {
520 		int isolation = java.sql.Connection.TRANSACTION_READ_UNCOMMITTED;
521 		Session testSession = null;
522 		try {
523 			testSession = openSession();
524 			isolation = testSession.doReturningWork(
525 					new AbstractReturningWork<Integer>() {
526 						@Override
527 						public Integer execute(Connection connection) throws SQLException {
528 							return connection.getTransactionIsolation();
529 						}
530 					}
531 			);
532 		}
533 		catch( Throwable ignore ) {
534 		}
535 		finally {
536 			if ( testSession != null ) {
537 				try {
538 					testSession.close();
539 				}
540 				catch( Throwable ignore ) {
541 				}
542 			}
543 		}
544 		if ( isolation < java.sql.Connection.TRANSACTION_READ_COMMITTED ) {
545 			SkipLog.reportSkip( "environment does not support at least read committed isolation", scenario );
546 			return false;
547 		}
548 		else {
549 			return true;
550 		}
