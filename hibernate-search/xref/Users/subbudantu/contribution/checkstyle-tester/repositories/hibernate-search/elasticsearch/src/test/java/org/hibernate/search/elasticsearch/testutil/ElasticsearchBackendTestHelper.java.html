7   package org.hibernate.search.elasticsearch.testutil;
8   
9   import java.util.ArrayList;
10  import java.util.List;
11  
12  import org.hibernate.search.elasticsearch.client.impl.JestClient;
13  import org.hibernate.search.elasticsearch.impl.ElasticsearchIndexManager;
14  import org.hibernate.search.elasticsearch.impl.ElasticsearchIndexNameNormalizer;
15  import org.hibernate.search.engine.service.spi.ServiceManager;
16  import org.hibernate.search.engine.service.spi.ServiceReference;
17  import org.hibernate.search.indexes.spi.IndexManager;
18  import org.hibernate.search.test.TestResourceManager;
19  import org.hibernate.search.test.util.BackendTestHelper;
20  
21  import io.searchbox.core.Count;
22  import io.searchbox.core.CountResult;
23  
24  /**
25   * {@link BackendTestHelper} implementation based on Elasticsearch.
26   *
27   * @author Gunnar Morling
28   */
29  public class ElasticsearchBackendTestHelper extends BackendTestHelper {
30  
31  	private TestResourceManager resourceManager;
32  
33  	public ElasticsearchBackendTestHelper(TestResourceManager resourceManager) {
34  		this.resourceManager = resourceManager;
35  	}
36  
37  	@Override
38  	public int getNumberOfDocumentsInIndex(Class<?> entityType) {
39  		ServiceManager serviceManager = resourceManager.getExtendedSearchIntegrator().getServiceManager();
40  
41  		IndexManager[] indexManagers = resourceManager.getExtendedSearchIntegrator()
42  				.getIndexBinding( entityType )
43  				.getIndexManagers();
44  
45  		List<String> indexNames = new ArrayList<>( indexManagers.length );
46  
47  		for ( IndexManager indexManager : indexManagers ) {
48  			indexNames.add( ( (ElasticsearchIndexManager)indexManager ).getActualIndexName() );
49  		}
50  
51  		try ( ServiceReference<JestClient> client = serviceManager.requestReference( JestClient.class ) ) {
52  			Count request = new Count.Builder()
53  					.addIndex( indexNames )
54  					.addType( entityType.getName() )
55  					.build();
56  
57  			CountResult response = client.get().executeRequest( request );
58  
59  			return response.getCount().intValue();
60  		}
61  	}
62  
63  	@Override
64  	public int getNumberOfDocumentsInIndex(String indexName) {
65  		ServiceManager serviceManager = resourceManager.getExtendedSearchIntegrator().getServiceManager();
66  
67  		try ( ServiceReference<JestClient> client = serviceManager.requestReference( JestClient.class ) ) {
68  			Count request = new Count.Builder()
69  					.addIndex( ElasticsearchIndexNameNormalizer.getElasticsearchIndexName( indexName ) )
70  					.build();
71  
72  			CountResult response = client.get().executeRequest( request );
73  
74  			return response.getCount().intValue();
75  		}
76  	}
77  
78  	@Override
79  	public int getNumberOfDocumentsInIndexByQuery(String indexName, String fieldName, String value) {
80  		ServiceManager serviceManager = resourceManager.getExtendedSearchIntegrator().getServiceManager();
81  		String query = value.contains( "*" ) ? "wildcard" : "term";
82  
83  		try ( ServiceReference<JestClient> client = serviceManager.requestReference( JestClient.class ) ) {
84  			Count request = new Count.Builder()
85  					.addIndex( ElasticsearchIndexNameNormalizer.getElasticsearchIndexName( indexName ) )
86  					.query( "{ \"query\" : { \"" + query + "\" : { \"" + fieldName + "\" : \"" + value + "\" } } }" )
87  					.build();
88  
89  			CountResult response = client.get().executeRequest( request );
90  
91  			return response.getCount().intValue();
92  		}
93  	}
94  }
