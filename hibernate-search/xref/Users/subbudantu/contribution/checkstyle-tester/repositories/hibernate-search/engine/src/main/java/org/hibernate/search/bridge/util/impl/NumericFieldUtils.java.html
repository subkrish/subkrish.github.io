22  /**
23   * Utility class to handle numeric fields.
24   *
25   * @author Gustavo Fernandes
26   * @author Hardy Ferentschik
27   */
28  public final class NumericFieldUtils {
29  
30  	private static final Log log = LoggerFactory.make();
31  
32  	private NumericFieldUtils() {
33  		//not allowed
34  	}

36  	public static Query createNumericRangeQuery(String fieldName, Object from, Object to,
37  												boolean includeLower, boolean includeUpper) {
38  
39  		Class<?> numericClass;
40  
41  		if ( from != null ) {
42  			numericClass = from.getClass();
43  		}
44  		else if ( to != null ) {
45  			numericClass = to.getClass();
46  		}
47  		else {
48  			throw log.rangeQueryWithNullToAndFromValue( fieldName );
49  		}
50  
51  		if ( Double.class.isAssignableFrom( numericClass ) ) {
52  			return NumericRangeQuery.newDoubleRange( fieldName, (Double) from, (Double) to, includeLower, includeUpper );
53  		}
54  		if ( Byte.class.isAssignableFrom( numericClass ) ) {
55  			return NumericRangeQuery.newIntRange( fieldName, ( (Byte) from ).intValue(), ( (Byte) to ).intValue(), includeLower, includeUpper );
56  		}
57  		if ( Short.class.isAssignableFrom( numericClass ) ) {
58  			return NumericRangeQuery.newIntRange( fieldName, ( (Short) from ).intValue(), ( (Short) to ).intValue(), includeLower, includeUpper );
59  		}
60  		if ( Long.class.isAssignableFrom( numericClass ) ) {
61  			return NumericRangeQuery.newLongRange( fieldName, (Long) from, (Long) to, includeLower, includeUpper );
62  		}
63  		if ( Integer.class.isAssignableFrom( numericClass ) ) {
64  			return NumericRangeQuery.newIntRange( fieldName, (Integer) from, (Integer) to, includeLower, includeUpper );
65  		}
66  		if ( Float.class.isAssignableFrom( numericClass ) ) {
67  			return NumericRangeQuery.newFloatRange( fieldName, (Float) from, (Float) to, includeLower, includeUpper );
68  		}
69  		if ( Date.class.isAssignableFrom( numericClass ) ) {
70  			Long fromValue = from != null ? ((Date) from).getTime() : null;
71  			Long toValue = to != null ? ((Date) to).getTime() : null;
72  			return NumericRangeQuery.newLongRange( fieldName, fromValue, toValue, includeLower, includeUpper );
73  		}
74  		if ( Calendar.class.isAssignableFrom( numericClass ) ) {
75  			Long fromValue = from != null ? ((Calendar) from).getTime().getTime() : null;
76  			Long toValue = to != null ? ((Calendar) to).getTime().getTime() : null;
77  			return NumericRangeQuery.newLongRange( fieldName, fromValue, toValue, includeLower, includeUpper );
78  		}
79  		if ( JavaTimeBridgeProvider.isActive() ) {
80  			if ( java.time.Duration.class.isAssignableFrom( numericClass ) ) {
81  				Long fromValue = from != null ? ( (java.time.Duration) from ).toNanos() : null;
82  				Long toValue = to != null ? ( (java.time.Duration) to ).toNanos() : null;
83  				return NumericRangeQuery.newLongRange( fieldName, fromValue, toValue, includeLower, includeUpper );
84  			}
85  			if ( java.time.Year.class.isAssignableFrom( numericClass ) ) {
86  				Integer fromValue = from != null ? ( (java.time.Year) from ).getValue() : null;
87  				Integer toValue = to != null ? ( (java.time.Year) to ).getValue() : null;
88  				return NumericRangeQuery.newIntRange( fieldName, fromValue, toValue, includeLower, includeUpper );
89  			}
90  			if ( java.time.Instant.class.isAssignableFrom( numericClass ) ) {
91  				Long fromValue = from != null ? ( (java.time.Instant) from ).toEpochMilli() : null;
92  				Long toValue = to != null ? ( (java.time.Instant) to ).toEpochMilli() : null;
93  				return NumericRangeQuery.newLongRange( fieldName, fromValue, toValue, includeLower, includeUpper );
94  			}
95  		}
96  
97  		throw log.numericRangeQueryWithNonNumericToAndFromValues( fieldName );
98  	}
99  
100 	/**
101 	 * Will create a {@code RangeQuery} matching exactly the provided value: lower
102 	 * and upper value match, and bounds are included. This should perform
103 	 * as efficiently as a TermQuery.
104 	 *
105 	 * @param fieldName the field name the query targets
106 	 * @param value the value to match
107 	 * @return the created {@code Query}
108 	 */
109 	public static Query createExactMatchQuery(String fieldName, Object value) {
110 		return createNumericRangeQuery( fieldName, value, value, true, true );
111 	}
112 
113 	/**
114 	 * When the type of {@code RangeQuery} needs to be guessed among keyword based ranges or numeric based
115 	 * range queries, the parameter type defines the strategy.
116 	 *
117 	 * Note that this is currently only used by the Infinispan backend as a fallback and it should be used with a lot
118 	 * of caution as it does not take into account backend specific behaviors.
119 	 * For instance, when indexing on Elasticsearch, Dates require a keyword range query.
120 	 *
121 	 * This should match the default {@code FieldBridge} used for each type.
122 	 * @param value on Object
123 	 * @return true if the value argument is of any type which is by default indexed as a NumericField
124 	 */
125 	public static boolean requiresNumericRangeQuery(Object value) {
126 		if ( value == null ) {
127 			return false;
128 		}
129 		return value instanceof Double ||
130 				value instanceof Byte ||
131 				value instanceof Short ||
132 				value instanceof Long ||
133 				value instanceof Integer ||
134 				value instanceof Float ||
135 				value instanceof Date ||
136 				value instanceof Calendar ||
137 				( JavaTimeBridgeProvider.isActive() && (
138 						value instanceof java.time.Instant ||
139 						value instanceof java.time.Year ||
140 						value instanceof java.time.Duration
141 					));
142 	}
143 
144 	/**
145 	 * Indicates whether the considered {@code FieldBridge} is a numeric one.
146 	 *
147 	 * @param fieldBridge the considered {@code FieldBridge}
148 	 * @return true if the considered {@code FieldBridge} is a numeric {@code FieldBridge}
149 	 */
150 	public static boolean isNumericFieldBridge(FieldBridge fieldBridge) {
151 		EncodingBridge encodingBridge = BridgeAdaptorUtils.unwrapAdaptorOnly( fieldBridge, EncodingBridge.class );
152 		return !NumericEncodingType.UNKNOWN.equals( getNumericEncoding( encodingBridge ) );
153 	}
154 
155 	/**
156 	 * Indicates whether the considered {@code FieldBridge}, or its {@link ContainerBridge#getElementBridge() element bridge},
157 	 * is a numeric one.
158 	 *
159 	 * @param fieldBridge the considered {@code FieldBridge}
160 	 * @return true if the considered {@code FieldBridge} is a numeric {@code FieldBridge}
161 	 */
162 	public static boolean isNumericContainerOrNumericFieldBridge(FieldBridge fieldBridge) {
163 		EncodingBridge encodingBridge = BridgeAdaptorUtils.unwrapAdaptorAndContainer( fieldBridge, EncodingBridge.class );
164 		return !NumericEncodingType.UNKNOWN.equals( getNumericEncoding( encodingBridge ) );
165 	}
166 
167 	private static NumericEncodingType getNumericEncoding(EncodingBridge encodingBridge) {
168 		if ( encodingBridge != null ) {
169 			return encodingBridge.getEncodingType();
170 		}
171 		else {
172 			return NumericEncodingType.UNKNOWN;
173 		}
174 	}
