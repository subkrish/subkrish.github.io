42  /**
43   * Lucene full text search aware session.

45   * @author Emmanuel Bernard
46   * @author John Griffin

48   */
49  final class FullTextSessionImpl extends SessionDelegatorBaseImpl implements FullTextSession, SessionImplementor {
50  
51  	private static final Log log = LoggerFactory.make();
52  
53  	private transient ExtendedSearchIntegrator searchFactory;
54  	private transient SearchFactory searchFactoryAPI;
55  
56  	private final TransactionContext transactionContext;
57  
58  	public FullTextSessionImpl(org.hibernate.Session session) {
59  		super( (SessionImplementor) session );
60  		if ( session == null ) {
61  			throw log.getNullSessionPassedToFullTextSessionCreationException();
62  		}
63  		this.transactionContext = new EventSourceTransactionContext( (EventSource) session );
64  	}
65  
66  	/**
67  	 * Execute a Lucene query and retrieve managed objects of type entities (or their indexed subclasses)
68  	 * If entities is empty, include all indexed entities
69  	 *
70  	 * @param entities must be immutable for the lifetime of the query object
71  	 */
72  	@Override
73  	public FullTextQuery createFullTextQuery(org.apache.lucene.search.Query luceneQuery, Class<?>... entities) {
74  		HSQuery hsQuery = getSearchIntegrator().createHSQuery( luceneQuery, entities );
75  		return createFullTextQuery( hsQuery );
76  	}
77  
78  	@Override
79  	public FullTextQuery createFullTextQuery(QueryDescriptor queryDescriptor, Class<?>... entities) {
80  		HSQuery hsQuery = queryDescriptor.createHSQuery( getSearchIntegrator() )
81  				.targetedEntities( Arrays.asList( entities ) );
82  		return createFullTextQuery( hsQuery );
83  	}
84  
85  	private FullTextQuery createFullTextQuery(HSQuery hsQuery, Class<?>... entities) {
86  		return new FullTextQueryImpl(
87  				hsQuery,
88  				delegate,
89  				new ParameterMetadataImpl( null, null )
90  		);
91  	}
92  
93  	@Override
94  	public <T> void purgeAll(Class<T> entityType) {
95  		purge( entityType, null );
96  	}
97  
98  	@Override
99  	public void flushToIndexes() {
100 		ExtendedSearchIntegrator extendedIntegrator = getSearchIntegrator();
101 		extendedIntegrator.getWorker().flushWorks( transactionContext );
102 	}

104 	@Override
105 	public <T> void purge(Class<T> entityType, Serializable id) {
106 		if ( entityType == null ) {
107 			return;
108 		}
109 
110 		Set<Class<?>> targetedClasses = getSearchIntegrator().getIndexedTypesPolymorphic(
111 				new Class[] {
112 						entityType
113 				}
114 		);
115 		if ( targetedClasses.isEmpty() ) {
116 			String msg = entityType.getName() + " is not an indexed entity or a subclass of an indexed entity";
117 			throw new IllegalArgumentException( msg );
118 		}
119 
120 		for ( Class<?> clazz : targetedClasses ) {
121 			if ( id == null ) {
122 				createAndPerformWork( clazz, null, WorkType.PURGE_ALL );
123 			}
124 			else {
125 				createAndPerformWork( clazz, id, WorkType.PURGE );
126 			}
127 		}
128 	}
129 
130 	private void createAndPerformWork(Class<?> clazz, Serializable id, WorkType workType) {
131 		Work work = new Work( delegate.getTenantIdentifier(), clazz, id, workType );
132 		getSearchIntegrator().getWorker().performWork( work, transactionContext );
133 	}
134 
135 	/**
136 	 * (Re-)index an entity.
137 	 * The entity must be associated with the session and non indexable entities are ignored.
138 	 *
139 	 * @param entity The entity to index - must not be <code>null</code>.
140 	 *
141 	 * @throws IllegalArgumentException if entity is null or not an @Indexed entity
142 	 */
143 	@Override
144 	public <T> void index(T entity) {
145 		if ( entity == null ) {
146 			throw new IllegalArgumentException( "Entity to index should not be null" );
147 		}
148 
149 		Class<?> clazz = HibernateHelper.getClass( entity );
150 		//TODO cache that at the FTSession level
151 		ExtendedSearchIntegrator extendedIntegrator = getSearchIntegrator();
152 		//not strictly necessary but a small optimization
153 		if ( extendedIntegrator.getIndexBinding( clazz ) == null ) {
154 			String msg = "Entity to index is not an @Indexed entity: " + entity.getClass().getName();
155 			throw new IllegalArgumentException( msg );
156 		}
157 		Serializable id = delegate.getIdentifier( entity );
158 		String tenantIdentifier = getTenantIdentifier();
159 
160 		Work work = new Work( tenantIdentifier, entity, id, WorkType.INDEX );
161 		Worker worker = extendedIntegrator.getWorker();
162 		worker.performWork( work, transactionContext );
163 
164 		//TODO
165 		//need to add elements in a queue kept at the Session level
166 		//the queue will be processed by a Lucene(Auto)FlushEventListener
167 		//note that we could keep this queue somewhere in the event listener in the mean time but that requires
168 		//a synchronized hashmap holding this queue on a per session basis plus some session house keeping (yuk)
169 		//another solution would be to subclass SessionImpl instead of having this LuceneSession delegation model
170 		//this is an open discussion
171 	}
172 
173 	@Override
174 	public MassIndexer createIndexer(Class<?>... types) {
175 		MassIndexerFactory massIndexerFactory = createMassIndexerFactory();
176 		MassIndexer massIndexer = massIndexerFactory.createMassIndexer( getSearchIntegrator(), getFactory(), types );
177 		if ( massIndexer instanceof MassIndexerWithTenant ) {
178 			( (MassIndexerWithTenant) massIndexer ).tenantIdentifier( getTenantIdentifier() );
179 		}
180 		return massIndexer;
181 	}
182 
183 	@Override
184 	public SearchFactory getSearchFactory() {
185 		if ( searchFactoryAPI == null ) {
186 			searchFactoryAPI = new SearchFactoryImpl( getSearchIntegrator() );
187 		}
188 		return searchFactoryAPI;
189 	}
190 
191 	private ExtendedSearchIntegrator getSearchIntegrator() {
192 		if ( searchFactory == null ) {
193 			searchFactory = ContextHelper.getSearchIntegrator( delegate );
194 		}
195 		return searchFactory;
196 	}
197 
198 	@Override
199 	public FullTextSharedSessionBuilder sessionWithOptions() {
200 		return new FullTextSharedSessionBuilderDelegator( super.sessionWithOptions() );
201 	}
202 
203 	private MassIndexerFactory createMassIndexerFactory() {
204 		MassIndexerFactory factory;
205 		Properties properties = getSearchIntegrator().getConfigurationProperties();
206 		String factoryClassName = properties.getProperty( MassIndexerFactory.MASS_INDEXER_FACTORY_CLASSNAME );
207 
208 		if ( factoryClassName != null ) {
209 			ExtendedSearchIntegrator extendedIntegrator = getSearchIntegrator();
210 			ServiceManager serviceManager = extendedIntegrator.getServiceManager();
211 			factory = ClassLoaderHelper.instanceFromName(
212 					MassIndexerFactory.class, factoryClassName, "Mass indexer factory", serviceManager
213 			);
214 		}
215 		else {
216 			factory = new DefaultMassIndexerFactory();
217 		}
218 		factory.initialize( properties );
219 		return factory;
220 	}
