68  	@Test
69  	public void testIndexAndSearchNumericField() {
70  		try ( Session session = openSession() ) {
71  			FullTextSession fts = Search.getFullTextSession( session );
72  			Transaction tx = fts.beginTransaction();
73  
74  			// Range Queries including lower and upper bounds
75  			assertEquals( "Query id ", 3, numericQueryFor( fts, "overriddenFieldName", 1, 3 ).size() );
76  			assertEquals( "Query by double range", 3, numericQueryFor( fts, "latitude", -10d, 10d ).size() );
77  			assertEquals( "Query by short range", 3, numericQueryFor( fts, "importance", (short) 11, (short) 13 ).size() );
78  			assertEquals( "Query by Short range", 3, numericQueryFor( fts, "fallbackImportance", Short.valueOf( "11" ), Short.valueOf( "13" ) ).size() );
79  			assertEquals( "Query by byte range", 3, numericQueryFor( fts, "popularity", (byte) 21, (byte) 23 ).size() );
80  			assertEquals( "Query by Byte range", 3, numericQueryFor( fts, "fallbackPopularity", Byte.valueOf( "21" ), Byte.valueOf( "23" ) ).size() );
81  			assertEquals( "Query by integer range", 4, numericQueryFor( fts, "ranking", 1, 2 ).size() );
82  			assertEquals( "Query by long range", 3, numericQueryFor( fts, "myCounter", 1L, 3L ).size() );
83  			assertEquals( "Query by multi-fields", 2, numericQueryFor( fts, "strMultiple", 0.7d, 0.9d ).size() );
84  			assertEquals( "Query on custom bridge by range", 4, numericQueryFor( fts, "visibleStars", -100L, 500L ).size() );
85  
86  			// Range Queries different bounds
87  			assertEquals( "Query by id excluding upper", 2, numericQueryFor( fts, "overriddenFieldName", 1, 3, true, false ).size() );
88  			assertEquals( "Query by id excluding upper and lower", 1, numericQueryFor( fts, "overriddenFieldName", 1, 3, false, false ).size() );
89  
90  			// Range Query for embedded entities
91  			assertEquals( "Range Query for indexed embedded", 2, numericQueryFor( fts, "country.idh", 0.9, 1d ).size() );
92  
93  			// Range Query across entities
94  			assertEquals( "Range Query across entities", 1, numericQueryFor( fts, "pinPoints.stars", 4, 5 ).size() );
95  
96  			// Exact Matching Queries
97  			assertEquals( "Query id exact", 1, doExactQuery( fts, "overriddenFieldName", 1 ).getId() );
98  			assertEquals( "Query double exact", 2, doExactQuery( fts, "latitude", -10d ).getId() );
99  			assertEquals( "Query short exact", 3, doExactQuery( fts, "importance", 12 ).getId() );
100 			assertEquals( "Query byte exact", 3, doExactQuery( fts, "popularity", 22 ).getId() );
101 			assertEquals( "Query integer exact", 3, doExactQuery( fts, "longitude", -20d ).getId() );
102 			assertEquals( "Query long exact", 4, doExactQuery( fts, "myCounter", 4L ).getId() );
103 			assertEquals( "Query multifield exact", 5, doExactQuery( fts, "strMultiple", 0.1d ).getId() );
104 			assertEquals( "Query on custom bridge exact", 3, doExactQuery( fts, "visibleStars", 1000L ).getId() );
105 
106 			tx.commit();
107 			fts.clear();
108 
109 			// Delete operation on Numeric Id with overriden field name:
110 			tx = fts.beginTransaction();
111 			List<?> allLocations = fts.createCriteria( Location.class ).list();
112 			for ( Object location : allLocations ) {
113 				fts.delete( location );
114 			}
115 			tx.commit();
116 			fts.clear();
117 			tx = fts.beginTransaction();
118 
119 			assertEquals( "Check for deletion on Query", 0, numericQueryFor( fts, "overriddenFieldName", 1, 6 ).size() );
120 			// and now check also for the real index contents:
121 			Query query = NumericFieldUtils.createNumericRangeQuery( "overriddenFieldName", 1, 6, true, true );
122 			FullTextQuery fullTextQuery = fts
123 					.createFullTextQuery( query, Location.class )
124 					.setProjection( ProjectionConstants.ID );
125 			assertEquals( "Check for deletion on index projection", 0, fullTextQuery.list().size() );
126 
127 			tx.commit();
128 		}
129 	}

133 	public void testNumericFieldProjections() {
134 		try ( Session session = openSession() ) {
135 			FullTextSession fullTextSession = Search.getFullTextSession( session );
136 			Transaction tx = fullTextSession.beginTransaction();
137 			Query latitudeQuery = NumericFieldUtils.createNumericRangeQuery( "latitude", -20d, -20d, true, true );
138 			List<?> list = fullTextSession.createFullTextQuery( latitudeQuery, Location.class )
139 					.setProjection( "latitude" )
140 					.list();
141 			Assert.assertEquals( 1, list.size() );
142 			Object[] firstProjection = (Object[]) list.get( 0 );
143 			Assert.assertEquals( 1, firstProjection.length );
144 			Assert.assertEquals( -20d, firstProjection[0] );
145 			List<?> listAgain = fullTextSession.createFullTextQuery( latitudeQuery, Location.class )
146 					.setProjection( "coordinatePair_x", "coordinatePair_y", "importance", "popularity" )
147 					.list();
148 			Assert.assertEquals( 1, listAgain.size() );
149 			Object[] secondProjection = (Object[]) listAgain.get( 0 );
150 			Assert.assertEquals( 4, secondProjection.length );
151 			Assert.assertEquals( 1d, secondProjection[0] );
152 			Assert.assertEquals( 2d, secondProjection[1] );
153 			Assert.assertEquals( (short) 10, secondProjection[2] );
154 			Assert.assertEquals( (byte) 20, secondProjection[3] );
155 			tx.commit();
156 		}
157 	}

220 	private Object getUnwrappedBridge(Class<?> clazz, String string, Class<?> expectedBridgeClass) {
221 		FieldBridge bridge = getExtendedSearchIntegrator().getIndexBinding( clazz ).getDocumentBuilder()
222 				.getMetadata().getDocumentFieldMetadataFor( string ).getFieldBridge();
223 		return unwrapBridge( bridge, expectedBridgeClass );
224 	}
225 
226 	private Object unwrapBridge(Object bridge, Class<?> expectedBridgeClass) {
227 		if ( bridge instanceof BridgeAdaptor ) {
228 			return ((BridgeAdaptor) bridge).unwrap( expectedBridgeClass );
229 		}
230 		else {
231 			return bridge;
232 		}
233 	}

235 	@Test
236 	@TestForIssue(jiraKey = "HSEARCH-1987")
237 	public void testSomeOfSeveralFieldsAreNumeric() {
238 		try ( Session session = openSession() ) {
239 			FullTextSession fullTextSession = Search.getFullTextSession( session );
240 
241 			Set<FieldDescriptor> fields = fullTextSession.getSearchFactory()
242 					.getIndexedTypeDescriptor( TouristAttraction.class )
243 					.getProperty( "rating" )
244 					.getIndexedFields();
245 
246 			assertThat( fields ).onProperty( "name" )
247 					.containsOnly( "rating", "ratingNumericPrecision1", "ratingNumericPrecision2" );
248 
249 			for ( FieldDescriptor field : fields ) {
250 				if ( "ratingNumericPrecision1".equals( field.getName() ) ) {
251 					assertThat( field.getType() ).isEqualTo( Type.NUMERIC );
252 					assertThat( field.as( NumericFieldSettingsDescriptor.class ).precisionStep() ).isEqualTo( 1 );
253 				}
254 				else if ( "ratingNumericPrecision2".equals( field.getName() ) ) {
255 					assertThat( field.getType() ).isEqualTo( Type.NUMERIC );
256 					assertThat( field.as( NumericFieldSettingsDescriptor.class ).precisionStep() ).isEqualTo( 2 );
257 				}
258 				else {
259 					assertThat( field.getType() ).isEqualTo( Type.BASIC );
260 				}
261 			}
262 		}
263 	}

316

320 			Location loc3 = new Location( 3, 3L, 0d, -20d, 1, "A text", 0.86d, countryFor( "Brazil", 0.813 ), BigDecimal.TEN, (short) 12, (byte) 22 );

368 	@SuppressWarnings("unchecked")
369 	private void cleanData() {
370 		try ( Session session = openSession() ) {
371 			FullTextSession fullTextSession = Search.getFullTextSession( session );
372 			Transaction tx = fullTextSession.beginTransaction();
373 			List<Location> locations = fullTextSession.createCriteria( Location.class ).list();
374 			for ( Location location : locations ) {
375 				fullTextSession.delete( location );
376 			}
377 
378 			List<Coordinate> coordinates = fullTextSession.createCriteria( Coordinate.class ).list();
379 			for ( Coordinate coordinate : coordinates ) {
380 				fullTextSession.delete( coordinate );
381 			}
382 
383 			List<PointOfInterest> pois = fullTextSession.createCriteria( PointOfInterest.class ).list();
384 			for ( PointOfInterest poi : pois ) {
385 				fullTextSession.delete( poi );
386 			}
387 
388 			List<Position> positions = fullTextSession.createCriteria( Position.class ).list();
389 			for ( Position position : positions ) {
390 				fullTextSession.delete( position );
391 			}
392 
393 			List<TouristAttraction> attractions = fullTextSession.createCriteria( TouristAttraction.class ).list();
394 			for ( TouristAttraction attraction : attractions ) {
395 				fullTextSession.delete( attraction );
396 			}
397 
398 			List<ScoreBoard> scoreboards = fullTextSession.createCriteria( ScoreBoard.class ).list();
399 			for ( ScoreBoard scoreboard : scoreboards ) {
400 				fullTextSession.delete( scoreboard );
401 			}
402 
403 			List<Score> scores = fullTextSession.createCriteria( Score.class ).list();
404 			for ( Score score : scores ) {
405 				fullTextSession.delete( score );
406 			}
407 
408 			tx.commit();
409 		}
