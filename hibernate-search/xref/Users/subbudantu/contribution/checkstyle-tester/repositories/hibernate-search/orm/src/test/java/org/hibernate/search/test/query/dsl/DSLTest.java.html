343 		Transaction transaction = fullTextSession.beginTransaction();
344 		final QueryBuilder monthQb = fullTextSession.getSearchFactory()
345 				.buildQueryBuilder().forEntity( Month.class ).get();
346 
347 		//must
348 		Query query = monthQb
349 				.bool()
350 				.must( monthQb.keyword().onField( "mythology" ).matching( "colder" ).createQuery() )
351 				.createQuery();
352 
353 		List<Month> results = fullTextSession.createFullTextQuery( query, Month.class ).list();
354 		assertEquals( 1, results.size() );
355 		assertEquals( "January", results.get( 0 ).getName() );
356 
357 		//must not + all
358 		query = monthQb
359 				.bool()
360 					.should( monthQb.all().createQuery() )
361 					.must( monthQb.keyword().onField( "mythology" ).matching( "colder" ).createQuery() )
362 						.not()
363 					.createQuery();
364 
365 		results = fullTextSession.createFullTextQuery( query, Month.class ).list();
366 		assertEquals( 2, results.size() );
367 		assertEquals( "February", results.get( 0 ).getName() );
368 		assertEquals( "March", results.get( 1 ).getName() );
369 
370 		//implicit must not + all (not recommended)
371 		query = monthQb
372 				.bool()
373 					.must( monthQb.keyword().onField( "mythology" ).matching( "colder" ).createQuery() )
374 						.not()
375 					.createQuery();
376 		results = fullTextSession.createFullTextQuery( query, Month.class ).list();
377 		assertEquals( 2, results.size() );
378 		assertEquals( "February", results.get( 0 ).getName() );
379 		assertEquals( "March", results.get( 1 ).getName() );
380 
381 		//all except (recommended)
382 		query = monthQb
383 				.all()
384 					.except( monthQb.keyword().onField( "mythology" ).matching( "colder" ).createQuery() )
385 				.createQuery();
386 
387 		results = fullTextSession.createFullTextQuery( query, Month.class ).list();
388 		assertEquals( 2, results.size() );
389 		assertEquals( "February", results.get( 0 ).getName() );
390 		assertEquals( "March", results.get( 1 ).getName() );
391 
392 		transaction.commit();
393 	}
