59  /**
60   * @author gunter.zeilinger@tiani.com
61   * @version $Revision: 12277 $ $Date: 2009-10-12 10:05:08 +0200 (Mo, 12 Okt 2009) $
62   * @since 05.10.2004
63   *
64   */
65  public abstract class AbstractModel extends BasicDatasetModel {
66  
67      private List childs = new ArrayList();
68      private List childsPK = null;
69  
70      protected AbstractModel() {
71          super();
72      }
73  
74      protected AbstractModel(Dataset ds) {
75          super(ds);
76      }
77  
78      public static boolean isModified(String prevVal, String newVal) {
79          return ! (prevVal == null || prevVal.length() == 0
80                  ? newVal == null || newVal.length() == 0
81                  : prevVal.equals(newVal));
82      }
83      
84      public boolean update( Dataset dsNew ) {
85          ByteBuffer bb = ds.getByteBuffer(PrivateTags.SeriesPk);
86          long pk = bb == null ? -1 : Convert.toLong(bb.array());
87          bb = ds.getByteBuffer(PrivateTags.SeriesPk);
88          long pkNew = bb == null ? -1 : Convert.toLong(bb.array());
89          if ( pk != pkNew ) {
90              return false;
91          }
92          this.ds = dsNew;
93          ds.setPrivateCreatorID(PrivateTags.CreatorID);
94          return true;
95      }
96  
97      /**
98       * Returns the childs of this model as List.
99       * 
100      * @return List of childs.
101      */
102     public List listOfChilds() {
103         return childs;
104     }
105 
106     /**
107      * Set the childs for this model.
108      * <p>
109      * This method should not be used directly! Use dedicated method of implementation class instead.
110      * <p>
111      * If param is null, return without any changes!
112      * <p>
113      * Resets <code>childsPK</code> to obtain new child pk's with <code>containsPK</code> method.
114      * 
115      * @param list New list of childs.
116      */
117     public void setChilds( List list ) {
118         if ( list != null ) {
119             childs = list;
120             childsPK = null;
121         }
122     }
123 
124     /**
125      * Checks if this model contains given child.
126      * 
127      * @param model A childs model
128      * @return true if this model contains given child.
129      */
130     public boolean contains( AbstractModel model ) {
131         return childs.contains( model );
132     }
133 
134     /**
135      * Checks if this model contains a child with given pk.
136      * <p>
137      * Use carefully, because pk's are only unique within a model type!
138      * 
139      * @param pk The childs pk
140      * @return 
141      */
142     public boolean containsPK( Long pk ) {
143         return childPKs().contains( pk );
144     }
145 
146     /**
147      * Gets the list of all clients pk's.
148      * <p>
149      * This method is used primary in method <code>containsPK</code>
150      * 
151      * @return List of childs pk's.
152      */
153     public List childPKs(){
154         if ( childsPK != null ) return childsPK;
155         childsPK = new ArrayList();
156         Iterator iter = listOfChilds().iterator();
157         while ( iter.hasNext() ) {
158             childsPK.add( new Long( ((AbstractModel) iter.next() ).getPk() ) );
159         }
160         return childsPK;
161     }
162 
163     /**
164      * Get the pk of this model instance.
165      * 
166      * @return pk
167      */
168     public abstract long getPk();
169 
170 }
