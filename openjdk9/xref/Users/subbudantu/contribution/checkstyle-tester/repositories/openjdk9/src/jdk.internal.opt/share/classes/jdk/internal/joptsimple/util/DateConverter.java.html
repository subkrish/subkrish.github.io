56  package jdk.internal.joptsimple.util;
57  
58  import java.text.DateFormat;
59  import java.text.ParsePosition;
60  import java.text.SimpleDateFormat;
61  import java.util.Date;
62  
63  import jdk.internal.joptsimple.ValueConversionException;
64  import jdk.internal.joptsimple.ValueConverter;
65  
66  /**
67   * Converts values to {@link Date}s using a {@link DateFormat} object.
68   *
69   * @author <a href="mailto:pholser@alumni.rice.edu">Paul Holser</a>
70   */
71  public class DateConverter implements ValueConverter<Date> {
72      private final DateFormat formatter;
73  
74      /**
75       * Creates a converter that uses the given date formatter/parser.
76       *
77       * @param formatter the formatter/parser to use
78       * @throws NullPointerException if {@code formatter} is {@code null}
79       */
80      public DateConverter( DateFormat formatter ) {
81          if ( formatter == null )
82              throw new NullPointerException( "illegal null formatter" );
83  
84          this.formatter = formatter;
85      }
86  
87      /**
88       * Creates a converter that uses a {@link SimpleDateFormat} with the given date/time pattern.  The date formatter
89       * created is not {@link SimpleDateFormat#setLenient(boolean) lenient}.
90       *
91       * @param pattern expected date/time pattern
92       * @return the new converter
93       * @throws NullPointerException if {@code pattern} is {@code null}
94       * @throws IllegalArgumentException if {@code pattern} is invalid
95       */
96      public static DateConverter datePattern( String pattern ) {
97          SimpleDateFormat formatter = new SimpleDateFormat( pattern );
98          formatter.setLenient( false );
99  
100         return new DateConverter( formatter );
101     }
102 
103     public Date convert( String value ) {
104         ParsePosition position = new ParsePosition( 0 );
105 
106         Date date = formatter.parse( value, position );
107         if ( position.getIndex() != value.length() )
108             throw new ValueConversionException( message( value ) );
109 
110         return date;
111     }
112 
113     public Class<Date> valueType() {
114         return Date.class;
115     }
116 
117     public String valuePattern() {
118         return formatter instanceof SimpleDateFormat
119             ? ( (SimpleDateFormat) formatter ).toPattern()
120             : "";
121     }
122 
123     private String message( String value ) {
124         String message = "Value [" + value + "] does not match date/time pattern";
125         if ( formatter instanceof SimpleDateFormat )
126             message += " [" + ( (SimpleDateFormat) formatter ).toPattern() + ']';
127 
128         return message;
129     }
130 }
